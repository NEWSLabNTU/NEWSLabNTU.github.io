287 // ── C trampolines ───────────────────────────────────────────────
288 //
289 // `ctx` is a pointer to this `ActionServer<A>` instance, so the
290 // trampoline reads the user's stored function pointer via the
291 // instance's own fields — no shared mutable statics.
292
293 static int32_t goal_trampoline(const uint8_t goal_id[16], const uint8_t* data, size_t len,
294 void* ctx) {
295 auto* self = static_cast<ActionServer*>(ctx);
296 if (!self) return static_cast<int32_t>(GoalResponse::Reject);
297 GoalType g;
298 if (GoalType::ffi_deserialize(data, len, &g) != 0) {
299 return static_cast<int32_t>(GoalResponse::Reject);
300 }
301 if (self->user_goal_fn_ctx_ != nullptr) {
302 return static_cast<int32_t>(self->user_goal_fn_ctx_(goal_id, g, self->user_goal_ctx_));
303 }
304 if (self->user_goal_fn_ != nullptr) {
305 return static_cast<int32_t>(self->user_goal_fn_(goal_id, g));
306 }
307 return static_cast<int32_t>(GoalResponse::Reject);
308 }
309
310 static int32_t cancel_trampoline(const uint8_t goal_id[16], void* ctx) {
311 auto* self = static_cast<ActionServer*>(ctx);
312 if (!self) return static_cast<int32_t>(CancelResponse::Accept);
313 if (self->user_cancel_fn_ctx_ != nullptr) {
314 return static_cast<int32_t>(self->user_cancel_fn_ctx_(goal_id, self->user_cancel_ctx_));
315 }
316 if (self->user_cancel_fn_ != nullptr) {
317 return static_cast<int32_t>(self->user_cancel_fn_(goal_id));
318 }
319 return static_cast<int32_t>(CancelResponse::Accept);
320 }
321
322 Result install_callbacks() {
323 bool goal_set = (user_goal_fn_ != nullptr) || (user_goal_fn_ctx_ != nullptr);
324 bool cancel_set = (user_cancel_fn_ != nullptr) || (user_cancel_fn_ctx_ != nullptr);
325 nros_cpp_goal_callback_t gcb = goal_set ? &goal_trampoline : nullptr;
326 nros_cpp_cancel_callback_t ccb = cancel_set ? &cancel_trampoline : nullptr;
327 return Result(nros_cpp_action_server_set_callbacks(storage_, gcb, ccb, this));
328 }

352template <typename A>
353Result Node::create_action_server(ActionServer<A>& out, const char* action_name, const QoS& qos) {
354 if (!initialized_) return Result(ErrorCode::NotInitialized);
355 nros_cpp_qos_t ffi_qos;
356 ffi_qos.reliability = static_cast<nros_cpp_qos_reliability_t>(qos.reliability_raw());
357 ffi_qos.durability = static_cast<nros_cpp_qos_durability_t>(qos.durability_raw());
358 ffi_qos.history = static_cast<nros_cpp_qos_history_t>(qos.history_raw());
359 ffi_qos.depth = qos.depth();
360 nros_cpp_ret_t ret = nros_cpp_action_server_create(
361 &handle_, action_name, A::TYPE_NAME, A::Goal::TYPE_HASH, ffi_qos, out.storage_);
362 if (ret != 0) return Result(ret);
363 // Register with executor — creates transport handles (3 queryables + 2 publishers).
364 // Deferred from create to avoid FreeRTOS QEMU deadlocks. Phase 87.6:
365 // names are passed at register-time (buffers live on the C++
366 // `nros::ActionServer<A>` class, not in the Rust struct).
367 ret = nros_cpp_action_server_register(out.storage_, executor_handle_, action_name,
368 A::TYPE_NAME, A::Goal::TYPE_HASH);
369 if (ret == 0) {
370 // Phase 87.6: copy action_name into the C++-owned buffer for
371 // `get_action_name()` accessor.
372 size_t name_len = 0;
373 while (action_name[name_len] != '\0' && name_len + 1 < sizeof(out.action_name_)) {
374 out.action_name_[name_len] = action_name[name_len];
375 ++name_len;
376 }
377 out.action_name_[name_len] = '\0';
378 out.executor_ = executor_handle_;
379 out.initialized_ = true;
380 }
381 return Result(ret);
382}
