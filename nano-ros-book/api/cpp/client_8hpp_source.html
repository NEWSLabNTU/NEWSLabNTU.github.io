60 Future<ResponseType> send_request(const RequestType& req) {
61 if (!initialized_) return Future<ResponseType>();
62
63 uint8_t req_buf[RequestType::SERIALIZED_SIZE_MAX];
64 size_t req_len = 0;
65 if (RequestType::ffi_serialize(&req, req_buf, sizeof(req_buf), &req_len) != 0) {
66 return Future<ResponseType>();
67 }
68
69 nros_cpp_ret_t ret = nros_cpp_service_client_send_request(storage_, req_buf, req_len);
70 if (ret != 0) return Future<ResponseType>();

72 return Future<ResponseType>(storage_, &nros_cpp_service_client_try_recv_reply,
73 0 // slot 0 (single outstanding request)
74 );
75 }
