379 // Global executor inline storage for init/shutdown free functions.
380 //
381 // Use a template-static-member trick instead of a function-local static.
382 // Function-local statics need __cxa_guard_acquire/release on first-call
383 // initialisation; on NuttX the resulting guard logic returns NULL for
384 // the storage pointer (observed empirically with LTO on armv7a-nuttx-eabihf,
385 // even with constant-initialisation `= {}`). A template static member is
386 // emitted into .bss like a file-scope variable and gets COMDAT-folded by
387 // the linker, sidestepping the guarded-init path entirely.
388 template <int = 0> struct GlobalStorageHolder {
389 alignas(8) static uint8_t storage[NROS_CPP_EXECUTOR_STORAGE_SIZE];
390 static bool initialized;
391 };
392 static uint8_t* global_storage() { return GlobalStorageHolder<>::storage; }
393 static bool& global_initialized() { return GlobalStorageHolder<>::initialized; }
394};
395
396// Out-of-class definitions for Node::GlobalStorageHolder<> — the template
397// machinery means these get emitted as COMDAT symbols, so multiple TUs
398// including this header all collapse to a single .bss allocation.
399template <int N>
400alignas(8) uint8_t Node::GlobalStorageHolder<N>::storage[NROS_CPP_EXECUTOR_STORAGE_SIZE] = {};
401template <int N> bool Node::GlobalStorageHolder<N>::initialized = false;
