657 // Global executor inline storage for init/shutdown free functions.
658 //
659 // Use a template-static-member trick instead of a function-local static.
660 // Function-local statics need __cxa_guard_acquire/release on first-call
661 // initialisation; on NuttX the resulting guard logic returns NULL for
662 // the storage pointer (observed empirically with LTO on armv7a-nuttx-eabihf,
663 // even with constant-initialisation `= {}`). A template static member is
664 // emitted into .bss like a file-scope variable and gets COMDAT-folded by
665 // the linker, sidestepping the guarded-init path entirely.
666 template <int = 0> struct GlobalStorageHolder {
667 alignas(8) static uint8_t storage[NROS_CPP_EXECUTOR_STORAGE_SIZE];
668 static bool initialized;
669 };
670 static uint8_t* global_storage() { return GlobalStorageHolder<>::storage; }
671 static bool& global_initialized() { return GlobalStorageHolder<>::initialized; }
672};
673
674// Out-of-class definitions for Node::GlobalStorageHolder<> — the template
675// machinery means these get emitted as COMDAT symbols, so multiple TUs
676// including this header all collapse to a single .bss allocation.
677template <int N>
678alignas(8) uint8_t Node::GlobalStorageHolder<N>::storage[NROS_CPP_EXECUTOR_STORAGE_SIZE] = {};
679template <int N> bool Node::GlobalStorageHolder<N>::initialized = false;

683inline Result init(const char* locator, uint8_t domain_id) {
684#if defined(NROS_CPP_STD) || (__STDC_HOSTED__ + 0)
685 // Phase 123.B.3 — on hosted builds, fall through to env vars
686 // ($NROS_LOCATOR / $ROS_DOMAIN_ID) so the no-arg `nros::init()`
687 // call works without `getenv()` boilerplate in user code.
688 // Explicit non-null `locator` / non-zero `domain_id` still win.
689 // Phase-287 W6 — the hard "tcp/127.0.0.1:7447" fallback moved BELOW the
690 // baked-macro check: threadx-linux is a HOSTED embedded target, and the
691 // eager default here shadowed its baked `NROS_ENTRY_LOCATOR` port.
692 // RFC-0045 / issue #206 — the env overlay (NROS_LOCATOR / ROS_DOMAIN_ID /
693 // NROS_NODE_NAME) moved into the shared Rust resolver behind
694 // nros_cpp_init (precedence model A: hosted env > this baked chain >
695 // compiled default; malformed or >232 ROS_DOMAIN_ID is an init ERROR,
696 // never a silent domain 0). This header only assembles the baked rung.
697#endif
698 // Baked compile-time locator (embedded gate) beats the local default but
699 // loses to an explicit arg; the hosted env rung (in the Rust resolver)
700 // overrides all of these (phase-287 W6 + #206; see the 3-arg overload).
701#ifdef NROS_ENTRY_LOCATOR
702 if (locator == nullptr) {
703 locator = NROS_ENTRY_LOCATOR;
704 }
705#endif
706#if defined(NROS_CPP_STD) || (__STDC_HOSTED__ + 0)
707 if (locator == nullptr) {
708 locator = "tcp/127.0.0.1:7447";
709 }
710#endif
711 // Phase 266 (W6) — unified default session name "node" across C, C++, and Rust.
712 return init(locator, domain_id, "node");
713}

715inline Result init(const char* locator, uint8_t domain_id, const char* session_name) {
716 // NROS_CPP_RET_INVALID_ARGUMENT = -3 (defined in nros_cpp_ffi.h
717 // which isn't included from this header — duplicate the value
718 // inline; generated header is the source of truth).
719 if (session_name == nullptr) {
720 return Result(-3);
721 }
722#if defined(NROS_CPP_STD) || (__STDC_HOSTED__ + 0)
723 // Issue #39 — apply the same `$NROS_LOCATOR` / `$ROS_DOMAIN_ID` env
724 // fallback as the 2-arg `init()` when `locator` is null / `domain_id` is
725 // 0. This makes `init_with_launch_auto()` (which delegates here with a
726 // null locator) honor the env overlay instead of passing a null locator
727 // to the backend → TransportError / degraded session.
728 // Phase-287 W6 — the hard local default moved BELOW the baked-macro check
729 // (threadx-linux is hosted; the eager default shadowed its baked port).
730 // RFC-0045 / issue #206 — the env overlay (NROS_LOCATOR / ROS_DOMAIN_ID /
731 // NROS_NODE_NAME) moved into the shared Rust resolver behind
732 // nros_cpp_init (precedence model A: hosted env > this baked chain >
733 // compiled default; malformed or >232 ROS_DOMAIN_ID is an init ERROR,
734 // never a silent domain 0). This header only assembles the baked rung.
735#endif
736 // Phase-287 W6 — compile-time connect defaults, so ONE portable source
737 // works native + embedded. `NROS_ENTRY_LOCATOR` / `NROS_ENTRY_DOMAIN_ID`
738 // are target compile definitions the embedded board gate bakes
739 // (NanoRosEntry.cmake; Kconfig on Zephyr via <nros/main.hpp>); on native
740 // they are undefined. Precedence (model A, RFC-0045/#206): env (hosted,
741 // applied in the Rust resolver) > explicit arg > baked macro > default.
742#ifdef NROS_ENTRY_LOCATOR
743 if (locator == nullptr) {
744 locator = NROS_ENTRY_LOCATOR;
745 }
746#endif
747#ifdef NROS_ENTRY_DOMAIN_ID
748 // Only the UNSET sentinel (0) folds the baked macro in; an explicit
749 // argument — including kDomainIdExplicitZero (255) for a literal
750 // domain 0 (issue #227) — passes through untouched.
751 if (domain_id == 0) {
752 domain_id = static_cast<uint8_t>(NROS_ENTRY_DOMAIN_ID);
753 }
754#endif
755#if defined(NROS_CPP_STD) || (__STDC_HOSTED__ + 0)
756 // Hosted local-router default — LAST, after env + baked macro.
757 if (locator == nullptr) {
758 locator = "tcp/127.0.0.1:7447";
759 }
760#endif
761 // Phase 128.C.1 — RMW-blind init. Every backend (cyclonedds,
762 // xrce, dds, zenoh, uorb, …) contributes its registration entry
763 // to the `RMW_INIT_ENTRIES` linker section via the
764 // `NROS_RMW_REGISTER_BACKEND` macro (C/C++) or
765 // `#[linkme::distributed_slice]` (Rust). The cffi runtime's
766 // section walker fires inside `nros_cpp_init` and dispatches to
767 // whichever backends were linked into this binary. No
768 // `#ifdef NROS_RMW_*` chain here, no CMake-driven fan-out — the
769 // user's `target_link_libraries(... NanoRos::Rmw::<name>)` is
770 // the only selector.
771 nros_cpp_ret_t ret =
772 nros_cpp_init(locator, domain_id, session_name, nullptr, Node::global_storage());
773 if (ret == 0) {
774 Node::global_initialized() = true;
775 }
776 return Result(ret);
777}
