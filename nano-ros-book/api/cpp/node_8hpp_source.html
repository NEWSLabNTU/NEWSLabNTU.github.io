388 // Global executor inline storage for init/shutdown free functions.
389 //
390 // Use a template-static-member trick instead of a function-local static.
391 // Function-local statics need __cxa_guard_acquire/release on first-call
392 // initialisation; on NuttX the resulting guard logic returns NULL for
393 // the storage pointer (observed empirically with LTO on armv7a-nuttx-eabihf,
394 // even with constant-initialisation `= {}`). A template static member is
395 // emitted into .bss like a file-scope variable and gets COMDAT-folded by
396 // the linker, sidestepping the guarded-init path entirely.
397 template <int = 0> struct GlobalStorageHolder {
398 alignas(8) static uint8_t storage[NROS_CPP_EXECUTOR_STORAGE_SIZE];
399 static bool initialized;
400 };
401 static uint8_t* global_storage() { return GlobalStorageHolder<>::storage; }
402 static bool& global_initialized() { return GlobalStorageHolder<>::initialized; }
403};
404
405// Out-of-class definitions for Node::GlobalStorageHolder<> — the template
406// machinery means these get emitted as COMDAT symbols, so multiple TUs
407// including this header all collapse to a single .bss allocation.
408template <int N>
409alignas(8) uint8_t Node::GlobalStorageHolder<N>::storage[NROS_CPP_EXECUTOR_STORAGE_SIZE] = {};
410template <int N> bool Node::GlobalStorageHolder<N>::initialized = false;

414inline Result init(const char* locator, uint8_t domain_id) {
415#if defined(NROS_CPP_STD) || (__STDC_HOSTED__ + 0)
416 // Phase 123.B.3 — on hosted builds, fall through to env vars
417 // ($NROS_LOCATOR / $ROS_DOMAIN_ID) so the no-arg `nros::init()`
418 // call works without `getenv()` boilerplate in user code.
419 // Explicit non-null `locator` / non-zero `domain_id` still win.
420 if (locator == nullptr) {
421 const char* env_loc = ::std::getenv("NROS_LOCATOR");
422 if (env_loc != nullptr && env_loc[0] != '\0') {
423 locator = env_loc;
424 } else {
425 locator = "tcp/127.0.0.1:7447";
426 }
427 }
428 if (domain_id == 0) {
429 const char* env_dom = ::std::getenv("ROS_DOMAIN_ID");
430 if (env_dom != nullptr && env_dom[0] != '\0') {
431 // Parse decimal digits inline — no <cstdlib> dep.
432 unsigned acc = 0;
433 for (const char* p = env_dom; *p; ++p) {
434 if (*p < '0' || *p > '9') {
435 acc = 0;
436 break;
437 }
438 acc = acc * 10 + static_cast<unsigned>(*p - '0');
439 if (acc > 232) {
440 acc = 0;
441 break;
442 }
443 }
444 domain_id = static_cast<uint8_t>(acc);
445 }
446 }
447#endif
448 return init(locator, domain_id, "nros_cpp");
449}

451inline Result init(const char* locator, uint8_t domain_id, const char* session_name) {
452 // NROS_CPP_RET_INVALID_ARGUMENT = -3 (defined in nros_cpp_ffi.h
453 // which isn't included from this header — duplicate the value
454 // inline; generated header is the source of truth).
455 if (session_name == nullptr) {
456 return Result(-3);
457 }
458 // Phase 128.C.1 — RMW-blind init. Every backend (cyclonedds,
459 // xrce, dds, zenoh, uorb, …) contributes its registration entry
460 // to the `RMW_INIT_ENTRIES` linker section via the
461 // `NROS_RMW_REGISTER_BACKEND` macro (C/C++) or
462 // `#[linkme::distributed_slice]` (Rust). The cffi runtime's
463 // section walker fires inside `nros_cpp_init` and dispatches to
464 // whichever backends were linked into this binary. No
465 // `#ifdef NROS_RMW_*` chain here, no CMake-driven fan-out — the
466 // user's `target_link_libraries(... NanoRos::Rmw::<name>)` is
467 // the only selector.
468 nros_cpp_ret_t ret =
469 nros_cpp_init(locator, domain_id, session_name, nullptr, Node::global_storage());
470 if (ret == 0) {
471 Node::global_initialized() = true;
472 }
473 return Result(ret);
474}

523// -- Phase 104.C.9 — NodeBuilder ----------------------------------------
524//
525// Mirrors Rust's `Executor::node_builder(name).rmw(...).locator(...).
526// domain_id(...).namespace(...).sched(...).build()` chain. The C++
527// wrapper is value-typed and stack-allocated; it accumulates options
528// into an inline `nros_cpp_node_options_t` and ships it to
529// `nros_cpp_node_create_ex` on `.build()`.

531// Usage:
532// ```cpp
533// nros::Node node;
534// NROS_TRY(executor.node_builder("egress")
535// .rmw("cyclonedds")
536// .domain_id(0)
537// .build(node));

584 Result build(Node& out) const {
585 if (!executor_handle_) return Result(ErrorCode::NotInitialized);
586 out.executor_handle_ = executor_handle_;
587 nros_cpp_ret_t ret =
588 nros_cpp_node_create_ex(executor_handle_, name_, &options_, &out.handle_);
589 if (ret == 0) {
590 out.initialized_ = true;
591 }
592 return Result(ret);
593 }
594
595 private:
596 static void copy_bounded(const char* src, uint8_t* dst, size_t* dst_len, size_t cap) {
597 size_t n = 0;
598 if (src != nullptr) {
599 while (src[n] != '\0' && n < cap) {
600 dst[n] = static_cast<uint8_t>(src[n]);
601 ++n;
602 }
603 }
604 // Zero out the tail so stale bytes don't leak across reuses.
605 for (size_t i = n; i < cap; ++i) {
606 dst[i] = 0;
607 }
608 *dst_len = n;
609 }

611 void* executor_handle_;
612 const char* name_;
613 nros_cpp_node_options_t options_;
