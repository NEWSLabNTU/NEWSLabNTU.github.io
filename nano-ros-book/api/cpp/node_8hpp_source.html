448 // Global executor inline storage for init/shutdown free functions.
449 //
450 // Use a template-static-member trick instead of a function-local static.
451 // Function-local statics need __cxa_guard_acquire/release on first-call
452 // initialisation; on NuttX the resulting guard logic returns NULL for
453 // the storage pointer (observed empirically with LTO on armv7a-nuttx-eabihf,
454 // even with constant-initialisation `= {}`). A template static member is
455 // emitted into .bss like a file-scope variable and gets COMDAT-folded by
456 // the linker, sidestepping the guarded-init path entirely.
457 template <int = 0> struct GlobalStorageHolder {
458 alignas(8) static uint8_t storage[NROS_CPP_EXECUTOR_STORAGE_SIZE];
459 static bool initialized;
460 };
461 static uint8_t* global_storage() { return GlobalStorageHolder<>::storage; }
462 static bool& global_initialized() { return GlobalStorageHolder<>::initialized; }
463};
464
465// Out-of-class definitions for Node::GlobalStorageHolder<> — the template
466// machinery means these get emitted as COMDAT symbols, so multiple TUs
467// including this header all collapse to a single .bss allocation.
468template <int N>
469alignas(8) uint8_t Node::GlobalStorageHolder<N>::storage[NROS_CPP_EXECUTOR_STORAGE_SIZE] = {};
470template <int N> bool Node::GlobalStorageHolder<N>::initialized = false;

474inline Result init(const char* locator, uint8_t domain_id) {
475#if defined(NROS_CPP_STD) || (__STDC_HOSTED__ + 0)
476 // Phase 123.B.3 — on hosted builds, fall through to env vars
477 // ($NROS_LOCATOR / $ROS_DOMAIN_ID) so the no-arg `nros::init()`
478 // call works without `getenv()` boilerplate in user code.
479 // Explicit non-null `locator` / non-zero `domain_id` still win.
480 if (locator == nullptr) {
481 const char* env_loc = ::std::getenv("NROS_LOCATOR");
482 if (env_loc != nullptr && env_loc[0] != '\0') {
483 locator = env_loc;
484 } else {
485 locator = "tcp/127.0.0.1:7447";
486 }
487 }
488 if (domain_id == 0) {
489 const char* env_dom = ::std::getenv("ROS_DOMAIN_ID");
490 if (env_dom != nullptr && env_dom[0] != '\0') {
491 // Parse decimal digits inline — no <cstdlib> dep.
492 unsigned acc = 0;
493 for (const char* p = env_dom; *p; ++p) {
494 if (*p < '0' || *p > '9') {
495 acc = 0;
496 break;
497 }
498 acc = acc * 10 + static_cast<unsigned>(*p - '0');
499 if (acc > 232) {
500 acc = 0;
501 break;
502 }
503 }
504 domain_id = static_cast<uint8_t>(acc);
505 }
506 }
507#endif
508 return init(locator, domain_id, "nros_cpp");
509}

511inline Result init(const char* locator, uint8_t domain_id, const char* session_name) {
512 // NROS_CPP_RET_INVALID_ARGUMENT = -3 (defined in nros_cpp_ffi.h
513 // which isn't included from this header — duplicate the value
514 // inline; generated header is the source of truth).
515 if (session_name == nullptr) {
516 return Result(-3);
517 }
518 // Phase 128.C.1 — RMW-blind init. Every backend (cyclonedds,
519 // xrce, dds, zenoh, uorb, …) contributes its registration entry
520 // to the `RMW_INIT_ENTRIES` linker section via the
521 // `NROS_RMW_REGISTER_BACKEND` macro (C/C++) or
522 // `#[linkme::distributed_slice]` (Rust). The cffi runtime's
523 // section walker fires inside `nros_cpp_init` and dispatches to
524 // whichever backends were linked into this binary. No
525 // `#ifdef NROS_RMW_*` chain here, no CMake-driven fan-out — the
526 // user's `target_link_libraries(... NanoRos::Rmw::<name>)` is
527 // the only selector.
528 nros_cpp_ret_t ret =
529 nros_cpp_init(locator, domain_id, session_name, nullptr, Node::global_storage());
530 if (ret == 0) {
531 Node::global_initialized() = true;
532 }
533 return Result(ret);
534}

583// -- Phase 104.C.9 — NodeBuilder ----------------------------------------
584//
585// Mirrors Rust's `Executor::node_builder(name).rmw(...).locator(...).
586// domain_id(...).namespace(...).sched(...).build()` chain. The C++
587// wrapper is value-typed and stack-allocated; it accumulates options
588// into an inline `nros_cpp_node_options_t` and ships it to
589// `nros_cpp_node_create_ex` on `.build()`.
590//
591// Usage:
592// ```cpp
593// nros::Node node;
594// NROS_TRY(executor.node_builder("egress")
595// .rmw("cyclonedds")
596// .domain_id(0)
597// .build(node));
598// ```
599
600class NodeBuilder {
601 public:
602 NodeBuilder(void* executor_handle, const char* name)
603 : executor_handle_(executor_handle), name_(name),
604 options_(nros_cpp_node_get_default_options()) {}

610 NodeBuilder& rmw(const char* name) {

612 return *this;
613 }
614
617 NodeBuilder& locator(const char* loc) {
618 copy_bounded(loc, options_.locator, &options_.locator_len, NROS_CPP_LOCATOR_LEN);
619 return *this;
620 }
621
624 NodeBuilder& domain_id(uint32_t id) {
625 options_.domain_id_override = id;
626 return *this;
627 }
628
631 NodeBuilder& namespace_(const char* ns) {
632 copy_bounded(ns, options_.namespace_, &options_.namespace_len, NROS_CPP_NAMESPACE_LEN);
633 return *this;
634 }
635
638 NodeBuilder& sched(uint8_t sc_id) {
639 options_.sched_context_id = sc_id;
640 return *this;
641 }
642
644 Result build(Node& out) const {
645 if (!executor_handle_) return Result(ErrorCode::NotInitialized);
646 out.executor_handle_ = executor_handle_;
647 nros_cpp_ret_t ret =
648 nros_cpp_node_create_ex(executor_handle_, name_, &options_, &out.handle_);
649 if (ret == 0) {
650 out.initialized_ = true;
651 }
652 return Result(ret);
653 }
654
655 private:
656 static void copy_bounded(const char* src, uint8_t* dst, size_t* dst_len, size_t cap) {
657 size_t n = 0;
658 if (src != nullptr) {
659 while (src[n] != '\0' && n < cap) {
660 dst[n] = static_cast<uint8_t>(src[n]);
661 ++n;
662 }
663 }
664 // Zero out the tail so stale bytes don't leak across reuses.
665 for (size_t i = n; i < cap; ++i) {
666 dst[i] = 0;
667 }
668 *dst_len = n;
669 }
670
671 void* executor_handle_;
672 const char* name_;
673 nros_cpp_node_options_t options_;
674};
