341 // Global executor inline storage for init/shutdown free functions.
342 //
343 // Use a template-static-member trick instead of a function-local static.
344 // Function-local statics need __cxa_guard_acquire/release on first-call
345 // initialisation; on NuttX the resulting guard logic returns NULL for
346 // the storage pointer (observed empirically with LTO on armv7a-nuttx-eabihf,
347 // even with constant-initialisation `= {}`). A template static member is
348 // emitted into .bss like a file-scope variable and gets COMDAT-folded by
349 // the linker, sidestepping the guarded-init path entirely.
350 template <int = 0> struct GlobalStorageHolder {
351 alignas(8) static uint8_t storage[NROS_CPP_EXECUTOR_STORAGE_SIZE];
352 static bool initialized;
353 };
354 static uint8_t* global_storage() { return GlobalStorageHolder<>::storage; }
355 static bool& global_initialized() { return GlobalStorageHolder<>::initialized; }
356};

360// including this header all collapse to a single .bss allocation.

367inline Result init(const char* locator, uint8_t domain_id) {
368#if defined(NROS_CPP_STD) || (__STDC_HOSTED__ + 0)
369 // Phase 123.B.3 — on hosted builds, fall through to env vars
370 // ($NROS_LOCATOR / $ROS_DOMAIN_ID) so the no-arg `nros::init()`
371 // call works without `getenv()` boilerplate in user code.
372 // Explicit non-null `locator` / non-zero `domain_id` still win.
373 if (locator == nullptr) {
374 const char* env_loc = ::std::getenv("NROS_LOCATOR");
375 if (env_loc != nullptr && env_loc[0] != '\0') {
376 locator = env_loc;
377 } else {
378 locator = "tcp/127.0.0.1:7447";
379 }
380 }
381 if (domain_id == 0) {
382 const char* env_dom = ::std::getenv("ROS_DOMAIN_ID");
383 if (env_dom != nullptr && env_dom[0] != '\0') {
384 // Parse decimal digits inline — no <cstdlib> dep.
385 unsigned acc = 0;
386 for (const char* p = env_dom; *p; ++p) {
387 if (*p < '0' || *p > '9') {
388 acc = 0;
389 break;
390 }
391 acc = acc * 10 + static_cast<unsigned>(*p - '0');
392 if (acc > 232) {
393 acc = 0;
394 break;
395 }
396 }
397 domain_id = static_cast<uint8_t>(acc);
398 }
399 }
400#endif
401 return init(locator, domain_id, "nros_cpp");
402}

404inline Result init(const char* locator, uint8_t domain_id, const char* session_name) {
405 // NROS_CPP_RET_INVALID_ARGUMENT = -3 (defined in nros_cpp_ffi.h

408 if (session_name == nullptr) {

411#ifdef NROS_RMW_CYCLONEDDS

413 // RMW backend (via nros-cpp's `NROS_CPP_RMW=cyclonedds` CMake
414 // option), register the backend's vtable with the runtime before
415 // the first session is created. Idempotent — re-registering
416 // the same vtable is a no-op.
417 {

431 int32_t reg_ret = nros_rmw_xrce_register();

436#endif

526// -- Phase 104.C.9 — NodeBuilder ----------------------------------------
527//
528// Mirrors Rust's `Executor::node_builder(name).rmw(...).locator(...).
529// domain_id(...).namespace(...).sched(...).build()` chain. The C++
530// wrapper is value-typed and stack-allocated; it accumulates options
531// into an inline `nros_cpp_node_options_t` and ships it to
532// `nros_cpp_node_create_ex` on `.build()`.
533//
534// Usage:
535// ```cpp
536// nros::Node node;
537// NROS_TRY(executor.node_builder("egress")
538// .rmw("dds")
539// .domain_id(0)
540// .build(node));
541// ```
542
543class NodeBuilder {
544 public:
545 NodeBuilder(void* executor_handle, const char* name)
546 : executor_handle_(executor_handle), name_(name),
547 options_(nros_cpp_node_get_default_options()) {}
548
553 NodeBuilder& rmw(const char* name) {
554 copy_bounded(name, options_.rmw_name, &options_.rmw_name_len, NROS_CPP_RMW_NAME_LEN);
555 return *this;
556 }
557
560 NodeBuilder& locator(const char* loc) {
561 copy_bounded(loc, options_.locator, &options_.locator_len, NROS_CPP_LOCATOR_LEN);
562 return *this;
563 }
564
567 NodeBuilder& domain_id(uint32_t id) {
568 options_.domain_id_override = id;
569 return *this;
570 }
571
574 NodeBuilder& namespace_(const char* ns) {
575 copy_bounded(ns, options_.namespace_, &options_.namespace_len, NROS_CPP_NAMESPACE_LEN);
576 return *this;
577 }
578
581 NodeBuilder& sched(uint8_t sc_id) {
582 options_.sched_context_id = sc_id;
583 return *this;
584 }

587 Result build(Node& out) const {
588 if (!executor_handle_) return Result(ErrorCode::NotInitialized);
589 out.executor_handle_ = executor_handle_;
590 nros_cpp_ret_t ret =
591 nros_cpp_node_create_ex(executor_handle_, name_, &options_, &out.handle_);
592 if (ret == 0) {
593 out.initialized_ = true;
594 }
595 return Result(ret);
596 }
597
598 private:
599 static void copy_bounded(const char* src, uint8_t* dst, size_t* dst_len, size_t cap) {
600 size_t n = 0;
601 if (src != nullptr) {
602 while (src[n] != '\0' && n < cap) {
603 dst[n] = static_cast<uint8_t>(src[n]);
604 ++n;
605 }
606 }
607 // Zero out the tail so stale bytes don't leak across reuses.
608 for (size_t i = n; i < cap; ++i) {
609 dst[i] = 0;
610 }
611 *dst_len = n;
612 }
613
614 void* executor_handle_;
615 const char* name_;
616 nros_cpp_node_options_t options_;
617};
