325 // Global executor inline storage for init/shutdown free functions.
326 //
327 // Use a template-static-member trick instead of a function-local static.
328 // Function-local statics need __cxa_guard_acquire/release on first-call
329 // initialisation; on NuttX the resulting guard logic returns NULL for
330 // the storage pointer (observed empirically with LTO on armv7a-nuttx-eabihf,
331 // even with constant-initialisation `= {}`). A template static member is
332 // emitted into .bss like a file-scope variable and gets COMDAT-folded by
333 // the linker, sidestepping the guarded-init path entirely.
334 template <int = 0> struct GlobalStorageHolder {
335 alignas(8) static uint8_t storage[NROS_CPP_EXECUTOR_STORAGE_SIZE];
336 static bool initialized;
337 };
338 static uint8_t* global_storage() { return GlobalStorageHolder<>::storage; }
339 static bool& global_initialized() { return GlobalStorageHolder<>::initialized; }
340};

344// including this header all collapse to a single .bss allocation.

351inline Result init(const char* locator, uint8_t domain_id) {
352#if defined(NROS_CPP_STD) || (__STDC_HOSTED__ + 0)
353 // Phase 123.B.3 — on hosted builds, fall through to env vars
354 // ($NROS_LOCATOR / $ROS_DOMAIN_ID) so the no-arg `nros::init()`
355 // call works without `getenv()` boilerplate in user code.
356 // Explicit non-null `locator` / non-zero `domain_id` still win.
357 if (locator == nullptr) {
358 const char* env_loc = ::std::getenv("NROS_LOCATOR");
359 if (env_loc != nullptr && env_loc[0] != '\0') {
360 locator = env_loc;
361 } else {
362 locator = "tcp/127.0.0.1:7447";
363 }
364 }
365 if (domain_id == 0) {
366 const char* env_dom = ::std::getenv("ROS_DOMAIN_ID");
367 if (env_dom != nullptr && env_dom[0] != '\0') {
368 // Parse decimal digits inline — no <cstdlib> dep.
369 unsigned acc = 0;
370 for (const char* p = env_dom; *p; ++p) {
371 if (*p < '0' || *p > '9') {
372 acc = 0;
373 break;
374 }
375 acc = acc * 10 + static_cast<unsigned>(*p - '0');
376 if (acc > 232) {
377 acc = 0;
378 break;
379 }
380 }
381 domain_id = static_cast<uint8_t>(acc);
382 }
383 }
384#endif
385 return init(locator, domain_id, "nros_cpp");
386}

516//
517// Mirrors Rust's `Executor::node_builder(name).rmw(...).locator(...).
518// domain_id(...).namespace(...).sched(...).build()` chain. The C++
519// wrapper is value-typed and stack-allocated; it accumulates options
520// into an inline `nros_cpp_node_options_t` and ships it to
521// `nros_cpp_node_create_ex` on `.build()`.

524// ```cpp
525// nros::Node node;
526// NROS_TRY(executor.node_builder("egress")
527// .rmw("dds")
528// .domain_id(0)
529// .build(node));
530// ```

557 options_.domain_id_override = id;

576 Result build(Node& out) const {
577 if (!executor_handle_) return Result(ErrorCode::NotInitialized);
578 out.executor_handle_ = executor_handle_;
579 nros_cpp_ret_t ret =
580 nros_cpp_node_create_ex(executor_handle_, name_, &options_, &out.handle_);
581 if (ret == 0) {
582 out.initialized_ = true;
583 }
584 return Result(ret);
585 }

587 private:
588 static void copy_bounded(const char* src, uint8_t* dst, size_t* dst_len, size_t cap) {
589 size_t n = 0;
590 if (src != nullptr) {
591 while (src[n] != '\0' && n < cap) {
592 dst[n] = static_cast<uint8_t>(src[n]);
593 ++n;
594 }
595 }
596 // Zero out the tail so stale bytes don't leak across reuses.
597 for (size_t i = n; i < cap; ++i) {
598 dst[i] = 0;
599 }
600 *dst_len = n;
601 }

604 const char* name_;
605 nros_cpp_node_options_t options_;
