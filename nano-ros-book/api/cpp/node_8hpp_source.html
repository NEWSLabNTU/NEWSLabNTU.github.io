643 // Global executor inline storage for init/shutdown free functions.
644 //
645 // Use a template-static-member trick instead of a function-local static.
646 // Function-local statics need __cxa_guard_acquire/release on first-call
647 // initialisation; on NuttX the resulting guard logic returns NULL for
648 // the storage pointer (observed empirically with LTO on armv7a-nuttx-eabihf,
649 // even with constant-initialisation `= {}`). A template static member is
650 // emitted into .bss like a file-scope variable and gets COMDAT-folded by
651 // the linker, sidestepping the guarded-init path entirely.
652 template <int = 0> struct GlobalStorageHolder {
653 alignas(8) static uint8_t storage[NROS_CPP_EXECUTOR_STORAGE_SIZE];
654 static bool initialized;
655 };
656 static uint8_t* global_storage() { return GlobalStorageHolder<>::storage; }
657 static bool& global_initialized() { return GlobalStorageHolder<>::initialized; }
658};
659
660// Out-of-class definitions for Node::GlobalStorageHolder<> — the template
661// machinery means these get emitted as COMDAT symbols, so multiple TUs
662// including this header all collapse to a single .bss allocation.
663template <int N>
664alignas(8) uint8_t Node::GlobalStorageHolder<N>::storage[NROS_CPP_EXECUTOR_STORAGE_SIZE] = {};
665template <int N> bool Node::GlobalStorageHolder<N>::initialized = false;

669inline Result init(const char* locator, uint8_t domain_id) {
670#if defined(NROS_CPP_STD) || (__STDC_HOSTED__ + 0)
671 // Phase 123.B.3 — on hosted builds, fall through to env vars
672 // ($NROS_LOCATOR / $ROS_DOMAIN_ID) so the no-arg `nros::init()`
673 // call works without `getenv()` boilerplate in user code.
674 // Explicit non-null `locator` / non-zero `domain_id` still win.
675 if (locator == nullptr) {
676 const char* env_loc = ::std::getenv("NROS_LOCATOR");
677 if (env_loc != nullptr && env_loc[0] != '\0') {
678 locator = env_loc;
679 } else {
680 locator = "tcp/127.0.0.1:7447";
681 }
682 }
683 if (domain_id == 0) {
684 const char* env_dom = ::std::getenv("ROS_DOMAIN_ID");
685 if (env_dom != nullptr && env_dom[0] != '\0') {
686 // Parse decimal digits inline — no <cstdlib> dep.
687 unsigned acc = 0;
688 for (const char* p = env_dom; *p; ++p) {
689 if (*p < '0' || *p > '9') {
690 acc = 0;
691 break;
692 }
693 acc = acc * 10 + static_cast<unsigned>(*p - '0');
694 if (acc > 232) {
695 acc = 0;
696 break;
697 }
698 }
699 domain_id = static_cast<uint8_t>(acc);
700 }
701 }
702#endif
703 // Phase 266 (W6) — unified default session name "node" across C, C++, and Rust.
704 return init(locator, domain_id, "node");
705}

707inline Result init(const char* locator, uint8_t domain_id, const char* session_name) {
708 // NROS_CPP_RET_INVALID_ARGUMENT = -3 (defined in nros_cpp_ffi.h
709 // which isn't included from this header — duplicate the value
710 // inline; generated header is the source of truth).
711 if (session_name == nullptr) {
712 return Result(-3);
713 }
714#if defined(NROS_CPP_STD) || (__STDC_HOSTED__ + 0)
715 // Issue #39 — apply the same `$NROS_LOCATOR` / `$ROS_DOMAIN_ID` env
716 // fallback as the 2-arg `init()` when `locator` is null / `domain_id` is
717 // 0. This makes `init_with_launch_auto()` (which delegates here with a
718 // null locator) honor the env overlay instead of passing a null locator
719 // to the backend → TransportError / degraded session.
720 if (locator == nullptr) {
721 const char* env_loc = ::std::getenv("NROS_LOCATOR");
722 if (env_loc != nullptr && env_loc[0] != '\0') {
723 locator = env_loc;
724 } else {
725 locator = "tcp/127.0.0.1:7447";
726 }
727 }
728 if (domain_id == 0) {
729 const char* env_dom = ::std::getenv("ROS_DOMAIN_ID");
730 if (env_dom != nullptr && env_dom[0] != '\0') {
731 unsigned acc = 0;
732 for (const char* p = env_dom; *p; ++p) {
733 if (*p < '0' || *p > '9') {
734 acc = 0;
735 break;
736 }
737 acc = acc * 10 + static_cast<unsigned>(*p - '0');
738 if (acc > 232) {
739 acc = 0;
740 break;
741 }
742 }
743 domain_id = static_cast<uint8_t>(acc);
744 }
745 }
746#endif
747 // Phase 128.C.1 — RMW-blind init. Every backend (cyclonedds,
748 // xrce, dds, zenoh, uorb, …) contributes its registration entry
749 // to the `RMW_INIT_ENTRIES` linker section via the
750 // `NROS_RMW_REGISTER_BACKEND` macro (C/C++) or
751 // `#[linkme::distributed_slice]` (Rust). The cffi runtime's
752 // section walker fires inside `nros_cpp_init` and dispatches to
753 // whichever backends were linked into this binary. No
754 // `#ifdef NROS_RMW_*` chain here, no CMake-driven fan-out — the
755 // user's `target_link_libraries(... NanoRos::Rmw::<name>)` is
756 // the only selector.
757 nros_cpp_ret_t ret =
758 nros_cpp_init(locator, domain_id, session_name, nullptr, Node::global_storage());
759 if (ret == 0) {
760 Node::global_initialized() = true;
761 }
762 return Result(ret);
763}

929 static void copy_bounded(const char* src, uint8_t* dst, size_t* dst_len, size_t cap) {
930 size_t n = 0;
931 if (src != nullptr) {
932 while (src[n] != '\0' && n < cap) {
933 dst[n] = static_cast<uint8_t>(src[n]);
934 ++n;
935 }
936 }
