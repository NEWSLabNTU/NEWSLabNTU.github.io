84 Result wait(void* executor_handle, uint32_t timeout_ms, T& out, uint32_t poll_ms = 10) {
85 if (slot_ < 0) return Result(ErrorCode::Error);
86 if (poll_ms == 0) poll_ms = 1;
87 // Phase 89.2: budget by wall-clock. Accumulating `step` per iteration
88 // breaks when the underlying `zpico_spin_once` returns early on a
89 // signaled condvar (keep-alives, discovery gossip) — the 500-step
90 // default-timeout loop can then collapse into milliseconds and
91 // return ErrorCode::Timeout before the reply has a chance to land.
92 const uint64_t start_ns = nros_cpp_time_ns();
93 const uint64_t budget_ns = static_cast<uint64_t>(timeout_ms) * 1000000ULL;
94 while (true) {
95 nros_cpp_ret_t ret =
96 nros_cpp_spin_once(executor_handle, static_cast<int32_t>(poll_ms));
97 // Transient conditions: keep polling. Anything else propagates.
98 // - Ok (0): nothing to dispatch this round.
99 // - Timeout (-2): spin_once returned after its timeout — normal.
100 // - TryAgain (-6): transport hint to retry.
101 if (ret != 0 && ret != static_cast<nros_cpp_ret_t>(ErrorCode::Timeout) &&
102 ret != static_cast<nros_cpp_ret_t>(ErrorCode::TryAgain)) {
103 return Result(ret);
104 }
105 if (is_ready()) return try_take(out);
106 const uint64_t now_ns = nros_cpp_time_ns();
107 if (now_ns - start_ns >= budget_ns) break;
108 }
109 return Result(ErrorCode::Timeout);
110 }
