29// ============================================================================
30// A) std::function callback wrappers for Timer and GuardCondition
31// ============================================================================
32//
33// Lifetime: the heap-allocated std::function is owned by the Timer /
34// GuardCondition instance via `attach_std_closure(unique_ptr)`. The Rust
35// side receives a raw pointer into the same std::function; the Timer's
36// destructor cancels the Rust-side callback before the unique_ptr is
37// dropped, so the raw pointer is never dereferenced after free.
38
39namespace detail {
40
42inline void std_function_trampoline(void* context) {
43 auto* fn = static_cast<std::function<void()>*>(context);
44 (*fn)();
45}
46
47} // namespace detail

60inline Result create_timer(Node& node, Timer& out, std::chrono::milliseconds period,
61 std::function<void()> callback) {
62 auto fn =
63 std::unique_ptr<std::function<void()>>(new std::function<void()>(std::move(callback)));
64 auto* raw = fn.get();
65 Result r = node.create_timer(out, static_cast<uint64_t>(period.count()),
66 detail::std_function_trampoline, raw);
67 if (r.ok()) {
68 out.attach_std_closure(std::move(fn));
