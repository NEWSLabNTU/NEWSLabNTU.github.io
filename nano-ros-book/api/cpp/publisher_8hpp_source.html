263Result Node::create_publisher(Publisher<M>& out, const char* topic, const QoS& qos) {
264 if (!initialized_) return Result(ErrorCode::NotInitialized);
265 nros_cpp_qos_t ffi_qos;
266 ffi_qos.reliability = static_cast<nros_cpp_qos_reliability_t>(qos.reliability_raw());
267 ffi_qos.durability = static_cast<nros_cpp_qos_durability_t>(qos.durability_raw());
268 ffi_qos.history = static_cast<nros_cpp_qos_history_t>(qos.history_raw());
269 ffi_qos.liveliness_kind = static_cast<nros_cpp_qos_liveliness_t>(qos.liveliness_raw());
270 ffi_qos.depth = qos.depth();
271 ffi_qos.deadline_ms = qos.deadline_ms();
272 ffi_qos.lifespan_ms = qos.lifespan_ms();
273 ffi_qos.liveliness_lease_ms = qos.liveliness_lease_ms();
274 ffi_qos.avoid_ros_namespace_conventions = qos.avoid_ros_namespace_conventions() ? 1 : 0;
275 nros_cpp_ret_t ret = nros_cpp_publisher_create(&handle_, topic, M::TYPE_NAME, M::TYPE_HASH,
276 ffi_qos, out.storage_);
277 if (ret == 0) {
278 // Topic name lives C++-side; copy + null-terminate into the
279 // fixed-size buffer. Truncation is silent.
280 size_t topic_len = 0;
281 while (topic[topic_len] != '\0' && topic_len + 1 < sizeof(out.topic_name_)) {
282 out.topic_name_[topic_len] = topic[topic_len];
283 ++topic_len;
284 }
285 out.topic_name_[topic_len] = '\0';
286 out.initialized_ = true;
287 }
288 return Result(ret);
289}
