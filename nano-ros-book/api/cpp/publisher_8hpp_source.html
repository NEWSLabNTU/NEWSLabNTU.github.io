166Result Node::create_publisher(Publisher<M>& out, const char* topic, const QoS& qos) {
167 if (!initialized_) return Result(ErrorCode::NotInitialized);
168 nros_cpp_qos_t ffi_qos;
169 ffi_qos.reliability = static_cast<nros_cpp_qos_reliability_t>(qos.reliability_raw());
170 ffi_qos.durability = static_cast<nros_cpp_qos_durability_t>(qos.durability_raw());
171 ffi_qos.history = static_cast<nros_cpp_qos_history_t>(qos.history_raw());
172 ffi_qos.depth = qos.depth();
173 nros_cpp_ret_t ret = nros_cpp_publisher_create(&handle_, topic, M::TYPE_NAME, M::TYPE_HASH,
174 ffi_qos, out.storage_);
175 if (ret == 0) {
176 // Topic name lives C++-side; copy + null-terminate into the
177 // fixed-size buffer. Truncation is silent.
178 size_t topic_len = 0;
179 while (topic[topic_len] != '\0' && topic_len + 1 < sizeof(out.topic_name_)) {
180 out.topic_name_[topic_len] = topic[topic_len];
181 ++topic_len;
182 }
183 out.topic_name_[topic_len] = '\0';
184 out.initialized_ = true;
185 }
186 return Result(ret);
187}
