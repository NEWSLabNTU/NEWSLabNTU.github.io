66inline Result spin(uint32_t duration_ms, int32_t poll_ms = 10) {
67 if (!Node::global_initialized()) {
68 return Result(ErrorCode::NotInitialized);
69 }
70 uint32_t elapsed = 0;
71 Result last = Result::success();
72 while (elapsed < duration_ms) {
73 int32_t remaining = static_cast<int32_t>(duration_ms - elapsed);
74 int32_t timeout = remaining < poll_ms ? remaining : poll_ms;
75 last = Result(nros_cpp_spin_once(Node::global_storage(), timeout));
76 if (!last.ok()) return last;
77 elapsed += static_cast<uint32_t>(timeout);
78 }
79 return last;
80}
