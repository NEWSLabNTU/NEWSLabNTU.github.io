69 Result wait_next(void* executor_handle, uint32_t timeout_ms, T& out, uint32_t poll_ms = 10) {
70 if (!try_recv_fn_) return Result(ErrorCode::NotInitialized);
71 if (poll_ms == 0) poll_ms = 1;
72 uint32_t elapsed = 0;
73 while (elapsed < timeout_ms) {
74 uint32_t step = poll_ms;
75 if (elapsed + step > timeout_ms) step = timeout_ms - elapsed;
76 nros_cpp_ret_t ret = nros_cpp_spin_once(executor_handle, static_cast<int32_t>(step));
77 // Transient conditions: keep polling. Anything else propagates.
78 if (ret != 0 && ret != static_cast<nros_cpp_ret_t>(ErrorCode::Timeout) &&
79 ret != static_cast<nros_cpp_ret_t>(ErrorCode::TryAgain)) {
80 return Result(ret);
81 }
82 Result rn = try_next(out);
83 if (rn.ok()) return Result::success();
84 // TryAgain / NotInitialized / Error from try_next: keep polling
85 // unless we've hit a hard error that's not "no data yet".
86 if (rn.code() != ErrorCode::TryAgain && rn.code() != ErrorCode::NotInitialized) {
87 return rn;
88 }
89 elapsed += step;
90 }
91 return Result(ErrorCode::Timeout);
92 }
