25// ============================================================================
26// FFI Reentrancy Guard
27// ============================================================================
28
29/// Execute a closure with FFI reentrancy protection.
30///
31/// When the `ffi-sync` feature is enabled, wraps the closure in
32/// `critical_section::with()` to prevent concurrent access to zpico global
33/// state from mixed-priority tasks or ISRs.
34///
35/// When the feature is disabled, this is a zero-cost passthrough.
36#[allow(dead_code)] // used only when a platform feature is enabled
37#[inline(always)]
38pub(crate) fn ffi_guard<R>(f: impl FnOnce() -> R) -> R {
39    #[cfg(feature = "ffi-sync")]
40    {
41        return critical_section::with(|_cs| f());
42    }
43    #[cfg(not(feature = "ffi-sync"))]
44    f()
45}

78/// Error type for shim operations
79#[derive(Debug, Clone, Copy, PartialEq, Eq)]
80pub enum ZpicoError {
81    /// Generic error
82    Generic,
83    /// Configuration error
84    Config,
85    /// Session error
86    Session,
87    /// Task creation error
88    Task,
89    /// Invalid key expression
90    KeyExpr,
91    /// Resource limit reached
92    Full,
93    /// Invalid handle
94    Invalid,
95    /// Publish error
96    Publish,
97    /// Session not open
98    NotOpen,
99    /// Query timeout (no reply received)
100    Timeout,
101}
102
103#[cfg(any(
104    feature = "platform-posix",
105    feature = "platform-zephyr",
106    feature = "platform-bare-metal",
107    feature = "platform-freertos",
108    feature = "platform-nuttx",
109    feature = "platform-threadx",
110    test
111))]
112impl ZpicoError {
113    fn from_code(code: i32) -> Self {
114        match code {
115            -1 => ZpicoError::Generic,
116            -2 => ZpicoError::Config,
117            -3 => ZpicoError::Session,
118            -4 => ZpicoError::Task,
119            -5 => ZpicoError::KeyExpr,
120            -6 => ZpicoError::Full,
121            -7 => ZpicoError::Invalid,
122            -8 => ZpicoError::Publish,
123            _ => ZpicoError::Generic,
124        }
125    }
126}

152/// A 16-byte Zenoh session ID
153#[derive(Clone, Copy, Debug, PartialEq, Eq)]
154pub struct ZenohId {
155    /// The raw 16-byte ID
156    pub id: [u8; 16],
157}
158
159impl ZenohId {
160    /// Create a new ZenohId from raw bytes
161    pub fn from_bytes(bytes: [u8; 16]) -> Self {
162        Self { id: bytes }
163    }
164
165    // `to_hex_string()` removed — was dead code requiring `alloc`.
166    // Use `to_hex_bytes()` instead (alloc-free, writes into caller-provided buffer).
167
168    /// Format the ID into a fixed-size buffer (for no_std)
169    ///
170    /// Returns the number of bytes written (always 32).
171    pub fn to_hex_bytes(&self, buf: &mut [u8; 32]) {
172        const HEX_CHARS: &[u8; 16] = b"0123456789abcdef";
173        // LSB-first order
174        for i in 0..16 {
175            let byte = self.id[15 - i];
176            buf[i * 2] = HEX_CHARS[(byte >> 4) as usize];
177            buf[i * 2 + 1] = HEX_CHARS[(byte & 0xf) as usize];
178        }
179    }
180}

328    ///
329    /// The locator should be a null-terminated string like `b"tcp/127.0.0.1:7447\0"`.

331    /// # Errors

337            let ret = unsafe { zpico_init(locator.as_ptr().cast()) };
338            if ret < 0 {

342            let ret = unsafe { zpico_open() };
343            if ret < 0 {
344                return Err(ZpicoError::from_code(ret));
345            }

348                _private: PhantomData,
349            })

354    ///
355    /// Byte slices for locator and mode must be null-terminated C strings.
356    ///
357    /// # Arguments
358    ///
359    /// * `locator` - Null-terminated locator (e.g., `b"tcp/127.0.0.1:7447\0"`), or `None` for peer mode
360    /// * `mode` - Null-terminated mode string (`b"client\0"` or `b"peer\0"`)

363    /// # Errors

391            }

393            let ret = unsafe { zpico_open() };
394            if ret < 0 {
395                return Err(ZpicoError::from_code(ret));
396            }
397

542    /// Combined poll and keepalive operation
543    ///
544    /// This is equivalent to calling `poll()` and performing any necessary
545    /// keepalive operations.
546    ///
547    /// # Arguments
548    ///
549    /// * `timeout_ms` - Maximum time to wait (0 = non-blocking)
550    ///
551    /// # Returns
552    ///
553    /// Number of events processed, or error
554    pub fn spin_once(&self, timeout_ms: u32) -> Result<i32> {
555        // When FFI guard is enabled, each `zpico_spin_once` call must
556        // run inside a `critical_section::with()` to serialise against
557        // mixed-priority tasks that also touch zenoh-pico globals.
558        //
559        // Phase 77.19 — the inner call is now `zpico_spin_once(remaining_ms)`
560        // instead of a busy `zpico_spin_once(0)` tight-loop. On platforms
561        // that have event-driven wake primitives (POSIX condvar, Zephyr
562        // condvar, FreeRTOS binary semaphore, NuttX `sem_timedwait` — all
563        // landed in 77.16 / 77.17), this returns immediately on data
564        // arrival instead of burning CPU until the deadline. On
565        // polled-smoltcp / polled-serial bare-metal the inner call is
566        // still a tight loop but the outer loop no longer adds a second
567        // layer of busy-waiting on top.
568        //
569        // The critical section is held for up to `remaining_ms` per
570        // iteration, which is fine today because the only active
571        // `ffi-sync` consumer is bare-metal RTIC mixed-priority, and
572        // those examples always call with `timeout_ms == 0` (single pass,
573        // never enters the loop). If a future consumer calls with
574        // `timeout_ms > 0` on a cortex-m RTIC target, this will hold
575        // IRQs off for that duration and should be revisited (split the
576        // wait and the zpico-state touch into separate CS regions).
577        #[cfg(feature = "ffi-sync")]
578        {
579            let ret = ffi_guard(|| unsafe { zpico_spin_once(0) });
580            if ret < 0 {
581                return Err(ZpicoError::from_code(ret));
582            }
583            if ret > 0 || timeout_ms == 0 {
584                return Ok(ret);
585            }
586            // Loop with guarded blocking spin_once calls until timeout
587            let mut clock = [0u8; 16];
588            unsafe { zpico_sys::zpico_clock_start(clock.as_mut_ptr()) };
589            loop {
590                let elapsed =
591                    unsafe { zpico_sys::zpico_clock_elapsed_ms_since(clock.as_mut_ptr()) };
592                let elapsed_u32 = elapsed as u32;
593                if elapsed_u32 >= timeout_ms {
594                    return Ok(0);
595                }
596                let remaining_ms = timeout_ms - elapsed_u32;
597                let ret = ffi_guard(|| unsafe { zpico_spin_once(remaining_ms) });
598                if ret < 0 {
599                    return Err(ZpicoError::from_code(ret));
600                }
601                if ret > 0 {
602                    return Ok(ret);
603                }
604            }
605        }
606        #[cfg(not(feature = "ffi-sync"))]
607        {
608            let ret = unsafe { zpico_spin_once(timeout_ms) };
609            if ret < 0 {
610                return Err(ZpicoError::from_code(ret));
611            }
612            Ok(ret)
613        }
614    }

697        attachment: Option<&[u8]>,
698    ) -> Result<()> {
699        let (att_ptr, att_len) = match attachment {
700            Some(att) => (att.as_ptr(), att.len()),
701            None => (core::ptr::null(), 0),
702        };
703
704        let ret = ffi_guard(|| unsafe {
705            zpico_query_reply(
706                queryable_handle,
707                keyexpr.as_ptr().cast(),
708                data.as_ptr(),
709                data.len(),
710                att_ptr,
711                att_len,
712            )
713        });
714        if ret < 0 {
715            return Err(ZpicoError::from_code(ret));
716        }
717        Ok(())
718    }
719
720    /// Start a non-blocking query (for async service client).
721    ///
722    /// Returns a slot handle on success that can be polled with [`get_check()`](Self::get_check).
723    pub fn get_start(&self, keyexpr: &[u8], payload: &[u8], timeout_ms: u32) -> Result<i32> {
724        let (payload_ptr, payload_len) = if payload.is_empty() {
725            (core::ptr::null(), 0)
726        } else {
727            (payload.as_ptr(), payload.len())
728        };
729
730        let ret = ffi_guard(|| unsafe {
731            zpico_sys::zpico_get_start(
732                keyexpr.as_ptr().cast(),
733                payload_ptr,
734                payload_len,
735                timeout_ms,
736            )
737        });
738
739        if ret < 0 {
740            return Err(ZpicoError::from_code(ret));
741        }
742
743        Ok(ret)
744    }
745
746    /// Check for a reply to a pending non-blocking query.
747    ///
748    /// Returns `Ok(Some(len))` when a reply has arrived, `Ok(None)` if still
749    /// pending, or `Err` on failure/timeout.
750    pub fn get_check(&self, handle: i32, reply_buf: &mut [u8]) -> Result<Option<usize>> {
751        let ret = ffi_guard(|| unsafe {
752            zpico_sys::zpico_get_check(handle, reply_buf.as_mut_ptr(), reply_buf.len())
753        });
754
755        if ret > 0 {
756            Ok(Some(ret as usize))
757        } else if ret == 0 {
758            Ok(None)
759        } else {
760            if ret == -9 {
761                return Err(ZpicoError::Timeout);
762            }
763            Err(ZpicoError::from_code(ret))
764        }
765    }
766
767    /// Start a non-blocking liveliness query.
768    ///
769    /// Returns a slot handle on success that can be polled with
770    /// [`liveliness_get_check()`](Self::liveliness_get_check).
771    /// `keyexpr` must be a null-terminated byte slice.
772    pub fn liveliness_get_start(&self, keyexpr: &[u8], timeout_ms: u32) -> Result<i32> {
773        let ret = ffi_guard(|| unsafe {
774            zpico_sys::zpico_liveliness_get_start(keyexpr.as_ptr().cast(), timeout_ms)
775        });
776
777        if ret < 0 {
778            return Err(ZpicoError::from_code(ret));
779        }
780        Ok(ret)
781    }
782
783    /// Poll a pending liveliness query.
784    ///
785    /// Returns `Ok(true)` once at least one matching liveliness token has
786    /// reported back, `Ok(false)` while the query is still in flight with
787    /// no replies yet. Returns `Err(ZpicoError::Timeout)` when the query
788    /// dropper fired without seeing any matching token (no server visible
789    /// within the timeout).
790    pub fn liveliness_get_check(&self, handle: i32) -> Result<bool> {
791        let ret =
792            ffi_guard(|| unsafe { zpico_sys::zpico_liveliness_get_check(handle) });
793
794        if ret == 1 {
795            Ok(true)
796        } else if ret == 0 {
797            Ok(false)
798        } else if ret == -9 {
799            Err(ZpicoError::Timeout)
800        } else {
801            Err(ZpicoError::from_code(ret))
802        }
803    }
804}

873    ///

878        let (att_ptr, att_len) = match attachment {
879            Some(att) => (att.as_ptr(), att.len()),
880            None => (core::ptr::null(), 0),
881        };
882
883        let ret = ffi_guard(|| unsafe {
884            zpico_publish_with_attachment(self.handle, data.as_ptr(), data.len(), att_ptr, att_len)
885        });
886        if ret < 0 {
887            return Err(ZpicoError::from_code(ret));
888        }
889        Ok(())
890    }
