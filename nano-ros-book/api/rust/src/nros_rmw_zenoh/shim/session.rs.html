19/// Zenoh session wrapping nros-rmw-zenoh Context
20///
21/// This session requires manual polling via `spin_once()` or `poll()`.
22/// There are no background threads.
23pub struct ZenohSession {
24    context: Context,
25    /// Phase 124.B.3 — executor wake callback. Installed by
26    /// `set_wake_callback`; non-null after the runtime has wired
27    /// the executor through the cffi vtable. Invoked by
28    /// `drive_io` after work was observed; the runtime cb does
29    /// flag-write + condvar-signal atomically so the executor
30    /// wakes from `wake_cv` instead of polling on a deadline.
31    wake_cb: core::sync::atomic::AtomicPtr<core::ffi::c_void>,
32    wake_ctx: core::sync::atomic::AtomicPtr<core::ffi::c_void>,
33}
34
35impl ZenohSession {
36    /// Create a new shim session with the given configuration
37    ///
38    /// # Arguments
39    ///
40    /// * `config` - Transport configuration with locator and mode
41    ///
42    /// # Returns
43    ///
44    /// A new session or error if connection fails
45    pub fn new(config: &TransportConfig) -> Result<Self, TransportError> {
46        // Build the locator string with null terminator
47        let locator = match (&config.mode, config.locator) {
48            (SessionMode::Client, Some(loc)) => {
49                // Create null-terminated locator
50                let mut buf = [0u8; LOCATOR_BUFFER_SIZE];
51                let bytes = loc.as_bytes();
52                if bytes.len() >= buf.len() {
53                    return Err(TransportError::InvalidArgument);
54                }
55                buf[..bytes.len()].copy_from_slice(bytes);
56                buf[bytes.len()] = 0; // Null terminator
57                buf
58            }
59            (SessionMode::Client, None) => {
60                return Err(TransportError::InvalidArgument);
61            }
62            (SessionMode::Peer, _) => {
63                // Peer mode - pass null locator
64                [0u8; LOCATOR_BUFFER_SIZE]
65            }
66        };
67
68        // Build mode string
69        let mode: &[u8] = match config.mode {
70            SessionMode::Client => b"client\0",
71            SessionMode::Peer => b"peer\0",
72        };
73
74        // Build null-terminated property strings on the stack
75        // Each key/value is at most 64 bytes
76        let mut key_bufs = [[0u8; CONFIG_PROPERTY_SIZE]; MAX_SESSION_PROPERTIES];
77        let mut val_bufs = [[0u8; CONFIG_PROPERTY_SIZE]; MAX_SESSION_PROPERTIES];
78        let mut c_props: [crate::zpico::zpico_property_t; MAX_SESSION_PROPERTIES] =
79            unsafe { core::mem::zeroed() };
80
81        let mut prop_count = 0usize;
82
83        // Copy explicit properties from config
84        for i in 0..config.properties.len().min(MAX_SESSION_PROPERTIES) {
85            let (key, value) = config.properties[i];
86            let key_bytes = key.as_bytes();
87            let val_bytes = value.as_bytes();
88            if key_bytes.len() >= CONFIG_PROPERTY_SIZE || val_bytes.len() >= CONFIG_PROPERTY_SIZE {
89                continue; // Skip oversized properties
90            }
91            key_bufs[prop_count][..key_bytes.len()].copy_from_slice(key_bytes);
92            key_bufs[prop_count][key_bytes.len()] = 0;
93            val_bufs[prop_count][..val_bytes.len()].copy_from_slice(val_bytes);
94            val_bufs[prop_count][val_bytes.len()] = 0;
95            c_props[prop_count] = crate::zpico::zpico_property_t {
96                key: key_bufs[prop_count].as_ptr().cast(),
97                value: val_bufs[prop_count].as_ptr().cast(),
98            };
99            prop_count += 1;
100        }
101
102        // Read ZENOH_* env vars as defaults (explicit properties take precedence)
103        #[cfg(feature = "std")]
104        {
105            let env_mappings: &[(&str, &str)] = &[
106                ("ZENOH_MULTICAST_SCOUTING", "multicast_scouting"),
107                ("ZENOH_SCOUTING_TIMEOUT", "scouting_timeout_ms"),
108                ("ZENOH_LISTEN", "listen"),
109                // TLS configuration
110                ("ZENOH_TLS_ROOT_CA_CERTIFICATE", "root_ca_certificate"),
111                (
112                    "ZENOH_TLS_ROOT_CA_CERTIFICATE_BASE64",
113                    "root_ca_certificate_base64",
114                ),
115                ("ZENOH_TLS_VERIFY_NAME_ON_CONNECT", "verify_name_on_connect"),
116            ];
117            for &(env_name, prop_key) in env_mappings {
118                if let Ok(val) = std::env::var(env_name) {
119                    let already_set = config.properties.iter().any(|(k, _)| *k == prop_key);
120                    if !already_set && prop_count < MAX_SESSION_PROPERTIES {
121                        let key_bytes = prop_key.as_bytes();
122                        let val_bytes = val.as_bytes();
123                        if key_bytes.len() < CONFIG_PROPERTY_SIZE
124                            && val_bytes.len() < CONFIG_PROPERTY_SIZE
125                        {
126                            key_bufs[prop_count][..key_bytes.len()].copy_from_slice(key_bytes);
127                            key_bufs[prop_count][key_bytes.len()] = 0;
128                            val_bufs[prop_count][..val_bytes.len()].copy_from_slice(val_bytes);
129                            val_bufs[prop_count][val_bytes.len()] = 0;
130                            c_props[prop_count] = crate::zpico::zpico_property_t {
131                                key: key_bufs[prop_count].as_ptr().cast(),
132                                value: val_bufs[prop_count].as_ptr().cast(),
133                            };
134                            prop_count += 1;
135                        }
136                    }
137                }
138            }
139        }
140
141        let locator_opt = if config.mode == SessionMode::Peer && config.locator.is_none() {
142            None
143        } else {
144            Some(locator.as_slice())
145        };
146
147        let context = Context::with_config(locator_opt, mode, &c_props[..prop_count])
148            .map_err(TransportError::from)?;
149
150        // Register the reply waker callback for async service client support
151        super::service::register_reply_waker();
152
153        Ok(Self {
154            context,
155            wake_cb: core::sync::atomic::AtomicPtr::new(core::ptr::null_mut()),
156            wake_ctx: core::sync::atomic::AtomicPtr::new(core::ptr::null_mut()),
157        })
158    }

172    /// Combined poll and keepalive operation
173    ///
174    /// This is the recommended way to drive the session. Call this
175    /// periodically (e.g., every 10ms) from your main loop or RTIC task.
176    ///
177    /// # Arguments
178    ///
179    /// * `timeout_ms` - Maximum time to wait (0 = non-blocking)
180    ///
181    /// # Returns
182    ///
183    /// Number of events processed, or error

185        self.context
186            .spin_once(timeout_ms)
187            .map_err(TransportError::from)
188    }

203    /// Declare a liveliness token for ROS 2 discovery
204    ///
205    /// This creates a liveliness token at the given key expression,
206    /// allowing ROS 2 nodes using rmw_zenoh to discover this entity.
207    ///
208    /// The key expression should be null-terminated.
209    pub fn declare_liveliness(&self, keyexpr: &[u8]) -> Result<LivelinessToken, TransportError> {
210        self.context
211            .declare_liveliness(keyexpr)
212            .map_err(TransportError::from)
213    }

215    /// Declare a node liveliness token for ROS 2 participant discovery
216    ///
217    /// Creates an NN liveliness token so ROS 2 tools (`ros2 node list`)
218    /// can discover this node. The token is kept alive for the session lifetime.
219    pub fn declare_node_liveliness(
220        &self,
221        domain_id: u32,
222        namespace: &str,
223        node_name: &str,
224    ) -> Option<LivelinessToken> {
225        self.declare_entity_liveliness(|zid| {
226            Ros2Liveliness::node_keyexpr::<256>(domain_id, zid, namespace, node_name)
227        })
228    }
229
230    /// Helper: build a liveliness keyexpr using a closure and declare it.
231    fn declare_entity_liveliness(
232        &self,
233        build_keyexpr: impl FnOnce(&ZenohId) -> heapless::String<256>,
234    ) -> Option<LivelinessToken> {
235        let zid = self.context.zid().ok()?;
236        let keyexpr = build_keyexpr(&zid);
237
238        #[cfg(feature = "std")]
239        log::debug!("liveliness keyexpr: {}", keyexpr.as_str());
240
241        let mut buf = [0u8; 257];
242        let bytes = keyexpr.as_bytes();
243        if bytes.len() < buf.len() {
244            buf[..bytes.len()].copy_from_slice(bytes);
245            buf[bytes.len()] = 0;
246            self.context.declare_liveliness(&buf[..=bytes.len()]).ok()
247        } else {
248            None
249        }
250    }

343    fn drive_io(&mut self, timeout_ms: i32) -> Result<(), Self::Error> {
344        let res = self.spin_once(timeout_ms as u32);
345        // Phase 124.B.3 — when zenoh's spin_once observed any work
346        // (n > 0), call the runtime-supplied wake callback so the
347        // executor's `wake_cv` is signalled (flag-write +
348        // condvar-signal happen atomically inside the cb).
349        // Best-effort: if the cb hasn't been installed yet the
350        // executor still drains via its deadline-bound cv-wait.
351        if matches!(res, Ok(n) if n > 0) {
352            let cb = self.wake_cb.load(core::sync::atomic::Ordering::Acquire);
353            if !cb.is_null() {
354                let ctx = self.wake_ctx.load(core::sync::atomic::Ordering::Acquire);
355                // SAFETY: cb was installed via `set_wake_callback`
356                // and points at a runtime-owned function; ctx
357                // points at WakeCtx allocated for the executor's
358                // lifetime.
359                let f: unsafe extern "C" fn(*mut core::ffi::c_void) =
360                    unsafe { core::mem::transmute(cb) };
361                unsafe { f(ctx) };
362            }
363        }
364        res.map(|_| ())
365    }
366
367    fn set_wake_callback(
368        &mut self,
369        cb: Option<unsafe extern "C" fn(ctx: *mut core::ffi::c_void)>,
370        ctx: *mut core::ffi::c_void,
371    ) {
372        let cb_ptr = cb
373            .map(|f| f as *mut core::ffi::c_void)
374            .unwrap_or(core::ptr::null_mut());
375        self.wake_cb
376            .store(cb_ptr, core::sync::atomic::Ordering::Release);
377        self.wake_ctx
378            .store(ctx, core::sync::atomic::Ordering::Release);
379    }

381    /// Phase 110.0 — bound the executor's `drive_io` wait against
382    /// zenoh-pico's transport keepalive interval.
383    ///
384    /// zenoh-pico does not expose its internal "next keepalive
385    /// timestamp" through FFI, so the shim returns a conservative
386    /// upper-bound: `Z_TRANSPORT_LEASE / Z_TRANSPORT_LEASE_EXPIRE_FACTOR`.
387    /// With the upstream defaults (lease = 10 000 ms, factor = 3) that
388    /// caps wake-late to ~3.3 s on a quiet link — the runtime never
389    /// blocks longer than one keepalive interval before returning
390    /// control to the executor. Tracking the precise per-call
391    /// timestamp would need a hook in `_z_send_keep_alive`; out of
392    /// scope for the v1 surface.
393    fn next_deadline_ms(&self) -> Option<u32> {
394        // Z_TRANSPORT_LEASE = 10000 ms, Z_TRANSPORT_LEASE_EXPIRE_FACTOR = 3
395        const ZENOH_KEEPALIVE_INTERVAL_MS: u32 = 10_000 / 3;
396        Some(ZENOH_KEEPALIVE_INTERVAL_MS)
397    }

399    fn ping_session(&mut self, _timeout_ms: i32) -> Result<(), Self::Error> {
400        // Phase 124.F.2 — zenoh-pico's closest match to a true ping
401        // is `zp_send_keep_alive`. Fire one frame; success means the
402        // transport accepted it (TCP / serial / shm send returned OK,
403        // i.e. the link is still alive from the local side). Failure
404        // surfaces as `Timeout` per the 124.F.1 semantics so callers
405        // can tear down + re-open the session on a dead link.
406        //
407        // The `timeout_ms` argument is ignored — the underlying call
408        // is synchronous and non-blocking. We don't honour the budget
409        // because the call returns within microseconds either way.
410        // True round-trip ping would need a `z_send_ping` API that
411        // zenoh-pico hasn't yet exposed; deferred to a follow-up
412        // when upstream lands one.
413        let rc = unsafe { zpico_sys::zpico_send_keep_alive() };
414        if rc == 0 {
415            Ok(())
416        } else {
417            Err(TransportError::Timeout)
418        }
419    }

421    fn supported_qos_policies(&self) -> nros_rmw::QosPolicyMask {
422        // Phase 108.B/C — zenoh-pico's wire protocol has no native
423        // DDS QoS, so the shim emulates everything:
424        // - Reliability maps to zenoh congestion-control (CORE).
425        // - Durability VOLATILE / History / Depth honoured at the
426        //   subscriber buffer level (CORE).
427        // - DEADLINE: clock-based check at try_recv_raw (sub) /
428        //   publish_raw (pub). 108.C.zenoh.2.
429        // - LIFESPAN: subscriber filters samples whose attachment
430        //   timestamp is older than `now - lifespan_ms`. 108.C.zenoh.3.
431        // - LIVELINESS_AUTOMATIC: zenoh runtime declares the token
432        //   automatically when the publisher is created
433        //   (`Ros2Liveliness::publisher_keyexpr`); subscribers track
434        //   alive-state via a periodic poll of the wildcard liveliness
435        //   keyexpr. Per-publisher count surfaced via
436        //   `zpico_liveliness_get_count` (108.C.zenoh.4-followup).
437        // - LIVELINESS_MANUAL_BY_TOPIC / MANUAL_BY_NODE: shim-side
438        //   keepalive timer. `Publisher::assert_liveliness()`
439        //   refreshes the lease; `publish_raw` checks for expiry and
440        //   fires `LivelinessLost` rate-limited to ≤ 1 per lease.
441        //   (108.C.zenoh.4-followup).
442        // - LIVELINESS_LEASE: caller-supplied lease duration honoured
443        //   for all liveliness kinds.
444        use nros_rmw::QosPolicyMask;
445        QosPolicyMask::CORE
446            | QosPolicyMask::DEADLINE
447            | QosPolicyMask::LIFESPAN
448            | QosPolicyMask::LIVELINESS_AUTOMATIC
449            | QosPolicyMask::LIVELINESS_MANUAL_BY_TOPIC
450            | QosPolicyMask::LIVELINESS_MANUAL_BY_NODE
451            | QosPolicyMask::LIVELINESS_LEASE
452    }
