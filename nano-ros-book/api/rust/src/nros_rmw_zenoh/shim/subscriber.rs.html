27/// Shared buffer for subscriber callbacks
28///
29/// This buffer stores the most recent message received by the subscriber,
30/// including the RMW attachment data for MessageInfo support.
31/// The callback writes to this buffer, and `try_recv_raw` reads from it.
32pub(super) struct SubscriberBuffer {
33    /// Buffer for received payload data (statically allocated)
34    pub(super) data: [u8; SUBSCRIBER_BUFFER_SIZE],
35    /// Buffer for received attachment data (33 or 37 bytes depending on safety-e2e)
36    pub(super) attachment: [u8; SUBSCRIBER_ATTACHMENT_BUF_SIZE],
37    /// Flag indicating new data is available
38    pub(super) has_data: AtomicBool,
39    /// Flag indicating the incoming message exceeded the buffer capacity.
40    /// Set by the callback when `len > data.len()`. Checked by `try_recv_raw`
41    /// which returns `Err(MessageTooLarge)` and clears this flag.
42    pub(super) overflow: AtomicBool,
43    /// Flag indicating a reader is currently accessing this buffer.
44    /// Set by `try_recv_raw` / `process_raw_in_place` before reading, cleared
45    /// after. The callback checks this flag and drops the message if locked,
46    /// preventing a data race where the callback overwrites the buffer mid-read.
47    pub(super) locked: AtomicBool,
48    /// Length of valid payload data
49    pub(super) len: AtomicUsize,
50    /// Length of valid attachment data
51    pub(super) attachment_len: AtomicUsize,
52    /// Async waker — registered by `Future::poll()`, woken from callback
53    /// when data arrives. Enables event-driven async without busy-polling.
54    pub(super) waker: AtomicWaker,
55}
56
57impl SubscriberBuffer {
58    pub(super) const fn new() -> Self {
59        Self {
60            data: [0u8; SUBSCRIBER_BUFFER_SIZE],
61            attachment: [0u8; SUBSCRIBER_ATTACHMENT_BUF_SIZE],
62            has_data: AtomicBool::new(false),
63            overflow: AtomicBool::new(false),
64            locked: AtomicBool::new(false),
65            len: AtomicUsize::new(0),
66            attachment_len: AtomicUsize::new(0),
67            waker: AtomicWaker::new(),
68        }
69    }
70}

72/// Static buffers for subscribers.
73///
74/// Count matches `ZPICO_MAX_SUBSCRIBERS` from zpico-sys (the C shim
75/// allocates the same number of subscriber entries). We use static buffers
76/// because the shim callback mechanism requires a static context pointer.
77static mut SUBSCRIBER_BUFFERS: [SubscriberBuffer; ZPICO_MAX_SUBSCRIBERS] =
78    [const { SubscriberBuffer::new() }; ZPICO_MAX_SUBSCRIBERS];
79
80/// Next available buffer index
81pub(super) static NEXT_BUFFER_INDEX: AtomicUsize = AtomicUsize::new(0);

83// ============================================================================
84// SubscriberBufferRef — safe accessor wrapper
85// ============================================================================

87/// Safe accessor for a statically-allocated subscriber buffer.
88///
89/// Encapsulates the `unsafe` access to `SUBSCRIBER_BUFFERS` by validating
90/// the index once at construction time. Subsequent accesses via [`get()`]
91/// are safe because the index is guaranteed in-bounds.
92///
93/// # Safety invariant
94///
95/// `SUBSCRIBER_BUFFERS` is a module-level `static mut` with a fixed address
96/// and element count equal to `ZPICO_MAX_SUBSCRIBERS`. The index is validated
97/// at construction and never changes, so every `get()` / `get_mut()` call
98/// dereferences a valid, in-bounds element.
99pub(super) struct SubscriberBufferRef {
100    index: usize,
101}
102
103impl SubscriberBufferRef {
104    /// Create a new buffer reference with bounds validation.
105    ///
106    /// # Panics
107    ///
108    /// Panics if `index >= ZPICO_MAX_SUBSCRIBERS`.
109    pub(super) fn new(index: usize) -> Self {
110        assert!(
111            index < ZPICO_MAX_SUBSCRIBERS,
112            "subscriber buffer index out of bounds: {index} >= {ZPICO_MAX_SUBSCRIBERS}"
113        );
114        Self { index }
115    }
116
117    /// Get an immutable reference to the subscriber buffer.
118    ///
119    /// Safety is guaranteed by the bounds check at construction time.
120    /// All shared fields use atomic types, preventing data races.
121    pub(super) fn get(&self) -> &SubscriberBuffer {
122        // Safety: index was validated at construction time.
123        // SUBSCRIBER_BUFFERS is a module-level static with fixed address.
124        unsafe { &SUBSCRIBER_BUFFERS[self.index] }
125    }
126
127    /// Get a mutable reference to the subscriber buffer.
128    ///
129    /// Only called from callbacks, which are invoked synchronously
130    /// (single-threaded) by zenoh-pico — no concurrent mutable access.
131    pub(super) fn get_mut(&mut self) -> &mut SubscriberBuffer {
132        // Safety: index was validated at construction time.
133        // Mutable access is only used by callbacks invoked synchronously
134        // by zenoh-pico, so there are no concurrent mutable accesses.
135        unsafe { &mut SUBSCRIBER_BUFFERS[self.index] }
136    }
137}
138
139/// Notify callback invoked by the C shim after direct-write to the static buffer.
140///
141/// The payload is already in `SUBSCRIBER_BUFFERS[buffer_index].data`. This callback
142/// only stores the length, attachment, and signals data availability.
143extern "C" fn subscriber_notify_callback(
144    len: usize,
145    attachment: *const u8,
146    attachment_len: usize,
147    ctx: *mut core::ffi::c_void,
148) {
149    let buffer_index = ctx as usize;
150    if buffer_index >= ZPICO_MAX_SUBSCRIBERS {
151        return;
152    }
153
154    let mut buf_ref = SubscriberBufferRef {
155        index: buffer_index,
156    };
157    let buffer = buf_ref.get_mut();
158
159    if len > buffer.data.len() {
160        // Overflow: the C shim called us with the oversized length so we can flag it.
161        buffer.overflow.store(true, Ordering::Release);
162        buffer.has_data.store(true, Ordering::Release);
163    } else {
164        // Payload already written by C shim — just store metadata
165        buffer.overflow.store(false, Ordering::Release);
166        buffer.len.store(len, Ordering::Release);
167
168        // Copy attachment data if present
169        if !attachment.is_null() && attachment_len > 0 {
170            let att_copy_len = attachment_len.min(buffer.attachment.len());
171            // Safety: attachment pointer is valid for att_copy_len bytes (from C shim)
172            unsafe {
173                core::ptr::copy_nonoverlapping(
174                    attachment,
175                    buffer.attachment.as_mut_ptr(),
176                    att_copy_len,
177                );
178            }
179            buffer.attachment_len.store(att_copy_len, Ordering::Release);
180        } else {
181            buffer.attachment_len.store(0, Ordering::Release);
182        }
183
184        buffer.has_data.store(true, Ordering::Release);
185    }
186
187    // Wake any async task waiting for data on this subscriber
188    buffer.waker.wake();
189
190    // Wake the executor spin loop (if waiting)
191    #[cfg(feature = "std")]
192    signal_executor_wake();
193}

199/// Zenoh subscriber wrapping nros-rmw-zenoh ZenohSubscriber
200pub struct ZenohSubscriber {
201    /// The subscriber handle (kept alive to maintain subscription)
202    _subscriber: crate::zpico::Subscriber<'static>,
203    /// Safe accessor for the static subscriber buffer
204    buf: SubscriberBufferRef,
205    /// Liveliness token for ROS 2 graph discovery (kept alive for subscriber lifetime)
206    _liveliness: Option<super::LivelinessToken>,
207    /// E2E safety validator (tracks sequence numbers, validates CRC)
208    #[cfg(feature = "safety-e2e")]
209    safety_validator: nros_rmw::SafetyValidator,
210    /// Phantom to indicate we don't own the buffer
211    _phantom: PhantomData<()>,
212}
213
214impl ZenohSubscriber {
215    /// Create a new subscriber for the given topic
216    pub fn new(
217        context: &Context,
218        topic: &nros_rmw::TopicInfo,
219        liveliness: Option<super::LivelinessToken>,
220    ) -> Result<Self, TransportError> {
221        // Allocate a buffer index
222        let buffer_index = NEXT_BUFFER_INDEX.fetch_add(1, Ordering::SeqCst);
223        if buffer_index >= ZPICO_MAX_SUBSCRIBERS {
224            // Roll back and return error
225            NEXT_BUFFER_INDEX.fetch_sub(1, Ordering::SeqCst);
226            return Err(TransportError::SubscriberCreationFailed);
227        }
228
229        let mut buf = SubscriberBufferRef::new(buffer_index);
230
231        // Generate the topic key with wildcard for type hash
232        let key: heapless::String<KEYEXPR_STRING_SIZE> = topic.to_key_wildcard();
233
234        #[cfg(feature = "std")]
235        log::debug!("Subscriber data keyexpr: {}", key.as_str());
236
237        // Create null-terminated keyexpr
238        let mut keyexpr_buf = [0u8; KEYEXPR_BUFFER_SIZE];
239        let bytes = key.as_bytes();
240        if bytes.len() >= keyexpr_buf.len() {
241            return Err(TransportError::TopicNameInvalid);
242        }
243        keyexpr_buf[..bytes.len()].copy_from_slice(bytes);
244        keyexpr_buf[bytes.len()] = 0;
245
246        // Create subscriber with direct-write: the C shim reads payload directly
247        // into SUBSCRIBER_BUFFERS[buffer_index].data via z_bytes_reader_read(),
248        // avoiding the z_bytes_to_slice() malloc.
249        let subscriber = unsafe {
250            let buffer = buf.get_mut();
251            let buf_ptr = buffer.data.as_mut_ptr();
252            let buf_capacity = buffer.data.len();
253            // AtomicBool is guaranteed to have the same in-memory representation
254            // as bool on all Rust targets (size 1, align 1). The C shim reads
255            // this via __atomic_load_n(ptr, __ATOMIC_ACQUIRE), which requires a
256            // pointer to the underlying bool storage — hence the cast.
257            let locked_ptr = buffer.locked.as_ptr() as *const bool;
258            let sub_result = context.declare_subscriber_direct_write_raw(
259                &keyexpr_buf,
260                buf_ptr,
261                buf_capacity,
262                locked_ptr,
263                subscriber_notify_callback,
264                buffer_index as *mut core::ffi::c_void,
265            );
266            match sub_result {
267                Ok(s) => core::mem::transmute::<
268                    crate::zpico::Subscriber<'_>,
269                    crate::zpico::Subscriber<'static>,
270                >(s),
271                Err(e) => return Err(TransportError::from(e)),
272            }
273        };
274
275        Ok(Self {
276            _subscriber: subscriber,
277            buf,
278            _liveliness: liveliness,
279            #[cfg(feature = "safety-e2e")]
280            safety_validator: nros_rmw::SafetyValidator::new(),
281            _phantom: PhantomData,
282        })
283    }
284}

286impl ZenohSubscriber {
287    /// Try to receive a validated message with E2E integrity status.
288    ///
289    /// Checks CRC-32 integrity and sequence continuity. Returns
290    /// `(payload_len, IntegrityStatus)` so the caller can decide whether
291    /// to trust the data.
292    ///
293    /// The payload bytes are written to `buf[..len]`.
294    #[cfg(feature = "safety-e2e")]
295    pub fn try_recv_validated(
296        &mut self,
297        buf: &mut [u8],
298    ) -> Result<Option<(usize, nros_rmw::IntegrityStatus)>, TransportError> {
299        let buffer = self.buf.get();
300
301        if !buffer.has_data.load(Ordering::Acquire) {
302            return Ok(None);
303        }
304
305        // Check for overflow
306        if buffer.overflow.load(Ordering::Acquire) {
307            buffer.overflow.store(false, Ordering::Release);
308            buffer.has_data.store(false, Ordering::Release);
309            return Err(TransportError::MessageTooLarge);
310        }
311
312        let len = buffer.len.load(Ordering::Acquire);
313        if len > buf.len() {
314            buffer.has_data.store(false, Ordering::Release);
315            return Err(TransportError::BufferTooSmall);
316        }
317
318        // Lock buffer to prevent callback from overwriting during copy
319        buffer.locked.store(true, Ordering::Release);
320
321        // Copy payload data
322        // Safety: data is valid up to len bytes, buffer is locked
323        unsafe {
324            core::ptr::copy_nonoverlapping(buffer.data.as_ptr(), buf.as_mut_ptr(), len);
325        }
326
327        // Parse attachment for sequence number and CRC
328        let attachment_len = buffer.attachment_len.load(Ordering::Acquire);
329        let (message_seq, crc_valid) = if attachment_len >= RMW_ATTACHMENT_SIZE {
330            // Extract sequence number (bytes 0..8, LE)
331            let att = &buffer.attachment;
332            let seq = i64::from_le_bytes([
333                att[0], att[1], att[2], att[3], att[4], att[5], att[6], att[7],
334            ]);
335
336            // Check for CRC (bytes 33..37)
337            let crc_result = if attachment_len >= RMW_ATTACHMENT_SIZE + SAFETY_CRC_SIZE {
338                let received_crc = u32::from_le_bytes([
339                    att[RMW_ATTACHMENT_SIZE],
340                    att[RMW_ATTACHMENT_SIZE + 1],
341                    att[RMW_ATTACHMENT_SIZE + 2],
342                    att[RMW_ATTACHMENT_SIZE + 3],
343                ]);
344                let computed_crc = nros_rmw::crc32(&buf[..len]);
345                Some(received_crc == computed_crc)
346            } else {
347                // No CRC in attachment (sender doesn't have safety-e2e)
348                None
349            };
350
351            (seq, crc_result)
352        } else {
353            // No attachment at all — cannot validate
354            (0, None)
355        };
356
357        buffer.locked.store(false, Ordering::Release);
358        buffer.has_data.store(false, Ordering::Release);
359
360        let status = self.safety_validator.validate(message_seq, crc_valid);
361        Ok(Some((len, status)))
362    }
363
364    /// Try to receive raw data along with message info from attachment
365    ///
366    /// Returns `Ok(Some((len, info)))` if data is available, where:
367    /// - `len` is the number of bytes written to the buffer
368    /// - `info` is the parsed message info (if attachment was present)
369    ///
370    /// Returns `Ok(None)` if no data is available.
371    pub fn try_recv_with_info(
372        &mut self,
373        buf: &mut [u8],
374    ) -> Result<Option<(usize, Option<MessageInfo>)>, TransportError> {
375        let buffer = self.buf.get();
376
377        if !buffer.has_data.load(Ordering::Acquire) {
378            return Ok(None);
379        }
380
381        // Check for overflow (message exceeded static buffer capacity)
382        if buffer.overflow.load(Ordering::Acquire) {
383            buffer.overflow.store(false, Ordering::Release);
384            buffer.has_data.store(false, Ordering::Release);
385            return Err(TransportError::MessageTooLarge);
386        }
387
388        let len = buffer.len.load(Ordering::Acquire);
389        if len > buf.len() {
390            // Clear has_data to avoid permanently stuck subscription — the oversized
391            // message is dropped, but the subscription recovers on the next message.
392            buffer.has_data.store(false, Ordering::Release);
393            return Err(TransportError::BufferTooSmall);
394        }
395
396        // Lock buffer to prevent callback from overwriting during copy
397        buffer.locked.store(true, Ordering::Release);
398
399        // Copy payload data
400        // Safety: Data is valid up to len bytes, buffer is locked
401        unsafe {
402            core::ptr::copy_nonoverlapping(buffer.data.as_ptr(), buf.as_mut_ptr(), len);
403        }
404
405        // Parse attachment if present
406        let attachment_len = buffer.attachment_len.load(Ordering::Acquire);
407        let message_info = if attachment_len > 0 {
408            let attachment_slice = &buffer.attachment[..attachment_len];
409            MessageInfo::from_attachment(attachment_slice)
410        } else {
411            None
412        };
413
414        buffer.locked.store(false, Ordering::Release);
415        buffer.has_data.store(false, Ordering::Release);
416
417        Ok(Some((len, message_info)))
418    }
419
420    /// Process the received message in-place with message info, without copying.
421    ///
422    /// Calls `f` with a reference to the raw CDR bytes and optional message info.
423    /// The buffer is locked during `f`.
424    ///
425    /// Returns `Ok(true)` if a message was available and `f` was called.
426    pub fn process_raw_in_place_with_info(
427        &mut self,
428        f: impl FnOnce(&[u8], Option<MessageInfo>),
429    ) -> Result<bool, TransportError> {
430        let buffer = self.buf.get();
431
432        if !buffer.has_data.load(Ordering::Acquire) {
433            return Ok(false);
434        }
435
436        if buffer.overflow.load(Ordering::Acquire) {
437            buffer.overflow.store(false, Ordering::Release);
438            buffer.has_data.store(false, Ordering::Release);
439            return Err(TransportError::MessageTooLarge);
440        }
441
442        let len = buffer.len.load(Ordering::Acquire);
443
444        buffer.locked.store(true, Ordering::Release);
445
446        // Parse attachment while locked (attachment is small: 33-37 bytes)
447        let attachment_len = buffer.attachment_len.load(Ordering::Acquire);
448        let message_info = if attachment_len > 0 {
449            let attachment_slice = &buffer.attachment[..attachment_len];
450            MessageInfo::from_attachment(attachment_slice)
451        } else {
452            None
453        };
454
455        f(&buffer.data[..len], message_info);
456
457        buffer.locked.store(false, Ordering::Release);
458        buffer.has_data.store(false, Ordering::Release);
459
460        Ok(true)
461    }
462}

464impl Subscriber for ZenohSubscriber {
465    type Error = TransportError;
466
467    fn has_data(&self) -> bool {
468        self.buf.get().has_data.load(Ordering::Acquire)
469    }
470
471    fn register_waker(&self, waker: &core::task::Waker) {
472        self.buf.get().waker.register(waker);
473    }
474
475    fn try_recv_raw(&mut self, buf: &mut [u8]) -> Result<Option<usize>, Self::Error> {
476        let buffer = self.buf.get();
477
478        if !buffer.has_data.load(Ordering::Acquire) {
479            return Ok(None);
480        }
481
482        // Check for overflow (message exceeded static buffer capacity)
483        if buffer.overflow.load(Ordering::Acquire) {
484            buffer.overflow.store(false, Ordering::Release);
485            buffer.has_data.store(false, Ordering::Release);
486            return Err(TransportError::MessageTooLarge);
487        }
488
489        let len = buffer.len.load(Ordering::Acquire);
490        if len > buf.len() {
491            // Clear has_data to avoid permanently stuck subscription — the oversized
492            // message is dropped, but the subscription recovers on the next message.
493            buffer.has_data.store(false, Ordering::Release);
494            return Err(TransportError::BufferTooSmall);
495        }
496
497        // Lock buffer to prevent callback from overwriting during copy
498        buffer.locked.store(true, Ordering::Release);
499
500        // Copy data
501        // Safety: Data is valid up to len bytes, buffer is locked
502        unsafe {
503            core::ptr::copy_nonoverlapping(buffer.data.as_ptr(), buf.as_mut_ptr(), len);
504        }
505
506        buffer.locked.store(false, Ordering::Release);
507        buffer.has_data.store(false, Ordering::Release);
508
509        Ok(Some(len))
510    }
511
512    fn process_raw_in_place(&mut self, f: impl FnOnce(&[u8])) -> Result<bool, Self::Error> {
513        let buffer = self.buf.get();
514
515        if !buffer.has_data.load(Ordering::Acquire) {
516            return Ok(false);
517        }
518
519        if buffer.overflow.load(Ordering::Acquire) {
520            buffer.overflow.store(false, Ordering::Release);
521            buffer.has_data.store(false, Ordering::Release);
522            return Err(TransportError::MessageTooLarge);
523        }
524
525        let len = buffer.len.load(Ordering::Acquire);
526
527        // Lock buffer, process in-place, then unlock
528        buffer.locked.store(true, Ordering::Release);
529        f(&buffer.data[..len]);
530        buffer.locked.store(false, Ordering::Release);
531        buffer.has_data.store(false, Ordering::Release);
532
533        Ok(true)
534    }
535
536    fn try_recv_raw_with_info(
537        &mut self,
538        buf: &mut [u8],
539    ) -> Result<Option<(usize, Option<nros_core::MessageInfo>)>, Self::Error> {
540        // Delegate to the inherent method which parses the zenoh attachment
541        match self.try_recv_with_info(buf)? {
542            Some((len, zenoh_info)) => {
543                let core_info = zenoh_info.map(|zi| {
544                    let mut info = nros_core::MessageInfo::new();
545                    info.set_publication_sequence_number(zi.sequence_number);
546                    info.set_source_timestamp(nros_core::Time::from_nanos(zi.timestamp_ns));
547                    info.set_publisher_gid(zi.publisher_gid);
548                    info
549                });
550                Ok(Some((len, core_info)))
551            }
552            None => Ok(None),
553        }
554    }
555
556    #[cfg(feature = "safety-e2e")]
557    fn try_recv_validated(
558        &mut self,
559        buf: &mut [u8],
560    ) -> Result<Option<(usize, nros_rmw::IntegrityStatus)>, Self::Error> {
561        // Delegate to the inherent safety validation method
562        ZenohSubscriber::try_recv_validated(self, buf)
563    }
564
565    fn deserialization_error(&self) -> Self::Error {
566        TransportError::DeserializationError
567    }
568}

574#[cfg(feature = "lending")]
575mod borrowing {
576    use super::*;
577    use core::sync::atomic::Ordering as CoreOrdering;
578
579    /// Backend-lent read-only view into the subscriber's static receive
580    /// buffer. Holds the buffer's `locked` flag for the lifetime of the
581    /// view so the C-side notify callback can't overwrite the bytes
582    /// while the user is reading them. On drop the lock is released and
583    /// `has_data` cleared (consume-on-borrow semantics, matching
584    /// `try_recv_raw`).
585    pub struct ZenohView<'a> {
586        bytes: &'a [u8],
587        buffer: &'a SubscriberBuffer,
588    }
589
590    impl<'a> AsRef<[u8]> for ZenohView<'a> {
591        fn as_ref(&self) -> &[u8] {
592            self.bytes
593        }
594    }
595
596    impl<'a> Drop for ZenohView<'a> {
597        fn drop(&mut self) {
598            // Release the buffer lock first so `locked` is consistent
599            // with `has_data`. The C callback gates writes on `locked`.
600            self.buffer.locked.store(false, CoreOrdering::Release);
601            self.buffer.has_data.store(false, CoreOrdering::Release);
602        }
603    }
604
605    impl nros_rmw::SlotBorrowing for ZenohSubscriber {
606        type View<'a>
607            = ZenohView<'a>
608        where
609            Self: 'a;
610
611        fn try_borrow(&mut self) -> Result<Option<Self::View<'_>>, TransportError> {
612            // SubscriberBufferRef::get() returns a &SubscriberBuffer whose
613            // backing storage is 'static (lives in SUBSCRIBER_BUFFERS).
614            // Re-tie that 'static reference to the lifetime of `&mut self`
615            // by wrapping it in ZenohView (whose `'_` is implicit on
616            // `Self::View<'_>` and bound by `Self: 'a` in the trait def).
617            let buffer = self.buf.get();
618
619            if !buffer.has_data.load(CoreOrdering::Acquire) {
620                return Ok(None);
621            }
622            if buffer.overflow.load(CoreOrdering::Acquire) {
623                buffer.overflow.store(false, CoreOrdering::Release);
624                buffer.has_data.store(false, CoreOrdering::Release);
625                return Err(TransportError::MessageTooLarge);
626            }
627
628            let len = buffer.len.load(CoreOrdering::Acquire);
629
630            // Lock against the C callback before we hand out a borrow
631            // into `buffer.data` so the callback can't overwrite the
632            // bytes while the user is reading.
633            buffer.locked.store(true, CoreOrdering::Release);
634
635            // SAFETY: data is valid up to len; locked=true blocks the
636            // notify callback from overwriting until ZenohView::drop.
637            let bytes = unsafe { core::slice::from_raw_parts(buffer.data.as_ptr(), len) };
638
639            Ok(Some(ZenohView { bytes, buffer }))
640        }
641    }
642}

658    /// Simulate a subscription callback by writing directly to the subscriber buffer.
659    /// Mirrors the logic in `subscriber_callback_with_attachment` (post-40.4: checks locked).
660    pub(in crate::shim) fn simulate_subscription_callback(slot: usize, payload: &[u8]) {
661        let mut buf_ref = SubscriberBufferRef::new(slot);
662        let buffer = buf_ref.get_mut();
663
664        // Post-40.4: check locked flag — drop message if reader is processing
665        if buffer.locked.load(Ordering::Acquire) {
666            return;
667        }
668
669        if payload.len() > buffer.data.len() {
670            buffer.overflow.store(true, Ordering::Release);
671            buffer.has_data.store(true, Ordering::Release);
672        } else {
673            buffer.overflow.store(false, Ordering::Release);
674            buffer.data[..payload.len()].copy_from_slice(payload);
675            buffer.len.store(payload.len(), Ordering::Release);
676            buffer.attachment_len.store(0, Ordering::Release);
677            buffer.has_data.store(true, Ordering::Release);
678        }
679    }

681    /// Reset a subscriber buffer to idle state.
682    pub(in crate::shim) fn reset_subscriber_buffer(slot: usize) {
683        let mut buf_ref = SubscriberBufferRef::new(slot);
684        let buffer = buf_ref.get_mut();
685        buffer.has_data.store(false, Ordering::Release);
686        buffer.overflow.store(false, Ordering::Release);
687        buffer.locked.store(false, Ordering::Release);
688        buffer.len.store(0, Ordering::Release);
689        buffer.attachment_len.store(0, Ordering::Release);
690    }

692    /// Try to receive from a subscriber buffer slot.
693    /// Replicates `try_recv_raw` logic for testing without a zenoh session.
694    pub(in crate::shim) fn try_recv_subscription(
695        slot: usize,
696        recv_buf: &mut [u8],
697    ) -> Result<Option<usize>, TransportError> {
698        let buf_ref = SubscriberBufferRef::new(slot);
699        let buffer = buf_ref.get();
700
701        if !buffer.has_data.load(Ordering::Acquire) {
702            return Ok(None);
703        }
704
705        if buffer.overflow.load(Ordering::Acquire) {
706            buffer.overflow.store(false, Ordering::Release);
707            buffer.has_data.store(false, Ordering::Release);
708            return Err(TransportError::MessageTooLarge);
709        }
710
711        let len = buffer.len.load(Ordering::Acquire);
712        if len > recv_buf.len() {
713            buffer.has_data.store(false, Ordering::Release);
714            return Err(TransportError::BufferTooSmall);
715        }
716
717        // Safety: Data is valid up to len bytes
718        unsafe {
719            core::ptr::copy_nonoverlapping(buffer.data.as_ptr(), recv_buf.as_mut_ptr(), len);
720        }
721        buffer.has_data.store(false, Ordering::Release);
722
723        Ok(Some(len))
724    }

726    /// Process subscription data in-place (mirrors `process_raw_in_place` logic).
727    fn process_in_place_subscription(
728        slot: usize,
729    ) -> Result<Option<alloc::vec::Vec<u8>>, TransportError> {
730        let buf_ref = SubscriberBufferRef::new(slot);
731        let buffer = buf_ref.get();
732
733        if !buffer.has_data.load(Ordering::Acquire) {
734            return Ok(None);
735        }
736
737        if buffer.overflow.load(Ordering::Acquire) {
738            buffer.overflow.store(false, Ordering::Release);
739            buffer.has_data.store(false, Ordering::Release);
740            return Err(TransportError::MessageTooLarge);
741        }
742
743        let len = buffer.len.load(Ordering::Acquire);
744        buffer.locked.store(true, Ordering::Release);
745
746        // Read data in-place (equivalent to closure in process_raw_in_place)
747        let data = buffer.data[..len].to_vec();
748
749        buffer.locked.store(false, Ordering::Release);
750        buffer.has_data.store(false, Ordering::Release);
751
752        Ok(Some(data))
753    }
