1//! Phase 212.N.10 — workspace pkg-index + `$(find <pkg>)` resolver.
2//!
3//! Language-agnostic build-time mechanism shared by the Rust proc-macro
4//! family (`nros::main!`, `nros::launch!`) and the future C++ cmake fn
5//! `nros_entry(...)`. Per design-doc §11.4:
6//!
7//! 1. **Workspace root detection** — walk up from a starting dir
8//!    (`CARGO_MANIFEST_DIR` / `CMAKE_SOURCE_DIR`) looking for, in
9//!    order:
10//!    1. `NROS_WORKSPACE_ROOT` env var (explicit override).
11//!    2. `.colcon_workspace` marker.
12//!    3. `Cargo.toml` containing `[workspace]`.
13//!    4. `.git/` (last-resort fallback).
14//! 2. **Pkg-index build** — recurse from workspace root, collect every
15//!    `package.xml`. Pkg name = `<name>` element; pkg dir = parent dir.
16//! 3. **Cache** — emit `$OUT_DIR/.nros-pkg-index.json` keyed on combined
17//!    `package.xml` mtimes (build-script callers only; CLI callers
18//!    skip when `$OUT_DIR` is unset).
19//!
20//! The substitution resolver (`resolve_find_substitution`) feeds the
21//! N.11 launch.xml parser's `$(find <pkg>)/path/rest` substitution.
22

36/// Workspace-root markers consulted by [`detect_workspace_root`]. The
37/// `NROS_WORKSPACE_ROOT` env var short-circuits the walk entirely.
38const COLCON_WORKSPACE_MARKER: &str = ".colcon_workspace";
39const COLCON_IGNORE_MARKER: &str = "COLCON_IGNORE";
40const NROS_IGNORE_MARKER: &str = ".nros-ignore";
41
42/// Directory names skipped during the recursive `package.xml` walk —
43/// build / cache artefacts and source-control internals.
44const SKIP_DIRS: &[&str] = &[
45    "target",
46    "build",
47    ".git",
48    ".cargo",
49    "node_modules",
50    "__pycache__",
51];

53/// In-memory pkg-name → pkg-source-dir index.
54#[derive(Clone, Debug, PartialEq, Eq)]
55pub struct PkgIndex {
56    workspace_root: PathBuf,
57    pkgs: BTreeMap<String, PathBuf>,
58}
59
60impl PkgIndex {
61    /// Workspace root the index was built from.
62    pub fn workspace_root(&self) -> &Path {
63        &self.workspace_root
64    }
65
66    /// Look up a package directory by name. Errors with a diagnostic
67    /// listing the known packages when `name` is unknown — keeps the
68    /// "did you mean" surface within reach of higher-level callers.
69    pub fn resolve_pkg(&self, name: &str) -> Result<&Path> {
70        match self.pkgs.get(name) {
71            Some(dir) => Ok(dir.as_path()),
72            None => {
73                let mut known: Vec<&str> = self.pkgs.keys().map(String::as_str).collect();
74                known.sort_unstable();
75                bail!(
76                    "pkg `{name}` not found in workspace `{}`. Known pkgs: [{}]",
77                    self.workspace_root.display(),
78                    known.join(", "),
79                )
80            }
81        }
82    }
83
84    /// Parse a `$(find <pkg>)/optional/rest/of/path` expression and
85    /// return the absolute resolved path as a String. Errors when the
86    /// expression does not begin with `$(find <pkg>)` or when `<pkg>`
87    /// is unknown.
88    ///
89    /// Whitespace inside `$(find … )` is tolerated (leading + trailing
90    /// trimmed). The trailing `/rest` may be absent.
91    pub fn resolve_find_substitution(&self, expr: &str) -> Result<String> {
92        let trimmed = expr.trim_start();
93        let rest = trimmed
94            .strip_prefix("$(find")
95            .ok_or_else(|| eyre!("expected `$(find <pkg>)` prefix, got: {expr:?}"))?;
96        let after_find = rest.strip_prefix(' ').or_else(|| rest.strip_prefix('\t'));
97        let after_find = match after_find {
98            Some(s) => s,
99            // Empty `$(find)` w/o a space between `find` and `<pkg>`.
100            None => bail!("missing pkg name after `$(find`: {expr:?}"),
101        };
102        let close = after_find
103            .find(')')
104            .ok_or_else(|| eyre!("unterminated `$(find …)` substitution: {expr:?}"))?;
105        let pkg_name = after_find[..close].trim();
106        if pkg_name.is_empty() {
107            bail!("empty pkg name inside `$(find )`: {expr:?}");
108        }
109        let after_close = &after_find[close + 1..];
110        let pkg_dir = self.resolve_pkg(pkg_name)?;
111        let joined = if after_close.is_empty() {
112            pkg_dir.to_path_buf()
113        } else {
114            // Strip a leading separator so PathBuf::join doesn't treat the
115            // rest as absolute.
116            let tail = after_close.trim_start_matches('/');
117            if tail.is_empty() {
118                pkg_dir.to_path_buf()
119            } else {
120                pkg_dir.join(tail)
121            }
122        };
123        Ok(joined.to_string_lossy().into_owned())
124    }
125
126    /// Iterate (pkg-name, pkg-dir) pairs in sorted-by-name order.
127    pub fn pkgs(&self) -> impl Iterator<Item = (&str, &Path)> {
128        self.pkgs.iter().map(|(k, v)| (k.as_str(), v.as_path()))
129    }
130}

132/// On-disk shape of `$OUT_DIR/.nros-pkg-index.json`. The mtime stamps
133/// invalidate the cache as soon as any contributing `package.xml`
134/// changes — keyed on the absolute path so multiple workspaces caching
135/// into a shared OUT_DIR don't collide.
136#[derive(Clone, Debug, Serialize, Deserialize, PartialEq, Eq)]
137struct CachedIndex {
138    workspace_root: PathBuf,
139    entries: BTreeMap<String, CachedEntry>,
140}
141
142#[derive(Clone, Debug, Serialize, Deserialize, PartialEq, Eq)]
143struct CachedEntry {
144    dir: PathBuf,
145    /// Modified time as `(secs, nanos)` since the unix epoch. `0` when
146    /// the underlying filesystem doesn't surface a modified time.
147    mtime: (u64, u32),
148}
149
150/// Build the workspace pkg-index by recursive `package.xml` walk.
151///
152/// Writes the result to `$OUT_DIR/.nros-pkg-index.json` when `$OUT_DIR`
153/// is set (build-script context). CLI callers without `$OUT_DIR`
154/// rebuild the index every call — fast enough for hundreds of pkgs.
155pub fn build_pkg_index(workspace_root: &Path) -> Result<PkgIndex> {
156    let workspace_root = workspace_root
157        .canonicalize()
158        .with_context(|| format!("canonicalize workspace root `{}`", workspace_root.display()))?;
159
160    let mut package_xml_paths: Vec<PathBuf> = Vec::new();
161    let walker = WalkDir::new(&workspace_root)
162        .follow_links(false)
163        .into_iter();
164    let filter = |entry: &walkdir::DirEntry| -> bool {
165        if entry.depth() == 0 {
166            return true;
167        }
168        let file_name = entry.file_name().to_string_lossy();
169        if entry.file_type().is_dir() {
170            // Skip well-known build / cache / vcs dirs.
171            if SKIP_DIRS.iter().any(|name| file_name == *name) {
172                return false;
173            }
174            if file_name.starts_with("build-") {
175                return false;
176            }
177            // Ament convention: a dir carrying `COLCON_IGNORE` or
178            // `.nros-ignore` is opaque to discovery.
179            let path = entry.path();
180            if path.join(COLCON_IGNORE_MARKER).exists() || path.join(NROS_IGNORE_MARKER).exists() {
181                return false;
182            }
183        }
184        true
185    };
186
187    for entry in walker.filter_entry(filter) {
188        let entry =
189            entry.with_context(|| format!("walk workspace root `{}`", workspace_root.display()))?;
190        if !entry.file_type().is_file() {
191            continue;
192        }
193        if entry.file_name() == "package.xml" {
194            package_xml_paths.push(entry.into_path());
195        }
196    }
197    // Deterministic iteration order so the cache is stable.
198    package_xml_paths.sort();
199
200    // Try the OUT_DIR cache first.
201    let cache_path = std::env::var_os("OUT_DIR")
202        .map(PathBuf::from)
203        .map(|d| d.join(".nros-pkg-index.json"));
204    if let Some(cache_path) = cache_path.as_ref()
205        && let Some(cached) = load_cache_if_fresh(cache_path, &workspace_root, &package_xml_paths)?
206    {
207        return Ok(cached);
208    }
209
210    // Build fresh.
211    let mut pkgs: BTreeMap<String, PathBuf> = BTreeMap::new();
212    let mut seen_dirs: BTreeSet<PathBuf> = BTreeSet::new();
213    for pkg_xml in &package_xml_paths {
214        let name = read_package_xml_name(pkg_xml)
215            .with_context(|| format!("parse `{}`", pkg_xml.display()))?;
216        let dir = pkg_xml
217            .parent()
218            .map(Path::to_path_buf)
219            .unwrap_or_else(|| workspace_root.clone());
220        if !seen_dirs.insert(dir.clone()) {
221            // Two `package.xml` in the same dir — shouldn't happen, but
222            // be defensive.
223            continue;
224        }
225        if let Some(prev) = pkgs.get(&name) {
226            bail!(
227                "duplicate pkg name `{name}` in workspace `{}`: `{}` and `{}`",
228                workspace_root.display(),
229                prev.display(),
230                dir.display(),
231            );
232        }
233        pkgs.insert(name, dir);
234    }
235
236    let index = PkgIndex {
237        workspace_root: workspace_root.clone(),
238        pkgs,
239    };
240
241    // Persist the cache if we have an OUT_DIR.
242    if let Some(cache_path) = cache_path {
243        let cached = CachedIndex {
244            workspace_root: workspace_root.clone(),
245            entries: index
246                .pkgs
247                .iter()
248                .map(|(name, dir)| {
249                    let pkg_xml = dir.join("package.xml");
250                    let mtime = read_mtime(&pkg_xml).unwrap_or((0, 0));
251                    (
252                        name.clone(),
253                        CachedEntry {
254                            dir: dir.clone(),
255                            mtime,
256                        },
257                    )
258                })
259                .collect(),
260        };
261        if let Some(parent) = cache_path.parent() {
262            let _ = fs::create_dir_all(parent);
263        }
264        let json = serde_json::to_string_pretty(&cached).context("serialise pkg-index cache")?;
265        let _ = fs::write(&cache_path, json);
266    }
267
268    Ok(index)
269}

271/// Walk up from `start` looking for a workspace root. Per design-doc §11.4
272/// the marker order is:
273/// 1. `NROS_WORKSPACE_ROOT` env var.
274/// 2. `.colcon_workspace` marker in any ancestor.
275/// 3. `Cargo.toml` containing `[workspace]`.
276/// 4. `.git/` directory.
277///
278/// `COLCON_IGNORE` ancestors do NOT terminate the walk (they merely
279/// shadow themselves from discovery — handled inside [`build_pkg_index`]).
280pub fn detect_workspace_root(start: &Path) -> Result<PathBuf> {
281    if let Some(override_) = std::env::var_os("NROS_WORKSPACE_ROOT") {
282        let p = PathBuf::from(override_);
283        if !p.exists() {
284            bail!(
285                "NROS_WORKSPACE_ROOT=`{}` does not exist on disk",
286                p.display()
287            );
288        }
289        return p
290            .canonicalize()
291            .with_context(|| format!("canonicalize NROS_WORKSPACE_ROOT=`{}`", p.display()));
292    }
293
294    let start = if start.is_absolute() {
295        start.to_path_buf()
296    } else {
297        std::env::current_dir()
298            .with_context(|| "read current dir for workspace-root detection")?
299            .join(start)
300    };
301    let start = start.canonicalize().unwrap_or(start);
302
303    // Pass 1: `.colcon_workspace` marker.
304    if let Some(root) = walk_ancestors(&start, |dir| dir.join(COLCON_WORKSPACE_MARKER).is_file()) {
305        return Ok(root);
306    }
307    // Pass 2: `Cargo.toml` with `[workspace]`.
308    if let Some(root) = walk_ancestors(&start, |dir| is_cargo_workspace_root(dir)) {
309        return Ok(root);
310    }
311    // Pass 3: `.git/` directory.
312    if let Some(root) = walk_ancestors(&start, |dir| dir.join(".git").is_dir()) {
313        return Ok(root);
314    }
315    bail!(
316        "no workspace root found above `{}` (looked for $NROS_WORKSPACE_ROOT, \
317         {COLCON_WORKSPACE_MARKER}, Cargo.toml [workspace], .git/)",
318        start.display()
319    )
320}

322fn walk_ancestors(start: &Path, mut pred: impl FnMut(&Path) -> bool) -> Option<PathBuf> {
323    let mut current = Some(start);
324    while let Some(dir) = current {
325        if pred(dir) {
326            return Some(dir.to_path_buf());
327        }
328        current = dir.parent();
329    }
330    None
331}

333fn is_cargo_workspace_root(dir: &Path) -> bool {
334    let cargo = dir.join("Cargo.toml");
335    if !cargo.is_file() {
336        return false;
337    }
338    let Ok(raw) = fs::read_to_string(&cargo) else {
339        return false;
340    };
341    // Cheap textual sniff: `[workspace]` table header. A full TOML parse
342    // would be more robust but adds a dep at the walk-up step; the
343    // sniff matches the same heuristic `cargo` itself uses for the
344    // "is this a workspace root?" prompt.
345    raw.lines().any(|line| {
346        let l = line.trim_start();
347        l == "[workspace]" || l.starts_with("[workspace.") || l.starts_with("[workspace ]")
348    })
349}

351/// Parse the `<name>` element out of a `package.xml`. The element may
352/// nest inside `<package>` (ament convention); we accept any depth so
353/// long as a top-level `<name>` text node exists.
354fn read_package_xml_name(path: &Path) -> Result<String> {
355    let file = fs::File::open(path).with_context(|| format!("open `{}`", path.display()))?;
356    let mut reader = Reader::from_reader(BufReader::new(file));
357    reader.config_mut().trim_text(true);
358
359    let mut buf = Vec::new();
360    let mut in_name = false;
361    let mut depth = 0_i32;
362    let mut name: Option<String> = None;
363    loop {
364        match reader.read_event_into(&mut buf) {
365            Ok(Event::Start(e)) => {
366                depth += 1;
367                // `<name>` lives at depth 2: `<package> <name>...</name> </package>`
368                // but be lenient — accept the first `<name>` text node.
369                if depth >= 1 && e.local_name().as_ref() == b"name" {
370                    in_name = true;
371                }
372            }
373            Ok(Event::End(e)) => {
374                depth -= 1;
375                if e.local_name().as_ref() == b"name" {
376                    in_name = false;
377                }
378            }
379            Ok(Event::Text(e)) => {
380                if in_name && name.is_none() {
381                    let raw = e.unescape().context("unescape <name> text")?;
382                    let trimmed = raw.trim().to_string();
383                    if !trimmed.is_empty() {
384                        name = Some(trimmed);
385                    }
386                }
387            }
388            Ok(Event::Empty(_)) => {}
389            Ok(Event::Eof) => break,
390            Ok(_) => {}
391            Err(e) => {
392                return Err(eyre!(
393                    "XML parse error in `{}` at position {}: {e}",
394                    path.display(),
395                    reader.error_position()
396                ));
397            }
398        }
399        buf.clear();
400    }
401    name.ok_or_else(|| {
402        eyre!(
403            "no `<name>` element found in `{}` (expected ament package.xml)",
404            path.display()
405        )
406    })
407}

409fn read_mtime(path: &Path) -> Option<(u64, u32)> {
410    let meta = fs::metadata(path).ok()?;
411    let mtime = meta.modified().ok()?;
412    let dur = mtime.duration_since(SystemTime::UNIX_EPOCH).ok()?;
413    Some((dur.as_secs(), dur.subsec_nanos()))
414}

416fn load_cache_if_fresh(
417    cache_path: &Path,
418    workspace_root: &Path,
419    package_xml_paths: &[PathBuf],
420) -> Result<Option<PkgIndex>> {
421    let raw = match fs::read_to_string(cache_path) {
422        Ok(s) => s,
423        Err(_) => return Ok(None),
424    };
425    let cached: CachedIndex = match serde_json::from_str(&raw) {
426        Ok(v) => v,
427        Err(_) => return Ok(None),
428    };
429    if cached.workspace_root != workspace_root {
430        return Ok(None);
431    }
432    if cached.entries.len() != package_xml_paths.len() {
433        return Ok(None);
434    }
435    // Every cached entry's dir must still host a package.xml whose
436    // mtime matches.
437    for entry in cached.entries.values() {
438        let pkg_xml = entry.dir.join("package.xml");
439        let Some(mtime) = read_mtime(&pkg_xml) else {
440            return Ok(None);
441        };
442        if mtime != entry.mtime {
443            return Ok(None);
444        }
445    }
446    let pkgs = cached
447        .entries
448        .into_iter()
449        .map(|(name, entry)| (name, entry.dir))
450        .collect();
451    Ok(Some(PkgIndex {
452        workspace_root: workspace_root.to_path_buf(),
453        pkgs,
454    }))
