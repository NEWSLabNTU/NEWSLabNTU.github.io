1//! Per-field message capacity configuration (RFC-0033, Phase 229).
2//!
3//! Generated message bindings store unbounded sequence/string fields in
4//! fixed-capacity containers. Historically the capacity was a single hardcoded
5//! constant (`*_DEFAULT_SEQUENCE_CAPACITY` = 64, `*_DEFAULT_STRING_CAPACITY` =
6//! 256) shared by the Rust, C, and C++ generators. This module reads a
7//! `nros-codegen.toml` into one [`CapacityResolver`] that all three generators
8//! consult — a single resolver / three emitters is what makes the configuration
9//! language-agnostic.
10//!
11//! Only **unbounded** fields consult the resolver. Explicit `.msg` bounds
12//! (`uint8[<=N]`, `string<=N`) are authoritative and resolved by the caller
13//! before reaching [`CapacityResolver::resolve`].
14//!
15//! Precedence (highest wins): `.msg` bound (caller) → `[fields]` → `[types]` →
16//! `[packages]` → `[defaults]` → built-in constant.
17

27/// How a field's local storage is realized. See RFC-0033 "Storage modes".
28#[derive(Debug, Clone, Copy, PartialEq, Eq, Default, Deserialize)]
29#[serde(rename_all = "lowercase")]
30pub enum StorageMode {
31    /// `heapless::Vec<T, N>` / fixed `[N]` array — `N` elems always inline.
32    #[default]
33    Owned,
34    /// `alloc::Vec<T>` — dynamic, needs `alloc`/`std` (Phase 229.5).
35    Heap,
36    /// `&'a [T]` into the CDR receive buffer — zero-copy (Phase 229.6 / issue 0007).
37    Borrowed,
38}
39
40impl StorageMode {
41    /// Phase-1 supports only `owned`; `heap`/`borrowed` land in 229.5 / 229.6.
42    pub fn is_phase1_supported(self) -> bool {
43        matches!(self, StorageMode::Owned)
44    }
45
46    /// Token used in config + diagnostics.
47    pub fn as_str(self) -> &'static str {
48        match self {
49            StorageMode::Owned => "owned",
50            StorageMode::Heap => "heap",
51            StorageMode::Borrowed => "borrowed",
52        }
53    }
54}

56/// Which kind of field is being resolved — selects the built-in default.
57#[derive(Debug, Clone, Copy, PartialEq, Eq)]
58pub enum FieldKind {
59    Sequence,
60    String,
61}
62
63impl FieldKind {
64    fn builtin_default(self) -> usize {
65        match self {
66            FieldKind::Sequence => NROS_DEFAULT_SEQUENCE_CAPACITY,
67            FieldKind::String => NROS_DEFAULT_STRING_CAPACITY,
68        }
69    }
70}
71
72/// Resolved storage for one field.
73#[derive(Debug, Clone, Copy, PartialEq, Eq)]
74pub struct FieldStorage {
75    pub cap: usize,
76    pub mode: StorageMode,
77}
78
79/// A config entry value: either an integer (owned shorthand) or `{ cap, mode }`.
80#[derive(Debug, Clone, Copy, Deserialize)]
81#[serde(untagged)]
82enum CapEntry {
83    /// `field = 64` → `{ cap = 64, mode = "owned" }`.
84    Int(usize),
85    /// `field = { cap = 2_000_000, mode = "borrowed" }`.
86    Table {
87        cap: usize,
88        #[serde(default)]
89        mode: StorageMode,
90    },
91}
92
93impl CapEntry {
94    fn resolve(self) -> FieldStorage {
95        match self {
96            CapEntry::Int(cap) => FieldStorage {
97                cap,
98                mode: StorageMode::Owned,
99            },
100            CapEntry::Table { cap, mode } => FieldStorage { cap, mode },
101        }
102    }
103}

105/// `sequence` / `string` overrides at the `[defaults]`, `[packages.*]`, and
106/// `[types.*]` levels. Each accepts the same int-or-table form as `[fields]`.
107#[derive(Debug, Clone, Copy, Default, Deserialize)]
108#[serde(deny_unknown_fields)]
109struct LevelCaps {
110    #[serde(default)]
111    sequence: Option<CapEntry>,
112    #[serde(default)]
113    string: Option<CapEntry>,
114}
115
116impl LevelCaps {
117    fn pick(&self, kind: FieldKind) -> Option<CapEntry> {
118        match kind {
119            FieldKind::Sequence => self.sequence,
120            FieldKind::String => self.string,
121        }
122    }
123
124    /// Per-key deep merge: `over` wins where it specifies a value.
125    fn merge_over(&mut self, over: LevelCaps) {
126        if over.sequence.is_some() {
127            self.sequence = over.sequence;
128        }
129        if over.string.is_some() {
130            self.string = over.string;
131        }
132    }
133}
134
135#[derive(Debug, Clone, Default, Deserialize)]
136#[serde(deny_unknown_fields)]
137struct RawConfig {
138    #[serde(default)]
139    defaults: LevelCaps,
140    /// keyed by package, e.g. `"sensor_msgs"`.
141    #[serde(default)]
142    packages: BTreeMap<String, LevelCaps>,
143    /// keyed by `"pkg/Msg"`, e.g. `"sensor_msgs/Image"`.
144    #[serde(default)]
145    types: BTreeMap<String, LevelCaps>,
146    /// keyed by `"pkg/Msg.field"`, e.g. `"sensor_msgs/Image.data"`.
147    #[serde(default)]
148    fields: BTreeMap<String, CapEntry>,
149}
150
151impl RawConfig {
152    /// Deep-merge `over` onto `self`; `over` (the app file) wins.
153    fn merge_over(&mut self, over: RawConfig) {
154        self.defaults.merge_over(over.defaults);
155        for (k, v) in over.packages {
156            self.packages.entry(k).or_default().merge_over(v);
157        }
158        for (k, v) in over.types {
159            self.types.entry(k).or_default().merge_over(v);
160        }
161        // Fields are atomic entries: the app entry replaces the workspace entry.
162        self.fields.extend(over.fields);
163    }
164}

166/// The conventional config filename discovered by [`CapacityResolver::discover`].
167pub const CODEGEN_CONFIG_FILENAME: &str = "nros-codegen.toml";
168
169/// Error parsing or loading a `nros-codegen.toml`.
170#[derive(Debug, thiserror::Error)]
171pub enum ConfigError {
172    #[error("failed to parse codegen config: {0}")]
173    Parse(#[from] toml::de::Error),
174    #[error("failed to read codegen config '{path}': {source}")]
175    Read {
176        path: PathBuf,
177        source: std::io::Error,
178    },
179}

181/// Resolves per-field storage from a merged `nros-codegen.toml`. One instance
182/// feeds all three language backends.
183#[derive(Debug, Clone, Default)]
184pub struct CapacityResolver {
185    raw: RawConfig,
186}
187
188impl CapacityResolver {
189    /// An empty resolver — every field falls through to its built-in default,
190    /// reproducing pre-Phase-229 output byte-for-byte.
191    pub fn empty() -> Self {
192        Self::default()
193    }
194
195    /// Parse a single `nros-codegen.toml` body.
196    pub fn from_toml_str(s: &str) -> Result<Self, ConfigError> {
197        Ok(Self {
198            raw: toml::from_str(s)?,
199        })
200    }
201
202    /// Load a single `nros-codegen.toml` from `path`.
203    pub fn from_file(path: &Path) -> Result<Self, ConfigError> {
204        let body = std::fs::read_to_string(path).map_err(|source| ConfigError::Read {
205            path: path.to_path_buf(),
206            source,
207        })?;
208        Self::from_toml_str(&body)
209    }
210
211    /// Discover and merge `nros-codegen.toml` files by walking up from
212    /// `start_dir` to the filesystem root (or until `stop_dir`, inclusive).
213    /// Files are merged root-most → `start_dir`, so the **closest** file (the
214    /// app) wins over ancestors (the workspace). Missing files are skipped; an
215    /// absent chain yields an empty resolver (built-in defaults).
216    pub fn discover(start_dir: &Path, stop_dir: Option<&Path>) -> Result<Self, ConfigError> {
217        // Collect candidate dirs from start upward, then reverse so the
218        // root-most is merged first and the closest file wins.
219        let mut dirs: Vec<&Path> = Vec::new();
220        let mut cur = Some(start_dir);
221        while let Some(dir) = cur {
222            dirs.push(dir);
223            if stop_dir == Some(dir) {
224                break;
225            }
226            cur = dir.parent();
227        }
228
229        let mut resolver = Self::empty();
230        for dir in dirs.into_iter().rev() {
231            let candidate = dir.join(CODEGEN_CONFIG_FILENAME);
232            if candidate.is_file() {
233                resolver = resolver.merged_with(Self::from_file(&candidate)?);
234            }
235        }
236        Ok(resolver)
237    }
238
239    /// Build a resolver from an optional explicit config path plus discovery
240    /// from `start_dir`. The explicit file (if any) is merged **last** so a
241    /// `--codegen-config` flag wins over any discovered `nros-codegen.toml`.
242    pub fn resolve_for(
243        explicit: Option<&Path>,
244        start_dir: &Path,
245        stop_dir: Option<&Path>,
246    ) -> Result<Self, ConfigError> {
247        let mut resolver = Self::discover(start_dir, stop_dir)?;
248        if let Some(path) = explicit {
249            resolver = resolver.merged_with(Self::from_file(path)?);
250        }
251        Ok(resolver)
252    }
253
254    /// Merge another config on top of this one; `over` (e.g. the app file) wins
255    /// over `self` (e.g. the workspace file).
256    pub fn merged_with(mut self, over: CapacityResolver) -> Self {
257        self.raw.merge_over(over.raw);
258        self
259    }
260
261    /// Resolve storage for an **unbounded** field. Bounded fields are resolved
262    /// by the caller from the `.msg` bound and must not reach this method.
263    pub fn resolve(
264        &self,
265        package: &str,
266        message: &str,
267        field: &str,
268        kind: FieldKind,
269    ) -> FieldStorage {
270        // 2 — per-field
271        let field_key = format!("{package}/{message}.{field}");
272        if let Some(e) = self.raw.fields.get(&field_key) {
273            return e.resolve();
274        }
275        // 3 — per-type
276        let type_key = format!("{package}/{message}");
277        if let Some(e) = self.raw.types.get(&type_key).and_then(|l| l.pick(kind)) {
278            return e.resolve();
279        }
280        // 4 — per-package
281        if let Some(e) = self.raw.packages.get(package).and_then(|l| l.pick(kind)) {
282            return e.resolve();
283        }
284        // 5 — global defaults
285        if let Some(e) = self.raw.defaults.pick(kind) {
286            return e.resolve();
287        }
288        // 6 — built-in
289        FieldStorage {
290            cap: kind.builtin_default(),
291            mode: StorageMode::Owned,
292        }
293    }
294}

305        let r = CapacityResolver::empty();
306        let s = r.resolve("std_msgs", "String", "data", STR);
307        assert_eq!(s.cap, NROS_DEFAULT_STRING_CAPACITY);
308        assert_eq!(s.mode, StorageMode::Owned);
309        let q = r.resolve("sensor_msgs", "Image", "data", SEQ);
310        assert_eq!(q.cap, NROS_DEFAULT_SEQUENCE_CAPACITY);
311        assert_eq!(q.mode, StorageMode::Owned);

360    fn precedence_field_beats_type_beats_package_beats_defaults() {
361        let r = CapacityResolver::from_toml_str(
362            r#"
