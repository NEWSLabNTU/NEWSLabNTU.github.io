31/// Generate a complete ROS 2 message package with both RMW and idiomatic layers
32pub fn generate_message_package(
33    package_name: &str,
34    message_name: &str,
35    message: &Message,
36    all_dependencies: &HashSet<String>,
37) -> Result<GeneratedPackage, GeneratorError> {
38    // Extract dependencies from this specific message
39    let msg_deps = extract_dependencies(message);
40
41    // Combine with externally provided dependencies
42    let mut all_deps: Vec<String> = all_dependencies.iter().cloned().collect();
43    all_deps.extend(msg_deps);
44    all_deps.sort();
45    all_deps.dedup();
46
47    // Check if we need serde's big-array feature
48    let needs_big_array_feature = needs_big_array(message);
49
50    // Generate Cargo.toml
51    let cargo_toml_template = CargoTomlTemplate {
52        package_name,
53        dependencies: &all_deps,
54        needs_big_array: needs_big_array_feature,
55    };
56    let cargo_toml = cargo_toml_template.render()?;
57
58    // Generate build.rs
59    let build_rs_template = BuildRsTemplate;
60    let build_rs = build_rs_template.render()?;
61
62    // Generate lib.rs
63    let lib_rs_template = LibRsTemplate {
64        has_messages: true,
65        has_services: false,
66        has_actions: false,
67    };
68    let lib_rs = lib_rs_template.render()?;
69
70    // Generate RMW layer message
71    let rmw_fields: Vec<RmwField> = message
72        .fields
73        .iter()
74        .map(|f| RmwField {
75            name: escape_keyword(&f.name),
76            rust_type: rust_type_for_field(&f.field_type, true, Some(package_name)),
77            default_value: f
78                .default_value
79                .as_ref()
80                .map(constant_value_to_rust)
81                .unwrap_or_default(),
82        })
83        .collect();
84
85    let rmw_constants: Vec<MessageConstant> = message
86        .constants
87        .iter()
88        .map(|c| MessageConstant {
89            name: c.name.clone(),
90            rust_type: rust_type_for_constant(&c.constant_type),
91            value: constant_value_to_rust(&c.value),
92        })
93        .collect();
94
95    let message_module = &to_snake_case(message_name);
96
97    let message_rmw_template = MessageRmwTemplate {
98        package_name,
99        message_name,
100        message_module,
101        fields: rmw_fields,
102        constants: rmw_constants,
103    };
104    let message_rmw = message_rmw_template.render()?;
105
106    // Generate idiomatic layer message
107    let idiomatic_fields: Vec<IdiomaticField> = message
108        .fields
109        .iter()
110        .map(|f| IdiomaticField {
111            name: escape_keyword(&f.name),
112            rust_type: rust_type_for_field(&f.field_type, false, Some(package_name)),
113            default_value: f
114                .default_value
115                .as_ref()
116                .map(constant_value_to_rust)
117                .unwrap_or_default(),
118            kind: determine_field_kind(&f.field_type),
119        })
120        .collect();
121
122    let idiomatic_constants: Vec<MessageConstant> = message
123        .constants
124        .iter()
125        .map(|c| MessageConstant {
126            name: c.name.clone(),
127            rust_type: rust_type_for_constant(&c.constant_type),
128            value: constant_value_to_rust(&c.value),
129        })
130        .collect();
131
132    let message_idiomatic_template = MessageIdiomaticTemplate {
133        package_name,
134        message_name,
135        message_module,
136        fields: idiomatic_fields,
137        constants: idiomatic_constants,
138    };
139    let message_idiomatic = message_idiomatic_template.render()?;
140
141    Ok(GeneratedPackage {
142        cargo_toml,
143        build_rs,
144        lib_rs,
145        message_rmw,
146        message_idiomatic,
147    })
148}

157/// Generate a nros message package
158pub fn generate_nros_message_package(
159    package_name: &str,
160    message_name: &str,
161    message: &Message,
162    all_dependencies: &HashSet<String>,
163    package_version: &str,
164    edition: RosEdition,
165    resolver: &CapacityResolver,
166) -> Result<GeneratedNrosPackage, GeneratorError> {
167    // Extract dependencies from this specific message
168    let msg_deps = extract_dependencies(message);
169
170    // Combine with externally provided dependencies
171    let mut all_deps: Vec<String> = all_dependencies.iter().cloned().collect();
172    all_deps.extend(msg_deps);
173    all_deps.sort();
174    all_deps.dedup();
175
176    // Generate Cargo.toml
177    let cargo_toml_template = CargoNrosTomlTemplate {
178        package_name,
179        package_version,
180        dependencies: &all_deps,
181        has_actions: false,
182    };
183    let cargo_toml = cargo_toml_template.render()?;
184
185    // Generate lib.rs
186    let lib_rs_template = LibNrosRsTemplate {
187        has_messages: true,
188        has_services: false,
189        has_actions: false,
190    };
191    let lib_rs = lib_rs_template.render()?;
192
193    // Generate message fields
194    let fields: Vec<NrosField> = message
195        .fields
196        .iter()
197        .map(|f| field_to_nros_field(f, package_name, message_name, resolver))
198        .collect::<Result<_, _>>()?;
199
200    // Generate constants
201    let constants: Vec<MessageConstant> = message
202        .constants
203        .iter()
204        .map(|c| MessageConstant {
205            name: c.name.clone(),
206            rust_type: nros_type_for_constant(&c.constant_type),
207            value: constant_value_to_rust(&c.value),
208        })
209        .collect();
210
211    let type_hash = edition.type_hash();
212
213    let has_fields = !fields.is_empty();
214    let has_large_array = fields.iter().any(|f| f.is_large_array);
215    let has_borrowed = fields.iter().any(|f| f.is_borrowed);
216    let schema = build_nros_message_schema(package_name, message_name, &message.fields);
217    let message_template = MessageNrosTemplate {
218        package_name,
219        message_name,
220        type_hash,
221        fields,
222        constants,
223        has_fields,
224        has_large_array,
225        has_borrowed,
226        inline_mode: false,
227        schema_helper_consts: schema.helper_consts,
228        schema_fields_block: schema.fields_block,
229        schema_type_name: schema.nros_type_name,
230    };
231    let message_rs = message_template.render()?;
232
233    Ok(GeneratedNrosPackage {
234        cargo_toml,
235        lib_rs,
236        message_rs,
237    })
238}

240/// Generate a single message's Rust code in inline mode.
241///
242/// Unlike `generate_nros_message_package`, this only returns the rendered
243/// message code (no Cargo.toml or lib.rs). Cross-package references use
244/// `super::super::super::pkg::msg::Type` paths.
245pub fn generate_nros_inline_message(
246    package_name: &str,
247    message_name: &str,
248    message: &Message,
249    edition: RosEdition,
250    resolver: &CapacityResolver,
251) -> Result<String, GeneratorError> {
252    let mode = NrosCodegenMode::Inline;
253    let fields: Vec<NrosField> = message
254        .fields
255        .iter()
256        .map(|f| field_to_nros_field_with_mode(f, package_name, message_name, resolver, mode))
257        .collect::<Result<_, _>>()?;
258
259    let constants: Vec<MessageConstant> = message
260        .constants
261        .iter()
262        .map(|c| MessageConstant {
263            name: c.name.clone(),
264            rust_type: nros_type_for_constant(&c.constant_type),
265            value: constant_value_to_rust(&c.value),
266        })
267        .collect();
268
269    let type_hash = edition.type_hash();
270    let has_fields = !fields.is_empty();
271    let has_large_array = fields.iter().any(|f| f.is_large_array);
272    let has_borrowed = fields.iter().any(|f| f.is_borrowed);
273    let schema = build_nros_message_schema(package_name, message_name, &message.fields);
274
275    let template = MessageNrosTemplate {
276        package_name,
277        message_name,
278        type_hash,
279        fields,
280        constants,
281        has_fields,
282        has_large_array,
283        has_borrowed,
284        inline_mode: true,
285        schema_helper_consts: schema.helper_consts,
286        schema_fields_block: schema.fields_block,
287        schema_type_name: schema.nros_type_name,
288    };
289
290    Ok(template.render()?)
291}

305/// Generate C code for a message type
306pub fn generate_c_message_package(
307    package_name: &str,
308    message_name: &str,
309    message: &Message,
310    type_hash: &str,
311    resolver: &CapacityResolver,
312) -> Result<GeneratedCPackage, GeneratorError> {
313    let c_pkg_name = to_c_package_name(package_name);
314    let msg_snake = to_snake_case(message_name);
315
316    // Build struct and guard names
317    let struct_name = format!("{}_msg_{}", c_pkg_name, msg_snake);
318    let guard_name = format!(
319        "{}_MSG_{}_H",
320        c_pkg_name.to_uppercase(),
321        msg_snake.to_uppercase()
322    );
323    let constant_prefix = format!(
324        "{}_MSG_{}",
325        c_pkg_name.to_uppercase(),
326        msg_snake.to_uppercase()
327    );
328    let header_name = format!("{}_msg_{}.h", c_pkg_name, msg_snake);
329    let source_name = format!("{}_msg_{}.c", c_pkg_name, msg_snake);
330
331    // Extract dependencies — both cross-package (umbrella includes) and
332    // intra-package (per-type includes for types in the same package).
333    let mut dependencies = Vec::new();
334    let mut type_includes = Vec::new();
335    for field in &message.fields {
336        let field_type = match &field.field_type {
337            FieldType::NamespacedType { .. } => Some(&field.field_type),
338            FieldType::Array { element_type, .. }
339            | FieldType::Sequence { element_type }
340            | FieldType::BoundedSequence { element_type, .. } => {
341                if matches!(element_type.as_ref(), FieldType::NamespacedType { .. }) {
342                    Some(element_type.as_ref())
343                } else {
344                    None
345                }
346            }
347            _ => None,
348        };
349        if let Some(FieldType::NamespacedType { package, name }) = field_type {
350            let pkg = package.as_deref().unwrap_or(package_name);
351            let dep = to_c_package_name(pkg);
352            let header_filename =
353                format!("{}_msg_{}.h", to_c_package_name(pkg), to_snake_case(name));
354            let type_header = if dep != c_pkg_name {
355                // Cross-package: include with subdirectory path
356                if !dependencies.contains(&dep) {
357                    dependencies.push(dep.clone());
358                }
359                format!("{}/msg/{}", dep, header_filename)
360            } else {
361                // Intra-package: include from same msg/ directory
362                header_filename
363            };
364            if !type_includes.contains(&type_header) {
365                type_includes.push(type_header);
366            }
367        }
368    }
369    dependencies.sort();
370    type_includes.sort();
371
372    // Build C fields
373    let fields: Vec<CField> = message
374        .fields
375        .iter()
376        .map(|field| {
377            build_c_field(
378                &field.name,
379                &field.field_type,
380                Some(package_name),
381                message_name,
382                resolver,
383            )
384        })
385        .collect::<Result<_, _>>()?;
386
387    // Build C constants
388    let constants: Vec<CConstant> = message
389        .constants
390        .iter()
391        .map(|constant| CConstant {
392            name: constant.name.clone(),
393            c_type: c_type_for_constant(&constant.constant_type),
394            value: constant_value_to_rust(&constant.value),
395        })
396        .collect();
397
398    let has_fields = !fields.is_empty();
399    let has_borrowed = fields.iter().any(|f| f.is_borrowed);
400
401    // Generate header
402    let header_template = MessageCHeaderTemplate {
403        package_name,
404        message_name,
405        type_hash,
406        guard_name,
407        struct_name: struct_name.clone(),
408        constant_prefix,
409        fields: fields.clone(),
410        constants,
411        dependencies,
412        type_includes,
413        has_fields,
414        has_borrowed,
415    };
416    let header = header_template.render()?;
417
418    // Generate source
419    let source_template = MessageCSourceTemplate {
420        package_name,
421        message_name,
422        type_hash,
423        header_name: header_name.clone(),
424        struct_name,
425        fields,
426        has_fields,
427        has_borrowed,
428    };
429    let source = source_template.render()?;
430
431    Ok(GeneratedCPackage {
432        header,
433        source,
434        header_name,
435        source_name,
436    })
