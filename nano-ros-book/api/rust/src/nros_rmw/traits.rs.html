1//! Transport abstraction traits.
2//!
3//! Defines the backend-agnostic interface that transport implementations
4//! (zenoh-pico, XRCE-DDS) must satisfy. The core trait hierarchy is:
5//!
6//! - [`Session`] — connection lifecycle and handle creation
7//! - [`Publisher`] / [`Subscriber`] — pub/sub data transport
8//! - [`ServiceServerTrait`] / [`ServiceClientTrait`] — request/reply
9//! - [`Rmw`] — top-level factory that creates sessions
10

83/// Action information for action client/server
84///
85/// Actions in ROS 2 use 5 communication channels:
86/// - `send_goal` service: `<action_name>/_action/send_goal`
87/// - `cancel_goal` service: `<action_name>/_action/cancel_goal`
88/// - `get_result` service: `<action_name>/_action/get_result`
89/// - `feedback` topic: `<action_name>/_action/feedback`
90/// - `status` topic: `<action_name>/_action/status`

96    pub type_name: &'a str,

99    /// Domain ID (default: 0)
100    pub domain_id: u32,
101}

130    }

133    /// Returns: `<action>/_action/get_result`

135        self.sub_name::<N>("get_result")
136    }

138    /// Generate the feedback topic name
139    /// Returns: `<action>/_action/feedback`
140    pub fn feedback_key<const N: usize>(&self) -> heapless::String<N> {
141        self.sub_name::<N>("feedback")
142    }
143
144    /// Generate the status topic name
145    /// Returns: `<action>/_action/status`
146    pub fn status_key<const N: usize>(&self) -> heapless::String<N> {
147        self.sub_name::<N>("status")
148    }
149
150    /// Generate a sub-entity name for an action component
151    /// Returns: `<action>/_action/<suffix>` (e.g., `fibonacci/_action/send_goal`)
152    ///
153    /// The caller is responsible for constructing the full key expression
154    /// by wrapping this name in a `ServiceInfo` or `TopicInfo` with the
155    /// correct sub-service/sub-topic type name.
156    fn sub_name<const N: usize>(&self, suffix: &str) -> heapless::String<N> {
157        let mut name = heapless::String::new();
158        let action_stripped = self.name.trim_matches('/');
159        let _ = core::fmt::write(
160            &mut name,
161            format_args!("/{}/_action/{}", action_stripped, suffix),
162        );
163        name
164    }
165}

199/// Transport error types.
200///
201/// No longer `Copy` — the `Backend` / `BackendDynamic` variants carry a
202/// string diagnostic, which can't be `Copy`. Rust callers that used to
203/// copy a `TransportError` value repeatedly now need `.clone()` or
204/// `ref` in match arms. C/C++ callers are unaffected — both map
205/// `TransportError` to integer codes (`nros_ret_t` / `ErrorCode`)
206/// before crossing the FFI boundary.
207#[derive(Debug, Clone, PartialEq, Eq)]
208pub enum TransportError {
209    /// Failed to connect to transport
210    ConnectionFailed,
211    /// Connection was closed
212    Disconnected,
213    /// Failed to create publisher
214    PublisherCreationFailed,
215    /// Failed to create subscriber
216    SubscriberCreationFailed,
217    /// Failed to create service server
218    ServiceServerCreationFailed,
219    /// Failed to create service client
220    ServiceClientCreationFailed,
221    /// Failed to publish message
222    PublishFailed,
223    /// Failed to send service request
224    ServiceRequestFailed,
225    /// Failed to send service reply
226    ServiceReplyFailed,
227    /// Serialization error
228    SerializationError,
229    /// Deserialization error
230    DeserializationError,
231    /// Buffer too small
232    BufferTooSmall,
233    /// Incoming message exceeded the static buffer capacity
234    MessageTooLarge,
235    /// Timeout waiting for message
236    Timeout,
237    /// Invalid configuration
238    InvalidConfig,
239    /// Resource (slot, buffer, queue) momentarily unavailable. Retry.
240    /// Phase 99: returned by `try_loan` when arena slots are full and
241    /// by `try_borrow` when no message is ready (alternative to
242    /// `Ok(None)` for backends that prefer the error variant).
243    WouldBlock,
244    /// Requested allocation exceeds backend capacity. Phase 99:
245    /// `try_loan(len)` returns this when `len` > arena slot size.
246    TooLarge,
247    /// Failed to start background tasks
248    TaskStartFailed,
249    /// Failed to poll for incoming messages
250    PollFailed,
251    /// Failed to send keepalive
252    KeepaliveFailed,
253    /// Failed to send join message
254    JoinFailed,
255    /// Caller supplied a NULL pointer, an out-of-range value, or an
256    /// inconsistent argument combination. Phase 102.1.
257    InvalidArgument,
258    /// The backend does not implement this operation. Optional
259    /// callbacks return this; the runtime then falls back to a
260    /// default path. Phase 102.1.
261    Unsupported,
262    /// Memory allocation failed. Returned by backends on `std` /
263    /// `alloc`-equipped platforms when heap allocation fails.
264    /// Bare-metal backends generally do not produce this — they
265    /// preallocate at session-open time. Phase 102.1.
266    BadAlloc,
267    /// Publisher and subscriber QoS profiles do not match in a way
268    /// the backend cannot reconcile. Phase 102.1.
269    IncompatibleQos,
270    /// Topic, service, or action name failed validation. Phase 102.1.
271    TopicNameInvalid,
272    /// A request referenced a node that does not exist in this
273    /// session. Phase 102.1.
274    NodeNameNonExistent,
275    /// The backend does not support loaned messages on this entity,
276    /// or the loan slot is currently in use. Phase 102.1.
277    LoanNotSupported,
278    /// No data was available on a non-blocking receive. Distinct
279    /// from `Timeout`: fires immediately, not after a bounded wait.
280    /// Phase 102.1.
281    NoData,
282    /// Backend-specific error with a `'static` diagnostic string.
283    ///
284    /// Useful for zenoh-pico / XRCE-DDS return codes that map to a
285    /// fixed set of known messages. `no_std`-compatible.
286    Backend(&'static str),
287    /// Backend-specific error with an owned diagnostic string.
288    ///
289    /// Available only with the `alloc` feature. Use this when the
290    /// diagnostic is formatted at runtime (e.g. from a C error code
291    /// plus a socket address).
292    #[cfg(feature = "alloc")]
293    BackendDynamic(alloc::string::String),
294}

345/// Full DDS-shaped QoS profile. Matches the field set of upstream
346/// `rmw_qos_profile_t`.
347///
348/// Backends advertise per-policy support via
349/// [`Session::supported_qos_policies`]; entities created with a
350/// profile the active backend can't honour return
351/// [`TransportError::IncompatibleQos`] synchronously at create time
352/// — no silent downgrade.
353///
354/// Zero-valued time-window fields ("off") mean infinite — the policy
355/// is effectively disabled for the entity.
356#[derive(Debug, Clone, Copy, PartialEq, Eq)]
357pub struct QosSettings {
358    /// History policy
359    pub history: QosHistoryPolicy,
360    /// Reliability policy
361    pub reliability: QosReliabilityPolicy,
362    /// Durability policy
363    pub durability: QosDurabilityPolicy,
364    /// Liveliness policy
365    pub liveliness_kind: QosLivelinessPolicy,
366    /// History depth (only used if history is KeepLast)
367    pub depth: u32,
368    /// Subscriber max-inter-arrival / publisher offered-rate, ms.
369    /// `0` = infinite (no deadline check).
370    pub deadline_ms: u32,
371    /// Sample expiry, ms. Subscribers filter samples older than this.
372    /// `0` = infinite (no expiry).
373    pub lifespan_ms: u32,
374    /// Liveliness lease, ms. `0` = infinite.
375    pub liveliness_lease_ms: u32,
376    /// If `true`, topic-name encoding skips the `/rt/` ROS prefix.
377    pub avoid_ros_namespace_conventions: bool,
378}

611#[derive(Debug, Clone)]
612pub struct TransportConfig<'a> {

616    pub mode: SessionMode,

618    ///

625    /// - `"add_timestamp"` - Add timestamps to messages (`"true"` or `"false"`)
626    pub properties: &'a [(&'a str, &'a str)],
627}

631        Self {

639/// Middleware-agnostic session configuration.
640///
641/// `RmwConfig` provides a uniform interface that any RMW backend can
642/// interpret. Backends map the universal fields to their own connection
643/// parameters and interpret `properties` for anything backend-specific.
644///
645/// # Examples
646///
647/// ```
648/// use nros_rmw::{RmwConfig, SessionMode};
649///
650/// let config = RmwConfig {
651///     locator: "tcp/192.168.1.1:7447",
652///     mode: SessionMode::Client,
653///     domain_id: 0,
654///     node_name: "talker",
655///     namespace: "",
656///     properties: &[],
657/// };
658/// ```
659#[derive(Debug, Clone, Copy)]
660pub struct RmwConfig<'a> {
661    /// Middleware-specific connection string.
662    ///
663    /// - zenoh: `"tcp/192.168.1.1:7447"` or `"udp/224.0.0.224:7447"`
664    /// - XRCE-DDS: `"udp/192.168.1.1:2019"`
665    pub locator: &'a str,
666    /// Session mode (zenoh: client/peer; XRCE-DDS: always client)
667    pub mode: SessionMode,
668    /// ROS 2 domain ID (maps to DDS domain or zenoh key prefix)
669    pub domain_id: u32,
670    /// Node name (e.g., `"talker"`)
671    pub node_name: &'a str,
672    /// Node namespace (e.g., `""` or `"/ns1"`)
673    pub namespace: &'a str,
674    /// Backend-specific key/value properties.
675    ///
676    /// Uniform escape hatch for backend-specific tuning that doesn't fit
677    /// the universal fields above. Each backend documents the keys it
678    /// understands; unknown keys are ignored. Passing `&[]` is always
679    /// valid.
680    ///
681    /// Examples:

700/// Locator transport protocol
701#[derive(Debug, Clone, Copy, PartialEq, Eq)]
702pub enum LocatorProtocol {
703    /// TCP transport (e.g., "tcp/127.0.0.1:7447")
704    Tcp,
705    /// UDP transport (e.g., "udp/192.168.1.50:2019" — common for XRCE-DDS)
706    Udp,
707    /// Serial/UART transport (e.g., "serial//dev/ttyUSB0#baudrate=115200")
708    Serial,
709    /// Unknown protocol
710    Unknown,
711}
712
713/// Parse the protocol from a locator string
714pub fn locator_protocol(locator: &str) -> LocatorProtocol {
715    if locator.starts_with("tcp/") {
716        LocatorProtocol::Tcp
717    } else if locator.starts_with("udp/") {
718        LocatorProtocol::Udp
719    } else if locator.starts_with("serial/") {
720        LocatorProtocol::Serial
721    } else {
722        LocatorProtocol::Unknown
723    }
724}
725
726/// Validate a locator string format.
727///
728/// Returns `Ok(())` if the locator is well-formed, or an error message describing
729/// the problem. This provides early feedback before zenoh-pico or XRCE-DDS rejects
730/// a bad locator.
731///
732/// Supported formats:
733/// - TCP: `tcp/<host>:<port>` (e.g., `tcp/127.0.0.1:7447`)
734/// - UDP: `udp/<host>:<port>` (e.g., `udp/192.168.1.50:2019`)
735/// - Serial: `serial/<device>#baudrate=<rate>` (e.g., `serial//dev/ttyUSB0#baudrate=115200`)
736pub fn validate_locator(locator: &str) -> Result<(), &'static str> {
737    match locator_protocol(locator) {
738        LocatorProtocol::Tcp => {
739            let rest = &locator[4..]; // skip "tcp/"
740            if !rest.contains(':') {
741                return Err("TCP locator must contain host:port (e.g., tcp/127.0.0.1:7447)");
742            }
743            Ok(())
744        }
745        LocatorProtocol::Udp => {
746            let rest = &locator[4..]; // skip "udp/"
747            if !rest.contains(':') {
748                return Err("UDP locator must contain host:port (e.g., udp/192.168.1.50:2019)");
749            }
750            Ok(())
751        }
752        LocatorProtocol::Serial => {
753            let rest = &locator[7..]; // skip "serial/"
754            if rest.is_empty() {
755                return Err(
756                    "serial locator must specify device (e.g., serial//dev/ttyUSB0#baudrate=115200)",
757                );
758            }
759            if !rest.contains("#baudrate=") {
760                return Err(
761                    "serial locator must include #baudrate=RATE (e.g., serial//dev/ttyUSB0#baudrate=115200)",
762                );
763            }
764            // Validate baudrate is numeric
765            if let Some(baud_str) = rest.split("#baudrate=").nth(1) {
766                let baud_str = baud_str.split('#').next().unwrap_or(baud_str);
767                if baud_str.parse::<u32>().is_err() {
768                    return Err("serial baudrate must be a number");
769                }
770            }
771            Ok(())
772        }
773        LocatorProtocol::Unknown => {
774            Err("unknown locator protocol (expected tcp/, udp/, or serial/)")
775        }
776    }
777}

789/// Transport session trait — the per-process anchor an RMW backend
790/// gives to the executor.
791///
792/// # Threading
793///
794/// `&mut self` on every method means the executor serialises all
795/// session calls onto a single thread. A backend may rely on this
796/// — no internal locking is required for `create_*` / `close` /
797/// `drive_io`. **Publisher / subscriber / service handles created
798/// from the session, however, are typically used from worker
799/// threads** and must carry their own synchronisation (see the
800/// [`Publisher`] / [`Subscriber`] trait docs).
801///
802/// # Calling pattern
803///
804/// 1. Open the session (backend-specific factory; not on this trait).
805/// 2. `create_*` for every entity at startup. Creating entities mid-
806///    flight after `drive_io` has run is allowed but not common.
807/// 3. The executor calls `drive_io` periodically. Worker threads
808///    publish / receive in parallel.
809/// 4. `close` once at shutdown. Entities must be dropped first.
810pub trait Session {
811    /// Error type for this session
812    type Error;
813    /// Publisher handle type
814    type PublisherHandle;
815    /// Subscriber handle type
816    type SubscriberHandle;
817    /// Service server handle type
818    type ServiceServerHandle;
819    /// Service client handle type
820    type ServiceClientHandle;
821
822    /// Create a publisher bound to this session.
823    ///
824    /// May allocate transport resources (zenoh declarations, DDS
825    /// writers). Returns a handle that outlives the call but not the
826    /// session — drop the handle before `close()`.
827    fn create_publisher(
828        &mut self,
829        topic: &TopicInfo,
830        qos: QosSettings,
831    ) -> Result<Self::PublisherHandle, Self::Error>;
832
833    /// Create a subscriber bound to this session.
834    ///
835    /// Subscribers may start receiving immediately after creation if
836    /// the transport supports late-joining publishers. Late messages
837    /// are buffered up to the QoS depth.
838    fn create_subscriber(
839        &mut self,
840        topic: &TopicInfo,
841        qos: QosSettings,
842    ) -> Result<Self::SubscriberHandle, Self::Error>;
843
844    /// Create a service server bound to this session. Replies are
845    /// matched to requests by the sequence number returned from
846    /// [`ServiceServerTrait::try_recv_request`].
847    fn create_service_server(
848        &mut self,
849        service: &ServiceInfo,
850    ) -> Result<Self::ServiceServerHandle, Self::Error>;
851
852    /// Create a service client bound to this session.
853    fn create_service_client(
854        &mut self,
855        service: &ServiceInfo,
856    ) -> Result<Self::ServiceClientHandle, Self::Error>;
857
858    /// Close the session, releasing transport resources. All entity
859    /// handles created from this session must already be dropped.
860    fn close(&mut self) -> Result<(), Self::Error>;
861
862    /// Drive transport I/O (poll network, dispatch callbacks).
863    ///
864    /// Both zenoh-pico and XRCE-DDS are pull-based: they require the
865    /// application to periodically call this method to read from the
866    /// network socket and dispatch incoming messages to subscriber
867    /// buffers.
868    ///
869    /// `timeout_ms` is the maximum time to wait for data (0 = non-blocking;
870    /// negative values mean "block indefinitely" — see Phase 84.D7 for the
871    /// planned migration to `core::time::Duration`).
872    ///
873    /// **Required**. There is no default body — both shipped backends
874    /// (zenoh and XRCE) must drive I/O, and a silent no-op default was a
875    /// trap for third-party implementers. If your backend genuinely
876    /// receives data via OS callbacks (push-based) and has nothing to do
877    /// here, return `Ok(())` explicitly.
878    fn drive_io(&mut self, timeout_ms: i32) -> Result<(), Self::Error>;
879
880    /// Phase 109 — report which QoS policies the active backend
881    /// honours. The runtime validates requested QoS against this mask
882    /// at entity-create time and returns
883    /// [`TransportError::IncompatibleQos`] if the requested profile
884    /// includes a policy the backend can't enforce. **No silent
885    /// downgrade.**
886    ///
887    /// Default returns [`QosPolicyMask::CORE`] — reliability +
888    /// durability VOLATILE + history + depth. Backends override per
889    /// supported policy.
890    fn supported_qos_policies(&self) -> QosPolicyMask {
891        QosPolicyMask::CORE
892    }
893}

895/// Bitmask of QoS policies a backend can honour. See
896/// [`Session::supported_qos_policies`].
897///
898/// `CORE` covers the policies every nano-ros backend implements:
899/// reliability, durability=VOLATILE, history, depth. Backends opt
900/// into additional policies by OR-ing the relevant flags.
901#[derive(Debug, Clone, Copy, PartialEq, Eq)]
902pub struct QosPolicyMask(pub u32);
903
904impl QosPolicyMask {
905    pub const RELIABILITY: Self = Self(1 << 0);
906    pub const DURABILITY_VOLATILE: Self = Self(1 << 1);
907    pub const DURABILITY_TRANSIENT_LOCAL: Self = Self(1 << 2);
908    pub const HISTORY: Self = Self(1 << 3);
909    pub const DEPTH: Self = Self(1 << 4);
910    pub const DEADLINE: Self = Self(1 << 5);
911    pub const LIFESPAN: Self = Self(1 << 6);
912    pub const LIVELINESS_AUTOMATIC: Self = Self(1 << 7);
913    pub const LIVELINESS_MANUAL_BY_TOPIC: Self = Self(1 << 8);
914    pub const LIVELINESS_MANUAL_BY_NODE: Self = Self(1 << 9);
915    pub const LIVELINESS_LEASE: Self = Self(1 << 10);
916    pub const AVOID_ROS_NAMESPACE_CONVENTIONS: Self = Self(1 << 11);
917
918    /// Policies every nano-ros backend implements.
919    pub const CORE: Self =
920        Self(Self::RELIABILITY.0 | Self::DURABILITY_VOLATILE.0 | Self::HISTORY.0 | Self::DEPTH.0);
921
922    /// `true` if `self` contains every policy in `other`.
923    pub const fn contains(self, other: Self) -> bool {
924        self.0 & other.0 == other.0
925    }
926
927    /// Bitwise OR of two masks.
928    pub const fn union(self, other: Self) -> Self {
929        Self(self.0 | other.0)
930    }
931}

935    fn bitor(self, rhs: Self) -> Self {
936        self.union(rhs)
937    }
938}

946/// Publisher trait for sending messages.
947///
948/// # Threading
949///
950/// `&self` on `publish_raw` — implementors must allow concurrent
951/// publishes from multiple threads. Internal locking (or lock-free
952/// queues) is the backend's responsibility.
953///
954/// # Buffer ownership
955///
956/// `data` in `publish_raw` is borrowed for the duration of the call.
957/// The backend must either send it inline or copy into its own
958/// buffer before returning — the slice is invalid after the call.
959///
960/// # Blocking
961///
962/// `publish_raw` is expected to be non-blocking on best-effort QoS
963/// and bounded-blocking on reliable QoS (waiting for outbound queue
964/// space). Backends should *not* block waiting for ack from a
965/// matched subscriber.
966pub trait Publisher {
967    /// Error type for publish operations
968    type Error;
969
970    /// Publish a CDR-serialised message.
971    ///
972    /// Returns once the message has been handed to the transport
973    /// (queued or fired-and-forgotten depending on QoS). Does **not**
974    /// wait for delivery.
975    fn publish_raw(&self, data: &[u8]) -> Result<(), Self::Error>;
976
977    /// Publish a typed message (serializes automatically)
978    fn publish<M: RosMessage>(&self, msg: &M, buf: &mut [u8]) -> Result<(), Self::Error> {
979        use nros_core::CdrWriter;
980
981        let mut writer = CdrWriter::new_with_header(buf).map_err(|_| self.buffer_error())?;
982        msg.serialize(&mut writer)
983            .map_err(|_| self.serialization_error())?;
984        let len = writer.position();
985        self.publish_raw(&buf[..len])
986    }
987
988    /// Return a buffer-too-small error (implementation specific)
989    fn buffer_error(&self) -> Self::Error;
990
991    /// Return a serialization error (implementation specific)
992    fn serialization_error(&self) -> Self::Error;
993
994    /// Phase 108 — `true` if the backend can generate this event for
995    /// this publisher. Default returns `false`; backends override per
996    /// supported event kind.
997    ///
998    /// Only [`EventKind::LivelinessLost`] and
999    /// [`EventKind::OfferedDeadlineMissed`] are publisher-side events;
1000    /// other kinds always return `false` here.
1001    fn supports_event(&self, _kind: crate::event::EventKind) -> bool {
1002        false
1003    }
1004
1005    /// Phase 108 — register a callback fired when the named status
1006    /// event occurs. `deadline_ms` applies to
1007    /// [`EventKind::OfferedDeadlineMissed`] only; ignored otherwise.
1008    /// Default impl returns the backend's "unsupported"-shaped error.
1009    #[cfg(feature = "alloc")]
1010    fn register_event_callback(
1011        &mut self,
1012        _kind: crate::event::EventKind,
1013        _deadline_ms: u32,
1014        _cb: crate::event::EventCallback,
1015    ) -> Result<(), Self::Error> {
1016        Err(self.unsupported_event_error())
1017    }
1018
1019    /// Phase 108 — backend's error variant for "this event kind is
1020    /// not supported." Default impl reuses `serialization_error()`
1021    /// since most backends share an `Unsupported` variant; backends
1022    /// override if they have a distinct `Unsupported` mapping.
1023    #[cfg(feature = "alloc")]
1024    fn unsupported_event_error(&self) -> Self::Error {
1025        self.serialization_error()
1026    }
1027
1028    /// Phase 109 — assert this publisher's liveliness manually.
1029    /// Required for publishers configured with
1030    /// `QosLivelinessPolicy::ManualByTopic`. No-op for other
1031    /// liveliness kinds. Default impl returns `Ok(())` (no-op);
1032    /// backends override when they implement manual liveliness.
1033    fn assert_liveliness(&self) -> Result<(), Self::Error> {
1034        Ok(())
1035    }
1036}

1038/// Subscriber trait for receiving messages.
1039///
1040/// # Threading
1041///
1042/// `&mut self` on `try_recv_raw` — the executor takes exclusive
1043/// ownership of the subscriber for the duration of a receive. A
1044/// backend that wants to allow concurrent receives must split into
1045/// per-thread sub-handles internally.
1046///
1047/// # Buffer ownership
1048///
1049/// `buf` is caller-owned. The implementation copies the next ready
1050/// message into `buf` and returns the byte count. The caller may
1051/// re-use or drop `buf` immediately after the call.
1052///
1053/// # Blocking
1054///
1055/// `try_recv_raw` is **non-blocking**: returns `Ok(None)` (or
1056/// equivalent for backends that map empty into a zero-length read)
1057/// when no message is ready. Use [`Session::drive_io`] to wait for
1058/// data; never sleep inside `try_recv_raw`.
1059pub trait Subscriber {
1060    /// Error type for receive operations
1061    type Error;
1062
1063    /// Check if data is available without consuming it.
1064    ///
1065    /// Non-destructive — does not advance the receive cursor.
1066    /// Conservative default returns `true` (always assume data may
1067    /// be available); backends should override with a real check
1068    /// to avoid spurious receive attempts.
1069    fn has_data(&self) -> bool {
1070        true
1071    }
1072
1073    /// Try to receive one message into `buf`.
1074    ///
1075    /// Non-blocking. On success returns `Ok(Some(len))` where `len`
1076    /// is the byte count written into `buf[..len]`. Returns
1077    /// `Ok(None)` if no message is ready. If `buf` is too small the
1078    /// backend may either truncate (and document it) or return an
1079    /// error (preferred).
1080    fn try_recv_raw(&mut self, buf: &mut [u8]) -> Result<Option<usize>, Self::Error>;
1081
1082    /// Try to receive a typed message (non-blocking)
1083    fn try_recv<M: RosMessage>(&mut self, buf: &mut [u8]) -> Result<Option<M>, Self::Error> {
1084        use nros_core::CdrReader;
1085
1086        match self.try_recv_raw(buf)? {
1087            Some(len) => {
1088                let mut reader = CdrReader::new_with_header(&buf[..len])
1089                    .map_err(|_| self.deserialization_error())?;
1090                let msg = M::deserialize(&mut reader).map_err(|_| self.deserialization_error())?;
1091                Ok(Some(msg))
1092            }
1093            None => Ok(None),
1094        }
1095    }
1096
1097    /// Process the received message in-place without copying.
1098    ///
1099    /// Calls `f` with a reference to the raw CDR bytes in the subscriber's
1100    /// internal receive buffer, avoiding a copy into a caller-provided buffer.
1101    /// While `f` executes the buffer is exclusively borrowed — any messages
1102    /// arriving from the transport during that time are dropped to prevent
1103    /// data races.
1104    ///
1105    /// Returns `Ok(true)` if a message was available and `f` was called,
1106    /// `Ok(false)` if no message was available.
1107    ///
1108    /// **Default body**: returns `Err(MessageTooLarge)` — the old default
1109    /// silently truncated anything larger than 1 KB into a stack buffer,
1110    /// which broke large messages with no diagnostic. Backends must
1111    /// override this with a real zero-copy path if they advertise support
1112    /// for `process_raw_in_place`; callers that hit the default should
1113    /// use `try_recv_raw` with a caller-sized buffer instead.
1114    fn process_raw_in_place(&mut self, f: impl FnOnce(&[u8])) -> Result<bool, Self::Error>
1115    where
1116        Self::Error: From<TransportError>,
1117    {
1118        let _ = f;
1119        Err(TransportError::MessageTooLarge.into())
1120    }
1121
1122    /// Try to receive raw data along with publisher metadata.
1123    ///
1124    /// When available, [`MessageInfo`](nros_core::MessageInfo) contains
1125    /// the publisher's GID (Global Identifier) and source timestamp,
1126    /// extracted from a transport-level attachment on the incoming message.
1127    ///
1128    /// Returns `Ok(Some((len, info)))` if data is available, where:
1129    /// - `len` is the number of bytes written to the buffer
1130    /// - `info` is the parsed publisher metadata (if attachment was present)
1131    ///
1132    /// Default: delegates to [`try_recv_raw`](Subscriber::try_recv_raw) with no info.
1133    fn try_recv_raw_with_info(
1134        &mut self,
1135        buf: &mut [u8],
1136    ) -> Result<Option<(usize, Option<nros_core::MessageInfo>)>, Self::Error> {
1137        self.try_recv_raw(buf).map(|opt| opt.map(|len| (len, None)))
1138    }
1139
1140    /// Try to receive raw data with E2E safety validation (CRC + sequence tracking).
1141    ///
1142    /// Returns `Ok(Some((len, status)))` if data is available, where:
1143    /// - `len` is the number of bytes written to the buffer
1144    /// - `status` is the integrity validation result
1145    ///
1146    /// Default: delegates to `try_recv_raw` with no CRC info.
1147    #[cfg(feature = "safety-e2e")]
1148    fn try_recv_validated(
1149        &mut self,
1150        buf: &mut [u8],
1151    ) -> Result<Option<(usize, crate::IntegrityStatus)>, Self::Error> {
1152        self.try_recv_raw(buf).map(|opt| {
1153            opt.map(|len| {
1154                (
1155                    len,
1156                    crate::IntegrityStatus {
1157                        gap: 0,
1158                        duplicate: false,
1159                        crc_valid: None,
1160                    },
1161                )
1162            })
1163        })
1164    }
1165
1166    /// Register an async waker to be notified when data arrives.
1167    ///
1168    /// Called from `Future::poll()` implementations to store the waker.
1169    /// The transport backend calls `waker.wake()` from its receive callback
1170    /// when new data is available, enabling event-driven async without
1171    /// busy-polling.
1172    ///
1173    /// Default: no-op (backends that don't support waking simply ignore this).
1174    fn register_waker(&self, _waker: &core::task::Waker) {}
1175
1176    /// Return a deserialization error (implementation specific)
1177    fn deserialization_error(&self) -> Self::Error;
1178
1179    /// Phase 108 — `true` if the backend can generate this event for
1180    /// this subscriber. Default returns `false`; backends override per
1181    /// supported event kind.
1182    ///
1183    /// Subscriber-side event kinds: [`EventKind::LivelinessChanged`],
1184    /// [`EventKind::RequestedDeadlineMissed`],
1185    /// [`EventKind::MessageLost`]. Publisher kinds always return
1186    /// `false` here.
1187    fn supports_event(&self, _kind: crate::event::EventKind) -> bool {
1188        false
1189    }
1190
1191    /// Phase 108 — register a callback fired when the named status
1192    /// event occurs. `deadline_ms` applies to
1193    /// [`EventKind::RequestedDeadlineMissed`] only; ignored otherwise.
1194    /// Default impl returns the backend's "unsupported"-shaped error.
1195    #[cfg(feature = "alloc")]
1196    fn register_event_callback(
1197        &mut self,
1198        _kind: crate::event::EventKind,
1199        _deadline_ms: u32,
1200        _cb: crate::event::EventCallback,
1201    ) -> Result<(), Self::Error> {
1202        Err(self.unsupported_event_error())
1203    }
1204
1205    /// Phase 108 — backend's error variant for "this event kind is
1206    /// not supported." Default reuses `deserialization_error()` for
1207    /// backends that don't have a distinct `Unsupported` mapping.
1208    #[cfg(feature = "alloc")]
1209    fn unsupported_event_error(&self) -> Self::Error {
1210        self.deserialization_error()
1211    }

1224// ============================================================================
1225//
1226// Backends that can lend a slot directly into their outbound buffer
1227// (zenoh-pico w/ unstable-zenoh-api, XRCE-DDS via uxr_prepare_output_stream,
1228// full DDS w/ SHM transport) implement these traits. Backends that cannot
1229// (uORB, default zenoh-pico) do NOT impl them — `EmbeddedRawPublisher` then
1230// falls back to its per-publisher arena and memcpys at commit time. Both
1231// paths land at `Publisher::publish_raw` for the actual wire write; only
1232// the user-side copy is eliminated when lending is available.
1233//
1234// Selection is **compile-time** via the `lending` Cargo feature. Each
1235// backend crate forwards its own `lending` feature to `nros-rmw/lending`
1236// when it can satisfy the trait. nros-node's `rmw-lending` aggregates.
1237// User opting `nros/rmw-lending` w/ a non-lending backend (e.g. uORB)
1238// gets a clear compile error from the unsatisfied trait bound on the
1239// concrete `RmwPublisher`.

1241/// Backend can lend a writable slot into its outbound buffer.
1242///
1243/// The returned slot's lifetime is tied to `&self`; user fills it in
1244/// place, then calls [`commit_slot`](Self::commit_slot) to publish.
1245/// Dropping the slot without commit is a no-op (slot returned to free
1246/// pool); concurrent loan attempts that would exceed backend capacity
1247/// return [`TransportError::WouldBlock`] — never block.
1248#[cfg(feature = "lending")]
1249pub trait SlotLending: Publisher {
1250    /// Backend-owned writable slot. Holds a `&'a mut [u8]` and any
1251    /// state needed for commit_slot.
1252    type Slot<'a>: AsMut<[u8]> + 'a
1253    where
1254        Self: 'a;
1255
1256    /// Reserve a writable slot of `len` bytes from the backend's
1257    /// outbound buffer. Returns `Ok(None)` if the backend has no slot
1258    /// available (full); never blocks.
1259    fn try_lend_slot(&self, len: usize) -> Result<Option<Self::Slot<'_>>, Self::Error>;
1260
1261    /// Commit a previously-lent slot. Consumes the slot and triggers
1262    /// the actual wire write. Returns `Err` on backend send failure;
1263    /// the slot's bytes are lost in that case (caller must re-lend +
1264    /// re-fill to retry).
1265    fn commit_slot(&self, slot: Self::Slot<'_>) -> Result<(), Self::Error>;
1266}

1268/// Backend can lend a read-only view into its receive buffer.
1269///
1270/// The returned view's lifetime is tied to `&mut self` (subscriber-
1271/// exclusive); dropping the view releases any backend lock and lets
1272/// the next message advance into the buffer.
1273#[cfg(feature = "lending")]
1274pub trait SlotBorrowing: Subscriber {
1275    /// Backend-owned read-only view. Holds a `&'a [u8]` and any state
1276    /// needed to release the borrow on Drop.
1277    type View<'a>: AsRef<[u8]> + 'a
1278    where
1279        Self: 'a;
1280
1281    /// Try to borrow the next available message in place. Returns
1282    /// `Ok(None)` if no message is ready; never blocks.
1283    fn try_borrow(&mut self) -> Result<Option<Self::View<'_>>, Self::Error>;
1284}

1286/// Service server trait for handling requests.
1287///
1288/// # Threading
1289///
1290/// `&mut self` on `try_recv_request` and `send_reply` — the executor
1291/// owns the server while a request is being handled. Handler bodies
1292/// run synchronously on the executor thread; long handlers should
1293/// dispatch work to a worker queue and reply later via the recorded
1294/// `sequence_number`.
1295///
1296/// # Calling pattern
1297///
1298/// 1. Executor calls `try_recv_request(buf)`.
1299/// 2. If `Some(req)` returned, decode, run handler, encode reply.
1300/// 3. `send_reply(req.sequence_number, &reply_buf)`.
1301///
1302/// `sequence_number` is the canonical request → reply correlation
1303/// token; backends derive it from the wire-level metadata (zenoh
1304/// query id, DDS sample identity).
1305pub trait ServiceServerTrait {
1306    /// Error type for service operations
1307    type Error;
1308
1309    /// Check if a request is available without consuming it.
1310    ///
1311    /// Non-destructive. Default returns `true` (always assume one
1312    /// may be available); backends should override with a real
1313    /// check.
1314    fn has_request(&self) -> bool {
1315        true
1316    }
1317
1318    /// Try to receive a service request into `buf` (non-blocking).
1319    ///
1320    /// On success returns a `ServiceRequest` that borrows from
1321    /// `buf`. The borrow is released when the returned struct is
1322    /// dropped — typically before `send_reply` is called, since
1323    /// `send_reply` takes `&mut self`.
1324    fn try_recv_request<'a>(
1325        &mut self,
1326        buf: &'a mut [u8],
1327    ) -> Result<Option<ServiceRequest<'a>>, Self::Error>;
1328
1329    /// Send a reply for the given sequence number. Non-blocking
1330    /// from the application's perspective; the backend may queue
1331    /// the reply for transport-level transmission.
1332    fn send_reply(&mut self, sequence_number: i64, data: &[u8]) -> Result<(), Self::Error>;
1333
1334    /// Handle a service request with typed messages
1335    fn handle_request<S: RosService>(
1336        &mut self,
1337        req_buf: &mut [u8],
1338        reply_buf: &mut [u8],
1339        handler: impl FnOnce(&S::Request) -> S::Reply,
1340    ) -> Result<bool, Self::Error>
1341    where
1342        Self::Error: From<TransportError>,
1343    {
1344        use nros_core::{CdrReader, CdrWriter};
1345
1346        // First, try to receive a request and extract necessary data.
1347        // Capture the data slice's offset within `req_buf` so we can
1348        // re-borrow it after the `ServiceRequest` (which holds a
1349        // borrow into `req_buf`) is dropped. Some backends prepend a
1350        // header (DDS: 8-byte sequence number) and place the CDR
1351        // payload at a non-zero offset in the buffer; others (zenoh)
1352        // put it at offset 0. Reading from offset 0 unconditionally
1353        // would feed the prefix bytes to the CDR deserializer and
1354        // silently corrupt the request.
1355        let buf_start = req_buf.as_ptr() as usize;
1356        let (data_offset, data_len, sequence_number) = match self.try_recv_request(req_buf)? {
1357            Some(request) => {
1358                let offset = (request.data.as_ptr() as usize).saturating_sub(buf_start);
1359                (offset, request.data.len(), request.sequence_number)
1360            }
1361            None => return Ok(false),
1362        };
1363
1364        // Deserialize request from the captured offset.
1365        let mut reader = CdrReader::new_with_header(&req_buf[data_offset..data_offset + data_len])
1366            .map_err(|_| TransportError::DeserializationError)?;
1367        let req = S::Request::deserialize(&mut reader)
1368            .map_err(|_| TransportError::DeserializationError)?;
1369
1370        // Call handler
1371        let reply = handler(&req);
1372
1373        // Serialize reply
1374        let mut writer =
1375            CdrWriter::new_with_header(reply_buf).map_err(|_| TransportError::BufferTooSmall)?;
1376        reply
1377            .serialize(&mut writer)
1378            .map_err(|_| TransportError::SerializationError)?;
1379        let len = writer.position();
1380
1381        // Send reply (now we can borrow self mutably again)
1382        self.send_reply(sequence_number, &reply_buf[..len])?;
1383        Ok(true)
1384    }
1385
1386    /// Handle a service request where the handler returns `Box<S::Reply>`
1387    ///
1388    /// Identical to `handle_request` but the handler returns a heap-allocated reply.
1389    /// This is needed for services with large response types (e.g., parameter services
1390    /// where `Vec<ParameterValue, 64>` is ~1MB+) that would overflow the stack.
1391    #[cfg(feature = "alloc")]
1392    fn handle_request_boxed<S: RosService>(
1393        &mut self,
1394        req_buf: &mut [u8],
1395        reply_buf: &mut [u8],
1396        handler: impl FnOnce(&S::Request) -> alloc::boxed::Box<S::Reply>,
1397    ) -> Result<bool, Self::Error>
1398    where
1399        Self::Error: From<TransportError>,
1400    {
1401        use nros_core::{CdrReader, CdrWriter};
1402
1403        let buf_start = req_buf.as_ptr() as usize;
1404        let (data_offset, data_len, sequence_number) = match self.try_recv_request(req_buf)? {
1405            Some(request) => {
1406                let offset = (request.data.as_ptr() as usize).saturating_sub(buf_start);
1407                (offset, request.data.len(), request.sequence_number)
1408            }
1409            None => return Ok(false),
1410        };
1411
1412        let mut reader = CdrReader::new_with_header(&req_buf[data_offset..data_offset + data_len])
1413            .map_err(|_| TransportError::DeserializationError)?;
1414        let req = S::Request::deserialize(&mut reader)
1415            .map_err(|_| TransportError::DeserializationError)?;
1416
1417        let reply = handler(&req);
1418
1419        let mut writer =
1420            CdrWriter::new_with_header(reply_buf).map_err(|_| TransportError::BufferTooSmall)?;
1421        reply
1422            .serialize(&mut writer)
1423            .map_err(|_| TransportError::SerializationError)?;
1424        let len = writer.position();
1425
1426        self.send_reply(sequence_number, &reply_buf[..len])?;
1427        Ok(true)
1428    }
1429}

1431/// Service client trait for sending requests.
1432///
1433/// # Threading
1434///
1435/// `&mut self` on every method — the client is single-owner. For
1436/// fan-out request patterns, create one client per worker thread.
1437///
1438/// # Calling pattern
1439///
1440/// All in-tree backends route blocking waits through the executor:
1441///
1442/// 1. `send_request_raw(buf)` — non-blocking; returns once the
1443///    request is queued for transmission.
1444/// 2. The executor's `drive_io` runs.
1445/// 3. `try_recv_reply_raw(buf)` — non-blocking; returns
1446///    `Ok(Some(len))` when the reply is back.
1447///
1448/// The deprecated [`call_raw`](Self::call_raw) blocking path is
1449/// kept for backwards compatibility but should not be called.
1450pub trait ServiceClientTrait {
1451    /// Error type for service operations
1452    type Error;
1453
1454    /// Send a service request and wait for reply (blocking).
1455    ///
1456    /// **Deprecated — do not call.** The default body returns `Timeout`
1457    /// immediately without polling. Use `Client::call` →
1458    /// `Promise::wait(executor, timeout_ms)` which lets the executor
1459    /// drive I/O while waiting instead of busy-looping on
1460    /// `try_recv_reply_raw` with no sleep (which starves the transport
1461    /// on FreeRTOS / Zephyr single-threaded schedulers).
1462    ///
1463    /// Backends that still need an internal blocking path should
1464    /// override this with a real sleep-between-polls implementation,
1465    /// but all in-tree backends (zenoh, XRCE) route blocking waits
1466    /// through the executor.
1467    #[deprecated(note = "use Client::call → Promise::wait with an executor instead")]
1468    fn call_raw(&mut self, request: &[u8], _reply_buf: &mut [u8]) -> Result<usize, Self::Error>
1469    where
1470        Self::Error: From<TransportError>,
1471    {
1472        let _ = request;
1473        Err(TransportError::Timeout.into())
1474    }
1475
1476    /// Send a service request without waiting for a reply (non-blocking).
1477    ///
1478    /// The caller must subsequently poll [`try_recv_reply_raw`](Self::try_recv_reply_raw)
1479    /// to retrieve the reply.
1480    fn send_request_raw(&mut self, request: &[u8]) -> Result<(), Self::Error>;
1481
1482    /// Poll for a reply to the most recently sent request (non-blocking).
1483    ///
1484    /// Returns `Ok(Some(len))` when a reply has arrived, `Ok(None)` if not yet
1485    /// available, or `Err` on failure.
1486    fn try_recv_reply_raw(&mut self, reply_buf: &mut [u8]) -> Result<Option<usize>, Self::Error>;
1487
1488    /// Send a typed service request without waiting for a reply (non-blocking).
1489    ///
1490    /// Serializes the request into `req_buf` and calls [`send_request_raw`](Self::send_request_raw).
1491    fn send_request<S: RosService>(
1492        &mut self,
1493        request: &S::Request,
1494        req_buf: &mut [u8],
1495    ) -> Result<(), Self::Error>
1496    where
1497        Self::Error: From<TransportError>,
1498    {
1499        use nros_core::CdrWriter;
1500
1501        let mut writer =
1502            CdrWriter::new_with_header(req_buf).map_err(|_| TransportError::BufferTooSmall)?;
1503        request
1504            .serialize(&mut writer)
1505            .map_err(|_| TransportError::SerializationError)?;
1506        let req_len = writer.position();
1507
1508        self.send_request_raw(&req_buf[..req_len])
1509    }
1510
1511    /// Poll for a typed reply to the most recently sent request (non-blocking).
1512    ///
1513    /// Calls [`try_recv_reply_raw`](Self::try_recv_reply_raw) and deserializes if available.
1514    fn try_recv_reply<S: RosService>(
1515        &mut self,
1516        reply_buf: &mut [u8],
1517    ) -> Result<Option<S::Reply>, Self::Error>
1518    where
1519        Self::Error: From<TransportError>,
1520    {
1521        use nros_core::CdrReader;
1522
1523        match self.try_recv_reply_raw(reply_buf)? {
1524            Some(len) => {
1525                let mut reader = CdrReader::new_with_header(&reply_buf[..len])
1526                    .map_err(|_| TransportError::DeserializationError)?;
1527                let reply = S::Reply::deserialize(&mut reader)
1528                    .map_err(|_| TransportError::DeserializationError)?;
1529                Ok(Some(reply))
1530            }
1531            None => Ok(None),
1532        }
1533    }
1534
1535    /// Register an async waker to be notified when a reply arrives.
1536    ///
1537    /// Called from `Future::poll()` implementations to store the waker.
1538    /// The transport backend calls `waker.wake()` from its reply callback
1539    /// when a response is available, enabling event-driven async without
1540    /// busy-polling.
1541    ///
1542    /// Default: no-op (backends that don't support waking simply ignore this).
1543    fn register_waker(&self, _waker: &core::task::Waker) {}
1544
1545    /// Begin a server-discovery query on this client (non-blocking).
1546    ///
1547    /// Models `rclcpp::ClientBase::wait_for_service` machinery: the backend
1548    /// fires off a discovery probe (typically a Zenoh liveliness query
1549    /// against the matching server's wildcarded liveliness keyexpr) and
1550    /// the caller polls [`poll_server_discovery`](Self::poll_server_discovery)
1551    /// to collect the result.
1552    ///
1553    /// Default impl: no-op success. Backends without a discovery channel
1554    /// (or those that always assume the server is reachable) can leave
1555    /// this default and have `poll_server_discovery` return
1556    /// `Ok(Some(true))` immediately.
1557    fn start_server_discovery(&mut self, _timeout_ms: u32) -> Result<(), Self::Error> {
1558        Ok(())
1559    }
1560
1561    /// Poll an in-flight server-discovery query.
1562    ///
1563    /// - `Ok(Some(true))` — at least one matching server has reported
1564    ///   back; safe to send the first request.
1565    /// - `Ok(Some(false))` — discovery query finished without finding
1566    ///   any matching server (timeout / no-replies).
1567    /// - `Ok(None)` — query still in flight.
1568    /// - `Err(_)` — transport-level failure unrelated to server presence.
1569    ///
1570    /// Default impl: returns `Ok(Some(true))` (i.e., "server is always
1571    /// assumed reachable"). The Zenoh backend overrides this with a
1572    /// liveliness-token check.
1573    fn poll_server_discovery(&mut self) -> Result<Option<bool>, Self::Error> {
1574        Ok(Some(true))
1575    }
1576
1577    /// Synchronous, non-blocking check of whether a matching server is
1578    /// currently visible.
1579    ///
1580    /// Mirrors `rclcpp::ClientBase::service_is_ready`. Backends that lack
1581    /// discovery should keep the default `true` so existing call sites
1582    /// don't regress.
1583    ///
1584    /// Default impl: always `true`.
1585    fn is_server_ready(&self) -> bool {
1586        true
1587    }
1588
1589    /// Call a service with typed messages (blocking).
1590    ///
1591    /// **Deprecated — do not call.** The default body returns `Timeout`
1592    /// immediately without polling. Use `Client::call` on the executor
1593    /// instead, which drives I/O while waiting. See
1594    /// [`call_raw`](Self::call_raw) for the same reasoning.
1595    #[deprecated(note = "use Client::call → Promise::wait with an executor instead")]
1596    fn call<S: RosService>(
1597        &mut self,
1598        request: &S::Request,
1599        req_buf: &mut [u8],
1600        _reply_buf: &mut [u8],
1601    ) -> Result<S::Reply, Self::Error>
1602    where
1603        Self::Error: From<TransportError>,
1604    {
1605        use nros_core::CdrWriter;
1606
1607        // Serialize request so the error surface matches the old impl
1608        // for the "bad request" path; but skip the receive busy-loop.
1609        let mut writer =
1610            CdrWriter::new_with_header(req_buf).map_err(|_| TransportError::BufferTooSmall)?;
1611        request
1612            .serialize(&mut writer)
1613            .map_err(|_| TransportError::SerializationError)?;
1614        Err(TransportError::Timeout.into())
1615    }
1616}
