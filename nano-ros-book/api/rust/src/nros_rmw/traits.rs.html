1//! Transport abstraction traits.
2//!
3//! Defines the backend-agnostic interface that transport implementations
4//! (zenoh-pico, XRCE-DDS) must satisfy. The core trait hierarchy is:
5//!
6//! - [`Session`] — connection lifecycle and handle creation
7//! - [`Publisher`] / [`Subscriber`] — pub/sub data transport
8//! - [`ServiceServerTrait`] / [`ServiceClientTrait`] — request/reply
9//! - [`Rmw`] — top-level factory that creates sessions

13/// Topic information for pub/sub
14#[derive(Debug, Clone)]
15pub struct TopicInfo<'a> {
16    /// Topic name (e.g., "/chatter")
17    pub name: &'a str,
18    /// ROS type name (e.g., "std_msgs::msg::dds_::String_")
19    pub type_name: &'a str,
20    /// Type hash for compatibility checking
21    pub type_hash: &'a str,
22    /// Domain ID (default: 0)
23    pub domain_id: u32,
24    /// Node name for liveliness token generation.
25    /// `None` means no node association — no liveliness token will be declared.
26    pub node_name: Option<&'a str>,
27    /// Node namespace for liveliness token generation (default: "/").
28    /// In ROS 2, "/" is the root namespace and the standard default.
29    pub namespace: &'a str,
30    /// Phase 231 (RFC-0038) — receive-buffer size hint, bytes. The executor sets
31    /// this from the subscription's `RX_BUF`; a backend may use it to route the
32    /// subscription to a size-class receive buffer (zenoh-pico: small vs large).
33    /// `0` = unset (backend picks its default). Ignored by backends that don't
34    /// size-class their receive storage.
35    pub rx_buffer_hint: usize,
36}

165    /// Generate a sub-entity name for an action component
166    /// Returns: `<action>/_action/<suffix>` (e.g., `fibonacci/_action/send_goal`)
167    ///
168    /// The caller is responsible for constructing the full key expression
169    /// by wrapping this name in a `ServiceInfo` or `TopicInfo` with the
170    /// correct sub-service/sub-topic type name.
171    fn sub_name<const N: usize>(&self, suffix: &str) -> heapless::String<N> {
172        let mut name = heapless::String::new();
173        let action_stripped = self.name.trim_matches('/');
174        let _ = core::fmt::write(
175            &mut name,
176            format_args!("/{}/_action/{}", action_stripped, suffix),
177        );
178        name
179    }
180}

214/// Transport error types.
215///
216/// No longer `Copy` — the `Backend` / `BackendDynamic` variants carry a
217/// string diagnostic, which can't be `Copy`. Rust callers that used to
218/// copy a `TransportError` value repeatedly now need `.clone()` or
219/// `ref` in match arms. C/C++ callers are unaffected — both map
220/// `TransportError` to integer codes (`nros_ret_t` / `ErrorCode`)
221/// before crossing the FFI boundary.
222#[derive(Debug, Clone, PartialEq, Eq)]
223pub enum TransportError {
224    /// Failed to connect to transport
225    ConnectionFailed,
226    /// Connection was closed
227    Disconnected,
228    /// Failed to create publisher
229    PublisherCreationFailed,
230    /// Failed to create subscriber
231    SubscriberCreationFailed,
232    /// Failed to create service server
233    ServiceServerCreationFailed,
234    /// Failed to create service client
235    ServiceClientCreationFailed,
236    /// Failed to publish message
237    PublishFailed,
238    /// Failed to send service request
239    ServiceRequestFailed,
240    /// Failed to send service reply
241    ServiceReplyFailed,
242    /// Serialization error
243    SerializationError,
244    /// Deserialization error
245    DeserializationError,
246    /// Buffer too small
247    BufferTooSmall,
248    /// Incoming message exceeded the static buffer capacity
249    MessageTooLarge,
250    /// Timeout waiting for message
251    Timeout,
252    /// Invalid configuration
253    InvalidConfig,
254    /// Resource (slot, buffer, queue) momentarily unavailable. Retry.
255    /// Phase 99: returned by `try_loan` when arena slots are full and
256    /// by `try_borrow` when no message is ready (alternative to
257    /// `Ok(None)` for backends that prefer the error variant).
258    WouldBlock,
259    /// Requested allocation exceeds backend capacity. Phase 99:
260    /// `try_loan(len)` returns this when `len` > arena slot size.
261    TooLarge,
262    /// Failed to start background tasks
263    TaskStartFailed,
264    /// Failed to poll for incoming messages
265    PollFailed,
266    /// Failed to send keepalive
267    KeepaliveFailed,
268    /// Failed to send join message
269    JoinFailed,
270    /// Caller supplied a NULL pointer, an out-of-range value, or an
271    /// inconsistent argument combination. Phase 102.1.
272    InvalidArgument,
273    /// The backend does not implement this operation. Optional
274    /// callbacks return this; the runtime then falls back to a
275    /// default path. Phase 102.1.
276    Unsupported,
277    /// Memory allocation failed. Returned by backends on `std` /
278    /// `alloc`-equipped platforms when heap allocation fails.
279    /// Bare-metal backends generally do not produce this — they
280    /// preallocate at session-open time. Phase 102.1.
281    BadAlloc,
282    /// Publisher and subscriber QoS profiles do not match in a way
283    /// the backend cannot reconcile. Phase 102.1.
284    IncompatibleQos,
285    /// Topic, service, or action name failed validation. Phase 102.1.
286    TopicNameInvalid,
287    /// A request referenced a node that does not exist in this
288    /// session. Phase 102.1.
289    NodeNameNonExistent,
290    /// The backend does not support loaned messages on this entity,
291    /// or the loan slot is currently in use. Phase 102.1.
292    LoanNotSupported,
293    /// No data was available on a non-blocking receive. Distinct
294    /// from `Timeout`: fires immediately, not after a bounded wait.
295    /// Phase 102.1.
296    NoData,
297    /// Phase 115.A.2 — caller passed a versioned vtable struct
298    /// (e.g. `NrosTransportOps`) with an `abi_version` the runtime
299    /// doesn't know. Maps to `NROS_RMW_RET_INCOMPATIBLE_ABI` at
300    /// the C boundary.
301    IncompatibleAbi,
302    /// Backend-specific error with a `'static` diagnostic string.
303    ///
304    /// Useful for zenoh-pico / XRCE-DDS return codes that map to a
305    /// fixed set of known messages. `no_std`-compatible.
306    Backend(&'static str),
307    /// Backend-specific error with an owned diagnostic string.
308    ///
309    /// Available only with the `alloc` feature. Use this when the
310    /// diagnostic is formatted at runtime (e.g. from a C error code
311    /// plus a socket address).
312    #[cfg(feature = "alloc")]
313    BackendDynamic(alloc::string::String),
314}

365/// Phase 211.H — which side of a topic a [`QosOverride`] targets.
366/// Mirrors the `<role>` segment of a ROS 2

369pub enum QosOverrideRole {

371    Publisher,
372    /// `qos_overrides.<topic>.subscription.*`
373    Subscription,
374}

376/// Phase 211.H — a single policy value a [`QosOverride`] sets. A typed enum
377/// (not a string) so the codegen that bakes these from the plan catches an
378/// unknown policy / mistyped value at generation time rather than silently
379/// no-op-ing at runtime.
380#[derive(Debug, Clone, Copy, PartialEq, Eq)]
381pub enum QosOverrideValue {
382    /// `.reliability` → Reliable / BestEffort.
383    Reliability(QosReliabilityPolicy),
384    /// `.durability` → Volatile / TransientLocal.
385    Durability(QosDurabilityPolicy),
386    /// `.history` → KeepLast / KeepAll.
387    History(QosHistoryPolicy),
388    /// `.depth` → KeepLast depth.
389    Depth(u32),
390}

392/// Phase 211.H — one per-topic QoS override, lowered from a ROS 2
393/// `qos_overrides.<topic>.<role>.<policy>` launch parameter by the planner and
394/// baked into a `&'static [QosOverride]` table by the entry codegen. The node
395/// folds the matching entries into the entity's [`QosSettings`] at
396/// `create_publisher` / `create_subscription` time (setup-time, single
397/// linear scan, no alloc), *before* the backend-compat `validate_against` —
398/// so an override the active RMW can't honour still errors loudly, never a
399/// silent downgrade.
400#[derive(Debug, Clone, Copy, PartialEq, Eq)]
401pub struct QosOverride {
402    /// The resolved (remapped) topic name the override targets, e.g.
403    /// `"/chatter"`. Matched exactly against the entity's topic.
404    pub topic: &'static str,
405    /// Publisher or subscription side.
406    pub role: QosOverrideRole,
407    /// The policy + value to set.
408    pub value: QosOverrideValue,
409}

411/// Full DDS-shaped QoS profile. Matches the field set of upstream
412/// `rmw_qos_profile_t`.
413///
414/// Backends advertise per-policy support via
415/// [`Session::supported_qos_policies`]; entities created with a
416/// profile the active backend can't honour return
417/// [`TransportError::IncompatibleQos`] synchronously at create time
418/// — no silent downgrade.
419///
420/// Zero-valued time-window fields ("off") mean infinite — the policy
421/// is effectively disabled for the entity.
422#[derive(Debug, Clone, Copy, PartialEq, Eq)]
423pub struct QosSettings {
424    /// History policy
425    pub history: QosHistoryPolicy,
426    /// Reliability policy
427    pub reliability: QosReliabilityPolicy,
428    /// Durability policy
429    pub durability: QosDurabilityPolicy,
430    /// Liveliness policy
431    pub liveliness_kind: QosLivelinessPolicy,
432    /// History depth (only used if history is KeepLast)
433    pub depth: u32,
434    /// Subscriber max-inter-arrival / publisher offered-rate, ms.
435    /// `0` = infinite (no deadline check).
436    pub deadline_ms: u32,
437    /// Sample expiry, ms. Subscribers filter samples older than this.
438    /// `0` = infinite (no expiry).
439    pub lifespan_ms: u32,
440    /// Liveliness lease, ms. `0` = infinite.
441    pub liveliness_lease_ms: u32,
442    /// If `true`, topic-name encoding skips the `/rt/` ROS prefix.
443    pub avoid_ros_namespace_conventions: bool,
444}

452impl QosSettings {
453    /// Phase 211.H — fold the plan's `qos_overrides` matching `topic` + `role`
454    /// into this profile, returning the overridden profile. Setup-time only
455    /// (called from `create_publisher`/`create_subscription`): a single linear
456    /// scan over the baked `&'static` table, no alloc, RT-safe. Later entries
457    /// win on a duplicate `(topic, role, policy)` (last-write), matching the
458    /// planner's sorted, de-conflicted emit. Non-matching entries are ignored,
459    /// so passing the whole node table to every entity is cheap + correct.
460    #[must_use]
461    pub fn apply_overrides(
462        mut self,
463        topic: &str,
464        role: QosOverrideRole,
465        overrides: &[QosOverride],
466    ) -> Self {
467        for ovr in overrides {
468            if ovr.topic == topic && ovr.role == role {
469                match ovr.value {
470                    QosOverrideValue::Reliability(r) => self.reliability = r,
471                    QosOverrideValue::Durability(d) => self.durability = d,
472                    QosOverrideValue::History(h) => self.history = h,
473                    QosOverrideValue::Depth(d) => self.depth = d,
474                }
475            }
476        }
477        self
478    }
479}

832/// Locator transport protocol
833#[derive(Debug, Clone, Copy, PartialEq, Eq)]
834pub enum LocatorProtocol {
835    /// TCP transport (e.g., "tcp/127.0.0.1:7447")
836    Tcp,
837    /// UDP transport (e.g., "udp/192.168.1.50:2019" — common for XRCE-DDS)
838    Udp,
839    /// Serial/UART transport (e.g., "serial//dev/ttyUSB0#baudrate=115200")
840    Serial,
841    /// Unknown protocol
842    Unknown,
843}
844
845/// Parse the protocol from a locator string
846pub fn locator_protocol(locator: &str) -> LocatorProtocol {
847    if locator.starts_with("tcp/") {
848        LocatorProtocol::Tcp
849    } else if locator.starts_with("udp/") {
850        LocatorProtocol::Udp
851    } else if locator.starts_with("serial/") {
852        LocatorProtocol::Serial
853    } else {
854        LocatorProtocol::Unknown
855    }
856}
857
858/// Validate a locator string format.
859///
860/// Returns `Ok(())` if the locator is well-formed, or an error message describing
861/// the problem. This provides early feedback before zenoh-pico or XRCE-DDS rejects
862/// a bad locator.
863///
864/// Supported formats:
865/// - TCP: `tcp/<host>:<port>` (e.g., `tcp/127.0.0.1:7447`)
866/// - UDP: `udp/<host>:<port>` (e.g., `udp/192.168.1.50:2019`)
867/// - Serial: `serial/<device>#baudrate=<rate>` (e.g., `serial//dev/ttyUSB0#baudrate=115200`)
868pub fn validate_locator(locator: &str) -> Result<(), &'static str> {
869    match locator_protocol(locator) {
870        LocatorProtocol::Tcp => {
871            let rest = &locator[4..]; // skip "tcp/"
872            if !rest.contains(':') {
873                return Err("TCP locator must contain host:port (e.g., tcp/127.0.0.1:7447)");
874            }
875            Ok(())
876        }
877        LocatorProtocol::Udp => {
878            let rest = &locator[4..]; // skip "udp/"
879            if !rest.contains(':') {
880                return Err("UDP locator must contain host:port (e.g., udp/192.168.1.50:2019)");
881            }
882            Ok(())
883        }
884        LocatorProtocol::Serial => {
885            let rest = &locator[7..]; // skip "serial/"
886            if rest.is_empty() {
887                return Err(
888                    "serial locator must specify device (e.g., serial//dev/ttyUSB0#baudrate=115200)",
889                );
890            }
891            if !rest.contains("#baudrate=") {
892                return Err(
893                    "serial locator must include #baudrate=RATE (e.g., serial//dev/ttyUSB0#baudrate=115200)",
894                );
895            }
896            // Validate baudrate is numeric
897            if let Some(baud_str) = rest.split("#baudrate=").nth(1) {
898                let baud_str = baud_str.split('#').next().unwrap_or(baud_str);
899                if baud_str.parse::<u32>().is_err() {
900                    return Err("serial baudrate must be a number");
901                }
902            }
903            Ok(())
904        }
905        LocatorProtocol::Unknown => {
906            Err("unknown locator protocol (expected tcp/, udp/, or serial/)")
907        }
908    }
909}

921/// Transport session trait — the per-process anchor an RMW backend
922/// gives to the executor.
923///
924/// # Threading
925///
926/// `&mut self` on every method means the executor serialises all
927/// session calls onto a single thread. A backend may rely on this
928/// — no internal locking is required for `create_*` / `close` /
929/// `drive_io`. **Publisher / subscriber / service handles created
930/// from the session, however, are typically used from worker
931/// threads** and must carry their own synchronisation (see the
932/// [`Publisher`] / [`Subscriber`] trait docs).
933///
934/// # Calling pattern
935///
936/// 1. Open the session (backend-specific factory; not on this trait).
937/// 2. `create_*` for every entity at startup. Creating entities mid-
938///    flight after `drive_io` has run is allowed but not common.
939/// 3. The executor calls `drive_io` periodically. Worker threads
940///    publish / receive in parallel.
941/// 4. `close` once at shutdown. Entities must be dropped first.
942pub trait Session {
943    /// Error type for this session
944    type Error;
945    /// Publisher handle type
946    type PublisherHandle;
947    /// Subscriber handle type
948    type SubscriberHandle;
949    /// Service server handle type
950    type ServiceServerHandle;
951    /// Service client handle type
952    type ServiceClientHandle;
953
954    /// Create a publisher bound to this session.
955    ///
956    /// May allocate transport resources (zenoh declarations, DDS
957    /// writers). Returns a handle that outlives the call but not the
958    /// session — drop the handle before `close()`.
959    fn create_publisher(
960        &mut self,
961        topic: &TopicInfo,
962        qos: QosSettings,
963    ) -> Result<Self::PublisherHandle, Self::Error>;
964
965    /// Create a subscriber bound to this session.
966    ///
967    /// Subscribers may start receiving immediately after creation if
968    /// the transport supports late-joining publishers. Late messages
969    /// are buffered up to the QoS depth.
970    fn create_subscriber(
971        &mut self,
972        topic: &TopicInfo,
973        qos: QosSettings,
974    ) -> Result<Self::SubscriberHandle, Self::Error>;
975
976    /// Create a service server bound to this session. Replies are
977    /// matched to requests by the sequence number returned from
978    /// [`ServiceServerTrait::try_recv_request`].
979    ///
980    /// `qos` is applied to both the request and reply endpoints (a
981    /// service is two DDS topics; rmw uses one profile for both). The
982    /// default is [`QosSettings::services_default`]
983    /// (RELIABLE+VOLATILE+KEEP_LAST(10)).
984    fn create_service_server(
985        &mut self,
986        service: &ServiceInfo,
987        qos: QosSettings,
988    ) -> Result<Self::ServiceServerHandle, Self::Error>;
989
990    /// Create a service client bound to this session.
991    ///
992    /// `qos` is applied to both the request and reply endpoints (a
993    /// service is two DDS topics; rmw uses one profile for both). The
994    /// default is [`QosSettings::services_default`]
995    /// (RELIABLE+VOLATILE+KEEP_LAST(10)).
996    fn create_service_client(
997        &mut self,
998        service: &ServiceInfo,
999        qos: QosSettings,
1000    ) -> Result<Self::ServiceClientHandle, Self::Error>;
1001
1002    /// Close the session, releasing transport resources. All entity
1003    /// handles created from this session must already be dropped.
1004    fn close(&mut self) -> Result<(), Self::Error>;
1005
1006    /// Drive transport I/O (poll network, dispatch callbacks).
1007    ///
1008    /// Both zenoh-pico and XRCE-DDS are pull-based: they require the
1009    /// application to periodically call this method to read from the
1010    /// network socket and dispatch incoming messages to subscriber
1011    /// buffers.
1012    ///
1013    /// `timeout_ms` is the maximum time to wait for data (0 = non-blocking;
1014    /// negative values mean "block indefinitely" — see Phase 84.D7 for the
1015    /// planned migration to `core::time::Duration`).
1016    ///
1017    /// **Required**. There is no default body — both shipped backends
1018    /// (zenoh and XRCE) must drive I/O, and a silent no-op default was a
1019    /// trap for third-party implementers. If your backend genuinely
1020    /// receives data via OS callbacks (push-based) and has nothing to do
1021    /// here, return `Ok(())` explicitly.
1022    fn drive_io(&mut self, timeout_ms: i32) -> Result<(), Self::Error>;
1023
1024    /// Phase 109 — report which QoS policies the active backend
1025    /// honours. The runtime validates requested QoS against this mask
1026    /// at entity-create time and returns
1027    /// [`TransportError::IncompatibleQos`] if the requested profile
1028    /// includes a policy the backend can't enforce. **No silent
1029    /// downgrade.**
1030    ///
1031    /// Default returns [`QosPolicyMask::CORE`] — reliability +
1032    /// durability VOLATILE + history + depth. Backends override per
1033    /// supported policy.
1034    fn supported_qos_policies(&self) -> QosPolicyMask {
1035        QosPolicyMask::CORE
1036    }
1037
1038    /// Phase 110.0 — backend's next internal-event deadline in
1039    /// milliseconds from now (lease keepalive, heartbeat, reader
1040    /// ACK-NACK timeout, etc.).
1041    ///
1042    /// The executor caps its `drive_io` timeout against
1043    /// `min(user_timeout, timer_deadline, this)` so quiet links don't
1044    /// wake early, see no user-visible work, and round-trip back into
1045    /// `drive_io`. Returns `None` when the backend has no internal
1046    /// deadlines or chooses not to expose them.
1047    ///
1048    /// Default `None` keeps existing backends working unchanged; opt-in
1049    /// per backend.
1050    fn next_deadline_ms(&self) -> Option<u32> {
1051        None
1052    }
1053
1054    /// Phase 124.B.1 — install (or clear, when `cb.is_none()`) the
1055    /// executor wake callback. The runtime calls this once per
1056    /// session after `open` with `cb` pointing at a runtime-owned
1057    /// function and `ctx` pointing at the executor's wake state.
1058    /// The backend stores `(cb, ctx)` in its per-session state and
1059    /// calls `cb(ctx)` whenever its transport notification path
1060    /// fires (datagram arrival, condvar wake, etc.) — the runtime
1061    /// cb does flag-write + condvar-signal atomically, so a
1062    /// `spin_once` blocked on the wake condvar resumes immediately
1063    /// instead of waiting for the next poll iteration.
1064    ///
1065    /// # Safety
1066    ///
1067    /// When `cb` is `Some`, `ctx` must remain valid until the callback is
1068    /// cleared or the session is closed. The backend may invoke `cb(ctx)` from
1069    /// its transport notification path.
1070    ///
1071    /// Default body: ignore the call. Poll-only backends (XRCE,
1072    /// bare-metal) leave the default in place; the executor still
1073    /// drains them on its deadline-bound cv-wait boundary.
1074    unsafe fn set_wake_callback(
1075        &mut self,
1076        cb: Option<unsafe extern "C" fn(ctx: *mut core::ffi::c_void)>,
1077        ctx: *mut core::ffi::c_void,
1078    ) {
1079        let _ = (cb, ctx);
1080    }
1081
1082    /// Phase 130.4 — does this backend actually honour
1083    /// [`set_wake_callback`]?
1084    ///
1085    /// `true` means the backend installs the callback and will
1086    /// fire it from its async notify path (worker thread, ISR,
1087    /// signalfd, …). `false` (the default) means
1088    /// `set_wake_callback` was a no-op — the executor must drive
1089    /// I/O for the caller's full timeout because no async wake
1090    /// will pre-empt it.
1091    ///
1092    /// The executor uses this to choose between the wake-primitive
1093    /// wait (`NodeWake::wait_ms` / `std::Condvar::wait_timeout_while`)
1094    /// and a direct `drive_io(timeout_ms)`. Poll-only backends
1095    /// (XRCE-DDS-Client, bare-metal smoltcp) return `false`;
1096    /// event-driven backends (zenoh-pico with an RX task that
1097    /// invokes the callback on packet arrival) return `true`.
1098    ///
1099    /// [`set_wake_callback`]: Self::set_wake_callback
1100    fn supports_wake_callback(&self) -> bool {
1101        false
1102    }
1103
1104    /// Phase 124.F.1 — session-level connectivity probe.
1105    ///
1106    /// Sends a wire-level round-trip probe and waits up to
1107    /// `timeout_ms`. `Ok(())` on reply, `Err(TransportError::Timeout)`
1108    /// on no reply, `Err(TransportError::Unsupported)` when the
1109    /// backend can't probe (DDS without participant introspection).
1110    /// Lesson from micro-ROS's `rmw_uros_ping_agent`.
1111    ///
1112    /// Default body: `Err(Unsupported)`. Backends with a native
1113    /// ping API (zenoh: `z_send_ping`; XRCE:
1114    /// `uxr_ping_agent_session_until_timeout`) opt in by overriding.
1115    fn ping_session(&mut self, timeout_ms: i32) -> Result<(), Self::Error>
1116    where
1117        Self::Error: From<TransportError>,
1118    {
1119        let _ = timeout_ms;
1120        Err(TransportError::Unsupported.into())
1121    }

1124/// Bitmask of QoS policies a backend can honour. See
1125/// [`Session::supported_qos_policies`].
1126///
1127/// `CORE` covers the policies every nano-ros backend implements:
1128/// reliability, durability=VOLATILE, history, depth. Backends opt
1129/// into additional policies by OR-ing the relevant flags.
1130#[derive(Debug, Clone, Copy, PartialEq, Eq)]
1131pub struct QosPolicyMask(pub u32);
1132
1133impl QosPolicyMask {
1134    pub const RELIABILITY: Self = Self(1 << 0);
1135    pub const DURABILITY_VOLATILE: Self = Self(1 << 1);
1136    pub const DURABILITY_TRANSIENT_LOCAL: Self = Self(1 << 2);
1137    pub const HISTORY: Self = Self(1 << 3);
1138    pub const DEPTH: Self = Self(1 << 4);
1139    pub const DEADLINE: Self = Self(1 << 5);
1140    pub const LIFESPAN: Self = Self(1 << 6);
1141    pub const LIVELINESS_AUTOMATIC: Self = Self(1 << 7);
1142    pub const LIVELINESS_MANUAL_BY_TOPIC: Self = Self(1 << 8);
1143    pub const LIVELINESS_MANUAL_BY_NODE: Self = Self(1 << 9);
1144    pub const LIVELINESS_LEASE: Self = Self(1 << 10);
1145    pub const AVOID_ROS_NAMESPACE_CONVENTIONS: Self = Self(1 << 11);
1146
1147    /// Policies every nano-ros backend implements.
1148    pub const CORE: Self =
1149        Self(Self::RELIABILITY.0 | Self::DURABILITY_VOLATILE.0 | Self::HISTORY.0 | Self::DEPTH.0);
1150
1151    /// `true` if `self` contains every policy in `other`.
1152    pub const fn contains(self, other: Self) -> bool {
1153        self.0 & other.0 == other.0
1154    }
1155
1156    /// Bitwise OR of two masks.
1157    pub const fn union(self, other: Self) -> Self {
1158        Self(self.0 | other.0)
1159    }
1160}

1162impl core::ops::BitOr for QosPolicyMask {
1163    type Output = Self;
1164    fn bitor(self, rhs: Self) -> Self {
1165        self.union(rhs)
1166    }
1167}

1169impl core::ops::BitOrAssign for QosPolicyMask {
1170    fn bitor_assign(&mut self, rhs: Self) {
1171        self.0 |= rhs.0;
1172    }

1175impl QosSettings {
1176    /// Compute the set of QoS policies actually requested by this profile.
1177    ///
1178    /// Zero-valued time fields and `LivelinessKind::None` count as "not
1179    /// requesting" the corresponding policy — the cheap default. The
1180    /// `CORE` bits (reliability, durability=VOLATILE, history, depth)
1181    /// are always present because every nano-ros backend honours them.
1182    pub fn required_policies(&self) -> QosPolicyMask {
1183        let mut mask = QosPolicyMask::CORE;
1184        if self.durability == QosDurabilityPolicy::TransientLocal {
1185            mask = QosPolicyMask(
1186                (mask.0 & !QosPolicyMask::DURABILITY_VOLATILE.0)
1187                    | QosPolicyMask::DURABILITY_TRANSIENT_LOCAL.0,
1188            );
1189        }
1190        if self.deadline_ms != 0 {
1191            mask |= QosPolicyMask::DEADLINE;
1192        }
1193        if self.lifespan_ms != 0 {
1194            mask |= QosPolicyMask::LIFESPAN;
1195        }
1196        match self.liveliness_kind {
1197            QosLivelinessPolicy::None => {}
1198            QosLivelinessPolicy::Automatic => mask |= QosPolicyMask::LIVELINESS_AUTOMATIC,
1199            QosLivelinessPolicy::ManualByTopic => mask |= QosPolicyMask::LIVELINESS_MANUAL_BY_TOPIC,
1200            QosLivelinessPolicy::ManualByNode => mask |= QosPolicyMask::LIVELINESS_MANUAL_BY_NODE,
1201        }
1202        if self.liveliness_lease_ms != 0 {
1203            mask |= QosPolicyMask::LIVELINESS_LEASE;
1204        }
1205        if self.avoid_ros_namespace_conventions {
1206            mask |= QosPolicyMask::AVOID_ROS_NAMESPACE_CONVENTIONS;
1207        }
1208        mask
1209    }
1210
1211    /// Returns `Err(TransportError::IncompatibleQos)` if any policy this
1212    /// profile requires is missing from the backend's `supported` mask.
1213    /// Used at entity-create time to enforce the **no silent
1214    /// degradation** contract.
1215    pub fn validate_against(&self, supported: QosPolicyMask) -> Result<(), TransportError> {
1216        if supported.contains(self.required_policies()) {
1217            Ok(())
1218        } else {
1219            Err(TransportError::IncompatibleQos)
1220        }
1221    }
1222}

1224/// Publisher trait for sending messages.
1225///
1226/// # Threading
1227///
1228/// `&self` on `publish_raw` — implementors must allow concurrent
1229/// publishes from multiple threads. Internal locking (or lock-free
1230/// queues) is the backend's responsibility.
1231///
1232/// # Buffer ownership
1233///
1234/// `data` in `publish_raw` is borrowed for the duration of the call.
1235/// The backend must either send it inline or copy into its own
1236/// buffer before returning — the slice is invalid after the call.
1237///
1238/// # Blocking
1239///
1240/// `publish_raw` is expected to be non-blocking on best-effort QoS
1241/// and bounded-blocking on reliable QoS (waiting for outbound queue
1242/// space). Backends should *not* block waiting for ack from a
1243/// matched subscriber.
1244pub trait Publisher {
1245    /// Error type for publish operations
1246    type Error;
1247
1248    /// Publish a CDR-serialised message.
1249    ///
1250    /// Returns once the message has been handed to the transport
1251    /// (queued or fired-and-forgotten depending on QoS). Does **not**
1252    /// wait for delivery.
1253    fn publish_raw(&self, data: &[u8]) -> Result<(), Self::Error>;
1254
1255    /// Phase 128.F.4 — publish with an opaque attachment block.
1256    ///
1257    /// `attachment` rides alongside the payload at the wire layer.
1258    /// Receivers can read it back via
1259    /// [`Subscriber::try_recv_raw_with_attachment`].
1260    ///
1261    /// Primary use case: cross-RMW bridges stamp a `bridge_origin`
1262    /// tag (the source backend's RMW name) so a paired return
1263    /// bridge can deterministically drop echoed frames.
1264    ///
1265    /// Default body delegates to [`publish_raw`](Self::publish_raw)
1266    /// and discards the attachment — backends that do not natively
1267    /// carry attachments (XRCE today, DDS without a user-data hook)
1268    /// see no change. Backends with native attachment support
1269    /// (zenoh-pico's `z_publisher_put_options::attachment`,
1270    /// Cyclone DDS user-data) override to write the bytes onto the
1271    /// wire.
1272    fn publish_raw_with_attachment(
1273        &self,
1274        data: &[u8],
1275        _attachment: &[u8],
1276    ) -> Result<(), Self::Error> {
1277        self.publish_raw(data)
1278    }
1279
1280    /// Phase 124.E.1 — streamed publish.
1281    ///
1282    /// `size_cb` reports the total payload length once; `chunk_cb`
1283    /// fills the slot in chunks. Saves the per-publisher staging
1284    /// buffer when the message is large enough to dominate the
1285    /// device's `.bss`.
1286    ///
1287    /// Default body — the **staging-buffer fallback** (124.E.2).
1288    /// Asks `size_cb` for the total length, fills a stack-allocated
1289    /// `[u8; NROS_MAX_STREAM_CHUNK]` via `chunk_cb`, then forwards
1290    /// to `publish_raw`. Returns `Err(BufferTooSmall)` if the total
1291    /// exceeds the stack cap so the caller can drop back to a
1292    /// regular `publish_raw` with a heap-sized buffer.
1293    ///
1294    /// Concrete backends opt in by overriding to stream straight
1295    /// into the network buffer (zenoh: write into the zenoh-pico
1296    /// outbound buffer; XRCE: micro-CDR streaming APIs).
1297    ///
1298    /// # Safety
1299    ///
1300    /// The caller must ensure `user_ctx` is valid for every invocation of
1301    /// `size_cb` and `chunk_cb` during this call, and that both callbacks obey
1302    /// their out-pointer contracts.
1303    ///
1304    /// `size_cb` and `chunk_cb` may be called from the same thread
1305    /// that called `publish_streamed`. Backends MUST NOT defer the
1306    /// calls past the function return; the caller's `user_ctx`
1307    /// pointer is only guaranteed valid for the duration of the
1308    /// call.
1309    unsafe fn publish_streamed(
1310        &self,
1311        size_cb: unsafe extern "C" fn(out_total_len: *mut usize, user_ctx: *mut core::ffi::c_void),
1312        chunk_cb: unsafe extern "C" fn(
1313            out_buf: *mut u8,
1314            cap: usize,
1315            out_written: *mut usize,
1316            user_ctx: *mut core::ffi::c_void,
1317        ),
1318        user_ctx: *mut core::ffi::c_void,
1319    ) -> Result<(), Self::Error>
1320    where
1321        Self::Error: From<TransportError>,
1322    {
1323        /// Default staging-buffer cap. Stack-allocated, so embedded
1324        /// callers don't pay for it unless they actually invoke this
1325        /// fallback. 4 KiB matches typical RTPS frag-size +
1326        /// micro-XRCE message ceilings.
1327        const STAGE_CAP: usize = 4096;
1328
1329        let mut total = 0usize;
1330        // SAFETY: caller's contract on `size_cb` matches our trait
1331        // doc — fire once with a writable `*mut usize` slot.
1332        unsafe { size_cb(&mut total as *mut usize, user_ctx) };
1333        if total > STAGE_CAP {
1334            return Err(TransportError::BufferTooSmall.into());
1335        }
1336        let mut stage = [0u8; STAGE_CAP];
1337        let mut written_so_far = 0usize;
1338        while written_so_far < total {
1339            let mut chunk_written = 0usize;
1340            let remaining = total - written_so_far;
1341            // SAFETY: `chunk_cb` writes ≤ `cap` bytes to
1342            // `out_buf` and reports the count via `out_written`.
1343            unsafe {
1344                chunk_cb(
1345                    stage.as_mut_ptr().add(written_so_far),
1346                    remaining,
1347                    &mut chunk_written as *mut usize,
1348                    user_ctx,
1349                );
1350            }
1351            if chunk_written == 0 {
1352                // Caller signalled EOF early — treat the partial
1353                // write as a malformed sequence; reporting it as
1354                // BufferTooSmall keeps the surface tight without
1355                // adding a new variant.
1356                return Err(TransportError::BufferTooSmall.into());
1357            }
1358            written_so_far += chunk_written;
1359        }
1360        self.publish_raw(&stage[..total])
1361    }
1362
1363    /// Publish a typed message (serializes automatically)
1364    fn publish<M: RosMessage>(&self, msg: &M, buf: &mut [u8]) -> Result<(), Self::Error> {
1365        use nros_core::CdrWriter;
1366
1367        let mut writer = CdrWriter::new_with_header(buf).map_err(|_| self.buffer_error())?;
1368        msg.serialize(&mut writer)
1369            .map_err(|_| self.serialization_error())?;
1370        let len = writer.position();
1371        self.publish_raw(&buf[..len])
1372    }
1373
1374    /// Return a buffer-too-small error (implementation specific)
1375    fn buffer_error(&self) -> Self::Error;
1376
1377    /// Return a serialization error (implementation specific)
1378    fn serialization_error(&self) -> Self::Error;
1379
1380    /// Phase 108 — `true` if the backend can generate this event for
1381    /// this publisher. Default returns `false`; backends override per
1382    /// supported event kind.
1383    ///
1384    /// Only [`EventKind::LivelinessLost`](crate::event::EventKind::LivelinessLost) and
1385    /// [`EventKind::OfferedDeadlineMissed`](crate::event::EventKind::OfferedDeadlineMissed) are publisher-side events;
1386    /// other kinds always return `false` here.
1387    fn supports_event(&self, _kind: crate::event::EventKind) -> bool {
1388        false
1389    }
1390
1391    /// Phase 108 — register a callback fired when the named status
1392    /// event occurs. `deadline_ms` applies to
1393    /// [`EventKind::OfferedDeadlineMissed`](crate::event::EventKind::OfferedDeadlineMissed) only; ignored otherwise.
1394    /// Default impl returns the backend's "unsupported"-shaped error.
1395    ///
1396    /// # Safety
1397    ///
1398    /// `cb` and `user_ctx` must remain valid for the entity's
1399    /// lifetime. Caller (typically `nros-node`'s typed wrapper) is
1400    /// responsible for keeping the closure / context arena alive.
1401    unsafe fn register_event_callback(
1402        &mut self,
1403        _kind: crate::event::EventKind,
1404        _deadline_ms: u32,
1405        _cb: crate::event::EventCallback,
1406        _user_ctx: *mut core::ffi::c_void,
1407    ) -> Result<(), Self::Error> {
1408        Err(self.unsupported_event_error())
1409    }
1410
1411    /// Phase 108 — backend's error variant for "this event kind is
1412    /// not supported." Default impl reuses `serialization_error()`
1413    /// since most backends share an `Unsupported` variant; backends
1414    /// override if they have a distinct `Unsupported` mapping.
1415    fn unsupported_event_error(&self) -> Self::Error {
1416        self.serialization_error()
1417    }
1418
1419    /// Phase 109 — assert this publisher's liveliness manually.
1420    /// Required for publishers configured with
1421    /// `QosLivelinessPolicy::ManualByTopic`. No-op for other
1422    /// liveliness kinds. Default impl returns `Ok(())` (no-op);
1423    /// backends override when they implement manual liveliness.
1424    fn assert_liveliness(&self) -> Result<(), Self::Error> {
1425        Ok(())
1426    }
1427}

1429/// Subscriber trait for receiving messages.
1430///
1431/// # Threading
1432///
1433/// `&mut self` on `try_recv_raw` — the executor takes exclusive
1434/// ownership of the subscriber for the duration of a receive. A
1435/// backend that wants to allow concurrent receives must split into
1436/// per-thread sub-handles internally.
1437///
1438/// # Buffer ownership
1439///
1440/// `buf` is caller-owned. The implementation copies the next ready
1441/// message into `buf` and returns the byte count. The caller may
1442/// re-use or drop `buf` immediately after the call.
1443///
1444/// # Blocking
1445///
1446/// `try_recv_raw` is **non-blocking**: returns `Ok(None)` (or
1447/// equivalent for backends that map empty into a zero-length read)
1448/// when no message is ready. Use [`Session::drive_io`] to wait for
1449/// data; never sleep inside `try_recv_raw`.
1450pub trait Subscriber {
1451    /// Error type for receive operations
1452    type Error;
1453
1454    /// Check if data is available without consuming it.
1455    ///
1456    /// Non-destructive — does not advance the receive cursor.
1457    /// Conservative default returns `true` (always assume data may
1458    /// be available); backends should override with a real check
1459    /// to avoid spurious receive attempts.
1460    fn has_data(&self) -> bool {
1461        true
1462    }
1463
1464    /// Try to receive one message into `buf`.
1465    ///
1466    /// Non-blocking. On success returns `Ok(Some(len))` where `len`
1467    /// is the byte count written into `buf[..len]`. Returns
1468    /// `Ok(None)` if no message is ready. If `buf` is too small the
1469    /// backend may either truncate (and document it) or return an
1470    /// error (preferred).
1471    fn try_recv_raw(&mut self, buf: &mut [u8]) -> Result<Option<usize>, Self::Error>;
1472
1473    /// Phase 128.F.4 — receive with attachment bytes alongside the
1474    /// payload.
1475    ///
1476    /// On success returns `Ok(Some((payload_len, attachment_len)))`
1477    /// with the payload written into `buf[..payload_len]` and the
1478    /// attachment (if any) written into
1479    /// `att_buf[..attachment_len]`. `attachment_len == 0` means the
1480    /// incoming sample carried no attachment.
1481    ///
1482    /// Default body falls back to [`try_recv_raw`](Self::try_recv_raw)
1483    /// and reports a 0-length attachment. Backends with native
1484    /// attachment support override to populate `att_buf`. Cross-RMW
1485    /// bridges use the attachment to read the `bridge_origin` tag
1486    /// stamped by the sending side.
1487    fn try_recv_raw_with_attachment(
1488        &mut self,
1489        buf: &mut [u8],
1490        _att_buf: &mut [u8],
1491    ) -> Result<Option<(usize, usize)>, Self::Error> {
1492        match self.try_recv_raw(buf)? {
1493            Some(len) => Ok(Some((len, 0))),
1494            None => Ok(None),
1495        }
1496    }
1497
1498    /// Phase 124.D.1 — burst-take.
1499    ///
1500    /// Drain up to `max_msgs` queued samples into the contiguous
1501    /// `buf` block in one call, with the i-th sample at
1502    /// `buf[i * per_msg_cap .. i * per_msg_cap + out_lens[i]]`.
1503    /// Returns the number of messages actually delivered. Partial
1504    /// drains MUST report the count, not error out.
1505    ///
1506    /// Default body loop-drives `try_recv_raw` so callers can
1507    /// commit to the batched API regardless of backend support.
1508    /// Concrete backends opt in by overriding with a native batch
1509    /// take (zenoh queue drain, `dds_take(max_samples)`).
1510    fn try_recv_sequence(
1511        &mut self,
1512        buf: &mut [u8],
1513        per_msg_cap: usize,
1514        max_msgs: usize,
1515        out_lens: &mut [usize],
1516    ) -> Result<usize, Self::Error> {
1517        if per_msg_cap == 0 || max_msgs == 0 {
1518            return Ok(0);
1519        }
1520        let limit = max_msgs.min(out_lens.len());
1521        let mut count = 0;
1522        for i in 0..limit {
1523            let slot = &mut buf[i * per_msg_cap..(i + 1) * per_msg_cap];
1524            match self.try_recv_raw(slot)? {
1525                Some(len) => {
1526                    out_lens[i] = len;
1527                    count += 1;
1528                }
1529                None => break,
1530            }
1531        }
1532        Ok(count)
1533    }
1534
1535    /// Try to receive a typed message (non-blocking)
1536    fn try_recv<M: RosMessage>(&mut self, buf: &mut [u8]) -> Result<Option<M>, Self::Error> {
1537        use nros_core::CdrReader;
1538
1539        match self.try_recv_raw(buf)? {
1540            Some(len) => {
1541                let mut reader = CdrReader::new_with_header(&buf[..len])
1542                    .map_err(|_| self.deserialization_error())?;
1543                let msg = M::deserialize(&mut reader).map_err(|_| self.deserialization_error())?;
1544                Ok(Some(msg))
1545            }
1546            None => Ok(None),
1547        }
1548    }
1549
1550    /// Process the received message in-place without copying.
1551    ///
1552    /// Calls `f` with a reference to the raw CDR bytes in the subscriber's
1553    /// internal receive buffer, avoiding a copy into a caller-provided buffer.
1554    /// While `f` executes the buffer is exclusively borrowed — any messages
1555    /// arriving from the transport during that time are dropped to prevent
1556    /// data races.
1557    ///
1558    /// Returns `Ok(true)` if a message was available and `f` was called,
1559    /// `Ok(false)` if no message was available.
1560    ///
1561    /// **Default body**: returns `Err(MessageTooLarge)` — the old default
1562    /// silently truncated anything larger than 1 KB into a stack buffer,
1563    /// which broke large messages with no diagnostic. Backends must
1564    /// override this with a real zero-copy path if they advertise support
1565    /// for `process_raw_in_place`; callers that hit the default should
1566    /// use `try_recv_raw` with a caller-sized buffer instead.
1567    fn process_raw_in_place(&mut self, f: impl FnOnce(&[u8])) -> Result<bool, Self::Error>
1568    where
1569        Self::Error: From<TransportError>,
1570    {
1571        let _ = f;
1572        Err(TransportError::MessageTooLarge.into())
1573    }
1574
1575    /// Whether this backend implements the in-place dispatch methods
1576    /// ([`process_raw_in_place`](Subscriber::process_raw_in_place) /
1577    /// [`process_raw_in_place_with_info`](Subscriber::process_raw_in_place_with_info))
1578    /// with a real zero-copy borrow.
1579    ///
1580    /// The executor consults this at subscription registration to choose the
1581    /// **in-place** arena dispatch (borrow + deserialize from the backend slot, no
1582    /// arena buffer) over the **buffered** dispatch (copy into an arena buffer
1583    /// first). Backends that leave the in-place methods at their unsupported
1584    /// default return `false` (the default) and keep the buffered path. (RFC-0038,
1585    /// Phase 231 Wave 0.2.)
1586    fn supports_process_in_place(&self) -> bool {
1587        false
1588    }
1589
1590    /// In-place processing variant that also surfaces publisher metadata.
1591    ///
1592    /// Same borrow contract as
1593    /// [`process_raw_in_place`](Subscriber::process_raw_in_place): `f` receives
1594    /// the raw CDR bytes plus the parsed [`MessageInfo`](nros_core::MessageInfo)
1595    /// — the co-located attachment (publisher GID / sequence / source timestamp),
1596    /// or `None` when no attachment was present — for the duration of the call;
1597    /// the slot is released after `f` returns. `Ok(true)` = a message was
1598    /// available and `f` was called; `Ok(false)` = none ready.
1599    ///
1600    /// **Default body**: returns the unsupported error (mirrors
1601    /// `process_raw_in_place`). Backends that advertise in-place support override
1602    /// this with a real zero-copy path; callers that hit the default should use
1603    /// the buffered [`try_recv_raw_with_info`](Subscriber::try_recv_raw_with_info)
1604    /// path instead. (RFC-0038, Phase 231 Wave 0.1.)
1605    fn process_raw_in_place_with_info(
1606        &mut self,
1607        f: impl FnOnce(&[u8], Option<nros_core::MessageInfo>),
1608    ) -> Result<bool, Self::Error>
1609    where
1610        Self::Error: From<TransportError>,
1611    {
1612        let _ = f;
1613        Err(TransportError::MessageTooLarge.into())
1614    }
1615
1616    /// Try to receive raw data along with publisher metadata.
1617    ///
1618    /// When available, [`MessageInfo`](nros_core::MessageInfo) contains
1619    /// the publisher's GID (Global Identifier) and source timestamp,
1620    /// extracted from a transport-level attachment on the incoming message.
1621    ///
1622    /// Returns `Ok(Some((len, info)))` if data is available, where:
1623    /// - `len` is the number of bytes written to the buffer
1624    /// - `info` is the parsed publisher metadata (if attachment was present)
1625    ///
1626    /// Default: delegates to [`try_recv_raw`](Subscriber::try_recv_raw) with no info.
1627    fn try_recv_raw_with_info(
1628        &mut self,
1629        buf: &mut [u8],
1630    ) -> Result<Option<(usize, Option<nros_core::MessageInfo>)>, Self::Error> {
1631        self.try_recv_raw(buf).map(|opt| opt.map(|len| (len, None)))
1632    }
1633
1634    /// Try to receive raw data with E2E safety validation (CRC + sequence tracking).
1635    ///
1636    /// Returns `Ok(Some((len, status)))` if data is available, where:
1637    /// - `len` is the number of bytes written to the buffer
1638    /// - `status` is the integrity validation result
1639    ///
1640    /// Default: delegates to `try_recv_raw` with no CRC info.
1641    #[cfg(feature = "safety-e2e")]
1642    fn try_recv_validated(
1643        &mut self,
1644        buf: &mut [u8],
1645    ) -> Result<Option<(usize, crate::IntegrityStatus)>, Self::Error> {
1646        self.try_recv_raw(buf).map(|opt| {
1647            opt.map(|len| {
1648                (
1649                    len,
1650                    crate::IntegrityStatus {
1651                        gap: 0,
1652                        duplicate: false,
1653                        crc_valid: None,
1654                    },
1655                )
1656            })
1657        })
1658    }
1659
1660    /// Register an async waker to be notified when data arrives.
1661    ///
1662    /// Called from `Future::poll()` implementations to store the waker.
1663    /// The transport backend calls `waker.wake()` from its receive callback
1664    /// when new data is available, enabling event-driven async without
1665    /// busy-polling.
1666    ///
1667    /// Default: no-op (backends that don't support waking simply ignore this).
1668    fn register_waker(&self, _waker: &core::task::Waker) {}
1669
1670    /// Return a deserialization error (implementation specific)
1671    fn deserialization_error(&self) -> Self::Error;
1672
1673    /// Phase 108 — `true` if the backend can generate this event for
1674    /// this subscriber. Default returns `false`; backends override per
1675    /// supported event kind.
1676    ///
1677    /// Subscriber-side event kinds:
1678    /// [`EventKind::LivelinessChanged`](crate::event::EventKind::LivelinessChanged),
1679    /// [`EventKind::RequestedDeadlineMissed`](crate::event::EventKind::RequestedDeadlineMissed),
1680    /// [`EventKind::MessageLost`](crate::event::EventKind::MessageLost).
1681    /// Publisher kinds always return `false` here.
1682    fn supports_event(&self, _kind: crate::event::EventKind) -> bool {
1683        false
1684    }
1685
1686    /// Phase 108 — register a callback fired when the named status
1687    /// event occurs. `deadline_ms` applies to
1688    /// [`EventKind::RequestedDeadlineMissed`](crate::event::EventKind::RequestedDeadlineMissed) only; ignored otherwise.
1689    /// Default impl returns the backend's "unsupported"-shaped error.
1690    ///
1691    /// # Safety
1692    ///
1693    /// `cb` and `user_ctx` must remain valid for the entity's
1694    /// lifetime. Caller (typically `nros-node`'s typed wrapper) is
1695    /// responsible for keeping the closure / context arena alive.
1696    unsafe fn register_event_callback(
1697        &mut self,
1698        _kind: crate::event::EventKind,
1699        _deadline_ms: u32,
1700        _cb: crate::event::EventCallback,
1701        _user_ctx: *mut core::ffi::c_void,
1702    ) -> Result<(), Self::Error> {
1703        Err(self.unsupported_event_error())
1704    }
1705
1706    /// Phase 108 — backend's error variant for "this event kind is
1707    /// not supported." Default reuses `deserialization_error()` for
1708    /// backends that don't have a distinct `Unsupported` mapping.
1709    fn unsupported_event_error(&self) -> Self::Error {
1710        self.deserialization_error()
1711    }
1712}

1723// Phase 99 — zero-copy raw API: SlotLending / SlotBorrowing
1724// ============================================================================
1725//
1726// Backends that can lend a slot directly into their outbound buffer
1727// (zenoh-pico w/ unstable-zenoh-api, XRCE-DDS via uxr_prepare_output_stream,
1728// full DDS w/ SHM transport) implement these traits. Backends that cannot
1729// (uORB, default zenoh-pico) do NOT impl them — `EmbeddedRawPublisher` then
1730// falls back to its per-publisher arena and memcpys at commit time. Both
1731// paths land at `Publisher::publish_raw` for the actual wire write; only
1732// the user-side copy is eliminated when lending is available.
1733//
1734// Selection is **compile-time** via the `lending` Cargo feature. Each
1735// backend crate forwards its own `lending` feature to `nros-rmw/lending`
1736// when it can satisfy the trait. nros-node's `rmw-lending` aggregates.
1737// User opting `nros/rmw-lending` w/ a non-lending backend (e.g. uORB)
1738// gets a clear compile error from the unsatisfied trait bound on the
1739// concrete `RmwPublisher`.
1740
1741/// Backend can lend a writable slot into its outbound buffer.
1742///
1743/// The returned slot's lifetime is tied to `&self`; user fills it in
1744/// place, then calls [`commit_slot`](Self::commit_slot) to publish.
1745/// Dropping the slot without commit is a no-op (slot returned to free
1746/// pool); concurrent loan attempts that would exceed backend capacity
1747/// return [`TransportError::WouldBlock`] — never block.
1748#[cfg(feature = "lending")]
1749pub trait SlotLending: Publisher {
1750    /// Backend-owned writable slot. Holds a `&'a mut [u8]` and any
1751    /// state needed for commit_slot.
1752    type Slot<'a>: AsMut<[u8]> + 'a
1753    where
1754        Self: 'a;
1755
1756    /// Reserve a writable slot of `len` bytes from the backend's
1757    /// outbound buffer. Returns `Ok(None)` if the backend has no slot
1758    /// available (full); never blocks.
1759    fn try_lend_slot(&self, len: usize) -> Result<Option<Self::Slot<'_>>, Self::Error>;
1760
1761    /// Commit a previously-lent slot. Consumes the slot and triggers
1762    /// the actual wire write. Returns `Err` on backend send failure;
1763    /// the slot's bytes are lost in that case (caller must re-lend +
1764    /// re-fill to retry).
1765    fn commit_slot(&self, slot: Self::Slot<'_>) -> Result<(), Self::Error>;
1766}

1768/// Backend can lend a read-only view into its receive buffer.
1769///
1770/// The returned view's lifetime is tied to `&mut self` (subscriber-
1771/// exclusive); dropping the view releases any backend lock and lets
1772/// the next message advance into the buffer.
1773#[cfg(feature = "lending")]
1774pub trait SlotBorrowing: Subscriber {
1775    /// Backend-owned read-only view. Holds a `&'a [u8]` and any state
1776    /// needed to release the borrow on Drop.
1777    type View<'a>: AsRef<[u8]> + 'a
1778    where
1779        Self: 'a;
1780
1781    /// Try to borrow the next available message in place. Returns
1782    /// `Ok(None)` if no message is ready; never blocks.
1783    fn try_borrow(&mut self) -> Result<Option<Self::View<'_>>, Self::Error>;
1784}

1786/// Service server trait for handling requests.
1787///
1788/// # Threading
1789///
1790/// `&mut self` on `try_recv_request` and `send_reply` — the executor
1791/// owns the server while a request is being handled. Handler bodies
1792/// run synchronously on the executor thread; long handlers should
1793/// dispatch work to a worker queue and reply later via the recorded
1794/// `sequence_number`.
1795///
1796/// # Calling pattern
1797///
1798/// 1. Executor calls `try_recv_request(buf)`.
1799/// 2. If `Some(req)` returned, decode, run handler, encode reply.
1800/// 3. `send_reply(req.sequence_number, &reply_buf)`.
1801///
1802/// `sequence_number` is the canonical request → reply correlation
1803/// token; backends derive it from the wire-level metadata (zenoh
1804/// query id, DDS sample identity).
1805pub trait ServiceServerTrait {
1806    /// Error type for service operations
1807    type Error;
1808
1809    /// Check if a request is available without consuming it.
1810    ///
1811    /// Non-destructive. Default returns `true` (always assume one
1812    /// may be available); backends should override with a real
1813    /// check.
1814    fn has_request(&self) -> bool {
1815        true
1816    }
1817
1818    /// Phase 122.3.c.6.e — register a `Waker` for event-driven
1819    /// service servers. Mirrors the matching method on
1820    /// `SubscriberTrait` / `ServiceClientTrait`. Backends that
1821    /// surface incoming-request notifications wake `waker` when
1822    /// `has_request()` flips true. Default: no-op (backends without
1823    /// wake support ignore — caller falls back to polling).
1824    fn register_waker(&self, _waker: &core::task::Waker) {}
1825
1826    /// Try to receive a service request into `buf` (non-blocking).
1827    ///
1828    /// On success returns a `ServiceRequest` that borrows from
1829    /// `buf`. The borrow is released when the returned struct is
1830    /// dropped — typically before `send_reply` is called, since
1831    /// `send_reply` takes `&mut self`.
1832    fn try_recv_request<'a>(
1833        &mut self,
1834        buf: &'a mut [u8],
1835    ) -> Result<Option<ServiceRequest<'a>>, Self::Error>;
1836
1837    /// Send a reply for the given sequence number. Non-blocking
1838    /// from the application's perspective; the backend may queue
1839    /// the reply for transport-level transmission.
1840    fn send_reply(&mut self, sequence_number: i64, data: &[u8]) -> Result<(), Self::Error>;
1841
1842    /// Handle a service request with typed messages
1843    fn handle_request<S: RosService>(
1844        &mut self,
1845        req_buf: &mut [u8],
1846        reply_buf: &mut [u8],
1847        handler: impl FnOnce(&S::Request) -> S::Reply,
1848    ) -> Result<bool, Self::Error>
1849    where
1850        Self::Error: From<TransportError>,
1851    {
1852        use nros_core::{CdrReader, CdrWriter};
1853
1854        // First, try to receive a request and extract necessary data.
1855        // Capture the data slice's offset within `req_buf` so we can
1856        // re-borrow it after the `ServiceRequest` (which holds a
1857        // borrow into `req_buf`) is dropped. Some backends prepend a
1858        // header (DDS: 8-byte sequence number) and place the CDR
1859        // payload at a non-zero offset in the buffer; others (zenoh)
1860        // put it at offset 0. Reading from offset 0 unconditionally
1861        // would feed the prefix bytes to the CDR deserializer and
1862        // silently corrupt the request.
1863        let buf_start = req_buf.as_ptr() as usize;
1864        let (data_offset, data_len, sequence_number) = match self.try_recv_request(req_buf)? {
1865            Some(request) => {
1866                let offset = (request.data.as_ptr() as usize).saturating_sub(buf_start);
1867                (offset, request.data.len(), request.sequence_number)
1868            }
1869            None => return Ok(false),
1870        };
1871
1872        // Deserialize request from the captured offset.
1873        let mut reader = CdrReader::new_with_header(&req_buf[data_offset..data_offset + data_len])
1874            .map_err(|_| TransportError::DeserializationError)?;
1875        let req = S::Request::deserialize(&mut reader)
1876            .map_err(|_| TransportError::DeserializationError)?;
1877
1878        // Call handler
1879        let reply = handler(&req);
1880
1881        // Serialize reply
1882        let mut writer =
1883            CdrWriter::new_with_header(reply_buf).map_err(|_| TransportError::BufferTooSmall)?;
1884        reply
1885            .serialize(&mut writer)
1886            .map_err(|_| TransportError::SerializationError)?;
1887        let len = writer.position();
1888
1889        // Send reply (now we can borrow self mutably again)
1890        self.send_reply(sequence_number, &reply_buf[..len])?;
1891        Ok(true)
1892    }
1893
1894    /// Handle a service request where the handler returns `Box<S::Reply>`
1895    ///
1896    /// Identical to `handle_request` but the handler returns a heap-allocated reply.
1897    /// This is needed for services with large response types (e.g., parameter services
1898    /// where `Vec<ParameterValue, 64>` is ~1MB+) that would overflow the stack.
1899    #[cfg(feature = "alloc")]
1900    fn handle_request_boxed<S: RosService>(
1901        &mut self,
1902        req_buf: &mut [u8],
1903        reply_buf: &mut [u8],
1904        handler: impl FnOnce(&S::Request) -> alloc::boxed::Box<S::Reply>,
1905    ) -> Result<bool, Self::Error>
1906    where
1907        Self::Error: From<TransportError>,
1908    {
1909        use nros_core::{CdrReader, CdrWriter};
1910
1911        let buf_start = req_buf.as_ptr() as usize;
1912        let (data_offset, data_len, sequence_number) = match self.try_recv_request(req_buf)? {
1913            Some(request) => {
1914                let offset = (request.data.as_ptr() as usize).saturating_sub(buf_start);
1915                (offset, request.data.len(), request.sequence_number)
1916            }
1917            None => return Ok(false),
1918        };
1919
1920        let mut reader = CdrReader::new_with_header(&req_buf[data_offset..data_offset + data_len])
1921            .map_err(|_| TransportError::DeserializationError)?;
1922        let req = S::Request::deserialize(&mut reader)
1923            .map_err(|_| TransportError::DeserializationError)?;
1924
1925        let reply = handler(&req);
1926
1927        let mut writer =
1928            CdrWriter::new_with_header(reply_buf).map_err(|_| TransportError::BufferTooSmall)?;
1929        reply
1930            .serialize(&mut writer)
1931            .map_err(|_| TransportError::SerializationError)?;
1932        let len = writer.position();
1933
1934        self.send_reply(sequence_number, &reply_buf[..len])?;
1935        Ok(true)
1936    }
1937}

1939/// Service client trait for sending requests.
1940///
1941/// # Threading
1942///
1943/// `&mut self` on every method — the client is single-owner. For
1944/// fan-out request patterns, create one client per worker thread.
1945///
1946/// # Calling pattern
1947///
1948/// All in-tree backends route blocking waits through the executor:
1949///
1950/// 1. `send_request_raw(buf)` — non-blocking; returns once the
1951///    request is queued for transmission.
1952/// 2. The executor's `drive_io` runs.
1953/// 3. `try_recv_reply_raw(buf)` — non-blocking; returns
1954///    `Ok(Some(len))` when the reply is back.
1955///
1956/// The deprecated [`call_raw`](Self::call_raw) blocking path is
1957/// kept for backwards compatibility but should not be called.
1958pub trait ServiceClientTrait {
1959    /// Error type for service operations
1960    type Error;
1961
1962    /// Send a service request and wait for reply (blocking).
1963    ///
1964    /// **Deprecated — do not call.** The default body returns `Timeout`
1965    /// immediately without polling. Use `Client::call` →
1966    /// `Promise::wait(executor, timeout_ms)` which lets the executor
1967    /// drive I/O while waiting instead of busy-looping on
1968    /// `try_recv_reply_raw` with no sleep (which starves the transport
1969    /// on FreeRTOS / Zephyr single-threaded schedulers).
1970    ///
1971    /// Backends that still need an internal blocking path should
1972    /// override this with a real sleep-between-polls implementation,
1973    /// but all in-tree backends (zenoh, XRCE) route blocking waits
1974    /// through the executor.
1975    #[deprecated(note = "use Client::call → Promise::wait with an executor instead")]
1976    fn call_raw(&mut self, request: &[u8], _reply_buf: &mut [u8]) -> Result<usize, Self::Error>
1977    where
1978        Self::Error: From<TransportError>,
1979    {
1980        let _ = request;
1981        Err(TransportError::Timeout.into())
1982    }
1983
1984    /// Send a service request without waiting for a reply (non-blocking).
1985    ///
1986    /// The caller must subsequently poll [`try_recv_reply_raw`](Self::try_recv_reply_raw)
1987    /// to retrieve the reply.
1988    fn send_request_raw(&mut self, request: &[u8]) -> Result<(), Self::Error>;
1989
1990    /// Poll for a reply to the most recently sent request (non-blocking).
1991    ///
1992    /// Returns `Ok(Some(len))` when a reply has arrived, `Ok(None)` if not yet
1993    /// available, or `Err` on failure.
1994    fn try_recv_reply_raw(&mut self, reply_buf: &mut [u8]) -> Result<Option<usize>, Self::Error>;
1995
1996    /// Send a typed service request without waiting for a reply (non-blocking).
1997    ///
1998    /// Serializes the request into `req_buf` and calls [`send_request_raw`](Self::send_request_raw).
1999    fn send_request<S: RosService>(
2000        &mut self,
2001        request: &S::Request,
2002        req_buf: &mut [u8],
2003    ) -> Result<(), Self::Error>
2004    where
2005        Self::Error: From<TransportError>,
2006    {
2007        use nros_core::CdrWriter;
2008
2009        let mut writer =
2010            CdrWriter::new_with_header(req_buf).map_err(|_| TransportError::BufferTooSmall)?;
2011        request
2012            .serialize(&mut writer)
2013            .map_err(|_| TransportError::SerializationError)?;
2014        let req_len = writer.position();
2015
2016        self.send_request_raw(&req_buf[..req_len])
2017    }
2018
2019    /// Poll for a typed reply to the most recently sent request (non-blocking).
2020    ///
2021    /// Calls [`try_recv_reply_raw`](Self::try_recv_reply_raw) and deserializes if available.
2022    fn try_recv_reply<S: RosService>(
2023        &mut self,
2024        reply_buf: &mut [u8],
2025    ) -> Result<Option<S::Reply>, Self::Error>
2026    where
2027        Self::Error: From<TransportError>,
2028    {
2029        use nros_core::CdrReader;
2030
2031        match self.try_recv_reply_raw(reply_buf)? {
2032            Some(len) => {
2033                let mut reader = CdrReader::new_with_header(&reply_buf[..len])
2034                    .map_err(|_| TransportError::DeserializationError)?;
2035                let reply = S::Reply::deserialize(&mut reader)
2036                    .map_err(|_| TransportError::DeserializationError)?;
2037                Ok(Some(reply))
2038            }
2039            None => Ok(None),
2040        }
2041    }
2042
2043    /// Register an async waker to be notified when a reply arrives.
2044    ///
2045    /// Called from `Future::poll()` implementations to store the waker.
2046    /// The transport backend calls `waker.wake()` from its reply callback
2047    /// when a response is available, enabling event-driven async without
2048    /// busy-polling.
2049    ///
2050    /// Default: no-op (backends that don't support waking simply ignore this).
2051    fn register_waker(&self, _waker: &core::task::Waker) {}
2052
2053    /// Begin a server-discovery query on this client (non-blocking).
2054    ///
2055    /// Models `rclcpp::ClientBase::wait_for_service` machinery: the backend
2056    /// fires off a discovery probe (typically a Zenoh liveliness query
2057    /// against the matching server's wildcarded liveliness keyexpr) and
2058    /// the caller polls [`poll_server_discovery`](Self::poll_server_discovery)
2059    /// to collect the result.
2060    ///
2061    /// Default impl: no-op success. Backends without a discovery channel
2062    /// (or those that always assume the server is reachable) can leave
2063    /// this default and have `poll_server_discovery` return
2064    /// `Ok(Some(true))` immediately.
2065    fn start_server_discovery(&mut self, _timeout_ms: u32) -> Result<(), Self::Error> {
2066        Ok(())
2067    }
2068
2069    /// Poll an in-flight server-discovery query.
2070    ///
2071    /// - `Ok(Some(true))` — at least one matching server has reported
2072    ///   back; safe to send the first request.
2073    /// - `Ok(Some(false))` — discovery query finished without finding
2074    ///   any matching server (timeout / no-replies).
2075    /// - `Ok(None)` — query still in flight.
2076    /// - `Err(_)` — transport-level failure unrelated to server presence.
2077    ///
2078    /// Default impl: returns `Ok(Some(true))` (i.e., "server is always
2079    /// assumed reachable"). The Zenoh backend overrides this with a
2080    /// liveliness-token check.
2081    fn poll_server_discovery(&mut self) -> Result<Option<bool>, Self::Error> {
2082        Ok(Some(true))
2083    }
2084
2085    /// Synchronous, non-blocking check of whether a matching server is
2086    /// currently visible.
2087    ///
2088    /// Mirrors `rclcpp::ClientBase::service_is_ready`. Backends that lack
2089    /// discovery should keep the default `true` so existing call sites
2090    /// don't regress.
2091    ///
2092    /// Default impl: always `true`.
2093    fn is_server_ready(&self) -> bool {
2094        true
2095    }
2096
2097    /// Phase 124.C.1 — graph-aware server-availability probe.
2098    ///
2099    /// Returns `Ok(true)` if at least one matching server has been
2100    /// discovered, `Ok(false)` if none yet, or `Err(_)` if the
2101    /// backend cannot answer (e.g. XRCE — micro-XRCE-DDS-Client has
2102    /// no participant enumeration). Distinct from
2103    /// [`is_server_ready`](Self::is_server_ready), which collapses
2104    /// "don't know" and "no server" into the same `false` answer.
2105    ///
2106    /// User-facing surface: `Client<S>::server_available()` in Rust,
2107    /// `nros_client_server_available()` in C/C++. Clients use this
2108    /// to gate the first `call_raw` so a startup-ordering race
2109    /// (client opens before server's discovery announcement lands)
2110    /// doesn't surface as a request-side timeout.
2111    ///
2112    /// Default impl: `Err(TransportError::Unsupported)` — backends
2113    /// that support graph introspection (zenoh queryable interest,
2114    /// DDS built-in topic readers) opt in by overriding.
2115    fn server_available(&self) -> Result<bool, Self::Error>
2116    where
2117        Self::Error: From<TransportError>,
2118    {
2119        Err(TransportError::Unsupported.into())
2120    }
2121
2122    /// Call a service with typed messages (blocking).
2123    ///
2124    /// **Deprecated — do not call.** The default body returns `Timeout`
2125    /// immediately without polling. Use `Client::call` on the executor
2126    /// instead, which drives I/O while waiting. See
2127    /// [`call_raw`](Self::call_raw) for the same reasoning.
2128    #[deprecated(note = "use Client::call → Promise::wait with an executor instead")]
2129    fn call<S: RosService>(
2130        &mut self,
2131        request: &S::Request,
2132        req_buf: &mut [u8],
2133        _reply_buf: &mut [u8],
2134    ) -> Result<S::Reply, Self::Error>
2135    where
2136        Self::Error: From<TransportError>,
2137    {
2138        use nros_core::CdrWriter;
2139
2140        // Serialize request so the error surface matches the old impl
2141        // for the "bad request" path; but skip the receive busy-loop.
2142        let mut writer =
2143            CdrWriter::new_with_header(req_buf).map_err(|_| TransportError::BufferTooSmall)?;
2144        request
2145            .serialize(&mut writer)
2146            .map_err(|_| TransportError::SerializationError)?;
2147        Err(TransportError::Timeout.into())
2148    }
2149}
