1//! # nros
2//!
3//! A lightweight ROS 2 client library for embedded systems.
4//!
5//! This crate provides a unified API for building ROS 2 nodes in Rust,
6//! with support for `no_std` environments and embedded targets.
7//!
8//! ## Features
9//!
10//! - **no_std compatible**: Works on bare-metal and RTOS targets
11//! - **Zero-copy where possible**: Minimizes memory allocations
12//! - **Type-safe**: Compile-time verification of message types
13//! - **ROS 2 compatible**: Interoperates with standard ROS 2 nodes via rmw_zenoh
14//!
15//! ## Quick Start
16//!
17//! ```ignore
18//! use nros::prelude::*;
19//! use std_msgs::msg::Int32;
20//!
21//! let config = ExecutorConfig::from_env().node_name("my_node");
22//! let mut executor = Executor::open(&config)?;
23//!
24//! let node = executor.node_builder("my_node").build()?;
25//! let publisher = executor.node_mut(node).create_publisher::<Int32>("/my_topic")?;
26//! publisher.publish(&Int32 { data: 42 })?;
27//!
28//! executor.node_mut(node).create_subscription::<Int32, _>("/topic", |msg: &Int32| {
29//!     println!("Received: {}", msg.data);
30//! })?;
31//!
32//! executor.spin_blocking(SpinOptions::default());
33//! ```
34//!
35//! ## Executor Sizing
36//!
37//! The executor's static memory layout is controlled via environment variables
38//! at build time:
39//!
40//! - **`NROS_EXECUTOR_MAX_CBS`** (default 4) — maximum number of registered
41//!   callbacks (subscriptions + timers + services + guard conditions).
42//! - **`NROS_EXECUTOR_ARENA_SIZE`** (default 4096) — byte budget for storing
43//!   callback closures inline.
44//!
45//! For messages larger than the default 1024-byte receive buffer, size the
46//! subscription via the builder's `.rx_buffer::<N>()` knob (e.g.
47//! `node_mut(id).subscription(t).typed::<M>().rx_buffer::<4096>().build(cb)`).
48//!
49//! ## Transport Backends
50//!
51//! Phase 248 C5c — `nros` is RMW- and platform-AGNOSTIC. It carries only the
52//! `rmw-cffi` vtable; the concrete backend (zenoh / xrce / cyclonedds) enters
53//! the link graph via the board crate (embedded), the board-less app's own
54//! `nros-rmw-*` dep (native), or the `nros-c`/`nros-cpp` staticlib root (D3),
55//! and self-registers through the `RMW_INIT_ENTRIES` walker at `Executor::open`.
56//! The concrete session type is resolved automatically; advanced users can
57//! access it via `nros::internals::RmwSession`.
58//!
59//! ## Crate Features
60//!
61//! `nros` exposes only FUNCTIONAL features — `std`/`alloc`, the `rmw-cffi`
62//! vtable, `lending`, `bridge`/`config`, `param-services`,
63//! `lifecycle-services`, `safety-e2e`, `stream`, `ffi-sync`, and the ROS
64//! edition (`ros-humble`/`ros-iron`). There are NO `platform-*` or concrete
65//! `rmw-*` selector features (Phase 248 C7). Platform + RMW are selected by the
66//! board / staticlib root via dependencies, not `nros` features. The
67//! `zephyr_component_main!` entry macro is gated only on `rmw-cffi` (it's
68//! framework entry codegen, like `nros::main!`), not a platform feature.
69//!
70//! **ROS version** (select one):
71//! - `ros-humble` - ROS 2 Humble
72//! - `ros-iron` - ROS 2 Iron
73//!
74//! **Other**:
75//! - `std` (default) - Enable standard library support
76//! - `alloc` - Enable heap allocation without full std
77//!
78//! ## Further Reading
79//!
80//! - [`guide`] — tutorials: getting started, services, configuration,
81//!   ROS 2 interop, and troubleshooting
82//! - [Message Generation](https://github.com/jerry73204/nano-ros/blob/main/docs/guides/message-generation.md)
83//!   — codegen reference (all options, output structure, bundled interfaces)
84//! - [Environment Variables](https://github.com/jerry73204/nano-ros/blob/main/docs/reference/environment-variables.md)

87//!   — protocol details (key expressions, liveliness, attachments)

89//!   — working examples by platform (native, QEMU, ESP32, Zephyr)

283/// Define Zephyr's `rust_main` for a self-bringup Rust component package.
284///
285/// The macro is intended for `rust_cargo_application()` apps whose crate
286/// already invokes `nros::node!()`. It opens a Zephyr executor, registers
287/// the supplied component through [`ExecutorNodeRuntime`], and spins forever.
288// Phase 248 C7 (Method A) — gated on `rmw-cffi` only (needs `Executor`), NOT a
289// `platform-*` feature. This is a framework ENTRY macro (same category as
290// `nros::main!`'s zephyr `rust_main` codegen) — `#[macro_export]` so it emits
291// nothing unless a Zephyr example invokes it; the body's `::zephyr::*` /
292// `::nros_platform::zephyr::wait_network` resolve only in that zephyr-build
293// context (the example deps the `zephyr` crate + `nros-platform[platform-zephyr]`).
294#[cfg(feature = "rmw-cffi")]
295#[macro_export]
296macro_rules! zephyr_component_main {
297    ($node:ty) => {
298        #[unsafe(no_mangle)]
299        pub extern "C" fn rust_main() {
300            unsafe {
301                zephyr::set_logger().ok();
302            }
303            // Phase 248 C7 step 1 — relocated helper (was `$crate::platform::zephyr`).
304            let _ = ::nros_platform::zephyr::wait_network(2000);
305            // Phase 249 P1 — RMW register is board/platform-owned (Phase 248 C5a);
306            // the backend-agnostic `nros` crate cannot register (no backend dep), so
307            // the former no-op `$crate::__register_linked_rmw()` emit is removed. The
308            // Zephyr entry's explicit backend register lands via the board/platform
309            // boot path (verify under the phase-249 Zephyr e2e gate).
310            // Locator: `default_const()` = EMPTY locator → zenoh-pico
311            // multicast scouting, which native_sim NSOS can't satisfy.
312            // Bake `NROS_LOCATOR` at compile time (the example `build.rs`
313            // re-exports `CONFIG_NROS_ZENOH_LOCATOR` from Kconfig into that
314            // env). No baked value → falls back to the empty locator.
315            const BAKED_LOCATOR: ::core::option::Option<&str> = ::core::option_env!("NROS_LOCATOR");
316            let config = match BAKED_LOCATOR {
317                ::core::option::Option::Some(loc) if !loc.is_empty() => {
318                    $crate::ExecutorConfig::new(loc).node_name(<$node as $crate::Node>::NAME)
319                }
320                _ => {
321                    $crate::ExecutorConfig::default_const().node_name(<$node as $crate::Node>::NAME)
322                }
323            };
324            let executor = match $crate::Executor::open(&config) {
325                Ok(executor) => executor,
326                Err(_) => return,
327            };
328            let mut runtime = $crate::ExecutorNodeRuntime::from_executor(executor);
329            if runtime.register_node::<$node>().is_err() {
330                return;
331            }
332            // Readiness marker. The C/C++ Zephyr listeners print
333            // "Waiting for messages..." from their `main()` before the spin
334            // loop; the e2e harness polls for that substring to know the
335            // subscriber has declared before starting the talker (Phase 89.12).
336            // The Rust path's spin loop lives in this macro (the node only owns
337            // callbacks), so emit the same canonical marker here — without it a
338            // fully-working Rust listener never signals readiness and the e2e
339            // times out at 30 s (issue #35: the zenoh native_sim rust pubsub /
340            // service / action failures were this missing marker, not a
341            // transport fault — `Executor::open` + `register_node` had already
342            // succeeded).
343            ::log::info!("Waiting for messages");
344            loop {
345                let _ = runtime.spin_once(::core::time::Duration::from_millis(10));
346            }
347        }
348    };
349}

458    /// Open a new middleware session.
459    ///
460    /// Wraps the backend-specific session constructor behind a common signature.
461    /// Used by the C API (`nros-c`); Rust users should prefer `Executor::open()`.
462    ///
463    /// Phase 156 — consults `$NROS_RMW` (when std + the env var is set)
464    /// to pin the primary backend by name, mirroring what `Executor::open`
465    /// does for Rust callers. Without this, C bridges built with two
466    /// linked backends (e.g. xrce + dds) get whichever ctor fires
467    /// first via linkme — non-deterministic across link orderings +
468    /// often the wrong backend for the bridge's intended primary.
469    #[cfg(feature = "rmw-cffi")]
470    pub fn open_session(
471        locator: &str,
472        mode: nros_rmw::SessionMode,
473        domain_id: u32,
474        node_name: &str,
475    ) -> Result<RmwSession, nros_rmw::TransportError> {
476        use nros_rmw::Rmw;
477
478        // Phase 249 P4b.1 — every linked backend self-registered via
479        // its `.init_array` ctor before `main` (RFC-0042 §D3.3); no
480        // runtime section walk.
481
482        let config = nros_rmw::RmwConfig {
483            locator,
484            mode,
485            domain_id,
486            node_name,
487            namespace: "",
488            properties: &[],
489        };
490        // Phase 156 — honor `$NROS_RMW` env-var primary selector
491        // when present so C bridges built with multiple linked
492        // backends (e.g. xrce + dds) pin the primary deterministically
493        // instead of taking whichever linkme ctor fires first.
494        // Phase 155.B — propagate the real `TransportError` instead of
495        // collapsing every backend failure to `ConnectionFailed`. The
496        // C-side `nros_support_init` decodes the variant into a
497        // specific `NROS_RET_*` code so "init -> -X" tells the user
498        // which precondition the backend rejected.
499        #[cfg(feature = "std")]
500        if let Some(name) = std::env::var("NROS_RMW").ok().filter(|s| !s.is_empty()) {
501            return nros_rmw_cffi::CffiRmw::open_with_rmw(&name, &config);
502        }
503        nros_rmw_cffi::CffiRmw.open(&config)
504    }

507    ///
508    /// Delegates to [`Session::drive_io()`](nros_rmw::Session::drive_io),
509    /// which each backend implements appropriately (no-op for push-based,
510    /// poll for pull-based).
511    ///
