1//! Single source of truth for FFI opaque-storage sizes.
2//!
3//! Each `export_size!` invocation produces two artefacts:
4//!
5//! * `pub const FOO_SIZE: usize = core::mem::size_of::<T>();` — a normal
6//!   Rust const suitable for in-crate `const _: () = assert!(...)` checks and
7//!   direct use by `no_std` consumers.
8//! * `pub static __NROS_SIZE_FOO: [u8; FOO_SIZE] = [0; FOO_SIZE];` — an
9//!   array-sized static whose *symbol storage size* in the compiled rlib
10//!   equals `FOO_SIZE`. `nros-c`/`nros-cpp` build scripts read the sizes out
11//!   via [`nros_sizes_build::extract_sizes`](../../../nros-sizes-build/index.html)
12//!   to derive opaque-storage macros for the generated C/C++ headers.
13//!
14//! Feature gating follows the rest of the crate: the statics only exist when
15//! an RMW backend (`rmw-zenoh` / `rmw-xrce` / `rmw-dds` / `rmw-cffi`) is
16//! active, which is exactly the condition under which the `Rmw*` type
17//! aliases resolve. Workspace-level `cargo check` without any RMW feature
18//! sees this module as empty.
19

31    // Phase 77.25: per-name v0-mangled markers so the probe works
32    // under fat LTO. Each call to `export_size!(NAME = Ty)` expands to
33    // a distinct generic fn `__nros_size_NAME<const N: usize>` plus a
34    // monomorphised fn-pointer static. The monomorphisation's v0
35    // mangled symbol name contains both the name ("NAME") *and* the
36    // const-generic value (the size) — e.g. demangles as
37    // `nros::sizes::rmw_sizes::__nros_size_PUBLISHER_SIZE::<48>`.
38    // Symbol names survive LTO because the linker still needs them,
39    // even when the object file is LLVM bitcode and `object::parse`
40    // can't read symbol byte sizes. The original `__NROS_SIZE_<NAME>`
41    // static kept for backwards-compat is also emitted for consumers
42    // that still walk the legacy path.
43    #[doc(hidden)]
44    pub mod _size_markers {}
45
46    macro_rules! export_size {
47        ($vis:vis $name:ident = $ty:ty) => {
48            $vis const $name: usize = core::mem::size_of::<$ty>();
49            paste::paste! {
50                #[used]
51                #[unsafe(no_mangle)]
52                #[doc(hidden)]
53                pub static [<__NROS_SIZE_ $name>]: [u8; $name] = [0u8; $name];
54
55                #[doc(hidden)]
56                #[allow(non_snake_case)]
57                #[inline(never)]
58                pub fn [<__nros_size_ $name>]<const N: usize>() -> usize { N }
59
60                #[used]
61                #[doc(hidden)]
62                pub static [<__NROS_SIZE_FN_ $name>]: fn() -> usize =
63                    [<__nros_size_ $name>]::<{ $name }>;
64            }
65        };
66    }

76    // Phase 91.C: nros-c's `ActionServerInternal` embeds this nros-node
77    // type as a typed field. cbindgen (which can't recurse into deps)
78    // emits the field as `ActionServerRawHandle handle;` referencing a
79    // type it cannot define. nros-c's build.rs reads this size and emits

81    // so the cbindgen output is self-contained.

84    // Layout-mirror struct for `nros_c::action::ActionServerInternal`.
85    // ActionServerInternal lives in the `nros-c` crate (it embeds C-API
86    // pointer types like `*mut nros_action_server_t`), so it can't be
87    // referenced from `nros` directly. This mirror has the same `#[repr(C)]`
88    // field shape — `*mut c_void` and `unsafe extern "C" fn(*mut c_void, ...)`
89    // pointer slots — and therefore the same byte size, since fn-pointer
90    // size is independent of parameter types. nros-c asserts at compile
91    // time that `size_of::<ActionServerInternal>() ==
92    // size_of::<ActionServerInternalLayout>()`.
93    use core::ffi::c_void;
94    type CGoalCallbackLayout =
95        unsafe extern "C" fn(*mut c_void, *const c_void, *const u8, usize, *mut c_void) -> i32;
96    type CCancelCallbackLayout =
97        Option<unsafe extern "C" fn(*const c_void, i32, *mut c_void) -> i32>;
98    type CAcceptedCallbackLayout =
99        Option<unsafe extern "C" fn(*mut c_void, *const c_void, *mut c_void)>;
100
101    #[repr(C)]
102    #[doc(hidden)]
103    pub struct ActionServerInternalLayout {
104        pub handle: nros_node::ActionServerRawHandle,
105        pub executor_ptr: *mut c_void,
106        pub c_goal_callback: CGoalCallbackLayout,
107        pub c_cancel_callback: CCancelCallbackLayout,
108        pub c_accepted_callback: CAcceptedCallbackLayout,
109        pub c_context: *mut c_void,
110        pub server_ptr: *mut c_void,
111    }
112    export_size!(pub ACTION_SERVER_INTERNAL_SIZE = ActionServerInternalLayout);

114    // Layout-mirrors for nros-cpp's `CppActionServer` and `CppActionClient`.
115    //
116    // Same approach as `ActionServerInternalLayout` above: nros-cpp's
117    // wrapper structs live in a downstream crate but their byte sizes can
118    // be reconstructed from the field shape. This eliminates the
119    // hand-math in `nros-cpp/build.rs` (was Phase 87.11).
120    //
121    // The C++-side `nros::ActionServer<A>` / `nros::ActionClient<A>`
122    // classes hold opaque storage sized to these probe values. nros-cpp
123    // asserts `size_of::<CppActionServer>() == size_of::<CppActionServerLayout>()`
124    // (and the same for CppActionClient) so any field-shape drift in the
125    // real wrapper trips the build immediately.
126
127    type CppGoalCallbackLayout =
128        unsafe extern "C" fn(*const [u8; 16], *const u8, usize, *mut c_void) -> i32;
129    type CppCancelCallbackLayout = unsafe extern "C" fn(*const [u8; 16], *mut c_void) -> i32;
130
