1//! Phase 212.K.7.6.b — `nros_rmw_cyclonedds::register::<M>()` bridge.
2//!
3//! The Cyclone DDS RMW backend resolves topic-type descriptors via a
4//! runtime registry (`nros-rmw-cyclonedds::type_registry`) instead of the
5//! legacy static-init `descriptors.cpp` table. Each `nros-node` typed
6//! creator (`create_publisher`, `create_subscription`, `create_client`,
7//! `create_service`, `create_action_*`) routes through
8//! [`register_type::<M>`] *before* asking the cffi vtable to create the
9//! entity so the descriptor exists in the registry when
10//! `dds_create_topic` runs in the C++ bridge.
11//!
12//! # cfg gating (Phase 214.S.2 — auto-detected, not feature-gated)
13//!
14//! This module compiles to no-ops unless `cfg(rmw_cyclonedds_present)`
15//! is on. The cfg is emitted by `nros-node/build.rs` when:
16//!
17//! * the sibling `nros-rmw-cyclonedds-sys` shim crate (which carries
18//!   `links = "cyclonedds"`) is in the dep graph, surfacing
19//!   `DEP_CYCLONEDDS_PRESENT=1` to our build script; **or**
20//! * the private internal `__cyclonedds-link` feature is on (the umbrella
21//!   `nros/rmw-cyclonedds` activates it; this guarantees a direct edge to
22//!   the linking crate exists, since cargo's `DEP_*` env-vars only
23//!   propagate to *direct* dependents).
24//!
25//! Net effect: callers depend on `nros = { features = ["rmw-cyclonedds"] }`
26//! and the K.7.6.b hook lights up automatically — no user-facing feature
27//! flag on `nros-node` (was: `feature = "rmw-cyclonedds"`, dropped in
28//! Phase 214.S.2). Each typed creator calls [`register_type::<M>`]
29//! unconditionally; the body is empty when the cfg is off so zenoh/xrce
30//! paths pay nothing. With the cfg on, the caller pays one mutex
31//! acquisition + one `FnvIndexMap` lookup per creator invocation
32//! (idempotent; the registry caches the descriptor pointer on the
33//! first hit).
34//!
35//! # Trait bound — [`MessageForRmw`]
36//!
37//! The cyclonedds registry needs [`nros_serdes::schema::Message`] for its
38//! static field schema, but `nros-node`'s typed creators historically
39//! only constrain `M: nros_core::RosMessage`. Adding `Message` as a
40//! super-bound on `RosMessage` breaks every existing codegen-emitted msg
41//! crate (they impl `RosMessage` but not yet `Message`). Adding it as a
42//! per-method bound on every typed creator touches 30+ sites.
43//!
44//! Compromise: introduce a sealed helper trait
45//! [`MessageForRmw`] that is **the bound the typed creators use** in
46//! place of bare `M: RosMessage`. It is a blanket impl over
47//! `RosMessage` whose extra requirement is `Message` when
48//! `cfg(rmw_cyclonedds_present)` is on, and just `RosMessage` when off.
49//!
50//! Net effect: a msg crate that impls `RosMessage` works as-is for
51//! zenoh + xrce builds; for cyclonedds builds it must additionally impl
52//! `Message`. The codegen template (`nros-cli` K.7.1.b — separate
53//! agent / repo) emits both impls for every generated msg crate.
54//!
55//! # Error mapping
56//!
57//! `nros_rmw_cyclonedds::BuildError` flattens onto
58//! [`crate::NodeError::Transport`] with
59//! `TransportError::PublisherCreationFailed`. Backend-specific diagnostic
60//! is lost — the underlying `BuildError` is logged via the `log` crate
61//! when that feature is enabled. The choice not to add a dedicated
62//! `NodeError::CyclonedsTypeRegistrationFailed` variant is deliberate:
63//! the C/C++ FFI shim widens to a single `nros_ret_t`, and the failure
64//! mode (out-of-capacity registry, bridge error, etc.) is a "topic could
65//! not be created" from the caller's perspective.
66
67use nros_core::RosMessage;

69// ============================================================================
70// Phase 214.S.4.b — link-graph keep-alive for `nros-rmw-cyclonedds-sys`
71// ============================================================================
72//
73// The Cyclone backend's C++ register entry (`nros_rmw_cyclonedds_register`)
74// and its linkme self-register section live inside `nros-rmw-cyclonedds-sys`
75// (and its C++ companion `nros-rmw-cyclonedds`). Without an explicit Rust-
76// side reference to a symbol from that crate, rust-lld drops the rlib from
77// the final link and the C++ static-init register TU is pruned along with
78// it.
79//
80// Before S.4.b, example `src/main.rs` carried `nros_rmw_cyclonedds_sys::
81// register()` directly which doubled as the symbol-drag. S.4.b moves that
82// keep-alive in here so user crates can collapse their `rmw-cyclonedds`
83// feature to strict 1-entry parity (`["nros/rmw-cyclonedds"]`) with the
84// zenoh / xrce siblings — no `dep:nros-rmw-cyclonedds-sys` activator and
85// no direct `register()` call in user code. The linkme self-register
86// section inside `-sys` (Phase 128.B.3) fires the register on
87// `nros::init` via the cffi-rmw walker.
88//
89// Mirrors the existing pattern in `nros-platform::__FORCE_LINK_CFFI`
90// (`packages/core/nros-platform/src/lib.rs:83`) which keeps
91// `nros-platform-cffi` alive through `nros-rmw-zenoh`.
92//
93// Gated on a separate cfg from `rmw_cyclonedds_present` because the
94// `cyclonedds_register_smoke` test enables the latter via the lighter
95// `__cyclonedds-detect` feature path (no `-sys` dep, bridge symbols
96// from `nros-rmw-cyclonedds[bridge-stub]` dev-dep). Production
97// callers go through the umbrella's `__cyclonedds-link` which emits
98// BOTH `rmw_cyclonedds_present` AND `cyclonedds_link_keepalive`.
99// Pinning the `-sys` rlib under the bare detect path would force
100// rust-lld to whole-archive the production C++ libnros_rmw_cyclonedds
101// in alongside the bridge-stub Rust impls → duplicate-symbol link
102// failure on the smoke test.
103#[cfg(cyclonedds_link_keepalive)]
104#[used]
105#[doc(hidden)]
106pub static __FORCE_LINK_CYCLONEDDS_SYS: fn() -> Result<(), nros_rmw_cyclonedds_sys::RegisterError> =
107    nros_rmw_cyclonedds_sys::register;

109/// Bound used in place of bare `RosMessage` on typed creators.
110///
111/// Equivalent to `RosMessage` without `cfg(rmw_cyclonedds_present)`;
112/// equal to `RosMessage + nros_serdes::schema::Message` with it.
113///
114/// See module-level docs for the rationale.
115#[cfg(rmw_cyclonedds_present)]
116pub trait MessageForRmw: RosMessage + nros_serdes::schema::Message {}
117
118#[cfg(rmw_cyclonedds_present)]
119impl<T> MessageForRmw for T where T: RosMessage + nros_serdes::schema::Message {}
120
121#[cfg(not(rmw_cyclonedds_present))]
122pub trait MessageForRmw: RosMessage {}
123
124#[cfg(not(rmw_cyclonedds_present))]
125impl<T> MessageForRmw for T where T: RosMessage {}

131/// Register `M`'s cyclonedds topic descriptor with the runtime registry.
132///
133/// No-op when `cfg(rmw_cyclonedds_present)` is off. With the cfg on,
134/// delegates to `nros_rmw_cyclonedds::register::<M>()`. The first call
135/// for a given `M::TYPE_NAME` builds the descriptor via the C++ bridge
136/// and caches it; subsequent calls are O(1) lookups.
137///
138/// Returns `Ok(())` on success or `NodeError::Transport(
139/// TransportError::PublisherCreationFailed)` on any
140/// [`nros_rmw_cyclonedds::BuildError`].
141#[allow(unused_variables)] // M unused without the cfg
142#[inline]
143pub fn register_type<M: MessageForRmw>() -> Result<(), crate::NodeError> {
144    #[cfg(rmw_cyclonedds_present)]
145    {
146        // SAFETY-OF-INPUT: `M: Message` is enforced by the `MessageForRmw`
147        // bound under `rmw_cyclonedds_present`, so `register::<M>()`
148        // receives the schema it expects.
149        nros_rmw_cyclonedds::register::<M>().map_err(|err| {
150            #[cfg(feature = "log")]
151            log::error!(
152                "nros_rmw_cyclonedds::register::<{}>() failed: {:?}",
153                <M as nros_serdes::schema::Message>::TYPE_NAME,
154                err
155            );
156            let _ = err; // silence unused without log
157            crate::NodeError::Transport(nros_rmw::TransportError::PublisherCreationFailed)
158        })?;
159    }
160    Ok(())
161}
