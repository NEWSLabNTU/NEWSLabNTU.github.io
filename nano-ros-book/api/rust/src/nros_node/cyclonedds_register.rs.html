1//! Phase 212.K.7.6.b — `nros_rmw_cyclonedds::register::<M>()` bridge.
2//!
3//! The Cyclone DDS RMW backend resolves topic-type descriptors via a
4//! runtime registry (`nros-rmw-cyclonedds::type_registry`) instead of the
5//! legacy static-init `descriptors.cpp` table. Each `nros-node` typed
6//! creator (`create_publisher`, `create_subscription`, `create_client`,
7//! `create_service`, `create_action_*`) routes through
8//! [`register_type::<M>`] *before* asking the cffi vtable to create the
9//! entity so the descriptor exists in the registry when
10//! `dds_create_topic` runs in the C++ bridge.
11//!
12//! # Feature gating
13//!
14//! This module compiles to no-ops unless the `rmw-cyclonedds` feature is
15//! active on `nros-node`. Each typed creator calls
16//! [`register_type::<M>`] unconditionally; the body is empty without the
17//! feature so zenoh/xrce paths pay nothing. With the feature on, the
18//! caller pays one mutex acquisition + one `FnvIndexMap` lookup per
19//! creator invocation (idempotent; the registry caches the descriptor
20//! pointer on the first hit).
21//!
22//! # Trait bound — [`MessageForRmw`]
23//!
24//! The cyclonedds registry needs [`nros_serdes::schema::Message`] for its
25//! static field schema, but `nros-node`'s typed creators historically
26//! only constrain `M: nros_core::RosMessage`. Adding `Message` as a
27//! super-bound on `RosMessage` breaks every existing codegen-emitted msg
28//! crate (they impl `RosMessage` but not yet `Message`). Adding it as a
29//! per-method bound on every typed creator touches 30+ sites.
30//!
31//! Compromise: introduce a sealed helper trait
32//! [`MessageForRmw`] that is **the bound the typed creators use** in
33//! place of bare `M: RosMessage`. It is a blanket impl over
34//! `RosMessage` whose extra requirement is `Message` when the
35//! `rmw-cyclonedds` feature is on, and just `RosMessage` when it is off.
36//!
37//! Net effect: a msg crate that impls `RosMessage` works as-is for
38//! zenoh + xrce builds; for cyclonedds builds it must additionally impl
39//! `Message`. The codegen template (`nros-cli` K.7.1.b — separate
40//! agent / repo) emits both impls for every generated msg crate.
41//!
42//! # Error mapping
43//!
44//! `nros_rmw_cyclonedds::BuildError` flattens onto
45//! [`crate::NodeError::Transport`] with
46//! `TransportError::PublisherCreationFailed`. Backend-specific diagnostic
47//! is lost — the underlying `BuildError` is logged via the `log` crate
48//! when that feature is enabled. The choice not to add a dedicated
49//! `NodeError::CyclonedsTypeRegistrationFailed` variant is deliberate:
50//! the C/C++ FFI shim widens to a single `nros_ret_t`, and the failure
51//! mode (out-of-capacity registry, bridge error, etc.) is a "topic could
52//! not be created" from the caller's perspective.
53
54use nros_core::RosMessage;
55
56/// Bound used in place of bare `RosMessage` on typed creators.
57///
58/// Equivalent to `RosMessage` without the `rmw-cyclonedds` feature; equal
59/// to `RosMessage + nros_serdes::schema::Message` with it.
60///
61/// See module-level docs for the rationale.
62#[cfg(feature = "rmw-cyclonedds")]
63pub trait MessageForRmw: RosMessage + nros_serdes::schema::Message {}
64
65#[cfg(feature = "rmw-cyclonedds")]
66impl<T> MessageForRmw for T where T: RosMessage + nros_serdes::schema::Message {}
67
68#[cfg(not(feature = "rmw-cyclonedds"))]
69pub trait MessageForRmw: RosMessage {}
70
71#[cfg(not(feature = "rmw-cyclonedds"))]
72impl<T> MessageForRmw for T where T: RosMessage {}

78/// Register `M`'s cyclonedds topic descriptor with the runtime registry.
79///
80/// No-op when the `rmw-cyclonedds` feature is off. With the feature on,
81/// delegates to `nros_rmw_cyclonedds::register::<M>()`. The first call
82/// for a given `M::TYPE_NAME` builds the descriptor via the C++ bridge
83/// and caches it; subsequent calls are O(1) lookups.
84///
85/// Returns `Ok(())` on success or `NodeError::Transport(
86/// TransportError::PublisherCreationFailed)` on any
87/// [`nros_rmw_cyclonedds::BuildError`].
88#[allow(unused_variables)] // M unused without the feature
89#[inline]
90pub fn register_type<M: MessageForRmw>() -> Result<(), crate::NodeError> {
91    #[cfg(feature = "rmw-cyclonedds")]
92    {
93        // SAFETY-OF-INPUT: `M: Message` is enforced by the `MessageForRmw`
94        // bound under the cyclonedds feature, so `register::<M>()`
95        // receives the schema it expects.
96        nros_rmw_cyclonedds::register::<M>().map_err(|err| {
97            #[cfg(feature = "log")]
98            log::error!(
99                "nros_rmw_cyclonedds::register::<{}>() failed: {:?}",
100                <M as nros_serdes::schema::Message>::TYPE_NAME,
101                err
102            );
103            let _ = err; // silence unused without log
104            crate::NodeError::Transport(nros_rmw::TransportError::PublisherCreationFailed)
105        })?;
106    }
107    Ok(())
108}
