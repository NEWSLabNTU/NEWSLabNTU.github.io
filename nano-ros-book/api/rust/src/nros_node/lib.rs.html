1//! Node abstraction for nros
2//!
3//! This crate provides the high-level Node API for creating ROS 2 compatible
4//! publishers and subscribers on embedded systems.
5//!
6//! # Executor-Based API
7//!
8//! The executor-based API provides a unified interface that works on both
9//! std (desktop) and no_std (embedded) targets.
10//!
11//! ## Desktop Example
12//!
13//! ```ignore
14//! use nros::prelude::*;
15//! use std_msgs::msg::Int32;
16//!
17//! let config = ExecutorConfig::from_env().node_name("my_node");
18//! let mut executor: Executor = Executor::open(&config)?;
19//!
20//! // Register subscription callback
21//! executor.register_subscription::<Int32, _>("/topic", |msg: &Int32| {
22//!     println!("Received: {}", msg.data);
23//! })?;
24//!
25//! // Spin (processes callbacks)
26//! executor.spin_blocking(SpinOptions::default());
27//! ```
28//!
29//! ## Embedded Example
30//!
31//! ```ignore
32//! use nros::prelude::*;
33//! use std_msgs::msg::Int32;
34//!
35//! let config = ExecutorConfig { locator: "tcp/192.168.1.1:7447", ..Default::default() };
36//! let mut executor: Executor = Executor::open(&config)?;
37//!
38//! // Register subscription callback
39//! executor.register_subscription::<Int32, _>("/cmd", |msg: &Int32| {
40//!     // process message...
41//! })?;
42//!
43//! // In your main loop:
44//! loop {
45//!     executor.spin_once(core::time::Duration::from_millis(10));
46//!     // platform delay...
47//! }
48//! ```
49//!
50//! # Features
51//!
52//! - `std` - Enable standard library support (spin_blocking)
53//! - `alloc` - Enable heap allocation (parameter service boxed replies)
