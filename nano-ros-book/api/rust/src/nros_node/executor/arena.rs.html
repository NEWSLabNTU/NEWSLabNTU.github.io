228/// Buffer strategy selected by QoS depth at subscription registration time.
229///
230/// The buffer data lives in a trailing region immediately after the
231/// `SubBufferedEntry` struct in the arena.
232pub(crate) enum BufferStrategy {
233    /// `KEEP_LAST(1)`: 3 slots, latest-value semantics, writer never blocks.
234    Triple(TripleBuffer),
235    /// `KEEP_LAST(N)` where N > 1: N+1 slots, FIFO ordering, bounded drops.
236    Ring(SpscRing),
237}
238
239impl BufferStrategy {
240    /// Check if new data is available.
241    pub(crate) fn has_data(&self) -> bool {
242        match self {
243            BufferStrategy::Triple(tb) => tb.has_data(),
244            BufferStrategy::Ring(ring) => ring.has_data(),
245        }
246    }
247}
248
249/// Compute the number of buffer slots and trailing region size for a given
250/// QoS depth and per-slot buffer size.
251///
252/// Returns `(slot_count, trailing_bytes)`.
253pub(crate) fn buffered_region_size(depth: u32, slot_size: usize) -> (usize, usize) {
254    if depth <= 1 {
255        // Triple buffer: 3 fixed slots
256        (
257            TripleBuffer::SLOT_COUNT,
258            TripleBuffer::SLOT_COUNT * slot_size,
259        )
260    } else {
261        let d = depth as usize;
262        (SpscRing::slot_count(d), SpscRing::region_size(d, slot_size))
263    }
264}

286/// Drain the RMW subscriber handle into the buffer strategy.
287///
288/// Calls `try_recv_raw()` on the subscriber handle and writes received data
289/// into the triple buffer's write slot or the SPSC ring's next push slot.
290///
291/// # Safety
292/// `entry` must be a valid mutable reference to a `SubBufferedEntry`.
293unsafe fn drain_into_buffer<M, F>(entry: &mut SubBufferedEntry<M, F>) {
294    match &entry.buffer {
295        BufferStrategy::Triple(tb) => {
296            let slot = tb.write_slot();
297            if let Ok(Some(len)) = entry.handle.try_recv_raw(slot) {
298                tb.writer_publish(len);
299            }
300        }
301        BufferStrategy::Ring(ring) => {
302            // Drain all pending messages into ring slots
303            while let Some(slot) = ring.try_push() {
304                if let Ok(Some(len)) = entry.handle.try_recv_raw(slot) {
305                    ring.commit_push(len);
306                } else {
307                    break; // no more data
308                }
309            }
310        }
311    }
312}

314/// Monomorphized dispatch for buffered subscriptions.
315///
316/// First drains the RMW subscriber into the buffer strategy (triple buffer
317/// or SPSC ring), then dispatches from the buffer to the user callback.
318///
319/// # Safety
320/// `ptr` must point to a valid, aligned `SubBufferedEntry<M, F>`.
321pub(crate) unsafe fn sub_buffered_try_process<M, F>(
322    ptr: *mut u8,
323    _delta_ms: u64,
324) -> Result<bool, TransportError>
325where
326    M: RosMessage,
327    F: FnMut(&M),
328{
329    let entry = unsafe { &mut *(ptr as *mut SubBufferedEntry<M, F>) };
330
331    // Phase 1: drain RMW subscriber → buffer strategy
332    unsafe { drain_into_buffer(entry) };
333
334    // Phase 2: dispatch from buffer → user callback
335    match &entry.buffer {
336        BufferStrategy::Triple(tb) => match tb.reader_acquire() {
337            Some((data, len)) => {
338                let mut reader = CdrReader::new_with_header(&data[..len])
339                    .map_err(|_| TransportError::DeserializationError)?;
340                let msg = M::deserialize(&mut reader)
341                    .map_err(|_| TransportError::DeserializationError)?;
342                (entry.callback)(&msg);
343                Ok(true)
344            }
345            None => Ok(false),
346        },
347        BufferStrategy::Ring(ring) => {
348            let mut did_work = false;
349            while let Some((data, len)) = ring.try_pop() {
350                let mut reader = CdrReader::new_with_header(&data[..len])
351                    .map_err(|_| TransportError::DeserializationError)?;
352                let msg = M::deserialize(&mut reader)
353                    .map_err(|_| TransportError::DeserializationError)?;
354                (entry.callback)(&msg);
355                ring.commit_pop();
356                did_work = true;
357            }
358            Ok(did_work)
359        }
360    }
361}

394/// Monomorphized in-place dispatch for typed subscriptions.
395///
396/// Drains all pending messages from the backend, deserializing + invoking the
397/// callback directly from each borrowed slot. Returns `Ok(true)` if any message
398/// was dispatched.
399///
400/// # Safety
401/// `ptr` must point to a valid, aligned `SubInplaceEntry<M, F>`.
402pub(crate) unsafe fn sub_inplace_try_process<M, F>(
403    ptr: *mut u8,
404    _delta_ms: u64,
405) -> Result<bool, TransportError>
406where
407    M: RosMessage,
408    F: FnMut(&M),
409{
410    let entry = unsafe { &mut *(ptr as *mut SubInplaceEntry<M, F>) };
411    // Split-borrow the handle and callback (disjoint fields).
412    let SubInplaceEntry {
413        handle, callback, ..
414    } = entry;
415    let mut did_work = false;
416    loop {
417        let mut deser_err = false;
418        let processed =
419            handle.process_raw_in_place(|raw| match CdrReader::new_with_header(raw) {
420                Ok(mut reader) => match M::deserialize(&mut reader) {
421                    Ok(msg) => (callback)(&msg),
422                    Err(_) => deser_err = true,
423                },
424                Err(_) => deser_err = true,
425            })?;
426        if deser_err {
427            return Err(TransportError::DeserializationError);
428        }
429        if processed {
430            did_work = true;
431        } else {
432            break;
433        }
434    }
435    Ok(did_work)
436}

533// Borrowed (zero-copy) buffered subscription (Phase 229.6, issue 0007)
534// ============================================================================
535
536/// Buffered subscription entry for borrowed (zero-copy) message callbacks.
537///
538/// The callback receives `&B::View<'a>` — a lifetime-carrying message whose
539/// unbounded sequence/string fields borrow directly from the triple buffer's
540/// read slot (no arena copy, no `heapless::Vec` copy). The view is materialised
541/// per dispatch via [`DeserializeBorrowed`] and dropped before the slot is
542/// released, so the borrow never outlives the buffer.
543///
544/// **Triple-buffer only.** A borrowed view must reference exactly one
545/// well-defined slot for the duration of the callback; an SPSC ring (depth > 1)
546/// holds several samples in flight with no single such slot. Registration
547/// rejects `qos.depth > 1` for borrowed subscriptions, so `buffer` is always
548/// [`BufferStrategy::Triple`] here.
549#[repr(C)]
550pub(crate) struct SubBufferedBorrowedEntry<B, F> {
551    pub(crate) handle: session::RmwSubscriber,
552    pub(crate) buffer: BufferStrategy,
553    pub(crate) callback: F,
554    pub(crate) _phantom: PhantomData<B>,
555}

557/// Dispatch for borrowed (zero-copy) buffered subscriptions.
558///
559/// Drains the RMW handle into the triple buffer, then materialises a borrowed
560/// `B::View<'_>` over the read slot and hands it to the callback. The view
561/// borrows the slot; it is dropped at the end of the callback, before the next
562/// dispatch can publish over the slot.
563///
564/// # Safety
565/// `ptr` must point to a valid, aligned `SubBufferedBorrowedEntry<B, F>`.
566pub(crate) unsafe fn sub_buffered_borrowed_try_process<B, F>(
567    ptr: *mut u8,
568    _delta_ms: u64,
569) -> Result<bool, TransportError>
570where
571    B: BorrowedMessage,
572    F: for<'a> FnMut(&B::View<'a>),
573{
574    let entry = unsafe { &mut *(ptr as *mut SubBufferedBorrowedEntry<B, F>) };
575
576    // Borrowed subscriptions are triple-buffer only (enforced at registration).
577    let tb = match &entry.buffer {
578        BufferStrategy::Triple(tb) => tb,
579        // Unreachable: registration rejects depth > 1. Treat as no work.
580        BufferStrategy::Ring(_) => return Ok(false),
581    };
582
583    // Phase 1: drain RMW subscriber → triple buffer write slot.
584    {
585        let slot = tb.write_slot();
586        if let Ok(Some(len)) = entry.handle.try_recv_raw(slot) {
587            tb.writer_publish(len);
588        }
589    }
590
591    // Phase 2: borrow the read slot and deserialize a view over it (no copy).
592    match tb.reader_acquire() {
593        Some((data, len)) => {
594            let mut reader = CdrReader::new_with_header(&data[..len])
595                .map_err(|_| TransportError::DeserializationError)?;
596            let msg = <B::View<'_> as DeserializeBorrowed>::deserialize_borrowed(&mut reader)
597                .map_err(|_| TransportError::DeserializationError)?;
598            (entry.callback)(&msg);
599            Ok(true)
600        }
601        None => Ok(false),
602    }
603}

853/// Drain helper for C-style raw buffered entries.
854unsafe fn drain_into_buffer_raw_c(entry: &mut SubBufferedRawCEntry) {
855    match &entry.buffer {
856        BufferStrategy::Triple(tb) => {
857            let slot = tb.write_slot();
858            if let Ok(Some(len)) = entry.handle.try_recv_raw(slot) {
859                tb.writer_publish(len);
860            }
861        }
862        BufferStrategy::Ring(ring) => {
863            while let Some(slot) = ring.try_push() {
864                if let Ok(Some(len)) = entry.handle.try_recv_raw(slot) {
865                    ring.commit_push(len);
866                } else {
867                    break;
868                }
869            }
870        }
871    }
872}

898                ring.commit_pop();
899                did_work = true;
900            }
901            Ok(did_work)
902        }
903    }
904}

1232/// message's leading 4-byte encap (always a valid header). When `raw` already
1233/// begins with an encap (zenoh/XRCE) it is read directly; when it does not
1234/// (Cyclone `dds_stream` drops the inner encap of a nested message field) the
1235/// top-level encap is spliced in front into a `CAP`-byte scratch buffer first.
1236fn read_action_field<M: nros_serdes::Deserialize, const CAP: usize>(
1237    top_encap: &[u8],
1238    raw: &[u8],
1239) -> Option<M> {
1240    if payload_has_cdr_encap(raw) {
1241        let mut reader = CdrReader::new_with_header(raw).ok()?;
1242        return M::deserialize(&mut reader).ok();
1243    }
1244    if top_encap.len() < 4 || raw.len() + 4 > CAP {
1245        return None;
1246    }
1247    let mut buf = [0u8; CAP];
1248    buf[0..4].copy_from_slice(&top_encap[0..4]);
1249    buf[4..4 + raw.len()].copy_from_slice(raw);
1250    let mut reader = CdrReader::new_with_header(&buf[..4 + raw.len()]).ok()?;
1251    M::deserialize(&mut reader).ok()
1252}

1254pub(crate) unsafe fn action_client_raw_try_process<
1255    const GOAL_BUF: usize,
1256    const RESULT_BUF: usize,
1257    const FEEDBACK_BUF: usize,
1258>(
1259    ptr: *mut u8,
1260    _delta_ms: u64,
1261) -> Result<bool, TransportError> {
1262    let entry = unsafe {
1263        &mut *(ptr as *mut ActionClientRawArenaEntry<GOAL_BUF, RESULT_BUF, FEEDBACK_BUF>)
1264    };
1265    let ActionClientRawArenaEntry {
1266        core,
1267        goal_response_callback,
1268        feedback_callback,
1269        result_callback,
1270        context,
1271    } = entry;
1272
1273    let mut did_work = false;
1274
1275    // 1. Poll goal acceptance reply
1276    if let Ok(Some(total_len)) = core.try_recv_send_goal_reply() {
1277        if let Some(cb) = goal_response_callback {
1278            // Reply CDR: header (4) + accepted (u8) + stamp
1279            let accepted = total_len >= 5 && core.result_buffer[4] != 0;
1280            // Extract GoalId from the last sent goal
1281            let goal_id = nros_core::GoalId {
1282                uuid: {
1283                    let mut uuid = [0u8; 16];
1284                    let counter = core.goal_counter.to_le_bytes();
1285                    uuid[..8].copy_from_slice(&counter);
1286                    uuid
1287                },
1288            };
1289            unsafe { cb(&goal_id, accepted, *context) };
1290        }
1291        did_work = true;
1292    }
1293
1294    // 2. Poll feedback
1295    if let Ok(Some((goal_id, total_len))) = core.try_recv_feedback_raw() {
1296        if let Some(cb) = feedback_callback {
1297            // Feedback buffer layout from `publish_feedback_raw` in
1298            // `action_core.rs`:
1299            //   bytes 0..4   outer CDR header (`new_with_header`)
1300            //   bytes 4..20  GoalId.uuid (16 bytes, fixed `uint8[16]`,
1301            //                no length prefix — ROS 2
1302            //                `unique_identifier_msgs/UUID`; see
1303            //                `action_core::write_goal_id`)
1304            //   bytes 20..   payload — exactly the bytes the caller
1305            //                of `publish_feedback_raw` handed in
1306            //                (typed serializers like `ffi_serialize`
1307            //                write a CDR header at the front).
1308            //
1309            // 233.6: the GoalId carries NO sequence-length prefix (it did
1310            // pre-233.6, which made the offset `4 + 4 + 16`; that framing
1311            // self-matched nano-ros peers but a real `rcl_action` peer
1312            // rejects the extra 4 bytes).
1313            const FEEDBACK_PAYLOAD_OFFSET: usize = 4 + 16;
1314            if total_len > FEEDBACK_PAYLOAD_OFFSET {
1315                // #175 — same encap restoration as the result path below: Cyclone's
1316                // typed `FeedbackMessage` framing carries `feedback` as a nested
1317                // field, so `dds_stream` drops the inner encap and the fields arrive
1318                // raw here. Splice the top-level encap (`feedback_buffer[0..4]`) when
1319                // absent; zenoh/XRCE keep it and pass through.
1320                let raw = &core.feedback_buffer[FEEDBACK_PAYLOAD_OFFSET..total_len];
1321                if payload_has_cdr_encap(raw) {
1322                    unsafe { cb(&goal_id, raw.as_ptr(), raw.len(), *context) };
1323                } else {
1324                    let mut spliced = [0u8; FEEDBACK_BUF];
1325                    let n = raw.len().min(FEEDBACK_BUF - 4);
1326                    spliced[0..4].copy_from_slice(&core.feedback_buffer[0..4]);
1327                    spliced[4..4 + n].copy_from_slice(&raw[..n]);
1328                    unsafe { cb(&goal_id, spliced.as_ptr(), 4 + n, *context) };
1329                }
1330            }
1331        }
1332        did_work = true;
1333    }
1334
1335    // 3. Poll result reply
1336    if let Ok(Some(total_len)) = core.try_recv_get_result_reply() {
1337        if let Some(cb) = result_callback {
1338            // Reply layout from `try_handle_get_result_raw` in
1339            // `action_core.rs`:
1340            //   bytes 0..4   outer CDR header (`new_with_header`)
1341            //   byte  4      status (i8)
1342            //   bytes 5..8   align(4) pad
1343            //   bytes 8..    payload — exactly the bytes the caller
1344            //                of `complete_goal_raw` handed in (typed
1345            //                serializers like `ffi_serialize` write
1346            //                a CDR header at the front, which is
1347            //                why the alignment pad above is sized
1348            //                to land the payload at a 4-byte boundary).
1349            //
1350            // The trampoline forwards `payload` to the C/C++
1351            // callback verbatim — the cpp wrapper expects to see
1352            // the inner CDR header that `ffi_serialize` wrote.
1353            // Earlier code used `result_offset = 5` and skipped
1354            // only the status byte; that leaked the 3 alignment
1355            // pad bytes into the payload prefix and blew up
1356            // `ffi_deserialize`, surfacing as an empty result on
1357            // the cpp/xrce action client (Phase 96.1 follow-up).
1358            const RESULT_PAYLOAD_OFFSET: usize = 8;
1359            if total_len >= RESULT_PAYLOAD_OFFSET {
1360                let status_byte = core.result_buffer[4];
1361                let status = match status_byte {
1362                    4 => nros_core::GoalStatus::Succeeded,
1363                    5 => nros_core::GoalStatus::Canceled,
1364                    6 => nros_core::GoalStatus::Aborted,
1365                    _ => nros_core::GoalStatus::Unknown,
1366                };
1367                // Extract GoalId from the last sent goal
1368                let goal_id = nros_core::GoalId {
1369                    uuid: {
1370                        let mut uuid = [0u8; 16];
1371                        let counter = core.goal_counter.to_le_bytes();
1372                        uuid[..8].copy_from_slice(&counter);
1373                        uuid
1374                    },
1375                };
1376                // #175 — restore the result's CDR encapsulation header if the
1377                // backend's typed reply framing dropped it. Cyclone sends the
1378                // `GetResult_Response` as a TYPED DDS sample whose `result` is a
1379                // NESTED field (no per-message encap), so `dds_stream` consumes
1380                // the inner encap the server serialised and the fields arrive raw
1381                // at `RESULT_PAYLOAD_OFFSET`. The consumer (`CallbackCtx::message`
1382                // / `ffi_deserialize`) reads with `new_with_header` and would eat
1383                // the first data word (e.g. a sequence length) as the encap. A raw
1384                // CDR field never begins with an encoding identifier
1385                // (`00 {00|01|06|07|0a|0b}`), so detect the missing header and
1386                // splice the reply's top-level encap (`result_buffer[0..4]`, always
1387                // valid) in front. Zenoh/XRCE preserve the inner encap, so their
1388                // payload already starts with one and is passed through unchanged.
1389                let raw = &core.result_buffer[RESULT_PAYLOAD_OFFSET..total_len];
1390                if payload_has_cdr_encap(raw) {
1391                    unsafe { cb(&goal_id, status, raw.as_ptr(), raw.len(), *context) };
1392                } else {
1393                    let mut spliced = [0u8; RESULT_BUF];
1394                    let n = raw.len().min(RESULT_BUF - 4);
1395                    spliced[0..4].copy_from_slice(&core.result_buffer[0..4]);
1396                    spliced[4..4 + n].copy_from_slice(&raw[..n]);
1397                    unsafe { cb(&goal_id, status, spliced.as_ptr(), 4 + n, *context) };
1398                }
1399            }
1400        }
1401        did_work = true;
1402    }
1403
1404    Ok(did_work)
1405}

1407/// Monomorphized raw service-client dispatch function.
1408///
1409/// Checks `reply_ready` (set by the transport waker) before calling
1410/// `try_recv_reply_raw`. This avoids blind polling on every spin tick —
1411/// the only cost per tick is an atomic load when no reply is pending.
1412///
1413/// # Safety
1414/// `ptr` must point to a valid, aligned `ServiceClientRawArenaEntry<REPLY_BUF>`.
1415pub(crate) unsafe fn service_client_raw_try_process<const REPLY_BUF: usize>(
1416    ptr: *mut u8,
1417    _delta_ms: u64,
1418) -> Result<bool, TransportError> {
1419    use core::sync::atomic::Ordering;
1420    use nros_rmw::ServiceClientTrait;
1421    let entry = unsafe { &mut *(ptr as *mut ServiceClientRawArenaEntry<REPLY_BUF>) };
1422
1423    if !entry.pending {
1424        return Ok(false);
1425    }
1426
1427    // Clear the waker flag if set (consumed by this check).
1428    entry.reply_ready.store(false, Ordering::Release);
1429
1430    match entry.handle.try_recv_reply_raw(&mut entry.reply_buffer) {
1431        Ok(Some(len)) => {
1432            entry.pending = false;
1433            if let Some(cb) = entry.callback {
1434                unsafe { cb(entry.reply_buffer.as_ptr(), len, entry.context) };
1435            }
1436            Ok(true)
1437        }
1438        Ok(None) => Ok(false),
1439        Err(_) => {
1440            entry.pending = false;
1441            Err(TransportError::ServiceRequestFailed)
1442        }
1443    }
1444}

1472/// Monomorphized typed service-client dispatch (RFC-0041, Phase 239.1).
1473///
1474/// Mirrors [`service_client_raw_try_process`] but deserializes the reply into
1475/// `Svc::Reply` and invokes the typed closure. Single in-flight request gated by
1476/// `hdr.pending`; the reply view is dropped before return (no escape).
1477///
1478/// # Safety
1479/// `ptr` must point to a valid, aligned `ServiceClientCallbackEntry<Svc, F, REPLY_BUF>`.
1480pub(crate) unsafe fn service_client_callback_try_process<Svc, F, const REPLY_BUF: usize>(
1481    ptr: *mut u8,
1482    _delta_ms: u64,
1483) -> Result<bool, TransportError>
1484where
1485    Svc: RosService,
1486    F: FnMut(&Svc::Reply),
1487{
1488    use core::sync::atomic::Ordering;
1489    use nros_rmw::ServiceClientTrait;
1490    let entry = unsafe { &mut *(ptr as *mut ServiceClientCallbackEntry<Svc, F, REPLY_BUF>) };
1491
1492    if !entry.hdr.pending {
1493        return Ok(false);
1494    }
1495    entry.hdr.reply_ready.store(false, Ordering::Release);
1496
1497    match entry
1498        .hdr
1499        .handle
1500        .try_recv_reply_raw(&mut entry.hdr.reply_buffer)
1501    {
1502        Ok(Some(len)) => {
1503            entry.hdr.pending = false;
1504            let mut reader = CdrReader::new_with_header(&entry.hdr.reply_buffer[..len])
1505                .map_err(|_| TransportError::DeserializationError)?;
1506            // Fully-qualify the `Deserialize` trait (mirrors the
1507            // `DeserializeBorrowed` call above): arena.rs imports
1508            // `DeserializeBorrowed` but not `Deserialize`, so the bare
1509            // `Svc::Reply::deserialize` only resolved when a default/std feature
1510            // happened to glob it into scope — under `rmw-cffi` (embedded) it
1511            // failed E0599. The fully-qualified path resolves under every feature.
1512            let reply = <Svc::Reply as nros_serdes::Deserialize>::deserialize(&mut reader)
1513                .map_err(|_| TransportError::DeserializationError)?;
1514            (entry.callback)(&reply);
1515            Ok(true)
1516        }
1517        Ok(None) => Ok(false),
1518        Err(_) => {
1519            entry.hdr.pending = false;
1520            Err(TransportError::ServiceRequestFailed)
1521        }
1522    }
1523}

1554/// Reconstruct a `GoalId` from the core's monotonically increasing counter

1558    let mut uuid = [0u8; 16];
1559    uuid[..8].copy_from_slice(&counter.to_le_bytes());

1563/// Decode one raw feedback slot (outer header + GoalId at [4..20] + inner-CDR
1564/// payload at `offset`) and invoke the typed feedback closure (Phase 239.5).
1565#[inline]
1566fn dispatch_feedback<A, F, const FEEDBACK_BUF: usize>(
1567    data: &[u8],
1568    offset: usize,
1569    on_feedback: &mut F,
1570) where
1571    A: RosAction,
1572    F: FnMut(&nros_core::GoalId, &A::Feedback),
1573{
1574    if data.len() <= offset {
1575        return;
1576    }
1577    let mut uuid = [0u8; 16];
1578    uuid.copy_from_slice(&data[4..20]);
1579    let goal_id = nros_core::GoalId { uuid };
1580    // #175 — restore the feedback's per-message encap if a typed transport
1581    // framing (Cyclone) stripped it; the message's top-level encap is `data[0..4]`.
1582    if let Some(fb) = read_action_field::<A::Feedback, FEEDBACK_BUF>(&data[0..4], &data[offset..]) {
1583        on_feedback(&goal_id, &fb);
1584    }
1585}

1587/// Monomorphized typed action-client dispatch (RFC-0041, Phase 239.2). Mirrors
1588/// [`action_client_raw_try_process`] but deserializes the feedback / result
1589/// payloads into `A::Feedback` / `A::Result` and invokes the typed closures.
1590///
1591/// # Safety
1592/// `ptr` must point to a valid, aligned `ActionClientCallbackEntry<…>`.
1593#[allow(clippy::type_complexity)]
1594pub(crate) unsafe fn action_client_callback_try_process<
1595    A,
1596    GRespF,
1597    FbF,
1598    ResF,
1599    const GOAL_BUF: usize,
1600    const RESULT_BUF: usize,
1601    const FEEDBACK_BUF: usize,
1602>(
1603    ptr: *mut u8,
1604    _delta_ms: u64,
1605) -> Result<bool, TransportError>
1606where
1607    A: RosAction,
1608    GRespF: FnMut(&nros_core::GoalId, bool),
1609    FbF: FnMut(&nros_core::GoalId, &A::Feedback),
1610    ResF: FnMut(&nros_core::GoalId, nros_core::GoalStatus, &A::Result),
1611{
1612    let entry = unsafe {
1613        &mut *(ptr as *mut ActionClientCallbackEntry<
1614            A,
1615            GRespF,
1616            FbF,
1617            ResF,
1618            GOAL_BUF,
1619            RESULT_BUF,
1620            FEEDBACK_BUF,
1621        >)
1622    };
1623    let ActionClientCallbackEntry {
1624        core,
1625        feedback_buffer,
1626        on_goal_response,
1627        on_feedback,
1628        on_result,
1629        _phantom,
1630    } = entry;
1631
1632    let mut did_work = false;
1633
1634    // 1. Goal-acceptance reply (single-outstanding → gated single buffer).
1635    if let Ok(Some(total_len)) = core.try_recv_send_goal_reply() {
1636        let accepted = total_len >= 5 && core.result_buffer[4] != 0;
1637        let goal_id = goal_id_from_counter(core.goal_counter);
1638        on_goal_response(&goal_id, accepted);
1639        did_work = true;
1640    }
1641
1642    // 2. Feedback — a stream: drain `feedback_subscriber` into the QoS-depth ring
1643    //    (Phase 239.5), then dispatch each slot. Each slot holds the raw feedback
1644    //    message: outer CDR header (4) + GoalId (16) + inner-CDR payload; see the
1645    //    raw dispatcher for the layout rationale (233.6).
1646    {
1647        const FEEDBACK_PAYLOAD_OFFSET: usize = 4 + 16;
1648        match feedback_buffer {
1649            BufferStrategy::Triple(tb) => {
1650                let slot = tb.write_slot();
1651                if let Ok(Some(len)) = core.feedback_subscriber.try_recv_raw(slot) {
1652                    tb.writer_publish(len);
1653                }
1654                if let Some((data, len)) = tb.reader_acquire() {
1655                    dispatch_feedback::<A, _, FEEDBACK_BUF>(
1656                        &data[..len],
1657                        FEEDBACK_PAYLOAD_OFFSET,
1658                        on_feedback,
1659                    );
1660                    did_work = true;
1661                }
1662            }
1663            BufferStrategy::Ring(ring) => {
1664                while let Some(slot) = ring.try_push() {
1665                    match core.feedback_subscriber.try_recv_raw(slot) {
1666                        Ok(Some(len)) => ring.commit_push(len),
1667                        _ => break,
1668                    }
1669                }
1670                while let Some((data, len)) = ring.try_pop() {
1671                    dispatch_feedback::<A, _, FEEDBACK_BUF>(
1672                        &data[..len],
1673                        FEEDBACK_PAYLOAD_OFFSET,
1674                        on_feedback,
1675                    );
1676                    ring.commit_pop();
1677                    did_work = true;
1678                }
1679            }
1680        }
1681    }
1682
1683    // 3. Result reply — status at byte 4, payload at [8 ..] (header + status +
1684    //    align pad); see the raw dispatcher (Phase 96.1).
1685    if let Ok(Some(total_len)) = core.try_recv_get_result_reply() {
1686        const RESULT_PAYLOAD_OFFSET: usize = 8;
1687        if total_len >= RESULT_PAYLOAD_OFFSET {
1688            let status = match core.result_buffer[4] {
1689                4 => nros_core::GoalStatus::Succeeded,
1690                5 => nros_core::GoalStatus::Canceled,
1691                6 => nros_core::GoalStatus::Aborted,
1692                _ => nros_core::GoalStatus::Unknown,
1693            };
1694            let goal_id = goal_id_from_counter(core.goal_counter);
1695            // #175 — restore the result's per-message encap if a typed transport
1696            // framing (Cyclone) stripped it; the reply's top-level encap is
1697            // `result_buffer[0..4]`.
1698            if let Some(res) = read_action_field::<A::Result, RESULT_BUF>(
1699                &core.result_buffer[0..4],
1700                &core.result_buffer[RESULT_PAYLOAD_OFFSET..total_len],
1701            ) {
1702                on_result(&goal_id, status, &res);
1703            }
1704        }
1705        did_work = true;
1706    }
1707
1708    Ok(did_work)
1709}

1711/// Monomorphized raw service dispatch function.
1712///
1713/// # Safety
1714/// `ptr` must point to a valid, aligned `SrvRawEntry<REQ_BUF, REPLY_BUF>`.
1715pub(crate) unsafe fn srv_raw_try_process<const REQ_BUF: usize, const REPLY_BUF: usize>(
1716    ptr: *mut u8,
1717    _delta_ms: u64,
1718) -> Result<bool, TransportError> {
1719    let entry = unsafe { &mut *(ptr as *mut SrvRawEntry<REQ_BUF, REPLY_BUF>) };
1720    let SrvRawEntry {
1721        handle,
1722        req_buffer,
1723        reply_buffer,
1724        callback,
1725        context,
1726    } = entry;
1727    let buf_start = req_buffer.as_ptr() as usize;
1728    let (data_offset, data_len, seq_num) = match handle.try_recv_request(req_buffer) {
1729        Ok(Some(request)) => {
1730            let offset = (request.data.as_ptr() as usize).saturating_sub(buf_start);
1731            let len = request.data.len();
1732            let seq = request.sequence_number;
1733            (offset, len, seq)
1734        }
1735        Ok(None) => return Ok(false),
1736        Err(_) => return Err(TransportError::ServiceReplyFailed),
1737    };
1738
1739    let mut resp_len: usize = 0;
1740    let ok = unsafe {
1741        (*callback)(
1742            req_buffer.as_ptr().add(data_offset),
1743            data_len,
1744            reply_buffer.as_mut_ptr(),
1745            REPLY_BUF,
1746            &mut resp_len,
1747            *context,
1748        )
1749    };
1750    if ok && resp_len > 0 {
1751        handle
1752            .send_reply(seq_num, &reply_buffer[..resp_len])
1753            .map_err(|_| TransportError::ServiceReplyFailed)?;
1754    }
1755    Ok(true)
1756}
