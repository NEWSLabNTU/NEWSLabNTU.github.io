1//! RFC-0052 / phase-296 W3b.4/.5 — on-target contract monitors.
2//!
3//! The baked shape mirrors Phase 211.H's `qos_overrides`: codegen emits a
4//! `&'static [MonitorSpec]` table (plus one `static PubMonitorCell` per
5//! contracted publisher) from the SystemModel's contract layer; the entry
6//! installs it on the executor before entity creation. An uncontracted
7//! image bakes an empty table — every path below dead-code-eliminates.
8//!
9//! Publish counting is an atomic bump on the publisher handle (no clock,
10//! no lock on the hot path); the rate check runs on spin ticks over a
11//! ~[`RATE_CHECK_INTERVAL_US`] window and pushes violations into a small
12//! ring the entry glue drains into the `nros-diagnostics` reporter.
13//!
14//! W3b.5 adds three more rules on the same drain:
15//! - `max-age-runtime` — subscriber take-age (`epoch_now - header.stamp`
16//!   peeked from the raw CDR buffer at [`RosMessage::STAMP_OFFSET`],
17//!   recorded into a [`SubMonitorCell`] on the take path).
18//! - `max-latency-runtime` — node-path (take → publish) latency: the
19//!   dispatch elapsed time is attributed to every monitored publisher
20//!   whose counter advanced during that dispatch (an upper bound on
21//!   take → publish, measured on the executor's monotonic clock).
22//! - `deadline-miss-runtime` — a dispatched callback ran past its bound
23//!   SchedContext's `deadline_us`; what ELSE happens is the tier's
24//!   [`DeadlineAction`](super::sched_context::DeadlineAction).
25
26use core::sync::atomic::{AtomicU32, Ordering};
27
28/// One contracted publisher's counters. Baked as a `static` by codegen
29/// (or declared by the fixture); the publisher handle bumps `count` on
30/// every publish, the executor reads deltas on spin ticks.
31#[derive(Debug, Default)]
32pub struct PubMonitorCell {
33    pub count: AtomicU32,
34    /// W3b.5 — max observed take→publish latency (µs) in the current
35    /// check window. Written by the dispatch loop (fetch_max), drained
36    /// (swap 0) by the latency check.
37    pub max_latency_us: AtomicU32,
38}
39
40impl PubMonitorCell {
41    pub const fn new() -> Self {
42        Self {
43            count: AtomicU32::new(0),
44            max_latency_us: AtomicU32::new(0),
45        }
46    }
47}

49/// One contracted subscriber's take-age accumulator (W3b.5). The take
50/// path records `epoch_now - header.stamp` per message (fetch_max); the
51/// age check drains it (swap 0) per window.
52#[derive(Debug, Default)]
53pub struct SubMonitorCell {
54    /// Max observed take-age (ms) in the current check window.
55    pub max_age_ms: AtomicU32,
56}
57
58impl SubMonitorCell {
59    pub const fn new() -> Self {
60        Self {
61            max_age_ms: AtomicU32::new(0),
62        }
63    }
64
65    /// Take-path hook: record one message's age. `stamp_us` is the
66    /// peeked `header.stamp` as µs since the UNIX epoch, `epoch_now_us`
67    /// the receive-side wall clock. A stamp from the future clamps to 0.
68    pub fn observe(&self, stamp_us: u64, epoch_now_us: u64) {
69        let age_ms = (epoch_now_us.saturating_sub(stamp_us) / 1_000).min(u32::MAX as u64) as u32;
70        self.max_age_ms.fetch_max(age_ms, Ordering::Relaxed);
71    }
72}
73
74/// Peek `Time { i32 sec; u32 nanosec }` little-endian at `offset` in a
75/// raw CDR receive buffer (encapsulation header included) and return µs
76/// since the UNIX epoch. `None` when the buffer is too short or the
77/// stamp is pre-epoch/zero (unstamped messages never fire age monitors).
78pub fn peek_stamp_us(raw: &[u8], offset: usize) -> Option<u64> {
79    let sec_b = raw.get(offset..offset + 4)?;
80    let nsec_b = raw.get(offset + 4..offset + 8)?;
81    let sec = i32::from_le_bytes([sec_b[0], sec_b[1], sec_b[2], sec_b[3]]);
82    let nsec = u32::from_le_bytes([nsec_b[0], nsec_b[1], nsec_b[2], nsec_b[3]]);
83    if sec <= 0 {
84        return None;
85    }
86    Some(sec as u64 * 1_000_000 + nsec as u64 / 1_000)
87}

89/// One monitored publisher endpoint.
90#[derive(Debug, Clone, Copy)]
91pub struct MonitorSpec {
92    /// Topic name EXACTLY as the node passes it to `create_publisher`
93    /// (the SystemModel's wiring carries the same resolved name).
94    pub topic: &'static str,
95    /// Endpoint ref for violation reports (`<node FQN>/<endpoint>` — the
96    /// SystemModel contract key).
97    pub fqn: &'static str,
98    /// Declared publisher guarantee, in milli-Hz (fixed point: Hz × 1000).
99    /// 0 = no rate contract on this endpoint.
100    pub min_rate_hz_milli: u32,
101    /// W3b.5 — node-path budget (ms) for paths whose OUTPUT is this
102    /// endpoint (`contracts.node_paths[..].max_latency_ms`). 0 = no
103    /// latency contract.
104    pub max_latency_ms: u32,
105    /// The endpoint's counter cell.
106    pub cell: &'static PubMonitorCell,
107}

109/// One monitored subscriber endpoint (W3b.5 age contracts). Separate

118    /// Declared max take-age (ms). 0 = no age contract.

120    /// The endpoint's age accumulator.

124/// Rate-check window (µs). Matches play_launch's ~5 s time-based trigger
125/// so both runtimes converge on comparable cadence.
126pub const RATE_CHECK_INTERVAL_US: u64 = 5_000_000;
127
128/// Max monitored endpoints per executor (const table, no_std).
129pub const MAX_MONITORS: usize = 8;
130/// Violation ring depth.
131pub const MAX_VIOLATIONS: usize = 8;

133/// A detected contract violation, in the play_launch rule-id vocabulary.
134#[derive(Debug, Clone)]
135pub struct Violation {
136    /// `"rate-hierarchy-runtime"` | `"max-age-runtime"` |
137    /// `"max-latency-runtime"` | `"deadline-miss-runtime"`.
138    pub rule: &'static str,
139    /// Violating endpoint ref (from the spec's `fqn`; the SC name for
140    /// deadline misses).
141    pub fqn: &'static str,
142    /// Measured value. Unit is per-rule: milli-Hz for the rate rule, ms
143    /// for age/latency, µs for deadline misses.
144    pub measured: u32,
145    /// Declared bound, same unit as `measured`.
146    pub declared: u32,
147}

149/// Per-spec accounting state (parallel to the spec table).
150#[derive(Debug, Clone, Copy, Default)]
151pub(crate) struct MonitorState {
152    /// Window opened (a plain bool, not a 0-sentinel on the timestamp —
153    /// `now_us == 0` is a legitimate first sample on freshly-started
154    /// monotonic clocks).
155    pub(crate) opened: bool,
156    pub(crate) window_start_us: u64,
157    pub(crate) count_at_window_start: u32,
158    /// Suppress duplicate reports: only re-report after a clean window.
159    pub(crate) violated_last_window: bool,
160    /// W3b.5 — separate dedup for the latency rule on the same spec row.
161    pub(crate) latency_violated_last_window: bool,
162}
163
164/// Pure rate check over one window boundary. Returns `Some(violation)`
165/// when the window elapsed AND the measured rate is below the declared
166/// minimum (and we didn't already report last window).
167///
168/// Extracted from the executor so the math is unit-testable without a
169/// session: publish counting is injected via the cell, time via `now_us`.
170pub(crate) fn check_rate(
171    spec: &MonitorSpec,
172    state: &mut MonitorState,
173    now_us: u64,
174) -> Option<Violation> {
175    if spec.min_rate_hz_milli == 0 {
176        return None;
177    }
178    let count = spec.cell.count.load(Ordering::Relaxed);
179    if !state.opened {
180        // First observation: open the window, no verdict yet.
181        state.opened = true;
182        state.window_start_us = now_us;
183        state.count_at_window_start = count;
184        return None;
185    }
186    let window_us = now_us.saturating_sub(state.window_start_us);
187    if window_us < RATE_CHECK_INTERVAL_US {
188        return None;
189    }
190    let published = count.wrapping_sub(state.count_at_window_start) as u64;
191    // milli-Hz = published * 1e3 / window_s = published * 1e9 / window_us
192    let measured_milli_hz =
193        (published.saturating_mul(1_000_000_000) / window_us.max(1)).min(u32::MAX as u64) as u32;
194
195    // Roll the window.
196    state.window_start_us = now_us;
197    state.count_at_window_start = count;
198
199    if measured_milli_hz < spec.min_rate_hz_milli {
200        if state.violated_last_window {
201            return None; // still violated — already reported
202        }
203        state.violated_last_window = true;
204        Some(Violation {
205            rule: "rate-hierarchy-runtime",
206            fqn: spec.fqn,
207            measured: measured_milli_hz,
208            declared: spec.min_rate_hz_milli,
209        })
210    } else {
211        state.violated_last_window = false;
212        None
213    }
214}

216/// Pure latency check: drains the spec cell's window-max take→publish
217/// latency and fires when it exceeds the declared node-path budget.
218/// Same report-once-until-recovery semantics as the rate rule; runs on
219/// every monitor tick (the cell accumulates between ticks, so no window
220/// bookkeeping is needed — draining IS the window roll).
221pub(crate) fn check_latency(spec: &MonitorSpec, state: &mut MonitorState) -> Option<Violation> {
222    if spec.max_latency_ms == 0 {
223        return None;
224    }
225    let max_us = spec.cell.max_latency_us.swap(0, Ordering::Relaxed);
226    let max_ms = max_us / 1_000;
227    if max_ms > spec.max_latency_ms {
228        if state.latency_violated_last_window {
229            return None;
230        }
231        state.latency_violated_last_window = true;
232        Some(Violation {
233            rule: "max-latency-runtime",
234            fqn: spec.fqn,
235            measured: max_ms,
236            declared: spec.max_latency_ms,
237        })
238    } else {
239        // A quiet window (no dispatch attributed) also counts as clean —
240        // recovery resets the dedup like the rate rule's clean window.
241        state.latency_violated_last_window = false;
242        None
243    }
244}

252/// Pure age check: drains the sub cell's window-max take-age and fires

270    } else {
271        state.violated_last_window = false;
272        None
273    }

292    #[test]
293    fn slow_publisher_fires_once_until_recovery() {
294        CELL.count.store(0, Ordering::Relaxed);
295        let s = spec(100_000); // 100 Hz declared
296        let mut st = MonitorState::default();
297
298        // t=0: opens window.
299        assert!(check_rate(&s, &mut st, 0).is_none());
300        // 5 publishes in 5 s = 1 Hz — violation.
301        CELL.count.store(5, Ordering::Relaxed);
302        let v = check_rate(&s, &mut st, RATE_CHECK_INTERVAL_US).expect("fires");
303        assert_eq!(v.rule, "rate-hierarchy-runtime");
304        assert_eq!(v.fqn, "/demo/talker/chatter");
305        assert_eq!(v.measured, 1_000);
306        assert_eq!(v.declared, 100_000);
307        // Still slow next window — suppressed (no re-report spam).
308        CELL.count.store(10, Ordering::Relaxed);
309        assert!(check_rate(&s, &mut st, 2 * RATE_CHECK_INTERVAL_US).is_none());
310        // Recovers (500 publishes in 5 s = 100 Hz) — clean window resets.
311        CELL.count.store(510, Ordering::Relaxed);
312        assert!(check_rate(&s, &mut st, 3 * RATE_CHECK_INTERVAL_US).is_none());
313        // Degrades again — fires again.
314        CELL.count.store(511, Ordering::Relaxed);
315        assert!(check_rate(&s, &mut st, 4 * RATE_CHECK_INTERVAL_US).is_some());
316    }

376    #[test]
377    fn peek_stamp_reads_le_time_and_rejects_unstamped() {
378        // Encapsulation header (4B) + sec=100 nsec=5000 at offset 4.
379        let mut raw = [0u8; 12];
380        raw[4..8].copy_from_slice(&100i32.to_le_bytes());
381        raw[8..12].copy_from_slice(&5_000u32.to_le_bytes());
382        assert_eq!(peek_stamp_us(&raw, 4), Some(100_000_005));
383        // Zero / negative sec = unstamped: no age sample.
384        assert_eq!(peek_stamp_us(&[0u8; 12], 4), None);
385        // Short buffer: no panic, no sample.
386        assert_eq!(peek_stamp_us(&raw[..8], 4), None);
387    }

390    fn slow_path_fires_latency_once_until_recovery() {
391        static C3: PubMonitorCell = PubMonitorCell::new();

393            topic: "/cmd",

413        C3.max_latency_us.store(1_000, Ordering::Relaxed);
414        assert!(check_latency(&s, &mut st).is_none());
415        C3.max_latency_us.store(30_000, Ordering::Relaxed);
