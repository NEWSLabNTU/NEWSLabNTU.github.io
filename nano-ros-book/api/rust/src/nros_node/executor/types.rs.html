6// SpinOnceResult
7// ============================================================================
8
9/// Result of a single spin iteration
10///
11/// Contains counts of how many items were processed during `spin_once()`,
12/// plus error counts for transport failures that would otherwise be silently dropped.
13#[derive(Debug, Clone, Copy, Default, PartialEq, Eq)]
14pub struct SpinOnceResult {
15    /// Number of subscription callbacks invoked
16    pub subscriptions_processed: usize,
17    /// Number of timers that fired
18    pub timers_fired: usize,
19    /// Number of service requests handled
20    pub services_handled: usize,
21    /// Number of subscription processing errors (e.g., BufferTooSmall, MessageTooLarge)
22    pub subscription_errors: usize,
23    /// Number of service processing errors (e.g., BufferTooSmall)
24    pub service_errors: usize,
25}
26
27impl SpinOnceResult {
28    /// Create a new empty result
29    pub const fn new() -> Self {
30        Self {
31            subscriptions_processed: 0,
32            timers_fired: 0,
33            services_handled: 0,
34            subscription_errors: 0,
35            service_errors: 0,
36        }
37    }
38
39    /// Check if any work was done (errors are not counted as work)
40    pub const fn any_work(&self) -> bool {
41        self.subscriptions_processed > 0 || self.timers_fired > 0 || self.services_handled > 0
42    }
43
44    /// Total number of callbacks successfully invoked (errors excluded)
45    pub const fn total(&self) -> usize {
46        self.subscriptions_processed + self.timers_fired + self.services_handled
47    }
48
49    /// Check if any errors occurred during this spin iteration
50    pub const fn any_errors(&self) -> bool {
51        self.subscription_errors > 0 || self.service_errors > 0
52    }
53
54    /// Total number of errors across all handle types
55    pub const fn total_errors(&self) -> usize {
56        self.subscription_errors + self.service_errors
57    }
58}

64/// Result from a single period of polling execution (`no_std` compatible).
65///
66/// Contains the work performed and the remaining time the caller should sleep.
67/// The caller is responsible for the actual delay (platform-specific).
68///
69/// # Example
70///
71/// ```ignore
72/// loop {
73///     let r = executor.spin_one_period(10, elapsed_ms);
74///     platform_sleep_ms(r.remaining_ms);
75/// }
76/// ```
77#[derive(Debug, Clone, Copy)]
78pub struct SpinPeriodPollingResult {
79    /// Work performed during this iteration
80    pub work: SpinOnceResult,
81    /// Remaining time in ms that the caller should sleep
82    pub remaining_ms: u64,
83}

108/// Options controlling blocking spin behavior.
109///
110/// Used with `Executor::spin_blocking`
111/// to control when the spin loop exits.
112#[derive(Debug, Clone, Default)]
113pub struct SpinOptions {
114    /// Stop after this duration (in milliseconds)
115    pub timeout_ms: Option<u64>,
116    /// Only process immediately available work (single iteration)
117    pub only_next: bool,
118    /// Stop after processing this many callbacks total
119    pub max_callbacks: Option<usize>,
120}
121
122impl SpinOptions {
123    /// Create default spin options (spin forever until halted)
124    pub const fn new() -> Self {
125        Self {
126            timeout_ms: None,
127            only_next: false,
128            max_callbacks: None,
129        }
130    }
131
132    /// Set a timeout duration
133    pub const fn timeout_ms(mut self, ms: u64) -> Self {
134        self.timeout_ms = Some(ms);
135        self
136    }
137
138    /// Only process one round of work (equivalent to spin_once)
139    pub const fn spin_once() -> Self {
140        Self {
141            timeout_ms: None,
142            only_next: true,
143            max_callbacks: None,
144        }
145    }
146
147    /// Stop after processing N callbacks
148    pub const fn max_callbacks(mut self, n: usize) -> Self {
149        self.max_callbacks = Some(n);
150        self
151    }
152}

158/// Configuration for opening an embedded executor session.
159///
160/// Provides a backend-agnostic builder for configuring the middleware
161/// connection. The active Cargo feature (`rmw-zenoh`, `rmw-xrce`, or
162/// `rmw-cffi`) determines which backend is used.
163///
164/// # Example
165///
166/// ```ignore
167/// use nros::prelude::*;
168///
169/// let config = ExecutorConfig::new("tcp/127.0.0.1:7447")
170///     .node_name("talker")
171///     .domain_id(0);
172/// let mut executor: Executor = Executor::open(&config)?;
173/// ```
174pub struct ExecutorConfig<'a> {
175    /// Middleware-specific connection string.
176    pub locator: &'a str,
177    /// Session mode (client or peer).
178    pub mode: SessionMode,
179    /// ROS 2 domain ID.
180    pub domain_id: u32,
181    /// Node name.
182    pub node_name: &'a str,
183    /// Node namespace.
184    pub namespace: &'a str,
185}

243#[cfg(feature = "std")]
244struct EnvCache {
245    locator: std::string::String,
246    domain_id: u32,
247    mode: SessionMode,
248}
249
250#[cfg(feature = "std")]
251static ENV_CACHE: std::sync::OnceLock<EnvCache> = std::sync::OnceLock::new();
252
253#[cfg(feature = "std")]
254fn env_cache() -> &'static EnvCache {
255    ENV_CACHE.get_or_init(|| {
256        // Prefer NROS_LOCATOR / NROS_SESSION_MODE; accept legacy ZENOH_*
257        // names with a stderr deprecation warning.
258        let locator = std::env::var("NROS_LOCATOR")
259            .or_else(|_| {
260                std::env::var("ZENOH_LOCATOR").inspect(|_| {
261                    std::eprintln!("nros: ZENOH_LOCATOR is deprecated; use NROS_LOCATOR instead");
262                })
263            })
264            .unwrap_or_else(|_| std::string::String::from("tcp/127.0.0.1:7447"));
265        let domain_id = std::env::var("ROS_DOMAIN_ID")
266            .ok()
267            .and_then(|s| s.parse().ok())
268            .unwrap_or(0);
269        let mode_str = std::env::var("NROS_SESSION_MODE")
270            .or_else(|_| {
271                std::env::var("ZENOH_MODE").inspect(|_| {
272                    std::eprintln!("nros: ZENOH_MODE is deprecated; use NROS_SESSION_MODE instead");
273                })
274            })
275            .ok();
276        let mode = match mode_str.as_deref() {
277            Some("peer") => SessionMode::Peer,
278            _ => SessionMode::Client,
279        };
280        EnvCache {
281            locator,
282            domain_id,
283            mode,
284        }
285    })
286}

288#[cfg(feature = "std")]
289impl ExecutorConfig<'static> {
290    /// Create a configuration from environment variables.
291    ///
292    /// Reads:
293    /// - `NROS_LOCATOR` — Middleware locator (default: `"tcp/127.0.0.1:7447"`).
294    ///   Legacy name `ZENOH_LOCATOR` is accepted with a deprecation warning.
295    /// - `ROS_DOMAIN_ID` — ROS 2 domain ID (default: `0`).
296    /// - `NROS_SESSION_MODE` — Session mode: `"client"` or `"peer"` (default:
297    ///   `"client"`). Legacy name `ZENOH_MODE` is accepted with a deprecation warning.
298    ///
299    /// Env-var values are cached in a process-global `OnceLock` on the
300    /// first call and reused for the process lifetime — repeated calls
301    /// do NOT re-read the environment and do NOT accrete memory. The
302    /// returned `&'static str` fields point into the cache.
303    pub fn from_env() -> Self {
304        let cache = env_cache();
305        Self {
306            locator: cache.locator.as_str(),
307            mode: cache.mode,
308            domain_id: cache.domain_id,
309            node_name: "node",
310            namespace: "",
311        }
312    }
313}

319/// Error type for generic embedded node operations.
320///
321/// Not `Copy` — `NodeError::Transport` wraps a [`TransportError`] which
322/// carries owned diagnostic strings (`Backend` / `BackendDynamic`). Rust
323/// callers that matched on `NodeError` by value may need `ref` arms or
324/// `.clone()`; C/C++ callers are unaffected (they see an integer
325/// `nros_ret_t`).
326#[derive(Debug, Clone, PartialEq, Eq)]
327pub enum NodeError {
328    /// Transport-level error.
329    Transport(TransportError),
330    /// Node name exceeds 64 bytes.
331    NameTooLong,
332    /// CDR serialization failed.
333    Serialization,
334    /// CDR deserialization failed.
335    Deserialization,
336    /// Buffer too small for message.
337    BufferTooSmall,
338    /// Action server/client creation failed.
339    ActionCreationFailed,
340    /// Service request failed.
341    ServiceRequestFailed,
342    /// Service reply failed.
343    ServiceReplyFailed,
344    /// Operation timed out.
345    Timeout,
346    /// A required subsystem has not been initialized (e.g. parameter
347    /// services have not been registered on the executor).
348    NotInitialized,
349    /// The client / action client already has a request in flight that
350    /// hasn't been consumed. Phase 84.D3: fixes the hazard where dropping
351    /// a `Promise` without awaiting its reply left the stale reply queued
352    /// to be delivered to the *next* call. Resolve by either polling the
353    /// existing promise to completion or calling `reset_in_flight()`.
354    RequestInFlight,
355    /// Phase 110.B — `create_sched_context` ran out of slots
356    /// (`MAX_SC` exceeded).
357    NoSchedContextSlot,
358    /// Phase 110.B — `bind_handle_to_sched_context` was called with
359    /// an out-of-range handle, an empty entry slot, or an unknown
360    /// `SchedContextId`.
361    InvalidSchedContextBinding,
362    /// Phase 104.C.2 — `Executor::node_builder(...).build()` was
363    /// called when the per-Executor node table is full
364    /// (`NROS_EXECUTOR_MAX_NODES` reached).
365    NodeTableFull,
366    /// Phase 104.C.2 — requested RMW backend does not match the
367    /// Executor's open session (single-session restriction lifts in
368    /// Phase 104.C.3 when the per-Node session cache lands).
369    BackendMismatch,
370}

375    }
376}

398/// Index into the executor's `entries[]` array. Phase 110.A caps at
399/// 64 to match the existing readiness bitmap width; if a future
400/// MAX_HANDLES bump goes past 64 the type widens accordingly.
401#[cfg(any(has_rmw, test))]
402pub(crate) type DescIdx = u8;
403
404/// Sort key used to order callbacks within a `ReadySet`.
405///
406/// Phase 110.A: registration-order — `sort_key` mirrors `desc_idx`
407/// numerically so `FifoReadySet` preserves bit-for-bit dispatch order.
408/// Phase 110.B will widen this to encode an EDF deadline ahead of
409/// `desc_idx`.
410#[cfg(any(has_rmw, test))]
411#[allow(dead_code)] // Phase 110.A — wired in 110.A.b spin_once rewire.
412pub(crate) type SortKey = u32;
413
414/// One ready callback queued for dispatch. Stored in the `ReadySet`,
415/// consumed by the `Dispatcher`. Full handle metadata (callback fn,
416/// data offset, kind) is reconstructed from
417/// `Executor::entries[desc_idx]` at dispatch time so the ready set
418/// itself stays compact.
419#[cfg(any(has_rmw, test))]
420#[allow(dead_code)] // Phase 110.A — wired in 110.A.b spin_once rewire.
421#[derive(Debug, Clone, Copy, Eq, PartialEq)]
422pub(crate) struct ActiveJob {
423    pub sort_key: SortKey,
424    pub desc_idx: DescIdx,
425}
426
427/// How aggressively the dispatcher drains the `ReadySet`.
428///
429/// `Latched` (default) preserves today's `spin_once` semantics:
430/// callbacks that become ready *during* dispatch wait for the next
431/// cycle. `Greedy` re-runs the activator after each callback so newly
432/// ready entries fire in the same cycle — soft-RT pipelines that want
433/// chain-style propagation use this.
434#[cfg(any(has_rmw, test))]
435#[allow(dead_code)] // Phase 110.B introduces the user-facing knob.
436#[derive(Debug, Clone, Copy, Eq, PartialEq, Default)]
437pub(crate) enum DrainMode {
438    #[default]
439    Latched,
440    Greedy,
441}

454// ============================================================================
455// HandleSet
456// ============================================================================
457
458/// A set of handle IDs, represented as a bitset.
459///
460/// Supports up to 64 handles. Construct via `HandleId` operators:
461/// ```ignore
462/// let set = imu | gps | lidar;  // HandleSet from 3 handles
463/// ```
464#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
465pub struct HandleSet(pub(crate) u64);
466
467impl HandleSet {
468    /// Empty set.
469    pub const EMPTY: Self = Self(0);
470
471    /// Insert a handle into the set.
472    pub const fn insert(self, id: HandleId) -> Self {
473        Self(self.0 | (1u64 << id.0))
474    }
475
476    /// Check if the set contains a handle.
477    pub const fn contains(self, id: HandleId) -> bool {
478        self.0 & (1u64 << id.0) != 0
479    }
480
481    /// Union of two sets.
482    pub const fn union(self, other: Self) -> Self {
483        Self(self.0 | other.0)
484    }
485
486    /// Number of handles in the set.
487    pub const fn len(self) -> u32 {
488        self.0.count_ones()
489    }
490
491    /// Check if the set is empty.
492    pub const fn is_empty(self) -> bool {
493        self.0 == 0
494    }
495}
496
497impl core::ops::BitOr for HandleId {
498    type Output = HandleSet;
499    fn bitor(self, rhs: HandleId) -> HandleSet {
500        HandleSet::EMPTY.insert(self).insert(rhs)
501    }
502}
503
504impl core::ops::BitOr<HandleId> for HandleSet {
505    type Output = HandleSet;
506    fn bitor(self, rhs: HandleId) -> HandleSet {
507        self.insert(rhs)
508    }
509}
510
511impl core::ops::BitOr for HandleSet {
512    type Output = HandleSet;
513    fn bitor(self, rhs: HandleSet) -> HandleSet {
514        self.union(rhs)
515    }
516}

522/// Snapshot of handle readiness at the start of a spin iteration.
523///
524/// Passed to [`Trigger::Predicate`] functions. Query by [`HandleId`].
525pub struct ReadinessSnapshot {
526    pub(crate) bits: u64,
527    pub(crate) count: usize,
528}
529
530impl ReadinessSnapshot {
531    /// Check if a specific handle has data.
532    pub const fn is_ready(&self, id: HandleId) -> bool {
533        self.bits & (1u64 << id.0) != 0
534    }
535
536    /// Check if all handles in the set have data.
537    pub const fn all_ready(&self, set: HandleSet) -> bool {
538        self.bits & set.0 == set.0
539    }
540
541    /// Check if any handle in the set has data.
542    pub const fn any_ready(&self, set: HandleSet) -> bool {
543        self.bits & set.0 != 0
544    }
545
546    /// Number of handles that have data.
547    pub const fn ready_count(&self) -> u32 {
548        self.bits.count_ones()
549    }
550
551    /// Total registered handles.
552    pub const fn total(&self) -> usize {
553        self.count
554    }
555}

574// ============================================================================
575// Trigger
576// ============================================================================
577
578/// Executor-level trigger condition.
579///
580/// Controls when the executor dispatches callbacks during `spin_once()`.
581/// The trigger is evaluated after polling the transport but before any
582/// callback dispatch.
583#[derive(Clone, Copy, Default)]
584pub enum Trigger {
585    /// Fire if any registered handle has data (default).
586    #[default]
587    Any,
588    /// Fire only when ALL non-timer handles have data.
589    All,
590    /// Fire only when a specific handle has data.
591    One(HandleId),
592    /// Fire only when every handle in the set has data.
593    AllOf(HandleSet),
594    /// Fire when any handle in the set has data.
595    AnyOf(HandleSet),
596    /// Always fire, regardless of data availability.
597    Always,
598    /// Custom predicate over a readiness snapshot.
599    Predicate(fn(&ReadinessSnapshot) -> bool),
600    /// Custom predicate with C-compatible signature and context pointer.
601    ///
602    /// The callback receives a `bool` array of readiness flags (one per handle),
603    /// the count of handles, and a user-provided context pointer.
604    /// Used by the C API to bridge `nros_executor_trigger_t` to the Rust trigger system.
605    RawPredicate {
606        /// C trigger callback
607        callback: unsafe extern "C" fn(
608            ready: *const bool,
609            count: usize,
610            context: *mut core::ffi::c_void,
611        ) -> bool,
612        /// User-provided context pointer passed to the callback
613        context: *mut core::ffi::c_void,
614    },
615}
616
617// Manual Debug impl because fn pointers don't impl Debug well
618impl core::fmt::Debug for Trigger {
619    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
620        match self {
621            Self::Any => write!(f, "Any"),
622            Self::All => write!(f, "All"),
623            Self::One(id) => f.debug_tuple("One").field(id).finish(),
624            Self::AllOf(set) => f.debug_tuple("AllOf").field(set).finish(),
625            Self::AnyOf(set) => f.debug_tuple("AnyOf").field(set).finish(),
626            Self::Always => write!(f, "Always"),
627            Self::Predicate(_) => write!(f, "Predicate(...)"),
628            Self::RawPredicate { .. } => write!(f, "RawPredicate(...)"),
629        }
630    }
631}

772// GuardConditionHandle
773// ============================================================================
774
775/// Handle for triggering a guard condition from outside the executor.
776///
777/// Obtained from `Executor::register_guard_condition`.
778/// Safe to use from any thread — the inner `&'static AtomicBool` is inherently
779/// `Send + Sync`.
780pub struct GuardConditionHandle {
781    // The AtomicBool lives in the executor's arena, which is never moved or
782    // deallocated while handles exist. The 'static lifetime is asserted at
783    // construction time (see `new()`).
784    flag: &'static portable_atomic::AtomicBool,
785    /// Phase 124.B.5 — runtime wake callback. On std + rmw-cffi
786    /// builds the executor sets this to `nros_rmw_runtime_wake_cb`
787    /// with `ctx` pointing at the executor's WakeCtx. `trigger`
788    /// invokes it after writing the arena flag so a `spin_once`
789    /// blocked on `wake_cv` resumes immediately (sub-poll wake
790    /// latency). Bare/no-std builds leave it `None` — the arena
791    /// flag is observed on the next spin iteration as before.
792    wake_cb: Option<unsafe extern "C" fn(ctx: *mut core::ffi::c_void)>,
793    wake_ctx: *mut core::ffi::c_void,
794}
795
796// SAFETY: `wake_cb` is a plain function pointer; `wake_ctx` points
797// at a WakeCtx Arc allocated on Executor::new and never freed before
798// Executor::drop. Both are safe to share across threads.
799unsafe impl Send for GuardConditionHandle {}
800unsafe impl Sync for GuardConditionHandle {}
801
802impl GuardConditionHandle {
803    /// Create a handle from a raw pointer to an arena-allocated `AtomicBool`.
804    ///
805    /// # Safety
806    ///
807    /// The pointed-to `AtomicBool` must outlive this handle. This is guaranteed
808    /// when the `AtomicBool` lives in the executor arena (which is never moved
809    /// or deallocated while handles exist).
810    #[cfg(any(has_rmw, test))]
811    pub(crate) unsafe fn new(flag: *const portable_atomic::AtomicBool) -> Self {
812        // SAFETY: Caller guarantees the AtomicBool outlives this handle.
813        Self {
814            flag: unsafe { &*flag },
815            wake_cb: None,
816            wake_ctx: core::ptr::null_mut(),
817        }
818    }
819
820    /// Phase 124.B.5 — install the executor's wake callback. Called
821    /// once at handle creation; the executor passes its
822    /// `nros_rmw_runtime_wake_cb` + WakeCtx pointer here so
823    /// `trigger()` can signal the wake condvar from any thread / ISR.
824    #[cfg(any(all(feature = "std", feature = "rmw-cffi"), test))]
825    #[allow(dead_code)] // Wired by register_guard_condition under cfg.
826    pub(crate) fn set_wake_cb(
827        &mut self,
828        cb: unsafe extern "C" fn(ctx: *mut core::ffi::c_void),
829        ctx: *mut core::ffi::c_void,
830    ) {
831        self.wake_cb = Some(cb);
832        self.wake_ctx = ctx;
833    }
834
835    /// Trigger the guard condition.
836    ///
837    /// The executor will invoke the associated callback on the next spin iteration.
838    /// On std + rmw-cffi builds, `trigger` also signals the executor's
839    /// wake condvar so a blocked `spin_once` resumes immediately.
840    pub fn trigger(&self) {
841        self.flag.store(true, portable_atomic::Ordering::Release);
842        if let Some(cb) = self.wake_cb {
843            // SAFETY: cb + ctx installed by Executor::register_guard_condition;
844            // ctx points at WakeCtx valid for Executor's lifetime.
845            unsafe { cb(self.wake_ctx) };
846        }
847    }
848}
