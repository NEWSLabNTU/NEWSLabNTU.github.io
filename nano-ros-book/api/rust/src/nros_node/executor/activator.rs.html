1//! Phase 110.A — `Activator` trait.
2//!
3//! The activator decides *what* should fire this cycle. It is invoked
4//! after `Session::drive_io` returns, walks the executor's `entries[]`
5//! looking at `meta.has_data`, evaluates the configured `Trigger`,
6//! and writes the resulting jobs into a [`ReadySet`](super::ready_set::ReadySet).
7//!
8//! 110.A defines the trait; 110.A.b rewires `spin_once` to drive
9//! activation through this trait instead of the inline bitmap scan.
10//! The default impl reproduces the pre-refactor scan + trigger logic
11//! bit-for-bit.
12
13use super::ready_set::ReadySet;
14
15/// Context handed to [`Activator::scan`]. Phase 110.A keeps this
16/// opaque — the default activator impl borrows the executor's
17/// `entries[]`, `arena_ptr`, and `trigger` directly. 110.A.b widens
18/// this once the inline scan moves into the trait method.
19#[allow(dead_code)] // Phase 110.A — wired in 110.A.b spin_once rewire.
20pub(crate) struct ActivatorCtx<'a> {
21    _phantom: core::marker::PhantomData<&'a ()>,
22}
23
24#[allow(dead_code)] // Phase 110.A — wired in 110.A.b spin_once rewire.
25pub(crate) trait Activator {
26    /// Walk the executor's entries, evaluate the trigger, and insert
27    /// every callback that should fire into `ready`. Idempotent on
28    /// already-set entries (`ReadySet::insert` is itself idempotent).
29    fn scan<R: ReadySet>(&self, ctx: &ActivatorCtx<'_>, ready: &mut R);
30}
