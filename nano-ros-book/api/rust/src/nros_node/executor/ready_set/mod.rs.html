1//! Phase 110.A — `ReadySet` trait + default `FifoReadySet` impl.
2//!
3//! `ReadySet` abstracts the queue + selection layer between
4//! [`Activator`](super::activator::Activator) and
5//! [`Dispatcher`](super::dispatcher::Dispatcher).
6//!
7//! 110.A ships only `FifoReadySet` (registration-order, bit-for-bit
8//! parity with the pre-refactor `spin_once`). Phase 110.B adds
9//! `EdfReadySet`; phase 110.C adds the bucketed variants.
10//!
11//! ## Invariants
12//!
13//! - `insert` is **idempotent** — a second insert for an already-ready
14//!   `desc_idx` is a no-op. This matches default ROS 2 behavior:
15//!   one ready bit per callback regardless of how many messages are
16//!   queued at the rmw layer; the callback drains its rmw queue per
17//!   QoS depth itself.
18//! - `pop_next` removes the lowest-key entry and returns it.
19//! - `clear` empties the set in O(1).
20//!
21//! ## Capacity
22//!
23//! The const-generic `N` caps the number of distinct `DescIdx` values
24//! the set can track. Phase 110.A holds N = 64 to match the existing
25//! `u64` readiness bitmap width; future MAX_HANDLES bumps will widen
26//! the storage accordingly (likely `BitSet<N>`).
27
28use super::types::{ActiveJob, DescIdx};

30/// Capacity-overflow error returned from [`ReadySet::insert`].
31#[derive(Debug, Clone, Copy, PartialEq, Eq)]
32pub(crate) struct Overflow;
33
34// Phase 192.8 — the bucketed dispatch (`spin_once`) drives the sets through
35// `insert` + `pop_next` only; `clear` / `is_empty` / `contains` are exercised
36// solely by this module's unit tests (the `scheduler-os-priority` stub that
37// also delegates them is itself dead), so they are `#[cfg(test)]` rather than
38// kept alive with a blanket `#[allow(dead_code)]`.
39pub(crate) trait ReadySet {
40    /// Insert a job. Idempotent: a second insert for the same
41    /// `desc_idx` returns `Ok(())` without changing internal state.
42    fn insert(&mut self, job: ActiveJob) -> Result<(), Overflow>;
43    /// Pop the lowest-key job. Returns `None` when the set is empty.
44    fn pop_next(&mut self) -> Option<ActiveJob>;
45    #[cfg(test)]
46    fn clear(&mut self);
47    #[cfg(test)]
48    fn is_empty(&self) -> bool;
49    #[cfg(test)]
50    fn contains(&self, desc_idx: DescIdx) -> bool;
51}

53/// FIFO ready set backed by a 64-bit presence bitmap.
54///
55/// Selection order is the registration order of `desc_idx` (lowest
56/// bit first), which reproduces the pre-110.A `spin_once` behavior
57/// exactly.
58#[derive(Debug)]
59pub(crate) struct FifoReadySet<const N: usize> {
60    bits: u64,
61}
62
63impl<const N: usize> Default for FifoReadySet<N> {
64    fn default() -> Self {
65        Self::new()
66    }
67}
68
69impl<const N: usize> FifoReadySet<N> {
70    pub const fn new() -> Self {
71        // Capacity guard — 110.A bitmap width matches the existing
72        // executor scan. Wider sets land with the BitSet rewrite.
73        const {
74            assert!(N <= 64, "FifoReadySet capacity capped at 64");
75        }
76        Self { bits: 0 }
77    }
78
79    /// Bulk-set the presence bitmap. Test-only helper (Phase 192.8): the
80    /// runtime path inserts per-bit; only the unit tests set the bitmap in
81    /// one shot.
82    #[cfg(test)]
83    pub fn set_bits(&mut self, bits: u64) {
84        self.bits = bits;
85    }
86}
87
88impl<const N: usize> ReadySet for FifoReadySet<N> {
89    #[cfg(test)]
90    fn clear(&mut self) {
91        self.bits = 0;
92    }
93
94    #[cfg(test)]
95    fn is_empty(&self) -> bool {
96        self.bits == 0
97    }
98
99    fn insert(&mut self, job: ActiveJob) -> Result<(), Overflow> {
100        let idx = job.desc_idx as usize;
101        if idx >= N {
102            return Err(Overflow);
103        }
104        self.bits |= 1u64 << idx;
105        Ok(())
106    }
107
108    fn pop_next(&mut self) -> Option<ActiveJob> {
109        if self.bits == 0 {
110            return None;
111        }
112        let idx = self.bits.trailing_zeros() as DescIdx;
113        self.bits &= !(1u64 << idx);
114        Some(ActiveJob {
115            sort_key: idx as u32,
116            desc_idx: idx,
117        })
118    }
119
120    #[cfg(test)]
121    fn contains(&self, desc_idx: DescIdx) -> bool {
122        let idx = desc_idx as usize;
123        if idx >= 64 {
124            return false;
125        }
126        self.bits & (1u64 << idx) != 0
127    }
128}

137// =============================================================================
138// Phase 110.B.a — `EdfReadySet<N>`
139// =============================================================================

141/// Earliest-Deadline-First ready set backed by a min-heap on
142/// `(deadline_us, desc_idx)`.
143///
144/// Each insert places the job at its deadline-ordered slot;
145/// `pop_next` yields the earliest deadline first. Insert is
146/// idempotent on `desc_idx` — a second insert for an entry already
147/// in the heap returns `Ok(())` without enqueueing a duplicate
148/// (matches the FIFO contract; one ready bit per callback regardless
149/// of message count).
150///
151/// Capacity `N` caps the number of distinct entries; tracked via a
152/// presence bitmap in addition to the heap so duplicate-detection is
153/// O(1).
154///
155/// Phase 110.B.a (this commit) exposes the type + tests; 110.B.b
156/// wires it through `spin_once` once `SchedContext::Edf` callbacks
157/// can be bound on Executor.
158#[derive(Debug)]
159pub(crate) struct EdfReadySet<const N: usize> {
160    /// Min-heap on `(deadline_us, desc_idx)`. We keep the heap as a
161    /// fixed array + length; insertions sift up, pops sift down.
162    heap: [(u32, super::types::DescIdx); N],
163    len: usize,
164    /// Presence bitmap for O(1) idempotency check on `insert`.
165    present: u64,
166}
167
168impl<const N: usize> EdfReadySet<N> {
169    pub const fn new() -> Self {
170        const {
171            assert!(N <= 64, "EdfReadySet presence bitmap caps at 64");
172        }
173        Self {
174            heap: [(0, 0); N],
175            len: 0,
176            present: 0,
177        }
178    }
179
180    fn sift_up(&mut self, mut i: usize) {
181        while i > 0 {
182            let parent = (i - 1) / 2;
183            if self.heap[i].0 < self.heap[parent].0 {
184                self.heap.swap(i, parent);
185                i = parent;
186            } else {
187                return;
188            }
189        }
190    }
191
192    fn sift_down(&mut self, mut i: usize) {
193        loop {
194            let left = 2 * i + 1;
195            let right = 2 * i + 2;
196            let mut smallest = i;
197            if left < self.len && self.heap[left].0 < self.heap[smallest].0 {
198                smallest = left;
199            }
200            if right < self.len && self.heap[right].0 < self.heap[smallest].0 {
201                smallest = right;
202            }
203            if smallest == i {
204                return;
205            }
206            self.heap.swap(i, smallest);
207            i = smallest;
208        }
209    }
210}
211
212impl<const N: usize> Default for EdfReadySet<N> {
213    fn default() -> Self {
214        Self::new()
215    }
216}

218impl<const N: usize> ReadySet for EdfReadySet<N> {
219    #[cfg(test)]
220    fn clear(&mut self) {
221        self.len = 0;
222        self.present = 0;
223    }
224
225    #[cfg(test)]
226    fn is_empty(&self) -> bool {
227        self.len == 0
228    }
229
230    fn insert(&mut self, job: super::types::ActiveJob) -> Result<(), Overflow> {
231        let idx = job.desc_idx as usize;
232        if idx >= N || self.len >= N {
233            return Err(Overflow);
234        }
235        // Idempotent: skip when already in the heap.
236        if self.present & (1u64 << idx) != 0 {
237            return Ok(());
238        }
239        self.heap[self.len] = (job.sort_key, job.desc_idx);
240        self.len += 1;
241        self.present |= 1u64 << idx;
242        self.sift_up(self.len - 1);
243        Ok(())
244    }
245
246    fn pop_next(&mut self) -> Option<super::types::ActiveJob> {
247        if self.len == 0 {
248            return None;
249        }
250        let (key, desc_idx) = self.heap[0];
251        self.len -= 1;
252        if self.len > 0 {
253            self.heap[0] = self.heap[self.len];
254            self.sift_down(0);
255        }
256        self.present &= !(1u64 << desc_idx as usize);
257        Some(super::types::ActiveJob {
258            sort_key: key,
259            desc_idx,
260        })
261    }
262
263    #[cfg(test)]
264    fn contains(&self, desc_idx: super::types::DescIdx) -> bool {
265        let idx = desc_idx as usize;
266        if idx >= 64 {
267            return false;
268        }
269        self.present & (1u64 << idx) != 0
270    }
271}

273// =============================================================================
274// Phase 110.C — `BucketedFifoSet<NB, N>` + `BucketedEdfSet<NB, N>`
275// =============================================================================
276//
277// Splits a ready set across `NB` priority buckets (defaults to 3:
278// Critical / Normal / BestEffort). Within each bucket the underlying
279// FIFO bitmap or EDF heap behaves exactly as the un-bucketed variant.
280//
281// Selection: `pop_next` walks buckets ascending (lowest index first =
282// highest priority) and returns the first available job. This gives
283// strict static priority across buckets while preserving FIFO /
284// EDF ordering inside a bucket.
285//
286// Single-thread blocking note: an in-flight callback in a lower-
287// priority bucket blocks higher-priority work that becomes ready
288// during dispatch. For hard-RT preemption see Phase 110.D.
289
290#[derive(Debug)]
291pub(crate) struct BucketedFifoSet<const NB: usize, const N: usize> {
292    buckets: [FifoReadySet<N>; NB],
293}
294
295impl<const NB: usize, const N: usize> BucketedFifoSet<NB, N> {
296    pub const fn new() -> Self {
297        const {
298            assert!(NB >= 1, "BucketedFifoSet needs at least 1 bucket");
299        }
300        Self {
301            buckets: [const { FifoReadySet::<N>::new() }; NB],
302        }
303    }
304
305    pub fn insert_into(
306        &mut self,
307        bucket: usize,
308        job: super::types::ActiveJob,
309    ) -> Result<(), Overflow> {
310        if bucket >= NB {
311            return Err(Overflow);
312        }
313        self.buckets[bucket].insert(job)
314    }
315
316    /// Pop a job from a specific bucket, ignoring other buckets. The
317    /// dispatcher walks buckets in priority order itself (see `spin_once`),
318    /// so the bucketed set only exposes per-bucket draining.
319    pub fn pop_from(&mut self, bucket: usize) -> Option<super::types::ActiveJob> {
320        self.buckets.get_mut(bucket)?.pop_next()
321    }
322}
323
324impl<const NB: usize, const N: usize> Default for BucketedFifoSet<NB, N> {
325    fn default() -> Self {
326        Self::new()
327    }
328}
