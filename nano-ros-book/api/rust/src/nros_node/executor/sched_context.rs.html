1//! Phase 110.B — `SchedContext` API + supporting types.
2//!
3//! A `SchedContext` is a first-class scheduling capability. Multiple
4//! callbacks share one SC; one OS priority slot per Executor regardless
5//! of callback count. Inspired by seL4 MCS (Mixed-Criticality
6//! Scheduling).
7//!
8//! 110.B.a (this commit) lands the type surface + `EdfReadySet`. The
9//! Executor builder methods (`create_sched_context`,
10//! `register_subscription_in`, ...) and the cbindgen / C / C++ wrappers
11//! land in 110.B.b once the const-generic `Executor<MAX_HANDLES,
12//! MAX_SC>` reshape is sorted.
13
14use core::num::NonZeroU32;

16/// Optional time field with a sentinel `0` for "absent".
17///
18/// Phase 110.B keeps a stable `#[repr(transparent)]` u32 layout so
19/// cbindgen emits plain `uint32_t` for C consumers — `Option<NonZeroU32>`
20/// loses its niche optimization the moment a `#[repr(C)]` struct
21/// embeds it. Rust callers see the ergonomic
22/// [`get`](OptUs::get)-returning-`Option<NonZeroU32>` getter.
23///
24/// Sentinel `0` is physically meaningful for every time field on
25/// [`SchedContext`]: 0-period would mean infinite frequency, 0-budget
26/// means unbounded, 0-deadline means no deadline.
27#[allow(dead_code)] // Phase 110.B.a — wired in 110.B.b builder/dispatch.
28#[derive(Debug, Clone, Copy, Default, PartialEq, Eq)]
29#[repr(transparent)]
30pub struct OptUs(u32);
31
32#[allow(dead_code)] // Phase 110.B.a — wired in 110.B.b builder/dispatch.
33impl OptUs {
34    pub const NONE: Self = Self(0);
35
36    pub const fn from_us(us: u32) -> Self {
37        Self(us)
38    }
39
40    pub const fn from_nz(nz: NonZeroU32) -> Self {
41        Self(nz.get())
42    }
43
44    /// Returns the inner value or `None` when the sentinel is set.
45    pub const fn get(self) -> Option<NonZeroU32> {
46        NonZeroU32::new(self.0)
47    }
48
49    pub const fn is_some(self) -> bool {
50        self.0 != 0
51    }
52
53    pub const fn raw(self) -> u32 {
54        self.0
55    }
56}

58/// Scheduling class — picks the runtime queue + selection policy for
59/// the contained callbacks.
60///
61/// Phase 110.A only exercises `Fifo`; `Edf` lands with the
62/// `EdfReadySet` plumb-up in 110.B.b; `Sporadic` is post-v1 (110.E);
63/// `TimeTriggered` is post-v1 (110.G).
64#[allow(dead_code)] // Phase 110.B.a — wired in 110.B.b builder/dispatch.
65#[derive(Debug, Clone, Copy, Default, PartialEq, Eq)]
66pub enum SchedClass {
67    #[default]
68    Fifo,
69    Edf,
70    Sporadic,
71    BestEffort,
72    /// Deprecated as of Phase 110.G refactor — TT is now an
73    /// orthogonal slot-membership annotation via
74    /// `SchedContext.tt_window_offset_us` /
75    /// `tt_window_duration_us`, not a class. Keeping the variant
76    /// for one release so exhaustive matches don't break; treated
77    /// as `Fifo` in dispatch.
78    #[deprecated(
79        since = "0.1.0",
80        note = "use SchedContext.tt_window_offset_us + tt_window_duration_us instead; \
81                TT now cooperates with Fifo / Edf / Sporadic / BestEffort classes"
82    )]
83    TimeTriggered,
84}

86/// Criticality bucket for [`SchedContext`]. Phase 110.C uses this to
87/// pick which `BucketedFifoSet` / `BucketedEdfSet` slot a callback
88/// dispatches through; later phases (110.D) map it to OS priority.
89///
90/// Default `Normal` keeps existing single-bucket workloads unchanged
91/// — every default-Fifo SC sits in `Normal`, so dispatch order is
92/// bit-identical to pre-110.C when no callback opts in to `Critical`
93/// or `BestEffort`.
94///
95/// Single-thread non-preemption note: a `BestEffort` callback already
96/// running blocks `Critical` work that becomes ready mid-cycle. Hard-
97/// RT scenarios need 110.D's multi-executor preemption.
98#[allow(dead_code)] // Phase 110.C — wired in spin_once bucketed dispatch.
99#[derive(Debug, Clone, Copy, Default, PartialEq, Eq, Ord, PartialOrd)]
100pub enum Priority {
101    /// Highest-priority bucket. Drained first within a single
102    /// `spin_once` cycle; non-preemptive against in-flight lower-
103    /// priority callbacks (see Phase 110.D for preemption).
104    Critical = 0,
105    /// Default bucket. Most callbacks (and the auto-default Fifo SC)
106    /// live here.
107    #[default]
108    Normal = 1,
109    /// Lowest-priority bucket. Drained last; first to be skipped if a
110    /// future cycle-budget overrun forces an early return.
111    BestEffort = 2,
112}
113
114#[allow(dead_code)] // Phase 110.C — wired in spin_once bucketed dispatch.
115impl Priority {
116    pub const COUNT: usize = 3;
117
118    pub const fn index(self) -> usize {
119        self as usize
120    }
121}

123/// How an EDF deadline is interpreted relative to a callback firing.
124///
125/// - `Released`: deadline is `release_time + period`. Default for
126///   timer-triggered callbacks.
127/// - `Activated`: deadline is `activation_time + relative_deadline`.
128///   Default for event-triggered subscriptions.
129/// - `Inherited`: deadline travels in the message header — latency-
130///   aware pipelines extract it per-message at dispatch time.
131#[allow(dead_code)] // Phase 110.B.a — wired in 110.B.b builder/dispatch.
132#[derive(Debug, Clone, Copy, Default, PartialEq, Eq)]
133pub enum DeadlinePolicy {
134    Released,
135    #[default]
136    Activated,
137    Inherited,
138}

147/// First-class scheduling capability — one SC per scheduling concern,
148/// shared by every callback that should run under the same budget /
149/// period / deadline / class.
150///
151/// Phase 110.B.a defines the shape; 110.B.b's builder methods on
152/// Executor consume it.
153#[allow(dead_code)] // Phase 110.B.a — wired in 110.B.b builder/dispatch.
154#[derive(Debug, Clone, Copy, Default)]
155pub struct SchedContext {
156    pub class: SchedClass,
157    pub priority: Priority,
158    pub period_us: OptUs,
159    pub budget_us: OptUs,
160    pub deadline_us: OptUs,
161    pub deadline_policy: DeadlinePolicy,
162    /// Phase 110.F — opt-in OS-level priority for per-callback
163    /// dispatch. `0` (default) means "no per-callback OS priority"
164    /// — the executor's cooperative dispatch path runs every
165    /// callback bound to this SC. Non-zero values trigger the
166    /// per-priority worker-pool path (registered via
167    /// `Executor::register_os_priority_dispatcher`); each callback
168    /// then runs on a worker thread the OS scheduler has elevated
169    /// to that numeric priority.
170    ///
171    /// Numeric meaning is platform-defined (POSIX 1..99 for
172    /// SCHED_FIFO; FreeRTOS 0..configMAX_PRIORITIES-1; Zephyr
173    /// direction-flipped). Chain-priority assignment + chain
174    /// grouping happen at the orchestration layer and are out of
175    /// executor scope.
176    pub os_pri: u8,
177    /// Phase 110.G — time-triggered window offset within the
178    /// executor's major frame. `None` (sentinel `0`) = always
179    /// eligible (no TT gate); `Some(off)` + `tt_window_duration_us`
180    /// gates dispatch to the half-open interval
181    /// `[off, off + duration) mod major_frame`.
182    ///
183    /// Independent of `class` — a `Sporadic`-class SC can also be TT-
184    /// gated; both gates apply (skip dispatch when EITHER fails).
185    /// Pairs with `Executor::register_time_triggered_dispatcher`
186    /// which sets the major-frame length.
187    pub tt_window_offset_us: OptUs,
188    /// Phase 110.G — time-triggered window length. See
189    /// `tt_window_offset_us`.
190    pub tt_window_duration_us: OptUs,
191}

193/// Phase 110.E.b — atomic sporadic-server state for ISR-driven
194/// refill. ISR / timer-thread context calls `refill_thunk` to top up
195/// the budget; spin_once reads atomically without any `&mut` access.
196///
197/// Replaces the polled-clock `SporadicState` shape on platforms with
198/// a `PlatformTimer` impl. The Executor still keeps the legacy
199/// `SporadicState` path active on `feature = "std"` so the
200/// transition is non-breaking.
201#[allow(dead_code)] // Phase 110.E.b — wired in PlatformTimer integration.
202pub struct AtomicSporadicState {
203    pub budget_remaining_us: portable_atomic::AtomicU32,
204    /// Wraps every ~50 days at ms resolution; saturates per
205    /// `tick`'s monotonic-clock contract. portable-atomic provides
206    /// AtomicU32 even on RISC-V `riscv32imc` / Cortex-M0+ that lack
207    /// native 32-bit atomics.
208    pub last_refill_ms: portable_atomic::AtomicU32,
209    pub budget_capacity_us: u32,
210    pub period_us: u32,
211    /// Phase 110.E.b — cumulative count of dispatched callbacks
212    /// whose measured wall-clock runtime exceeded the SC's
213    /// `budget_us`. Bumped by the per-callback runtime closure
214    /// inside `Executor::spin_once` (std-only — the no_std fallback
215    /// continues to use the polled `SporadicState` path without
216    /// per-callback overrun accounting). Cooperative single-thread
217    /// dispatch can't preempt a runaway callback, so this counter
218    /// is the diagnostic signal — the design's oneshot-IRQ-and-
219    /// cancel pattern is structurally equivalent for non-preemptive
220    /// callbacks, and `last_overrun_us` carries the worst-case
221    /// observation for tuning. Both reset by `clear_overrun_stats`.
222    pub overrun_count: portable_atomic::AtomicU32,
223    /// Phase 110.E.b — most recent dispatch's overrun amount
224    /// (`measured_us - budget_us`). `0` when no overrun has been
225    /// observed since the last `clear_overrun_stats`. Used by
226    /// monitoring code that wants to size the budget against
227    /// worst-case observed runtime.
228    pub last_overrun_us: portable_atomic::AtomicU32,
229}
230
231#[allow(dead_code)] // Phase 110.E.b — wired in PlatformTimer integration.
232impl AtomicSporadicState {
233    pub const fn new(budget_us: u32, period_us: u32) -> Self {
234        Self {
235            budget_remaining_us: portable_atomic::AtomicU32::new(budget_us),
236            last_refill_ms: portable_atomic::AtomicU32::new(0),
237            budget_capacity_us: budget_us,
238            period_us,
239            overrun_count: portable_atomic::AtomicU32::new(0),
240            last_overrun_us: portable_atomic::AtomicU32::new(0),
241        }
242    }
243
244    /// Record one overrun: callback measured runtime exceeded the
245    /// SC's `budget_us`. Bumps `overrun_count` + stores the absolute
246    /// overrun amount in `last_overrun_us`. Called from the
247    /// per-callback runtime closure inside `Executor::spin_once`.
248    #[inline]
249    pub fn record_overrun(&self, overrun_us: u32) {
250        self.overrun_count
251            .fetch_add(1, portable_atomic::Ordering::Relaxed);
252        self.last_overrun_us
253            .store(overrun_us, portable_atomic::Ordering::Relaxed);
254    }
255
256    /// Reset both overrun statistics. Useful when tuning the budget
257    /// across windows (monitoring code logs + clears periodically).
258    #[inline]
259    pub fn clear_overrun_stats(&self) {
260        self.overrun_count
261            .store(0, portable_atomic::Ordering::Relaxed);
262        self.last_overrun_us
263            .store(0, portable_atomic::Ordering::Relaxed);
264    }
265
266    /// Read the budget atomically; spin_once consults this to decide
267    /// whether to skip the SC's entries this cycle.
268    pub fn has_budget(&self) -> bool {
269        self.budget_remaining_us
270            .load(portable_atomic::Ordering::Acquire)
271            > 0
272    }
273
274    /// Saturating subtract — used by spin_once after dispatching a
275    /// callback bound to this SC.
276    pub fn consume(&self, us: u32) {
277        let mut cur = self
278            .budget_remaining_us
279            .load(portable_atomic::Ordering::Acquire);
280        loop {
281            let next = cur.saturating_sub(us);
282            match self.budget_remaining_us.compare_exchange_weak(
283                cur,
284                next,
285                portable_atomic::Ordering::Release,
286                portable_atomic::Ordering::Acquire,
287            ) {
288                Ok(_) => return,
289                Err(observed) => cur = observed,
290            }
291        }
292    }
293}

295/// C-callable refill thunk that `PlatformTimer::create_periodic`
296/// invokes from the platform's timer context. Single atomic store —
297/// safe in any thread / ISR context.
298///
299/// # Safety
300/// `user_data` must point at a live `AtomicSporadicState`; the caller
301/// of `PlatformTimer::create_periodic` owns the lifetime contract.
302#[allow(dead_code)] // Phase 110.E.b — wired in PlatformTimer integration.
303pub extern "C" fn atomic_sporadic_refill_thunk(user_data: *mut core::ffi::c_void) {
304    if user_data.is_null() {
305        return;
306    }
307    let state = unsafe { &*(user_data as *const AtomicSporadicState) };
308    state
309        .budget_remaining_us
310        .store(state.budget_capacity_us, portable_atomic::Ordering::Release);
311}

313/// Phase 110.E — user-space sporadic-server runtime state.
314///
315/// Tracks remaining `budget_us` for the current period and the wall-
316/// clock instant of the last refill. The executor consults this state
317/// during dispatch: when `budget_remaining_us` reaches 0 the SC is
318/// suppressed until the next period boundary, at which point a refill
319/// resets the counter.
320///
321/// Refill cadence is polled — each `spin_once` checks whether the
322/// elapsed time since the last refill exceeds `period_us` and tops
323/// the budget back up. Less precise than an ISR-driven refill (Phase
324/// 110.E's per-platform timer hook is what gets that) but correct as
325/// an upper-bound bandwidth limiter.
326#[allow(dead_code)] // Phase 110.E — wired in spin_once Sporadic dispatch.
327#[derive(Debug, Clone, Copy)]
328pub struct SporadicState {
329    pub budget_remaining_us: u32,
330    pub budget_capacity_us: u32,
331    pub period_us: u32,
332    pub last_refill_ms: u64,
333}
334
335#[allow(dead_code)] // Phase 110.E — wired in spin_once Sporadic dispatch.
336impl SporadicState {
337    pub const fn new(budget_us: u32, period_us: u32) -> Self {
338        Self {
339            budget_remaining_us: budget_us,
340            budget_capacity_us: budget_us,
341            period_us,
342            last_refill_ms: 0,
343        }
344    }
345
346    /// Apply elapsed-time accounting since the previous spin. Returns
347    /// `true` if the SC has remaining budget after the refill check.
348    pub fn tick(&mut self, now_ms: u64, delta_us: u32) -> bool {
349        // Refill at period boundaries — coarse but correct.
350        if now_ms.saturating_sub(self.last_refill_ms) >= self.period_us as u64 / 1000 {
351            self.budget_remaining_us = self.budget_capacity_us;
352            self.last_refill_ms = now_ms;
353        }
354        self.budget_remaining_us = self.budget_remaining_us.saturating_sub(delta_us);
355        self.budget_remaining_us > 0
356    }
357}

390/// One slot in a time-triggered schedule.
391///
392/// Window `[offset_us, offset_us + duration_us)` within the major
393/// frame. `name` is a static-lifetime label for diagnostics
394/// (logging, panic messages); the runtime never inspects it.
395#[derive(Debug, Clone, Copy)]
396pub struct TimeTriggeredWindow {
397    pub offset_us: u32,
398    pub duration_us: u32,
399    pub name: &'static str,
400}

402impl TimeTriggeredWindow {
403    pub const fn new(offset_us: u32, duration_us: u32, name: &'static str) -> Self {
404        Self {
405            offset_us,
406            duration_us,
407            name,
408        }
409    }
410}

412/// Fixed-size, no_std-friendly cyclic schedule. `N` is the
413/// declared maximum window count; `window_count` is the active
414/// length (callers can build the array up to `N` and set
415/// `window_count` to the actual size used).
416#[derive(Debug)]
417pub struct TimeTriggeredSchedule<const N: usize> {
418    pub major_frame_us: u32,
419    pub windows: [TimeTriggeredWindow; N],
420    pub window_count: usize,
421}
422
423impl<const N: usize> TimeTriggeredSchedule<N> {
424    /// Construct a schedule from an exhaustive `[TimeTriggeredWindow; N]`
425    /// array; `window_count` is set to `N`.
426    pub const fn new_full(major_frame_us: u32, windows: [TimeTriggeredWindow; N]) -> Self {
427        Self {
428            major_frame_us,
429            windows,
430            window_count: N,
431        }
432    }
433
434    /// Validate the schedule: every window must fit inside
435    /// `[0, major_frame_us)` and windows must be non-overlapping
436    /// in offset-sorted order. Sliding-window check; O(N²) is fine
437    /// because TT schedules are small (rarely > 16 slots).
438    pub fn validate(&self) -> Result<(), TimeTriggeredScheduleError> {
439        if self.major_frame_us == 0 {
440            return Err(TimeTriggeredScheduleError::ZeroMajorFrame);
441        }
442        if self.window_count > N {
443            return Err(TimeTriggeredScheduleError::WindowCountOverflow);
444        }
445        for (i, w) in self.windows[..self.window_count].iter().enumerate() {
446            if w.duration_us == 0 {
447                return Err(TimeTriggeredScheduleError::ZeroWindowDuration { window: i });
448            }
449            let end = (w.offset_us as u64) + (w.duration_us as u64);
450            if end > self.major_frame_us as u64 {
451                return Err(TimeTriggeredScheduleError::WindowExceedsMajorFrame { window: i });
452            }
453            for (j, other) in self.windows[..self.window_count].iter().enumerate() {
454                if i == j {
455                    continue;
456                }
457                let o_end = (other.offset_us as u64) + (other.duration_us as u64);
458                let overlaps = (w.offset_us as u64) < o_end && (other.offset_us as u64) < end;
459                if overlaps {
460                    return Err(TimeTriggeredScheduleError::WindowsOverlap {
461                        window_a: i,
462                        window_b: j,
463                    });
464                }
465            }
466        }
467        Ok(())
468    }
469}
470
471/// Validation errors for a [`TimeTriggeredSchedule`].
472#[derive(Debug, Clone, Copy, PartialEq, Eq)]
473pub enum TimeTriggeredScheduleError {
474    ZeroMajorFrame,
475    WindowCountOverflow,
476    ZeroWindowDuration { window: usize },
477    WindowExceedsMajorFrame { window: usize },
478    WindowsOverlap { window_a: usize, window_b: usize },
479}
