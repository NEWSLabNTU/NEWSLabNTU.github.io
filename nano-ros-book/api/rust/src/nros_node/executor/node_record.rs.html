1//! Phase 104.C.2 — multi-Node-per-Executor storage.
2//!
3//! Mirrors the `rclcpp` pattern where a single `Executor` holds N
4//! `Node`s via `add_node(...)`. Each Node carries its own
5//! name/namespace + a reference to the Session that backs it +
6//! a default `SchedContext` (Phase 110) handles inherit unless
7//! overridden.
8//!
9//! For Phase 104.C.2 we land the *storage scaffold* + the builder
10//! API. Multi-Session-per-Executor dispatch is a follow-up
11//! (Phase 104.C.3) — today every Node in this list resolves to the
12//! Executor's primary session, which means `node_builder.rmw(name)`
13//! only accepts the same backend the Executor was opened against.
14//! Bridge use cases (two RMW backends concurrent in one Executor)
15//! light up when 104.C.3 adds the per-Node session ref.
16
17use super::{sched_context::SchedContextId, types::NodeError};

19/// Opaque handle returned by `Executor::node_builder(...).build()`.
20/// Used in 104.C.3+ to disambiguate handle ownership when multiple
21/// Nodes coexist in one Executor.
22#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
23pub struct NodeId(pub(crate) u8);
24
25impl NodeId {
26    /// Reserved id for the implicit "primary" Node that mirrors the
27    /// pre-Phase 104.C.2 single-Node Executor identity.
28    pub const PRIMARY: NodeId = NodeId(0);
29
30    /// Numeric index into the Executor's node table.
31    pub fn index(self) -> usize {
32        self.0 as usize
33    }
34
35    /// Phase 104.C.8.b / C.9.b — build a `NodeId` from a raw `u8` for
36    /// FFI consumers that store the index in their own struct
37    /// (`nros_node_t.node_id`, `nros_cpp_node_t.node_id`). The value
38    /// is not validated against the executor's `nodes` table — the
39    /// caller is responsible for only constructing ids that
40    /// `node_builder(...).build()` previously returned. Out-of-range
41    /// ids fail loudly at the next `with_node` / `_on(...)` call.
42    pub const fn from_raw(raw: u8) -> NodeId {
43        NodeId(raw)
44    }
45
46    /// Raw u8 form for FFI persistence.
47    pub const fn raw(self) -> u8 {
48        self.0
49    }
50}
51
52/// Per-Node metadata stored inside the Executor.
53///
54/// Phase 104.C.2 keeps the shape minimal — name, namespace,
55/// default SchedContext, optional rmw-name for diagnostics. Future
56/// items: per-Node session reference (104.C.3), per-Node liveliness
57/// state, per-Node parameter overrides.
58pub struct NodeRecord {
59    pub name: heapless::String<64>,
60    pub namespace: heapless::String<64>,
61    /// RMW backend the Node was created against. `None` for the
62    /// implicit primary Node populated from `Executor::open`.
63    pub rmw_name: Option<heapless::String<32>>,
64    /// Per-Node locator override. `None` = use the Executor's
65    /// session-level locator.
66    pub locator: Option<heapless::String<128>>,
67    /// Default `SchedContext` for handles created via this Node.
68    /// Handles may override per-call. `SchedContextId::default()` =
69    /// the executor's auto-created Fifo slot (slot 0).
70    pub default_sched: SchedContextId,
71    /// Phase 104.C.3 — session-slot index. `0` resolves to the
72    /// Executor's primary `session` field; `N >= 1` resolves to
73    /// `extra_sessions[N-1]`. Each Node may bind to a different
74    /// session, enabling multi-RMW bridges in one Executor.
75    pub session_idx: u8,
76}
77
78impl NodeRecord {
79    /// Construct the implicit "primary" NodeRecord that mirrors the
80    /// Executor's pre-104.C.2 single-Node identity. Currently unused
81    /// (the primary Node is implicit until 104.C.3 wires per-Node
82    /// dispatch); kept here for the upcoming migration where every
83    /// Executor will have an explicit entry at slot 0.
84    #[allow(dead_code)]
85    pub(crate) fn new_primary(name: heapless::String<64>, namespace: heapless::String<64>) -> Self {
86        Self {
87            name,
88            namespace,
89            rmw_name: None,
90            locator: None,
91            default_sched: SchedContextId(0),
92            session_idx: 0,
93        }
94    }
95}

97/// Builder returned by `Executor::node_builder(name)`. Chainable
98/// configuration; `.build()` registers the Node with the Executor
99/// and returns a [`NodeId`].
100///
101/// rclcpp-aligned API. Mirrors:
102///
103/// ```ignore
104/// rclcpp::Node::make_shared("my_node",
105///     rclcpp::NodeOptions().use_intra_process_comms(true))
106/// ```
107///
108/// Where rclcpp uses a single `NodeOptions` struct, we expose the
109/// individual setters directly on the builder — fewer cycles when
110/// the user only needs one option.
111pub struct NodeBuilder<'a, 'cfg> {
112    pub(crate) executor: &'a mut super::spin::Executor,
113    pub(crate) name: &'cfg str,
114    pub(crate) namespace: Option<&'cfg str>,
115    pub(crate) rmw_name: Option<&'cfg str>,
116    pub(crate) locator: Option<&'cfg str>,
117    pub(crate) domain_id: Option<u32>,
118    pub(crate) sched: Option<SchedContextId>,
119}
120
121impl<'a, 'cfg> NodeBuilder<'a, 'cfg> {
122    /// Select an RMW backend by name. `name` must match a backend
123    /// registered via `nros_rmw_cffi_register_named` (Phase 104.B.2).
124    ///
125    /// In Phase 104.C.2 (current), the name must match the backend
126    /// the Executor was opened against — bridge mode lands in
127    /// 104.C.3 when per-Node sessions are wired. Passing a name
128    /// that doesn't match the Executor's session returns
129    /// `Err(NodeError::BackendMismatch)` from `.build()`.
130    pub fn rmw(mut self, name: &'cfg str) -> Self {
131        self.rmw_name = Some(name);
132        self
133    }
134
135    /// Override the locator for this Node's session. Empty / unset =
136    /// use the Executor's locator.
137    pub fn locator(mut self, locator: &'cfg str) -> Self {
138        self.locator = Some(locator);
139        self
140    }
141
142    /// Override the domain id for this Node's session.
143    pub fn domain_id(mut self, domain_id: u32) -> Self {
144        self.domain_id = Some(domain_id);
145        self
146    }
147
148    /// Namespace for handles created via this Node. Empty = "/".
149    pub fn namespace(mut self, namespace: &'cfg str) -> Self {
150        self.namespace = Some(namespace);
151        self
152    }
153
154    /// Default [`SchedContext`](super::sched_context::SchedContext) for
155    /// handles registered via this Node. Phase 110 integration —
156    /// handles inherit this unless they pass their own SchedContext
157    /// at registration time.
158    pub fn sched(mut self, sched: SchedContextId) -> Self {
159        self.sched = Some(sched);
160        self
161    }
162
163    /// Phase 104.C.3 — pick a session slot for the Node being
164    /// built. Returns `0` for the primary session (no rmw override
165    /// or rmw matches existing) and `N >= 1` for an extra session
166    /// just opened via `CffiRmw::open_with_rmw`.
167    #[cfg(feature = "rmw-cffi")]
168    fn resolve_session_slot(&mut self) -> Result<u8, NodeError> {
169        let Some(rmw) = self.rmw_name else {
170            return Ok(0);
171        };
172
173        // Reuse an extra session if one already opened against the
174        // same rmw + locator. Slot 0 (primary) is opaque — we don't
175        // know its rmw name today; treat the first-named-rmw Node
176        // as the primary unless explicitly bridging.
177        for (i, sess) in self.executor.extra_sessions.iter().enumerate() {
178            let _ = sess;
179            // Phase 104.C.3 doesn't yet store rmw-name per session;
180            // dedupe by NodeRecord's stored rmw_name + locator.
181            if let Some(prev) = self.executor.nodes.iter().find(|n| {
182                n.session_idx as usize == i + 1
183                    && n.rmw_name.as_deref() == Some(rmw)
184                    && n.locator.as_deref() == self.locator
185            }) {
186                let _ = prev;
187                return Ok((i + 1) as u8);
188            }
189        }
190
191        // First Node naming this rmw → open a new session.
192        let mode = nros_rmw::SessionMode::Client;
193        let locator = self.locator.unwrap_or("");
194        let domain_id = self.domain_id.unwrap_or(0);
195        let cfg = nros_rmw::RmwConfig {
196            locator,
197            mode,
198            domain_id,
199            node_name: self.name,
200            namespace: self.namespace.unwrap_or(""),
201            properties: &[],
202        };
203        let session = nros_rmw_cffi::CffiRmw::open_with_rmw(rmw, &cfg)
204            .map_err(crate::executor::types::NodeError::Transport)?;
205        self.executor
206            .extra_sessions
207            .push(session)
208            .map_err(|_| NodeError::NodeTableFull)?;
209        let idx = self.executor.extra_sessions.len();
210        if idx > u8::MAX as usize {
211            return Err(NodeError::NodeTableFull);
212        }
213        // Phase 104.C.6.b — install the shared wake flag on the
214        // freshly opened extra session so its backend notifications
215        // can short-circuit `spin_once`.
216        #[cfg(feature = "std")]
217        self.executor.install_wake_signal_on_extra(idx - 1);
218        Ok(idx as u8)
219    }
220
221    #[cfg(not(feature = "rmw-cffi"))]
222    fn resolve_session_slot(&mut self) -> Result<u8, NodeError> {
223        // Without `rmw-cffi`, only the primary session exists. An
224        // rmw-name override is meaningless; treat as the primary.
225        Ok(0)
226    }
227
228    /// Register the Node with the Executor and return its
229    /// [`NodeId`]. Bumps `Executor.nodes.len()`; fails if the table
230    /// is full (`NROS_EXECUTOR_MAX_NODES` reached) or the name is
231    /// too long.
232    pub fn build(mut self) -> Result<NodeId, NodeError> {
233        if self.name.len() > 64 {
234            return Err(NodeError::NameTooLong);
235        }
236
237        // Phase 104.C.2 — single-session check. rmw mismatch is an
238        // error today; 104.C.3 will accept and open a new session
239        // via the session cache.
240        if let Some(_requested) = self.rmw_name {
241            // No accessor for the current session's rmw name yet —
242            // the registry first-registered slot drives the
243            // singleton. We accept any rmw name in C.2 (no
244            // validation) so consumer code is forward-compatible.
245            // C.3 adds the mismatch check + session-cache lookup.
246        }
247
248        let mut name_buf = heapless::String::<64>::new();
249        name_buf
250            .push_str(self.name)
251            .map_err(|_| NodeError::NameTooLong)?;
252
253        let mut ns_buf = heapless::String::<64>::new();
254        if let Some(ns) = self.namespace {
255            ns_buf.push_str(ns).map_err(|_| NodeError::NameTooLong)?;
256        } else {
257            ns_buf
258                .push_str(self.executor.namespace.as_str())
259                .map_err(|_| NodeError::NameTooLong)?;
260        }
261
262        let mut rmw_buf = None;
263        if let Some(rmw) = self.rmw_name {
264            let mut s = heapless::String::<32>::new();
265            s.push_str(rmw).map_err(|_| NodeError::NameTooLong)?;
266            rmw_buf = Some(s);
267        }
268
269        let mut loc_buf = None;
270        if let Some(loc) = self.locator {
271            let mut s = heapless::String::<128>::new();
272            s.push_str(loc).map_err(|_| NodeError::NameTooLong)?;
273            loc_buf = Some(s);
274        }
275
276        // Phase 104.C.3 — resolve session_idx. If no rmw was named,
277        // or the rmw name matches the implicit primary backend, use
278        // slot 0. Otherwise, open a fresh session via
279        // `CffiRmw::open_with_rmw` and stash it in `extra_sessions`.
280        let session_idx = self.resolve_session_slot()?;
281
282        let record = NodeRecord {
283            name: name_buf,
284            namespace: ns_buf,
285            rmw_name: rmw_buf,
286            locator: loc_buf,
287            default_sched: self.sched.unwrap_or(SchedContextId(0)),
288            session_idx,
289        };
290
291        self.executor
292            .nodes
293            .push(record)
294            .map_err(|_| NodeError::NodeTableFull)?;
295        let idx = self.executor.nodes.len() - 1;
296        if idx > u8::MAX as usize {
297            // heapless cap is far below u8::MAX; defensive only.
298            return Err(NodeError::NodeTableFull);
299        }
300        Ok(NodeId(idx as u8))
301    }
