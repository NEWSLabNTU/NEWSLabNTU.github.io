1//! Phase 110.A — `Dispatcher` trait.
2//!
3//! The dispatcher consumes a [`ReadySet`](super::ready_set::ReadySet)
4//! produced by [`Activator`](super::activator::Activator) and fires
5//! the corresponding callbacks. It binds the arena pointer at
6//! construction time (no raw `*mut u8` per call) and respects the
7//! configured [`DrainMode`](super::types::DrainMode) — `Latched`
8//! drains the snapshot only, `Greedy` re-runs the activator after
9//! each callback.
10//!
11//! 110.A defines the trait; 110.A.b rewires `spin_once` to drive
12//! dispatch through it. Default impl reproduces the pre-refactor
13//! `try_process` loop bit-for-bit.
14
15use super::{ready_set::ReadySet, types::SpinOnceResult};
16
17#[allow(dead_code)] // Phase 110.A — wired in 110.A.b spin_once rewire.
18pub(crate) trait Dispatcher {
19    /// Drain `ready` and fire each callback. Returns aggregate counts
20    /// for the cycle.
21    fn dispatch<R: ReadySet>(&mut self, ready: &mut R, delta_ms: u64) -> SpinOnceResult;
22}
