1//! Phase 218.E — ABI version guard between the prebuilt `nros` CLI and the
2//! `nros-core` runtime crate the consumer's `Cargo.lock` resolves.
3//!
4//! The CLI emits Rust / C / C++ code that targets `nros-core`, `nros-c`, and
5//! `nros-cpp` runtime ABIs. A version mismatch between the CLI binary and the
6//! runtime crates the consumer links surfaces as link errors, struct-layout
7//! mismatches, or — worst — silent runtime UB. The guard catches this BEFORE
8//! codegen runs.
9//!
10//! ## Surface
11//!
12//! - [`check_workspace`] — resolves the authoritative `Cargo.lock`, parses
13//!   it, finds the resolved `nros-core` version, compares to the CLI
14//!   binary's embedded version ([`CLI_VERSION`]), and either continues or
15//!   exits with an actionable error message naming both versions plus the
16//!   fix command. For consumers inside the nano-ros monorepo the lock is the
17//!   monorepo root lock (in-tree examples link the patched in-tree
18//!   `nros-core`, so a standalone crate's own — possibly stale — lock is not
19//!   authoritative); external consumers use the nearest `Cargo.lock`.
20//! - [`check_workspaces`] — the multi-workspace variant for verbs that resolve
21//!   inputs from more than one workspace (e.g. `nros codegen-system`).
22//!
23//! ## Opt-out
24//!
25//! `NROS_SKIP_VERSION_CHECK=1` in the environment bypasses the check, with a
26//! `warning:` line on stderr so the bypass is visible in CI logs.
27//!
28//! ## Match rule
29//!
30//! Strict equality on the full SemVer string today. The comparison fn
31//! ([`versions_match`]) is the single point to relax later (e.g. to
32//! "MAJOR.MINOR equal").
33//!
34//! ## Missing `Cargo.lock`
35//!
36//! If the consumer has not run `cargo generate-lockfile` yet, the guard
37//! warns-and-continues — the codegen step the verb is about to run will
38//! probably create the lock itself.
39

48/// The CLI binary's embedded version — baked at compile time from the
49/// `nros-cli-core` crate's `CARGO_PKG_VERSION`.
50///
51/// Today this is *the CLI's own version*; the `nros-cli` workspace bumps in
52/// lockstep with the runtime ABI it targets. A future refactor may split the
53/// "CLI version" from the "runtime ABI version" (e.g. via a build script that
54/// resolves the in-tree `nros-core` version); the call sites here would not
55/// change.
56pub const CLI_VERSION: &str = env!("CARGO_PKG_VERSION");
57
58/// Env-var name that bypasses the guard (any non-empty value opts out).
59pub const SKIP_ENV: &str = "NROS_SKIP_VERSION_CHECK";
60
61/// The codegen verb that triggered the check — flows into the error
62/// message so the user knows which command tripped the guard.
63#[derive(Debug, Clone, Copy)]
64pub enum Verb {
65    Codegen,
66    GenerateRust,
67    GenerateC,
68    GenerateCpp,
69    CodegenSystem,
70}
71
72impl Verb {
73    fn as_str(self) -> &'static str {
74        match self {
75            Verb::Codegen => "nros codegen",
76            Verb::GenerateRust => "nros generate-rust",
77            Verb::GenerateC => "nros generate c",
78            Verb::GenerateCpp => "nros generate cpp",
79            Verb::CodegenSystem => "nros codegen-system",
80        }
81    }
82}

84/// Walk up from `start` looking for a `Cargo.lock`. Returns `None` if none
85/// is found above the filesystem root.
86pub fn find_cargo_lock(start: &Path) -> Option<PathBuf> {
87    let mut cur: Option<&Path> = if start.is_file() {
88        start.parent()
89    } else {
90        Some(start)
91    };
92    while let Some(dir) = cur {
93        let candidate = dir.join("Cargo.lock");
94        if candidate.is_file() {
95            return Some(candidate);
96        }
97        cur = dir.parent();
98    }
99    None
100}

102/// Walk up from `start` to find the nano-ros monorepo root — the directory
103/// containing `packages/core/nros-core/Cargo.toml`. Returns `None` when
104/// `start` is not inside the monorepo (a genuine external consumer), in
105/// which case the nearest-`Cargo.lock` rule applies.
106pub fn find_monorepo_root(start: &Path) -> Option<PathBuf> {
107    const MARKER: &str = "packages/core/nros-core/Cargo.toml";
108    let mut cur: Option<&Path> = if start.is_file() {
109        start.parent()
110    } else {
111        Some(start)
112    };
113    while let Some(dir) = cur {
114        if dir.join(MARKER).is_file() {
115            return Some(dir.to_path_buf());
116        }
117        cur = dir.parent();
118    }
119    None
120}

122/// The monorepo root's `Cargo.lock`, if `start` is inside the monorepo and
123/// that lock exists.
124///
125/// In-tree examples / test crates are standalone projects that link the
126/// in-tree `nros-core` via `[patch.crates-io]`, so a standalone crate's own
127/// `Cargo.lock` can be stale (e.g. pinned to a pre-bump `0.1.0` long after
128/// the workspace moved to `0.5.0`) while the crate still builds against the
129/// current in-tree ABI. The monorepo root lock is the authoritative source
130/// for those consumers; external consumers fall back to the nearest lock.
131fn monorepo_root_lock(start: &Path) -> Option<PathBuf> {
132    let lock = find_monorepo_root(start)?.join("Cargo.lock");
133    lock.is_file().then_some(lock)
134}

136/// Parse a `Cargo.lock` (TOML) and pull out the resolved version of the named
137/// package, if present.
138///
139/// Returns `None` if the package is not in the lock — e.g. the consumer has
140/// not yet declared a dependency on `nros-core`. That is not an error; the
141/// caller treats it as "nothing to check".
142pub fn nros_core_version_in_lock(lock_path: &Path) -> Result<Option<String>> {
143    let body = std::fs::read_to_string(lock_path)
144        .wrap_err_with(|| format!("read Cargo.lock at {}", lock_path.display()))?;
145    let parsed: toml::Value = toml::from_str(&body)
146        .wrap_err_with(|| format!("parse Cargo.lock at {}", lock_path.display()))?;
147    let Some(packages) = parsed.get("package").and_then(|v| v.as_array()) else {
148        return Ok(None);
149    };
150    for pkg in packages {
151        let Some(name) = pkg.get("name").and_then(|n| n.as_str()) else {
152            continue;
153        };
154        if name != "nros-core" {
155            continue;
156        }
157        if let Some(ver) = pkg.get("version").and_then(|v| v.as_str()) {
158            return Ok(Some(ver.to_string()));
159        }
160    }
161    Ok(None)
162}

164/// Comparison rule. Strict equality today. The single point to relax later
165/// (e.g. compare only `MAJOR.MINOR`).
166pub fn versions_match(cli: &str, consumer: &str) -> bool {
167    cli == consumer
168}
169
170/// Run the guard against a single workspace anchor.
171///
172/// `start` is either a file (the `package.xml` / `args-file` / `system.toml`
173/// the verb received) or a directory (the workspace root); the guard walks up
174/// from there to find a `Cargo.lock`.
175///
176/// Returns `Ok(())` on match, opt-out, or no-lock-found.
177/// Returns `Err(...)` on mismatch with an actionable message.
178pub fn check_workspace(start: &Path, verb: Verb) -> Result<()> {
179    check_workspaces(std::slice::from_ref(&start), verb)
180}
181
182/// Multi-workspace variant — checks each anchor in turn and fails on the
183/// first mismatch. Used by verbs that resolve inputs from more than one
184/// workspace (e.g. `nros codegen-system` reading a bringup pkg from
185/// workspace A + member pkgs from workspace B).
186pub fn check_workspaces(starts: &[&Path], verb: Verb) -> Result<()> {
187    if env::var(SKIP_ENV).map(|v| !v.is_empty()).unwrap_or(false) {
188        warn_bypass(verb);
189        return Ok(());
190    }
191
192    let mut checked_locks: Vec<PathBuf> = Vec::new();
193    for start in starts {
194        // Prefer the nano-ros monorepo root lock for in-tree consumers: a
195        // standalone example/test under the nano-ros tree links the in-tree
196        // `nros-core` via `[patch.crates-io]`, so its own (possibly stale)
197        // `Cargo.lock` does not reflect the ABI it actually builds against.
198        // External consumers (no monorepo marker above `start`) keep the
199        // nearest-lock rule.
200        let lock = monorepo_root_lock(start).or_else(|| find_cargo_lock(start));
201        let Some(lock) = lock else {
202            // No lock — codegen verb will likely create one. Warn so the
203            // skip is visible.
204            warn_no_lock(start, verb);
205            continue;
206        };
207        if checked_locks.iter().any(|prev| prev == &lock) {
208            // Same workspace as a previous anchor.
209            continue;
210        }
211        let Some(consumer_version) = nros_core_version_in_lock(&lock)? else {
212            // `nros-core` not in lock — consumer hasn't declared a dep on it
213            // yet. Nothing to check.
214            checked_locks.push(lock);
215            continue;
216        };
217        if !versions_match(CLI_VERSION, &consumer_version) {
218            bail!(
219                "{}",
220                mismatch_message(verb, &lock, &consumer_version, CLI_VERSION)
221            );
222        }
223        checked_locks.push(lock);
224    }
225    Ok(())
226}

228fn warn_bypass(verb: Verb) {
229    let _ = writeln!(
230        std::io::stderr(),
231        "warning: {} ABI version guard bypassed via {SKIP_ENV}=1 \
232         (CLI nros-core = {CLI_VERSION})",
233        verb.as_str(),
234    );
235}
236
237fn warn_no_lock(start: &Path, verb: Verb) {
238    // Quiet unless verbose-ish: still noteworthy in CI logs. Only emit
239    // when stderr is a TTY OR when the env requests verbose tracing —
240    // otherwise pure-codegen verbs (which legitimately run before a lock
241    // exists) get spammy. Cheapest heuristic that still surfaces in CI: a
242    // single line, gated only on the env var.
243    if env::var("NROS_TRACE_ABI_GUARD")
244        .map(|v| !v.is_empty())
245        .unwrap_or(false)
246        || std::io::stderr().is_terminal()
247    {
248        let _ = writeln!(
249            std::io::stderr(),
250            "note: {} found no Cargo.lock at or above {}; skipping ABI version guard",
251            verb.as_str(),
252            start.display(),
253        );
254    }
255}

257fn mismatch_message(verb: Verb, lock: &Path, consumer: &str, cli: &str) -> String {
258    format!(
259        "{verb} aborted: ABI version mismatch between the `nros` CLI binary \
260         and the runtime `nros-core` your workspace resolves.\n  \
261         CLI binary nros-core version: {cli}\n  \
262         Workspace Cargo.lock at:      {lock}\n  \
263         Workspace nros-core version:  {consumer}\n\n\
264         The CLI emits Rust / C / C++ that targets a specific `nros-core` ABI; \
265         a mismatch can manifest as link errors, struct-layout mismatches, or \
266         silent runtime UB. Resolve by rebuilding the CLI against this workspace's \
267         pinned runtime:\n  \
268         cargo build --release --manifest-path /path/to/nano-ros/packages/cli/Cargo.toml --bin nros\n\
269         (or `just setup-cli` if the target workspace IS nano-ros itself).\n\n\
270         To bypass this guard for an intentional cross-version workflow, set \
271         {SKIP_ENV}=1.",
272        verb = verb.as_str(),
273        cli = cli,
274        lock = lock.display(),
275        consumer = consumer,
276    )
277}

283    #[test]
284    fn versions_match_is_strict_equality() {
285        assert!(versions_match("0.3.7", "0.3.7"));
286        assert!(!versions_match("0.3.7", "0.3.8"));
287        assert!(!versions_match("0.3.7", "0.0.999"));
288    }

290    #[test]
291    fn find_cargo_lock_walks_up() {
292        let tmp = tempdir_path("abi_guard_find_lock");
293        std::fs::create_dir_all(tmp.join("a/b/c")).unwrap();
294        std::fs::write(tmp.join("Cargo.lock"), "# stub\n").unwrap();
295        let found = find_cargo_lock(&tmp.join("a/b/c")).expect("lock found");
296        assert_eq!(found, tmp.join("Cargo.lock"));
297        let _ = std::fs::remove_dir_all(&tmp);
298    }

303        std::fs::create_dir_all(&tmp).unwrap();
304        let lock = tmp.join("Cargo.lock");

327    #[test]
328    fn monorepo_root_lock_wins_over_nested_stale_lock() {
329        let tmp = tempdir_path("abi_guard_monorepo");
330        // Monorepo root: the nros-core marker + a fresh root lock (0.5.0).
331        std::fs::create_dir_all(tmp.join("packages/core/nros-core")).unwrap();
332        std::fs::write(tmp.join("packages/core/nros-core/Cargo.toml"), "# stub\n").unwrap();
333        std::fs::write(
334            tmp.join("Cargo.lock"),
335            "[[package]]\nname = \"nros-core\"\nversion = \"0.5.0\"\n",
336        )
337        .unwrap();
338        // Nested standalone example with a STALE own-lock (0.1.0).
339        let ex = tmp.join("examples/x/rust/talker");
340        std::fs::create_dir_all(&ex).unwrap();
341        std::fs::write(
342            ex.join("Cargo.lock"),
343            "[[package]]\nname = \"nros-core\"\nversion = \"0.1.0\"\n",
344        )
345        .unwrap();
346
347        // The monorepo root is discovered from the nested dir...
348        let root = find_monorepo_root(&ex).expect("monorepo root found");
349        assert_eq!(root, tmp);
350        // ...and its lock (0.5.0) is the authoritative source, not the
351        // nested stale 0.1.0 lock that find_cargo_lock would return.
352        let nearest = find_cargo_lock(&ex).unwrap();
353        assert_eq!(
354            nros_core_version_in_lock(&nearest).unwrap().as_deref(),
355            Some("0.1.0")
356        );
357        assert_eq!(
358            nros_core_version_in_lock(&root.join("Cargo.lock"))
359                .unwrap()
360                .as_deref(),
361            Some("0.5.0")
362        );
363
364        // A dir outside any monorepo marker has no monorepo root.
365        let outside = tempdir_path("abi_guard_outside");
366        std::fs::create_dir_all(&outside).unwrap();
367        assert!(find_monorepo_root(&outside).is_none());
368
369        let _ = std::fs::remove_dir_all(&tmp);
370        let _ = std::fs::remove_dir_all(&outside);
371    }

392    fn tempdir_path(tag: &str) -> std::path::PathBuf {
393        let stamp = std::time::SystemTime::now()
394            .duration_since(std::time::UNIX_EPOCH)
395            .unwrap()
396            .as_nanos();
397        let dir =
398            std::env::temp_dir().join(format!("phase-218-e-{tag}-{}-{stamp}", std::process::id()));
399        let _ = std::fs::remove_dir_all(&dir);
400        dir
401    }
