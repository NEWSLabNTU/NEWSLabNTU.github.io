15/// Permissive envelope for extracting a `[component]` table out of a package's
16/// `nros.toml` (Phase 172 W.1 fold) while ignoring sibling tables
17/// (`[workspace]` / `[system]` / `[deploy]` / `[node]` / `[[transport]]`).
18/// Unknown keys are ignored on purpose — only `[component]` is read here.
19#[derive(Debug, Deserialize)]
20struct ComponentEnvelope {
21    #[serde(default)]
22    component: Option<ComponentConfig>,
23}
24
25/// Load a component declaration from a manifest path. Handles two forms:
26///
27/// - **Folded** (Phase 172 W.1): a `[component]` table inside a package's
28///   `nros.toml`. Returns `Ok(None)` when that file carries no `[component]`
29///   (it is a workspace-root / direct-mode manifest, not a component).
30/// - **Legacy**: the standalone whole-file form (`component_nros.toml` or
31///   `nros/components/*.toml`), which is deprecated and warns once per file.
32pub fn load_component_config(path: &Path) -> Result<Option<ComponentConfig>> {
33    let raw = fs::read_to_string(path)
34        .with_context(|| format!("failed to read component manifest {}", path.display()))?;
35    let is_nros_toml = path.file_name().and_then(|name| name.to_str()) == Some("nros.toml");
36    if is_nros_toml {
37        let envelope: ComponentEnvelope =
38            toml::from_str(&raw).with_context(|| format!("failed to parse {}", path.display()))?;
39        Ok(envelope.component)
40    } else {
41        warn_legacy_component_manifest(path);
42        let config: ComponentConfig = toml::from_str(&raw)
43            .with_context(|| format!("failed to parse component manifest {}", path.display()))?;
44        Ok(Some(config))
45    }
46}

48/// Emit the `component_nros.toml` deprecation notice at most once per file path
49/// for the life of the process (Phase 172 W.1 deprecation window).
50fn warn_legacy_component_manifest(path: &Path) {
51    static WARNED: LazyLock<Mutex<BTreeSet<PathBuf>>> =
52        LazyLock::new(|| Mutex::new(BTreeSet::new()));
53    if WARNED.lock().unwrap().insert(path.to_path_buf()) {
54        eprintln!(
55            "warning: `{}` is a deprecated standalone component manifest; fold it into the \
56             package's `nros.toml` as a `[component]` table (Phase 172 W.1). The standalone \
57             form still works during the deprecation window.",
58            path.display()
59        );
60    }
61}

69#[derive(Debug, Clone)]
70pub struct Package {
71    pub name: String,
72    pub root: PathBuf,
73    pub package_xml: PathBuf,
74    pub nros_toml: Option<PathBuf>,
75    pub launch_files: Vec<PathBuf>,
76    pub manifest_files: Vec<PathBuf>,
77    pub metadata_files: Vec<PathBuf>,
78    /// Component-declaration candidates that tie a package to a
79    /// `nros::component!` export + its source-metadata path (Phase 126.B.7).
80    /// In preference order: the package's folded `nros.toml` `[component]`
81    /// table (W.1), the legacy standalone `component_nros.toml`, then any
82    /// `nros/components/*.toml`. An `nros.toml` without a `[component]` table
83    /// is filtered out at parse time (`load_component_config`).
84    pub component_config_files: Vec<PathBuf>,
85}

87impl Workspace {
88    pub fn discover(root: &Path) -> Result<Self> {
89        let mut packages = Vec::new();
90        let root = root.to_path_buf();
91        if root.join("package.xml").is_file() {
92            packages.push(discover_package(&root)?);
93        }
94        let src = root.join("src");
95        if src.is_dir() {
96            for entry in fs::read_dir(&src)? {
97                let entry = entry?;
98                let path = entry.path();
99                if path.join("package.xml").is_file() {
100                    packages.push(discover_package(&path)?);
101                }
102            }
103        }
104        packages.sort_by(|a, b| a.name.cmp(&b.name));
105        Ok(Self { root, packages })
106    }
107
108    pub fn source_metadata_files(&self) -> Vec<PathBuf> {
109        unique_paths(
110            self.packages
111                .iter()
112                .flat_map(|pkg| pkg.metadata_files.iter().cloned()),
113        )
114    }
115
116    pub fn manifest_files(&self) -> Vec<PathBuf> {
117        unique_paths(
118            self.packages
119                .iter()
120                .flat_map(|pkg| pkg.manifest_files.iter().cloned()),
121        )
122    }
123
124    pub fn package_nros_toml(&self, package: &str) -> Option<PathBuf> {
125        self.packages
126            .iter()
127            .find(|pkg| pkg.name == package)
128            .and_then(|pkg| pkg.nros_toml.clone())
129    }

131    /// Iterate every component declaration in the workspace — folded
132    /// `nros.toml` `[component]` tables (W.1) and legacy standalone
133    /// `component_nros.toml` / `nros/components/*.toml` files — as
134    /// `(package_root, manifest_path, parsed_config)` tuples, deduped by
135    /// `(package, component)`. Used by the metadata command to detect packages
136    /// that
137    /// declared themselves nros components but lack the
138    /// `nros::component!` export (their `[metadata].source_metadata`
139    /// path doesn't exist on disk — see Phase 126.B.7 acceptance
140    /// criterion).
141    pub fn component_declarations(&self) -> Result<Vec<ComponentDeclaration>> {
142        let mut out = Vec::new();
143        for pkg in &self.packages {
144            // Dedup by `(package, component)` within a package, first-wins. The
145            // folded `nros.toml` sorts ahead of a legacy `component_nros.toml`,
146            // so when both declare the same component the folded form wins and
147            // the legacy file is ignored (it still warns once on read).
148            let mut seen = BTreeSet::new();
149            for manifest_path in &pkg.component_config_files {
150                // A package `nros.toml` is a candidate only if it actually
151                // carries a `[component]` table (W.1 fold); skip it otherwise.
152                let Some(config) = load_component_config(manifest_path)? else {
153                    continue;
154                };
155                if !seen.insert((config.package.clone(), config.component.clone())) {
156                    continue;
157                }
158                out.push(ComponentDeclaration {
159                    package_root: pkg.root.clone(),
160                    manifest_path: manifest_path.clone(),
161                    config,
162                });
163            }
164        }
165        Ok(out)
166    }
167}

169/// Parsed component manifest paired with its on-disk location.
170#[derive(Debug, Clone)]
171pub struct ComponentDeclaration {
172    /// Package root the manifest belongs to. `source_metadata` paths
173    /// in the manifest resolve relative to this directory.
174    pub package_root: PathBuf,
175    /// Absolute path to the manifest the declaration came from — a package's
176    /// folded `nros.toml` (W.1) or a legacy standalone `component_nros.toml` /
177    /// `nros/components/*.toml`.
178    pub manifest_path: PathBuf,
179    pub config: ComponentConfig,
180}
181
182impl ComponentDeclaration {
183    /// Absolute path to the `[metadata].source_metadata` file the
184    /// component is expected to emit. Relative paths resolve against
185    /// `package_root`.
186    pub fn source_metadata_path(&self) -> PathBuf {
187        let raw = Path::new(&self.config.metadata.source_metadata);
188        if raw.is_absolute() {
189            raw.to_path_buf()
190        } else {
191            self.package_root.join(raw)
192        }
193    }
194}

196fn discover_package(root: &Path) -> Result<Package> {
197    let package_xml = root.join("package.xml");
198    let parsed = PackageXml::parse(&package_xml)
199        .wrap_err_with(|| format!("failed to parse {}", package_xml.display()))?;
200    Ok(Package {
201        name: parsed.name,
202        root: root.to_path_buf(),
203        package_xml,
204        nros_toml: root
205            .join("nros.toml")
206            .is_file()
207            .then(|| root.join("nros.toml")),
208        launch_files: collect_files(
209            root,
210            &["launch"],
211            &["launch.py", "launch.xml", "launch.yaml", "launch.yml"],
212        )?,
213        manifest_files: collect_files(
214            root,
215            &["manifest", "manifests"],
216            &["launch.yaml", "launch.yml"],
217        )?,
218        metadata_files: collect_files(root, &["metadata", "nros", "target/nros"], &["json"])?,
219        component_config_files: discover_component_configs(root)?,
220    })
221}

223/// Locate component declaration candidates. Preference order (W.1 fold):
224/// the package's `nros.toml` (read for a `[component]` table — the canonical
225/// folded form), then the deprecated standalone `component_nros.toml` at the
226/// package root, then any `nros/components/*.toml`. Whether a candidate is
227/// actually a component is decided at parse time (`load_component_config`
228/// returns `None` for an `nros.toml` with no `[component]`).
229fn discover_component_configs(root: &Path) -> Result<Vec<PathBuf>> {
230    let mut out = Vec::new();
231    // W.1 fold: a package `nros.toml` may carry a `[component]` table.
232    let folded = root.join("nros.toml");
233    if folded.is_file() {
234        out.push(folded);
235    }
236    let primary = root.join("component_nros.toml");
237    if primary.is_file() {
238        out.push(primary);
239    }
240    // The multi-component glob is order-independent — sort it for determinism,
241    // but keep it *after* the root candidates so the folded `nros.toml` and the
242    // legacy `component_nros.toml` retain their preference order.
243    let components_dir = root.join("nros").join("components");
244    if components_dir.is_dir() {
245        let mut globbed = Vec::new();
246        for entry in fs::read_dir(&components_dir)? {
247            let path = entry?.path();
248            if path.extension().is_some_and(|ext| ext == "toml") {
249                globbed.push(path);
250            }
251        }
252        globbed.sort();
253        out.extend(globbed);
254    }
255    Ok(out)
256}

258fn collect_files(root: &Path, dirs: &[&str], suffixes: &[&str]) -> Result<Vec<PathBuf>> {
259    let mut out = Vec::new();
260    for dir in dirs {
261        let path = root.join(dir);
262        if path.is_dir() {
263            collect_matching(&path, suffixes, &mut out)?;
264        }
265    }
266    out.sort();
267    Ok(out)
268}
269
270fn collect_matching(dir: &Path, suffixes: &[&str], out: &mut Vec<PathBuf>) -> Result<()> {
271    for entry in fs::read_dir(dir)? {
272        let entry = entry?;
273        let path = entry.path();
274        if path.is_dir() {
275            collect_matching(&path, suffixes, out)?;
276        } else if suffixes.iter().any(|suffix| {
277            path.file_name()
278                .and_then(|name| name.to_str())
279                .is_some_and(|name| name.ends_with(suffix))
280        }) {
281            out.push(path);
282        }
283    }
284    Ok(())
285}

287pub fn unique_paths<I>(paths: I) -> Vec<PathBuf>
288where
289    I: IntoIterator<Item = PathBuf>,
290{
291    let mut seen = BTreeSet::new();
292    paths
293        .into_iter()
294        .filter(|path| seen.insert(path.clone()))
295        .collect()
296}

303    /// RAII scratch directory under the system temp dir (no `tempfile` dep).
304    struct Scratch(PathBuf);
305    impl Scratch {
306        fn new(tag: &str) -> Self {
307            static N: AtomicU32 = AtomicU32::new(0);
308            let dir = std::env::temp_dir().join(format!(
309                "nros-ws-test-{}-{}-{}",
310                tag,
311                std::process::id(),
312                N.fetch_add(1, Ordering::Relaxed)
313            ));
314            fs::create_dir_all(&dir).unwrap();
315            Scratch(dir)
316        }
317        fn write(&self, rel: &str, body: &str) {
318            let path = self.0.join(rel);
319            fs::create_dir_all(path.parent().unwrap()).unwrap();
320            fs::write(path, body).unwrap();
321        }
322    }
323    impl Drop for Scratch {
324        fn drop(&mut self) {
325            let _ = fs::remove_dir_all(&self.0);
326        }
327    }

360    #[test]
361    fn folds_component_table_in_package_nros_toml() {
362        let s = Scratch::new("fold");
363        s.write("src/demo_pkg/package.xml", PKG_XML);
364        // A package nros.toml carrying [workspace]-unrelated sibling tables plus
365        // the folded [component] — sibling tables must be ignored.
366        s.write(
367            "src/demo_pkg/nros.toml",
368            &format!("[[transport]]\nid = \"t\"\nkind = \"udp\"\n{COMPONENT_TABLE}"),
369        );
370
371        let ws = Workspace::discover(&s.0).unwrap();
372        let decls = ws.component_declarations().unwrap();
373        assert_eq!(decls.len(), 1, "folded [component] is discovered");
374        assert_eq!(decls[0].config.package, "demo_pkg");
375        assert_eq!(decls[0].config.component, "talker");
376        assert!(decls[0].manifest_path.ends_with("nros.toml"));
377    }
