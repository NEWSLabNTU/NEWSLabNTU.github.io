1//! Phase 212.L.6 — launch file synthesis + multi-launch resolution.
2//!
3//! Every nros verb that needs a launch file (`plan`, `launch`,
4//! `codegen-system`) shares the same resolution policy:
5//!
6//! 1. Honour an explicit `--file <path>` argument.
7//! 2. Pick `<pkg>/launch/<pkg-name>.launch.xml` if it exists.
8//! 3. Pick `<pkg>/launch/system.launch.xml` if it exists.
9//! 4. Pick the single `*.launch.xml` under `<pkg>/launch/` when there is
10//!    exactly one.
11//! 5. For component / application pkgs (Cargo.toml or CMakeLists.txt
12//!    present) with no launch file, **synthesise** a minimal
13//!    `<launch><node pkg="…" exec="…" /></launch>` body in memory.
14//! 6. Otherwise (Path A bringup pkg with no launch file, or multi-launch
15//!    ambiguity), hard-error with a candidate list.
16//!
17//! The synthesised XML is returned as a string — callers materialise it
18//! to a temp file when they need to hand it to the external
19//! `play_launch_parser` binary. It is never persisted alongside the
20//! bringup.
21

29/// One launch input — either an on-disk file or an in-memory XML body
30/// synthesised by [`resolve_launch`] for self-bringup pkgs.
31#[derive(Clone, Debug, PartialEq, Eq)]
32pub enum LaunchInput {
33    /// Pre-existing file on disk (resolved by steps 1-4 of the policy).
34    File(PathBuf),
35    /// In-memory XML body synthesised for a self-bringup pkg (step 5).
36    Synth(String),
37}
38
39impl LaunchInput {
40    /// Materialise the launch input to a real filesystem path so it can be
41    /// fed to the external `play_launch_parser` binary. For an on-disk
42    /// [`LaunchInput::File`] this is a no-op; for [`LaunchInput::Synth`]
43    /// it writes a uniquely-named temp file and returns an RAII guard
44    /// that removes the file on drop.
45    pub fn materialise(&self) -> Result<MaterialisedLaunch> {
46        match self {
47            LaunchInput::File(p) => Ok(MaterialisedLaunch {
48                path: p.clone(),
49                _guard: None,
50            }),
51            LaunchInput::Synth(body) => {
52                let dir = std::env::temp_dir();
53                let stamp = std::time::SystemTime::now()
54                    .duration_since(std::time::UNIX_EPOCH)
55                    .map(|d| d.as_nanos())
56                    .unwrap_or(0);
57                let path = dir.join(format!(
58                    "nros-launch-synth-{}-{stamp}.launch.xml",
59                    std::process::id()
60                ));
61                fs::write(&path, body)
62                    .map_err(|e| eyre!("write synthesised launch to {}: {e}", path.display()))?;
63                Ok(MaterialisedLaunch {
64                    path: path.clone(),
65                    _guard: Some(TempPathGuard(path)),
66                })
67            }
68        }
69    }
70}

72/// On-disk handle to a launch file. Owns a temp-file guard when the
73/// underlying input was synthesised — dropping cleans up the temp file.
74pub struct MaterialisedLaunch {
75    pub path: PathBuf,
76    _guard: Option<TempPathGuard>,
77}

79/// RAII: deletes the wrapped path on drop. Used by [`LaunchInput::materialise`]
80/// to clean up the synthesised temp file once the parser returns.
81struct TempPathGuard(PathBuf);
82
83impl Drop for TempPathGuard {
84    fn drop(&mut self) {
85        let _ = fs::remove_file(&self.0);
86    }
87}

89/// Errors specific to launch resolution. Wrapped in `eyre` at the
90/// callsite; declared as a real enum so tests can match on the failure
91/// mode without parsing strings.
92#[derive(Debug, thiserror::Error)]
93pub enum LaunchResolveError {
94    #[error(
95        "no launch file resolved under {pkg}/launch and pkg is a Path A bringup \
96         (no Cargo.toml / CMakeLists.txt) — synthesis is disallowed for multi-node bringups"
97    )]
98    PathABringupNoLaunch { pkg: PathBuf },
99
100    #[error(
101        "multiple launch files under {pkg}/launch ({candidates:?}) — pass `--file <name>` \
102         to pick one (try one of: {candidates:?})"
103    )]
104    AmbiguousMultiLaunch {
105        pkg: PathBuf,
106        candidates: Vec<String>,
107    },
108
109    #[error(
110        "package {pkg} declares multiple executables ({candidates:?}); \
111         pass `--exec <name>` to pick one"
112    )]
113    AmbiguousExec {
114        pkg: PathBuf,
115        candidates: Vec<String>,
116    },
117
118    #[error(
119        "--file {file:?} does not resolve to an existing launch file (tried \
120         {pkg}/launch/{file}, ./{file}, and absolute)"
121    )]
122    FileArgUnresolved { pkg: PathBuf, file: String },
123
124    #[error(
125        "could not determine pkg name for {pkg} (no Cargo.toml [package].name, no project() in CMakeLists.txt)"
126    )]
127    UnknownPkgName { pkg: PathBuf },
128
129    #[error(
130        "could not determine executable for {pkg} (no [[bin]] in Cargo.toml, no add_executable in CMakeLists.txt)"
131    )]
132    UnknownExec { pkg: PathBuf },
133}

135/// Spec resolver — see module docs.
136///
137/// `pkg_dir` is the bringup / component / application pkg directory.
138/// `file_arg` is the user's `--file` value (passes step 1).
139/// `exec_arg` is the user's `--exec` value (consumed only by synth).
140pub fn resolve_launch(
141    pkg_dir: &Path,
142    file_arg: Option<&str>,
143    exec_arg: Option<&str>,
144) -> Result<LaunchInput> {
145    // Step 1 — explicit --file.
146    if let Some(file) = file_arg {
147        // <pkg>/launch/<file>
148        let in_launch = pkg_dir.join("launch").join(file);
149        if in_launch.is_file() {
150            return Ok(LaunchInput::File(in_launch));
151        }
152        // ./<file> (cwd-relative)
153        let cwd_rel = std::env::current_dir().ok().map(|c| c.join(file));
154        if let Some(p) = cwd_rel
155            && p.is_file()
156        {
157            return Ok(LaunchInput::File(p));
158        }
159        // absolute
160        let as_abs = Path::new(file);
161        if as_abs.is_absolute() && as_abs.is_file() {
162            return Ok(LaunchInput::File(as_abs.to_path_buf()));
163        }
164        return Err(LaunchResolveError::FileArgUnresolved {
165            pkg: pkg_dir.to_path_buf(),
166            file: file.to_string(),
167        }
168        .into());
169    }
170
171    // Need pkg name for steps 2 + 5; derive it now (used in error
172    // messages too). Allow failure here: a Path A bringup may have
173    // neither Cargo.toml nor CMakeLists.txt, in which case we still need
174    // to walk steps 2-4 against the directory name.
175    let pkg_name = discover_pkg_name(pkg_dir).unwrap_or_else(|_| {
176        pkg_dir
177            .file_name()
178            .and_then(|s| s.to_str())
179            .unwrap_or("")
180            .to_string()
181    });
182
183    // Step 2 — pkg-named convention.
184    if !pkg_name.is_empty() {
185        let pkg_named = pkg_dir
186            .join("launch")
187            .join(format!("{pkg_name}.launch.xml"));
188        if pkg_named.is_file() {
189            return Ok(LaunchInput::File(pkg_named));
190        }
191    }
192
193    // Step 3 — system.launch.xml convention.
194    let system_launch = pkg_dir.join("launch").join("system.launch.xml");
195    if system_launch.is_file() {
196        return Ok(LaunchInput::File(system_launch));
197    }
198
199    // Step 4 — single *.launch.xml.
200    let candidates = enumerate_launch_files(pkg_dir);
201    match candidates.len() {
202        0 => {
203            // Steps 5/6: synth eligibility check.
204            if is_self_bringup_eligible(pkg_dir) {
205                let exec_name = match exec_arg {
206                    Some(e) => e.to_string(),
207                    None => discover_exec_target(pkg_dir, &pkg_name)?,
208                };
209                Ok(LaunchInput::Synth(synthesise_xml(&pkg_name, &exec_name)))
210            } else {
211                Err(LaunchResolveError::PathABringupNoLaunch {
212                    pkg: pkg_dir.to_path_buf(),
213                }
214                .into())
215            }
216        }
217        1 => Ok(LaunchInput::File(
218            pkg_dir.join("launch").join(&candidates[0]),
219        )),
220        _ => Err(LaunchResolveError::AmbiguousMultiLaunch {
221            pkg: pkg_dir.to_path_buf(),
222            candidates,
223        }
224        .into()),
225    }
226}

228/// True when `pkg_dir` is a component or application pkg eligible for
229/// synthesis: has a `Cargo.toml` or `CMakeLists.txt`. Path A bringup
230/// pkgs (no Cargo.toml + no CMakeLists.txt + has system.toml) are
231/// disqualified.
232fn is_self_bringup_eligible(pkg_dir: &Path) -> bool {
233    let has_cargo = pkg_dir.join("Cargo.toml").is_file();
234    let has_cmake = pkg_dir.join("CMakeLists.txt").is_file();
235    has_cargo || has_cmake
236}

238/// Enumerate `*.launch.xml` files directly under `<pkg>/launch/` (no
239/// recursion). Returns sorted basenames.
240pub fn enumerate_launch_files(pkg_dir: &Path) -> Vec<String> {
241    let dir = pkg_dir.join("launch");
242    let mut out = Vec::new();
243    let Ok(rd) = fs::read_dir(&dir) else {
244        return out;
245    };
246    for entry in rd.flatten() {
247        let path = entry.path();
248        if !path.is_file() {
249            continue;
250        }
251        let Some(name) = path.file_name().and_then(|s| s.to_str()) else {
252            continue;
253        };
254        if name.ends_with(".launch.xml") {
255            out.push(name.to_string());
256        }
257    }
258    out.sort();
259    out
260}

262/// Read `pkg-name`:
263/// * Rust: `Cargo.toml` `[package].name`
264/// * C++: `project(<name> …)` in `CMakeLists.txt`
265pub fn discover_pkg_name(pkg_dir: &Path) -> Result<String> {
266    let cargo = pkg_dir.join("Cargo.toml");
267    if cargo.is_file() {
268        if let Some(name) = read_cargo_pkg_name(&cargo)? {
269            return Ok(name);
270        }
271    }
272    let cmake = pkg_dir.join("CMakeLists.txt");
273    if cmake.is_file() {
274        if let Some(name) = read_cmake_project_name(&cmake)? {
275            return Ok(name);
276        }
277    }
278    Err(LaunchResolveError::UnknownPkgName {
279        pkg: pkg_dir.to_path_buf(),
280    }
281    .into())
282}

284fn read_cargo_pkg_name(cargo_toml: &Path) -> Result<Option<String>> {
285    let raw =
286        fs::read_to_string(cargo_toml).map_err(|e| eyre!("read {}: {e}", cargo_toml.display()))?;
287    let doc: toml::Value =
288        toml::from_str(&raw).map_err(|e| eyre!("parse {}: {e}", cargo_toml.display()))?;
289    Ok(doc
290        .get("package")
291        .and_then(|t| t.get("name"))
292        .and_then(|v| v.as_str())
293        .map(|s| s.to_string()))
294}
295
296fn read_cmake_project_name(cmake: &Path) -> Result<Option<String>> {
297    let raw = fs::read_to_string(cmake).map_err(|e| eyre!("read {}: {e}", cmake.display()))?;
298    // Minimal scan: `project(<name> …)` — first `project(` call wins.
299    // Whitespace-tolerant; strips a trailing version token if present.
300    for line in raw.lines() {
301        let trimmed = line.trim();
302        if trimmed.starts_with('#') {
303            continue;
304        }
305        if let Some(rest) = trimmed
306            .strip_prefix("project(")
307            .or_else(|| trimmed.strip_prefix("project ("))
308        {
309            // first token up to whitespace, ')', or 'VERSION'
310            let mut name = String::new();
311            for ch in rest.chars() {
312                if ch.is_whitespace() || ch == ')' {
313                    break;
314                }
315                name.push(ch);
316            }
317            if !name.is_empty() {
318                return Ok(Some(name));
319            }
320        }
321    }
322    Ok(None)
323}

325/// Resolve the executable target name for synthesis:
326///
327/// * `--exec <name>` arg wins.
328/// * Single `[[bin]]` (Rust) → that bin's `name`.
329/// * Single `add_executable(<name> …)` (C++) → that target.
330/// * Component pkg w/ no bins (only `[lib]`) → the pkg name (codegen
331///   synth-main convention).
332/// * Multiple candidates → hard error.
333pub fn discover_exec_target(pkg_dir: &Path, pkg_name: &str) -> Result<String> {
334    // Rust path.
335    let cargo = pkg_dir.join("Cargo.toml");
336    if cargo.is_file() {
337        let bins = read_cargo_bins(&cargo)?;
338        match bins.len() {
339            0 => {
340                // Component pkg with no bins → use pkg name (codegen
341                // synthesises a main bin named after the pkg).
342                return Ok(pkg_name.to_string());
343            }
344            1 => return Ok(bins.into_iter().next().unwrap()),
345            _ => {
346                return Err(LaunchResolveError::AmbiguousExec {
347                    pkg: pkg_dir.to_path_buf(),
348                    candidates: bins,
349                }
350                .into());
351            }
352        }
353    }
354
355    // C++ path.
356    let cmake = pkg_dir.join("CMakeLists.txt");
357    if cmake.is_file() {
358        let execs = read_cmake_add_executables(&cmake)?;
359        match execs.len() {
360            0 => {
361                return Err(LaunchResolveError::UnknownExec {
362                    pkg: pkg_dir.to_path_buf(),
363                }
364                .into());
365            }
366            1 => return Ok(execs.into_iter().next().unwrap()),
367            _ => {
368                return Err(LaunchResolveError::AmbiguousExec {
369                    pkg: pkg_dir.to_path_buf(),
370                    candidates: execs,
371                }
372                .into());
373            }
374        }
375    }
376
377    Err(LaunchResolveError::UnknownExec {
378        pkg: pkg_dir.to_path_buf(),
379    }
380    .into())
381}

383/// Read `[[bin]]` names from a `Cargo.toml`. An implicit bin
384/// (`src/main.rs` w/o `[[bin]]`) defaults to the pkg name — captured by
385/// the no-bin → pkg-name fallback in [`discover_exec_target`].
386fn read_cargo_bins(cargo_toml: &Path) -> Result<Vec<String>> {
387    let raw =
388        fs::read_to_string(cargo_toml).map_err(|e| eyre!("read {}: {e}", cargo_toml.display()))?;
389    let doc: toml::Value =
390        toml::from_str(&raw).map_err(|e| eyre!("parse {}: {e}", cargo_toml.display()))?;
391    let mut out = Vec::new();
392    if let Some(bin_array) = doc.get("bin").and_then(|v| v.as_array()) {
393        for b in bin_array {
394            if let Some(name) = b.get("name").and_then(|v| v.as_str()) {
395                out.push(name.to_string());
396            }
397        }
398    }
399    Ok(out)
400}

402/// Scan `add_executable(<name> …)` calls in a CMakeLists.txt and return
403/// the named targets. Minimal text scan — sufficient for the in-tree
404/// canonical example shapes; non-trivial CMake-driven exec discovery is
405/// out of scope for synthesis.
406fn read_cmake_add_executables(cmake: &Path) -> Result<Vec<String>> {
407    let raw = fs::read_to_string(cmake).map_err(|e| eyre!("read {}: {e}", cmake.display()))?;
408    let mut out = Vec::new();
409    for line in raw.lines() {
410        let trimmed = line.trim();
411        if trimmed.starts_with('#') {
412            continue;
413        }
414        if let Some(rest) = trimmed
415            .strip_prefix("add_executable(")
416            .or_else(|| trimmed.strip_prefix("add_executable ("))
417        {
418            let mut name = String::new();
419            for ch in rest.chars() {
420                if ch.is_whitespace() || ch == ')' {
421                    break;
422                }
423                name.push(ch);
424            }
425            if !name.is_empty() {
426                out.push(name);
427            }
428        }
429    }
430    Ok(out)
431}

433/// Render the synth XML body for a single-node bringup.
434pub fn synthesise_xml(pkg_name: &str, exec_name: &str) -> String {
435    format!(
436        "<?xml version=\"1.0\"?>\n\
437         <!-- synthesised by nros (Phase 212.L.6) — not persisted to disk -->\n\
438         <launch>\n  <node pkg=\"{}\" exec=\"{}\" />\n</launch>\n",
439        xml_escape(pkg_name),
440        xml_escape(exec_name),
441    )
442}
443
444fn xml_escape(s: &str) -> String {
445    let mut out = String::with_capacity(s.len());
446    for ch in s.chars() {
447        match ch {
448            '<' => out.push_str("&lt;"),
449            '>' => out.push_str("&gt;"),
450            '&' => out.push_str("&amp;"),
451            '"' => out.push_str("&quot;"),
452            '\'' => out.push_str("&apos;"),
453            c => out.push(c),
454        }
455    }
456    out
457}
