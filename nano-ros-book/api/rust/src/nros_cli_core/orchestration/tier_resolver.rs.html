1//! Tier resolver — re-export shim (Phase 228.E).
2//!
3//! The resolver + its result types moved to the shared
4//! [`nros_orchestration_ir`] crate so the `nros::main!()` proc-macro can run
5//! the exact same resolution at compile time (the CLI can't be a proc-macro
6//! dep). This module re-exports them so existing `orchestration::tier_resolver::*`
7//! references stay valid.
8//!
9//! The one shape change: [`resolve_tiers`] now takes the decomposed
10//! `system.toml` pieces (`tiers`, `node_overrides`, `component_names`,
11//! `callback_groups`) instead of a whole `SystemToml`, so the leaf crate stays
12//! free of the full CLI config type. See [`crate::cmd::codegen_system`] for the
13//! call site that adapts a `SystemToml` to it.

25/// Phase 228 / 256 W4.2 — collect each system component's declared
26/// `callback_groups` from the component-package metadata in `cfg`, keyed by the
27/// system `[[component]].name` (the instance name `ResolvedTier.members` use).
28/// Shared by the C bake (`codegen_system`) and the Rust codegen (`generate`).
29pub fn collect_callback_groups(
30    cfg: &NrosConfig,
31    components: &[SystemComponentEntry],
32) -> BTreeMap<String, Vec<CallbackGroupDecl>> {
33    let mut map = BTreeMap::new();
34    for c in components {
35        let Some(pkg) = cfg.component_packages.get(&c.pkg) else {
36            continue;
37        };
38        // Single-node pkg → node_or_component; multi-node pkg → match by name/class.
39        let groups = pkg
40            .nros
41            .node_or_component()
42            .filter(|m| !m.callback_groups.is_empty())
43            .map(|m| m.callback_groups.clone())
44            .or_else(|| {
45                pkg.nros
46                    .nodes_or_components()
47                    .values()
48                    .find(|m| {
49                        m.name.as_deref() == Some(c.name.as_str())
50                            || m.class.as_deref() == Some(c.class.as_str())
51                    })
52                    .map(|m| m.callback_groups.clone())
53            })
54            .unwrap_or_default();
55        if !groups.is_empty() {
56            map.insert(c.name.clone(), groups);
57        }
58    }
59    map
60}

62/// Adapt a full [`SystemToml`] + its components' `callback_groups` to the
63/// decomposed [`resolve_tiers`] signature.
64pub fn resolve_system_tiers(
65    system: &SystemToml,
66    callback_groups: &BTreeMap<String, Vec<CallbackGroupDecl>>,
67    target_rtos: &str,
68) -> Result<ResolvedTierTable, TierResolveError> {
69    let component_names: BTreeSet<&str> =
70        system.components.iter().map(|c| c.name.as_str()).collect();
71    resolve_tiers(
72        &system.tiers,
73        &system.node_overrides,
74        &component_names,
75        callback_groups,
76        target_rtos,
77    )
78}

80/// Best-effort RTOS name for tier resolution from the selected deploy target.
81/// Defaults to `posix` (native); embedded targets refine it from the deploy
82/// `board`/`kind` hint.
83pub fn derive_target_rtos(system: &SystemToml, target: Option<&str>) -> String {
84    target
85        .and_then(|t| system.deploy.get(t))
86        .and_then(|d| d.board.as_deref().or(d.kind.as_deref()))
87        .map(|hint| {
88            for rtos in ["freertos", "zephyr", "threadx", "nuttx"] {
89                if hint.contains(rtos) {
90                    return rtos.to_string();
91                }
92            }
93            "posix".to_string()
94        })
95        .unwrap_or_else(|| "posix".to_string())
96}
