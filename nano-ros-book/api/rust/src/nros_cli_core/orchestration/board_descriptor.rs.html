1//! Data-driven board profiles (Phase 195.C).
2//!
3//! The `nros` CLI is shipped from a *separate* repo, so it must carry **no**
4//! baked-in knowledge of the nano-ros workspace layout. Every per-board fact
5//! — which board crate to depend on, the rustc target, the `.cargo/config.toml`
6//! body, the kernel-port / libc paths, the generated entry-point shape — lives
7//! in a `nros-board.toml` descriptor *in the workspace* and is read at runtime.
8//!
9//! Discovery is uniform: every `packages/boards/*/nros-board.toml` is loaded
10//! (crate-backed boards put the file in their crate dir; the crate-less host
11//! boards — `posix`, `zephyr`, `orin-spe` — get a descriptor-only dir under
12//! `packages/boards/`). A file holds a `[[board]]` array so one crate can back
13//! several boards (e.g. `nros-board-stm32f4` → `stm32f429` + `stm32f407`,
14//! differing only by `chip`).
15//!
16//! Layout paths in `cargo_config` are stored **relative** and written with the
17//! `${workspace}` placeholder; the CLI substitutes the workspace root it
18//! discovered at render time, so the binary stays workspace-agnostic.
19

40/// Rust toolchain a generated package pins.

84/// The per-board pieces the entry-point renderer interpolates into the shared
85/// board-run entry shape. `None` path interpolation here — these reference only
86/// the board crate name.
87#[derive(Debug, Clone, Deserialize)]
88pub struct BoardEntry {
89    /// Board rlib invoked as `<crate>::run(<crate>::Config::default(), ..)`.
90    pub crate_name: String,
91    /// Doc comment emitted directly above the entry fn.
92    #[serde(default)]
93    pub comment: String,
94    /// Attribute(s) + `fn` signature line(s) preceding the fn body.
95    pub signature: String,
96    /// Crate-root `use`s / items pinned above the entry (panic handler, etc.).
97    #[serde(default)]
98    pub crate_root_extra: String,
99    /// Builder-chain suffix appended inside the closure; empty for most boards.
100    #[serde(default)]
101    pub closure_extra: String,
102}

104/// One board profile. Mirrors the old hardcoded `PlatformProfile` +
105/// `BoardEntry`, but every field is owned data read from `nros-board.toml`.
106#[derive(Debug, Clone, Deserialize)]
107pub struct BoardDescriptor {
108    /// Board name + accepted aliases (the values a user passes as `board`).
109    pub names: Vec<String>,
110    pub platform: PlatformKind,
111    /// rustc target triple this board pins, if any (`None` → take from plan).
112    #[serde(default)]
113    pub target: Option<String>,
114    pub toolchain: Toolchain,
115    /// The `nros/<feature>` selected (e.g. `platform-posix`).
116    pub platform_feature: String,
117    /// Extra local default-feature aliases beyond `nros/<feature>`.
118    #[serde(default)]
119    pub local_aliases: Vec<String>,
120    pub link_kind: LinkKind,
121    pub entry_kind: EntryKind,
122    pub net_stack: NetStack,
123    /// esp-hal / stm32 chip feature; `None` for non-chip platforms.
124    #[serde(default)]
125    pub chip: Option<String>,
126    /// Board crate to depend on; `None` for crate-less host boards
127    /// (posix / zephyr / orin-spe) that pull static or `nros-platform-cffi` deps.
128    #[serde(default)]
129    pub board_crate: Option<String>,
130    /// Board-crate path relative to the workspace root; defaults to
131    /// `packages/boards/<board_crate>` when omitted.
132    #[serde(default)]
133    pub crate_path: Option<String>,
134    /// Extra features to enable on the board crate dependency.
135    #[serde(default)]
136    pub board_features: Vec<String>,
137    /// Verbatim `.cargo/config.toml` body, with `${workspace}` placeholders for
138    /// any layout path. `None` for boards that need no config (posix/zephyr/…).
139    #[serde(default)]
140    pub cargo_config: Option<String>,
141    /// Generated entry-point pieces; `None` for hosted boards that emit the
142    /// default `fn main` shape.
143    #[serde(default)]
144    pub entry: Option<BoardEntry>,
145    /// Disambiguate two descriptors sharing a `names` entry by requiring this
146    /// substring in the requested target (e.g. `"riscv64"` for threadx-riscv64,
147    /// so `board = "threadx"` picks riscv64 vs linux by target).
148    #[serde(default)]
149    pub target_contains: Option<String>,
150}
151
152impl BoardDescriptor {
153    /// Board-crate path relative to the workspace root, applying the
154    /// `packages/boards/<board_crate>` default.
155    pub fn crate_path_rel(&self) -> Option<String> {
156        self.crate_path.clone().or_else(|| {
157            self.board_crate
158                .as_ref()
159                .map(|c| format!("packages/boards/{c}"))
160        })
161    }
162
163    /// Render `cargo_config` with `${workspace}` resolved to `workspace`.
164    pub fn cargo_config_rendered(&self, workspace: &Path) -> Option<String> {
165        let ws = path_for_template(workspace);
166        self.cargo_config
167            .as_ref()
168            .map(|body| body.replace("${workspace}", &ws))
169    }
170}
171
172/// Escape a path for embedding inside a double-quoted TOML string.
173fn path_for_template(path: &Path) -> String {
174    path.to_string_lossy()
175        .replace('\\', "\\\\")
176        .replace('"', "\\\"")
177}

179#[derive(Debug, Deserialize)]
180struct BoardFile {
181    #[serde(default, rename = "board")]
182    boards: Vec<BoardDescriptor>,
183}
184
185/// Every board descriptor discovered under `<workspace>/packages/boards`.
186#[derive(Debug, Default)]
187pub struct BoardCatalog {
188    descriptors: Vec<BoardDescriptor>,
189}
190
191impl BoardCatalog {
192    /// Load every `packages/boards/*/nros-board.toml` under `workspace`.
193    pub fn load(workspace: &Path) -> Result<Self, BoardLoadError> {
194        let boards_dir = workspace.join("packages/boards");
195        let mut descriptors = Vec::new();
196        let entries = std::fs::read_dir(&boards_dir)
197            .map_err(|e| BoardLoadError::Io(boards_dir.clone(), e))?;
198        for entry in entries {
199            let entry = entry.map_err(|e| BoardLoadError::Io(boards_dir.clone(), e))?;
200            let descriptor_path = entry.path().join("nros-board.toml");
201            if !descriptor_path.is_file() {
202                continue;
203            }
204            let text = std::fs::read_to_string(&descriptor_path)
205                .map_err(|e| BoardLoadError::Io(descriptor_path.clone(), e))?;
206            let file: BoardFile = toml::from_str(&text)
207                .map_err(|e| BoardLoadError::Parse(descriptor_path.clone(), e))?;
208            descriptors.extend(file.boards);
209        }
210        Ok(Self { descriptors })
211    }
212
213    /// Build a catalog from already-parsed descriptors (tests / in-memory).
214    pub fn from_descriptors(descriptors: Vec<BoardDescriptor>) -> Self {
215        Self { descriptors }
216    }
217
218    pub fn descriptors(&self) -> &[BoardDescriptor] {
219        &self.descriptors
220    }
221
222    /// Resolve a `(board, target)` pair to its descriptor.
223    ///
224    /// A board name may be claimed by two descriptors (e.g. `threadx` →
225    /// `threadx-riscv64` vs `threadx-linux`); the one whose `target_contains`
226    /// matches the requested target wins, else the unconstrained one. As a last
227    /// resort an unknown board on a `*-linux*` target resolves to `posix`
228    /// (mirrors the old `target.contains("linux")` fallback).
229    pub fn resolve(&self, board: &str, target: &str) -> Option<&BoardDescriptor> {
230        let named: Vec<&BoardDescriptor> = self
231            .descriptors
232            .iter()
233            .filter(|d| d.names.iter().any(|n| n == board))
234            .collect();
235        if !named.is_empty() {
236            // Prefer a target-qualified match, then the unconstrained one.
237            return named
238                .iter()
239                .find(|d| {
240                    d.target_contains
241                        .as_ref()
242                        .is_some_and(|sub| target.contains(sub.as_str()))
243                })
244                .or_else(|| named.iter().find(|d| d.target_contains.is_none()))
245                .copied();
246        }
247        if target.contains("linux") {
248            return self
249                .descriptors
250                .iter()
251                .find(|d| d.platform == PlatformKind::Posix);
252        }
253        None
254    }
255}
256
257/// Error loading or parsing board descriptors.
258#[derive(Debug)]
259pub enum BoardLoadError {
260    Io(std::path::PathBuf, std::io::Error),
261    Parse(std::path::PathBuf, toml::de::Error),
262}
263
264impl std::fmt::Display for BoardLoadError {
265    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
266        match self {
267            BoardLoadError::Io(path, e) => write!(f, "reading {}: {e}", path.display()),
268            BoardLoadError::Parse(path, e) => write!(f, "parsing {}: {e}", path.display()),
269        }
270    }
271}

326        let cat = catalog();
