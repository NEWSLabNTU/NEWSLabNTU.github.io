3use serde_json::{Map, Value};
4use std::path::{Path, PathBuf};
5
6#[derive(Debug, Clone)]
7pub struct ParameterInputs<'a> {
8    pub source_metadata: Option<&'a Value>,
9    pub package_nros: Option<&'a Value>,
10    pub launch_params: &'a [(String, String)],
11    pub param_files: &'a [String],
12    pub overlays: &'a [Value],
13}
14
15pub fn effective_parameters(inputs: ParameterInputs<'_>) -> Value {
16    let mut out = Map::new();
17    if let Some(metadata) = inputs.source_metadata {
18        merge_object(&mut out, metadata.get("parameter_defaults"));
19        merge_object(&mut out, metadata.pointer("/parameters/defaults"));
20        merge_source_parameter_array(&mut out, metadata.get("parameters"));
21        merge_object(&mut out, metadata.get("parameters"));
22    }
23    if let Some(package_nros) = inputs.package_nros {
24        merge_object(&mut out, package_nros.get("parameters"));
25    }
26    if !inputs.param_files.is_empty() {
27        out.insert(
28            "parameter_files".to_string(),
29            Value::Array(
30                inputs
31                    .param_files
32                    .iter()
33                    .map(|path| Value::String(path.clone()))
34                    .collect(),
35            ),
36        );
37    }
38    for (key, value) in inputs.launch_params {
39        out.insert(key.clone(), parse_scalar(value));
40    }
41    for overlay in inputs.overlays {
42        merge_object(&mut out, overlay.get("parameters"));
43        merge_object(&mut out, overlay.pointer("/overlays/parameters"));
44    }
45    Value::Object(out)
46}

48/// Phase 256 Wave 0 — a parsed `nros.toml` overlay paired with the file it came
49/// from. The provenance primitive the config-SSoT endgame needs: it lets the
50/// per-block deprecation warnings (Waves 1-5) NAME the offending file, and powers
51/// `nros config show` provenance + `nros check`'s legacy-overlay flag (Waves 6-7).
52#[derive(Debug, Clone)]
53pub struct SourcedToml {
54    pub path: PathBuf,
55    pub value: Value,
56}
57
58/// Parse each `nros.toml` path into a [`SourcedToml`], preserving file
59/// attribution. Same parse as [`load_toml_values`] — that fn is now a thin
60/// projection of this (drops the path).
61pub fn load_sourced_toml_values(paths: &[PathBuf]) -> eyre::Result<Vec<SourcedToml>> {
62    paths
63        .iter()
64        .map(|path| {
65            let raw = std::fs::read_to_string(path)?;
66            let value: toml::Value = toml::from_str(&raw)?;
67            Ok(SourcedToml {
68                path: path.clone(),
69                value: serde_json::to_value(value)?,
70            })
71        })
72        .collect()
73}
74
75pub fn load_toml_values(paths: &[PathBuf]) -> eyre::Result<Vec<Value>> {
76    Ok(load_sourced_toml_values(paths)?
77        .into_iter()
78        .map(|s| s.value)
79        .collect())
80}

82/// The file that **last** declared top-level `block` across the sourced overlays,
83/// or `None` if no overlay carries it. Last-wins matches the overlay merge
84/// semantics (`schema_build_json` / `collect_*` all let later overlays override
85/// earlier), so this is the file whose value actually reached the plan — the one
86/// a deprecation warning or provenance column should name.
87pub fn last_block_source<'a>(sourced: &'a [SourcedToml], block: &str) -> Option<&'a Path> {
88    sourced
89        .iter()
90        .rev()
91        .find(|s| s.value.get(block).is_some())
92        .map(|s| s.path.as_path())
93}

95fn merge_source_parameter_array(out: &mut Map<String, Value>, value: Option<&Value>) {
96    let Some(Value::Array(parameters)) = value else {
97        return;
98    };
99    for parameter in parameters {
100        let Some(name) = parameter.get("name").and_then(Value::as_str) else {
101            continue;
102        };
103        let Some(default) = parameter.get("default") else {
104            continue;
105        };
106        out.insert(name.to_string(), default.clone());
107    }
108}
109
110fn merge_object(out: &mut Map<String, Value>, value: Option<&Value>) {
111    let Some(Value::Object(map)) = value else {
112        return;
113    };
114    for (key, value) in map {
115        out.insert(key.clone(), value.clone());
116    }
117}
118
119fn parse_scalar(value: &str) -> Value {
120    if let Ok(parsed) = value.parse::<bool>() {
121        return Value::Bool(parsed);
122    }
123    if let Ok(parsed) = value.parse::<i64>() {
124        return Value::Number(parsed.into());
125    }
126    if let Ok(parsed) = value.parse::<f64>()
127        && let Some(number) = serde_json::Number::from_f64(parsed)
128    {
129        return Value::Number(number);
130    }
131    Value::String(value.to_string())
132}

135mod tests {
136    use super::*;
137    use serde_json::json;

139    #[test]
140    fn launch_params_override_source_defaults() {
141        let source = json!({"parameter_defaults": {"rate": 10, "frame": "map"}});
142        let launch = vec![("rate".to_string(), "20".to_string())];
143        let value = effective_parameters(ParameterInputs {
144            source_metadata: Some(&source),
145            package_nros: None,
146            launch_params: &launch,
147            param_files: &[],
148            overlays: &[],
149        });
150        assert_eq!(value["rate"], 20);
151        assert_eq!(value["frame"], "map");
152    }

154    /// Phase 256 Wave 0 — the sourced loader keeps file attribution, and
155    /// `last_block_source` returns the LAST overlay declaring a block (last-wins,
156    /// matching the merge), or `None` when no overlay carries it.
157    #[test]
158    fn sourced_toml_tracks_provenance_per_block() {
159        let dir = tempfile::tempdir().unwrap();
160        let a = dir.path().join("a.toml");
161        let b = dir.path().join("b.toml");
162        std::fs::write(
163            &a,
164            "[lifecycle]\nautostart=\"all\"\n[build]\nrmw=\"zenoh\"\n",
165        )
166        .unwrap();
167        std::fs::write(&b, "[build]\nrmw=\"cyclonedds\"\n").unwrap();
168
169        let sourced = load_sourced_toml_values(&[a.clone(), b.clone()]).unwrap();
170        assert_eq!(sourced.len(), 2);
171        assert_eq!(sourced[0].path, a);
172        assert_eq!(sourced[0].value["lifecycle"]["autostart"], "all");
173
174        // `[build]` last-declared in b; `[lifecycle]` only in a.
175        assert_eq!(last_block_source(&sourced, "build"), Some(b.as_path()));
176        assert_eq!(last_block_source(&sourced, "lifecycle"), Some(a.as_path()));
177        assert_eq!(last_block_source(&sourced, "nonexistent"), None);
178
179        // `load_toml_values` is the path-dropping projection of the same parse.
180        let plain = load_toml_values(&[a, b]).unwrap();
181        assert_eq!(
182            plain,
183            sourced.into_iter().map(|s| s.value).collect::<Vec<_>>()
184        );
185    }
