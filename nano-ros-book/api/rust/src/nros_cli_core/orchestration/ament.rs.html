1//! Phase 212.B.4 — `[package.metadata.ament]` reader entry point.
2//!
3//! The strict schema lives at [`cargo_metadata_schema::PackageMetadataAment`];
4//! this module is the path-based entry point that callers (`nros emit
5//! package-xml`'s `render_for_pkg`, `nros check`, and the workspace
6//! loader's per-pkg cargo metadata sweep) reach for when they have a pkg
7//! dir on disk and want the parsed table without re-implementing the
8//! Cargo.toml read-and-extract.
9//!
10//! Usage:
11//!
12//! ```ignore
13//! use std::path::Path;
14//! use nros_cli_core::orchestration::ament::parse_ament_metadata;
15//!
16//! let ament = parse_ament_metadata(Path::new("./talker_pkg"))?;
17//! println!("license = {:?}", ament.license);
18//! ```
19//!
20//! Behaviour:
21//!
22//! * `<pkg-dir>/Cargo.toml` absent → returns `PackageMetadataAment::default()`
23//!   (no ament block authored).
24//! * Present but no `[package.metadata.ament]` → same.
25//! * Present and authored → strict-mode parse; serde rejects unknown
26//!   fields with the underlying diagnostic verbatim.
27
28use std::path::Path;
29
30use eyre::{Context, Result, eyre};
31use serde::Deserialize;
32
33use super::cargo_metadata_schema::PackageMetadataAment;
34
35/// Re-export so callsites can write `ament::AmentMetadata` if they prefer
36/// the role-named alias to the schema struct name.
37pub type AmentMetadata = PackageMetadataAment;
38
39/// Read `<pkg-dir>/Cargo.toml` and return its `[package.metadata.ament]`
40/// table parsed against the strict schema. Returns
41/// `PackageMetadataAment::default()` when the manifest exists but the
42/// table is absent — matching the policy of every other Phase 212
43/// per-pkg reader (silent default is fine; only typos surface as
44/// hard errors).
45pub fn parse_ament_metadata(pkg_dir: &Path) -> Result<AmentMetadata> {
46    let cargo_toml = pkg_dir.join("Cargo.toml");
47    if !cargo_toml.is_file() {
48        return Ok(AmentMetadata::default());
49    }
50    let raw = std::fs::read_to_string(&cargo_toml)
51        .with_context(|| format!("read {}", cargo_toml.display()))?;
52    let manifest: AmentBearingManifest =
53        toml::from_str(&raw).with_context(|| format!("parse {}", cargo_toml.display()))?;
54    let ament = manifest
55        .package
56        .and_then(|p| p.metadata)
57        .and_then(|m| m.ament)
58        .unwrap_or_default();
59    Ok(ament)
60}

62/// Minimal `Cargo.toml` projection that exposes only the ament sub-table.
63/// We avoid `#[serde(deny_unknown_fields)]` here on the outer structs
64/// because a real Cargo manifest carries dozens of fields we don't model
65/// — the strictness lives on [`PackageMetadataAment`] itself.
66#[derive(Debug, Deserialize)]
67struct AmentBearingManifest {
68    #[serde(default)]
69    package: Option<AmentBearingPackage>,
70}
71
72#[derive(Debug, Deserialize)]
73struct AmentBearingPackage {
74    #[serde(default)]
75    metadata: Option<AmentBearingPackageMetadata>,
76}
77
78#[derive(Debug, Deserialize)]
79struct AmentBearingPackageMetadata {
80    #[serde(default)]
81    ament: Option<PackageMetadataAment>,
82}
83
84/// Convenience: parse a raw `[package.metadata.ament]` table out of an
85/// already-deserialised `serde_json::Value` (the shape `cargo metadata`
86/// hands back). Mirrors the in-loader fn in
87/// [`super::nros_config`] so external callers can reuse the same parse
88/// path without going through cargo metadata themselves.
89///
90/// Returns `Ok(PackageMetadataAment::default())` when the input value has
91/// no `ament` key.
92pub fn parse_ament_metadata_value(value: &serde_json::Value) -> Result<AmentMetadata> {
93    let Some(ament) = value.get("ament") else {
94        return Ok(AmentMetadata::default());
95    };
96    PackageMetadataAment::deserialize(ament.clone())
97        .map_err(|e| eyre!("invalid [package.metadata.ament]: {e}"))
98}

100#[cfg(test)]
101mod tests {
102    use super::*;
103    use std::fs;
104    use std::path::PathBuf;
105    use std::sync::atomic::{AtomicU64, Ordering};
106    use std::time::{SystemTime, UNIX_EPOCH};
107
108    static SEQ: AtomicU64 = AtomicU64::new(0);
109
110    fn scratch_pkg(name: &str) -> PathBuf {
111        let n = SEQ.fetch_add(1, Ordering::Relaxed);
112        let stamp = SystemTime::now()
113            .duration_since(UNIX_EPOCH)
114            .unwrap()
115            .as_nanos();
116        let dir = std::env::temp_dir().join(format!(
117            "nros-ament-{name}-{}-{stamp}-{n}",
118            std::process::id()
119        ));
120        let _ = fs::remove_dir_all(&dir);
121        fs::create_dir_all(&dir).expect("scratch");
122        dir
123    }
124
125    #[test]
126    fn parse_ament_metadata_basic_round_trip() {
127        let p = scratch_pkg("basic");
128        fs::write(
129            p.join("Cargo.toml"),
130            r#"
131[package]
132name = "talker_pkg"
133version = "0.1.0"
134edition = "2021"
135
136[package.metadata.ament]
137description = "Talker."
138license = "Apache-2.0"
139maintainer = { name = "Ada Lovelace", email = "ada@example.com" }
140buildtool_depend = ["ament_cargo"]
141exec_depend = ["std_msgs", "rcl_interfaces"]
142build_depend = ["std_msgs"]
143"#,
144        )
145        .unwrap();
146        let ament = parse_ament_metadata(&p).expect("parse");
147        assert_eq!(ament.description.as_deref(), Some("Talker."));
148        assert_eq!(ament.license.as_deref(), Some("Apache-2.0"));
149        let m = ament.maintainer.as_ref().expect("maintainer");
150        assert_eq!(m.name, "Ada Lovelace");
151        assert_eq!(m.email, "ada@example.com");
152        assert_eq!(ament.buildtool_depend, vec!["ament_cargo"]);
153        assert_eq!(ament.exec_depend, vec!["std_msgs", "rcl_interfaces"]);
154        assert_eq!(ament.build_depend, vec!["std_msgs"]);
155    }
156
157    #[test]
158    fn parse_ament_metadata_returns_default_when_no_cargo_toml() {
159        let p = scratch_pkg("nocargo");
160        // No Cargo.toml at all.
161        let ament = parse_ament_metadata(&p).expect("parse");
162        assert_eq!(ament, AmentMetadata::default());
163    }

173version = "0.1.0"
174edition = "2021"
175"#,

178        let ament = parse_ament_metadata(&p).expect("parse");
179        assert_eq!(ament, AmentMetadata::default());
180    }
181
182    #[test]
183    fn parse_ament_metadata_rejects_typo() {
184        let p = scratch_pkg("typo");
185        fs::write(
186            p.join("Cargo.toml"),
187            r#"
188[package]
189name = "p"
190version = "0.1.0"
191edition = "2021"
192
193[package.metadata.ament]
194descripshun = "typo here"
195"#,
196        )
197        .unwrap();
198        let err = parse_ament_metadata(&p).expect_err("typo must reject");
199        let msg = format!("{err:#}");
200        assert!(
201            msg.contains("descripshun") || msg.contains("unknown field"),
202            "diagnostic: {msg}"
203        );
204    }
