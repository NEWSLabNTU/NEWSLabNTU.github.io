1//! Phase 215.C.2 — strict reader for `[package.metadata.nros.board]`.
2//!
3//! Each `packages/boards/nros-board-*` crate carries TWO faces of the same
4//! board manifest:
5//!
6//! * `board.cmake` — sidecar consumed by the Zephyr cmake module
7//!   `zephyr/cmake/nano_ros_use_board.cmake`.
8//! * `[package.metadata.nros.board]` in `Cargo.toml` — consumed by Rust /
9//!   `nros` CLI tooling (drift audit, `nros board info`).
10//!
11//! Both must stay in lock-step; Phase 215.F drift audit guards that
12//! invariant. This module is the strict reader for the Cargo.toml face —
13//! mirrors the discipline of `cargo_metadata_schema.rs` (Phase 212.B):
14//! `deny_unknown_fields` so typos surface at parse time instead of being
15//! silently dropped.

21/// `[package.metadata.nros.board]` — strict schema.
22///
23/// Field semantics mirror the `NROS_BOARD_*` cmake variables defined in
24/// Phase 215.A.1:
25///
26/// | Cargo field         | `board.cmake` variable        |
27/// |---------------------|-------------------------------|
28/// | `zephyr_board`      | `NROS_BOARD_ZEPHYR_ID`        |
29/// | `toolchain`         | `NROS_BOARD_TOOLCHAIN`        |
30/// | `gated`             | `NROS_BOARD_GATED_PKGS`       |
31/// | `default_rmw`       | `NROS_BOARD_DEFAULT_RMW`      |
32/// | `default_transport` | `NROS_BOARD_DEFAULT_TRANSPORT`|
33/// | `runner`            | `NROS_BOARD_RUNNER`           |
34/// | `prj_conf`          | `NROS_BOARD_PRJ_CONF`         |
35/// | `board_conf`        | `NROS_BOARD_BOARD_CONF`       |
36/// | `board_overlay`     | `NROS_BOARD_BOARD_OVERLAY`    |
37///
38/// `prj_conf` / `board_conf` / `board_overlay` are RELATIVE to the
39/// host `Cargo.toml`'s directory (the cmake face stores the absolute
40/// form after resolution via the board crate dir).
41#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
42#[serde(deny_unknown_fields)]
43pub struct BoardMetadata {
44    /// Zephyr `BOARD` string (`fvp_baser_aemv8r/fvp_aemv8r_aarch64/smp`).
45    pub zephyr_board: String,
46    /// SDK abi target (e.g. `aarch64-zephyr-elf`).
47    pub toolchain: String,
48    /// Optional semicolon-list (in cmake) of `[features.<flag>]` gates.
49    /// Defaults to empty — boards w/o gated pkgs simply omit the field.
50    #[serde(default, skip_serializing_if = "Vec::is_empty")]
51    pub gated: Vec<String>,
52    /// `cyclonedds` / `zenoh` / `xrce`.
53    pub default_rmw: String,
54    /// `ethernet` / `serial` / …
55    pub default_transport: String,
56    /// `armfvp` / `qemu` / `native` / …
57    pub runner: String,
58    /// Relative path to `prj.conf` (relative to `Cargo.toml`).
59    pub prj_conf: String,
60    /// Relative path to per-board hwv2 `<board>.conf` overlay.
61    pub board_conf: String,
62    /// Relative path to per-board DTS overlay.
63    pub board_overlay: String,
64
65    // -----------------------------------------------------------------------
66    // Phase 215.J.1 — downstream Zephyr consumer provisioning contract.
67    //
68    // These mirror the `NROS_BOARD_{ZEPHYR_LINE,REQUIRES_RUST,RUST_TARGETS,
69    // RMW_SOURCE}` cmake variables and drive `nros setup board <name>
70    // --zephyr-workspace <dir>`. All optional so pre-215.J board manifests
71    // (and non-Zephyr boards) still parse — a board that omits them simply
72    // isn't provisionable via `nros setup board`.
73    // -----------------------------------------------------------------------
74    /// Zephyr support line (e.g. `3.7`) → `scripts/zephyr/patches/<line>.sh`.
75    #[serde(default, skip_serializing_if = "Option::is_none")]
76    pub zephyr_line: Option<String>,
77    /// Whether the board requires the Rust language module (`CONFIG_RUST`).
78    #[serde(default, skip_serializing_if = "Option::is_none")]
79    pub requires_rust: Option<bool>,
80    /// rustup target triples to add (e.g. `["aarch64-unknown-none"]`).
81    #[serde(default, skip_serializing_if = "Vec::is_empty")]
82    pub rust_targets: Vec<String>,
83    /// `[source.*]` name in `nros-sdk-index.toml` for the board's RMW source
84    /// (e.g. `cyclonedds-src`).
85    #[serde(default, skip_serializing_if = "Option::is_none")]
86    pub rmw_source: Option<String>,
87}

89/// Parse `[package.metadata.nros.board]` from a board crate's `Cargo.toml`.
90///
91/// Strict on absent table — callers that need a fallback should handle the
92/// `Err` themselves. Strict on unknown fields (`deny_unknown_fields`).
93pub fn parse_board_metadata(cargo_toml: &Path) -> Result<BoardMetadata, eyre::Report> {
94    let raw = std::fs::read_to_string(cargo_toml).map_err(|e| {
95        eyre::eyre!(
96            "failed to read {} for `[package.metadata.nros.board]`: {e}",
97            cargo_toml.display()
98        )
99    })?;
100    let value: toml::Value = toml::from_str(&raw)
101        .map_err(|e| eyre::eyre!("invalid TOML in {}: {e}", cargo_toml.display()))?;
102    let table = value
103        .get("package")
104        .and_then(|p| p.get("metadata"))
105        .and_then(|m| m.get("nros"))
106        .and_then(|n| n.get("board"))
107        .ok_or_else(|| {
108            eyre::eyre!(
109                "no `[package.metadata.nros.board]` table in {}",
110                cargo_toml.display()
111            )
112        })?;
113    let cloned = table.clone();
114    let board: BoardMetadata = cloned.try_into().map_err(|e| {
115        eyre::eyre!(
116            "invalid `[package.metadata.nros.board]` in {}: {e}",
117            cargo_toml.display()
118        )
119    })?;
120    Ok(board)
121}

127/// One drift mismatch surfaced by [`compute_drift`].
128#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
129pub struct DriftEntry {
130    /// Cargo-side field name (e.g. `"runner"`).
131    pub field: &'static str,
132    /// Value in `[package.metadata.nros.board]`.
133    pub cargo_metadata: String,
134    /// Value parsed out of `board.cmake`.
135    pub board_cmake: String,
136}

138/// Map a `BoardMetadata` field → the `board.cmake` variable name it
139/// mirrors (Phase 215.A.1).
140const FIELD_MAP: &[(&str, &str)] = &[
141    ("zephyr_board", "NROS_BOARD_ZEPHYR_ID"),
142    ("toolchain", "NROS_BOARD_TOOLCHAIN"),
143    ("default_rmw", "NROS_BOARD_DEFAULT_RMW"),
144    ("default_transport", "NROS_BOARD_DEFAULT_TRANSPORT"),
145    ("runner", "NROS_BOARD_RUNNER"),
146    ("prj_conf", "NROS_BOARD_PRJ_CONF"),
147    ("board_conf", "NROS_BOARD_BOARD_CONF"),
148    ("board_overlay", "NROS_BOARD_BOARD_OVERLAY"),
149    // Phase 215.J.1 — provisioning contract (plain-string fields).
150    ("zephyr_line", "NROS_BOARD_ZEPHYR_LINE"),
151    ("rmw_source", "NROS_BOARD_RMW_SOURCE"),

154/// Tokenise `board.cmake` and return a `name → value` map for every
155/// `set(NROS_BOARD_<KEY> <value> …)` call. Values are returned with
156/// surrounding quotes stripped. Cache annotations (`CACHE STRING "doc"`)
157/// are tolerated — only the variable's value is captured. Semicolon-
158/// delimited lists are preserved verbatim; downstream consumers can
159/// `.split(';')` as needed.
160pub fn parse_board_cmake(source: &str) -> BTreeMap<String, String> {
161    let normalised = join_multiline_set_calls(source);
162    let mut out: BTreeMap<String, String> = BTreeMap::new();
163    for raw_line in normalised.lines() {
164        let line = raw_line.split('#').next().unwrap_or("").trim();
165        if line.is_empty() {
166            continue;
167        }
168        let lower = line.to_ascii_lowercase();
169        if !lower.starts_with("set(") {
170            continue;
171        }
172        let after_set = &line["set(".len()..];
173        let body = after_set.trim_end_matches(')').trim();
174
175        let mut tokens = tokenise_cmake_args(body);
176        let var = tokens.next().unwrap_or_default();
177        if !var.starts_with("NROS_BOARD_") {
178            continue;
179        }
180        let mut value: Option<String> = None;
181        for tok in tokens.by_ref() {
182            let upper = tok.to_ascii_uppercase();
183            if matches!(upper.as_str(), "CACHE" | "FORCE" | "PARENT_SCOPE") {
184                continue;
185            }
186            if value.is_some() {
187                break;
188            }
189            value = Some(tok);
190        }
191        if let Some(v) = value {
192            out.insert(var, v);
193        }
194    }
195    out
196}

198/// Collapse multi-line `set(NAME\n  "value"\n)` calls onto a single
199/// line so the per-line parser sees the full call. Tracks paren depth
200/// only outside double-quoted strings; comments (`#…\n`) are stripped
201/// before counting since CMake treats `#` as a line comment.
202fn join_multiline_set_calls(source: &str) -> String {
203    let mut out = String::with_capacity(source.len());
204    let mut depth: usize = 0;
205    let mut in_string = false;
206    let mut chars = source.chars().peekable();
207    while let Some(c) = chars.next() {
208        if !in_string && c == '#' {
209            // Strip to end of line.
210            while let Some(&n) = chars.peek() {
211                if n == '\n' {
212                    break;
213                }
214                chars.next();
215            }
216            continue;
217        }
218        if c == '"' {
219            in_string = !in_string;
220            out.push(c);
221            continue;
222        }
223        if !in_string {
224            if c == '(' {
225                depth += 1;
226            } else if c == ')' {
227                depth = depth.saturating_sub(1);
228            }
229            if c == '\n' && depth > 0 {
230                // Inside an open `set(...)` call — collapse the newline
231                // (and any leading whitespace on the next line) to a
232                // single space so the per-line parser sees one statement.
233                out.push(' ');
234                while let Some(&n) = chars.peek() {
235                    if n == ' ' || n == '\t' {
236                        chars.next();
237                    } else {
238                        break;
239                    }
240                }
241                continue;
242            }
243        }
244        out.push(c);
245    }
246    out
247}

249/// Minimal CMake `set()` argument tokeniser. Handles bare words and
250/// double-quoted strings; CMake bracket syntax `[[…]]` is NOT supported
251/// (the board.cmake schema doesn't use it).
252fn tokenise_cmake_args(s: &str) -> impl Iterator<Item = String> + '_ {
253    let mut iter = s.chars().peekable();
254    std::iter::from_fn(move || {
255        while let Some(&c) = iter.peek() {
256            if c.is_whitespace() {
257                iter.next();
258            } else {
259                break;
260            }
261        }
262        let first = iter.next()?;
263        let mut tok = String::new();
264        if first == '"' {
265            while let Some(c) = iter.next() {
266                if c == '\\' {
267                    if let Some(esc) = iter.next() {
268                        tok.push(esc);
269                    }
270                } else if c == '"' {
271                    return Some(tok);
272                } else {
273                    tok.push(c);
274                }
275            }
276            Some(tok)
277        } else {
278            tok.push(first);
279            while let Some(&c) = iter.peek() {
280                if c.is_whitespace() {
281                    break;
282                }
283                tok.push(c);
284                iter.next();
285            }
286            Some(tok)
287        }
288    })
289}

291/// Compare the typed Cargo.toml view to the parsed board.cmake map.
292///
293/// Path-shaped fields (`prj_conf` / `board_conf` / `board_overlay`)
294/// compare by basename — the Cargo.toml face stores them relative to
295/// `Cargo.toml`, while the cmake face stores absolute paths post-
296/// `${CMAKE_CURRENT_LIST_DIR}` resolution. Basename comparison keeps
297/// the audit meaningful without canonicalising both surfaces.
298///
299/// A board.cmake variable that is not authored is treated as
300/// "no opinion" — not drift.
301pub fn compute_drift(cargo: &BoardMetadata, cmake: &BTreeMap<String, String>) -> Vec<DriftEntry> {
302    let mut out = Vec::new();
303    for &(field, cmake_var) in FIELD_MAP {
304        let cargo_val = cargo_field(cargo, field);
305        let Some(cmake_val) = cmake.get(cmake_var) else {
306            continue;
307        };
308        let (lhs, rhs) = match field {
309            "prj_conf" | "board_conf" | "board_overlay" => (
310                basename(&cargo_val).to_string(),
311                basename(cmake_val).to_string(),
312            ),
313            _ => (cargo_val.clone(), cmake_val.clone()),
314        };
315        if lhs != rhs {
316            out.push(DriftEntry {
317                field,
318                cargo_metadata: cargo_val,
319                board_cmake: cmake_val.clone(),
320            });
321        }
322    }
323    if let Some(cmake_gated) = cmake.get("NROS_BOARD_GATED_PKGS") {
324        let mut cargo_gated = cargo.gated.clone();
325        cargo_gated.sort();
326        let mut cmake_gated_v: Vec<String> = cmake_gated
327            .split(';')
328            .filter(|s| !s.is_empty())
329            .map(|s| s.to_string())
330            .collect();
331        cmake_gated_v.sort();
332        if cargo_gated != cmake_gated_v {
333            out.push(DriftEntry {
334                field: "gated",
335                cargo_metadata: cargo.gated.join(";"),
336                board_cmake: cmake_gated.clone(),
337            });
338        }
339    }
340    // Phase 215.J.1 — `rust_targets` is a semicolon list (same shape as
341    // `gated`); order-insensitive compare.
342    if let Some(cmake_targets) = cmake.get("NROS_BOARD_RUST_TARGETS") {
343        let mut cargo_targets = cargo.rust_targets.clone();
344        cargo_targets.sort();
345        let mut cmake_targets_v: Vec<String> = cmake_targets
346            .split(';')
347            .filter(|s| !s.is_empty())
348            .map(|s| s.to_string())
349            .collect();
350        cmake_targets_v.sort();
351        if cargo_targets != cmake_targets_v {
352            out.push(DriftEntry {
353                field: "rust_targets",
354                cargo_metadata: cargo.rust_targets.join(";"),
355                board_cmake: cmake_targets.clone(),
356            });
357        }
358    }
359    // Phase 215.J.1 — `requires_rust` is a bool on the Cargo side, a cmake
360    // boolean token (`y`/`ON`/…) on the cmake side. Compare normalised.
361    if let Some(cmake_req) = cmake.get("NROS_BOARD_REQUIRES_RUST") {
362        let cargo_req = cargo.requires_rust.unwrap_or(false);
363        if cargo_req != cmake_bool(cmake_req) {
364            out.push(DriftEntry {
365                field: "requires_rust",
366                cargo_metadata: cargo_req.to_string(),
367                board_cmake: cmake_req.clone(),
368            });
369        }
370    }
371    out
372}

374fn cargo_field(m: &BoardMetadata, field: &str) -> String {
375    match field {
376        "zephyr_board" => m.zephyr_board.clone(),
377        "toolchain" => m.toolchain.clone(),
378        "default_rmw" => m.default_rmw.clone(),
379        "default_transport" => m.default_transport.clone(),
380        "runner" => m.runner.clone(),
381        "prj_conf" => m.prj_conf.clone(),
382        "board_conf" => m.board_conf.clone(),
383        "board_overlay" => m.board_overlay.clone(),
384        "zephyr_line" => m.zephyr_line.clone().unwrap_or_default(),
385        "rmw_source" => m.rmw_source.clone().unwrap_or_default(),
386        other => panic!("cargo_field: unknown field {other}"),
387    }
388}

390/// Normalise a `board.cmake` boolean token (`y`/`yes`/`on`/`true`/`1`,
391/// case-insensitive) to a Rust bool. Anything else is `false`.
392fn cmake_bool(v: &str) -> bool {
393    matches!(
394        v.trim().to_ascii_lowercase().as_str(),
395        "y" | "yes" | "on" | "true" | "1"
396    )
397}

399fn basename(p: &str) -> &str {
400    p.rsplit(|c| c == '/' || c == '\\').next().unwrap_or(p)
401}

614    fn parses_board_cmake_multiline_set() {
615        // The real FVP board.cmake at Phase 215.A.2 uses the multi-line
616        // `set(NAME\n    "value")` form. The parser must collapse
617        // multi-line calls onto a single statement before tokenising.
618        // Regression test for Phase 215.A/C verification.
619        let src = r#"
620# multiline form (as in nros-board-fvp-aemv8r-smp/board.cmake)
621set(NROS_BOARD_ZEPHYR_ID
622    "fvp_baser_aemv8r/fvp_aemv8r_aarch64/smp")
623set(NROS_BOARD_TOOLCHAIN
624    "aarch64-zephyr-elf")
625set(NROS_BOARD_PRJ_CONF
626    "${CMAKE_CURRENT_LIST_DIR}/prj.conf")
627"#;
628        let m = parse_board_cmake(src);
629        assert_eq!(
630            m.get("NROS_BOARD_ZEPHYR_ID").map(String::as_str),
631            Some("fvp_baser_aemv8r/fvp_aemv8r_aarch64/smp"),
632            "multi-line ZEPHYR_ID must round-trip",
633        );
634        assert_eq!(
635            m.get("NROS_BOARD_TOOLCHAIN").map(String::as_str),
636            Some("aarch64-zephyr-elf"),
637        );
638        assert!(
639            m.get("NROS_BOARD_PRJ_CONF")
640                .map(|s| s.contains("prj.conf"))
641                .unwrap_or(false),
642            "multi-line PRJ_CONF must round-trip",
643        );
644    }

716    #[test]
717    fn drift_compute_provisioning_contract() {
718        // Phase 215.J.1 — provisioning fields participate in the drift audit.
719        let cargo = BoardMetadata {
720            zephyr_board: "x".into(),
721            toolchain: "y".into(),
722            gated: vec![],
723            default_rmw: "cyclonedds".into(),
724            default_transport: "ethernet".into(),
725            runner: "armfvp".into(),
726            prj_conf: "prj.conf".into(),
727            board_conf: "x.conf".into(),
728            board_overlay: "x.overlay".into(),
729            zephyr_line: Some("3.7".into()),
730            requires_rust: Some(true),
731            rust_targets: vec!["aarch64-unknown-none".into()],
732            rmw_source: Some("cyclonedds-src".into()),
733        };
734
735        // Agreement: cmake uses the `y` boolean token + matching values.
736        let mut cmake: BTreeMap<String, String> = BTreeMap::new();
737        cmake.insert("NROS_BOARD_ZEPHYR_LINE".into(), "3.7".into());
738        cmake.insert("NROS_BOARD_REQUIRES_RUST".into(), "y".into());
739        cmake.insert(
740            "NROS_BOARD_RUST_TARGETS".into(),
741            "aarch64-unknown-none".into(),
742        );
743        cmake.insert("NROS_BOARD_RMW_SOURCE".into(), "cyclonedds-src".into());
744        assert!(
745            compute_drift(&cargo, &cmake).is_empty(),
746            "provisioning contract agrees → no drift"
747        );
748
749        // Drift: rmw_source + requires_rust disagree.
750        let mut bad = cmake.clone();
751        bad.insert("NROS_BOARD_RMW_SOURCE".into(), "iceoryx-src".into());
752        bad.insert("NROS_BOARD_REQUIRES_RUST".into(), "n".into());
753        let drift = compute_drift(&cargo, &bad);
754        let fields: Vec<&str> = drift.iter().map(|d| d.field).collect();
755        assert!(fields.contains(&"rmw_source"), "rmw_source drift: {drift:?}");
756        assert!(
757            fields.contains(&"requires_rust"),
758            "requires_rust drift: {drift:?}"
759        );
760    }
