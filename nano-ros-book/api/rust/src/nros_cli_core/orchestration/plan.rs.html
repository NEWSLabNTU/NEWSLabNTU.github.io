457/// Phase 172.C — dispatch concurrency class of a [`PlanCallbackGroup`],
458/// mirroring rclcpp's two callback-group kinds.
459#[derive(Clone, Copy, Debug, PartialEq, Eq, Serialize, Deserialize)]
460#[serde(rename_all = "snake_case")]
461pub enum CallbackGroupKind {
462    /// Members never run concurrently with one another (serialized
463    /// dispatch) — the safe default for dataflow-coupled pipeline stages
464    /// that may share state.
465    MutuallyExclusive,
466    /// Members may run concurrently — inferred for callbacks with no
467    /// detected dataflow coupling.
468    Reentrant,
469}

471/// Phase 172.C — an inferred (or overridden) callback group. Each callback
472/// belongs to exactly one group; the group's [`CallbackGroupKind`] decides
473/// whether its members serialize or may run concurrently. Derived from the
474/// 172.B callback chains: every chain becomes one mutually-exclusive group
475/// (its stages serialize), and every callback outside any chain becomes its
476/// own reentrant group (no coupling detected → concurrent-safe). 172.G
477/// assigns scheduling tiers on top of this grouping.
478#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
479#[serde(deny_unknown_fields)]
480pub struct PlanCallbackGroup {
481    /// Stable group id, e.g. `group/<chain-head>` or `group/<callback-id>`.
482    pub id: String,
483    /// Serialize vs concurrent dispatch.
484    pub kind: CallbackGroupKind,
485    /// Callback ids that belong to this group (chain order for chain
486    /// groups; a single callback for reentrant groups).
487    pub callbacks: Vec<String>,
488    /// `true` when the planner inferred this group from the chains;
489    /// `false` when it came from an explicit `[[group]]` override.
490    pub inferred: bool,
491}

598/// Phase 173.5 — physical transport a `[[transport]]` entry selects.
599/// The kind always comes from `nros.toml`; the per-kind value (ip /
600/// baudrate / device) lands wherever that platform's net stack reads it
601/// (board `Config` for `NanoRosOwned`, an RTOS config fragment for
602/// `RtosOwned` — see [`super::generate`] / Phase 173.7).
603#[derive(Clone, Copy, Debug, PartialEq, Eq, Serialize, Deserialize)]
604#[serde(rename_all = "lowercase")]
605pub enum TransportKind {
606    Ethernet,
607    Wifi,
608    Serial,
609    Can,
610}
611
612impl TransportKind {
613    /// The board crate Cargo feature that enables this transport.
614    pub fn cargo_feature(self) -> &'static str {
615        match self {
616            TransportKind::Ethernet => "ethernet",
617            TransportKind::Wifi => "wifi",
618            TransportKind::Serial => "serial",
619            TransportKind::Can => "can",
620        }
621    }
622}

624/// Phase 173.5 — one transport⟷RMW binding from `nros.toml`'s
625/// `[[transport]]` array. Two or more entries put the build in **bridge
626/// mode** (each transport runs its own RMW session;
627/// `Executor::open_multi` consumes the resulting `SessionSpec`s).
628///
629/// `rmw`/`locator` are optional per entry; when absent they fall back to
630/// the top-level `build.rmw` / the platform default. The generator —
631/// not hand-written code — turns these into the board transport
632/// feature(s), the per-transport `Config` values, and the RMW deps.
633#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
634#[serde(deny_unknown_fields)]
635pub struct PlanTransport {
636    pub kind: TransportKind,
637    /// Stable transport id used to bind a node/instance to this session
638    /// (`SystemComponent.transport`). `None` ⇒ defaults to `rmw` (works when
639    /// each transport has a distinct rmw). Phase 172.K.
640    #[serde(default)]
641    pub id: Option<String>,
642    /// IPv4 CIDR (`"10.0.2.50/24"`) or `"dhcp"` — ethernet/wifi only.
643    pub ip: Option<String>,
644    /// WiFi SSID — wifi only. Phase 172.K.
645    #[serde(default)]
646    pub ssid: Option<String>,
647    /// WiFi password — wifi only. Phase 172.K.
648    #[serde(default)]
649    pub password: Option<String>,
650    /// Ethernet MAC (`"02:00:00:00:00:01"`) — ethernet only. `None` ⇒
651    /// the board's fixed/fused MAC. (Phase 172.J — replaces
652    /// `config.toml`'s `[network].mac`.)
653    #[serde(default)]
654    pub mac: Option<String>,
655    /// Default IPv4 gateway (`"10.0.2.2"`) — ethernet only. `None` ⇒ a
656    /// flat link with no gateway. (Phase 172.J — replaces
657    /// `config.toml`'s `[network].gateway`.)
658    #[serde(default)]
659    pub gateway: Option<String>,
660    /// NIC name(s) this transport multi-homes over (`["eth0", "eth1"]`) —
661    /// ethernet / wifi only. One session folds every listed interface into a
662    /// *single* discovery graph (Phase 172.K.7); this is the opposite intent
663    /// from declaring multiple `[[transport]]` entries (which open *separate*
664    /// sessions). Empty ⇒ the backend's default (all / any interface). The
665    /// generator maps the list per backend — zenoh listen/connect per NIC +
666    /// `scouting.multicast.interface`; Cyclone `<General><Interfaces>`.
667    #[serde(default, skip_serializing_if = "Vec::is_empty")]
668    pub interfaces: Vec<String>,
669    /// Device handle (`"UART0"`, `"CAN0"`) — serial / can only.
670    pub device: Option<String>,
671    /// Line rate (serial baud / CAN bitrate) — serial / can only.
672    pub baudrate: Option<u32>,
673    /// RMW that rides this transport. `None` ⇒ inherit `build.rmw`.
674    pub rmw: Option<String>,
675    /// Zenoh/DDS locator seeding this transport's session. `None` ⇒
676    /// platform / env default.
677    pub locator: Option<String>,
678    /// Phase 172 WP-B — ROS domain this transport's session joins. `None` ⇒
679    /// the build/system default (0). Lets a bridge open same-rmw sessions on
680    /// distinct domains (multi-domain in-binary); the session = (rmw, locator,
681    /// domain). Additive; skip-when-absent so single-domain plans round-trip
682    /// unchanged.
683    #[serde(default, skip_serializing_if = "Option::is_none")]
684    pub domain: Option<u32>,
685}

687#[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]
688#[serde(deny_unknown_fields)]
689pub struct PlanBuildOptions {
690    pub target: String,
691    pub board: String,
692    pub rmw: String,
693    pub profile: String,
694    pub features: Vec<String>,
695    pub cfg: ParameterTable,
696    /// Phase 173.5 — `nros.toml` `[[transport]]` entries. Empty ⇒
697    /// zero-config single-transport build (board default transport +
698    /// the single linked RMW). Defaulted so pre-173.5 plans parse;
699    /// skip-when-empty so the stable pretty fixtures (zero-config
700    /// builds) round-trip without an empty `"transports": []`.
701    #[serde(default, skip_serializing_if = "Vec::is_empty")]
702    pub transports: Vec<PlanTransport>,
703    /// Phase 255 Wave 5 — the extra RMW backends a single binary links to host
704    /// cross-RMW `[[bridge]]`s (the union of every bridged `[[domain]]`'s RMW,
705    /// from `system.toml`). Empty ⇒ single-RMW build (just `rmw`), byte-identical
706    /// to pre-255. Supersedes the `[[transport]].rmw` overlay multi-RMW path —
707    /// bridges are topology and belong in `system.toml`, read by both codegen
708    /// paths. `rmw_set` unions this with `rmw` / `[[transport]].rmw`.
709    #[serde(default, skip_serializing_if = "Vec::is_empty")]
710    pub bridged_rmws: Vec<String>,
711    /// Phase 204.15 — coherent size/speed *intent* (`size` | `speed` |
712    /// `balanced` | `debug`). The generated package build fans it out to RUSTFLAGS (`-C
713    /// opt-level/lto/codegen-units/strip`) on top of the cargo profile, so one
714    /// knob tunes the Rust layer without per-crate `[profile.*]` edits. `None` ⇒
715    /// today's behaviour (profile only). Per-layer overrides + the cc/CMake
716    /// fan-out are tracked follow-ups (204.15).
717    #[serde(default, skip_serializing_if = "Option::is_none")]
718    pub optimize: Option<String>,
719    /// Phase 204.15 (increment 2) — per-layer `[build.cargo]` override table.
720    /// Refines the Rust/cargo layer *on top of* the `optimize` baseline
721    /// (precedence: `optimize` baseline → `[build.cargo]` field). Lets a build
722    /// tune one cargo-profile field without disturbing the coherent intent —
723    /// e.g. `optimize = "size"` + `[build.cargo] debug = true, strip = false`
724    /// keeps Rust debuginfo while everything else stays size-tuned. `None` ⇒
725    /// the `optimize` baseline alone.
726    #[serde(default, skip_serializing_if = "Option::is_none")]
727    pub cargo: Option<PlanCargoOverrides>,
728    /// Phase 204.15 (increment 3) — per-layer `[build.cc]` override for the C/C++
729    /// layer. `debug`/`cflags` are exported as `CFLAGS`/`CXXFLAGS` which `cc-rs`
730    /// *appends* to its computed flags (every zenoh-pico/XRCE/net.c/lwIP
731    /// `cc::Build`, no build.rs edit): `debug = true` adds `-g` without disturbing
732    /// the opt level → the C-side of the "debug one layer" case. `opt_level` →
733    /// `NROS_CC_OPT` (build scripts that honor it override their hardcoded opt; 204.9).
734    #[serde(default, skip_serializing_if = "Option::is_none")]
735    pub cc: Option<PlanCcOverrides>,
736    /// Phase 195.C — workspace root, populated at generate time (NOT part of
737    /// the plan wire format). Lets `profile()` load board descriptors from
738    /// `<workspace>/packages/boards/*/nros-board.toml` so the CLI carries no
739    /// baked-in board layout. `None` outside a generate run.
740    #[serde(skip)]
741    pub workspace_root: Option<std::path::PathBuf>,
742}

744/// Phase 204.15 (increment 2) — `[build.cargo]` per-layer override fields. Each
745/// maps to a cargo `[profile.release]` key and, when present, replaces the value
746/// the `optimize` intent produced. Values are kept as raw JSON so `opt_level`
747/// accepts both `3` (number) and `"z"` (string); `lto`/`strip` accept a bool or
748/// string; rendering picks the right TOML literal. Unknown JSON shapes are

758    pub debug: Option<serde_json::Value>,
759    #[serde(default, skip_serializing_if = "Option::is_none")]
760    pub strip: Option<serde_json::Value>,
761    #[serde(default, skip_serializing_if = "Option::is_none")]
762    pub codegen_units: Option<serde_json::Value>,
763    #[serde(default, skip_serializing_if = "Option::is_none")]
764    pub panic: Option<serde_json::Value>,
765}

781impl PlanBuildOptions {
782    /// `true` when more than one transport is declared — the build runs
783    /// multiple RMW sessions via `Executor::open_multi` (bridge mode).
784    pub fn is_bridge(&self) -> bool {
785        self.transports.len() > 1
786    }
787
788    /// Phase 204.7 — `true` when the declared transports carry **no IP link**
789    /// (every transport is serial/CAN, none ethernet/wifi). The generated
790    /// `.cargo/config.toml` then bakes `NROS_LINK_IP=0` so the zenoh-pico /
791    /// XRCE TCP+UDP link C is dropped (with `--gc-sections`, ~‑33 KB BSS on a
792    /// bare-metal serial build). Empty transports ⇒ `false` (zero-config build
793    /// keeps the board default IP link).
794    pub fn drops_ip_link(&self) -> bool {
795        !self.transports.is_empty()
796            && self
797                .transports
798                .iter()
799                .all(|t| matches!(t.kind, TransportKind::Serial | TransportKind::Can))
800    }
801
802    /// Validate the `[[transport]]` array against per-kind field rules.
803    /// Returns the list of human-readable problems (empty ⇒ valid) so
804    /// the caller can surface them all at once rather than one per run.
805    pub fn validate_transports(&self) -> Vec<String> {
806        let mut problems = Vec::new();
807        for (i, t) in self.transports.iter().enumerate() {
808            let at = format!("transport[{i}] (kind = {:?})", t.kind);
809            match t.kind {
810                TransportKind::Ethernet => {
811                    if t.device.is_some() || t.baudrate.is_some() {
812                        problems.push(format!("{at}: `device`/`baudrate` are serial/can-only"));
813                    }
814                    if t.ssid.is_some() || t.password.is_some() {
815                        problems.push(format!("{at}: `ssid`/`password` are wifi-only"));
816                    }
817                }
818                TransportKind::Wifi => {
819                    // wifi carries ssid/password (+ optional static ip/gateway);
820                    // mac is ethernet-only, device/baudrate are serial/can-only.
821                    if t.device.is_some() || t.baudrate.is_some() {
822                        problems.push(format!("{at}: `device`/`baudrate` are serial/can-only"));
823                    }
824                    if t.mac.is_some() {
825                        problems.push(format!("{at}: `mac` is ethernet-only"));
826                    }
827                }
828                TransportKind::Serial | TransportKind::Can => {
829                    if t.ip.is_some() {
830                        problems.push(format!("{at}: `ip` is ethernet/wifi-only"));
831                    }
832                    if t.mac.is_some() {
833                        problems.push(format!("{at}: `mac` is ethernet-only"));
834                    }
835                    if t.gateway.is_some() {
836                        problems.push(format!("{at}: `gateway` is ethernet/wifi-only"));
837                    }
838                    if !t.interfaces.is_empty() {
839                        problems.push(format!("{at}: `interfaces` is ethernet/wifi-only"));
840                    }
841                    if t.ssid.is_some() || t.password.is_some() {
842                        problems.push(format!("{at}: `ssid`/`password` are wifi-only"));
843                    }
844                }
845            }
846        }
847        problems
848    }
849}
