1//! Phase 187.3 — the install store, provenance, lockfile, and fetch/source-build
2//! engine behind `nros setup`.
3//!
4//! A tool always lands at the same versioned prefix
5//! `$NROS_HOME/sdk/<tool>/<version>/`, whether fetched (prebuilt `dist`) or
6//! source-built — downstream resolves the prefix, provenance-agnostic
7//! (`.nros-provenance` records which). The store is shared across workspaces;
8//! `nros-sdk.lock` pins what's installed. See
9//! `docs/design/nros-setup-toolchain-management.md`.

26/// The shared SDK store root: `$NROS_HOME/sdk`, else `~/.nros/sdk`, else
27/// `./.nros/sdk`.
28pub fn store_root() -> PathBuf {
29    if let Some(h) = std::env::var_os("NROS_HOME") {
30        return PathBuf::from(h).join("sdk");
31    }
32    if let Some(h) = std::env::var_os("HOME") {
33        return PathBuf::from(h).join(".nros").join("sdk");
34    }
35    PathBuf::from(".nros").join("sdk")
36}

43/// How a tool was installed; persisted to `<prefix>/.nros-provenance`.
44#[derive(Clone, Copy, Debug, PartialEq, Eq, Serialize, Deserialize)]
45#[serde(rename_all = "lowercase")]
46pub enum ProvenanceKind {
47    Prebuilt,
48    Source,
49}
50
51#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
52pub struct Provenance {
53    pub kind: ProvenanceKind,
54    pub version: String,
55    #[serde(default, skip_serializing_if = "Option::is_none")]
56    pub sha256: Option<String>,
57}
58
59impl Provenance {
60    fn marker(prefix: &Path) -> PathBuf {
61        prefix.join(".nros-provenance")
62    }
63
64    /// Read the provenance marker, if the prefix is populated.
65    pub fn read(prefix: &Path) -> Option<Self> {
66        let raw = std::fs::read_to_string(Self::marker(prefix)).ok()?;
67        toml::from_str(&raw).ok()
68    }
69
70    pub fn write(&self, prefix: &Path) -> Result<()> {
71        std::fs::create_dir_all(prefix)
72            .wrap_err_with(|| format!("create prefix {}", prefix.display()))?;
73        std::fs::write(Self::marker(prefix), toml::to_string(self)?)
74            .wrap_err("write .nros-provenance")?;
75        Ok(())
76    }
77}

79/// `nros-sdk.lock` — the exact installed toolset, for reproducibility.
80#[derive(Clone, Debug, Default, PartialEq, Eq, Serialize, Deserialize)]
81pub struct SdkLock {
82    #[serde(default)]
83    pub tool: BTreeMap<String, LockEntry>,
84}
85
86#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
87pub struct LockEntry {
88    pub version: String,
89    pub provenance: ProvenanceKind,
90    #[serde(default, skip_serializing_if = "Option::is_none")]
91    pub sha256: Option<String>,
92}
93
94impl SdkLock {
95    /// Load the lockfile; a missing file is an empty lock (not an error).
96    pub fn load(path: &Path) -> Result<Self> {
97        match std::fs::read_to_string(path) {
98            Ok(raw) => toml::from_str(&raw).wrap_err("invalid nros-sdk.lock"),
99            Err(e) if e.kind() == std::io::ErrorKind::NotFound => Ok(Self::default()),
100            Err(e) => Err(e).wrap_err_with(|| format!("read {}", path.display())),
101        }
102    }
103
104    pub fn record(&mut self, tool: &str, p: &Provenance) {
105        self.tool.insert(
106            tool.to_string(),
107            LockEntry {
108                version: p.version.clone(),
109                provenance: p.kind,
110                sha256: p.sha256.clone(),
111            },
112        );
113    }
114
115    pub fn save(&self, path: &Path) -> Result<()> {
116        std::fs::write(path, toml::to_string_pretty(self)?)
117            .wrap_err_with(|| format!("write {}", path.display()))
118    }
119}

121/// The decided install action for a tool on a host.
122#[derive(Clone, Debug, PartialEq, Eq)]
123pub enum InstallAction {
124    /// Already at the prefix (idempotent skip).
125    Present,
126    /// Fetch + verify + unpack the prebuilt artifact.
127    Prebuilt { url: String, sha256: String },
128    /// Build from source into the same prefix.
129    Source {
130        git: String,
131        git_ref: String,
132        configure: Option<String>,
133        install: Option<String>,
134    },
135    /// No prebuilt for this host and no source recipe.
136    Unavailable,
137}
138
139/// Decide how to install `tool` on `host`, given the prefix's current state.
140/// Pure — does no I/O beyond reading the provenance marker.
141pub fn plan_install(tool: &ToolPackage, host: &str, prefix: &Path) -> InstallAction {
142    if Provenance::read(prefix).is_some() {
143        return InstallAction::Present;
144    }
145    if let Some(d) = tool.dist_for(host) {
146        return InstallAction::Prebuilt {
147            url: d.url.clone(),
148            sha256: d.sha256.clone(),
149        };
150    }
151    if let Some(s) = &tool.source {
152        return InstallAction::Source {
153            git: s.git.clone(),
154            git_ref: s.git_ref.clone(),
155            configure: s.configure.clone(),
156            install: s.install.clone(),
157        };
158    }
159    InstallAction::Unavailable
160}

162/// Execute an install action into `prefix`, returning the recorded provenance.
163/// Side-effecting (shells out to curl / sha256sum / tar / git); real-run only.
164pub fn execute(
165    action: &InstallAction,
166    tool: &str,
167    version: &str,
168    prefix: &Path,
169) -> Result<Provenance> {
170    match action {
171        InstallAction::Present => Provenance::read(prefix)
172            .ok_or_else(|| eyre!("{tool}: present but no provenance marker")),
173        InstallAction::Prebuilt { url, sha256 } => {
174            std::fs::create_dir_all(prefix)
175                .wrap_err_with(|| format!("create {}", prefix.display()))?;
176            let archive = prefix.with_extension("download");
177            sh(
178                &[
179                    "curl",
180                    "-L",
181                    "--fail",
182                    "--silent",
183                    "--show-error",
184                    "-o",
185                    &archive.to_string_lossy(),
186                    url,
187                ],
188                None,
189            )
190            .wrap_err_with(|| format!("download {url}"))?;
191            verify_sha256(&archive, sha256)?;
192            sh(
193                &[
194                    "tar",
195                    "-xf",
196                    &archive.to_string_lossy(),
197                    "-C",
198                    &prefix.to_string_lossy(),
199                ],
200                None,
201            )
202            .wrap_err("unpack prebuilt archive")?;
203            let _ = std::fs::remove_file(&archive);
204            let p = Provenance {
205                kind: ProvenanceKind::Prebuilt,
206                version: version.to_string(),
207                sha256: Some(sha256.clone()),
208            };
209            p.write(prefix)?;
210            Ok(p)
211        }
212        InstallAction::Source {
213            git,
214            git_ref,
215            configure,
216            install,
217        } => {
218            let src = prefix.with_extension("src");
219            let _ = std::fs::remove_dir_all(&src);
220            sh(
221                &[
222                    "git",
223                    "clone",
224                    "--depth",
225                    "1",
226                    "--branch",
227                    git_ref,
228                    git,
229                    &src.to_string_lossy(),
230                ],
231                None,
232            )
233            .wrap_err_with(|| format!("git clone {git} @ {git_ref}"))?;
234            let prefix_abs = prefix.to_string_lossy().to_string();
235            if let Some(cfg) = configure {
236                sh(
237                    &["sh", "-c", &cfg.replace("{prefix}", &prefix_abs)],
238                    Some(&src),
239                )
240                .wrap_err("configure step")?;
241            }
242            if let Some(inst) = install {
243                sh(
244                    &["sh", "-c", &inst.replace("{prefix}", &prefix_abs)],
245                    Some(&src),
246                )
247                .wrap_err("install step")?;
248            }
249            let p = Provenance {
250                kind: ProvenanceKind::Source,
251                version: version.to_string(),
252                sha256: None,
253            };
254            p.write(prefix)?;
255            Ok(p)
256        }
257        InstallAction::Unavailable => {
258            bail!("{tool} {version}: no prebuilt for this host and no source recipe in the index")
259        }
260    }
261}

263/// Verify `path`'s sha256 equals `expected` (shells out to `sha256sum`, falling
264/// back to `shasum -a 256` on macOS).
265fn verify_sha256(path: &Path, expected: &str) -> Result<()> {
266    let out = Command::new("sha256sum")
267        .arg(path)
268        .output()
269        .or_else(|_| {
270            Command::new("shasum")
271                .args(["-a", "256"])
272                .arg(path)
273                .output()
274        })
275        .wrap_err("run sha256sum / shasum")?;
276    if !out.status.success() {
277        bail!("sha256sum failed for {}", path.display());
278    }
279    let got = String::from_utf8_lossy(&out.stdout);
280    let got = got.split_whitespace().next().unwrap_or("");
281    if !got.eq_ignore_ascii_case(expected) {
282        bail!(
283            "sha256 mismatch for {}: expected {expected}, got {got}",
284            path.display()
285        );
286    }
287    Ok(())
288}

290/// Outcome of [`provision_source`] — for the `nros setup` disposition line.
291#[derive(Clone, Debug, PartialEq, Eq)]
292pub enum SourceDisposition {
293    /// Fetched into `dest` (clone or submodule update).
294    Provisioned,
295    /// `dest` already present — left untouched (idempotent skip).
296    AlreadyPresent,
297    /// No fetch step declared (version-only `[source.*]`).
298    NoFetch,
299    /// `--dry-run`: what would have happened.
300    Planned,
301}
302
303/// Phase 195.B — provision a `[source.*]` package into its workspace-relative
304/// `dest` from index data (never a path baked into the binary). Idempotent: an
305/// already-present `dest` is left untouched. Two modes (see
306/// [`SourcePackage::provision`]):
307///
308/// Both modes honor the source's `shallow` (default true → `--depth 1`,
309/// fetch-by-SHA so a lagging pin is still a real depth-1 checkout) and
310/// `recursive` (default true → descend the source's own nested submodules).
311///
312/// - **Clone:** shallow → `git init` + `git fetch --depth 1 origin <ref>` +
313///   `git checkout FETCH_HEAD` (`ref` may be a sha, so `--branch` can't be
314///   used); full → `git clone <git> <dest>` + `git checkout <ref>`. Then, if
315///   recursive, `git submodule update --init --recursive [--depth 1]`.
316/// - **Submodule:** `git -C <workspace> submodule update --init [--recursive]
317///   [--depth 1] -- <submodule>` (inherently idempotent; checks out the
318///   superproject's recorded commit, kept in lockstep with the index ref via
319///   the SSOT rule).
320pub fn provision_source(
321    name: &str,
322    src: &SourcePackage,
323    workspace: &Path,
324    dry_run: bool,
325    shallow_override: Option<bool>,
326) -> Result<SourceDisposition> {
327    // `--full` / `--shallow` (per-invocation) wins over the index `shallow`.
328    let shallow = shallow_override.unwrap_or(src.shallow);
329    match src.provision() {
330        SourceProvision::None => Ok(SourceDisposition::NoFetch),
331        SourceProvision::Submodule => {
332            let path = src.submodule.as_deref().expect("submodule mode has a path");
333            if dry_run {
334                return Ok(SourceDisposition::Planned);
335            }
336            // Fast path: `git submodule update --init [--recursive] [--depth 1]`.
337            // CAVEAT: `--depth 1` shallow-fetches the submodule's BRANCH TIP, not
338            // the pinned gitlink SHA — so a pin that lags its tip and isn't an
339            // advertised ref (e.g. PX4-Autopilot's 1.15.x sha vs `main`) fails the
340            // checkout here. We catch that and fall back to an explicit
341            // depth-1 fetch-by-SHA (GitHub serves reachable SHAs), which `git
342            // clone --branch` / `submodule update` can't express directly.
343            let workspace_s = workspace.to_string_lossy();
344            let mut args: Vec<&str> =
345                vec!["git", "-C", &workspace_s, "submodule", "update", "--init"];
346            if src.recursive {
347                args.push("--recursive");
348            }
349            if shallow {
350                args.push("--depth");
351                args.push("1");
352            }
353            args.push("--");
354            args.push(path);
355            let fast = sh(&args, None);
356            if let Err(e) = fast {
357                if !shallow {
358                    return Err(e)
359                        .wrap_err_with(|| format!("git submodule update {path} (source {name})"));
360                }
361                // By-SHA fallback. `submodule update` already initialised the
362                // gitdir + worktree (only its checkout of the lagging pin failed),
363                // so fetch the exact gitlink SHA at depth 1 and check it out, then
364                // descend if recursive.
365                let sha = sh_capture(&["git", "-C", &workspace_s, "ls-tree", "HEAD", path], None)
366                    .wrap_err_with(|| format!("read gitlink sha for {path} (source {name})"))?
367                    .split_whitespace()
368                    .nth(2)
369                    .map(str::to_owned)
370                    .ok_or_else(|| eyre!("no gitlink sha for {path} (source {name})"))?;
371                let subdir = workspace.join(path);
372                let subdir_s = subdir.to_string_lossy();
373                sh(
374                    &[
375                        "git", "-C", &subdir_s, "fetch", "--depth", "1", "origin", &sha,
376                    ],
377                    None,
378                )
379                .wrap_err_with(|| format!("git fetch --depth 1 {sha} ({path}, source {name})"))?;
380                sh(&["git", "-C", &subdir_s, "checkout", "-q", &sha], None)
381                    .wrap_err_with(|| format!("git checkout {sha} ({path}, source {name})"))?;
382                if src.recursive {
383                    sh(
384                        &[
385                            "git",
386                            "-C",
387                            &subdir_s,
388                            "submodule",
389                            "update",
390                            "--init",
391                            "--recursive",
392                            "--depth",
393                            "1",
394                            "--recommend-shallow",
395                        ],
396                        None,
397                    )
398                    .wrap_err_with(|| {
399                        format!("git submodule update --recursive ({path}, source {name})")
400                    })?;
401                }
402            }
403            Ok(SourceDisposition::Provisioned)
404        }
405        SourceProvision::Clone => {
406            let git = src.git.as_deref().expect("clone mode has a git url");
407            let git_ref = src.git_ref.as_deref().expect("clone mode has a ref");
408            let dest = src.dest.as_deref().expect("clone mode has a dest");
409            let dest_abs = workspace.join(dest);
410            // Idempotent: a present, non-empty dest is left as-is (don't clobber
411            // a contributor checkout / in-progress work). `nros setup` on a
412            // fresh tree provisions; a populated tree is a skip.
413            let present = dest_abs
414                .read_dir()
415                .map(|mut d| d.next().is_some())
416                .unwrap_or(false);
417            if present {
418                return Ok(SourceDisposition::AlreadyPresent);
419            }
420            if dry_run {
421                return Ok(SourceDisposition::Planned);
422            }
423            if let Some(parent) = dest_abs.parent() {
424                std::fs::create_dir_all(parent)
425                    .wrap_err_with(|| format!("create source dest parent {}", parent.display()))?;
426            }
427            let dest_str = dest_abs.to_string_lossy();
428            if shallow {
429                // Shallow clone of a possibly-SHA `ref`: `git clone --branch`
430                // can't take a sha, so init + fetch-by-ref at depth 1 (works for
431                // sha/tag/branch via the server's reachable-SHA support) + check
432                // out the fetched commit (detached, same as a sha checkout).
433                sh(&["git", "init", "-q", &dest_str], None)
434                    .wrap_err_with(|| format!("git init {dest_str} (source {name})"))?;
435                sh(
436                    &["git", "-C", &dest_str, "remote", "add", "origin", git],
437                    None,
438                )
439                .wrap_err_with(|| format!("git remote add (source {name})"))?;
440                sh(
441                    &[
442                        "git", "-C", &dest_str, "fetch", "-q", "--depth", "1", "origin", git_ref,
443                    ],
444                    None,
445                )
446                .wrap_err_with(|| format!("git fetch --depth 1 {git_ref} (source {name})"))?;
447                sh(
448                    &["git", "-C", &dest_str, "checkout", "-q", "FETCH_HEAD"],
449                    None,
450                )
451                .wrap_err_with(|| format!("git checkout FETCH_HEAD (source {name})"))?;
452            } else {
453                sh(&["git", "clone", git, &dest_str], None)
454                    .wrap_err_with(|| format!("git clone {git} (source {name})"))?;
455                sh(&["git", "-C", &dest_str, "checkout", git_ref], None)
456                    .wrap_err_with(|| format!("git checkout {git_ref} (source {name})"))?;
457            }
458            if src.recursive {
459                // A cloned source may carry its own nested submodules.
460                let mut sub: Vec<&str> = vec![
461                    "git",
462                    "-C",
463                    &dest_str,
464                    "submodule",
465                    "update",
466                    "--init",
467                    "--recursive",
468                ];
469                if shallow {
470                    sub.push("--depth");
471                    sub.push("1");
472                }
473                sh(&sub, None).wrap_err_with(|| {
474                    format!("git submodule update --recursive (source {name})")
475                })?;
476            }
477            Ok(SourceDisposition::Provisioned)
478        }
479    }
480}

482fn sh(args: &[&str], cwd: Option<&Path>) -> Result<()> {
483    let (cmd, rest) = args.split_first().ok_or_else(|| eyre!("empty command"))?;
484    let mut c = Command::new(cmd);
485    c.args(rest);
486    if let Some(d) = cwd {
487        c.current_dir(d);
488    }
489    let status = c.status().wrap_err_with(|| format!("spawn {cmd}"))?;
490    if !status.success() {
491        bail!("`{}` failed ({status})", args.join(" "));
492    }
493    Ok(())
494}
495
496/// Run a command and capture trimmed stdout (for reading a gitlink SHA, etc.).
497fn sh_capture(args: &[&str], cwd: Option<&Path>) -> Result<String> {
498    let (cmd, rest) = args.split_first().ok_or_else(|| eyre!("empty command"))?;
499    let mut c = Command::new(cmd);
500    c.args(rest);
501    if let Some(d) = cwd {
502        c.current_dir(d);
503    }
504    let out = c.output().wrap_err_with(|| format!("spawn {cmd}"))?;
505    if !out.status.success() {
506        bail!("`{}` failed ({})", args.join(" "), out.status);
507    }
508    Ok(String::from_utf8_lossy(&out.stdout).trim().to_string())
509}

628            git: Some("https://example/lwip.git".into()),
629            git_ref: Some("STABLE-2_2_0".into()),

634        };

640        // Clone mode, empty dest, dry-run → Planned (no git run).
