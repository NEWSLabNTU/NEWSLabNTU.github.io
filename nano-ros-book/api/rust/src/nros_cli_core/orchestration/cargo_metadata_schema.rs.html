1//! Phase 212.B — `[workspace.metadata.nros]` + `[package.metadata.nros]` +
2//! `[package.metadata.ament]` + `<bringup>/system.toml` data contracts.
3//!
4//! These are the *user-authored* TOML surfaces introduced by Phase 212. They
5//! live in standard cargo manifest tables (`[workspace.metadata.…]` /
6//! `[package.metadata.…]`) so that cargo treats them as opaque user data and
7//! pure-cargo workflows (no `nros build` verb) keep working. The
8//! `<bringup>/system.toml` is the per-system declarative file owned by the
9//! `<system>_bringup` package.
10//!
11//! Vocabulary discipline (per the Phase 212 doc): every field name is a strict
12//! subset of names that already appear in `nros-sdk-index.toml`,
13//! `app_config.h`, or the existing planner schema. No second TOML dialect.
14//!
15//! Every struct here uses `#[serde(deny_unknown_fields)]` so typos surface as
16//! parse errors at the user's terminal instead of being silently dropped.
17

28/// `[workspace.metadata.nros]` in a workspace-root `Cargo.toml`.
29///
30/// Thin pointer (see `docs/design/multi-node-workspace-layout.md` §5). The
31/// authoritative system spec lives in `<bringup>/system.toml`; this table

37    /// args resolves the system via this pointer.

39    pub default_system: Option<String>,
40    /// Optional workspace-wide RMW override — rare, intended for
41    /// `nros plan --override` workflows.
42    #[serde(default, skip_serializing_if = "Option::is_none")]
43    pub rmw_override: Option<String>,
44}

50/// `[package.metadata.nros]` in a component / application package's
51/// `Cargo.toml`.
52///
53/// Three top-level shapes (mutually exclusive):
54///
55/// * Single-node crate — `[package.metadata.nros.node]` describes the
56///   one node the crate exposes. (The Phase 212.N.12 rename made
57///   `node` the canonical key; `[package.metadata.nros.component]`
58///   remains accepted as a deprecated alias — declaring both is a
59///   hard error. See `nros_config::parse_package_metadata_nros`.)
60/// * Multi-component crate — `[package.metadata.nros.components.<Name>]`
61///   table-of-tables enumerates each.
62/// * Application crate — `[package.metadata.nros.application]` describes a
63///   native-only application pkg (per Phase 212.L.2).
64///
65/// Phase 212.L.7 also adds an optional per-target deploy table at
66/// `[package.metadata.nros.deploy.<target>]`, used both by application pkgs
67/// and by self-bringup component pkgs (component pkg w/ `[deploy.*]` and no
68/// sibling bringup acts as its own bringup).
69///
70/// At most one of `component` / `components` / `application` may be present;
71/// the loader validates this after deserialization (a serde untagged enum
72/// would lose the precise `deny_unknown_fields` error, so we keep the fields
73/// flat and reject conflicts in [`PackageMetadataNros::validate`]).
74#[derive(Clone, Debug, Default, PartialEq, Serialize, Deserialize)]
75#[serde(deny_unknown_fields)]
76pub struct PackageMetadataNros {
77    #[serde(default, skip_serializing_if = "Option::is_none")]
78    pub component: Option<ComponentMetadata>,
79    /// Phase 212.N.12 (in-flight) — `node` is the forward-looking spelling of
80    /// the `component` shape. The reader accepts BOTH spellings during the
81    /// in-flight rename (Phase 212.B.2 task spec). Mutually exclusive with
82    /// `component` / `components` / `nodes` / `application` (validated below).
83    /// The shape is identical to [`ComponentMetadata`] so codegen can treat
84    /// the two interchangeably until N.12 retires the `component` spelling.
85    #[serde(default, skip_serializing_if = "Option::is_none")]
86    pub node: Option<ComponentMetadata>,
87    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
88    pub components: BTreeMap<String, ComponentMetadata>,
89    /// Phase 212.N.12 in-flight — `nodes` is the forward-looking spelling
90    /// of the `components` (multi-shape) table. Same shape, accepted as an
91    /// alias during the rename. Mutually exclusive with `components`.
92    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
93    pub nodes: BTreeMap<String, ComponentMetadata>,
94    /// Phase 212.L.2 — `[package.metadata.nros.application]`. Application
95    /// pkgs are native-only orchestration roots; they MUST NOT name an RTOS
96    /// in their `deploy = […]` allow-list. (The `nros check` lint enforces
97    /// the no-RTOS rule; the schema only accepts the field.)
98    #[serde(default, skip_serializing_if = "Option::is_none")]
99    pub application: Option<ApplicationMetadata>,
100    /// Phase 212.N.7 — `[package.metadata.nros.entry]`. Entry pkgs declare
101    /// which `[deploy.<target>]` block they run on (the firmware bin pulls
102    /// in the per-board shim + emits `run_plan(runtime)`). Strict schema —
103    /// `deploy = "<board>"` is the only field today.
104    #[serde(default, skip_serializing_if = "Option::is_none")]
105    pub entry: Option<EntryMetadata>,
106    /// Phase 212.L.7 / L.8 — per-target deploy tables, keyed by target name
107    /// (`native`, `qemu-mps2-an385`, …). Populates both application pkgs and
108    /// self-bringup component pkgs (component pkg w/ `[deploy.*]` and no
109    /// sibling bringup eats its own bringup role).
110    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
111    pub deploy: BTreeMap<String, DeployTargetMetadata>,
112    /// Phase 212.B.2 stub (`[package.metadata.nros.domain]`) — opaque
113    /// pass-through during the schema in-flight window. The full typed shape
114    /// lands with system.toml's F.4 work. Captured as `toml::Value` so
115    /// `deny_unknown_fields` still surfaces typos elsewhere while letting
116    /// users author the table.
117    #[serde(default, skip_serializing_if = "Option::is_none")]
118    pub domain: Option<toml::Value>,
119    /// Phase 212.B.2 stub (`[package.metadata.nros.bridge]`) — opaque
120    /// pass-through, same rationale as `domain`.
121    #[serde(default, skip_serializing_if = "Option::is_none")]
122    pub bridge: Option<toml::Value>,
123    /// Phase 212.B.2 stub (`[package.metadata.nros.embedded]`) — opaque
124    /// pass-through. Will eventually hold board-specific embedded knobs
125    /// (`linker_script` / `stack_size` / …); kept opaque so the reader
126    /// doesn't break the moment a board author authors the table.
127    #[serde(default, skip_serializing_if = "Option::is_none")]
128    pub embedded: Option<toml::Value>,
129}
130
131impl PackageMetadataNros {
132    /// Reject manifests that combine more than one of `component` /
133    /// `node` / `components` / `nodes` / `application` — the shapes are
134    /// mutually exclusive per the Phase 212.L design doc plus the Phase
135    /// 212.N.12 rename (the `node` / `nodes` spellings are aliases of
136    /// `component` / `components`, not new categories).
137    pub fn validate(&self) -> Result<(), String> {
138        let has_component = self.component.is_some();
139        let has_node = self.node.is_some();
140        let has_components = !self.components.is_empty();
141        let has_nodes = !self.nodes.is_empty();
142        let has_application = self.application.is_some();
143        let count = [
144            has_component,
145            has_node,
146            has_components,
147            has_nodes,
148            has_application,
149        ]
150        .into_iter()
151        .filter(|b| *b)
152        .count();
153        if count > 1 {
154            return Err("`[package.metadata.nros]` carries more than one of \
155                 `component` / `node` / `components` / `nodes` / `application` — pick exactly \
156                 one shape (Phase 212.L.2 / L.7; N.12 rename in flight — `node` / \
157                 `nodes` are the forward-looking spellings of `component` / `components`)"
158                .to_string());
159        }
160        Ok(())
161    }
162
163    /// Phase 212.N.12 in-flight rename — `component` and `node` are aliases.
164    /// Returns the present shape, preferring `node` (the forward spelling)
165    /// over `component`. Callers reading the per-pkg shape go through this
166    /// accessor so the N.12 sweep can later flip the storage field without
167    /// touching every read site.
168    pub fn node_or_component(&self) -> Option<&ComponentMetadata> {
169        self.node.as_ref().or(self.component.as_ref())
170    }
171
172    /// Phase 212.N.12 in-flight rename — multi-shape accessor. Returns
173    /// `nodes` when present, else `components`. Empty if neither populated.
174    pub fn nodes_or_components(&self) -> &BTreeMap<String, ComponentMetadata> {
175        if !self.nodes.is_empty() {
176            &self.nodes
177        } else {
178            &self.components
179        }
180    }
181
182    /// True when this manifest is a *self-bringup-eligible* component or
183    /// application pkg (Phase 212.L.7): it declares its component/application
184    /// surface AND at least one `[package.metadata.nros.deploy.<target>]`
185    /// table. The planner / codegen path treats such a pkg as its own
186    /// degenerate 1-component bringup when no sibling bringup pkg points at
187    /// it.
188    pub fn is_self_bringup_eligible(&self) -> bool {
189        let has_role = self.component.is_some()
190            || self.node.is_some()
191            || !self.components.is_empty()
192            || !self.nodes.is_empty()
193            || self.application.is_some();
194        has_role && !self.deploy.is_empty()
195    }
196}

198/// `[package.metadata.nros.entry]` — Phase 212.N.7.
199///
200/// Marks an Entry pkg (the firmware bin) so the planner can route it to the
201/// right `[deploy.<target>]` block. Today the only field is `deploy =
202/// "<board>"` (the deploy-target key in the workspace deploy map). The
203/// reader keeps this strict so a typo on `deploy =` surfaces immediately.
204#[derive(Clone, Debug, Default, PartialEq, Eq, Serialize, Deserialize)]
205#[serde(deny_unknown_fields)]
206pub struct EntryMetadata {
207    /// Board / deploy-target key (e.g. `"freertos"`, `"zephyr"`).
208    pub deploy: String,
209}

211/// `[package.metadata.nros.node]` (single shape, canonical post Phase
212/// 212.N.12) — or `[package.metadata.nros.component]` (deprecated alias)
213/// — or `[package.metadata.nros.components.<Name>]` (multi shape).
214///
215/// Pure deployment intent — no build-system knobs (Cargo + CMake own those).
216#[derive(Clone, Debug, Default, PartialEq, Serialize, Deserialize)]
217#[serde(deny_unknown_fields)]
218pub struct ComponentMetadata {
219    /// Phase 212.L.4 — fully-qualified class name (`<pkg-dir>::<UserClass>`).
220    /// Lint-enforced by `nros check` to match the host pkg name; codegen
221    /// uses it to land at the right Rust type.
222    #[serde(default, skip_serializing_if = "Option::is_none")]
223    pub class: Option<String>,
224    /// Short component instance name (used as the planner / codegen
225    /// instance identifier when the pkg is its own self-bringup, per
226    /// Phase 212.L.7).
227    #[serde(default, skip_serializing_if = "Option::is_none")]
228    pub name: Option<String>,
229    /// Default namespace the component is mounted at. Absent → `/`.
230    #[serde(default, skip_serializing_if = "Option::is_none")]
231    pub default_namespace: Option<String>,
232    /// Raw ROS parameter declarations. Values stay as `toml::Value` here so
233    /// the planner can do its own type-aware lowering (mirrors the existing
234    /// `params::ParameterTable` resolution path).
235    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
236    pub parameters: BTreeMap<String, toml::Value>,
237    /// `from` → `to` topic / service remaps, mirroring rclpy / rclcpp.
238    /// Aliased to [`RemapRule`] (already `{from, to}`-shaped in
239    /// `super::schema`) to avoid creating a duplicate type.
240    #[serde(default, skip_serializing_if = "Vec::is_empty")]
241    pub remaps: Vec<RemapRule>,
242}

244/// `[package.metadata.nros.application]` — Phase 212.L.2.
245///
246/// Application pkgs are native-only orchestration roots: they wire several
247/// component pkgs together but MUST NOT name an RTOS target in their
248/// `deploy = […]` allow-list. The allow-list semantics are enforced by
249/// `nros check`; the schema only accepts the field.
250///
251/// The application's per-target deploy block lives on the outer
252/// `[package.metadata.nros.deploy.<target>]` (shared with self-bringup
253/// component pkgs), not nested under `[application]`.
254#[derive(Clone, Debug, Default, PartialEq, Eq, Serialize, Deserialize)]
255#[serde(deny_unknown_fields)]
256pub struct ApplicationMetadata {
257    /// Allow-list of deploy targets — keys into the outer
258    /// `[package.metadata.nros.deploy.<target>]` map. Must not include an
259    /// RTOS target name (lint-enforced by `nros check`).
260    #[serde(default, skip_serializing_if = "Vec::is_empty")]
261    pub deploy: Vec<String>,
262    /// Optional short app name; falls back to the Cargo `[package].name`
263    /// when absent.
264    #[serde(default, skip_serializing_if = "Option::is_none")]
265    pub name: Option<String>,
266}

270/// Per-target deploy parameters baked into the bringup tree by `nros

275pub struct DeployTargetMetadata {

279    /// RMW backend (`zenoh` / `xrce` / `cyclonedds`).

282    /// Baked ROS_DOMAIN_ID — embedded targets bake at build time.

285    /// Optional RMW locator URI (e.g. `tcp/127.0.0.1:7447`).
286    #[serde(default, skip_serializing_if = "Option::is_none")]
287    pub locator: Option<String>,
288}

319/// `<bringup>/system.toml` — the authoritative system spec.
320///
321/// Sections (see `docs/design/workspace-layout-by-case.md` Case 3/4 and
322/// `multi-node-workspace-layout.md` §4):
323///
324/// * `[system]` — name, RMW, domain, optional locator.
325/// * `[[component]]` — one entry per node/component.
326/// * `[deploy.<target>]` — per-target deploy block (`kind = "self" | "qemu"
327///   | "flash" | …`).
328/// * `[[domain]]` — optional per-system domain routing.
329/// * `[[bridge]]` — optional cross-RMW bridges.
330#[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]
331#[serde(deny_unknown_fields)]
332pub struct SystemToml {
333    pub system: SystemHeader,
334    #[serde(default, rename = "component", skip_serializing_if = "Vec::is_empty")]
335    pub components: Vec<SystemComponentEntry>,
336    /// `[deploy.<target>]` — keyed by target name (e.g. `native`,
337    /// `qemu-mps2-an385`, `flash-stm32f4-disco`).
338    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
339    pub deploy: BTreeMap<String, DeployTarget>,
340    #[serde(default, rename = "domain", skip_serializing_if = "Vec::is_empty")]
341    pub domains: Vec<SystemDomainEntry>,
342    #[serde(default, rename = "bridge", skip_serializing_if = "Vec::is_empty")]
343    pub bridges: Vec<SystemBridgeEntry>,
344}

380/// `[deploy.<target>]` block.
381///
382/// Per the F.4 §12 known-gap #2 resolution (path a — relax parser): both
383/// `kind` and `target` are OPTIONAL. The deploy block is configuration-
384/// by-target — the `<target>` map key (e.g. `native`, `qemu-mps2-an385`,
385/// `threadx-linux`, `platformio`) already names the runner, and the
386/// runner stage derives sensible defaults for `kind` / `target` from
387/// the target name when these fields are absent. Strict
388/// `deny_unknown_fields` is preserved — widening the schema, not
389/// loosening the policy.
390///
391/// `nros check` is the place to surface a heads-up when `kind`/`target`
392/// is absent AND the runner can't synthesise defaults from the target
393/// name; that's a lint, not a parser error.
394#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
395#[serde(deny_unknown_fields)]
396pub struct DeployTarget {
397    /// `"self"`, `"qemu"`, `"flash"`, … — interpreted by the runner stage.
398    /// Optional (F.4 §12 gap #2): absent ⇒ runner derives from the
399    /// `<target>` map key.
400    #[serde(default, skip_serializing_if = "Option::is_none")]
401    pub kind: Option<String>,
402    /// Target triple / board id / runner key. The exact semantics depend
403    /// on `kind`. Optional (F.4 §12 gap #2): absent ⇒ runner derives
404    /// from the `<target>` map key + the chosen `kind`.
405    #[serde(default, skip_serializing_if = "Option::is_none")]
406    pub target: Option<String>,
407    /// Optional path (relative to the bringup pkg root) to a
408    /// `launch/*.launch.xml` used for this deploy.
409    #[serde(default, skip_serializing_if = "Option::is_none")]
410    pub launch: Option<String>,
411    /// Optional board identifier (e.g. `mps2_an385`, `qemu_riscv64`).
412    #[serde(default, skip_serializing_if = "Option::is_none")]
413    pub board: Option<String>,
414    /// Optional framework identifier for runners that surface one — today
415    /// PlatformIO (`"espidf"`, `"arduino"`, …) and indirectly ESP-IDF. Held
416    /// verbatim and forwarded to the runner stage; no schema-level
417    /// validation. Resolves F.4 §12 known gap #3 (the platformio fixture
418    /// authored `framework = "espidf"` against a `DeployTarget` that
419    /// didn't know the field).
420    #[serde(default, skip_serializing_if = "Option::is_none")]
421    pub framework: Option<String>,
422}

450    /// Round-trip a `[workspace.metadata.nros]` golden through parse +
451    /// serialize + reparse and compare structs.
452    #[test]
453    fn workspace_metadata_round_trip() {
454        let raw = r#"
455default_system = "demo_bringup"
456rmw_override = "cyclonedds"
457"#;
458        let v1: WorkspaceMetadataNros = toml::from_str(raw).expect("parse golden");
459        assert_eq!(v1.default_system.as_deref(), Some("demo_bringup"));
460        assert_eq!(v1.rmw_override.as_deref(), Some("cyclonedds"));
461
462        let reserialized = toml::to_string(&v1).expect("serialize");
463        let v2: WorkspaceMetadataNros = toml::from_str(&reserialized).expect("reparse");
464        assert_eq!(v1, v2);
465    }

485    /// `[package.metadata.nros.component]` single-shape round-trip.

487    fn package_metadata_single_component_round_trip() {
488        let raw = r#"
489[component]
490default_namespace = "/demo"
491
492[component.parameters]
493rate_hz = 10
494greeting = "hello"
495
496[[component.remaps]]
497from = "chatter"
498to = "topic/chatter"
499"#;
500        let v1: PackageMetadataNros = toml::from_str(raw).expect("parse");
501        v1.validate().expect("single-shape is valid");
502        let component = v1.component.as_ref().expect("component present");
