1//! Phase 212.B — `[workspace.metadata.nros]` + `[package.metadata.nros]` +
2//! `[package.metadata.ament]` + `<bringup>/system.toml` data contracts.
3//!
4//! These are the *user-authored* TOML surfaces introduced by Phase 212. They
5//! live in standard cargo manifest tables (`[workspace.metadata.…]` /
6//! `[package.metadata.…]`) so that cargo treats them as opaque user data and
7//! pure-cargo workflows (no CLI build wrapper) keep working. The
8//! `<bringup>/system.toml` is the per-system declarative file owned by the
9//! `<system>_bringup` package.
10//!
11//! Vocabulary discipline (per the Phase 212 doc): every field name is a strict
12//! subset of names that already appear in `nros-sdk-index.toml`,
13//! `app_config.h`, or the existing planner schema. No second TOML dialect.
14//!
15//! Every struct here uses `#[serde(deny_unknown_fields)]` so typos surface as
16//! parse errors at the user's terminal instead of being silently dropped.
17

36/// `[workspace.metadata.nros]` in a workspace-root `Cargo.toml`.
37///
38/// Thin pointer (see `docs/design/0024-multi-node-workspace-layout.md` §5). The
39/// authoritative system spec lives in `<bringup>/system.toml`; this table
40/// only disambiguates which bringup the workspace defaults to plus a small
41/// set of rarely-used workspace-wide overrides.
42///
43/// Per the Phase 212.L.7 redesign, `default_system` may name EITHER a
44/// bringup package (`<system>_bringup`) OR a Node/Entry pkg that eats its
45/// own Entry role (single-pkg `cargo run` dev loop). The launcher walks
46/// the workspace and resolves the pointer against either category.
47///
48/// All fields are optional; an absent `[workspace.metadata.nros]` table
49/// parses as `WorkspaceMetadataNros::default()`.
50#[derive(Clone, Debug, Default, PartialEq, Eq, Serialize, Deserialize)]
51#[serde(deny_unknown_fields)]
52pub struct WorkspaceMetadataNros {
53    /// Bringup package name (`<system>_bringup`) OR Entry pkg name
54    /// (Phase 212.L.7 self-entry shape). `nros plan` /
55    /// `nros codegen-system` with no `--bringup` hint resolves the
56    /// system via this pointer.
57    #[serde(default, skip_serializing_if = "Option::is_none")]
58    pub default_system: Option<String>,
59    /// Optional workspace-wide RMW override — rare, intended for
60    /// `nros plan --override` workflows. Values are `"zenoh"` /
61    /// `"xrce"` / `"cyclonedds"`.
62    #[serde(default, skip_serializing_if = "Option::is_none")]
63    pub rmw_override: Option<String>,
64    /// Optional workspace-wide `ROS_DOMAIN_ID` override. When present,
65    /// `nros plan` / `nros codegen-system` propagate it into the
66    /// generated `system_config.h` instead of the per-deploy /
67    /// `[system].domain_id` value. Rare — used for one-off bring-ups
68    /// against a shared ROS 2 graph.
69    #[serde(default, skip_serializing_if = "Option::is_none")]
70    pub domain_id_override: Option<u32>,
71}

77/// `[package.metadata.nros]` in a component / application package's
78/// `Cargo.toml`.
79///
80/// Three top-level shapes (mutually exclusive):
81///
82/// * Single-node crate — `[package.metadata.nros.node]` describes the
83///   one node the crate exposes. (The Phase 212.N.12 rename made
84///   `node` the canonical key; `[package.metadata.nros.component]`
85///   remains accepted as a deprecated alias — declaring both is a
86///   hard error. See `nros_config::parse_package_metadata_nros`.)
87/// * Multi-component crate — `[package.metadata.nros.components.<Name>]`
88///   table-of-tables enumerates each.
89/// * Application crate — `[package.metadata.nros.application]` describes a
90///   native-only application pkg (per Phase 212.L.2).
91///
92/// Phase 212.L.7 also adds an optional per-target deploy table at
93/// `[package.metadata.nros.deploy.<target>]`, used both by application pkgs
94/// and by self-bringup component pkgs (component pkg w/ `[deploy.*]` and no
95/// sibling bringup acts as its own bringup).
96///
97/// At most one of `component` / `components` / `application` may be present;
98/// the loader validates this after deserialization (a serde untagged enum
99/// would lose the precise `deny_unknown_fields` error, so we keep the fields
100/// flat and reject conflicts in [`PackageMetadataNros::validate`]).
101#[derive(Clone, Debug, Default, PartialEq, Serialize, Deserialize)]
102#[serde(deny_unknown_fields)]
103pub struct PackageMetadataNros {
104    #[serde(default, skip_serializing_if = "Option::is_none")]
105    pub component: Option<ComponentMetadata>,
106    /// Phase 212.N.12 (in-flight) — `node` is the forward-looking spelling of
107    /// the `component` shape. The reader accepts BOTH spellings during the
108    /// in-flight rename (Phase 212.B.2 task spec). Mutually exclusive with
109    /// `component` / `components` / `nodes` / `application` (validated below).
110    /// The shape is identical to [`ComponentMetadata`] so codegen can treat
111    /// the two interchangeably until N.12 retires the `component` spelling.
112    #[serde(default, skip_serializing_if = "Option::is_none")]
113    pub node: Option<ComponentMetadata>,
114    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
115    pub components: BTreeMap<String, ComponentMetadata>,
116    /// Phase 212.N.12 in-flight — `nodes` is the forward-looking spelling
117    /// of the `components` (multi-shape) table. Same shape, accepted as an
118    /// alias during the rename. Mutually exclusive with `components`.
119    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
120    pub nodes: BTreeMap<String, ComponentMetadata>,
121    /// Phase 212.L.2 — `[package.metadata.nros.application]`. Application
122    /// pkgs are native-only orchestration roots; they MUST NOT name an RTOS
123    /// in their `deploy = […]` allow-list. (The `nros check` lint enforces
124    /// the no-RTOS rule; the schema only accepts the field.)
125    #[serde(default, skip_serializing_if = "Option::is_none")]
126    pub application: Option<ApplicationMetadata>,
127    /// Phase 212.N.7 — `[package.metadata.nros.entry]`. Entry pkgs declare
128    /// which `[deploy.<target>]` block they run on (the firmware bin pulls
129    /// in the per-board shim + emits `run_plan(runtime)`). Strict schema —
130    /// `deploy = "<board>"` is the only field today.
131    #[serde(default, skip_serializing_if = "Option::is_none")]
132    pub entry: Option<EntryMetadata>,
133    /// Phase 212.L.7 / L.8 — per-target deploy tables, keyed by target name
134    /// (`native`, `qemu-mps2-an385`, …). Populates both application pkgs and
135    /// self-bringup component pkgs (component pkg w/ `[deploy.*]` and no
136    /// sibling bringup eats its own bringup role).
137    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
138    pub deploy: BTreeMap<String, DeployTargetMetadata>,
139    /// Phase 212.B.2 stub (`[package.metadata.nros.domain]`) — opaque
140    /// pass-through during the schema in-flight window. The full typed shape
141    /// lands with system.toml's F.4 work. Captured as `toml::Value` so
142    /// `deny_unknown_fields` still surfaces typos elsewhere while letting
143    /// users author the table.
144    #[serde(default, skip_serializing_if = "Option::is_none")]
145    pub domain: Option<toml::Value>,
146    /// Phase 212.B.2 stub (`[package.metadata.nros.bridge]`) — opaque
147    /// pass-through, same rationale as `domain`.
148    #[serde(default, skip_serializing_if = "Option::is_none")]
149    pub bridge: Option<toml::Value>,
150    /// Phase 212.B.2 stub (`[package.metadata.nros.embedded]`) — opaque
151    /// pass-through. Will eventually hold board-specific embedded knobs
152    /// (`linker_script` / `stack_size` / …); kept opaque so the reader
153    /// doesn't break the moment a board author authors the table.
154    #[serde(default, skip_serializing_if = "Option::is_none")]
155    pub embedded: Option<toml::Value>,
156}

158impl PackageMetadataNros {
159    /// Reject manifests that combine more than one of `component` /
160    /// `node` / `components` / `nodes` / `application` — the shapes are
161    /// mutually exclusive per the Phase 212.L design doc plus the Phase
162    /// 212.N.12 rename (the `node` / `nodes` spellings are aliases of
163    /// `component` / `components`, not new categories).
164    pub fn validate(&self) -> Result<(), String> {
165        let has_component = self.component.is_some();
166        let has_node = self.node.is_some();
167        let has_components = !self.components.is_empty();
168        let has_nodes = !self.nodes.is_empty();
169        let has_application = self.application.is_some();
170        let count = [
171            has_component,
172            has_node,
173            has_components,
174            has_nodes,
175            has_application,
176        ]
177        .into_iter()
178        .filter(|b| *b)
179        .count();
180        if count > 1 {
181            return Err("`[package.metadata.nros]` carries more than one of \
182                 `component` / `node` / `components` / `nodes` / `application` — pick exactly \
183                 one shape (Phase 212.L.2 / L.7; N.12 rename in flight — `node` / \
184                 `nodes` are the forward-looking spellings of `component` / `components`)"
185                .to_string());
186        }
187        Ok(())
188    }
189
190    /// Phase 212.N.12 in-flight rename — `component` and `node` are aliases.
191    /// Returns the present shape, preferring `node` (the forward spelling)
192    /// over `component`. Callers reading the per-pkg shape go through this
193    /// accessor so the N.12 sweep can later flip the storage field without
194    /// touching every read site.
195    pub fn node_or_component(&self) -> Option<&ComponentMetadata> {
196        self.node.as_ref().or(self.component.as_ref())
197    }
198
199    /// Phase 212.N.12 in-flight rename — multi-shape accessor. Returns
200    /// `nodes` when present, else `components`. Empty if neither populated.
201    pub fn nodes_or_components(&self) -> &BTreeMap<String, ComponentMetadata> {
202        if !self.nodes.is_empty() {
203            &self.nodes
204        } else {
205            &self.components
206        }
207    }
208
209    /// True when this manifest is a *self-bringup-eligible* component or
210    /// application pkg (Phase 212.L.7): it declares its component/application
211    /// surface AND at least one `[package.metadata.nros.deploy.<target>]`
212    /// table. The planner / codegen path treats such a pkg as its own
213    /// degenerate 1-component bringup when no sibling bringup pkg points at
214    /// it.
215    pub fn is_self_bringup_eligible(&self) -> bool {
216        let has_role = self.component.is_some()
217            || self.node.is_some()
218            || !self.components.is_empty()
219            || !self.nodes.is_empty()
220            || self.application.is_some();
221        has_role && !self.deploy.is_empty()
222    }
223}

331/// `[package.metadata.ament]` — the source of truth for `nros emit
332/// package-xml` (Phase 212.G). Mirrors ament/colcon's `package.xml`
333/// vocabulary 1-to-1.
334///
335/// Vocabulary (Phase 212.B.4):
336///
337/// * `description` / `license` — passthrough to `<description>` /
338///   `<license>` in the emitted `package.xml`. When absent the emitter
339///   falls back to a synthesised description + `"Apache-2.0"`.
340/// * `maintainer = { name, email }` — populates the single
341///   `<maintainer email="…">…</maintainer>` row. Multiple maintainers
342///   are not modelled yet — ROS allows several `<maintainer>` rows,
343///   but every in-tree fixture authors at most one.
344/// * `build_depend` / `exec_depend` / `test_depend` /
345///   `buildtool_depend` — each row emits a corresponding `<*_depend>`
346///   entry. Sorted + deduped at emit time so list ordering doesn't
347///   drift between edits.
348/// * `build_type` — `<export><build_type>…</build_type></export>`.
349///   Defaults differ between component pkgs (`ament_cargo`) and
350///   bringup pkgs (`ament_cmake`).

352#[serde(deny_unknown_fields)]
353pub struct PackageMetadataAment {
354    /// `<description>` body. Absent → emitter synthesises a generic
355    /// "`nano-ros component package <name>`" line.
356    #[serde(default, skip_serializing_if = "Option::is_none")]
357    pub description: Option<String>,
358    /// `<license>` body (SPDX identifier). Absent → `"Apache-2.0"`.
359    #[serde(default, skip_serializing_if = "Option::is_none")]
360    pub license: Option<String>,
361    /// `<maintainer email="…">name</maintainer>` row. Absent →
362    /// emitter falls back to a placeholder `Developer <dev@example.com>`.
363    #[serde(default, skip_serializing_if = "Option::is_none")]
364    pub maintainer: Option<AmentMaintainer>,
365    /// `<build_depend>` rows.
366    #[serde(default, skip_serializing_if = "Vec::is_empty")]
367    pub build_depend: Vec<String>,
368    /// `<buildtool_depend>` rows (e.g. `"ament_cargo"`, `"ament_cmake"`).
369    /// Phase 212.B.4: explicit dependency category so users opting in
370    /// to colcon interop can author `buildtool_depend = ["ament_cmake"]`
371    /// without polluting `<build_depend>`.
372    #[serde(default, skip_serializing_if = "Vec::is_empty")]
373    pub buildtool_depend: Vec<String>,
374    /// `<exec_depend>` rows.
375    #[serde(default, skip_serializing_if = "Vec::is_empty")]
376    pub exec_depend: Vec<String>,
377    /// `<test_depend>` rows.
378    #[serde(default, skip_serializing_if = "Vec::is_empty")]
379    pub test_depend: Vec<String>,
380    /// `<export><build_type>…</build_type></export>` body. Component
381    /// pkgs default to `"ament_cargo"`; bringup pkgs default to
382    /// `"ament_cmake"`.
383    #[serde(default, skip_serializing_if = "Option::is_none")]
384    pub build_type: Option<String>,
385}

456    /// Phase 254 — `[param_services]`: the external ROS 2 parameter server.

459}

485impl SystemToml {
486    /// Phase 255 — the RMW backend name for `target`, applying the RFC-0031
487    /// precedence (highest wins): the CLI `--rmw` flag, then `[deploy.<target>].rmw`,
488    /// then `[system].rmw`, then the `"zenoh"` default. Both codegen paths (the
489    /// planner's board-feature lowering + the bake's C `#define`) resolve through
490    /// this single helper so a given target gets exactly one RMW — no duality.
491    pub fn resolved_rmw(&self, target: Option<&str>, cli: Option<&str>) -> String {
492        if let Some(c) = cli {
493            return c.to_string();
494        }
495        if let Some(t) = target
496            && let Some(dt) = self.deploy.get(t)
497            && let Some(r) = &dt.rmw
498        {
499            return r.clone();
500        }
501        if self.system.rmw.is_empty() {
502            "zenoh".to_string()
503        } else {
504            self.system.rmw.clone()
505        }
506    }
507
508    /// Phase 255 Wave 5 — the multi-RMW link set a single binary needs when it
509    /// hosts cross-RMW `[[bridge]]`s. A single binary must link the **union** of
510    /// every bridged session's RMW so it can speak both sides. Returned in
511    /// declaration order, deduped, and always seeded with the system default
512    /// (`resolved_rmw(None, None)`) so the local side links too.
513    ///
514    /// Each `[[bridge]]` names two endpoints (`from`/`to`). An endpoint is an
515    /// `<rmw>:<domain>` session selector (e.g. `cyclone:default`) — the RMW is
516    /// the prefix before `:`. A bare endpoint with no `:` is read as a
517    /// `[[domain]]` name and resolved to that domain's `rmw`. Unresolvable bare
518    /// endpoints are skipped (the `nros check` provenance pass flags them; issue
519    /// 0076 §A).
520    ///
521    /// Empty `[[bridge]]` set ⇒ a single-element vec (just the default) ⇒ the
522    /// build is byte-identical to a non-bridged single-RMW system.
523    pub fn bridged_rmws(&self) -> Vec<String> {
524        let mut set: Vec<String> = vec![self.resolved_rmw(None, None)];
525        let domain_rmw = |name: &str| -> Option<String> {
526            self.domains
527                .iter()
528                .find(|d| d.name == name)
529                .map(|d| d.rmw.clone())
530        };
531        for bridge in &self.bridges {
532            for endpoint in [&bridge.from, &bridge.to] {
533                // `<rmw>:<domain>` → the RMW is the prefix; a bare name is a
534                // `[[domain]]` reference resolved to its `rmw`.
535                let rmw = match endpoint.split_once(':') {
536                    Some((rmw, _domain)) => Some(rmw.to_string()),
537                    None => domain_rmw(endpoint),
538                };
539                if let Some(rmw) = rmw
540                    && !rmw.is_empty()
541                    && !set.contains(&rmw)
542                {
543                    set.push(rmw);
544                }
545            }
546        }
547        set
548    }
549}
