1//! Phase 212.B — `[workspace.metadata.nros]` + `[package.metadata.nros]` +
2//! `[package.metadata.ament]` + `<bringup>/system.toml` data contracts.
3//!
4//! These are the *user-authored* TOML surfaces introduced by Phase 212. They
5//! live in standard cargo manifest tables (`[workspace.metadata.…]` /
6//! `[package.metadata.…]`) so that cargo treats them as opaque user data and
7//! pure-cargo workflows (no `nros build` verb) keep working. The
8//! `<bringup>/system.toml` is the per-system declarative file owned by the
9//! `<system>_bringup` package.
10//!
11//! Vocabulary discipline (per the Phase 212 doc): every field name is a strict
12//! subset of names that already appear in `nros-sdk-index.toml`,
13//! `app_config.h`, or the existing planner schema. No second TOML dialect.
14//!
15//! Every struct here uses `#[serde(deny_unknown_fields)]` so typos surface as
16//! parse errors at the user's terminal instead of being silently dropped.
17

28/// `[workspace.metadata.nros]` in a workspace-root `Cargo.toml`.
29///
30/// Thin pointer (see `docs/design/multi-node-workspace-layout.md` §5). The
31/// authoritative system spec lives in `<bringup>/system.toml`; this table
32/// only disambiguates which bringup the workspace defaults to plus a small
33/// set of rarely-used workspace-wide overrides.
34///
35/// Per the Phase 212.L.7 redesign, `default_system` may name EITHER a
36/// bringup package (`<system>_bringup`) OR a Node/Entry pkg that eats its
37/// own Entry role (single-pkg `cargo run` dev loop). The launcher walks
38/// the workspace and resolves the pointer against either category.
39///
40/// All fields are optional; an absent `[workspace.metadata.nros]` table
41/// parses as `WorkspaceMetadataNros::default()`.
42#[derive(Clone, Debug, Default, PartialEq, Eq, Serialize, Deserialize)]
43#[serde(deny_unknown_fields)]
44pub struct WorkspaceMetadataNros {
45    /// Bringup package name (`<system>_bringup`) OR Entry pkg name
46    /// (Phase 212.L.7 self-entry shape). `nros plan` / `nros launch` /
47    /// `nros codegen-system` with no `--bringup` hint resolves the
48    /// system via this pointer.
49    #[serde(default, skip_serializing_if = "Option::is_none")]
50    pub default_system: Option<String>,
51    /// Optional workspace-wide RMW override — rare, intended for
52    /// `nros plan --override` workflows. Values are `"zenoh"` /
53    /// `"xrce"` / `"cyclonedds"`.
54    #[serde(default, skip_serializing_if = "Option::is_none")]
55    pub rmw_override: Option<String>,
56    /// Optional workspace-wide `ROS_DOMAIN_ID` override. When present,
57    /// `nros plan` / `nros codegen-system` propagate it into the
58    /// generated `system_config.h` instead of the per-deploy /
59    /// `[system].domain_id` value. Rare — used for one-off bring-ups
60    /// against a shared ROS 2 graph.
61    #[serde(default, skip_serializing_if = "Option::is_none")]
62    pub domain_id_override: Option<u32>,
63}

69/// `[package.metadata.nros]` in a component / application package's
70/// `Cargo.toml`.
71///
72/// Three top-level shapes (mutually exclusive):
73///
74/// * Single-node crate — `[package.metadata.nros.node]` describes the
75///   one node the crate exposes. (The Phase 212.N.12 rename made
76///   `node` the canonical key; `[package.metadata.nros.component]`
77///   remains accepted as a deprecated alias — declaring both is a
78///   hard error. See `nros_config::parse_package_metadata_nros`.)
79/// * Multi-component crate — `[package.metadata.nros.components.<Name>]`
80///   table-of-tables enumerates each.
81/// * Application crate — `[package.metadata.nros.application]` describes a
82///   native-only application pkg (per Phase 212.L.2).
83///
84/// Phase 212.L.7 also adds an optional per-target deploy table at
85/// `[package.metadata.nros.deploy.<target>]`, used both by application pkgs
86/// and by self-bringup component pkgs (component pkg w/ `[deploy.*]` and no
87/// sibling bringup acts as its own bringup).
88///
89/// At most one of `component` / `components` / `application` may be present;
90/// the loader validates this after deserialization (a serde untagged enum
91/// would lose the precise `deny_unknown_fields` error, so we keep the fields
92/// flat and reject conflicts in [`PackageMetadataNros::validate`]).
93#[derive(Clone, Debug, Default, PartialEq, Serialize, Deserialize)]
94#[serde(deny_unknown_fields)]
95pub struct PackageMetadataNros {
96    #[serde(default, skip_serializing_if = "Option::is_none")]
97    pub component: Option<ComponentMetadata>,
98    /// Phase 212.N.12 (in-flight) — `node` is the forward-looking spelling of
99    /// the `component` shape. The reader accepts BOTH spellings during the
100    /// in-flight rename (Phase 212.B.2 task spec). Mutually exclusive with
101    /// `component` / `components` / `nodes` / `application` (validated below).
102    /// The shape is identical to [`ComponentMetadata`] so codegen can treat
103    /// the two interchangeably until N.12 retires the `component` spelling.
104    #[serde(default, skip_serializing_if = "Option::is_none")]
105    pub node: Option<ComponentMetadata>,
106    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
107    pub components: BTreeMap<String, ComponentMetadata>,
108    /// Phase 212.N.12 in-flight — `nodes` is the forward-looking spelling
109    /// of the `components` (multi-shape) table. Same shape, accepted as an
110    /// alias during the rename. Mutually exclusive with `components`.
111    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
112    pub nodes: BTreeMap<String, ComponentMetadata>,
113    /// Phase 212.L.2 — `[package.metadata.nros.application]`. Application
114    /// pkgs are native-only orchestration roots; they MUST NOT name an RTOS
115    /// in their `deploy = […]` allow-list. (The `nros check` lint enforces
116    /// the no-RTOS rule; the schema only accepts the field.)
117    #[serde(default, skip_serializing_if = "Option::is_none")]
118    pub application: Option<ApplicationMetadata>,
119    /// Phase 212.N.7 — `[package.metadata.nros.entry]`. Entry pkgs declare
120    /// which `[deploy.<target>]` block they run on (the firmware bin pulls
121    /// in the per-board shim + emits `run_plan(runtime)`). Strict schema —
122    /// `deploy = "<board>"` is the only field today.
123    #[serde(default, skip_serializing_if = "Option::is_none")]
124    pub entry: Option<EntryMetadata>,
125    /// Phase 212.L.7 / L.8 — per-target deploy tables, keyed by target name
126    /// (`native`, `qemu-mps2-an385`, …). Populates both application pkgs and
127    /// self-bringup component pkgs (component pkg w/ `[deploy.*]` and no
128    /// sibling bringup eats its own bringup role).
129    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
130    pub deploy: BTreeMap<String, DeployTargetMetadata>,
131    /// Phase 212.B.2 stub (`[package.metadata.nros.domain]`) — opaque
132    /// pass-through during the schema in-flight window. The full typed shape
133    /// lands with system.toml's F.4 work. Captured as `toml::Value` so
134    /// `deny_unknown_fields` still surfaces typos elsewhere while letting
135    /// users author the table.
136    #[serde(default, skip_serializing_if = "Option::is_none")]
137    pub domain: Option<toml::Value>,
138    /// Phase 212.B.2 stub (`[package.metadata.nros.bridge]`) — opaque
139    /// pass-through, same rationale as `domain`.
140    #[serde(default, skip_serializing_if = "Option::is_none")]
141    pub bridge: Option<toml::Value>,
142    /// Phase 212.B.2 stub (`[package.metadata.nros.embedded]`) — opaque
143    /// pass-through. Will eventually hold board-specific embedded knobs
144    /// (`linker_script` / `stack_size` / …); kept opaque so the reader
145    /// doesn't break the moment a board author authors the table.
146    #[serde(default, skip_serializing_if = "Option::is_none")]
147    pub embedded: Option<toml::Value>,
148}
149
150impl PackageMetadataNros {
151    /// Reject manifests that combine more than one of `component` /
152    /// `node` / `components` / `nodes` / `application` — the shapes are
153    /// mutually exclusive per the Phase 212.L design doc plus the Phase
154    /// 212.N.12 rename (the `node` / `nodes` spellings are aliases of
155    /// `component` / `components`, not new categories).
156    pub fn validate(&self) -> Result<(), String> {
157        let has_component = self.component.is_some();
158        let has_node = self.node.is_some();
159        let has_components = !self.components.is_empty();
160        let has_nodes = !self.nodes.is_empty();
161        let has_application = self.application.is_some();
162        let count = [
163            has_component,
164            has_node,
165            has_components,
166            has_nodes,
167            has_application,
168        ]
169        .into_iter()
170        .filter(|b| *b)
171        .count();
172        if count > 1 {
173            return Err("`[package.metadata.nros]` carries more than one of \
174                 `component` / `node` / `components` / `nodes` / `application` — pick exactly \
175                 one shape (Phase 212.L.2 / L.7; N.12 rename in flight — `node` / \
176                 `nodes` are the forward-looking spellings of `component` / `components`)"
177                .to_string());
178        }
179        Ok(())
180    }
181
182    /// Phase 212.N.12 in-flight rename — `component` and `node` are aliases.
183    /// Returns the present shape, preferring `node` (the forward spelling)
184    /// over `component`. Callers reading the per-pkg shape go through this
185    /// accessor so the N.12 sweep can later flip the storage field without
186    /// touching every read site.
187    pub fn node_or_component(&self) -> Option<&ComponentMetadata> {
188        self.node.as_ref().or(self.component.as_ref())
189    }
190
191    /// Phase 212.N.12 in-flight rename — multi-shape accessor. Returns
192    /// `nodes` when present, else `components`. Empty if neither populated.
193    pub fn nodes_or_components(&self) -> &BTreeMap<String, ComponentMetadata> {
194        if !self.nodes.is_empty() {
195            &self.nodes
196        } else {
197            &self.components
198        }
199    }
200
201    /// True when this manifest is a *self-bringup-eligible* component or
202    /// application pkg (Phase 212.L.7): it declares its component/application
203    /// surface AND at least one `[package.metadata.nros.deploy.<target>]`
204    /// table. The planner / codegen path treats such a pkg as its own
205    /// degenerate 1-component bringup when no sibling bringup pkg points at
206    /// it.
207    pub fn is_self_bringup_eligible(&self) -> bool {
208        let has_role = self.component.is_some()
209            || self.node.is_some()
210            || !self.components.is_empty()
211            || !self.nodes.is_empty()
212            || self.application.is_some();
213        has_role && !self.deploy.is_empty()
214    }
215}

218///
219/// Marks an Entry pkg (the firmware bin) so the planner can route it to the

230/// `[package.metadata.nros.node]` (single shape, canonical post Phase
231/// 212.N.12) — or `[package.metadata.nros.component]` (deprecated alias)
232/// — or `[package.metadata.nros.components.<Name>]` (multi shape).
233///
234/// Pure deployment intent — no build-system knobs (Cargo + CMake own those).
235#[derive(Clone, Debug, Default, PartialEq, Serialize, Deserialize)]
236#[serde(deny_unknown_fields)]
237pub struct ComponentMetadata {
238    /// Phase 212.L.4 — fully-qualified class name (`<pkg-dir>::<UserClass>`).
239    /// Lint-enforced by `nros check` to match the host pkg name; codegen
240    /// uses it to land at the right Rust type.
241    #[serde(default, skip_serializing_if = "Option::is_none")]
242    pub class: Option<String>,
243    /// Short component instance name (used as the planner / codegen
244    /// instance identifier when the pkg is its own self-bringup, per
245    /// Phase 212.L.7).
246    #[serde(default, skip_serializing_if = "Option::is_none")]
247    pub name: Option<String>,
248    /// Default namespace the component is mounted at. Absent → `/`.
249    #[serde(default, skip_serializing_if = "Option::is_none")]
250    pub default_namespace: Option<String>,
251    /// Raw ROS parameter declarations. Values stay as `toml::Value` here so
252    /// the planner can do its own type-aware lowering (mirrors the existing
253    /// `params::ParameterTable` resolution path).
254    #[serde(default, skip_serializing_if = "BTreeMap::is_empty")]
255    pub parameters: BTreeMap<String, toml::Value>,
256    /// `from` → `to` topic / service remaps, mirroring rclpy / rclcpp.
257    /// Aliased to [`RemapRule`] (already `{from, to}`-shaped in
258    /// `super::schema`) to avoid creating a duplicate type.
259    #[serde(default, skip_serializing_if = "Vec::is_empty")]
260    pub remaps: Vec<RemapRule>,
261}

263/// `[package.metadata.nros.application]` — Phase 212.L.2.
264///
265/// Application pkgs are native-only orchestration roots: they wire several
266/// component pkgs together but MUST NOT name an RTOS target in their

278    /// RTOS target name (lint-enforced by `nros check`).
279    #[serde(default, skip_serializing_if = "Vec::is_empty")]
280    pub deploy: Vec<String>,
281    /// Optional short app name; falls back to the Cargo `[package].name`
282    /// when absent.
283    #[serde(default, skip_serializing_if = "Option::is_none")]
284    pub name: Option<String>,
285}

317/// `[package.metadata.ament]` — the source of truth for `nros emit
318/// package-xml` (Phase 212.G). Mirrors ament/colcon's `package.xml`
319/// vocabulary 1-to-1.
320///
321/// Vocabulary (Phase 212.B.4):
322///
323/// * `description` / `license` — passthrough to `<description>` /
324///   `<license>` in the emitted `package.xml`. When absent the emitter
325///   falls back to a synthesised description + `"Apache-2.0"`.
326/// * `maintainer = { name, email }` — populates the single
327///   `<maintainer email="…">…</maintainer>` row. Multiple maintainers
328///   are not modelled yet — ROS allows several `<maintainer>` rows,
329///   but every in-tree fixture authors at most one.
330/// * `build_depend` / `exec_depend` / `test_depend` /
331///   `buildtool_depend` — each row emits a corresponding `<*_depend>`
332///   entry. Sorted + deduped at emit time so list ordering doesn't
333///   drift between edits.
334/// * `build_type` — `<export><build_type>…</build_type></export>`.
335///   Defaults differ between component pkgs (`ament_cargo`) and
336///   bringup pkgs (`ament_cmake`).
337#[derive(Clone, Debug, Default, PartialEq, Eq, Serialize, Deserialize)]
338#[serde(deny_unknown_fields)]
339pub struct PackageMetadataAment {
340    /// `<description>` body. Absent → emitter synthesises a generic
341    /// "`nano-ros component package <name>`" line.
342    #[serde(default, skip_serializing_if = "Option::is_none")]
343    pub description: Option<String>,
344    /// `<license>` body (SPDX identifier). Absent → `"Apache-2.0"`.
345    #[serde(default, skip_serializing_if = "Option::is_none")]
346    pub license: Option<String>,
347    /// `<maintainer email="…">name</maintainer>` row. Absent →
348    /// emitter falls back to a placeholder `Developer <dev@example.com>`.
349    #[serde(default, skip_serializing_if = "Option::is_none")]
350    pub maintainer: Option<AmentMaintainer>,
351    /// `<build_depend>` rows.
352    #[serde(default, skip_serializing_if = "Vec::is_empty")]
353    pub build_depend: Vec<String>,
354    /// `<buildtool_depend>` rows (e.g. `"ament_cargo"`, `"ament_cmake"`).
355    /// Phase 212.B.4: explicit dependency category so users opting in
356    /// to colcon interop can author `buildtool_depend = ["ament_cmake"]`
357    /// without polluting `<build_depend>`.
358    #[serde(default, skip_serializing_if = "Vec::is_empty")]
359    pub buildtool_depend: Vec<String>,
360    /// `<exec_depend>` rows.
361    #[serde(default, skip_serializing_if = "Vec::is_empty")]
362    pub exec_depend: Vec<String>,
363    /// `<test_depend>` rows.
364    #[serde(default, skip_serializing_if = "Vec::is_empty")]
365    pub test_depend: Vec<String>,
366    /// `<export><build_type>…</build_type></export>` body. Component
367    /// pkgs default to `"ament_cargo"`; bringup pkgs default to
368    /// `"ament_cmake"`.
369    #[serde(default, skip_serializing_if = "Option::is_none")]
370    pub build_type: Option<String>,
371}

393/// Sections (see `docs/design/workspace-layout-by-case.md` Case 3/4 and

452/// `[deploy.<target>]` block.
453///
454/// Per the F.4 §12 known-gap #2 resolution (path a — relax parser): both
455/// `kind` and `target` are OPTIONAL. The deploy block is configuration-
456/// by-target — the `<target>` map key (e.g. `native`, `qemu-mps2-an385`,
457/// `threadx-linux`, `platformio`) already names the runner, and the
458/// runner stage derives sensible defaults for `kind` / `target` from
459/// the target name when these fields are absent. Strict
460/// `deny_unknown_fields` is preserved — widening the schema, not
461/// loosening the policy.
462///
463/// `nros check` is the place to surface a heads-up when `kind`/`target`
464/// is absent AND the runner can't synthesise defaults from the target
465/// name; that's a lint, not a parser error.
466#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
467#[serde(deny_unknown_fields)]
468pub struct DeployTarget {
469    /// `"self"`, `"qemu"`, `"flash"`, … — interpreted by the runner stage.
470    /// Optional (F.4 §12 gap #2): absent ⇒ runner derives from the
471    /// `<target>` map key.
472    #[serde(default, skip_serializing_if = "Option::is_none")]
473    pub kind: Option<String>,
474    /// Target triple / board id / runner key. The exact semantics depend
475    /// on `kind`. Optional (F.4 §12 gap #2): absent ⇒ runner derives
476    /// from the `<target>` map key + the chosen `kind`.
477    #[serde(default, skip_serializing_if = "Option::is_none")]
478    pub target: Option<String>,
479    /// Optional path (relative to the bringup pkg root) to a
480    /// `launch/*.launch.xml` used for this deploy.

482    pub launch: Option<String>,

485    pub board: Option<String>,

492    #[serde(default, skip_serializing_if = "Option::is_none")]
