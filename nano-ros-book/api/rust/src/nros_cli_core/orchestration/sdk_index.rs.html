1//! Phase 187.1 — the SDK package index that `nros setup` reads.
2//!
3//! `nros-sdk-index.toml` is the versioned manifest of host toolchains/tools.
4//! Each `[tool.*]` carries a per-host prebuilt `dist` (GitHub Release asset URL,
5//! sha256) **and** a `[tool.*.source]` recipe used when no `dist` matches the
6//! host — both install into the same `$NROS_HOME/sdk/<tool>/<version>/` prefix.
7//! `[source.*]` packages build with the app (target-compiled, never prebuilt);
8//! `[gated.*]` are license-gated (never fetched/built — instruct + env check).
9//!
10//! This module is the format + loader (the rest of `nros setup` — board
11//! resolution, fetch/cache, the CI release gate — is Phase 187.2–187.5). See
12//! `docs/design/nros-setup-toolchain-management.md`.
13

19/// The whole `nros-sdk-index.toml`.
20#[derive(Clone, Debug, Default, PartialEq, Eq, Serialize, Deserialize)]
21#[serde(deny_unknown_fields)]
22pub struct SdkIndex {
23    /// Prebuilt host tools (qemu, cross-gcc, zenohd, …), keyed by tool name.
24    #[serde(default)]
25    pub tool: BTreeMap<String, ToolPackage>,
26    /// Source packages built with the app (kernels, small C libs), by name.
27    #[serde(default)]
28    pub source: BTreeMap<String, SourcePackage>,
29    /// License-gated packages (never hosted/built), by name.
30    #[serde(default)]
31    pub gated: BTreeMap<String, GatedPackage>,
32    /// RMW → host package set (Phase 191.6.a). The RMW axis is orthogonal to the
33    /// board/platform axis: a board lists only its platform/toolchain packages,
34    /// the chosen RMW contributes its host daemon/tool (`zenohd` / `xrce-agent`
35    /// / `cyclonedds`). `nros setup <board> --rmw <name>` resolves
36    /// `board.packages ∪ rmw.packages` — no `board×rmw` pair enumeration.
37    #[serde(default)]
38    pub rmw: BTreeMap<String, RmwEntry>,
39    /// Board → required package set (Phase 191.1). The board→toolchain SSOT that
40    /// ships with the index — replaces board-name keyword guessing in
41    /// `resolve_packages`. Keyed by the canonical board id the user passes to
42    /// `nros setup <board>`.
43    #[serde(default)]
44    pub board: BTreeMap<String, BoardEntry>,
45    /// Named source groupings not tied to a single board/rmw (Phase 197.2) —
46    /// e.g. `[reference.px4]`. Consumed by `tools/setup.sh --with-reference`,
47    /// NOT by `nros setup`.
48    #[serde(default)]
49    pub reference: BTreeMap<String, ReferenceEntry>,
50}

56    /// `[source.*]` names this reference set pulls.

61/// An RMW's host package set — the orthogonal RMW axis (Phase 191.6.a).
62#[derive(Clone, Debug, Default, PartialEq, Eq, Serialize, Deserialize)]
63#[serde(deny_unknown_fields)]
64pub struct RmwEntry {
65    /// The index package names (`[tool]`/`[source]`/`[gated]`) this RMW's host
66    /// side needs — e.g. `["zenohd"]`, `["xrce-agent"]`, `["cyclonedds"]`.
67    #[serde(default)]
68    pub packages: Vec<String>,
69    /// `[source.*]` names built with the app for this RMW (Phase 197.2). Consumed
70    /// by `tools/setup.sh` (the local dev provisioner), NOT by `nros setup` —
71    /// recorded here so the index is the single source manifest.
72    #[serde(default)]
73    pub build_sources: Vec<String>,
74    /// Opt-in dev `[source.*]` (full upstream repos, for hacking on the RMW).
75    #[serde(default)]
76    pub dev_sources: Vec<String>,
77}

79/// A prebuilt host tool: a per-host `dist` map + an optional `source` fallback.
80#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
81#[serde(deny_unknown_fields)]
82pub struct ToolPackage {
83    pub version: String,
84    /// The exact upstream revision the prebuilt is built/repackaged from (Phase
85    /// 191.2) — e.g. ARM `13.2.rel1`, xPack `14.2.0-3`, a fork branch. The SSOT
86    /// the build scripts consume (as the `build-tool.yml` `upstream` input)
87    /// instead of hardcoding/hand-deriving it. For tools with a `source` recipe
88    /// this equals `source.ref`; recorded here too for dist-only tools.
89    #[serde(default)]
90    pub upstream: Option<String>,
91    /// host key (`<os>-<arch>`, e.g. `linux-x86_64`) → prebuilt artifact.
92    #[serde(default)]
93    pub dist: BTreeMap<String, DistArtifact>,
94    /// Build-from-source recipe used when no `dist` matches the host.
95    #[serde(default)]
96    pub source: Option<ToolSource>,
97}

99/// A board's required SDK package set — the board→toolchain SSOT (Phase 191.1).
100#[derive(Clone, Debug, Default, PartialEq, Eq, Serialize, Deserialize)]
101#[serde(deny_unknown_fields)]
102pub struct BoardEntry {
103    /// Target arch family (descriptive: `cortex-m3`, `riscv32`, `x86_64`, …).
104    #[serde(default)]
105    pub arch: Option<String>,
106    /// Platform / RTOS (descriptive: `bare-metal`, `freertos`, `posix`, …).
107    #[serde(default)]
108    pub platform: Option<String>,
109    /// The index package names (`[tool]`/`[source]`/`[gated]`) this board needs.
110    /// Explicit — no derivation, no board-name guessing. May be empty (e.g. an
111    /// ESP32-C3 board whose riscv32 toolchain is rustup-managed).
112    #[serde(default)]
113    pub packages: Vec<String>,
114    /// `[source.*]` names built with the app for this board (Phase 197.2).
115    /// Consumed by `tools/setup.sh`, NOT by `nros setup <board>` (they're
116    /// target-compiled with the app, not host tools) — recorded here so the
117    /// index is the single source manifest.
118    #[serde(default)]
119    pub build_sources: Vec<String>,
120    /// Opt-in dev `[source.*]` (full upstream repos, for in-tree development).
121    #[serde(default)]
122    pub dev_sources: Vec<String>,
123}

133/// The source-build fallback recipe — installs into the same prefix as `dist`.
134#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
135#[serde(deny_unknown_fields)]
136pub struct ToolSource {
137    pub git: String,

139    #[serde(rename = "ref")]

143    pub configure: Option<String>,
144    /// Build + install step.
145    #[serde(default)]
146    pub install: Option<String>,

149/// A package compiled with the user's app for their chosen target.
150///
151/// Phase 195.B — `[source.*]` provisioning is data-driven: `nros setup`
152/// fetches the source into [`dest`](Self::dest) from index data, never a
153/// hardcoded `third-party/` path. `git`/`ref` record the canonical pin (the
154/// SSOT — so `.gitmodules` and the index can't drift); `submodule` is an
155/// optional *mode hint*:
156/// - **clone mode** (`git` + `ref` + `dest`, no `submodule`): fresh
157///   `git clone`@`ref`.
158/// - **submodule mode** (`submodule` + `dest`, `git`/`ref` document the pin):
159///   `git submodule update --init <submodule>` — used when the canonical
160///   source is a committed submodule (the contributor checkout keeps it).
161///
162/// A source with no fetch fields at all has no provisioning step (e.g. a
163/// host-built package whose tree already lives in the workspace).
164#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
165#[serde(deny_unknown_fields)]
166pub struct SourcePackage {
167    pub version: String,
168    /// Git URL to clone (clone mode). Mutually exclusive with `submodule`.
169    #[serde(default)]
170    pub git: Option<String>,
171    /// Git ref (tag/branch/sha) to check out — pinned in lockstep with
172    /// `version`. Required in clone mode.
173    #[serde(default, rename = "ref")]
174    pub git_ref: Option<String>,
175    /// Workspace-relative destination the source is provisioned into. The
176    /// index is the SSOT — never a path baked into the `nros` binary.
177    #[serde(default)]
178    pub dest: Option<String>,
179    /// `.gitmodules` path when the canonical source is a committed submodule;
180    /// `nros setup` runs `git submodule update --init <path>` instead of a
181    /// fresh clone. `git`/`ref` still record the pin (SSOT) in this mode.
182    #[serde(default)]
183    pub submodule: Option<String>,
184    /// Shallow-fetch the submodule (`--depth 1`). Default `true` — pins lag the
185    /// upstream branch tip, and `git submodule update --depth 1` fetches the
186    /// pinned SHA directly (fetch-by-SHA), so this is a true depth-1 checkout,
187    /// not a deepen-to-reach-pin. Set `shallow = false` for a source whose
188    /// upstream rejects reachable-SHA shallow fetches. Submodule mode only.
189    #[serde(default = "default_true")]
190    pub shallow: bool,
191    /// Recurse into the source's own nested submodules (`--recursive`). Default
192    /// `true`. Only affects a source that *has* nested submodules; it never
193    /// pulls sibling top-level sources (e.g. PX4-Autopilot is a separate
194    /// `[source.*]`, not nested in `px4-rs`). Set `recursive = false` to pin a
195    /// source to its top tree only. Submodule mode only.
196    #[serde(default = "default_true")]
197    pub recursive: bool,
198}
199
200fn default_true() -> bool {
201    true
202}
203
204// Hand-rolled so `SourcePackage::default()` matches the serde defaults
205// (`shallow`/`recursive` default to `true`); a `#[derive(Default)]` would make
206// the bools `false` and silently diverge from a TOML-parsed entry.
207impl Default for SourcePackage {
208    fn default() -> Self {
209        Self {
210            version: String::new(),
211            git: None,
212            git_ref: None,
213            dest: None,
214            submodule: None,
215            shallow: true,
216            recursive: true,
217        }
218    }
219}

221/// How a [`SourcePackage`] is provisioned (Phase 195.B).
222#[derive(Clone, Copy, Debug, PartialEq, Eq)]
223pub enum SourceProvision {
224    /// `git clone <git> @ <ref>` into `dest`.
225    Clone,
226    /// `git submodule update --init <submodule>` (dest is the submodule path).
227    Submodule,
228    /// No fetch step — the tree already lives in the workspace.
229    None,
230}
231
232impl SourcePackage {
233    /// Which provisioning mode this entry declares (Phase 195.B).
234    pub fn provision(&self) -> SourceProvision {
235        if self.submodule.is_some() {
236            SourceProvision::Submodule
237        } else if self.git.is_some() {
238            SourceProvision::Clone
239        } else {
240            SourceProvision::None
241        }
242    }
243}

256impl SdkIndex {
257    /// Read, parse, + validate an `nros-sdk-index.toml`.
258    pub fn load(path: &Path) -> Result<Self> {
259        let raw = std::fs::read_to_string(path)
260            .wrap_err_with(|| format!("failed to read SDK index {}", path.display()))?;
261        let idx =
262            Self::parse(&raw).wrap_err_with(|| format!("invalid SDK index {}", path.display()))?;
263        idx.validate()
264            .wrap_err_with(|| format!("invalid SDK index {}", path.display()))?;
265        Ok(idx)
266    }
267
268    /// Parse from a string (schema only — no cross-reference validation, so unit
269    /// tests can parse partial fixtures). [`load`] additionally [`validate`]s.
270    pub fn parse(raw: &str) -> Result<Self> {
271        toml::from_str(raw).wrap_err("invalid nros-sdk-index.toml schema")
272    }
273
274    /// Phase 191.4 — every `[board.*].packages` name must be a defined
275    /// `[tool]`/`[source]`/`[gated]` package. Phase 191.6.a extends this to
276    /// `[rmw.*].packages`. Catches typos/renames that would otherwise silently
277    /// skip (a board's/RMW's tool would just not install).
278    pub fn validate(&self) -> Result<()> {
279        let known = |pkg: &str| {
280            self.tool.contains_key(pkg)
281                || self.source.contains_key(pkg)
282                || self.gated.contains_key(pkg)
283        };
284        for (board, entry) in &self.board {
285            for pkg in &entry.packages {
286                if !known(pkg) {
287                    bail!(
288                        "board '{board}' references undefined package '{pkg}' \
289                         (not a [tool]/[source]/[gated] entry)"
290                    );
291                }
292            }
293        }
294        for (rmw, entry) in &self.rmw {
295            for pkg in &entry.packages {
296                if !known(pkg) {
297                    bail!(
298                        "rmw '{rmw}' references undefined package '{pkg}' \
299                         (not a [tool]/[source]/[gated] entry)"
300                    );
301                }
302            }
303        }
304        // Phase 195.B — a `[source.*]` provisioning recipe must be coherent so
305        // `nros setup` can act on it without guessing. `submodule` mode needs a
306        // `dest`; clone mode (a `git` with no `submodule`) needs both `ref` and
307        // `dest`. `git`/`ref` may accompany `submodule` (they record the pin).
308        for (name, src) in &self.source {
309            match src.provision() {
310                SourceProvision::Clone => {
311                    if src.git_ref.is_none() {
312                        bail!("source '{name}' has `git` but no `ref` (clone needs a pinned ref)");
313                    }
314                    if src.dest.is_none() {
315                        bail!("source '{name}' has `git` but no `dest` (where to provision it)");
316                    }
317                }
318                SourceProvision::Submodule => {
319                    if src.dest.is_none() {
320                        bail!("source '{name}' has `submodule` but no `dest`");
321                    }
322                }
323                SourceProvision::None => {}
324            }
325        }
326        Ok(())
327    }
328}

330impl ToolPackage {
331    /// The prebuilt artifact for `host` (e.g. `linux-x86_64`), if one exists.
332    pub fn dist_for(&self, host: &str) -> Option<&DistArtifact> {
333        self.dist.get(host)
334    }
335
336    /// Whether this tool can be installed on `host` — a matching prebuilt, or a
337    /// source recipe to fall back to. (`false` ⇒ no prebuilt + no source.)
338    pub fn installable_on(&self, host: &str) -> bool {
339        self.dist.contains_key(host) || self.source.is_some()
340    }
341}
342
343/// The current host key (`<os>-<arch>`), matching `dist` map keys.
344pub fn host_key() -> String {
345    let arch = match std::env::consts::ARCH {
346        "aarch64" => "arm64",
347        other => other, // x86_64, riscv64, …
348    };
349    format!("{}-{arch}", std::env::consts::OS) // linux / macos / windows
350}

381    fn parses_tool_source_and_gated_sections() {
382        let idx = SdkIndex::parse(SAMPLE).expect("sample parses");
383        assert_eq!(idx.tool.len(), 2);
384        assert_eq!(idx.source.len(), 1);
385        assert_eq!(idx.gated.len(), 1);
386
387        let qemu = &idx.tool["qemu"];
388        assert_eq!(qemu.version, "11.0-nros1");
389        assert_eq!(qemu.dist_for("linux-x86_64").unwrap().sha256, "aa");
390        assert!(qemu.dist_for("windows-x86_64").is_none());
391        let src = qemu.source.as_ref().expect("qemu has a source recipe");
392        assert_eq!(src.git_ref, "v11.0-nros1"); // the `ref` key
393        assert!(src.configure.as_deref().unwrap().contains("{prefix}"));
