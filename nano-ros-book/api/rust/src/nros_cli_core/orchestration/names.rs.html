1//! ROS name normalization helpers used by the host planner.
2
3use serde::Serialize;
4
5#[derive(Debug, Clone, Serialize)]
6pub struct ResolvedName {
7    pub source: String,
8    pub resolved: String,
9    pub remapped_from: Option<String>,
10}
11
12pub fn normalize_namespace(namespace: Option<&str>) -> String {
13    let raw = namespace.unwrap_or("/");
14    if raw.is_empty() || raw == "/" {
15        return "/".to_string();
16    }
17    normalize_absolute(raw)
18}
19
20pub fn node_fqn(namespace: Option<&str>, name: Option<&str>, fallback: &str) -> String {
21    let ns = normalize_namespace(namespace);
22    let node_name = name.unwrap_or(fallback).trim_matches('/');
23    if node_name.is_empty() {
24        ns
25    } else if ns == "/" {
26        format!("/{node_name}")
27    } else {
28        format!("{ns}/{node_name}")
29    }
30}
31
32pub fn resolve_entity_name(
33    namespace: &str,
34    node_name: &str,
35    source_name: &str,
36    remaps: &[(String, String)],
37) -> ResolvedName {
38    let remapped = remaps
39        .iter()
40        .find(|(from, _)| from == source_name)
41        .map(|(_, to)| to.as_str());
42    let effective = remapped.unwrap_or(source_name);
43    ResolvedName {
44        source: source_name.to_string(),
45        resolved: resolve_without_remap(namespace, node_name, effective),
46        remapped_from: remapped.map(|_| source_name.to_string()),
47    }
48}
49
50pub fn resolve_without_remap(namespace: &str, node_name: &str, name: &str) -> String {
51    let namespace = normalize_namespace(Some(namespace));
52    if let Some(rest) = name.strip_prefix("~/") {
53        let node = node_name.trim_matches('/');
54        return normalize_absolute(&join_ros(&join_ros(&namespace, node), rest));
55    }
56    if name == "~" {
57        let node = node_name.trim_matches('/');
58        return normalize_absolute(&join_ros(&namespace, node));
59    }
60    if name.starts_with('/') {
61        normalize_absolute(name)
62    } else {
63        normalize_absolute(&join_ros(&namespace, name))
64    }
65}
66
67fn join_ros(prefix: &str, suffix: &str) -> String {
68    let prefix = prefix.trim_end_matches('/');
69    let suffix = suffix.trim_start_matches('/');
70    match (prefix.is_empty(), suffix.is_empty()) {
71        (true, true) => "/".to_string(),
72        (true, false) => format!("/{suffix}"),
73        (false, true) => prefix.to_string(),
74        (false, false) => format!("{prefix}/{suffix}"),
75    }
76}
77
78fn normalize_absolute(name: &str) -> String {
79    let mut out = String::from("/");
80    out.push_str(
81        &name
82            .split('/')
83            .filter(|part| !part.is_empty())
84            .collect::<Vec<_>>()
85            .join("/"),
86    );
87    if out.len() > 1 && out.ends_with('/') {
88        out.pop();
89    }
90    out
91}

93#[cfg(test)]
94mod tests {
95    use super::*;
96
97    #[test]
98    fn resolves_private_names_under_node() {
99        assert_eq!(
100            resolve_without_remap("/robot", "driver", "~/cmd"),
101            "/robot/driver/cmd"
102        );
103    }
104
105    #[test]
106    fn resolves_relative_names_under_namespace() {
107        assert_eq!(
108            resolve_without_remap("robot", "driver", "scan"),
109            "/robot/scan"
110        );
111    }
112
113    #[test]
114    fn applies_exact_remap_before_resolution() {
115        let remaps = vec![("~/cmd".to_string(), "/mux/cmd".to_string())];
116        let resolved = resolve_entity_name("/robot", "driver", "~/cmd", &remaps);
117        assert_eq!(resolved.resolved, "/mux/cmd");
118        assert_eq!(resolved.remapped_from.as_deref(), Some("~/cmd"));
119    }
