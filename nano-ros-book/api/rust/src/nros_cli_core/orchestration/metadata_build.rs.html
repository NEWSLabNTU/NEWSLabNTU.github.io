1//! Phase 172.E (driver) — metadata-mode build + run.
2//!
3//! Produces a component's `source-metadata.json` by compiling a tiny **host**
4//! harness that links the component crate, runs its `Component::register`
5//! against the in-memory `MetadataRecorder` (no transport, no RTOS task), and
6//! serializes the recorder via `to_source_metadata_json`. This is the "compile
7//! each component in a host-side metadata mode and invoke its entry path with a
8//! fake `ComponentContext`" step from the workflow design — the input `nros
9//! metadata` collects + the planner consumes.
10//!
11//! Scope (chosen 2026-05-28): the **driver** only. Hardening this execution
12//! (resource limits, fs/network sandbox for untrusted component crates) is the
13//! deferred 172.E sandbox; it wraps the `cargo` invocation here when it lands.
14
15use std::{

20use eyre::{Result, WrapErr, bail, eyre};

22#[derive(Debug, Clone)]
23pub struct MetadataBuildOptions {
24    /// Component id `crate::module` (e.g. `demo_pkg::talker`).
25    pub component_id: String,
26    /// ROS package name (the `package` field of the emitted metadata).
27    pub package: String,
28    /// Component name (`Component::NAME`).
29    pub component: String,
30    pub executable: Option<String>,
31    pub exported_symbol: Option<String>,
32    /// The component crate directory (a Cargo path dependency of the harness).
33    pub component_dir: PathBuf,
34    /// nano-ros workspace root (for the `nros` path dependency).
35    pub nano_ros_workspace: PathBuf,
36    /// Where the harness writes the source-metadata JSON.
37    pub output_path: PathBuf,
38    /// Scratch directory for the generated harness crate.
39    pub harness_dir: PathBuf,
40}

42/// `crate::module` → `crate::module::Component` (the registered type). Mirrors
43/// the generator's `rust_component_type_path`.
44fn component_type_path(component_id: &str) -> Option<String> {
45    let mut parts = component_id.split("::").filter(|p| !p.is_empty());
46    let krate = parts.next()?;
47    let module = parts.next()?;
48    Some(format!("{krate}::{module}::Component"))
49}
50
51fn crate_name(component_id: &str) -> Option<&str> {
52    component_id.split("::").next().filter(|s| !s.is_empty())
53}
54
55pub fn render_harness_cargo_toml(o: &MetadataBuildOptions) -> Result<String> {
56    let krate = crate_name(&o.component_id)
57        .ok_or_else(|| eyre!("component id '{}' has no crate segment", o.component_id))?;
58    Ok(format!(
59        "[package]\n\
60         name = \"nros-metadata-probe\"\n\
61         version = \"0.0.0\"\n\
62         edition = \"2024\"\n\
63         publish = false\n\n\
64         [[bin]]\n\
65         name = \"probe\"\n\
66         path = \"src/main.rs\"\n\n\
67         [dependencies]\n\
68         nros = {{ path = {nros:?}, features = [\"std\"] }}\n\
69         {krate} = {{ path = {comp:?} }}\n",
70        nros = o
71            .nano_ros_workspace
72            .join("packages/core/nros")
73            .display()
74            .to_string(),
75        comp = o.component_dir.display().to_string(),
76    ))
77}

79pub fn render_harness_main(o: &MetadataBuildOptions) -> Result<String> {
80    let type_path = component_type_path(&o.component_id)
81        .ok_or_else(|| eyre!("component id '{}' is not `crate::module`", o.component_id))?;
82    let exe = o
83        .executable
84        .as_deref()
85        .map(|e| format!("\n        .executable({e:?})"))
86        .unwrap_or_default();
87    let sym = o
88        .exported_symbol
89        .as_deref()
90        .map(|s| format!("\n        .exported_symbol({s:?})"))
91        .unwrap_or_default();
92    Ok(format!(
93        "// Generated metadata-mode harness (Phase 172.E). Records {type_path}'s\n\
94         // declarations against an in-memory recorder; opens no transport.\n\
95         fn main() {{\n\
96         \x20   // Bare type ⇒ default capacity const-params.\n\
97         \x20   let mut recorder: nros::MetadataRecorder = nros::MetadataRecorder::default();\n\
98         \x20   nros::record_component_metadata::<{type_path}>(&mut recorder)\n\
99         \x20       .expect(\"component register (metadata mode)\");\n\
100         \x20   let export = nros::SourceMetadataExport::new({pkg:?}, {comp:?}){exe}{sym};\n\
101         \x20   let json = recorder\n\
102         \x20       .to_source_metadata_json(&export)\n\
103         \x20       .expect(\"serialize source metadata\");\n\
104         \x20   std::fs::write({out:?}, json).expect(\"write source metadata\");\n\
105         }}\n",
106        pkg = o.package,
107        comp = o.component,
108        out = o.output_path.display().to_string(),
109    ))
110}

112/// Generate the harness crate, then `cargo run` it so it writes the
113/// source-metadata JSON to `output_path`.
114pub fn build_metadata(o: &MetadataBuildOptions) -> Result<()> {
115    let src = o.harness_dir.join("src");
116    std::fs::create_dir_all(&src).wrap_err_with(|| format!("create {}", src.display()))?;
117    write_if_changed(
118        &o.harness_dir.join("Cargo.toml"),
119        &render_harness_cargo_toml(o)?,
120    )?;
121    write_if_changed(&src.join("main.rs"), &render_harness_main(o)?)?;
122    if let Some(parent) = o.output_path.parent() {
123        std::fs::create_dir_all(parent).wrap_err_with(|| format!("create {}", parent.display()))?;
124    }
125
126    let manifest = o.harness_dir.join("Cargo.toml");
127    let target_dir = o.harness_dir.join("target");
128    let status = Command::new("cargo")
129        .arg("run")
130        .arg("--quiet")
131        .arg("--manifest-path")
132        .arg(&manifest)
133        .arg("--target-dir")
134        .arg(&target_dir)
135        // The harness inherits no pinned toolchain so a generated
136        // `rust-toolchain.toml` elsewhere can't force a re-resolve.
137        .env_remove("RUSTUP_TOOLCHAIN")
138        .stdout(Stdio::inherit())
139        .stderr(Stdio::inherit())
140        .status()
141        .wrap_err_with(|| format!("run metadata-mode harness for '{}'", o.component_id))?;
142    if !status.success() {
143        bail!(
144            "metadata-mode harness failed (exit {}) for component '{}'",
145            status.code().unwrap_or(-1),
146            o.component_id
147        );
148    }
149    if !o.output_path.is_file() {
150        bail!(
151            "metadata-mode harness produced no source metadata at {}",
152            o.output_path.display()
153        );
154    }
155    Ok(())
156}
157
158fn write_if_changed(path: &Path, contents: &str) -> Result<()> {
159    if std::fs::read_to_string(path).ok().as_deref() == Some(contents) {
160        return Ok(());
161    }
162    std::fs::write(path, contents).wrap_err_with(|| format!("write {}", path.display()))
163}

165#[cfg(test)]
166mod tests {
167    use super::*;
168
169    fn opts() -> MetadataBuildOptions {
170        MetadataBuildOptions {
171            component_id: "demo_pkg::talker".into(),
172            package: "demo_pkg".into(),
173            component: "talker".into(),
174            executable: Some("talker".into()),
175            exported_symbol: Some("nros_component_talker".into()),
176            component_dir: PathBuf::from("/ws/src/demo_pkg"),
177            nano_ros_workspace: PathBuf::from("/nano-ros"),
178            output_path: PathBuf::from("/out/talker.metadata.json"),
179            harness_dir: PathBuf::from("/scratch/probe"),
180        }
181    }
182
183    #[test]
184    fn type_path_and_crate_name() {
185        assert_eq!(
186            component_type_path("demo_pkg::talker").as_deref(),
187            Some("demo_pkg::talker::Component")
188        );
189        assert_eq!(crate_name("demo_pkg::talker"), Some("demo_pkg"));
190        assert_eq!(component_type_path("nocrate"), None); // needs crate::module
191    }
192
193    #[test]
194    fn harness_main_calls_record_and_serialize() {
195        let main = render_harness_main(&opts()).unwrap();
196        assert!(main.contains("record_component_metadata::<demo_pkg::talker::Component>"));
197        assert!(main.contains("SourceMetadataExport::new(\"demo_pkg\", \"talker\")"));
198        assert!(main.contains(".executable(\"talker\")"));
199        assert!(main.contains(".exported_symbol(\"nros_component_talker\")"));
200        assert!(main.contains("to_source_metadata_json"));
201        assert!(main.contains("/out/talker.metadata.json"));
202    }
203
204    #[test]
205    fn harness_cargo_toml_path_deps_nros_std_and_component() {
206        let toml = render_harness_cargo_toml(&opts()).unwrap();
207        assert!(
208            toml.contains(
209                "nros = { path = \"/nano-ros/packages/core/nros\", features = [\"std\"] }"
210            )
211        );
212        assert!(toml.contains("demo_pkg = { path = \"/ws/src/demo_pkg\" }"));
213    }
214
215    #[test]
216    fn harness_main_omits_optional_export_fields_when_absent() {
217        let mut o = opts();
218        o.executable = None;
219        o.exported_symbol = None;
220        let main = render_harness_main(&o).unwrap();
221        assert!(main.contains("SourceMetadataExport::new(\"demo_pkg\", \"talker\");"));
222        assert!(!main.contains(".executable("));
223        assert!(!main.contains(".exported_symbol("));
224    }
225}
