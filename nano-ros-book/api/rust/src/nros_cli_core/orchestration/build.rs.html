31/// Schema tag baked into the staleness fingerprint so a future change to
32/// what the stamp covers invalidates every old stamp.
33const STAMP_SCHEMA: &str = "nano-ros/build-stamp/v1";
34/// Stamp file under the generated package root recording the last
35/// successful generation's input fingerprint.
36const STAMP_FILE: &str = ".nros-build-stamp";

38pub fn build_generated_package(options: &BuildOptions) -> Result<GeneratedPackage> {
39    // Read the plan once as bytes so the fingerprint sees exactly what the
40    // generator consumes (parse separately for the interface cache).
41    let plan_bytes = fs::read(&options.plan_path)
42        .wrap_err_with(|| format!("failed to read {}", options.plan_path.display()))?;
43    let plan: NrosPlan = serde_json::from_slice(&plan_bytes)
44        .wrap_err_with(|| format!("failed to parse {}", options.plan_path.display()))?;
45
46    // Phase 172.D — skip regeneration when the generation inputs are
47    // unchanged. The generated package's contents are a pure function of
48    // the plan, the baked-in paths, and the generator version; component
49    // *source* is deliberately NOT in the fingerprint — generation never
50    // reads it, and cargo's own incremental fingerprinting (below) owns
51    // recompilation staleness for component edits. So the stamp gates
52    // generation only; cargo always runs and is itself the recompile gate.
53    let fingerprint = generation_fingerprint(&plan_bytes, options);
54    let generated = if generation_is_fresh(&options.output_dir, &fingerprint, options.force) {
55        eprintln!(
56            "nros build: generated package up to date (plan + generator unchanged) \
57             — skipping regeneration"
58        );
59        GeneratedPackage {
60            root: options.output_dir.clone(),
61            manifest_path: options.output_dir.join("Cargo.toml"),
62            plan_path: options.plan_path.clone(),
63        }
64    } else {
65        write_interface_cache(&options.output_dir, &plan)?;
66        let generated = generate_package(&GenerateOptions {
67            package_name: options.package_name.clone(),
68            output_dir: options.output_dir.clone(),
69            plan_path: options.plan_path.clone(),
70            nros_path: options.workspace_root.join("packages/core/nros"),
71            nros_orchestration_path: options
72                .workspace_root
73                .join("packages/core/nros-orchestration"),
74            component_workspace: options.component_workspace.clone(),
75        })?;
76        // Record the fingerprint only after a clean generation. Written
77        // before cargo on purpose: the stamp tracks generation freshness,
78        // not build success — a failed cargo run still leaves a valid
79        // generated tree, and the next run re-runs cargo (which retries the
80        // compile) without needlessly regenerating.
81        write_stamp(&options.output_dir, &fingerprint)?;
82        generated
83    };
84
85    let mut cmd = Command::new("cargo");
86    cmd.args(generated_cargo_args(
87        &generated.manifest_path,
88        &generated_target_dir(&generated.root),
89        &plan.build,
90        options.release,
91        options.target.as_deref(),
92        &options.cargo_args,
93    ))
94    .stdout(Stdio::inherit())
95    .stderr(Stdio::inherit())
96    .current_dir(&generated.root);
97    // Phase 126.M5.nuttx — strip any inherited rustup pin so the
98    // generated package's `rust-toolchain.toml` (e.g. NuttX's
99    // pinned nightly + rust-src) is actually honoured. Without
100    // this, an outer `cargo test` invocation propagates its own
101    // RUSTUP_TOOLCHAIN through, and rustup never re-resolves
102    // against the generated package's overrides.
103    cmd.env_remove("RUSTUP_TOOLCHAIN");
104
105    // Phase 204.15 inc 3 — `[build.cc]` fans out to the C/C++ layer via env that
106    // `cc-rs` *appends* to its computed flags (every zenoh-pico/XRCE/net.c/lwIP
107    // `cc::Build`), no build.rs edit. `debug=true` adds `-g` without disturbing
108    // the opt level → the C-side "debug one layer" case (Rust stays stripped via
109    // its profile). `opt_level` → `NROS_CC_OPT` for build scripts that honor it
110    // (204.9). Appended to any inherited CFLAGS/CXXFLAGS.
111    if let Some(cc) = &plan.build.cc {
112        let mut extra: Vec<String> = Vec::new();
113        if cc.debug == Some(true) {
114            extra.push("-g".to_string());
115        }
116        extra.extend(cc.cflags.iter().cloned());
117        if !extra.is_empty() {
118            let suffix = extra.join(" ");
119            for var in ["CFLAGS", "CXXFLAGS"] {
120                let mut v = std::env::var(var).unwrap_or_default();
121                if !v.is_empty() {
122                    v.push(' ');
123                }
124                v.push_str(&suffix);
125                cmd.env(var, v);
126            }
127        }
128        if let Some(opt) = &cc.opt_level {
129            cmd.env("NROS_CC_OPT", opt);
130        }
131    }
132
133    let status = cmd
134        .status()
135        .wrap_err("failed to invoke generated cargo build")?;
136    if !status.success() {
137        return Err(eyre!(
138            "generated package build failed (exit {})",
139            status.code().unwrap_or(-1)
140        ));
141    }
142
143    Ok(generated)
144}

146#[derive(Serialize)]
147struct InterfaceCacheManifest<'a> {
148    schema: &'static str,
149    system: &'a str,
150    generated_by: &'a str,
151    interfaces: &'a [super::plan::PlanInterface],
152}
153
154fn write_interface_cache(generated_dir: &Path, plan: &NrosPlan) -> Result<()> {
155    let Some(system_dir) = generated_dir.parent() else {
156        return Ok(());
157    };
158    let interfaces_dir = system_dir.join("interfaces");
159    let manifest = serde_json::to_string_pretty(&InterfaceCacheManifest {
160        schema: "nano-ros/interface-cache/v1",
161        system: &plan.system,
162        generated_by: &plan.trace.generated_by,
163        interfaces: &plan.interfaces,
164    })?;
165
166    for lang in ["rust", "c", "cpp"] {
167        let lang_dir = interfaces_dir.join(lang);
168        fs::create_dir_all(&lang_dir).wrap_err_with(|| {
169            format!(
170                "failed to create generated interface cache dir {}",
171                lang_dir.display()
172            )
173        })?;
174        write_if_changed(&lang_dir.join("manifest.json"), &manifest)?;
175    }
176    Ok(())
177}
178
179fn write_if_changed(path: &Path, contents: &str) -> Result<()> {
180    if fs::read_to_string(path).ok().as_deref() == Some(contents) {
181        return Ok(());
182    }
183    fs::write(path, contents).wrap_err_with(|| format!("failed to write {}", path.display()))
184}

186/// Fingerprint of every input that determines the *generated* package's
187/// contents: the generator version, the plan bytes, and the paths baked into
188/// the generated manifest/build script. Component source is intentionally
189/// excluded — generation never reads it, and cargo owns recompilation
190/// staleness for those edits (see `build_generated_package`).
191fn generation_fingerprint(plan_bytes: &[u8], options: &BuildOptions) -> String {
192    let mut h = std::collections::hash_map::DefaultHasher::new();
193    STAMP_SCHEMA.hash(&mut h);
194    // A generator upgrade can change the templates even when the plan is
195    // byte-identical, so the CLI version is part of the key.
196    env!("CARGO_PKG_VERSION").hash(&mut h);
197    plan_bytes.hash(&mut h);
198    options.package_name.hash(&mut h);
199    options.workspace_root.hash(&mut h);
200    options.component_workspace.hash(&mut h);
201    format!("{:016x}", h.finish())
202}

204/// A generation is fresh when not forced, the generated crate is present, and
205/// the recorded stamp matches the current input fingerprint.
206fn generation_is_fresh(output_dir: &Path, fingerprint: &str, force: bool) -> bool {
207    !force
208        && output_dir.join("Cargo.toml").is_file()
209        && read_stamp(output_dir).as_deref() == Some(fingerprint)
210}
211
212fn read_stamp(output_dir: &Path) -> Option<String> {
213    fs::read_to_string(output_dir.join(STAMP_FILE))
214        .ok()
215        .map(|s| s.trim().to_string())
216}
217
218fn write_stamp(output_dir: &Path, fingerprint: &str) -> Result<()> {
219    let path = output_dir.join(STAMP_FILE);
220    fs::write(&path, fingerprint)
221        .wrap_err_with(|| format!("failed to write build stamp {}", path.display()))
222}

224fn generated_cargo_args(
225    manifest_path: &Path,
226    target_dir: &Path,
227    build: &PlanBuildOptions,
228    release_override: bool,
229    target_override: Option<&str>,
230    passthrough: &[String],
231) -> Vec<String> {
232    let mut args = vec![
233        "build".to_string(),
234        "--manifest-path".to_string(),
235        manifest_path.display().to_string(),
236        "--target-dir".to_string(),
237        target_dir.display().to_string(),
238    ];
239
240    match target_override {
241        Some(target) if !target.is_empty() => {
242            args.push("--target".to_string());
243            args.push(target.to_string());
244        }
245        _ if !build.target.is_empty() => {
246            args.push("--target".to_string());
247            args.push(build.target.clone());
248        }
249        _ => {}
250    }
251
252    if release_override || build.profile == "release" {
253        args.push("--release".to_string());
254    } else if !matches!(build.profile.as_str(), "" | "debug" | "dev") {
255        args.push("--profile".to_string());
256        args.push(build.profile.clone());
257    }
258
259    args.extend(passthrough.iter().cloned());
260    args
261}

263fn generated_target_dir(generated_root: &Path) -> PathBuf {
264    generated_root
265        .parent()
266        .map(|parent| parent.join("target"))
267        .unwrap_or_else(|| generated_root.join("target"))
268}

400    #[test]
401    fn fingerprint_is_stable_and_input_sensitive() {
402        let o = opts(PathBuf::from("/out"));
403        let base = generation_fingerprint(b"plan-A", &o);
404
405        // Identical inputs → identical fingerprint.
406        assert_eq!(base, generation_fingerprint(b"plan-A", &o));
407        // Different plan bytes → different fingerprint.
408        assert_ne!(base, generation_fingerprint(b"plan-B", &o));
409
410        // Each baked-in path / name participates in the key.
411        let mut o2 = o.clone();
412        o2.package_name = "other".to_string();
413        assert_ne!(base, generation_fingerprint(b"plan-A", &o2));
414        let mut o3 = o.clone();
415        o3.workspace_root = PathBuf::from("/ws/other");
416        assert_ne!(base, generation_fingerprint(b"plan-A", &o3));
417        let mut o4 = o.clone();
418        o4.component_workspace = None;
419        assert_ne!(base, generation_fingerprint(b"plan-A", &o4));
420    }

422    fn temp_dir(name: &str) -> PathBuf {
423        let stamp = SystemTime::now()
424            .duration_since(UNIX_EPOCH)
425            .unwrap()
426            .as_nanos();
427        let dir = std::env::temp_dir().join(format!("{name}-{}-{stamp}", std::process::id()));
428        let _ = std::fs::remove_dir_all(&dir);
429        std::fs::create_dir_all(&dir).unwrap();
430        dir
431    }

433    #[test]
434    fn freshness_requires_crate_present_matching_stamp_and_not_forced() {
435        let dir = temp_dir("nros-staleness");
436        let fp = "deadbeefdeadbeef";
437
438        // No generated crate yet → not fresh (must generate).
439        assert!(!generation_is_fresh(&dir, fp, false));
440
441        // Crate present but no stamp → not fresh.
442        std::fs::write(dir.join("Cargo.toml"), "[package]\n").unwrap();
443        assert!(!generation_is_fresh(&dir, fp, false));
444
445        // Matching stamp → fresh (skip regeneration).
446        write_stamp(&dir, fp).unwrap();
447        assert!(generation_is_fresh(&dir, fp, false));
448
449        // `--force` defeats a matching stamp.
450        assert!(!generation_is_fresh(&dir, fp, true));
451
452        // A changed fingerprint (new plan / generator) → stale.
453        assert!(!generation_is_fresh(&dir, "feedfacefeedface", false));
454
455        let _ = std::fs::remove_dir_all(&dir);
456    }
