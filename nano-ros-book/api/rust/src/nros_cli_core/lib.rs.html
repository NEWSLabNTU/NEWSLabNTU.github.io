1//! Shared library backing the `nros` CLI.
2//!
3//! `nros-cli` owns the canonical user-facing command surface. Codegen
4//! implementation details still live in the `cargo_nano_ros` library until
5//! that library is renamed or split.
6
7pub mod cmd;
8pub mod orchestration;
9
10use eyre::Result;
11
12/// Top-level dispatcher entry point — every binary front-end lands here.
13///
14/// `argv` is the post-clap parsed command structure. Each variant maps
15/// 1:1 to a `nros <verb>` invocation.
16pub fn run(cmd: cmd::Cmd) -> Result<()> {
17    match cmd {
18        cmd::Cmd::New(args) => cmd::new::run(args),
19        cmd::Cmd::Generate(args) => cmd::generate::run(args),
20        cmd::Cmd::GenerateRust(args) => cmd::generate::run_rust(args),
21        cmd::Cmd::Codegen(args) => cmd::codegen::run(args),
22        cmd::Cmd::CodegenSystem(args) => cmd::codegen_system::run(args),
23        cmd::Cmd::Metadata(args) => cmd::metadata::run(args),
24        cmd::Cmd::Migrate(sub) => match sub {
25            cmd::MigrateSub::Workspace(args) => cmd::migrate::run(args),
26        },
27        cmd::Cmd::Plan(args) => cmd::plan::run(args),
28        cmd::Cmd::Check(args) => cmd::check::run(args),
29        cmd::Cmd::Explain(args) => cmd::explain::run(args),
30        cmd::Cmd::Config(args) => cmd::config::run(args),
31        cmd::Cmd::Build(args) => cmd::build::run(args),
32        cmd::Cmd::Deploy(args) => cmd::deploy::run(args),
33        cmd::Cmd::Launch(args) => cmd::launch::run(args),
34        cmd::Cmd::Setup(args) => cmd::setup::run(args),
35        cmd::Cmd::Run(args) => cmd::run_target::run(args),
36        cmd::Cmd::Monitor(args) => cmd::monitor::run(args),
37        cmd::Cmd::Doctor(args) => cmd::doctor::run(args),
38        cmd::Cmd::Board(args) => cmd::board::run(args),
39        cmd::Cmd::Ws(args) => cmd::ws::run(args),
40        cmd::Cmd::Version => cmd::version::run(),
41        cmd::Cmd::Completions(args) => cmd::completions::run(args),
42        #[cfg(feature = "release")]
43        cmd::Cmd::Release(args) => cmd::release::run(args),
44    }
45}
