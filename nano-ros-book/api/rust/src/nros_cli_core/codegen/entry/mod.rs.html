1//! Phase 219.A — Entry-pkg codegen shared module.
2//!
3//! Lifts the pkg-index walk + launch.xml parse + register-call
4//! resolution out of the Rust `nros::main!()` proc-macro (which lives
5//! in `packages/core/nros-macros/src/main_macro.rs`) into one place
6//! every front-end can call. The three emitters
7//! ([`emit_rust`], [`emit_cpp`], [`emit_c`]) consume a single in-memory
8//! [`Plan`] IR so the per-language differences stay surface-level.
9//!
10//! Surface (per phase doc §3.2):
11//!
12//! ```ignore
13//! use nros_cli_core::codegen::entry::{Lang, Plan, plan_from_launch};
14//!
15//! let plan = plan_from_launch(PlanInput {
16//!     workspace: ws.as_path(),
17//!     launch_spec: "demo_bringup:system.launch.xml",
18//!     board: Some("native".into()),
19//!     arg_overrides: vec![],
20//! })?;
21//! let src = match lang {
22//!     Lang::Rust => emit_rust::emit(&plan),               // register-based
23//!     Lang::Cpp  => emit_cpp::emit_typed(&plan)?,         // typed (RFC-0043)
24//!     Lang::C    => emit_c::emit_typed(&plan)?,           // typed (phase-257)
25//! };
26//! ```
27//!
28//! Errors carry enough context that the CLI verb's `eyre::Result`
29//! wrapper passes them through verbatim.
30

45/// Target language for the emitted TU.
46#[derive(Clone, Copy, Debug, PartialEq, Eq)]
47pub enum Lang {
48    Rust,
49    Cpp,
50    C,
51}
52
53impl Lang {
54    pub fn parse(s: &str) -> Result<Self> {
55        match s {
56            "rust" => Ok(Lang::Rust),
57            "cpp" | "c++" | "cxx" => Ok(Lang::Cpp),
58            "c" => Ok(Lang::C),
59            other => bail!("unknown --lang `{other}` (expected one of: rust, cpp, c)"),
60        }
61    }
62}

64/// Caller-supplied inputs to [`plan_from_launch`].
65#[derive(Debug)]
66pub struct PlanInput<'a> {
67    /// Workspace root — the directory holding `src/<pkg>/package.xml`
68    /// trees. Typically the dir containing the workspace-root
69    /// `CMakeLists.txt` or `Cargo.toml`.
70    pub workspace: &'a Path,
71    /// `"<bringup_pkg>"` or `"<bringup_pkg>:<file>.launch.xml"`.
72    pub launch_spec: &'a str,
73    /// Board key (`"native"`, `"freertos"`, …) or an explicit C++
74    /// path-like (`"nros::board::NativeBoard"`). For the Rust emitter
75    /// this is treated as the `[package.metadata.nros.entry] deploy =`
76    /// value; the emitter dispatches to the matching board ZST.
77    /// `None` falls back to `"native"` — the only Entry-pkg target
78    /// currently supported by the C/C++ surface (Phase 212.L.2).
79    pub board: Option<String>,
80    /// Caller-supplied launch-arg overrides (forwarded to the parser).
81    pub arg_overrides: Vec<(String, String)>,
82}

84/// Resolved plan handed to one of the three emitters.
85#[derive(Clone, Debug, PartialEq, Eq)]
86pub struct Plan {
87    /// Board key from [`PlanInput::board`]; `"native"` by default.
88    pub board: String,
89    /// Per-node entries in launch-file order (top-scope first, then
90    /// each `<group>`'s children). Duplicates per `(pkg, exec)` are
91    /// preserved — multiple instances of the same Node pkg may occur
92    /// in a single launch.
93    pub nodes: Vec<PlanNode>,
94    /// Absolute paths of every file the plan read. Caller emits these
95    /// as `cargo:rerun-if-changed=` / `--depfile` entries for build-
96    /// system rebuild correctness.
97    pub depfile_paths: Vec<PathBuf>,
98    /// Bringup-pkg name from the launch spec (`"demo_bringup"` for
99    /// `"demo_bringup:system.launch.xml"`). Surfaced so emitters can
100    /// thread it into the generated banner.
101    pub bringup: String,
102    /// Resolved launch-file path; surfaces in the generated header
103    /// banner.
104    pub launch_file: PathBuf,
105}

107impl Plan {
108    /// Phase 211.F — partition a multi-host launch for a single target host.
109    ///
110    /// Returns a copy of the plan keeping only the nodes that belong on host
111    /// `host`: those whose `<node machine="…">` equals `host`, plus every
112    /// **unhosted** node (`host == None`) — an unhosted node is shared / runs
113    /// everywhere (matches ROS 2, where a node without `machine=` runs on the
114    /// local host). The multi-host deploy bakes one entry per host from these
115    /// partitions, each deployed to that host's `[deploy.<id>]` target. A
116    /// single-host launch (no `machine=` anywhere) is unaffected — every node
117    /// is unhosted, so `for_host` returns all of them for any host.
118    #[must_use]
119    pub fn for_host(&self, host: &str) -> Plan {
120        Plan {
121            board: self.board.clone(),
122            nodes: self
123                .nodes
124                .iter()
125                .filter(|n| match &n.host {
126                    Some(h) => h == host,
127                    None => true,
128                })
129                .cloned()
130                .collect(),
131            depfile_paths: self.depfile_paths.clone(),
132            bringup: self.bringup.clone(),
133            launch_file: self.launch_file.clone(),
134        }
135    }
136
137    /// Phase 211.F — the distinct target hosts named across the plan's nodes
138    /// (the `machine=` set), sorted + deduped. Empty for a single-host launch.
139    /// The multi-host deploy bakes one entry per entry in this set.
140    #[must_use]
141    pub fn hosts(&self) -> Vec<String> {
142        let mut hs: Vec<String> = self.nodes.iter().filter_map(|n| n.host.clone()).collect();
143        hs.sort();
144        hs.dedup();
145        hs
146    }
147}

149/// Phase 211.H (issue #52) — one per-topic QoS override on a node, decomposed
150/// from a `qos_overrides.<topic>.<role>.<policy>` launch param. The typed C++
151/// entry emitter (`emit_cpp`) bakes these into a static `nros_cpp_qos_override_t[]`
152/// + a `node.set_qos_overrides(...)` call before `configure(node)`, so the
153/// node's entities honour the override at create time. Fields are the raw
154/// decomposed strings; the emitter maps them to the C-ABI `{role, policy, value}`
155/// scalar codes.
156#[derive(Clone, Debug, PartialEq, Eq)]
157pub struct QosOverrideSpec {
158    /// Resolved topic (e.g. `"/chatter"`).
159    pub topic: String,
160    /// `"publisher"` or `"subscription"`.
161    pub role: String,
162    /// `"reliability"` / `"durability"` / `"history"` / `"depth"`.
163    pub policy: String,
164    /// Raw launch value (e.g. `"best_effort"`, `"transient_local"`, `"5"`).
165    pub value: String,
166}
167
168/// Decompose a node's launch params into [`QosOverrideSpec`]s. Mirrors the
169/// planner's `schema_qos_overrides`: a param named
170/// `qos_overrides.<topic>.<role>.<policy>` splits via `rsplitn(3, '.')` (so a
171/// `/`-bearing topic survives), and the result is sorted `(topic, role, policy)`
172/// for byte-stable codegen. Non-matching params are ignored.
173fn qos_overrides_from_params(params: &[crate::launch_parser::ParamSpec]) -> Vec<QosOverrideSpec> {
174    const QOS_OVERRIDE_PREFIX: &str = "qos_overrides.";
175    let mut out: Vec<QosOverrideSpec> = params
176        .iter()
177        .filter_map(|p| {
178            let rest = p.name.strip_prefix(QOS_OVERRIDE_PREFIX)?;
179            // rsplitn(3, '.') → [policy, role, topic]
180            let mut parts = rest.rsplitn(3, '.');
181            let policy = parts.next()?;
182            let role = parts.next()?;
183            let topic = parts.next()?;
184            if topic.is_empty() || role.is_empty() || policy.is_empty() {
185                return None;
186            }
187            Some(QosOverrideSpec {
188                topic: topic.to_string(),
189                role: role.to_string(),
190                policy: policy.to_string(),
191                value: p.value.clone(),
192            })
193        })
194        .collect();
195    out.sort_by(|a, b| (&a.topic, &a.role, &a.policy).cmp(&(&b.topic, &b.role, &b.policy)));
196    out
197}

199/// One Node-pkg invocation in launch order.
200///
201/// `pkg` is the cargo-style pkg name (sanitised via [`sanitize_pkg`]
202/// for symbol-name use). `exec` and `name` come straight from the
203/// launch XML; today only `pkg` drives codegen (the per-pkg mangled
204/// register symbol is keyed on it), but the `exec` / `name` fields
205/// stay on the IR so future `<param>` / `<remap>` routing has them.
206#[derive(Clone, Debug, PartialEq, Eq)]
207pub struct PlanNode {
208    pub pkg: String,
209    pub exec: String,
210    pub name: Option<String>,
211    pub namespace: Option<String>,
212    /// Phase 240.2 (RFC-0043) — fully-qualified C++ component class
213    /// (`"talker_pkg::Talker"`), from `nano_ros_node_register(CLASS …)` via the
214    /// cmake metadata. Required by the **typed** entry emitter (`emit_cpp_typed`)
215    /// which constructs the class; `None` for the legacy register-symbol path.
216    pub class_name: Option<String>,
217    /// Phase 240.2 — the component class header to `#include`
218    /// (`"talker_pkg/Talker.hpp"`). Paired with `class_name`.
219    pub class_header: Option<String>,
220    /// Phase 240.4 (RFC-0043) — component implementation language from the
221    /// cmake metadata (`"c"` / `"cpp"` / `"rust"`). `None` for the launch-only
222    /// legacy path. The **typed** entry emitter branches on it: a `"c"` node is
223    /// constructed via its C-ABI factory + `configure(node_handle, self)` seam
224    /// (`NROS_C_COMPONENT`), a `"cpp"` node via its C++ class + `configure(node)`.
225    pub lang: Option<String>,
226    /// Phase 242.4 (RFC-0044) — component *shape* from the cmake metadata:
227    /// `"rclcpp"` (IS-A-node, ctor-wired — construct-with-handle) or `"configure"`
228    /// (RFC-0043 default-construct + `configure(Node&)`). `None` ⇒ `"configure"`
229    /// (back-compat). The **typed** entry emitter branches construct on it: an
230    /// `"rclcpp"` C++ node is placement-new'd with the executor handle *after*
231    /// `nros::init` (the ctor owns the node); a `"configure"` node keeps the
232    /// 240.x static-construct-then-`configure(node)` path.
233    pub shape: Option<String>,
234    /// Phase 211.F — `<node machine="…">` target host (multi-host launch).
235    /// `None` for single-host / unhosted nodes. [`Plan::for_host`] partitions a
236    /// multi-host launch into per-host bakes: an entry for host `H` keeps nodes
237    /// whose `host == Some(H)` plus all unhosted (`None`) nodes.
238    pub host: Option<String>,
239    /// Phase 211.H (issue #52) — per-topic QoS overrides decomposed from this
240    /// node's `qos_overrides.<topic>.<role>.<policy>` launch params. Empty when
241    /// none. The typed C++ entry emitter bakes them into a
242    /// `node.set_qos_overrides(...)` call before `configure(node)`.
243    pub qos_overrides: Vec<QosOverrideSpec>,
244}

247    /// Per-pkg mangled register symbol per Phase 212.M.5.a.1:
248    /// `__nros_component_<sanitised_pkg>_register`.
249    pub fn register_symbol(&self) -> String {
250        format!("__nros_component_{}_register", sanitize_pkg(&self.pkg))
251    }
252
253    /// Cmake target name for the static lib the Node pkg's
254    /// `nano_ros_node_register()` produces:
255    /// `<pkg>_<exec>_component`. The Entry pkg's auto-link
256    /// (`nano_ros_entry(... LAUNCH …)`) consumes this string.
257    pub fn cmake_link_target(&self) -> String {
258        format!("{}_{}_component", sanitize_pkg(&self.pkg), self.exec)
259    }

262/// Sanitise a pkg name into a valid identifier (`-` → `_`).
263///
264/// Mirrors the rule the Rust `nros::node!()` macro and the cmake fn
265/// `nano_ros_node_register()` already apply (see
266/// `packages/core/nros-macros/src/main_macro.rs::pkg_to_crate_ident`).
267pub fn sanitize_pkg(pkg: &str) -> String {
268    let mut out = String::with_capacity(pkg.len());
269    for c in pkg.chars() {
270        if c.is_ascii_alphanumeric() || c == '_' {
271            out.push(c);
272        } else {
273            out.push('_');
274        }
275    }
276    out
277}

279/// Walk the workspace pkg-index, locate the bringup pkg, parse the
280/// launch XML, and lower the result into a [`Plan`].
281///
282/// Errors carry enough context that the CLI verb's `eyre::Result`
283/// wrapper passes them through verbatim.
284pub fn plan_from_launch(input: PlanInput<'_>) -> Result<Plan> {
285    let pkg_index = build_pkg_index(input.workspace)
286        .with_context(|| format!("build pkg-index from `{}`", input.workspace.display()))?;
287
288    let (bringup_name, file_override) = match input.launch_spec.split_once(':') {
289        Some((b, f)) => (b.trim().to_string(), Some(f.trim().to_string())),
290        None => (input.launch_spec.trim().to_string(), None),
291    };
292    if bringup_name.is_empty() {
293        bail!(
294            "empty bringup pkg name in launch spec `{}`",
295            input.launch_spec
296        );
297    }
298
299    let bringup_dir = pkg_index
300        .resolve_pkg(&bringup_name)
301        .with_context(|| format!("resolve bringup pkg `{bringup_name}`"))?
302        .to_path_buf();
303
304    // Pull every package.xml the index walked into the depfile list.
305    let mut depfile_paths: Vec<PathBuf> = Vec::new();
306    for (_, pkg_dir) in pkg_index.pkgs() {
307        depfile_paths.push(pkg_dir.join("package.xml"));
308    }
309
310    // Resolve the launch filename. With no `:file` override, consult
311    // bringup's `system.toml [system] default_launch`; fall back to
312    // `system.launch.xml` (matches the Rust proc-macro shape).
313    let launch_filename = match file_override {
314        Some(s) => s,
315        None => {
316            let system_toml = bringup_dir.join("system.toml");
317            if system_toml.exists() {
318                depfile_paths.push(system_toml.clone());
319                read_default_launch(&system_toml)
320                    .with_context(|| format!("parse `{}`", system_toml.display()))?
321                    .unwrap_or_else(|| "system.launch.xml".to_string())
322            } else {
323                "system.launch.xml".to_string()
324            }
325        }
326    };
327    let launch_path = bringup_dir.join("launch").join(&launch_filename);
328    if !launch_path.exists() {
329        bail!("launch file not found: `{}`", launch_path.display());
330    }
331    depfile_paths.push(launch_path.clone());
332
333    let desc: LaunchDescription = parse_launch_file(&launch_path, &pkg_index, &input.arg_overrides)
334        .with_context(|| format!("parse launch file `{}`", launch_path.display()))?;
335
336    // Walk every `<node>` (top-level + inside groups) and lower to a
337    // PlanNode. Order matches the source XML (top-scope first, then
338    // each group's children).
339    let mut nodes: Vec<PlanNode> = Vec::new();
340    let push_node = |n: &NodeSpec, out: &mut Vec<PlanNode>| {
341        out.push(PlanNode {
342            pkg: n.pkg.clone(),
343            exec: n.exec.clone(),
344            name: n.name.clone(),
345            namespace: n.namespace.clone(),
346            // Phase 240.2 — the launch path doesn't carry the C++ class/header
347            // yet (threaded from cmake metadata in 240.2b); the legacy
348            // register-symbol emitters ignore these.
349            class_name: None,
350            class_header: None,
351            lang: None,
352            shape: None,
353            host: n.machine.clone(),
354            qos_overrides: qos_overrides_from_params(&n.params),
355        });
356    };
357    for n in &desc.nodes {
358        push_node(n, &mut nodes);
359    }
360    for g in &desc.groups {
361        for n in &g.nodes {
362            push_node(n, &mut nodes);
363        }
364    }
365
366    if nodes.is_empty() {
367        bail!(
368            "launch file `{}` has no `<node>` entries — nothing to register",
369            launch_path.display()
370        );
371    }
372
373    // Sort + dedup the depfile entries — pkg-index revisits + sibling
374    // `<include>`s can list a single file twice.
375    depfile_paths.sort();
376    depfile_paths.dedup();
377
378    let board = input.board.unwrap_or_else(|| "native".to_string());
379
380    Ok(Plan {
381        board,
382        nodes,
383        depfile_paths,
384        bringup: bringup_name,
385        launch_file: launch_path,
386    })
387}

389/// `[system] default_launch = "<file>"` reader, mirrors the proc-macro
390/// helper in `packages/core/nros-macros/src/main_macro.rs`.
391fn read_default_launch(system_toml: &Path) -> Result<Option<String>> {
392    let raw = std::fs::read_to_string(system_toml)
393        .with_context(|| format!("read `{}`", system_toml.display()))?;
394    let v: toml::Value = toml::from_str(&raw).context("parse toml")?;
395    Ok(v.get("system")
396        .and_then(|s| s.get("default_launch"))
397        .and_then(|d| d.as_str())
398        .map(str::to_string))
399}

401/// Write the depfile in GNU-make `target: dep1 dep2 …` form so cmake's
402/// `CMAKE_CONFIGURE_DEPENDS` / make-style consumers ingest it directly.
403///
404/// The "target" line is the generated TU path (or whatever the caller
405/// supplies); each dep is one absolute path on its own continuation
406/// line so newline-quoting + path escaping stays trivial.
407pub fn write_depfile(target: &Path, deps: &[PathBuf], depfile: &Path) -> Result<()> {
408    let mut out = String::new();
409    out.push_str(&escape_make(&target.display().to_string()));
410    out.push(':');
411    for dep in deps {
412        out.push_str(" \\\n    ");
413        out.push_str(&escape_make(&dep.display().to_string()));
414    }
415    out.push('\n');
416    if let Some(parent) = depfile.parent() {
417        std::fs::create_dir_all(parent)
418            .with_context(|| format!("create depfile parent `{}`", parent.display()))?;
419    }
420    std::fs::write(depfile, out)
421        .with_context(|| format!("write depfile `{}`", depfile.display()))?;
422    Ok(())
423}

425/// Make-format target/dep escape: spaces → `\ `, `#` → `\#`.
426fn escape_make(s: &str) -> String {
427    let mut out = String::with_capacity(s.len());
428    for c in s.chars() {
429        match c {
430            ' ' => out.push_str("\\ "),
431            '#' => out.push_str("\\#"),
432            other => out.push(other),
433        }
434    }
435    out
436}
