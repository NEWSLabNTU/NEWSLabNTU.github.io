1//! C Entry-pkg TU emitter (typed, RFC-0043 / phase-257).
2//!
3//! Maps a [`Plan`] (see `super::mod`) onto a generated typed `main.c`: per launch
4//! node it creates a `nros_cpp_node_t` (`nros_cpp_node_create`) and calls the
5//! node's `NROS_C_COMPONENT` factory/configure seam
6//! (`__nros_c_component_<pkg>_{create,configure}`) on the real executor; `main`
7//! drives `nros_board_native_run_components`. The C counterpart of
8//! [`super::emit_cpp::emit_typed`]. The legacy non-typed emitter (the
9//! `EntryNodeRuntime` interpreter via `nros_board_native_run`) was retired in
10//! phase-257 Stage-3.
11
12use std::fmt::Write;
13
14use super::{Plan, sanitize_pkg};
15
16/// Phase 257 (W0-A) — a `lang == "c"` node is a `NROS_C_COMPONENT` typed
17/// component the C Entry installs via its `__nros_c_component_<pkg>_*` seam.
18fn is_c_node(n: &super::PlanNode) -> bool {
19    n.lang.as_deref() == Some("c")
20}
21
22/// Phase 257 (W0-A, RFC-0043) — emit the **typed** C Entry TU: route each launch
23/// node to the real executor via its `NROS_C_COMPONENT` factory/configure seam,
24/// driven by the C-ABI `nros_board_native_run_components`. The C counterpart of
25/// [`super::emit_cpp::emit_typed`]; pure-C workspaces only (every node is a C
26/// component — a non-C node has no C seam and is rejected).
27pub fn emit_typed(plan: &Plan) -> Result<String, String> {
28    for n in &plan.nodes {
29        if !is_c_node(n) {
30            return Err(format!(
31                "typed C entry emit: node pkg `{}` exec `{}` is lang `{}`, not `c` — \
32                 a `--lang c` typed Entry supports C components only (use `--lang cpp` for \
33                 a mixed C/C++/Rust workspace)",
34                n.pkg,
35                n.exec,
36                n.lang.as_deref().unwrap_or("<unset>")
37            ));
38        }
39    }
40
41    let mut out = String::new();
42    let _ = writeln!(
43        out,
44        "/* Generated by `nros codegen entry --lang c --typed` (RFC-0043)\n\
45         *   bringup = {bringup}\n\
46         *   launch  = {launch}\n\
47         *   board   = {board}\n\
48         *\n\
49         * DO NOT EDIT — regenerated at configure time. Routes each launch node\n\
50         * to the real executor via its `NROS_C_COMPONENT` factory/configure seam\n\
51         * (no synthesizing interpreter); `nros_board_native_run_components` owns\n\
52         * init/spin/shutdown.\n\
53         */",
54        bringup = plan.bringup,
55        launch = plan.launch_file.display(),
56        board = plan.board,
57    );
58    out.push('\n');
59
60    out.push_str("#include <nros/main.h>        /* nros_board_native_run_components */\n");
61    // The cbindgen FFI header is the superset: full `nros_cpp_node_t`, `nros_cpp_node_create`,
62    // and the `NROS_CPP_RET_*` codes. We do NOT include `<nros/component.h>` — it redefines
63    // `struct nros_cpp_qos_t` + the `nros_cpp_*` prototypes, which collide in C (struct
64    // redefinition is ill-formed). The entry only creates nodes; the component TUs own the
65    // pub/sub seam.
66    out.push_str("#include <nros/nros_cpp_ffi.h> /* nros_cpp_node_t + nros_cpp_node_create + ret codes */\n\n");
67
68    // Forward-declare the C-ABI factory + configure for each unique C pkg.
69    out.push_str("/* C component factory + configure seam (NROS_C_COMPONENT). The node's\n");
70    out.push_str(" * `nros_cpp_node_t*` + executor handle are handed to `configure`, which\n");
71    out.push_str(" * registers the component's real callbacks on the executor. */\n");
72    let mut seen_pkgs: Vec<String> = Vec::new();
73    for n in &plan.nodes {
74        let pkg = sanitize_pkg(&n.pkg);
75        if seen_pkgs.contains(&pkg) {
76            continue;
77        }
78        let _ = writeln!(out, "extern void* __nros_c_component_{pkg}_create(void);");
79        let _ = writeln!(
80            out,
81            "extern int32_t __nros_c_component_{pkg}_configure(const nros_cpp_node_t* node, void* executor, void* self);"
82        );
83        seen_pkgs.push(pkg);
84    }
85    out.push('\n');
86
87    // Static per-node storage — one `nros_cpp_node_t` per launch `<node>` row.
88    out.push_str("/* Static per-node storage (outlives the spin loop; no heap). */\n");
89    for (i, _n) in plan.nodes.iter().enumerate() {
90        let _ = writeln!(out, "static nros_cpp_node_t __nros_node_{i};");
91    }
92    out.push('\n');
93
94    // setup (post-init): create each node + wire its component's real callbacks.
95    out.push_str("/* setup (post-init): the executor handle is passed in; create each node +\n");
96    out.push_str(" * configure its component on the real executor. */\n");
97    out.push_str("static int32_t __nros_entry_setup(void* executor) {\n");
98    out.push_str("    if (executor == NULL) {\n");
99    out.push_str("        return (int32_t)NROS_CPP_RET_NOT_INIT;\n");
100    out.push_str("    }\n");
101    for (i, n) in plan.nodes.iter().enumerate() {
102        let pkg = sanitize_pkg(&n.pkg);
103        let node_name = n.name.as_deref().unwrap_or(&n.exec);
104        let name_lit = node_name.replace('\\', "\\\\").replace('"', "\\\"");
105        let _ = writeln!(out, "    {{");
106        let _ = writeln!(
107            out,
108            "        nros_cpp_ret_t nrc = nros_cpp_node_create(executor, \"{name_lit}\", \"/\", &__nros_node_{i});"
109        );
110        out.push_str("        if (nrc != NROS_CPP_RET_OK) return (int32_t)nrc;\n");
111        let _ = writeln!(
112            out,
113            "        void* self = __nros_c_component_{pkg}_create();"
114        );
115        let _ = writeln!(
116            out,
117            "        int32_t crc = __nros_c_component_{pkg}_configure(&__nros_node_{i}, executor, self);"
118        );
119        out.push_str("        if (crc != 0) return crc;\n");
120        let _ = writeln!(out, "    }}");
121    }
122    out.push_str("    return 0;\n");
123    out.push_str("}\n\n");
124
125    out.push_str("int main(int argc, char** argv) {\n");
126    out.push_str("    (void)argc;\n");
127    out.push_str("    (void)argv;\n");
128    out.push_str("    return nros_board_native_run_components(&__nros_entry_setup);\n");
129    out.push_str("}\n");
130
131    Ok(out)
132}

134#[cfg(test)]
135mod tests {
136    use super::*;
137    use crate::codegen::entry::PlanNode;
138    use std::path::PathBuf;
139
140    fn fixture_plan(nodes: &[(&str, &str)]) -> Plan {
141        Plan {
142            board: "native".into(),
143            nodes: nodes
144                .iter()
145                .map(|(pkg, exec)| PlanNode {
146                    pkg: (*pkg).into(),
147                    exec: (*exec).into(),
148                    name: None,
149                    namespace: None,
150                    class_name: None,
151                    class_header: None,
152                    lang: Some("c".into()),
153                    shape: None,
154                    host: None,
155                    qos_overrides: Vec::new(),
156                })
157                .collect(),
158            depfile_paths: Vec::new(),
159            bringup: "demo_bringup".into(),
160            launch_file: PathBuf::from("/tmp/system.launch.xml"),
161        }
162    }
163
164    #[test]
165    fn typed_emit_creates_node_and_configures_component() {
166        let plan = fixture_plan(&[("talker_pkg", "talker")]);
167        let src = emit_typed(&plan).expect("typed C emit ok");
168        // factory + configure seam forward-decls.
169        assert!(src.contains("extern void* __nros_c_component_talker_pkg_create(void);"));
170        assert!(src.contains(
171            "extern int32_t __nros_c_component_talker_pkg_configure(const nros_cpp_node_t* node, void* executor, void* self);"
172        ));
173        // setup creates the node + configures on the executor.
174        assert!(src.contains("nros_cpp_node_create(executor, \"talker\", \"/\", &__nros_node_0)"));
175        assert!(
176            src.contains("__nros_c_component_talker_pkg_configure(&__nros_node_0, executor, self)")
177        );
178        // main drives the real-executor lifecycle.
179        assert!(src.contains("nros_board_native_run_components(&__nros_entry_setup)"));
180    }
181
182    #[test]
183    fn typed_emit_preserves_launch_order() {
184        let plan = fixture_plan(&[("talker_pkg", "talker"), ("listener_pkg", "listener")]);
185        let src = emit_typed(&plan).expect("typed C emit ok");
186        let pos_talker = src.find("__nros_c_component_talker_pkg_create").unwrap();
187        let pos_listener = src.find("__nros_c_component_listener_pkg_create").unwrap();
188        assert!(pos_talker < pos_listener);
189    }
190
191    #[test]
192    fn typed_emit_dashed_pkg_names_sanitised() {
193        let plan = fixture_plan(&[("dashy-pkg", "talker")]);
194        let src = emit_typed(&plan).expect("typed C emit ok");
195        assert!(src.contains("__nros_c_component_dashy_pkg_create"));
196        assert!(!src.contains("dashy-pkg_create"));
197    }
198
199    #[test]
200    fn typed_emit_rejects_non_c_node() {
201        let mut plan = fixture_plan(&[("talker_pkg", "talker")]);
202        plan.nodes[0].lang = Some("cpp".into());
203        let err = emit_typed(&plan).unwrap_err();
204        assert!(err.contains("not `c`"), "{err}");
205        assert!(err.contains("talker_pkg"), "{err}");
206    }
207
208    #[test]
209    fn typed_emit_has_required_includes() {
210        let plan = fixture_plan(&[("talker_pkg", "talker")]);
211        let src = emit_typed(&plan).expect("typed C emit ok");
212        assert!(src.contains("#include <nros/main.h>"));
213        assert!(src.contains("#include <nros/nros_cpp_ffi.h>"));
214        assert!(!src.contains("#include <nros/component.h>"));
215        assert!(src.contains("int main(int argc, char** argv)"));
216    }
