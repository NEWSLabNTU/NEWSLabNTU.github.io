1//! `nros setup` — Phase 187.2: resolve a board's toolchain/SDK package set from
2//! the index and report the install plan. The actual fetch / source-build /
3//! cache is Phase 187.3; this verb does the CLI + board→package resolution +
4//! `--list` / `--licenses` / the per-host disposition plan.

30pub struct Args {

33    /// omit it for the legacy host-toolchain `nros setup <board>` flow below.

35    pub command: Option<SetupCommand>,
36
37    /// Board to set up (resolves its toolchain/SDK package set from the index
38    /// `[board.*]` table).

40
41    /// List every package in the index + its version.
42    #[arg(long)]
43    pub list: bool,
44
45    /// Show the license-gated packages + how to install them.
46    #[arg(long)]
47    pub licenses: bool,
48

53
54    /// Provision a single `[source.*]` package by name from the index (Phase

57    /// The `just <module> setup` recipes call this instead of inlining

59    #[arg(long = "source")]
60    pub sources: Vec<String>,

75    #[arg(long)]
76    pub rmw: Option<String>,
77
78    /// Resolve + print the plan without fetching/building anything.
79    #[arg(long)]
80    pub dry_run: bool,
81
82    /// Provision full git history instead of the per-source shallow default
83    /// (`--depth 1`). Use when you want `git log` / `blame` / branching in a
84    /// provisioned source or submodule. Overrides the index `shallow` for this
85    /// invocation only — no shared-file edit. (An already-shallow checkout is
86    /// deepened in place with `git -C <path> fetch --unshallow`.)
87    #[arg(long, conflicts_with = "shallow")]
88    pub full: bool,
89
90    /// Force shallow (`--depth 1`) even for sources that set `shallow = false`
91    /// in the index. The inverse of `--full`.
92    #[arg(long)]
93    pub shallow: bool,

134/// Per-invocation shallow override from `--full` / `--shallow`: `None` = use the
135/// per-source index default, `Some(false)` = full history, `Some(true)` = force
136/// shallow.
137fn shallow_override(args: &Args) -> Option<bool> {
138    if args.full {
139        Some(false)
140    } else if args.shallow {
141        Some(true)
142    } else {
143        None
144    }
145}

147pub fn run(args: Args) -> Result<()> {
148    if let Some(command) = args.command {
149        return match command {
150            SetupCommand::Board(b) => run_board(b),
151        };
152    }
153
154    let index = SdkIndex::load(&args.index)?;
155    let host = host_key();
156
157    if args.list {
158        print_list(&index);
159        return Ok(());
160    }
161    if args.licenses {
162        print_licenses(&index);
163        return Ok(());
164    }
165
166    if let Some(tool) = args.tool.as_deref() {
167        return install_single_tool(&index, tool, args.prefix.as_deref(), args.dry_run);
168    }
169
170    if !args.sources.is_empty() {
171        return provision_named_sources(
172            &index,
173            &args.index,
174            &args.sources,
175            args.dry_run,
176            shallow_override(&args),
177        );
178    }
179
180    let board = match args.board.as_deref() {
181        Some(b) => b,
182        None => {
183            bail!("nros setup: give a <board>, `--tool <name>`, `--list`, or `--licenses`")
184        }
185    };
186
187    let packages = resolve_packages_with_rmw(&index, board, args.rmw.as_deref())?;
188    eprintln!(
189        "nros setup: {board} (rmw {}) needs {} package(s):",
190        args.rmw.as_deref().unwrap_or("zenoh"),
191        packages.len()
192    );
193
194    let root = store_root();
195    let workspace = index_workspace(&args.index);
196    let lock_path = PathBuf::from(LOCK_FILE);
197    let mut lock = SdkLock::load(&lock_path)?;
198    let mut installed = false;
199
200    for name in &packages {
201        // `[tool.*]` packages install into the shared store; `[source.*]` are
202        // provisioned into their index-declared `dest` (Phase 195.B);
203        // `[gated.*]` are user-installed.
204        let Some(tool) = index.tool.get(*name) else {
205            if let Some(src) = index.source.get(*name) {
206                let disp =
207                    provision_source(name, src, &workspace, args.dry_run, shallow_override(&args))
208                        .wrap_err_with(|| format!("provision source {name}"))?;
209                eprintln!("  {:<22} {}", name, describe_source(src, &disp));
210                if matches!(disp, SourceDisposition::Provisioned) {
211                    installed = true;
212                }
213            } else {
214                eprintln!("  {:<22} {}", name, disposition(&index, name, &host));
215            }
216            continue;
217        };
218        let prefix = tool_prefix(&root, name, &tool.version);
219        let action = plan_install(tool, &host, &prefix);
220        eprintln!("  {:<22} {}", name, describe(&action, &tool.version, &host));
221
222        if args.dry_run {
223            continue;
224        }
225        match action {
226            InstallAction::Unavailable => {
227                bail!(
228                    "nros setup: {name} {} has no prebuilt for {host} and no source recipe \
229                     (add one to the index, or set up that host's toolchain manually)",
230                    tool.version
231                );
232            }
233            other => {
234                let provenance = execute(&other, name, &tool.version, &prefix)
235                    .wrap_err_with(|| format!("install {name} {}", tool.version))?;
236                lock.record(name, &provenance);
237                installed = true;
238                eprintln!("    → {}", prefix.display());
239            }
240        }
241    }
242
243    if args.dry_run {
244        eprintln!("(--dry-run: nothing installed)");
245    } else if installed {
246        lock.save(&lock_path)?;
247        eprintln!(
248            "nros setup: {board} ready; locked in {}",
249            lock_path.display()
250        );
251    } else {
252        eprintln!("nros setup: {board} — all packages already present");
253    }
254    Ok(())
255}

257/// Verify the consumer's Zephyr checkout matches the board's declared
258/// `zephyr_line` (major.minor), reading `<ws>/zephyr/VERSION`. Hard-errors on a
259/// mismatch: the line-specific patch set (`patches/<line>.sh`) and nano-ros's
260/// platform-zephyr code target that exact Zephyr API, so applying them to a
261/// different line drifts silently into deep compile errors (issue 0054). A
262/// missing/unparseable VERSION is a warning, not a hard stop (don't block an
263/// unusual-but-valid layout).
264fn verify_zephyr_line(zephyr_ws: &Path, zephyr_line: &str) -> Result<()> {
265    let version_file = zephyr_ws.join("zephyr").join("VERSION");
266    let text = match std::fs::read_to_string(&version_file) {
267        Ok(t) => t,
268        Err(e) => {
269            eprintln!(
270                "  warning: cannot read {} ({e}) — skipping Zephyr-line check \
271                 (board declares {zephyr_line}).",
272                version_file.display()
273            );
274            return Ok(());
275        }
276    };
277    let field = |key: &str| -> Option<String> {
278        text.lines().find_map(|l| {
279            let (k, v) = l.split_once('=')?;
280            (k.trim() == key).then(|| v.trim().to_string())
281        })
282    };
283    let (Some(major), Some(minor)) = (field("VERSION_MAJOR"), field("VERSION_MINOR")) else {
284        eprintln!(
285            "  warning: {} has no VERSION_MAJOR/MINOR — skipping Zephyr-line check.",
286            version_file.display()
287        );
288        return Ok(());
289    };
290    let actual = format!("{major}.{minor}");
291    if actual != zephyr_line {
292        bail!(
293            "nros setup board: consumer Zephyr is v{actual} but this board declares \
294             zephyr_line={zephyr_line}. The {zephyr_line} patch set and nano-ros's \
295             platform-zephyr code target the {zephyr_line} API; provisioning v{actual} \
296             would drift into deep compile errors. Pin the consumer's zephyr to the \
297             {zephyr_line} line (west.yml revision / submodule) and re-run."
298        );
299    }
300    eprintln!("  zephyr check: consumer is v{actual} — matches board line {zephyr_line}");
301    Ok(())
302}

304/// `nros setup board <name> --zephyr-workspace <dir>` (Phase 215.J.2).
305///
306/// Provisions a DOWNSTREAM Zephyr consumer's tree for a nano-ros board, driven
307/// entirely by the board's provisioning contract (`board.cmake` /
308/// `[package.metadata.nros.board]` — Phase 215.J.1). Reuses the existing
309/// index-driven `--source` fetch + the workspace-parameterized patch scripts
310/// (`scripts/zephyr/patches/<line>.sh $WORKSPACE`); no consumer-side
311/// duplication, no forked index logic. Idempotent: source provisioning skips
312/// when present, the patch scripts self-detect prior application, and
313/// `rustup target add` is a no-op when installed.
314fn run_board(args: BoardSetupArgs) -> Result<()> {
315    // 1. Resolve the nano-ros workspace root (board crate + scripts + index).
316    let root = match args.workspace {
317        Some(p) => p,
318        None => find_workspace_root().wrap_err(
319            "nros setup board: could not locate the nano-ros workspace root \
320             (pass --workspace <path> or set NROS_WORKSPACE_ROOT)",
321        )?,
322    };
323
324    // 2. Validate the consumer's Zephyr workspace.
325    let zephyr_ws = &args.zephyr_workspace;
326    if !zephyr_ws.join("zephyr").is_dir() {
327        bail!(
328            "nros setup board: --zephyr-workspace `{}` does not look like a Zephyr \
329             workspace (no `zephyr/` subdir). Point it at the consumer's west \
330             topdir (the tree containing `zephyr/` + `modules/`).",
331            zephyr_ws.display()
332        );
333    }
334
335    // 3. Read the board's provisioning contract.
336    let crate_dir = root
337        .join("packages")
338        .join("boards")
339        .join(format!("nros-board-{}", args.name));
340    let cargo_toml = crate_dir.join("Cargo.toml");
341    if !cargo_toml.is_file() {
342        bail!(
343            "nros setup board: no board crate at `{}` (check the board name; \
344             `nros board list` enumerates them)",
345            crate_dir.display()
346        );
347    }
348    let meta = parse_board_metadata(&cargo_toml)
349        .wrap_err_with(|| format!("read provisioning contract from {}", cargo_toml.display()))?;
350
351    let Some(zephyr_line) = meta.zephyr_line.as_deref() else {
352        bail!(
353            "nros setup board: `{}` has no `zephyr_line` in its provisioning \
354             contract — it is not a Zephyr consumer board (nothing to provision). \
355             Non-Zephyr boards are consumed via cargo path-deps, not `nros setup board`.",
356            args.name
357        );
358    };
359
360    // Verify the consumer's Zephyr matches the board's declared line BEFORE
361    // touching the tree — the line-specific patch set + nano-ros's
362    // platform-zephyr code target that exact Zephyr API, so a mismatched
363    // checkout drifts silently into deep compile errors (issue 0054).
364    verify_zephyr_line(zephyr_ws, zephyr_line)?;
365
366    eprintln!(
367        "nros setup board {}: provisioning consumer Zephyr tree at {}",
368        args.name,
369        zephyr_ws.display()
370    );
371    eprintln!("  contract: zephyr_line={zephyr_line}, requires_rust={}, rmw_source={}, rust_targets=[{}]",
372        meta.requires_rust.unwrap_or(false),
373        meta.rmw_source.as_deref().unwrap_or("-"),
374        meta.rust_targets.join(", "),
375    );
376
377    // (a) Fetch the board's RMW source — index-driven, into nano-ros's own
378    //     tree (the consumer links it via `nano_ros_use_board()` /
379    //     `add_subdirectory(packages/dds/...)`), same as `just zephyr setup`.
380    if let Some(rmw_source) = meta.rmw_source.as_deref() {
381        let index_path = if args.index.is_absolute() {
382            args.index.clone()
383        } else {
384            root.join(&args.index)
385        };
386        eprintln!("  (a) RMW source: nros setup --source {rmw_source}");
387        if args.dry_run {
388            eprintln!("      (--dry-run: skipped)");
389        } else {
390            let index = SdkIndex::load(&index_path).wrap_err_with(|| {
391                format!("load SDK index from {}", index_path.display())
392            })?;
393            provision_named_sources(
394                &index,
395                &index_path,
396                std::slice::from_ref(&rmw_source.to_string()),
397                false,
398                None,
399            )?;
400        }
401    }
402
403    // (b) Apply the zephyr-line patch set to the CONSUMER's tree. The patch
404    //     scripts already take the workspace dir as $1.
405    let patch_script = root
406        .join("scripts")
407        .join("zephyr")
408        .join("patches")
409        .join(format!("{zephyr_line}.sh"));
410    if !patch_script.is_file() {
411        bail!(
412            "nros setup board: no patch set for Zephyr line `{zephyr_line}` at {} \
413             (add scripts/zephyr/patches/{zephyr_line}.sh — see patches/README.md)",
414            patch_script.display()
415        );
416    }
417    eprintln!(
418        "  (b) zephyr patches: bash {} {}",
419        patch_script.display(),
420        zephyr_ws.display()
421    );
422    if !args.dry_run {
423        let status = Command::new("bash")
424            .arg(&patch_script)
425            .arg(zephyr_ws)
426            .status()
427            .wrap_err_with(|| format!("spawn {}", patch_script.display()))?;
428        if !status.success() {
429            bail!(
430                "nros setup board: patch set {} exited with {status}",
431                patch_script.display()
432            );
433        }
434    }
435
436    // (c) rustup target add (when the board requires Rust).
437    if meta.requires_rust.unwrap_or(false) && !meta.rust_targets.is_empty() {
438        eprintln!(
439            "  (c) rust targets: rustup target add {}",
440            meta.rust_targets.join(" ")
441        );
442        if !args.dry_run {
443            for target in &meta.rust_targets {
444                let status = Command::new("rustup")
445                    .args(["target", "add", target])
446                    .status()
447                    .wrap_err("spawn rustup (is it on PATH?)")?;
448                if !status.success() {
449                    bail!("nros setup board: `rustup target add {target}` failed ({status})");
450                }
451            }
452        }
453    }
454
455    // (d) Ensure the zephyr-lang-rust module is in the consumer's tree. The
456    //     module fetch itself is west-native (the board ships a
457    //     `west-downstream.yml` `import:false` fragment — Phase 215.J.3); here
458    //     we only verify + instruct, never edit the consumer's manifest.
459    if meta.requires_rust.unwrap_or(false) {
460        ensure_lang_rust_module(&crate_dir, zephyr_ws);
461    }
462
463    if args.dry_run {
464        eprintln!("nros setup board {}: (--dry-run: nothing changed)", args.name);
465    } else {
466        eprintln!(
467            "nros setup board {}: consumer Zephyr tree provisioned",
468            args.name
469        );
470    }
471    Ok(())
472}

474/// Phase 215.J.2 step (d) — verify the `zephyr-lang-rust` module is present in
475/// the consumer's tree; warn + point at the board's `west-downstream.yml`
476/// import fragment (Phase 215.J.3) when it is not. Never mutates the consumer's
477/// west manifest — the consumer keeps manifest authority.
478fn ensure_lang_rust_module(crate_dir: &Path, zephyr_ws: &Path) {
479    let module = zephyr_ws.join("modules").join("lang").join("rust");
480    if module.join("Kconfig").is_file() {
481        eprintln!("  (d) zephyr-lang-rust: present at {}", module.display());
482        return;
483    }
484    let fragment = crate_dir.join("west-downstream.yml");
485    eprintln!(
486        "  (d) zephyr-lang-rust: MISSING at {}.\n\
487         \x20     Add the board's import fragment to your west manifest, then `west update`:\n\
488         \x20       manifest:\n\
489         \x20         self:\n\
490         \x20           import:\n\
491         \x20             - file: {}\n\
492         \x20     (board-shipped `import:false` fragment — pins zephyr-lang-rust at\n\
493         \x20      nano-ros's supported rev; `name-allowlist` keeps it to that one module.)",
494        module.display(),
495        fragment.display()
496    );
497}

499/// Install one tool by name (`nros setup --tool <name>`). `prefix_override`
500/// (from `--prefix`) places it outside the shared store — e.g. `build/qemu`, the
501/// location the test harness already reads, so `just <module> setup` can delegate
502/// here with no harness change and no script-side path resolution. Prebuilt-or-
503/// source per the index (187.3); the lockfile is only updated for shared-store
504/// installs (a `--prefix` placement is workspace-local).
505fn install_single_tool(
506    index: &SdkIndex,
507    name: &str,
508    prefix_override: Option<&Path>,
509    dry_run: bool,
510) -> Result<()> {
511    let host = host_key();
512    let tool = index
513        .tool
514        .get(name)
515        .ok_or_else(|| eyre::eyre!("nros setup --tool: no [tool.{name}] in the index"))?;
516    let root = store_root();
517    let prefix = prefix_override
518        .map(Path::to_path_buf)
519        .unwrap_or_else(|| tool_prefix(&root, name, &tool.version));
520
521    let action = plan_install(tool, &host, &prefix);
522    eprintln!(
523        "nros setup --tool {name}: {} → {}",
524        describe(&action, &tool.version, &host),
525        prefix.display()
526    );
527    if dry_run {
528        eprintln!("(--dry-run: nothing installed)");
529        return Ok(());
530    }
531    match action {
532        InstallAction::Present => {}
533        InstallAction::Unavailable => bail!(
534            "nros setup --tool {name} {}: no prebuilt for {host} and no source recipe",
535            tool.version
536        ),
537        other => {
538            let prov = execute(&other, name, &tool.version, &prefix)
539                .wrap_err_with(|| format!("install {name} {}", tool.version))?;
540            // Only the shared store is tracked by the lock; --prefix is local.
541            if prefix_override.is_none() {
542                let lock_path = PathBuf::from(LOCK_FILE);
543                let mut lock = SdkLock::load(&lock_path)?;
544                lock.record(name, &prov);
545                lock.save(&lock_path)?;
546            }
547        }
548    }
549    Ok(())
550}

552/// Provision one or more `[source.*]` packages by name (`nros setup --source
553/// <name> …`) — the index-driven replacement for inline `git submodule update
554/// <path>` in the `just <module> setup` recipes (Phase 195.B; mirrors what
555/// 187.6 did for `qemu`/`zenohd` via `--tool`). The index is the SSOT: `dest`,
556/// `ref`, and `submodule` all come from data.
557fn provision_named_sources(
558    index: &SdkIndex,
559    index_path: &Path,
560    names: &[String],
561    dry_run: bool,
562    shallow_override: Option<bool>,
563) -> Result<()> {
564    let workspace = index_workspace(index_path);
565    for name in names {
566        let src = index
567            .source
568            .get(name.as_str())
569            .ok_or_else(|| eyre::eyre!("nros setup --source: no [source.{name}] in the index"))?;
570        let disp = provision_source(name, src, &workspace, dry_run, shallow_override)
571            .wrap_err_with(|| format!("provision source {name}"))?;
572        eprintln!(
573            "nros setup --source {name}: {}",
574            describe_source(src, &disp)
575        );
576    }
577    Ok(())
578}

580/// Phase 187.6 — lazy install support: resolve the
581/// board's index tools and install any not already in the store, so a first
582/// platform build needs no separate `nros setup` (the PlatformIO auto-install
583/// ergonomic). Only `[tool.*]` packages are installed; `[source.*]` build with
584/// the app and `[gated.*]` are user-provided. Opt out with `NROS_NO_AUTO_SETUP`.
585/// No-op (empty) when no index is found; an unavailable tool warns rather than
586/// fails so the downstream platform build surfaces the real miss (e.g. a system-installed
587/// toolchain the index doesn't host).
588///
589/// Returns the `bin/` dirs of the resolved tools present in the store — Method A
590/// callers ([`activate_store_path`]) prepend these to the env so every spawned
591/// child finds the toolchain, without any non-`nros` script resolving paths.
592pub fn ensure_tools(board: &str, workspace: Option<&Path>) -> Result<Vec<PathBuf>> {
593    if std::env::var_os("NROS_NO_AUTO_SETUP").is_some() {
594        return Ok(Vec::new());
595    }
596    let Some(index_path) = locate_index(workspace) else {
597        return Ok(Vec::new());
598    };
599    let index = SdkIndex::load(&index_path)?;
600    let host = host_key();
601    let root = store_root();
602    let ws = index_workspace(&index_path);
603    let lock_path = PathBuf::from(LOCK_FILE);
604    let mut lock = SdkLock::load(&lock_path)?;
605    let mut installed = false;
606    let mut bin_dirs = Vec::new();
607
608    // Unknown board ⇒ no known package set — warn + skip (lazy auto-setup is
609    // best-effort; the user provides tools). `nros setup` errors instead.
610    // Auto-setup defaults to the zenoh RMW host set (rmw=None). The default
611    // keeps the historical behaviour (e.g. native pulls `zenohd`).
612    let packages = match resolve_packages_with_rmw(&index, board, None) {
613        Ok(p) => p,
614        Err(_) => {
615            eprintln!(
616                "nros: board '{board}' not in the SDK index — skipping auto-setup \
617                 (provide its tools yourself, or add a [board.{board}] entry)"
618            );
619            return Ok(Vec::new());
620        }
621    };
622    for name in packages {
623        let Some(tool) = index.tool.get(name) else {
624            // Phase 195.B — provision `[source.*]` into its index `dest` so a
625            // first build/deploy gets the kernel/lib source with no `just`.
626            if let Some(src) = index.source.get(name) {
627                // Lazy auto-setup uses the index per-source default (no
628                // `--full`/`--shallow` to thread here).
629                match provision_source(name, src, &ws, false, None) {
630                    Ok(SourceDisposition::Provisioned) => {
631                        eprintln!(
632                            "nros: provisioned source {name} → {}",
633                            src.dest.as_deref().unwrap_or("-")
634                        );
635                        installed = true;
636                    }
637                    Ok(_) => {}
638                    Err(e) => eprintln!(
639                        "nros: source {name} provisioning failed ({e}) — provide it yourself if the build needs it"
640                    ),
641                }
642            }
643            continue; // gated / not-in-index — not a store tool
644        };
645        let prefix = tool_prefix(&root, name, &tool.version);
646        match plan_install(tool, &host, &prefix) {
647            InstallAction::Present => {}
648            InstallAction::Unavailable => {
649                eprintln!(
650                    "nros: {name} {} unavailable for {host} (no prebuilt, no source) — \
651                     install it yourself if the build needs it",
652                    tool.version
653                );
654                continue; // not in the store → nothing to add to PATH
655            }
656            action => {
657                eprintln!(
658                    "nros: auto-installing {name} {} (set NROS_NO_AUTO_SETUP to skip)",
659                    tool.version
660                );
661                let prov = execute(&action, name, &tool.version, &prefix)
662                    .wrap_err_with(|| format!("auto-setup {name} {}", tool.version))?;
663                lock.record(name, &prov);
664                installed = true;
665                eprintln!("    → {}", prefix.display());
666            }
667        }
668        let bin = prefix.join("bin");
669        if bin.is_dir() {
670            bin_dirs.push(bin);
671        }
672    }
673    if installed {
674        lock.save(&lock_path)?;
675    }
676    Ok(bin_dirs)
677}

679/// Method A — prepend the store `bin/` dirs (from [`ensure_tools`]) to this
680/// process's `PATH` so child platform-tool invocations (cargo, cmake,
681/// west, the `build[]`/`package[]` steps) find the toolchain on `PATH`. `nros`
682/// is the single resolver; non-`nros` scripts/code never hunt for SDK paths.
683/// A no-op when `dirs` is empty (no store tools / auto-setup skipped).
684pub fn activate_store_path(dirs: &[PathBuf]) {
685    if dirs.is_empty() {
686        return;
687    }
688    let mut parts: Vec<PathBuf> = dirs.to_vec();
689    if let Some(cur) = std::env::var_os("PATH") {
690        parts.extend(std::env::split_paths(&cur));
691    }
692    if let Ok(joined) = std::env::join_paths(parts) {
693        // SAFETY: a CLI invocation activating its own toolchain for the child
694        // processes it is about to spawn; set before any thread reads the env.
695        unsafe { std::env::set_var("PATH", joined) };
696    }
697}

699/// Locate the SDK index for auto-setup: cwd, then the passed workspace, then
700/// `$NROS_WORKSPACE`. `None` ⇒ auto-setup is a no-op (not every build runs near
701/// a nano-ros workspace). Shared with `nros doctor`'s license-gate check (187.7).
702pub(crate) fn locate_index(workspace: Option<&Path>) -> Option<PathBuf> {
703    let cwd = PathBuf::from("nros-sdk-index.toml");
704    if cwd.is_file() {
705        return Some(cwd);
706    }
707    let ws = workspace
708        .map(Path::to_path_buf)
709        .or_else(|| std::env::var_os("NROS_WORKSPACE").map(PathBuf::from));
710    ws.map(|w| w.join("nros-sdk-index.toml"))
711        .filter(|p| p.is_file())
712}
713
714/// The workspace root a `[source.*]` `dest` is resolved against: the directory
715/// containing the index (Phase 195.B — `dest` is workspace-relative index data,
716/// never a path baked into the binary). Falls back to `.` for a bare index name.
717fn index_workspace(index: &Path) -> PathBuf {
718    index
719        .parent()
720        .filter(|p| !p.as_os_str().is_empty())
721        .map(Path::to_path_buf)
722        .unwrap_or_else(|| PathBuf::from("."))
723}

725/// One-line description of a source's provisioning outcome (Phase 195.B).
726fn describe_source(
727    src: &crate::orchestration::sdk_index::SourcePackage,
728    disp: &SourceDisposition,
729) -> String {
730    use crate::orchestration::sdk_index::SourceProvision;
731    let mode = match src.provision() {
732        SourceProvision::Clone => format!(
733            "clone {}@{}",
734            src.git.as_deref().unwrap_or("?"),
735            src.git_ref.as_deref().unwrap_or("?")
736        ),
737        SourceProvision::Submodule => {
738            format!("submodule {}", src.submodule.as_deref().unwrap_or("?"))
739        }
740        SourceProvision::None => "built with the app".to_string(),
741    };
742    let outcome = match disp {
743        SourceDisposition::Provisioned => "provisioned",
744        SourceDisposition::AlreadyPresent => "already present (skip)",
745        SourceDisposition::NoFetch => "no fetch step",
746        SourceDisposition::Planned => "would provision (--dry-run)",
747    };
748    format!(
749        "source {} — {mode} → {} [{outcome}]",
750        src.version,
751        src.dest.as_deref().unwrap_or("-")
752    )
753}

755/// One-line description of the planned action (mirrors `disposition`, but for an
756/// already-resolved [`InstallAction`]).
757fn describe(action: &InstallAction, version: &str, host: &str) -> String {
758    match action {
759        InstallAction::Present => format!("present {version} (skip)"),
760        InstallAction::Prebuilt { .. } => format!("prebuilt {version} (dist {host})"),
761        InstallAction::Source { .. } => format!("source build {version} (no prebuilt for {host})"),
762        InstallAction::Unavailable => {
763            format!("UNAVAILABLE {version} (no prebuilt for {host}, no source)")
764        }
765    }
766}

768/// Resolve a board to its SDK package set from the index `[board.*]` table — the
769/// board→toolchain SSOT (Phase 191.1). No board-name guessing: an unknown board
770/// is a clear error listing the known boards, not a silent wrong package set
771/// (the failure mode the old keyword heuristic had — it mis-resolved ESP32-C3 as
772/// Xtensa). Adding a board is a `[board.<name>]` entry, no code change.
773pub fn resolve_packages<'i>(index: &'i SdkIndex, board: &str) -> Result<Vec<&'i str>> {
774    match index.board.get(board) {
775        Some(entry) => Ok(entry.packages.iter().map(String::as_str).collect()),
776        None => {
777            let mut known: Vec<&str> = index.board.keys().map(String::as_str).collect();
778            known.sort_unstable();
779            bail!(
780                "nros setup: unknown board '{board}'. Known boards: {}. \
781                 Add a [board.{board}] entry to nros-sdk-index.toml.",
782                if known.is_empty() {
783                    "(none in index)".to_string()
784                } else {
785                    known.join(", ")
786                }
787            )
788        }
789    }
790}

792/// Resolve `board.packages ∪ rmw.packages` (Phase 191.6.a). RMW is an axis
793/// orthogonal to the board: the board contributes its platform/toolchain
794/// packages, the chosen RMW its host daemon/tool (`zenohd` / `xrce-agent` /
795/// `cyclonedds`) — no `board×rmw` pair enumeration. `rmw=None` defaults to
796/// `zenoh`. A legacy index with no `[rmw.*]` table returns the board set
797/// unchanged; an unknown RMW name errors (listing the known ones).
798pub fn resolve_packages_with_rmw<'i>(
799    index: &'i SdkIndex,
800    board: &str,
801    rmw: Option<&str>,
802) -> Result<Vec<&'i str>> {
803    let mut packages = resolve_packages(index, board)?;
804    if index.rmw.is_empty() {
805        return Ok(packages); // legacy index without the RMW axis
806    }
807    let rmw = rmw.unwrap_or("zenoh");
808    match index.rmw.get(rmw) {
809        Some(entry) => {
810            for pkg in &entry.packages {
811                let p = pkg.as_str();
812                if !packages.contains(&p) {
813                    packages.push(p);
814                }
815            }
816        }
817        None => {
818            let mut known: Vec<&str> = index.rmw.keys().map(String::as_str).collect();
819            known.sort_unstable();
820            bail!(
821                "nros setup: unknown rmw '{rmw}'. Known RMWs: {}.",
822                known.join(", ")
823            );
824        }
825    }
826    Ok(packages)
827}

829/// How `name` would be provisioned on `host`, per the index.
830fn disposition(index: &SdkIndex, name: &str, host: &str) -> String {
831    if let Some(tool) = index.tool.get(name) {
832        if tool.dist_for(host).is_some() {
833            format!("prebuilt {} (dist {host})", tool.version)
834        } else if tool.source.is_some() {
835            format!("source build {} (no prebuilt for {host})", tool.version)
836        } else {
837            format!(
838                "UNAVAILABLE {} (no prebuilt for {host}, no source)",
839                tool.version
840            )
841        }
842    } else if let Some(src) = index.source.get(name) {
843        format!("source {} (built with the app)", src.version)
844    } else if let Some(g) = index.gated.get(name) {
845        format!(
846            "license-gated {} (set ${}{})",
847            g.version,
848            g.env,
849            g.installer
850                .as_deref()
851                .map(|i| format!(", via {i}"))
852                .unwrap_or_default()
853        )
854    } else {
855        "NOT in index (add to nros-sdk-index.toml — 187.5)".to_string()
856    }
857}

859fn print_list(index: &SdkIndex) {
860    eprintln!("nros setup --list:");
861    for (name, t) in &index.tool {
862        eprintln!("  [tool]   {name:<22} {}", t.version);
863    }
864    for (name, s) in &index.source {
865        eprintln!("  [source] {name:<22} {}", s.version);
866    }
867    for (name, g) in &index.gated {
868        eprintln!("  [gated]  {name:<22} {} (${})", g.version, g.env);
869    }
870}

872fn print_licenses(index: &SdkIndex) {
873    if index.gated.is_empty() {
874        eprintln!("nros setup --licenses: no license-gated packages");
875        return;
876    }

878    for (name, g) in &index.gated {

880            "  {name:<16} {} — set ${}{}",
881            g.version,
882            g.env,
883            g.installer
884                .as_deref()
885                .map(|i| format!(" (via {i})"))
886                .unwrap_or_default()
887        );
888    }
