1//! `nros new <name>` — Phase 111.A.4.
2//!
3//! Forwards to `cargo_nano_ros::scaffold::scaffold_package` so the CLI
4//! stays in lockstep with the shared scaffolding implementation.
5//! Use-case (`talker` / `listener` / `service` / `action`) and RMW-choice
6//! diversification are accepted at the CLI for forward-compat but
7//! currently affect only the printed "Next steps" banner — full
8//! per-use-case template trees land alongside the Phase 112 example
9//! sweep.
10
11use cargo_nano_ros::scaffold::{
12    ComponentScaffoldConfig, ScaffoldConfig, scaffold_component, scaffold_package,
13};
14use clap::Args as ClapArgs;
15use eyre::{Result, bail};
16use std::path::PathBuf;
17
18use crate::cmd::{
19    new_system::{BringupScaffold, scaffold_bringup},
20    scaffold_deploy::{DeployScaffold, scaffold_deploy},
21};
22
23#[derive(Debug, ClapArgs)]
24pub struct Args {
25    /// Project directory to create (project mode), or the literal keyword
26    /// `system` to enter Phase 212.F bringup-scaffold mode: `nros new system
27    /// <name>_bringup --components <pkg1,pkg2,...>`.
28    pub name: Option<PathBuf>,
29
30    /// Phase 212.F bringup-scaffold mode — the bringup package directory.
31    /// Only consumed when the first positional is the literal `system`.
32    pub system_name: Option<PathBuf>,
33
34    /// Phase 212.F — comma-separated component package names for
35    /// `nros new system <bringup> --components <list>`.
36    #[arg(long, value_delimiter = ',')]
37    pub components: Vec<String>,
38
39    /// Phase 212.F — repeatable single-component form (alternative to
40    /// `--components <a,b,c>` when commas in the shell are awkward). Merged
41    /// with `--components` at dispatch time.
42    #[arg(long = "component-name")]
43    pub component_name: Vec<String>,
44
45    /// Phase 212.F — workspace root holding the cargo `Cargo.toml` to
46    /// update. Defaults to the parent of the bringup dir.
47    #[arg(long)]
48    pub workspace_root: Option<PathBuf>,
49
50    /// Phase 212.F — parent dir under which the bringup pkg is created.
51    /// Defaults to the current working directory.
52    #[arg(long)]
53    pub into: Option<PathBuf>,
54
55    /// Phase 212.F — skip the optional `config/` sub-dir.
56    #[arg(long)]
57    pub no_config: bool,
58
59    /// Phase 212.F — skip the optional `README.md`.
60    #[arg(long)]
61    pub no_readme: bool,
62
63    /// Target platform (required in project mode)
64    #[arg(long, value_parser = ["native", "freertos", "nuttx", "threadx", "zephyr", "posix", "baremetal"])]
65    pub platform: Option<String>,
66
67    /// RMW backend
68    #[arg(long, value_parser = ["zenoh", "xrce", "cyclonedds"], default_value = "zenoh")]
69    pub rmw: String,
70
71    /// Source language
72    #[arg(long, value_parser = ["rust", "c", "cpp"], default_value = "rust")]
73    pub lang: String,
74
75    /// Use case template
76    #[arg(long = "use-case", value_parser = ["talker", "listener", "service", "action"], default_value = "talker")]
77    pub use_case: String,
78
79    /// Phase 172 W.3 — scaffold a planned-mode **component** (a reusable
80    /// library node with an `nros::Component` + a folded `[component]`
81    /// `nros.toml`) instead of a direct-mode binary project. Platform-agnostic
82    /// (platform/RMW are chosen at deploy time), so `--platform` is not needed.
83    #[arg(long)]
84    pub component: bool,
85
86    /// Scaffold a `[deploy.<name>]` target into the bringup package's
87    /// `system.toml` (RFC-0004 §4) instead of a project.
88    #[arg(long)]
89    pub deploy: Option<String>,
90
91    /// Deploy kind (deploy mode) — free-form runner key (`self`, `qemu`,
92    /// `flash`, …) written verbatim to `[deploy.<name>].kind`.
93    #[arg(long, default_value = "self")]
94    pub kind: String,
95
96    /// Cargo target triple / board id / runner key (deploy mode)
97    #[arg(long)]
98    pub target: Option<String>,
99
100    /// Board (deploy mode)
101    #[arg(long)]
102    pub board: Option<String>,
103
104    /// Deploy mode: pick the bringup package whose `system.toml` to edit
105    /// when the workspace exposes more than one.
106    #[arg(long)]
107    pub bringup: Option<String>,
108
109    /// Deploy mode: also set the bringup `[system].default_launch` (bootstrap)
110    #[arg(long)]
111    pub from_launch: Option<String>,
112
113    /// Deploy mode: fork an existing `[deploy.<name>]` profile
114    #[arg(long)]
115    pub from_profile: Option<String>,
116
117    /// Overwrite an existing directory / `[deploy.<name>]` table
118    #[arg(long)]
119    pub force: bool,
120}
121
122pub fn run(args: Args) -> Result<()> {
123    // Phase 212.F — system / bringup mode: `nros new system <name>_bringup
124    // --components <list>`. The literal `system` keyword as the first
125    // positional dispatches here.
126    if args
127        .name
128        .as_ref()
129        .and_then(|p| p.to_str())
130        .map(|s| s == "system")
131        .unwrap_or(false)
132    {
133        let bringup_path = args.system_name.clone().ok_or_else(|| {
134            eyre::eyre!("`nros new system <name>_bringup` requires a bringup pkg name")
135        })?;
136        // Phase 212.F: validate the user-supplied name early so
137        // `foo/bar`, `..`, absolute paths surface a clean diagnostic
138        // before we touch the filesystem.
139        crate::cmd::new_system::validate_bringup_name(&bringup_path)?;
140        // Merge --components <a,b,c> with repeatable --component-name <x>.
141        let mut components: Vec<String> = args.components.clone();
142        components.extend(args.component_name.clone());
143        if components.is_empty() {
144            bail!(
145                "`nros new system <bringup>` requires --components <pkg1,pkg2,...> \
146                 (at least one component); --component-name <x> may be repeated as an alternative"
147            );
148        }
149        let cwd = std::env::current_dir()?;
150        // --into <dir> overrides cwd as the parent directory for the bringup.
151        let into = args.into.clone().unwrap_or_else(|| cwd.clone());
152        let bringup_dir = if bringup_path.is_absolute() {
153            bringup_path
154        } else {
155            into.join(&bringup_path)
156        };
157        let pkg_name = bringup_dir
158            .file_name()
159            .and_then(|n| n.to_str())
160            .ok_or_else(|| eyre::eyre!("invalid bringup package name"))?
161            .to_string();
162        let workspace_root = args
163            .workspace_root
164            .clone()
165            .or_else(|| bringup_dir.parent().map(|p| p.to_path_buf()))
166            .unwrap_or_else(|| cwd.clone());
167        let out = scaffold_bringup(&BringupScaffold {
168            bringup_dir: bringup_dir.clone(),
169            pkg_name: pkg_name.clone(),
170            components: components.clone(),
171            workspace_root,
172            emit_config: !args.no_config,
173            emit_readme: !args.no_readme,
174            force: args.force,
175        })?;
176        eprintln!(
177            "nros new system: scaffolded bringup pkg {pkg_name} at {} ({} component(s))",
178            out.bringup_dir.display(),
179            components.len()
180        );
181        if let Some(ws) = out.workspace_cargo_toml.as_ref() {
182            eprintln!(
183                "nros new system: updated [workspace] exclude in {}",
184                ws.display()
185            );
186        }
187        let _ = out; // silence unused warning under future changes
188        return Ok(());
189    }
190
191    // Deploy mode: `nros new --deploy <name> [--kind <k>] ...` writes a
192    // `[deploy.<name>]` into the bringup package's `system.toml` (RFC-0004 §4).
193    if let Some(deploy_name) = args.deploy {
194        return scaffold_deploy(&DeployScaffold {
195            name: deploy_name,
196            kind: Some(args.kind),
197            target: args.target,
198            board: args.board,
199            from_launch: args.from_launch,
200            from_profile: args.from_profile,
201            workspace_root: std::env::current_dir()?,
202            bringup: args.bringup,
203            force: args.force,
204        });
205    }
206
207    let name = args
208        .name
209        .as_ref()
210        .ok_or_else(|| eyre::eyre!("`nros new <name>` requires a project name"))?
211        .file_name()
212        .and_then(|n| n.to_str())
213        .ok_or_else(|| eyre::eyre!("invalid project name"))?
214        .to_string();
215
216    // Component mode (Phase 172 W.3): a reusable planned-mode library node.
217    // Platform-agnostic. Phase 172 W.3 landed Rust; Phase 219.M landed C++;
218    // Phase 223 adds the C Node pkg scaffold using the same declarative
219    // §212.L.9 shape.
220    if args.component {
221        match args.lang.as_str() {
222            "rust" | "cpp" | "c" => {}
223            other => bail!(
224                "`nros new --component --lang {other}` is not supported. Use \
225                 `rust`, `c`, or `cpp`."
226            ),
227        }
228        return scaffold_component(&ComponentScaffoldConfig {
229            name,
230            use_case: args.use_case,
231            lang: args.lang,
232            force: args.force,
233        });
234    }
235
236    // Project mode.
237    let platform = args
238        .platform
239        .ok_or_else(|| eyre::eyre!("`nros new <name>` requires `--platform <p>`"))?;
240    scaffold_package(&ScaffoldConfig {
241        name,
242        lang: args.lang,
243        platform,
244        rmw: args.rmw,
245        use_case: args.use_case,
246        force: args.force,
247    })
248}
