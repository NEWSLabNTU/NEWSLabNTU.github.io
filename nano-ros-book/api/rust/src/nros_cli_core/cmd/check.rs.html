1//! `nros check` - validate a generated nros-plan.json, a root nros.toml, or
2//! (Phase 212.F) a `<bringup>` pkg directory for pure-declarative shape.
3
4use crate::cmd::bringup::lint_bringup;
5use crate::cmd::check_workspace::check_workspace;
6use crate::cmd::emit_package_xml::{DriftStatus, check_drift};
7use crate::orchestration::{planner::check_plan_file, root_config::WorkspaceConfig};
8use clap::Args as ClapArgs;
9use eyre::Result;
10use std::path::PathBuf;
11
12#[derive(Debug, Default, ClapArgs)]
13pub struct Args {
14    /// Path to nros-plan.json, a root nros.toml (Phase 172 WP-A), or a
15    /// `<bringup>` pkg directory when `--bringup` is set (Phase 212.F).
16    #[arg(default_value = "build/nros/nros-plan.json")]
17    pub plan: PathBuf,
18
19    /// Phase 212.G.2 — also check a package directory for generated
20    /// `package.xml` drift (a generator-marked file edited by hand).
21    /// May be passed multiple times.
22    #[arg(long = "package-xml-drift")]
23    pub package_xml_drift: Vec<PathBuf>,
24
25    /// Phase 212.F — lint the `plan` argument as a `<bringup>` package
26    /// directory: reject `Cargo.toml`, `CMakeLists.txt`, `src/`, or any
27    /// nested `add_executable(`. The bringup package must be pure
28    /// declarative (see docs/design/multi-node-workspace-layout.md §4).
29    #[arg(long)]
30    pub bringup: bool,
31
32    /// Phase 212.L — walk a workspace root and run L.4 / L.8 / L.11:
33    /// `<pkg>::<Class>` enforcement on every `[[component]]` row, stray
34    /// `system.toml` next to a component pkg, and per-pkg
35    /// `.cargo/config.toml` carrying `[patch.crates-io]` (warn-only). When
36    /// the flag is passed with no value the workspace defaults to the
37    /// current directory.
38    #[arg(long, num_args = 0..=1, default_missing_value = ".", value_name = "DIR")]
39    pub workspace: Option<PathBuf>,
40}
41
42pub fn run(args: Args) -> Result<()> {
43    // Phase 212.L — `--workspace [<dir>]` runs the workspace-walk lint.
44    if let Some(ws_root) = args.workspace.as_deref() {
45        let report = check_workspace(ws_root)?;
46        for w in &report.warnings {
47            eprintln!("nros check: warning: {w}");
48        }
49        eprintln!(
50            "nros check: ok (workspace {}, {} pkg(s), {} warning(s))",
51            ws_root.display(),
52            report.pkgs_visited,
53            report.warnings.len()
54        );
55        return Ok(());
56    }
57
58    // Phase 212.F — `--bringup` switches the `plan` argument into a directory
59    // path and runs the pure-declarative lint.
60    if args.bringup {
61        lint_bringup(&args.plan)?;
62        eprintln!(
63            "nros check: ok (bringup pkg {} is pure declarative)",
64            args.plan.display()
65        );
66        return Ok(());
67    }
68
69    // Phase 212.G.2 — drift sweep over any explicitly named pkg dirs runs
70    // first so warnings surface even when the plan check exits early.
71    for pkg_dir in &args.package_xml_drift {
72        match check_drift(pkg_dir)? {
73            DriftStatus::Drift { on_disk_path } => {
74                eprintln!(
75                    "nros check: warning: {} carries the generated marker but \
76                     differs from a fresh `nros emit package-xml` — \
77                     re-run the emit to discard local edits",
78                    on_disk_path.display()
79                );
80            }
81            DriftStatus::Absent | DriftStatus::Clean | DriftStatus::HandWritten => {}
82        }
83    }
84
85    // A `.toml` argument is the workspace-root deployment config; anything
86    // else is a generated plan. `WorkspaceConfig::load` validates as it parses.
87    if args.plan.extension().is_some_and(|e| e == "toml") {
88        let cfg = WorkspaceConfig::load(&args.plan)?;
89        let systems = cfg.systems.len() + usize::from(cfg.system.is_some());
90        eprintln!(
91            "nros check: ok ({} system(s), {} deploy target(s), {})",
92            systems,
93            cfg.deploy.len(),
94            args.plan.display()
95        );
96        return Ok(());
97    }
98
99    let report = check_plan_file(&args.plan)?;
100    if report.errors == 0 {
101        for message in &report.messages {
102            eprintln!("nros check: warning: {message}");
103        }
104        eprintln!(
105            "nros check: ok ({} warning(s), {})",
106            report.warnings,
107            args.plan.display()
108        );
109    }
110    Ok(())
111}
