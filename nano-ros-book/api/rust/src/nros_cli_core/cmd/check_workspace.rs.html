1//! Phase 212.L / O — workspace-walk lints for `nros check --workspace`.
2//!
3//! Lints land here:
4//!
5//! * **L.4 — `<pkg>::<Class>` enforcement.** Every `[[component]]` row in a
6//!   bringup `system.toml` carries `pkg = "<dir>"` + `class = "<dir>::<Type>"`.
7//!   The `class` MUST be prefixed by `<pkg>::` so the codegen path and a human
8//!   reader land at the same crate.
9//!
10//! * **L.8 — `system.toml` outside bringup is forbidden.** `system.toml` is a
11//!   bringup-pkg-only file. A component pkg (carries `Cargo.toml` or
12//!   `CMakeLists.txt`) with a stray `system.toml` next to it is rejected.
13//!
14//! * **L.11 — per-pkg `.cargo/config.toml` with `[patch.crates-io]` is a
15//!   warning.** Cargo reads `[patch.crates-io]` from both `Cargo.toml` AND
16//!   `.cargo/config.toml`; when both exist the config-file shadows the
17//!   manifest. Patches must live in the workspace-root `Cargo.toml` only.
18//!
19//! * **O.2 `entry-deploy-missing` (hard error).** A component pkg whose
20//!   `Cargo.toml` declares `[package.metadata.nros.entry]` MUST also set
21//!   `deploy = "<board>"` (per Phase 212.L.2 / N.7). A missing or empty
22//!   `deploy` field rejects with the `entry-deploy-missing` diagnostic.
23//!
24//! * **O.6 `application-rtos-deploy-forbidden` (hard error).** A component
25//!   pkg whose `Cargo.toml` declares `[package.metadata.nros.application]`
26//!   MUST only name `"native"` in its `deploy = […]` allow-list. Application
27//!   pkgs are native-only by definition (Phase 212.L.2 / M-F.1); naming an
28//!   RTOS rejects with the `application-rtos-deploy-forbidden` diagnostic.
29//!
30//! * **Phase 216.D.1 `dispatch-mismatch-<framework>-<strategy>` (hard
31//!   error).** Cross-validates the `(framework, DispatchStrategy)` matrix
32//!   per the locked 216.D.1 spec. For every Entry pkg in the workspace, the
33//!   walk resolves its board (path-dep carrying
34//!   `[package.metadata.nros.board] framework = …`) and every Node pkg in
35//!   its path-dep closure (carrying `[package.metadata.nros.node]
36//!   dispatch = …`). Rejected cells:
37//!
38//!   | framework | Inline | Deferred | FromIsr |
39//!   |-----------|--------|----------|---------|
40//!   | OwnedSpin | ok     | ok       | reject  |
41//!   | Rtic      | reject | ok       | ok      |
42//!   | Embassy   | reject | ok       | reject  |
43//!
44//!   Missing `[package.metadata.nros.node] dispatch = …` defaults to
45//!   Inline (matches the `Node::DISPATCH` trait const default). Missing
46//!   board metadata defaults to OwnedSpin (POSIX/RTOS, the
47//!   `BoardEntry::run` direct-exec path). Static-only check — does NOT
48//!   read the runtime ABI symbol `__nros_node_<pkg>_dispatch_strategy`
49//!   the `nros::node!()` macro emits, because that would require linking
50//!   the Node crate.
51//!
52//! The walk is `nros check --workspace [<dir>]`. Each recursively indexed
53//! `package.xml` directory is classified as a bringup pkg (has `system.toml`,
54//! no `Cargo.toml` / `CMakeLists.txt` / `src/`) or a component pkg (has
55//! `Cargo.toml` or `CMakeLists.txt`). Other dirs are skipped.
56

63/// Result of one workspace walk. Hard-error lints bail through `Result`;
64/// warnings flow back as a list so the caller can stamp the final
65/// "ok (N warning(s))" summary.
66#[derive(Debug, Default, PartialEq, Eq)]
67pub struct WorkspaceLintReport {
68    /// Number of pkg dirs visited (any kind).
69    pub pkgs_visited: usize,
70    /// Soft warnings collected during the walk (L.11 today).
71    pub warnings: Vec<String>,
72}
73
74/// Walk `workspace_root` and run the L.4 / L.8 / L.11 lints.
75///
76/// Hard-error lints (L.4 / L.8) bail with `eyre::Error` carrying a diagnostic
77/// that names the offending dir + the rule. Warnings (L.11) accumulate in
78/// the returned report.
79pub fn check_workspace(workspace_root: &Path) -> Result<WorkspaceLintReport> {
80    let mut report = WorkspaceLintReport::default();
81
82    let index = crate::pkg_index::build_pkg_index(workspace_root)
83        .wrap_err_with(|| format!("build package index under {}", workspace_root.display()))?;
84    let mut dirs: Vec<(String, std::path::PathBuf)> = index
85        .pkgs()
86        .map(|(name, path)| (name.to_string(), path.to_path_buf()))
87        .collect();
88    let mut seen: BTreeSet<std::path::PathBuf> =
89        dirs.iter().map(|(_, path)| path.clone()).collect();
90    let entries = fs::read_dir(workspace_root)
91        .wrap_err_with(|| format!("read {}", workspace_root.display()))?;
92    for entry in entries.flatten() {
93        let path = entry.path();
94        if !path.is_dir() || !seen.insert(path.clone()) {
95            continue;
96        }
97        let Some(name) = path.file_name().and_then(|n| n.to_str()) else {
98            continue;
99        };
100        dirs.push((name.to_string(), path));
101    }
102    // Deterministic order so diagnostics + warnings are reproducible.
103    dirs.sort_by(|a, b| a.0.cmp(&b.0));
104
105    for (name, pkg_dir) in dirs {
106        // Skip dotted dirs (.git, .cargo, .claude, …) and build output dirs.
107        if name.starts_with('.') || name == "target" || name == "build" {
108            continue;
109        }
110
111        let has_cargo = pkg_dir.join("Cargo.toml").is_file();
112        let has_cmake = pkg_dir.join("CMakeLists.txt").is_file();
113        let has_system = pkg_dir.join("system.toml").is_file();
114        let has_package_xml = pkg_dir.join("package.xml").is_file();
115        let has_src = pkg_dir.join("src").is_dir();
116
117        // Phase 212.F.2 — bringup detection is `package.xml + system.toml`
118        // presence, regardless of forbidden surfaces. A dir that LOOKS like
119        // a bringup but carries Cargo.toml / CMakeLists.txt / src/ must
120        // still be classified as a bringup so the pure-declarative lint can
121        // surface the offence (rather than mis-classifying it as a
122        // component and treating the system.toml as "stray").
123        let is_bringup = has_system && has_package_xml;
124        // Component shape: has Cargo.toml or CMakeLists.txt AND is NOT a
125        // bringup (the bringup classification wins when both are present so
126        // the F.2 forbidden-file lint catches the contamination).
127        let is_component = (has_cargo || has_cmake) && !is_bringup;
128
129        if !is_component && !is_bringup {
130            continue; // Not an nros-managed pkg dir — skip.
131        }
132        report.pkgs_visited += 1;
133
134        if is_component {
135            // L.8 — component pkg with stray `system.toml`.
136            if has_system {
137                bail!(
138                    "pkg {name}: stray system.toml next to {} — `system.toml` \
139                     lives ONLY in a Path A bringup pkg (no Cargo.toml / \
140                     CMakeLists.txt / src/); move it into a sibling \
141                     <system>_bringup/ dir (see \
142                     docs/design/0024-multi-node-workspace-layout.md §4)",
143                    if has_cargo {
144                        "Cargo.toml"
145                    } else {
146                        "CMakeLists.txt"
147                    }
148                );
149            }
150            // L.11 — per-pkg `.cargo/config.toml` shadowing patch.
151            let cargo_cfg = pkg_dir.join(".cargo/config.toml");
152            if cargo_cfg.is_file() {
153                let body = fs::read_to_string(&cargo_cfg).unwrap_or_default();
154                if has_patch_crates_io(&body) {
155                    report.warnings.push(format!(
156                        "pkg {name}: .cargo/config.toml carries \
157                         [patch.crates-io] — cargo reads patches from BOTH \
158                         Cargo.toml AND .cargo/config.toml and the config \
159                         file shadows the manifest; move the block to the \
160                         workspace-root Cargo.toml (auto-managed by \
161                         `nros ws sync`)"
162                    ));
163                }
164            }
165            // O.2 + O.6 — peek inside Cargo.toml's `[package.metadata.nros]`
166            // for the Entry-pkg / Application-pkg shape lints.
167            if has_cargo {
168                lint_cargo_metadata_nros(&pkg_dir.join("Cargo.toml"), &name)?;
169                // Phase 216.D.1 — framework × DispatchStrategy matrix.
170                // Runs only on Entry pkgs (presence of
171                // `[package.metadata.nros.entry]`); resolves the board's
172                // framework via path-dep walk and every Node pkg's
173                // dispatch.
174                lint_dispatch_matrix(&pkg_dir, &name)?;
175            }
176        }
177
178        if is_bringup {
179            // Phase 212.F.2 — run the pure-declarative surface lint on
180            // every bringup pkg discovered by the workspace walk. Catches
181            // Cargo.toml / CMakeLists.txt / src/ / include/ / lib/ +
182            // nested `add_executable(`. The L.4 `<pkg>::<Class>` lint runs
183            // unconditionally below; `<exec_depend>` drift is intentionally
184            // NOT in the workspace-walk pass (only on `--bringup <dir>`)
185            // because pre-spec fixtures stamp shells without the
186            // `<exec_depend>` rows and we don't want to regress them.
187            let _ = has_src;
188            crate::cmd::bringup::lint_bringup_forbidden_surfaces(&pkg_dir)?;
189            // L.4 — class prefix matches pkg.
190            lint_class_pkg_prefix(&pkg_dir, &name)?;
191        }
192    }
193
194    Ok(report)
195}

197/// L.4 helper. Read `<bringup>/system.toml`, verify each `[[component]]`
198/// row's `class` is `<pkg>::<Type>`-shaped. Public so `lint_bringup` can
199/// reuse it on the `--bringup <dir>` flow.
200pub fn lint_class_pkg_prefix(bringup_dir: &Path, bringup_pkg_name: &str) -> Result<()> {
201    let system_toml = bringup_dir.join("system.toml");
202    if !system_toml.is_file() {
203        return Ok(());
204    }
205    let raw = fs::read_to_string(&system_toml)
206        .wrap_err_with(|| format!("read {}", system_toml.display()))?;
207    let parsed: SystemToml =
208        toml::from_str(&raw).wrap_err_with(|| format!("parse {}", system_toml.display()))?;
209    let mut bad: Vec<String> = Vec::new();
210    for c in &parsed.components {
211        let prefix = format!("{}::", c.pkg);
212        if !c.class.starts_with(&prefix) {
213            bad.push(format!(
214                "[[component]] name=\"{}\" pkg=\"{}\" class=\"{}\" — class \
215                 must start with \"{}\"",
216                c.name, c.pkg, c.class, prefix
217            ));
218        }
219    }
220    if !bad.is_empty() {
221        bail!(
222            "bringup pkg {bringup_pkg_name}: system.toml component class \
223             mismatch — {}. The `class` field in a `[[component]]` row MUST \
224             be `<pkg>::<Type>` so codegen and humans land at the same crate.",
225            bad.join("; ")
226        );
227    }
228    Ok(())
229}

231/// Phase 212.O.2 + O.6 — peek inside a component pkg's `Cargo.toml`
232/// `[package.metadata.nros]` table and run the Entry / Application shape
233/// lints.
234///
235/// * **O.2 `entry-deploy-missing`** — `[package.metadata.nros.entry]` MUST
236///   carry `deploy = "<board>"` (non-empty string). Phase 212.L.2 / N.7.
237/// * **O.6 `application-rtos-deploy-forbidden`** — every entry in
238///   `[package.metadata.nros.application].deploy` MUST be `"native"`. Phase
239///   212.L.2 / M-F.1 — Application pkgs are native-only orchestration roots.
240///
241/// Both lints bail with an eyre error whose diagnostic id is embedded in the
242/// message body (`entry-deploy-missing` / `application-rtos-deploy-forbidden`).
243/// Diagnostic IDs are part of the stable contract used by `nros check`
244/// integration tests + downstream tooling.
245fn lint_cargo_metadata_nros(cargo_toml_path: &Path, pkg_name: &str) -> Result<()> {
246    let raw = match fs::read_to_string(cargo_toml_path) {
247        Ok(s) => s,
248        // Unreadable Cargo.toml is not this lint's problem — bail out silently
249        // (the rest of cargo / nros plan will surface a cleaner error).
250        Err(_) => return Ok(()),
251    };
252    let value: toml::Value = match toml::from_str(&raw) {
253        Ok(v) => v,
254        Err(_) => return Ok(()), // malformed manifest — not this lint's job
255    };
256
257    let nros = value
258        .get("package")
259        .and_then(|p| p.get("metadata"))
260        .and_then(|m| m.get("nros"));
261    let Some(nros) = nros else {
262        return Ok(());
263    };
264
265    // ---- O.2 entry-deploy-missing -----------------------------------------
266    if let Some(entry) = nros.get("entry") {
267        // `deploy` must be present, a string, and non-empty.
268        let deploy_ok = entry
269            .get("deploy")
270            .and_then(|v| v.as_str())
271            .is_some_and(|s| !s.trim().is_empty());
272        if !deploy_ok {
273            bail!(
274                "{pkg_name}: [package.metadata.nros.entry] missing or empty \
275                 'deploy' field — Entry pkg must name a board target (e.g. \
276                 deploy = \"native\" or deploy = \"qemu-mps2-an385-freertos\") \
277                 [diagnostic: entry-deploy-missing]"
278            );
279        }
280    }
281
282    // ---- O.6 application-rtos-deploy-forbidden ----------------------------
283    if let Some(app) = nros.get("application") {
284        // `deploy` is optional; when present it must be a list of strings,
285        // and every entry must be exactly "native".
286        if let Some(deploy) = app.get("deploy") {
287            let Some(arr) = deploy.as_array() else {
288                // Wrong type — leave to the strict schema validator; this
289                // lint only flags the RTOS-name policy.
290                return Ok(());
291            };
292            for entry in arr {
293                let Some(s) = entry.as_str() else {
294                    continue;
295                };
296                if s != "native" {
297                    bail!(
298                        "{pkg_name}: Application pkg may not deploy to RTOS \
299                         target '{s}' (Applications are native-only; use \
300                         Component pkg + Entry pkg for RTOS deployment) \
301                         [diagnostic: application-rtos-deploy-forbidden]"
302                    );
303                }
304            }
305        }
306    }
307
308    Ok(())
309}

311/// Cheap substring scan for `[patch.crates-io]` in a cargo config body.
312/// Avoids a full TOML parse so malformed user configs still flag.
313fn has_patch_crates_io(body: &str) -> bool {
314    for line in body.lines() {
315        let t = line.trim();
316        if t.starts_with('#') {
317            continue;
318        }
319        // Accept `[patch.crates-io]` exactly. Be tolerant of trailing comments
320        // and whitespace; reject `[patch.crates-io.foo]` (that's a dependency
321        // override entry, not the table header — but in practice users hit
322        // both with the same shadowing risk, so flag anything starting with
323        // the patch.crates-io path).
324        if t.starts_with("[patch.crates-io]") || t.starts_with("[patch.crates-io.") {
325            return true;
326        }
327    }
328    false
329}

332// Phase 216.D.1 — framework × DispatchStrategy matrix
333// ---------------------------------------------------------------------------
334
335/// Locked 216.D.1 framework axis. Boards without
336/// `[package.metadata.nros.board] framework = …` default to `OwnedSpin`
337/// (the POSIX / RTOS `BoardEntry::run` direct-exec path — no framework
338/// runtime owns the spin loop).
339#[derive(Debug, Clone, Copy, PartialEq, Eq)]
340enum Framework {
341    OwnedSpin,
342    Rtic,
343    Embassy,
344}
345
346impl Framework {
347    fn label(self) -> &'static str {
348        match self {
349            Framework::OwnedSpin => "owned-spin",
350            Framework::Rtic => "rtic",
351            Framework::Embassy => "embassy",
352        }
353    }
354}
355
356/// Locked 216.D.1 dispatch axis. Mirrors `nros_platform::DispatchStrategy`
357/// but lives here so `nros check` stays a static tool (no link against the
358/// runtime crate). Node pkgs missing the metadata key default to `Inline`
359/// to match the `Node::DISPATCH` trait const default.
360#[derive(Debug, Clone, Copy, PartialEq, Eq)]
361enum DispatchStrategy {
362    Inline,
363    Deferred,
364    FromIsr,
365}
366
367impl DispatchStrategy {
368    fn label(self) -> &'static str {
369        match self {
370            DispatchStrategy::Inline => "inline",
371            DispatchStrategy::Deferred => "deferred",
372            DispatchStrategy::FromIsr => "from-isr",
373        }
374    }
375}
376
377/// Apply the locked 216.D.1 matrix. Returns `Some(suggested_fix)` when the
378/// pair is rejected; `None` when the pair is allowed.
379fn matrix_reject_reason(fw: Framework, ds: DispatchStrategy) -> Option<&'static str> {
380    match (fw, ds) {
381        // OwnedSpin (POSIX / RTOS): Inline + Deferred ok; FromIsr rejected
382        // (no ISR fast-path on a hosted RTOS / POSIX runtime).
383        (Framework::OwnedSpin, DispatchStrategy::FromIsr) => Some(
384            "FromIsr is not supported on POSIX / RTOS (owned-spin) boards; \
385             change DISPATCH to Inline or Deferred, or deploy on a bare-metal \
386             framework board (rtic) that wires the ISR fast-path",
387        ),
388        // RTIC: Inline rejected (RTIC requires a Deferred hand-off to an
389        // `rtic::Mutex`-guarded task — running a callback inline on the
390        // spin task breaks the framework's priority model); Deferred ok;
391        // FromIsr ok (future — ISR-driven cell).
392        (Framework::Rtic, DispatchStrategy::Inline) => Some(
393            "RTIC requires Deferred dispatch (callbacks hand off to a \
394             framework-owned `#[task]` via an `rtic::Mutex`-guarded SPSC \
395             ring); set [package.metadata.nros.node] dispatch = \"deferred\" \
396             on the Node pkg",
397        ),
398        // Embassy: Inline rejected (same reason — Embassy needs a Spawner
399        // hand-off); Deferred ok; FromIsr rejected (Embassy's executor
400        // runs in thread mode — no ISR fast-path).
401        (Framework::Embassy, DispatchStrategy::Inline) => Some(
402            "Embassy requires Deferred dispatch (callbacks hand off to a \
403             framework-owned `async` task via the `embassy_executor::Spawner` \
404             escape); set [package.metadata.nros.node] dispatch = \"deferred\" \
405             on the Node pkg",
406        ),
407        (Framework::Embassy, DispatchStrategy::FromIsr) => Some(
408            "FromIsr is not supported on Embassy (the executor runs in \
409             thread mode — no ISR fast-path); change DISPATCH to Deferred, \
410             or deploy on an RTIC board",
411        ),
412        // All other (fw, ds) cells are allowed.
413        _ => None,
414    }
415}

417/// Parse `[package.metadata.nros.board] framework = "<rtic|embassy>"`.
418/// Missing table or unknown framework string → `Framework::OwnedSpin`
419/// (the matrix's default cell for POSIX / RTOS boards w/o framework
420/// metadata).
421fn read_board_framework(board_cargo_toml: &Path) -> Framework {
422    let Ok(raw) = fs::read_to_string(board_cargo_toml) else {
423        return Framework::OwnedSpin;
424    };
425    let Ok(value) = toml::from_str::<toml::Value>(&raw) else {
426        return Framework::OwnedSpin;
427    };
428    let fw = value
429        .get("package")
430        .and_then(|p| p.get("metadata"))
431        .and_then(|m| m.get("nros"))
432        .and_then(|n| n.get("board"))
433        .and_then(|b| b.get("framework"))
434        .and_then(|v| v.as_str());
435    match fw {
436        Some("rtic") => Framework::Rtic,
437        Some("embassy") => Framework::Embassy,
438        _ => Framework::OwnedSpin,
439    }
440}

442/// Parse `[package.metadata.nros.node] dispatch = "<inline|deferred|from_isr>"`.
443/// Missing key → `Inline` (matches `Node::DISPATCH` default). Returns
444/// `Err` on an unknown string so the lint surfaces a typo instead of
445/// silently defaulting.
446fn read_node_dispatch_strategy(node_cargo_toml: &Path) -> Result<Option<DispatchStrategy>> {
447    let Ok(raw) = fs::read_to_string(node_cargo_toml) else {
448        return Ok(None);
449    };
450    let Ok(value) = toml::from_str::<toml::Value>(&raw) else {
451        return Ok(None);
452    };
453    // Only treat this pkg as a Node pkg if it carries
454    // `[package.metadata.nros.node]`.
455    let node_tbl = value
456        .get("package")
457        .and_then(|p| p.get("metadata"))
458        .and_then(|m| m.get("nros"))
459        .and_then(|n| n.get("node"));
460    let Some(node_tbl) = node_tbl else {
461        return Ok(None);
462    };
463    let Some(ds_str) = node_tbl.get("dispatch").and_then(|v| v.as_str()) else {
464        // Node pkg w/o explicit dispatch → mirror the trait const default.
465        return Ok(Some(DispatchStrategy::Inline));
466    };
467    match ds_str {
468        "inline" => Ok(Some(DispatchStrategy::Inline)),
469        "deferred" => Ok(Some(DispatchStrategy::Deferred)),
470        // Accept both kebab + snake for the ISR variant; mirror the
471        // platform enum's Rust name (`FromIsr`).
472        "from_isr" | "from-isr" => Ok(Some(DispatchStrategy::FromIsr)),
473        other => bail!(
474            "{}: [package.metadata.nros.node] dispatch = \"{}\" — unknown \
475             dispatch strategy (expected one of: \"inline\", \"deferred\", \
476             \"from_isr\")",
477            node_cargo_toml.display(),
478            other
479        ),
480    }
481}

483/// Walk an Entry pkg's `[dependencies]` + `[dev-dependencies]` tables
484/// and collect every `path = "<rel>"` dep target. Returned paths are
485/// resolved against the Entry pkg's dir + canonicalised. Only path-deps
486/// are considered (registry / git deps have no in-tree manifest to read).
487fn collect_path_deps(entry_cargo_toml: &Path, entry_dir: &Path) -> Vec<std::path::PathBuf> {
488    let mut out = Vec::new();
489    let Ok(raw) = fs::read_to_string(entry_cargo_toml) else {
490        return out;
491    };
492    let Ok(value) = toml::from_str::<toml::Value>(&raw) else {
493        return out;
494    };
495    for tbl_name in ["dependencies", "dev-dependencies", "build-dependencies"] {
496        let Some(tbl) = value.get(tbl_name).and_then(|v| v.as_table()) else {
497            continue;
498        };
499        for (_dep_name, dep_spec) in tbl {
500            let Some(dep_tbl) = dep_spec.as_table() else {
501                continue;
502            };
503            let Some(rel) = dep_tbl.get("path").and_then(|v| v.as_str()) else {
504                continue;
505            };
506            let abs = entry_dir.join(rel);
507            // `canonicalize` resolves `..` segments and follows symlinks.
508            // Fall back to the joined path if canonicalisation fails (e.g.
509            // dep dir doesn't exist — a separate lint will catch that).
510            let resolved = fs::canonicalize(&abs).unwrap_or(abs);
511            out.push(resolved);
512        }
513    }
514    out
515}

517/// Phase 216.D.1 — framework × DispatchStrategy matrix entry point.
518///
519/// Triggered for every component pkg whose `Cargo.toml` carries
520/// `[package.metadata.nros.entry]` (= an Entry pkg). Walks the Entry pkg's
521/// path-deps, classifies each as `board` (carries `[package.metadata.
522/// nros.board]`) or `node` (carries `[package.metadata.nros.node]`), then
523/// runs the cross-validation matrix.
524///
525/// Bail conditions:
526/// * Two boards in the path-dep set carry conflicting `framework` values
527///   (operator error; the Entry pkg is multi-targeted in a way the matrix
528///   can't reason about).
529/// * A `(framework, dispatch)` pair lands in a rejected cell — diagnostic
530///   ID `dispatch-mismatch-<framework>-<strategy>`.
531///
532/// Silent on non-Entry pkgs and on Entry pkgs with no Node-pkg path-deps
533/// (e.g. a board-bringup Entry that defers all callbacks to the framework
534/// runtime without a user Node pkg).
535fn lint_dispatch_matrix(entry_dir: &Path, entry_pkg_name: &str) -> Result<()> {
536    let cargo_toml = entry_dir.join("Cargo.toml");
537    let Ok(raw) = fs::read_to_string(&cargo_toml) else {
538        return Ok(());
539    };
540    let Ok(value) = toml::from_str::<toml::Value>(&raw) else {
541        return Ok(());
542    };
543    // Only run on Entry pkgs.
544    let is_entry = value
545        .get("package")
546        .and_then(|p| p.get("metadata"))
547        .and_then(|m| m.get("nros"))
548        .and_then(|n| n.get("entry"))
549        .is_some();
550    if !is_entry {
551        return Ok(());
552    }
553
554    // Walk path-deps and bucket into boards / nodes.
555    let mut framework: Option<Framework> = None;
556    let mut framework_source: Option<std::path::PathBuf> = None;
557    // (node_pkg_dir, dispatch_strategy).
558    let mut nodes: Vec<(std::path::PathBuf, DispatchStrategy)> = Vec::new();
559
560    for dep_dir in collect_path_deps(&cargo_toml, entry_dir) {
561        let dep_cargo = dep_dir.join("Cargo.toml");
562        if !dep_cargo.is_file() {
563            continue;
564        }
565        let dep_raw = match fs::read_to_string(&dep_cargo) {
566            Ok(s) => s,
567            Err(_) => continue,
568        };
569        let dep_value: toml::Value = match toml::from_str(&dep_raw) {
570            Ok(v) => v,
571            Err(_) => continue,
572        };
573        let nros_meta = dep_value
574            .get("package")
575            .and_then(|p| p.get("metadata"))
576            .and_then(|m| m.get("nros"));
577        let Some(nros_meta) = nros_meta else {
578            continue;
579        };
580
581        // Board classification.
582        if nros_meta.get("board").is_some() {
583            let fw = read_board_framework(&dep_cargo);
584            if let Some(prev) = framework {
585                if prev != fw {
586                    bail!(
587                        "{entry_pkg_name}: conflicting board frameworks in \
588                         path-dep set — {} declared framework='{}', {} declared \
589                         framework='{}'. An Entry pkg may depend on at most one \
590                         board crate carrying `[package.metadata.nros.board]`. \
591                         [diagnostic: dispatch-mismatch-conflicting-boards]",
592                        framework_source
593                            .as_ref()
594                            .map(|p| p.display().to_string())
595                            .unwrap_or_else(|| "<unknown>".into()),
596                        prev.label(),
597                        dep_cargo.display(),
598                        fw.label(),
599                    );
600                }
601            } else {
602                framework = Some(fw);
603                framework_source = Some(dep_cargo.clone());
604            }
605        }
606
607        // Node classification (a pkg MAY be both a board + a node, but in
608        // practice no in-tree crate is — the two tables are mutually
609        // exclusive by convention).
610        if let Some(ds) = read_node_dispatch_strategy(&dep_cargo)? {
611            nodes.push((dep_dir, ds));
612        }
613    }
614
615    // No board metadata → default to OwnedSpin (POSIX / RTOS / native).
616    let framework = framework.unwrap_or(Framework::OwnedSpin);
617
618    // Apply the matrix per (framework, node).
619    for (node_dir, ds) in nodes {
620        if let Some(fix) = matrix_reject_reason(framework, ds) {
621            let node_label = node_dir
622                .file_name()
623                .and_then(|n| n.to_str())
624                .unwrap_or("<unknown>");
625            bail!(
626                "{entry_pkg_name}: framework × dispatch matrix rejects \
627                 ({fw} × {ds}) — Node pkg '{node_label}' declares \
628                 dispatch=\"{ds}\" but Entry pkg's board carries \
629                 framework=\"{fw}\". Fix: {fix} \
630                 [diagnostic: dispatch-mismatch-{fw}-{ds}]",
631                fw = framework.label(),
632                ds = ds.label(),
633                node_label = node_label,
634                fix = fix,
635            );
636        }
637    }
638
639    Ok(())
640}
