1//! Phase 212.L — workspace-walk lints for `nros check --workspace`.
2//!
3//! Three lints land here:
4//!
5//! * **L.4 — `<pkg>::<Class>` enforcement.** Every `[[component]]` row in a
6//!   bringup `system.toml` carries `pkg = "<dir>"` + `class = "<dir>::<Type>"`.
7//!   The `class` MUST be prefixed by `<pkg>::` so the codegen path and a human
8//!   reader land at the same crate.
9//!
10//! * **L.8 — `system.toml` outside bringup is forbidden.** `system.toml` is a
11//!   bringup-pkg-only file. A component pkg (carries `Cargo.toml` or
12//!   `CMakeLists.txt`) with a stray `system.toml` next to it is rejected.
13//!
14//! * **L.11 — per-pkg `.cargo/config.toml` with `[patch.crates-io]` is a
15//!   warning.** Cargo reads `[patch.crates-io]` from both `Cargo.toml` AND
16//!   `.cargo/config.toml`; when both exist the config-file shadows the
17//!   manifest. Patches must live in the workspace-root `Cargo.toml` only.
18//!
19//! The walk is `nros check --workspace [<dir>]`. Each immediate child of the
20//! workspace root is classified as a bringup pkg (has `system.toml`, no
21//! `Cargo.toml` / `CMakeLists.txt` / `src/`) or a component pkg (has
22//! `Cargo.toml` or `CMakeLists.txt`). Other dirs are skipped.
23

26use eyre::{Result, WrapErr, bail};

28use crate::orchestration::cargo_metadata_schema::SystemToml;

30/// Result of one workspace walk. Hard-error lints bail through `Result`;
31/// warnings flow back as a list so the caller can stamp the final
32/// "ok (N warning(s))" summary.
33#[derive(Debug, Default, PartialEq, Eq)]
34pub struct WorkspaceLintReport {
35    /// Number of pkg dirs visited (any kind).
36    pub pkgs_visited: usize,
37    /// Soft warnings collected during the walk (L.11 today).
38    pub warnings: Vec<String>,
39}
40
41/// Walk `workspace_root` and run the L.4 / L.8 / L.11 lints.
42///
43/// Hard-error lints (L.4 / L.8) bail with `eyre::Error` carrying a diagnostic
44/// that names the offending dir + the rule. Warnings (L.11) accumulate in
45/// the returned report.
46pub fn check_workspace(workspace_root: &Path) -> Result<WorkspaceLintReport> {
47    let mut report = WorkspaceLintReport::default();
48
49    let entries = fs::read_dir(workspace_root)
50        .wrap_err_with(|| format!("read {}", workspace_root.display()))?;
51    let mut dirs: Vec<std::path::PathBuf> = entries
52        .flatten()
53        .map(|e| e.path())
54        .filter(|p| p.is_dir())
55        .collect();
56    // Deterministic order so diagnostics + warnings are reproducible.
57    dirs.sort();
58
59    for pkg_dir in dirs {
60        let name = match pkg_dir.file_name().and_then(|n| n.to_str()) {
61            Some(n) => n.to_string(),
62            None => continue,
63        };
64        // Skip dotted dirs (.git, .cargo, .claude, …) and build output dirs.
65        if name.starts_with('.') || name == "target" || name == "build" {
66            continue;
67        }
68
69        let has_cargo = pkg_dir.join("Cargo.toml").is_file();
70        let has_cmake = pkg_dir.join("CMakeLists.txt").is_file();
71        let has_system = pkg_dir.join("system.toml").is_file();
72        let has_src = pkg_dir.join("src").is_dir();
73
74        // Component pkg shape: Cargo.toml or CMakeLists.txt sits at the root.
75        // Bringup pkg shape: only system.toml + package.xml (no source).
76        let is_component = has_cargo || has_cmake;
77        let is_bringup = has_system && !is_component && !has_src;
78
79        if !is_component && !is_bringup {
80            continue; // Not an nros-managed pkg dir — skip.
81        }
82        report.pkgs_visited += 1;
83
84        if is_component {
85            // L.8 — component pkg with stray `system.toml`.
86            if has_system {
87                bail!(
88                    "pkg {name}: stray system.toml next to {} — `system.toml` \
89                     lives ONLY in a Path A bringup pkg (no Cargo.toml / \
90                     CMakeLists.txt / src/); move it into a sibling \
91                     <system>_bringup/ dir (see \
92                     docs/design/multi-node-workspace-layout.md §4)",
93                    if has_cargo {
94                        "Cargo.toml"
95                    } else {
96                        "CMakeLists.txt"
97                    }
98                );
99            }
100            // L.11 — per-pkg `.cargo/config.toml` shadowing patch.
101            let cargo_cfg = pkg_dir.join(".cargo/config.toml");
102            if cargo_cfg.is_file() {
103                let body = fs::read_to_string(&cargo_cfg).unwrap_or_default();
104                if has_patch_crates_io(&body) {
105                    report.warnings.push(format!(
106                        "pkg {name}: .cargo/config.toml carries \
107                         [patch.crates-io] — cargo reads patches from BOTH \
108                         Cargo.toml AND .cargo/config.toml and the config \
109                         file shadows the manifest; move the block to the \
110                         workspace-root Cargo.toml (auto-managed by \
111                         `nros ws sync`)"
112                    ));
113                }
114            }
115        }
116
117        if is_bringup {
118            // L.4 — class prefix matches pkg.
119            lint_class_pkg_prefix(&pkg_dir, &name)?;
120        }
121    }
122
123    Ok(report)
124}

126/// L.4 helper. Read `<bringup>/system.toml`, verify each `[[component]]`
127/// row's `class` is `<pkg>::<Type>`-shaped. Public so `lint_bringup` can
128/// reuse it on the `--bringup <dir>` flow.
129pub fn lint_class_pkg_prefix(bringup_dir: &Path, bringup_pkg_name: &str) -> Result<()> {
130    let system_toml = bringup_dir.join("system.toml");
131    if !system_toml.is_file() {
132        return Ok(());
133    }
134    let raw = fs::read_to_string(&system_toml)
135        .wrap_err_with(|| format!("read {}", system_toml.display()))?;
136    let parsed: SystemToml =
137        toml::from_str(&raw).wrap_err_with(|| format!("parse {}", system_toml.display()))?;
138    let mut bad: Vec<String> = Vec::new();
139    for c in &parsed.components {
140        let prefix = format!("{}::", c.pkg);
141        if !c.class.starts_with(&prefix) {
142            bad.push(format!(
143                "[[component]] name=\"{}\" pkg=\"{}\" class=\"{}\" — class \
144                 must start with \"{}\"",
145                c.name, c.pkg, c.class, prefix
146            ));
147        }
148    }
149    if !bad.is_empty() {
150        bail!(
151            "bringup pkg {bringup_pkg_name}: system.toml component class \
152             mismatch — {}. The `class` field in a `[[component]]` row MUST \
153             be `<pkg>::<Type>` so codegen and humans land at the same crate.",
154            bad.join("; ")
155        );
156    }
157    Ok(())
158}

160/// Cheap substring scan for `[patch.crates-io]` in a cargo config body.
161/// Avoids a full TOML parse so malformed user configs still flag.
162fn has_patch_crates_io(body: &str) -> bool {
163    for line in body.lines() {
164        let t = line.trim();
165        if t.starts_with('#') {
166            continue;
167        }
168        // Accept `[patch.crates-io]` exactly. Be tolerant of trailing comments
169        // and whitespace; reject `[patch.crates-io.foo]` (that's a dependency
170        // override entry, not the table header — but in practice users hit
171        // both with the same shadowing risk, so flag anything starting with
172        // the patch.crates-io path).
173        if t.starts_with("[patch.crates-io]") || t.starts_with("[patch.crates-io.") {
174            return true;
175        }
176    }
177    false
178}

187    use std::path::PathBuf;

190    fn temp_root(tag: &str) -> PathBuf {
191        let stamp = SystemTime::now()
192            .duration_since(UNIX_EPOCH)
193            .unwrap()
194            .as_nanos();
195        let dir = std::env::temp_dir().join(format!(
196            "nros-check-ws-{tag}-{}-{stamp}",
197            std::process::id()
198        ));
199        let _ = fs::remove_dir_all(&dir);
200        fs::create_dir_all(&dir).unwrap();
201        dir
202    }
203
204    fn write_component_pkg(dir: &Path, name: &str) {
205        fs::create_dir_all(dir.join("src")).unwrap();
206        fs::write(
207            dir.join("Cargo.toml"),
208            format!("[package]\nname=\"{name}\"\nversion=\"0.1.0\"\n"),
209        )
210        .unwrap();
211        fs::write(dir.join("src/lib.rs"), "// stub\n").unwrap();
212    }
213
214    fn write_bringup_with_components(dir: &Path, components: &[(&str, &str, &str)]) {
215        fs::create_dir_all(dir).unwrap();
216        fs::write(
217            dir.join("package.xml"),
218            "<?xml version=\"1.0\"?><package format=\"3\">\
219             <name>demo_bringup</name><version>0.1.0</version></package>",
220        )
221        .unwrap();
222        let mut s = String::from("[system]\nname = \"demo\"\nrmw = \"zenoh\"\ndomain_id = 0\n");
223        for (pkg, class, cname) in components {
224            s.push_str(&format!(
225                "\n[[component]]\npkg = \"{pkg}\"\nclass = \"{class}\"\nname = \"{cname}\"\n"
226            ));
227        }
228        fs::write(dir.join("system.toml"), s).unwrap();
229    }
