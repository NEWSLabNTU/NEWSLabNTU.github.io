1//! `nros plan` - generate host-side orchestration plan.
2//!
3//! Phase 212.L.6: the positional `<launch_file>` now also accepts a
4//! **package directory** (Cargo / CMake pkg, or a bringup pkg). When a
5//! directory is passed we route through
6//! [`orchestration::launch_synth::resolve_launch`] which either picks a
7//! convention-named launch file under `<dir>/launch/` or synthesises a
8//! one-node `<launch>` body in-memory for self-bringup pkgs.
9
10use crate::orchestration::{
11    launch_synth::{is_self_entry_pkg, resolve_launch},
12    planner::{PlanOptions, plan_system},
13};
14use clap::Args as ClapArgs;
15use eyre::Result;
16use std::path::PathBuf;
17
18#[derive(Debug, ClapArgs)]
19pub struct Args {
20    /// System package name used for build/<system_pkg>/nros output.
21    /// When omitted, derived from the `<launch_file>` directory's
22    /// pkg name (Phase 212.L.7 self-bringup shape).
23    pub system_pkg: String,
24
25    /// ROS 2 launch file to parse, **or** a package directory to resolve
26    /// via the Phase 212.L.6 multi-launch policy (pkg-named →
27    /// system.launch.xml → single-file → synth for self-bringup pkgs).
28    /// When omitted in a Phase 212.L.7 single-arg invocation
29    /// (`nros plan <pkg-dir>`), defaults to the `<system_pkg>` argument
30    /// (treated as a directory path then).
31    #[arg(default_value = "")]
32    pub launch_file: PathBuf,
33
34    /// Precomputed play_launch record.json to use instead of parsing launch_file
35    #[arg(long)]
36    pub record: Option<PathBuf>,
37
38    /// Phase 212.L.6 — when `<launch_file>` is a directory, prefer
39    /// `<dir>/launch/<file>` (or cwd-relative / absolute as fallback).
40    #[arg(long = "file")]
41    pub file: Option<String>,
42
43    /// Phase 212.L.6 — disambiguates the synthesised `<node exec="…">`
44    /// when the package declares multiple `[[bin]]` / `add_executable`
45    /// targets.
46    #[arg(long = "exec")]
47    pub exec: Option<String>,
48
49    /// Workspace root containing colcon-like src/* packages
50    #[arg(long)]
51    pub workspace: Option<PathBuf>,
52
53    /// Output root for orchestration artifacts
54    #[arg(long)]
55    pub out_dir: Option<PathBuf>,
56
57    /// Existing source metadata JSON artifact
58    #[arg(long = "metadata")]
59    pub metadata: Vec<PathBuf>,
60
61    /// ROS launch manifest YAML artifact
62    #[arg(long = "manifest")]
63    pub manifests: Vec<PathBuf>,
64
65    /// nano-ros deployment overlay TOML
66    #[arg(long = "nros-toml")]
67    pub nros_toml: Vec<PathBuf>,
68
69    /// Launch arguments forwarded as name:=value or name=value
70    #[arg(trailing_var_arg = true, allow_hyphen_values = true)]
71    pub launch_args: Vec<String>,
72}
73
74pub fn run(args: Args) -> Result<()> {
75    let workspace_root = args.workspace.unwrap_or(std::env::current_dir()?);
76
77    // Phase 212.L.7 — single-arg self-bringup shape. When the user
78    // passes only `<pkg-dir>` (no positional launch_file), `clap`
79    // sees `launch_file` as the empty path; default it to the
80    // `system_pkg` argument treated as a directory path. The
81    // launch_synth resolver handles the rest.
82    let launch_input_path = if args.launch_file.as_os_str().is_empty() {
83        PathBuf::from(&args.system_pkg)
84    } else {
85        args.launch_file.clone()
86    };
87
88    // Derive the system_pkg from the dir name when the user passed a
89    // dir as `<system_pkg>` (single-arg self-bringup shape).
90    let system_pkg =
91        if args.launch_file.as_os_str().is_empty() && PathBuf::from(&args.system_pkg).is_dir() {
92            PathBuf::from(&args.system_pkg)
93                .file_name()
94                .and_then(|s| s.to_str())
95                .unwrap_or(&args.system_pkg)
96                .to_string()
97        } else {
98            args.system_pkg.clone()
99        };
100
101    let out_root = args
102        .out_dir
103        .unwrap_or_else(|| workspace_root.join("build").join(&system_pkg).join("nros"));
104
105    // Phase 212.L.6: the positional `launch_file` may be either an
106    // existing file (legacy path) or a package directory. Resolve to a
107    // real on-disk path that the external `play_launch_parser` binary
108    // can consume — synthesised XML is written to a temp file whose
109    // lifetime is tied to `_materialised` and removed when planning
110    // returns.
111    //
112    // Phase 212.L.7: when the pkg dir carries BOTH
113    // `[package.metadata.nros.node]` AND `[package.metadata.nros.entry]`
114    // it eats its own Entry role (single-pkg dev loop). The L.6 resolver
115    // covers both branches uniformly — real launch file under
116    // `<dir>/launch/` first, synthesised `<launch><node …/>` body when
117    // absent. We just log when L.7 self-entry mode kicks in so users
118    // see what shape the CLI picked.
119    let (resolved_path, _materialised) = if launch_input_path.is_dir() {
120        if is_self_entry_pkg(&launch_input_path) {
121            eprintln!(
122                "nros plan: {} is a self-entry pkg \
123                 ([package.metadata.nros.node] + [package.metadata.nros.entry]); \
124                 resolving launch via L.6 (real launch.xml or synth)",
125                launch_input_path.display()
126            );
127        }
128        let input = resolve_launch(
129            &launch_input_path,
130            args.file.as_deref(),
131            args.exec.as_deref(),
132        )?;
133        let materialised = input.materialise()?;
134        (materialised.path.clone(), Some(materialised))
135    } else {
136        (launch_input_path.clone(), None)
137    };
138
139    let output = plan_system(PlanOptions {
140        system_pkg,
141        workspace_root,
142        launch_file: resolved_path,
143        record_file: args.record,
144        out_root,
145        metadata_files: args.metadata,
146        manifest_files: args.manifests,
147        nros_toml_files: args.nros_toml,
148        launch_args: args.launch_args,
149    })?;
150
151    // `_materialised` keeps the synthesised temp file alive through
152    // `plan_system`; drop it now (RAII removes the temp file).
153    drop(_materialised);
154
155    eprintln!(
156        "nros plan: wrote {} and {}",
157        output.record_path.display(),
158        output.plan_path.display()
159    );
160    Ok(())
161}
