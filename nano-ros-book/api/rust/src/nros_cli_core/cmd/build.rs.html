1//! `nros build` — Phase 111.A.9 / Phase 172 flip.
2//!
3//! Two paths, auto-detected from the project root (cwd or `--project`):
4//!
5//! 1. **Deploy dispatch** (Phase 172 WP-A): when the root holds a loadable
6//!    workspace `nros.toml`, `nros build <name>` aliases `nros deploy <name>`
7//!    and bare `nros build` builds `[workspace].default`. The deploy runner
8//!    generates + builds the entry lib and runs the target's command steps.
9//!
10//! 2. **Project-flavor delegation**: otherwise detect the flavor and delegate
11//!    (Zephyr west, CMake, or Cargo) — precedence (highest first):
12//!
13//!      1. `prj.conf` present → Zephyr → `west build`
14//!      2. `CMakeLists.txt` present + no `Cargo.toml` → `cmake -B build && cmake --build build`
15//!      3. `Cargo.toml` present → `cargo build`
16//!
17//!    Mixed projects (Cargo.toml AND CMakeLists.txt) — common when a Rust
18//!    crate produces a `staticlib` consumed by C/C++ — go through the
19//!    cmake path. Heuristic: if `[lib].crate-type` in Cargo.toml contains
20//!    `staticlib` AND CMakeLists.txt exists, prefer cmake.
21//!
22//! The legacy `--launch` one-shot and `--system-plan` pre-planned-build paths
23//! were retired in the Phase 172 flip; building a system from a plan is now an
24//! orchestration-library call (`orchestration::build::build_generated_package`)
25//! driven by `nros deploy`.
26

41#[derive(Debug, ClapArgs)]
42pub struct Args {
43    /// Path to the project root (default: cwd)
44    #[arg(long)]
45    pub project: Option<PathBuf>,
46
47    /// nano-ros workspace root, forwarded to `nros deploy` for generated
48    /// path-dependency / vendor-pin resolution. Also honored via the
49    /// `NROS_WORKSPACE` environment variable inside the deploy runner.
50    #[arg(long)]
51    pub nano_ros_workspace: Option<PathBuf>,
52
53    /// Deploy target from the root nros.toml (Phase 172 WP-A): `nros build
54    /// <name>` is an alias for `nros deploy <name>`; bare `nros build` (in a
55    /// workspace root) builds `[workspace].default`.
56    pub deploy_name: Option<String>,
57
58    /// Trailing arguments forwarded verbatim to the underlying tool
59    #[arg(trailing_var_arg = true, allow_hyphen_values = true)]
60    pub passthrough: Vec<String>,
61}

63pub fn run(args: Args) -> Result<()> {
64    let root = match args.project.clone() {
65        Some(p) => p,
66        None => std::env::current_dir()?,
67    };
68
69    // Phase 172 WP-A + W.1 slice 3 — root nros.toml deploy dispatch.
70    //
71    // `nros build <name>` aliases `nros deploy <name>`: walk up (Cargo-style)
72    // to the enclosing `[workspace]` root, so it works from any member dir.
73    // Bare `nros build` deploys `[workspace].default` ONLY when the cwd itself
74    // is a workspace root — a member/standalone dir falls through to the
75    // project-flavor autodetect below (build the local project, like
76    // `cargo build`). A component / direct-mode `nros.toml` is not a workspace,
77    // so it never hijacks the bare build.
78    if let Some(name) = &args.deploy_name {
79        let root_toml = resolve_workspace_root(&root)?.ok_or_else(|| {
80            eyre!(
81                "nros build {name}: no workspace nros.toml found from {} — deploy targets live \
82                 in a [workspace] root",
83                root.display()
84            )
85        })?;
86        return deploy::run(deploy::Args {
87            name: Some(name.clone()),
88            config: root_toml,
89            nano_ros_workspace: args.nano_ros_workspace.clone(),
90            dry_run: false,
91        });
92    }
93    let root_toml = root.join("nros.toml");
94    if root_toml.is_file() && probe_manifest_kind(&root_toml)? == ManifestKind::Workspace {
95        let cfg = WorkspaceConfig::load(&root_toml)?;
96        let name = cfg.workspace.default.clone().ok_or_else(|| {
97            eyre!(
98                "nros build: ./nros.toml is a workspace but has no [workspace].default — \
99                 pass `nros build <name>`"
100            )
101        })?;
102        return deploy::run(deploy::Args {
103            name: Some(name),
104            config: root_toml,
105            nano_ros_workspace: args.nano_ros_workspace.clone(),
106            dry_run: false,
107        });
108    }
109
110    // Phase 187.6 (Method A): lazy-install host tools a native build needs
111    // (zenohd for the host router), then put them on PATH for the spawned build;
112    // no-op away from a nano-ros workspace / with NROS_NO_AUTO_SETUP.
113    let bins = crate::cmd::setup::ensure_tools("native", args.nano_ros_workspace.as_deref())?;
114    crate::cmd::setup::activate_store_path(&bins);
115
116    let flavor = detect_flavor(&root)?;
117    eprintln!("nros build: flavor = {flavor:?} ({})", root.display());
118
119    let mut cmd = match flavor {
120        Flavor::West => {
121            let mut c = Command::new("west");
122            c.arg("build");
123            c
124        }
125        Flavor::Cmake => {
126            // `cmake -B build && cmake --build build` chained as one
127            // shell, but we keep them as two child processes so we don't
128            // need a shell.
129            let configure = Command::new("cmake")
130                .current_dir(&root)
131                .args(["-B", "build", "-S", "."])
132                .args(&args.passthrough)
133                .status()
134                .wrap_err("failed to invoke `cmake -B build`")?;
135            if !configure.success() {
136                return Err(eyre!(
137                    "cmake configure failed (exit {})",
138                    configure.code().unwrap_or(-1)
139                ));
140            }
141            let mut c = Command::new("cmake");
142            c.arg("--build").arg("build");
143            c
144        }
145        Flavor::Cargo => {
146            let mut c = Command::new("cargo");
147            c.arg("build");
148            c
149        }
150    };
151    if !matches!(flavor, Flavor::Cmake) {
152        cmd.args(&args.passthrough);
153    }
154    cmd.current_dir(&root)
155        .stdout(Stdio::inherit())
156        .stderr(Stdio::inherit());
157
158    let status = cmd
159        .status()
160        .wrap_err_with(|| format!("failed to invoke build for {flavor:?}"))?;
161    if !status.success() {
162        return Err(eyre!("build failed (exit {})", status.code().unwrap_or(-1)));
163    }
164    Ok(())
165}

167#[derive(Debug)]
168enum Flavor {
169    West,
170    Cmake,
171    Cargo,
172}
173
174fn detect_flavor(root: &Path) -> Result<Flavor> {
175    let has_prj_conf = root.join("prj.conf").is_file();
176    let has_cmake = root.join("CMakeLists.txt").is_file();
177    let cargo_toml = root.join("Cargo.toml");
178    let has_cargo = cargo_toml.is_file();
179
180    if has_prj_conf {
181        return Ok(Flavor::West);
182    }
183
184    if has_cmake && has_cargo && produces_staticlib(&cargo_toml).unwrap_or(false) {
185        return Ok(Flavor::Cmake);
186    }
187    if has_cargo {
188        return Ok(Flavor::Cargo);
189    }
190    if has_cmake {
191        return Ok(Flavor::Cmake);
192    }
193    Err(eyre!(
194        "no build flavor detected at {}: expected prj.conf (Zephyr), \
195         CMakeLists.txt (CMake), or Cargo.toml (Rust)",
196        root.display()
197    ))
198}

200fn produces_staticlib(cargo_toml: &Path) -> Result<bool> {
201    let raw = fs::read_to_string(cargo_toml)?;
202    let doc: toml::Value = toml::from_str(&raw)?;
203    let Some(lib) = doc.get("lib") else {
204        return Ok(false);
205    };
206    let Some(crate_type) = lib.get("crate-type").or_else(|| lib.get("crate_type")) else {
207        return Ok(false);
208    };
209    Ok(match crate_type {
210        toml::Value::Array(arr) => arr.iter().any(|v| v.as_str() == Some("staticlib")),
211        toml::Value::String(s) => s == "staticlib",
212        _ => false,
213    })
