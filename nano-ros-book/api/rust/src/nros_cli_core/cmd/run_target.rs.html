1//! `nros run` — build → flash → monitor. Phase 111.A.10.
2//!
3//! v1 surface:
4//!   - Cargo + native target → `cargo run` (build + exec the default bin)
5//!   - Cargo + ESP32 (`xtensa-esp32*-none-elf` / `riscv32imc*` esp32c*) →
6//!     `espflash flash --monitor`
7//!   - Cmake / Zephyr → not yet wired; error with a clear pointer
8//!
9//! Multi-target QEMU launch (mps2-an385, nuttx-virt, threadx-…) goes
10//! through the existing `just <plat> run` recipes for now; v2 of this
11//! verb absorbs them.
12
13use clap::Args as ClapArgs;
14use eyre::{Result, WrapErr, eyre};
15use std::{
16    path::{Path, PathBuf},
17    process::{Command, Stdio},
18};
19
20#[derive(Debug, ClapArgs)]
21pub struct Args {
22    /// Path to the project root (default: cwd)
23    #[arg(long)]
24    pub project: Option<PathBuf>,
25
26    /// Named target/env (matches a `[env.<name>]` section in the
27    /// project config). Optional when the project has only one target.
28    #[arg(long)]
29    pub env: Option<String>,
30
31    /// Trailing arguments forwarded verbatim
32    #[arg(trailing_var_arg = true, allow_hyphen_values = true)]
33    pub passthrough: Vec<String>,
34}
35
36pub fn run(args: Args) -> Result<()> {
37    let root = match args.project {
38        Some(p) => p,
39        None => std::env::current_dir()?,
40    };
41
42    if root.join("prj.conf").is_file() {
43        return Err(eyre!(
44            "Zephyr `nros run` is not yet wired up. Use `west flash` after `nros build`."
45        ));
46    }
47
48    let cargo_toml = root.join("Cargo.toml");
49    if !cargo_toml.is_file() {
50        return Err(eyre!(
51            "no Cargo.toml at {}; cmake / west run paths are not yet wired",
52            root.display()
53        ));
54    }
55
56    let target = detect_cargo_target(&cargo_toml)?;
57    if let Some(t) = target.as_deref()
58        && (t.starts_with("xtensa-esp32") || t.starts_with("riscv32imc"))
59    {
60        return run_esp32(&root, &args.passthrough);
61    }
62
63    let mut cmd = Command::new("cargo");
64    cmd.arg("run").current_dir(&root);
65    cmd.args(&args.passthrough);
66    cmd.stdout(Stdio::inherit()).stderr(Stdio::inherit());
67    let status = cmd
68        .status()
69        .wrap_err_with(|| format!("failed to invoke `cargo run` in {}", root.display()))?;
70    if !status.success() {
71        return Err(eyre!(
72            "`cargo run` failed (exit {})",
73            status.code().unwrap_or(-1)
74        ));
75    }
76    Ok(())
77}
78
79fn run_esp32(root: &Path, passthrough: &[String]) -> Result<()> {
80    let mut cmd = Command::new("espflash");
81    cmd.arg("flash")
82        .arg("--monitor")
83        .current_dir(root)
84        .args(passthrough)
85        .stdout(Stdio::inherit())
86        .stderr(Stdio::inherit());
87    let status = cmd
88        .status()
89        .wrap_err("failed to invoke `espflash` (install with `cargo install espflash`)")?;
90    if !status.success() {
91        return Err(eyre!(
92            "`espflash flash --monitor` failed (exit {})",
93            status.code().unwrap_or(-1)
94        ));
95    }
96    Ok(())
97}

99/// Best-effort target detection from `.cargo/config.toml`. Returns
100/// `Some(target_triple)` when an explicit `[build].target` is set.
101fn detect_cargo_target(cargo_toml: &Path) -> Result<Option<String>> {
102    let cargo_dir = cargo_toml
103        .parent()
104        .ok_or_else(|| eyre!("invalid Cargo.toml path"))?;
105    let config = cargo_dir.join(".cargo").join("config.toml");
106    if !config.is_file() {
107        return Ok(None);
108    }
109    let raw = std::fs::read_to_string(&config)
110        .wrap_err_with(|| format!("failed to read {}", config.display()))?;
111    let doc: toml::Value = toml::from_str(&raw)?;
112    Ok(doc
113        .get("build")
114        .and_then(|b| b.get("target"))
115        .and_then(|t| t.as_str())
116        .map(|s| s.to_string()))
117}
