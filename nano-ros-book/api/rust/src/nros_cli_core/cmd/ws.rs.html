1//! `nros ws …` — workspace-level msg-pkg surface.
2//!
3//! Phase 210.B.3 + 210.D.1 (locked design). Subcommands:
4//!
5//! * `env` — print shell export for `NROS_INTERFACE_SEARCH_PATH`.
6//! * `sync` — scan workspace, codegen msg pkgs into
7//!   `generated/<pkg>/`, write `[patch.crates-io]`
8//!   block into the patch authority Cargo.toml so plain `cargo build`
9//!   resolves `local_msgs = "*"` to the generated crate.
10//!
11//! **Dual-mode (`cargo`-style):** every subcommand works on BOTH layouts —
12//! a multi-pkg colcon workspace (`<root>/src/<pkg>/package.xml`) AND a
13//! single standalone pkg (`<root>/package.xml`). Detection runs at command
14//! time:
15//!
16//!   * **colcon-mode** iff `<root>/src/` exists AND at least one
17//!     immediate subdir contains `package.xml`.
18//!   * **single-pkg mode** iff `<root>/package.xml` exists and the colcon
19//!     check fails.
20//!
21//! Mirrors `cargo build` which works at either a workspace root or a
22//! standalone pkg dir without special arg.
23//!
24//! See `docs/roadmap/phase-210-ros-convention-codegen.md` for the
25//! full design (patch authority detection, colcon-shape build dir,
26//! the chicken-egg motivation for a pre-cargo sync step).
27

41#[derive(Debug, Subcommand)]
42pub enum Sub {
43    /// Print shell export adding <dir> (default `./src`) to
44    /// `NROS_INTERFACE_SEARCH_PATH`. `eval "$(nros ws env)"`.
45    Env(EnvArgs),
46
47    /// Codegen workspace msg pkgs + write `[patch.crates-io]` block into
48    /// each Rust consumer's patch authority Cargo.toml. Pre-cargo step;
49    /// run once after editing `*.msg` files, then `cargo build` works.
50    Sync(SyncArgs),
51
52    /// List discovered msg + rust-consumer pkgs in the workspace (or
53    /// single pkg). Prints kind, name, dir per row. (Phase 210.F.3.)
54    List(ListArgs),
55
56    /// Freshness check — non-fatal sibling of `sync --check`. Prints a
57    /// one-line summary of `n up-to-date / n stale / n missing`.
58    Status(StatusArgs),
59
60    /// Remove `generated/` + the auto-managed
61    /// `[patch.crates-io]` block from each Rust consumer's patch authority
62    /// Cargo.toml. Leaves user-written sections alone.
63    Clean(CleanArgs),
64
65    /// Lint workspace pkgs: warn on missing `<member_of_group>
66    /// rosidl_interface_packages</member_of_group>` markers, malformed
67    /// `package.xml`, stale patch blocks. Mirrors the sync detection.
68    Doctor(DoctorArgs),
69}

158pub fn run(args: Args) -> Result<()> {
159    match args.command {
160        Sub::Env(a) => run_env(a),
161        Sub::Sync(a) => run_sync(a),
162        Sub::List(a) => run_list(a),
163        Sub::Status(a) => run_status(a),
164        Sub::Clean(a) => run_clean(a),
165        Sub::Doctor(a) => run_doctor(a),
166    }

173fn run_env(args: EnvArgs) -> Result<()> {
174    let abs = resolve_env_root(args.workspace.as_deref())?;
175    let abs_s = abs.display().to_string();
176    match args.shell {
177        Shell::Posix => {
178            println!("export NROS_INTERFACE_SEARCH_PATH=\"{abs_s}:${{NROS_INTERFACE_SEARCH_PATH:-}}\"");
179        }
180        Shell::Fish => {
181            println!("set -gx NROS_INTERFACE_SEARCH_PATH \"{abs_s}\" $NROS_INTERFACE_SEARCH_PATH");
182        }
183    }
184    Ok(())
185}

187/// Resolve the dir the cmake-side smart Find-stub will scan as a
188/// `NROS_INTERFACE_SEARCH_PATH` entry. Mirrors `sync`'s dual-mode
189/// detection so a `cd <my_pkg> && eval "$(nros ws env)"` from inside a
190/// standalone pkg works the same as one run at a colcon workspace root.
191///
192/// Resolution order:
193///   1. Explicit path arg → use it.
194///   2. `<cwd>/src/<sub>/package.xml` exists → use `<cwd>/src`.
195///   3. `<cwd>/package.xml` exists → use `<cwd>/..` (so smart Find-stub
196///      finds `<parent>/<my_pkg>/package.xml` from there).
197///   4. Fallback → `<cwd>/src` (legacy default; may not exist).
198fn resolve_env_root(arg: Option<&Path>) -> Result<PathBuf> {
199    if let Some(p) = arg {
200        return std::fs::canonicalize(p)
201            .map_err(|e| eyre!("ws env: {}: {e}", p.display()));
202    }
203    let cwd = std::env::current_dir()?;
204    let src = cwd.join("src");
205    if src.is_dir() && has_pkg_subdir(&src) {
206        return std::fs::canonicalize(&src)
207            .map_err(|e| eyre!("ws env: {}: {e}", src.display()));
208    }
209    if cwd.join("package.xml").is_file() {
210        let parent = cwd.parent().ok_or_else(|| {
211            eyre!("ws env: cwd {} is a standalone pkg but has no parent", cwd.display())
212        })?;
213        return std::fs::canonicalize(parent)
214            .map_err(|e| eyre!("ws env: {}: {e}", parent.display()));
215    }
216    // Fallback — caller might not be in a pkg/workspace dir. Use ./src
217    // and surface the error from canonicalize if it doesn't exist.
218    std::fs::canonicalize(&src)
219        .map_err(|e| eyre!("ws env: {}: {e}\n\
220                            (no `src/<pkg>/package.xml` colcon layout and no `package.xml` \
221                            at cwd — pass an explicit path arg)",
222                           src.display()))
223}

229/// Scanned workspace pkg.
230#[derive(Debug, Clone)]
231struct WsPkg {
232    name: String,
233    dir: PathBuf,
234    manifest: PathBuf,
235    /// True iff msg pkg (member_of_group=rosidl_interface_packages OR
236    /// msg/srv/action dirs).
237    is_msg_pkg: bool,
238    /// True iff `Cargo.toml` at root.
239    is_rust_pkg: bool,
240    /// Pkg names declared in `<*depend>` tags (filtered for ROS-meta).
241    deps: Vec<String>,
242}
243
244fn run_sync(args: SyncArgs) -> Result<()> {
245    let ws_root: PathBuf = match args.workspace {
246        Some(p) => std::fs::canonicalize(&p)
247            .wrap_err_with(|| format!("ws sync: {}", p.display()))?,
248        None => std::env::current_dir()?,
249    };
250    // Two layouts supported:
251    //  * `src/`-based: workspace root has src/, src/<pkg>/ subdirs (colcon
252    //    standard).
253    //  * Single-pkg: workspace root IS the pkg dir (package.xml at root).
254    //    Common for ported standalone examples (`examples/native/rust/talker`).
255    // Heuristic: colcon-style layout iff `src/` exists AND has at least one
256    // immediate subdir with `package.xml`. Falls through to single-pkg mode
257    // when the workspace root itself carries `package.xml` (the standalone
258    // example shape; `src/` may exist as the cargo source dir).
259    let colcon_layout = ws_root.join("src").is_dir()
260        && has_pkg_subdir(&ws_root.join("src"));
261    let single_pkg_mode = !colcon_layout && ws_root.join("package.xml").is_file();
262    let src_root = if colcon_layout {
263        ws_root.join("src")
264    } else if single_pkg_mode {
265        ws_root.clone()
266    } else {
267        bail!(
268            "ws sync: no `src/<pkg>/package.xml` and no `package.xml` at root \
269             under {} — expected colcon-style workspace or single-pkg dir",
270            ws_root.display()
271        );
272    };
273    let build_root = if args.build_dir.is_absolute() {
274        args.build_dir.clone()
275    } else {
276        ws_root.join(&args.build_dir)
277    };
278
279    let mut scan = Vec::new();
280    if single_pkg_mode {
281        scan_one_pkg_dir(&src_root, &mut scan)?;
282    } else {
283        scan_workspace(&src_root, &mut scan)?;
284    }
285    if scan.is_empty() {
286        println!("ws sync: no pkgs under {}", src_root.display());
287        return Ok(());
288    }
289    let msg_pkgs: Vec<&WsPkg> = scan.iter().filter(|p| p.is_msg_pkg).collect();
290    let topo = topo_sort_msg_pkgs(&msg_pkgs)?;
291
292    if args.verbose || args.dry_run {
293        println!(
294            "ws sync: scanned {} pkgs ({} msg, {} rust) under {}",
295            scan.len(),
296            msg_pkgs.len(),
297            scan.iter().filter(|p| p.is_rust_pkg).count(),
298            src_root.display()
299        );
300        println!("ws sync: topo order: {topo:?}");
301    }
302
303    if args.check {
304        return check_freshness(&ws_root, &build_root, &scan, &topo);
305    }
306
307    if args.dry_run {
308        for name in &topo {
309            let pkg = scan.iter().find(|p| &p.name == name).unwrap();
310            let out = build_root.join(name);
311            println!(
312                "ws sync: WOULD codegen {} from {} → {}",
313                name,
314                pkg.manifest.display(),
315                out.display()
316            );
317        }
318        return Ok(());
319    }
320
321    let edition = parse_edition(&args.ros_edition)?;
322
323    // Track every pkg we generate so a later iteration (or AMENT-dep walk)
324    // skips already-emitted ones. Keyed by pkg name.
325    let mut emitted: HashSet<String> = HashSet::new();
326
327    for name in &topo {
328        let pkg = scan.iter().find(|p| &p.name == name).unwrap();
329        // First materialize any AMENT-resolved cross-deps so the workspace
330        // pkg's deps closure exists in build/ too. Skips workspace deps
331        // (those are handled by topo order itself).
332        codegen_ament_deps_for(&pkg.deps, &scan, &build_root, edition, &mut emitted, args.verbose)?;
333        // Now generate the workspace pkg itself directly from its dir.
334        if !emitted.contains(name) {
335            codegen_workspace_pkg(pkg, &build_root, edition, args.verbose)?;
336            emitted.insert(name.clone());
337        }
338    }
339    // Also generate AMENT deps for every Rust consumer (pkg.xml deps).
340    let rust_consumers: Vec<&WsPkg> = scan
341        .iter()
342        .filter(|p| p.is_rust_pkg && !p.is_msg_pkg)
343        .collect();
344    for c in &rust_consumers {
345        codegen_ament_deps_for(&c.deps, &scan, &build_root, edition, &mut emitted, args.verbose)?;
346    }
347
348    if rust_consumers.is_empty() {
349        println!("ws sync: no Rust consumer pkgs — patch tables not written.");
350        return Ok(());
351    }
352
353    // Group consumers by patch authority. Cargo workspace covers many
354    // consumers via one umbrella; standalone pkgs are their own authority.
355    let all_emitted: Vec<String> = {
356        let mut v: Vec<String> = emitted.iter().cloned().collect();
357        v.sort();
358        v
359    };
360    let mut authority_to_pkgs: HashMap<PathBuf, Vec<String>> = HashMap::new();
361    for c in &rust_consumers {
362        let authority = find_patch_authority(&c.dir, &ws_root)?;
363        authority_to_pkgs
364            .entry(authority)
365            .or_default()
366            .extend(all_emitted.iter().cloned());
367    }
368    let nano_ros_path = args
369        .nano_ros_path
370        .or_else(|| std::env::var_os("NROS_REPO_DIR").map(PathBuf::from));
371    for (authority, pkgs) in authority_to_pkgs {
372        let mut unique = pkgs;
373        unique.sort();
374        unique.dedup();
375        write_patch_block(&authority, &build_root, &unique, nano_ros_path.as_deref())?;
376    }
377
378    println!("ws sync: done.");
379    Ok(())
380}

383fn parse_edition(s: &str) -> Result<RosEdition> {
384    match s.to_lowercase().as_str() {
385        "humble" => Ok(RosEdition::Humble),
386        "iron" => Ok(RosEdition::Iron),
387        other => bail!("ws sync: unknown ROS edition '{other}' (humble | iron)"),
388    }
389}

391// Generate the workspace pkg directly (using its dir as a synthetic share_dir
392// — `Package::from_share_dir` reads `package.xml` + scans msg/srv/action).
393fn codegen_workspace_pkg(
394    pkg: &WsPkg,
395    build_root: &Path,
396    edition: RosEdition,
397    verbose: bool,
398) -> Result<()> {
399    let out_dir = build_root;
400    std::fs::create_dir_all(&out_dir)
401        .wrap_err_with(|| format!("ws sync: mkdir {}", out_dir.display()))?;
402    if verbose {
403        println!("ws sync: codegen workspace pkg {} → {}", pkg.name, out_dir.display());
404    } else {
405        println!("ws sync: codegen {}", pkg.name);
406    }
407    let package = Package::from_share_dir(pkg.dir.clone())
408        .wrap_err_with(|| format!("ws sync: read pkg {}", pkg.dir.display()))?;
409    rosidl_bindgen::generator::generate_package(&package, &out_dir, edition)
410        .wrap_err_with(|| format!("ws sync: generate_package failed for {}", pkg.name))?;
411    // Codegen emits <out_dir>/<pkg>/{Cargo.toml,src/} with sibling `path =
412    // "../<dep>"` deps. We keep that flat layout (no extra `rust/`
413    // nesting) so the relative paths between generated crates resolve
414    // correctly without a rewrite pass. Our `nros_generator_rs` prefix
415    // already namespaces by language — the extra `rust/` colcon adds is
416    // there to coexist with `<pkg>/c/`, `<pkg>/cpp/`, etc. inside the
417    // same generator's output, which we don't have.
418    Ok(())
419}

421// Resolve AMENT-side deps (the per-pkg.xml `<depend>` tags not in workspace)
422// and codegen each via Package::from_share_dir over its AMENT share path.
423fn codegen_ament_deps_for(
424    deps: &[String],
425    scan: &[WsPkg],
426    build_root: &Path,
427    edition: RosEdition,
428    emitted: &mut HashSet<String>,
429    verbose: bool,
430) -> Result<()> {
431    // Pre-load ament index once per invocation.
432    static AMENT_INDEX: std::sync::OnceLock<Option<rosidl_bindgen::ament::AmentIndex>> =
433        std::sync::OnceLock::new();
434    let idx = AMENT_INDEX.get_or_init(|| {
435        rosidl_bindgen::ament::AmentIndex::from_env().ok()
436    });
437    let Some(idx) = idx else { return Ok(()) };
438
439    let in_workspace: HashSet<&str> = scan.iter().map(|p| p.name.as_str()).collect();
440    let mut to_resolve: Vec<String> =
441        deps.iter().filter(|d| !in_workspace.contains(d.as_str())).cloned().collect();
442
443    while let Some(dep) = to_resolve.pop() {
444        if emitted.contains(&dep) {
445            continue;
446        }
447        let Some(amented) = idx.packages().get(&dep).cloned() else {
448            // AMENT doesn't know — silently skip (smart-stub semantics).
449            continue;
450        };
451        // Codegen the AMENT pkg.
452        let out_dir = build_root;
453        std::fs::create_dir_all(&out_dir)?;
454        if verbose {
455            println!("ws sync: codegen AMENT pkg {} → {}", amented.name, out_dir.display());
456        } else {
457            println!("ws sync: codegen {}", amented.name);
458        }
459        rosidl_bindgen::generator::generate_package(&amented, &out_dir, edition)
460            .wrap_err_with(|| format!("ws sync: generate_package failed for {}", amented.name))?;
461        emitted.insert(amented.name.clone());
462        // Queue this pkg's own deps (parse its package.xml).
463        let pxml = amented.share_dir.join("package.xml");
464        if pxml.is_file() {
465            let body = std::fs::read_to_string(&pxml).unwrap_or_default();
466            for d in extract_pkg_deps(&body) {
467                if !in_workspace.contains(d.as_str()) && !emitted.contains(&d) {
468                    to_resolve.push(d);
469                }
470            }
471        }
472    }
473    Ok(())
474}

479fn has_pkg_subdir(dir: &Path) -> bool {
480    let Ok(entries) = std::fs::read_dir(dir) else {
481        return false;
482    };
483    for e in entries.flatten() {
484        if let Ok(t) = e.file_type() {
485            if t.is_dir() && e.path().join("package.xml").is_file() {
486                return true;
487            }
488        }
489    }
490    false
491}
492
493fn scan_one_pkg_dir(pkg_dir: &Path, out: &mut Vec<WsPkg>) -> Result<()> {
494    let manifest = pkg_dir.join("package.xml");
495    let body = std::fs::read_to_string(&manifest)?;
496    let Some(name) = extract_pkg_name(&body) else {
497        bail!("ws sync: single-pkg mode: package.xml at {} has no <name>",
498              manifest.display());
499    };
500    let is_msg_pkg = body.contains("rosidl_interface_packages")
501        || pkg_dir.join("msg").is_dir()
502        || pkg_dir.join("srv").is_dir()
503        || pkg_dir.join("action").is_dir();
504    let is_rust_pkg = pkg_dir.join("Cargo.toml").is_file();
505    let deps = extract_pkg_deps(&body);
506    out.push(WsPkg {
507        name,
508        dir: pkg_dir.to_path_buf(),
509        manifest,
510        is_msg_pkg,
511        is_rust_pkg,
512        deps,
513    });
514    Ok(())
515}

517fn scan_workspace(src_root: &Path, out: &mut Vec<WsPkg>) -> Result<()> {
518    for entry in std::fs::read_dir(src_root)? {
519        let entry = entry?;
520        if !entry.file_type()?.is_dir() {
521            continue;
522        }
523        let dir = entry.path();
524        let manifest = dir.join("package.xml");
525        if !manifest.is_file() {
526            continue;
527        }
528        let body = std::fs::read_to_string(&manifest)?;
529        let Some(name) = extract_pkg_name(&body) else {
530            continue;
531        };
532        let is_msg_pkg = body.contains("rosidl_interface_packages")
533            || dir.join("msg").is_dir()
534            || dir.join("srv").is_dir()
535            || dir.join("action").is_dir();
536        let is_rust_pkg = dir.join("Cargo.toml").is_file();
537        let deps = extract_pkg_deps(&body);
538        out.push(WsPkg {
539            name,
540            dir,
541            manifest,
542            is_msg_pkg,
543            is_rust_pkg,
544            deps,
545        });
546    }
547    Ok(())
548}

550fn extract_pkg_name(body: &str) -> Option<String> {
551    let start = body.find("<name>")? + "<name>".len();
552    let end = body[start..].find("</name>")? + start;
553    Some(body[start..end].trim().to_string())
554}
555
556fn extract_pkg_deps(body: &str) -> Vec<String> {
557    let mut deps = Vec::new();
558    for tag in &[
559        "<depend>",
560        "<build_depend>",
561        "<exec_depend>",
562        "<run_depend>",
563        "<build_export_depend>",
564    ] {
565        let close = tag.replace("<", "</");
566        let mut cursor = 0;
567        while let Some(rel) = body[cursor..].find(tag) {
568            let start = cursor + rel + tag.len();
569            let Some(rel_close) = body[start..].find(close.as_str()) else {
570                break;
571            };
572            let end = start + rel_close;
573            let name = body[start..end].trim().to_string();
574            if !name.is_empty() && !is_ros_meta_pkg(&name) {
575                deps.push(name);
576            }
577            cursor = end;
578        }
579    }
580    deps.sort();
581    deps.dedup();
582    deps
583}

585fn is_ros_meta_pkg(name: &str) -> bool {
586    name.starts_with("rosidl")
587        || name.starts_with("ament")
588        || name == "rclcpp"
589        || name == "rclpy"
590        || name.starts_with("rcl")
591        || name.starts_with("rmw")
592        || name.starts_with("launch")
593        || name == "catkin"
594}

596fn topo_sort_msg_pkgs(pkgs: &[&WsPkg]) -> Result<Vec<String>> {
597    let names: std::collections::HashSet<&str> =
598        pkgs.iter().map(|p| p.name.as_str()).collect();
599    let mut remaining: Vec<&&WsPkg> = pkgs.iter().collect();
600    let mut emitted: Vec<String> = Vec::new();
601    while !remaining.is_empty() {
602        let pick_idx = remaining.iter().position(|p| {
603            p.deps
604                .iter()
605                .filter(|d| names.contains(d.as_str()))
606                .all(|d| emitted.contains(d))
607        });
608        match pick_idx {
609            Some(idx) => emitted.push(remaining.remove(idx).name.clone()),
610            None => {
611                let names: Vec<&str> = remaining.iter().map(|p| p.name.as_str()).collect();
612                bail!("ws sync: dependency cycle (or missing dep) among {names:?}");
613            }
614        }
615    }
616    Ok(emitted)
617}

619// --- Patch authority -----------------------------------------------------------
620
621fn find_patch_authority(start: &Path, ws_root: &Path) -> Result<PathBuf> {
622    let mut cur = start.to_path_buf();
623    loop {
624        let cargo = cur.join("Cargo.toml");
625        if cargo.is_file() {
626            let body = std::fs::read_to_string(&cargo)?;
627            if has_workspace_table(&body) {
628                return Ok(cargo);
629            }
630        }
631        if cur == *ws_root {
632            return Ok(start.join("Cargo.toml"));
633        }
634        match cur.parent() {
635            Some(p) => cur = p.to_path_buf(),
636            None => return Ok(start.join("Cargo.toml")),
637        }
638    }
639}
640
641fn has_workspace_table(body: &str) -> bool {
642    body.lines().any(|l| {
643        let t = l.trim();
644        t == "[workspace]" || t.starts_with("[workspace]")
645    })
646}

648// --- Patch block writer --------------------------------------------------------
649
650const BEGIN: &str = "# === BEGIN nros-managed [patch.crates-io] ===";
651const END: &str = "# === END nros-managed [patch.crates-io] ===";
652
653fn write_patch_block(
654    authority: &Path,
655    build_root: &Path,
656    pkgs: &[String],
657    nano_ros_path: Option<&Path>,
658) -> Result<()> {
659    let mut body = std::fs::read_to_string(authority)
660        .wrap_err_with(|| format!("ws sync: read {}", authority.display()))?;
661    let block = render_patch_block(authority, build_root, pkgs, nano_ros_path);
662    body = replace_or_append_block(&body, &block);
663    std::fs::write(authority, body)
664        .wrap_err_with(|| format!("ws sync: write {}", authority.display()))?;
665    println!(
666        "ws sync: refreshed [patch.crates-io] block in {}",
667        authority.display()
668    );
669    Ok(())
670}
671
672fn render_patch_block(
673    authority: &Path,
674    build_root: &Path,
675    pkgs: &[String],
676    nano_ros_path: Option<&Path>,
677) -> String {
678    let authority_dir = authority.parent().unwrap();
679    let mut out = String::new();
680    out.push_str(&format!("\n{BEGIN}\n"));
681    // No timestamp — deterministic output keeps the committed Cargo.toml
682    // diff-stable across re-syncs (only path entries change when the user
683    // adds/removes a msg pkg). For "when did sync last run" debugging,
684    // grep the generated/<pkg>/Cargo.toml mtime.
685    out.push_str("# Auto-generated by `nros ws sync`. Do not edit between\n");
686    out.push_str("# the BEGIN/END markers — re-run sync instead.\n");
687    out.push_str("[patch.crates-io]\n");
688
689    // 1) Generated msg crates (path = generated/<pkg>).
690    for pkg in pkgs {
691        let crate_root = build_root.join(pkg);
692        let rel = pathdiff::diff_paths(&crate_root, authority_dir).unwrap_or(crate_root);
693        out.push_str(&format!("{pkg} = {{ path = \"{}\" }}\n", rel.display()));
694    }
695
696    // 2) Minimum runtime patches the GENERATED msg crates depend on.
697    //    Generated `<pkg>/Cargo.toml` carries `nros-core = "*"` +
698    //    `nros-serdes = "*"` (registry-style), so even when the user's
699    //    own Cargo.toml has direct path-deps for the larger nros-*
700    //    runtime (canonical 212 shape), the generated crates need
701    //    `[patch.crates-io]` entries for these two specific crates to
702    //    resolve. Larger nros-* runtime patches were dropped post-212
703    //    merge — they triggered cargo's "patch unused" warnings.
704    if let Some(nrp) = nano_ros_path {
705        for (cname, sub) in &[
706            ("nros-core", "packages/core/nros-core"),
707            ("nros-serdes", "packages/core/nros-serdes"),
708        ] {
709            let crate_root = nrp.join(sub);
710            if !crate_root.join("Cargo.toml").is_file() {
711                continue;
712            }
713            let rel = pathdiff::diff_paths(&crate_root, authority_dir).unwrap_or(crate_root);
714            out.push_str(&format!("{cname} = {{ path = \"{}\" }}\n", rel.display()));
715        }
716    }
717
718    out.push_str(&format!("{END}\n"));
719    out
720}
721
722fn replace_or_append_block(body: &str, block: &str) -> String {
723    if let (Some(b), Some(e)) = (body.find(BEGIN), body.find(END)) {
724        let end_line_end = e + END.len();
725        let block_trimmed = block.trim_start_matches('\n');
726        let mut out = String::new();
727        out.push_str(&body[..b]);
728        out.push_str(block_trimmed);
729        out.push_str(&body[end_line_end..]);
730        out
731    } else {
732        let mut out = body.to_string();
733        if !out.ends_with('\n') {
734            out.push('\n');
735        }
736        out.push_str(block);
737        out
738    }
739}

741// --- Check / freshness ---------------------------------------------------------
742
743fn check_freshness(
744    ws_root: &Path,
745    build_root: &Path,
746    scan: &[WsPkg],
747    topo: &[String],
748) -> Result<()> {
749    let mut stale = false;
750    for name in topo {
751        let pkg = scan.iter().find(|p| &p.name == name).unwrap();
752        let crate_root = build_root
753            
754            .join(name);
755        let cargo = crate_root.join("Cargo.toml");
756        if !cargo.is_file() {
757            eprintln!(
758                "ws sync --check: stale: {name} — no Cargo.toml at {}",
759                cargo.display()
760            );
761            stale = true;
762            continue;
763        }
764        let cargo_mt = std::fs::metadata(&cargo)?.modified()?;
765        for subdir in &["msg", "srv", "action"] {
766            let d = pkg.dir.join(subdir);
767            if !d.is_dir() {
768                continue;
769            }
770            for entry in std::fs::read_dir(d)? {
771                let entry = entry?;
772                if !entry.file_type()?.is_file() {
773                    continue;
774                }
775                let mt = entry.metadata()?.modified()?;
776                if mt > cargo_mt {
777                    eprintln!(
778                        "ws sync --check: stale: {name} — {} newer than generated crate",
779                        entry
780                            .path()
781                            .strip_prefix(ws_root)
782                            .unwrap_or(&entry.path())
783                            .display()
784                    );
785                    stale = true;
786                }
787            }
788        }
789    }
790    if stale {
791        bail!("ws sync --check: some pkgs stale — run `nros ws sync` first.");
792    }
793    println!("ws sync --check: all good.");
794    Ok(())
795}

797// =============================================================================
798// Phase 210.F.3 — `nros ws {list,status,clean,doctor}` sibling subcommands.
799// All dual-mode (single-pkg + colcon-style workspace), same detection as sync.
800// =============================================================================
801
802/// Run sync's scan+resolve step without codegen — for list/status/clean/
803/// doctor. Returns the workspace root + scanned pkgs + the resolved
804/// build_root. The optional `build_dir` arg defaults to `<ws_root>/build`.
805fn scan_for_query(
806    workspace: Option<&Path>,
807    build_dir: &Path,
808) -> Result<(PathBuf, Vec<WsPkg>, PathBuf)> {
809    let ws_root: PathBuf = match workspace {
810        Some(p) => std::fs::canonicalize(p)
811            .wrap_err_with(|| format!("ws: {}", p.display()))?,
812        None => std::env::current_dir()?,
813    };
814    let colcon_layout = ws_root.join("src").is_dir()
815        && has_pkg_subdir(&ws_root.join("src"));
816    let single_pkg_mode = !colcon_layout && ws_root.join("package.xml").is_file();
817    let src_root = if colcon_layout {
818        ws_root.join("src")
819    } else if single_pkg_mode {
820        ws_root.clone()
821    } else {
822        bail!(
823            "ws: no `src/<pkg>/package.xml` and no `package.xml` at root \
824             under {} — expected colcon-style workspace or single-pkg dir",
825            ws_root.display()
826        );
827    };
828    let mut scan = Vec::new();
829    if single_pkg_mode {
830        scan_one_pkg_dir(&src_root, &mut scan)?;
831    } else {
832        scan_workspace(&src_root, &mut scan)?;
833    }
834    let build_root = if build_dir.is_absolute() {
835        build_dir.to_path_buf()
836    } else {
837        ws_root.join(build_dir)
838    };
839    Ok((ws_root, scan, build_root))
840}

842// --- list ---------------------------------------------------------------------
843
844fn run_list(args: ListArgs) -> Result<()> {
845    // build_dir doesn't matter for list; use the default for the scan
846    // helper's signature.
847    let (ws_root, scan, _build_root) =
848        scan_for_query(args.workspace.as_deref(), Path::new("build"))?;
849    if scan.is_empty() {
850        println!("ws list: no pkgs found.");
851        return Ok(());
852    }
853    println!("ws list ({}):", ws_root.display());
854    let mut kinds = (0usize, 0usize); // (msg, rust)
855    for p in &scan {
856        let kind = match (p.is_msg_pkg, p.is_rust_pkg) {
857            (true, true) => "msg+rust",
858            (true, false) => "msg",
859            (false, true) => "rust",
860            (false, false) => "other",
861        };
862        if p.is_msg_pkg {
863            kinds.0 += 1;
864        }
865        if p.is_rust_pkg && !p.is_msg_pkg {
866            kinds.1 += 1;
867        }
868        println!(
869            "  {kind:9}  {:24}  {}",
870            p.name,
871            p.dir.strip_prefix(&ws_root).unwrap_or(&p.dir).display()
872        );
873    }
874    println!("ws list: {} msg, {} rust consumer", kinds.0, kinds.1);
875    Ok(())
876}

878// --- status -------------------------------------------------------------------
879
880fn run_status(args: StatusArgs) -> Result<()> {
881    let (ws_root, scan, build_root) =
882        scan_for_query(args.workspace.as_deref(), &args.build_dir)?;
883    let msg_pkgs: Vec<&WsPkg> = scan.iter().filter(|p| p.is_msg_pkg).collect();
884    if msg_pkgs.is_empty() {
885        println!("ws status: no msg pkgs.");
886        return Ok(());
887    }
888    let mut up_to_date = 0;
889    let mut stale = 0;
890    let mut missing = 0;
891    for pkg in &msg_pkgs {
892        let crate_root = build_root.join(&pkg.name);
893        let cargo = crate_root.join("Cargo.toml");
894        if !cargo.is_file() {
895            missing += 1;
896            continue;
897        }
898        let cargo_mt = match std::fs::metadata(&cargo).and_then(|m| m.modified()) {
899            Ok(t) => t,
900            Err(_) => {
901                missing += 1;
902                continue;
903            }
904        };
905        let mut pkg_stale = false;
906        for subdir in &["msg", "srv", "action"] {
907            let d = pkg.dir.join(subdir);
908            if !d.is_dir() {
909                continue;
910            }
911            for e in std::fs::read_dir(d)?.flatten() {
912                if e.file_type().map(|t| t.is_file()).unwrap_or(false) {
913                    if let Ok(mt) = e.metadata().and_then(|m| m.modified()) {
914                        if mt > cargo_mt {
915                            pkg_stale = true;
916                            break;
917                        }
918                    }
919                }
920            }
921            if pkg_stale {
922                break;
923            }
924        }
925        if pkg_stale {
926            stale += 1;
927        } else {
928            up_to_date += 1;
929        }
930    }
931    let _ = ws_root;
932    println!(
933        "ws status: {up_to_date} up-to-date, {stale} stale, {missing} missing \
934         (of {} msg pkgs)",
935        msg_pkgs.len()
936    );
937    Ok(())
938}

940// --- clean --------------------------------------------------------------------
941
942fn run_clean(args: CleanArgs) -> Result<()> {
943    let (ws_root, scan, build_root) =
944        scan_for_query(args.workspace.as_deref(), &args.build_dir)?;
945    let gen_dir = build_root;
946    if gen_dir.is_dir() {
947        if args.dry_run {
948            println!("ws clean: WOULD rm -rf {}", gen_dir.display());
949        } else {
950            std::fs::remove_dir_all(&gen_dir)
951                .wrap_err_with(|| format!("ws clean: rm {}", gen_dir.display()))?;
952            println!("ws clean: removed {}", gen_dir.display());
953        }
954    } else {
955        println!("ws clean: {} not present, skip", gen_dir.display());
956    }
957    // Strip the auto-managed patch block from every Rust consumer's patch
958    // authority Cargo.toml.
959    let rust_consumers: Vec<&WsPkg> =
960        scan.iter().filter(|p| p.is_rust_pkg).collect();
961    let mut authorities: std::collections::HashSet<PathBuf> =
962        std::collections::HashSet::new();
963    for c in &rust_consumers {
964        if let Ok(a) = find_patch_authority(&c.dir, &ws_root) {
965            authorities.insert(a);
966        }
967    }
968    for authority in authorities {
969        let body = match std::fs::read_to_string(&authority) {
970            Ok(b) => b,
971            Err(_) => continue,
972        };
973        let (Some(b), Some(e)) = (body.find(BEGIN), body.find(END)) else {
974            continue;
975        };
976        if args.dry_run {
977            println!("ws clean: WOULD strip patch block from {}", authority.display());
978            continue;
979        }
980        let end_line_end = e + END.len();
981        let mut out = String::new();
982        out.push_str(&body[..b]);
983        // Drop the preceding blank line we added when writing the block.
984        let trimmed = out.trim_end_matches('\n');
985        out.truncate(trimmed.len());
986        out.push('\n');
987        out.push_str(&body[end_line_end..]);
988        std::fs::write(&authority, out)
989            .wrap_err_with(|| format!("ws clean: write {}", authority.display()))?;
990        println!("ws clean: stripped patch block from {}", authority.display());
991    }
992    Ok(())
993}

995// --- doctor -------------------------------------------------------------------
996
997fn run_doctor(args: DoctorArgs) -> Result<()> {
998    let (ws_root, scan, build_root) =
999        scan_for_query(args.workspace.as_deref(), &args.build_dir)?;
1000    let mut warnings = 0;
1001    println!("ws doctor ({})", ws_root.display());
1002    for pkg in &scan {
1003        // (a) package.xml well-formed?
1004        let body = match std::fs::read_to_string(&pkg.manifest) {
1005            Ok(b) => b,
1006            Err(e) => {
1007                eprintln!("  ✗ {}: package.xml read error: {e}", pkg.name);
1008                warnings += 1;
1009                continue;
1010            }
1011        };
1012        // (b) msg pkg without member_of_group=rosidl_interface_packages
1013        let has_iface_group = body.contains("rosidl_interface_packages");
1014        let has_msg_dirs = pkg.dir.join("msg").is_dir()
1015            || pkg.dir.join("srv").is_dir()
1016            || pkg.dir.join("action").is_dir();
1017        if has_msg_dirs && !has_iface_group {
1018            eprintln!(
1019                "  ⚠ {}: has msg/srv/action dirs but pkg.xml lacks \
1020                 <member_of_group>rosidl_interface_packages</member_of_group> \
1021                 — upstream colcon won't classify it as an interface pkg",
1022                pkg.name
1023            );
1024            warnings += 1;
1025        }
1026        // (c) rust consumer: is the patch authority Cargo.toml sane?
1027        if pkg.is_rust_pkg && !pkg.is_msg_pkg {
1028            match find_patch_authority(&pkg.dir, &ws_root) {
1029                Ok(a) => {
1030                    let body = std::fs::read_to_string(&a).unwrap_or_default();
1031                    if !body.contains(BEGIN) {
1032                        eprintln!(
1033                            "  ⚠ {}: no nros-managed [patch.crates-io] block in \
1034                             patch authority ({}). Run `nros ws sync`.",
1035                            pkg.name, a.display()
1036                        );
1037                        warnings += 1;
1038                    }
1039                }
1040                Err(e) => {
1041                    eprintln!("  ⚠ {}: patch authority resolve failed: {e}", pkg.name);
1042                    warnings += 1;
1043                }
1044            }
1045        }
1046    }
1047    // (d) stale msg pkgs (same logic as status).
1048    let _ = build_root;
1049    if warnings == 0 {
1050        println!("ws doctor: no issues.");
1051    } else {
1052        println!("ws doctor: {warnings} warning(s).");
1053    }
1054    Ok(())
1055}
