1//! Phase 212.J — `nros launch` host-side launcher.
2//!
3//! Reads a bringup pkg's `system.toml` (+ optional resolved
4//! `nros-plan.json`) and spawns one host process per `[[component]]` with
5//! the env the runtime expects (`NROS_LOCATOR`, `ROS_DOMAIN_ID`, +
6//! per-component parameters / remaps from the component crate's
7//! `[package.metadata.nros.component]`).
8//!
9//! Lets users `nros launch <bringup>` on a desktop / native_sim host
10//! without depending on the ament index — `ros2 launch` requires
11//! `colcon build && source install/setup.bash`, this doesn't. This is
12//! the **canonical desktop launcher** for development; `ros2 launch`
13//! remains available for ament-installed consumers (separate path, no
14//! overlap). See `docs/design/multi-node-workspace-layout.md` §11 for
15//! the role of bringup pkgs and how `nros launch` reads `launch/` from
16//! the source tree, not from an ament install share path.
17//!
18//! Scope: **hosted/native targets only**. A `[deploy.<target>]` whose
19//! `kind` is `"qemu"` / `"flash"` is a clear error — those need their own
20//! runner (`nros run`), not host process spawn.
21

40const DEFAULT_LOCATOR: &str = "tcp/127.0.0.1:7447";

42#[derive(Debug, ClapArgs)]
43pub struct Args {
44    /// Bringup package directory (or pkg name, looked up under the workspace
45    /// root). Omit to use the workspace's
46    /// `[workspace.metadata.nros].default_system`.
47    pub bringup: Option<String>,
48
49    /// Workspace root holding the cargo workspace `Cargo.toml`. Defaults to
50    /// the current dir.
51    #[arg(long)]
52    pub workspace_root: Option<PathBuf>,
53
54    /// `[deploy.<target>]` to use. When unset, the launcher picks
55    /// `[system].default_target` (Phase 212.J.2) if present, else falls
56    /// back to `"native"` if that block exists, else the first deploy
57    /// entry in sorted order. Empty `[deploy]` is allowed — the launcher
58    /// falls back to baked defaults.
59    #[arg(long)]
60    pub target: Option<String>,
61
62    /// Cargo profile dir under `<workspace>/target/<profile>/<exec>`.
63    #[arg(long, default_value = "debug")]
64    pub profile: String,
65
66    /// Foreground mode: block until first child exits or signal arrives, then
67    /// propagate SIGTERM to all children. Mutually exclusive with
68    /// `--detach`. Default when neither is set.
69    #[arg(long, conflicts_with_all = ["detach", "stop"])]
70    pub foreground: bool,
71
72    /// Detach mode: write `<ws>/.nros/launch/<bringup>.pids` and return
73    /// immediately. The user later runs `nros launch --stop <pidfile>`.
74    #[arg(long, conflicts_with_all = ["foreground", "stop"])]
75    pub detach: bool,
76
77    /// Stop a previously-detached launch by sending SIGTERM to every PID in
78    /// the given pidfile.
79    #[arg(long, value_name = "PID-FILE")]
80    pub stop: Option<PathBuf>,
81
82    /// Phase 212.L.6 — multi-launch disambiguation. The launcher does not
83    /// drive component spawn from XML today (it spawns from `system.toml`'s
84    /// `[[component]]` list), but accepting `--file` keeps the verb
85    /// surface uniform with `nros plan` / `nros codegen-system`, and
86    /// invokes the shared resolver so bad input fails fast.
87    #[arg(long = "file")]
88    pub file: Option<String>,
89
90    /// Phase 212.L.6 — `<node exec="…">` override for synthesised
91    /// launches. Same uniform-surface motivation as `--file`.
92    #[arg(long = "exec")]
93    pub exec: Option<String>,
94}

96pub fn run(args: Args) -> Result<()> {
97    if let Some(pidfile) = args.stop.as_ref() {
98        return stop_pidfile(pidfile);
99    }
100
101    let workspace_root = args
102        .workspace_root
103        .clone()
104        .map(|p| p.canonicalize().unwrap_or(p))
105        .unwrap_or_else(|| std::env::current_dir().expect("current dir readable for nros launch"));
106
107    let bringup_dir = resolve_bringup_dir(&workspace_root, args.bringup.as_deref())?;
108
109    // Phase 212.L.6 — when `--file` or `--exec` are passed, run the
110    // shared resolver to validate the launch input early. The host
111    // launcher itself spawns from `system.toml::[[component]]`, not the
112    // launch XML, so a successful resolve is purely a sanity check;
113    // a failure surfaces with the resolver's specific error.
114    if args.file.is_some() || args.exec.is_some() {
115        let _ = crate::orchestration::launch_synth::resolve_launch(
116            &bringup_dir,
117            args.file.as_deref(),
118            args.exec.as_deref(),
119        )?;
120    }
121
122    let (system, target_name, target) = load_plan(&bringup_dir, args.target.as_deref())?;
123
124    // Embedded deploy kinds aren't host-spawnable. Surface a clean error.
125    if !is_host_kind(&target.kind) {
126        bail!(
127            "nros launch: [deploy.{target_name}].kind = \"{}\" is not a host target — \
128             use `nros run` for embedded deploys (qemu/flash/…)",
129            target.kind
130        );
131    }
132
133    let locator = system
134        .system
135        .locator
136        .clone()
137        .unwrap_or_else(|| DEFAULT_LOCATOR.to_string());
138    let domain_id = system.system.domain_id;
139
140    let mut commands: Vec<PreparedCommand> = Vec::new();
141    for comp in &system.components {
142        let prepared =
143            prepare_component(&workspace_root, &args.profile, comp, &locator, domain_id)?;
144        commands.push(prepared);
145    }
146
147    if commands.is_empty() {
148        bail!(
149            "nros launch: bringup {} declares no [[component]] entries — nothing to spawn",
150            bringup_dir.display()
151        );
152    }
153
154    // Default to foreground when neither flag is set.
155    if args.detach {
156        spawn_detached(&workspace_root, &bringup_dir, &commands)
157    } else {
158        spawn_foreground(&commands)
159    }
160}

162/// Resolve the bringup dir from an explicit arg + workspace metadata fallback.
163fn resolve_bringup_dir(workspace_root: &Path, arg: Option<&str>) -> Result<PathBuf> {
164    if let Some(raw) = arg {
165        // Absolute / relative path — accept directly when it's a dir.
166        let as_path = Path::new(raw);
167        let candidate = if as_path.is_absolute() {
168            as_path.to_path_buf()
169        } else {
170            workspace_root.join(as_path)
171        };
172        if candidate.is_dir() {
173            return Ok(candidate);
174        }
175        // Otherwise treat as a pkg name relative to workspace root.
176        let by_name = workspace_root.join(raw);
177        if by_name.is_dir() {
178            return Ok(by_name);
179        }
180        bail!(
181            "nros launch: bringup `{raw}` not found under {}",
182            workspace_root.display()
183        );
184    }
185
186    // Fall back to [workspace.metadata.nros].default_system.
187    let cargo_toml = workspace_root.join("Cargo.toml");
188    if !cargo_toml.is_file() {
189        bail!(
190            "nros launch: no bringup arg given and no Cargo.toml at {} — \
191             pass <bringup> explicitly",
192            workspace_root.display()
193        );
194    }
195    let raw = fs::read_to_string(&cargo_toml)
196        .wrap_err_with(|| format!("read {}", cargo_toml.display()))?;
197    let default_system = read_default_system(&raw).ok_or_else(|| {
198        eyre!(
199            "nros launch: no `[workspace.metadata.nros] default_system` in {} — \
200             pass <bringup> explicitly",
201            cargo_toml.display()
202        )
203    })?;
204    let dir = workspace_root.join(&default_system);
205    if !dir.is_dir() {
206        bail!(
207            "nros launch: default_system = \"{default_system}\" but {} is not a directory",
208            dir.display()
209        );
210    }
211    Ok(dir)
212}

214/// Pull the `default_system` field from a workspace `Cargo.toml` (in the
215/// `[workspace.metadata.nros]` table). Returns `None` when absent.
216fn read_default_system(cargo_toml_raw: &str) -> Option<String> {
217    #[derive(serde::Deserialize)]
218    struct Outer {
219        workspace: Option<WorkspaceTable>,
220    }
221    #[derive(serde::Deserialize)]
222    struct WorkspaceTable {
223        metadata: Option<MetadataTable>,
224    }
225    #[derive(serde::Deserialize)]
226    struct MetadataTable {
227        nros: Option<WorkspaceMetadataNros>,
228    }
229    let outer: Outer = toml::from_str(cargo_toml_raw).ok()?;
230    outer.workspace?.metadata?.nros?.default_system
231}

233/// Load the bringup's `system.toml` and pick a deploy target.
234fn load_plan(
235    bringup_dir: &Path,
236    target_arg: Option<&str>,
237) -> Result<(SystemToml, String, DeployTargetView)> {
238    let system_toml_path = bringup_dir.join("system.toml");
239    let raw = fs::read_to_string(&system_toml_path)
240        .wrap_err_with(|| format!("read {}", system_toml_path.display()))?;
241    let system: SystemToml =
242        toml::from_str(&raw).wrap_err_with(|| format!("parse {}", system_toml_path.display()))?;
243
244    // Pick the target. Phase 212.J.2 resolution order:
245    //   1. explicit `--target` arg,
246    //   2. `[system].default_target`,
247    //   3. `[deploy.native]` if present,
248    //   4. first deploy entry in sorted order (BTreeMap keys are sorted),
249    //   5. synthesise a `native` self-host default when `[deploy]` is empty.
250    let (target_name, target) = match target_arg {
251        Some(name) => {
252            let t = system.deploy.get(name).ok_or_else(|| {
253                eyre!(
254                    "nros launch: no [deploy.{name}] in {}",
255                    system_toml_path.display()
256                )
257            })?;
258            (
259                name.to_string(),
260                DeployTargetView {
261                    kind: t.kind.clone().unwrap_or_default(),
262                },
263            )
264        }
265        None => {
266            if let Some(default_name) = system.system.default_target.as_deref() {
267                let t = system.deploy.get(default_name).ok_or_else(|| {
268                    eyre!(
269                        "nros launch: [system].default_target = \"{default_name}\" but no \
270                         matching [deploy.{default_name}] in {}",
271                        system_toml_path.display()
272                    )
273                })?;
274                (
275                    default_name.to_string(),
276                    DeployTargetView {
277                        kind: t.kind.clone().unwrap_or_default(),
278                    },
279                )
280            } else if let Some(t) = system.deploy.get("native") {
281                (
282                    "native".to_string(),
283                    DeployTargetView {
284                        kind: t.kind.clone().unwrap_or_default(),
285                    },
286                )
287            } else {
288                match system.deploy.iter().next() {
289                    Some((n, t)) => (
290                        n.clone(),
291                        DeployTargetView {
292                            kind: t.kind.clone().unwrap_or_default(),
293                        },
294                    ),
295                    None => (
296                        "native".to_string(),
297                        DeployTargetView {
298                            kind: "self".to_string(),
299                        },
300                    ),
301                }
302            }
303        }
304    };
305    Ok((system, target_name, target))
306}

315/// Host-spawnable deploy kinds. The launcher is desktop / native_sim only.
316fn is_host_kind(kind: &str) -> bool {
317    matches!(kind, "self" | "native" | "host" | "")

320/// One prepared `Command` for a component, with the env baked in.
321struct PreparedCommand {
322    pkg: String,
323    binary: PathBuf,
324    env: Vec<(String, String)>,
325}
326
327fn prepare_component(
328    workspace_root: &Path,
329    profile: &str,
330    comp: &crate::orchestration::cargo_metadata_schema::SystemComponentEntry,
331    locator: &str,
332    domain_id: u32,
333) -> Result<PreparedCommand> {
334    let binary = resolve_binary(workspace_root, profile, &comp.pkg);
335    let (params, remaps, namespace) = load_component_meta(workspace_root, &comp.pkg, &comp.class)?;
336
337    let mut env: Vec<(String, String)> = Vec::new();
338    env.push(("NROS_LOCATOR".to_string(), locator.to_string()));
339    env.push(("ROS_DOMAIN_ID".to_string(), domain_id.to_string()));
340    if let Some(ns) = namespace {
341        env.push(("NROS_NAMESPACE".to_string(), ns));
342    }
343    if !params.is_empty() {
344        env.push(("NROS_PARAMETERS".to_string(), encode_params(&params)));
345    }
346    if !remaps.is_empty() {
347        env.push(("NROS_REMAPS".to_string(), encode_remaps(&remaps)));
348    }
349    Ok(PreparedCommand {
350        pkg: comp.pkg.clone(),
351        binary,
352        env,
353    })
354}

356fn resolve_binary(workspace_root: &Path, profile: &str, pkg: &str) -> PathBuf {
357    workspace_root.join("target").join(profile).join(pkg)

360/// Read `<pkg>/Cargo.toml` and surface the resolved component's parameters /
361/// remaps / default_namespace. Single-shape vs multi-shape:
362/// for multi-shape we match by `class` (the system.toml component's class
363/// field; for the multi case the user encodes the variant name in `class`,
364/// e.g. `talker_pkg::Talker` → look up `Talker`).
365fn load_component_meta(
366    workspace_root: &Path,
367    pkg: &str,
368    class: &str,
369) -> Result<(
370    std::collections::BTreeMap<String, toml::Value>,
371    Vec<crate::orchestration::schema::RemapRule>,
372    Option<String>,
373)> {
374    let cargo_toml = workspace_root.join(pkg).join("Cargo.toml");
375    if !cargo_toml.is_file() {
376        // Missing component manifest is not fatal — the user may have a
377        // standalone binary that doesn't carry the metadata table. We still
378        // spawn it with just NROS_LOCATOR/ROS_DOMAIN_ID.
379        return Ok((Default::default(), Vec::new(), None));
380    }
381    let raw = fs::read_to_string(&cargo_toml)
382        .wrap_err_with(|| format!("read {}", cargo_toml.display()))?;
383
384    #[derive(serde::Deserialize)]
385    struct Outer {
386        package: Option<PackageTable>,
387    }
388    #[derive(serde::Deserialize)]
389    struct PackageTable {
390        metadata: Option<MetadataTable>,
391    }
392    #[derive(serde::Deserialize)]
393    struct MetadataTable {
394        nros: Option<PackageMetadataNros>,
395    }
396    let outer: Outer =
397        toml::from_str(&raw).wrap_err_with(|| format!("parse {}", cargo_toml.display()))?;
398    let Some(meta) = outer.package.and_then(|p| p.metadata).and_then(|m| m.nros) else {
399        return Ok((Default::default(), Vec::new(), None));
400    };
401    if let Err(msg) = meta.validate() {
402        bail!(
403            "nros launch: invalid [package.metadata.nros] in {}: {msg}",
404            cargo_toml.display()
405        );
406    }
407    let resolved: Option<ComponentMetadata> = if let Some(c) = meta.component {
408        Some(c)
409    } else if !meta.components.is_empty() {
410        // Match by short class name (the part after `::`).
411        let short = class.rsplit("::").next().unwrap_or(class);
412        meta.components.get(short).cloned()
413    } else {
414        None
415    };
416    Ok(match resolved {
417        Some(c) => (c.parameters, c.remaps, c.default_namespace),
418        None => (Default::default(), Vec::new(), None),
419    })
420}

422/// Encode parameters as `key=value;…` (TOML-encoded value). Stable env shape
423/// the launched binary can parse; the binary owns its own param schema.
424fn encode_params(params: &std::collections::BTreeMap<String, toml::Value>) -> String {
425    params
426        .iter()
427        .map(|(k, v)| format!("{k}={}", encode_toml_value(v)))
428        .collect::<Vec<_>>()
429        .join(";")
430}
431
432fn encode_toml_value(v: &toml::Value) -> String {
433    match v {
434        toml::Value::String(s) => s.clone(),
435        other => other.to_string(),
436    }
437}

439fn encode_remaps(remaps: &[crate::orchestration::schema::RemapRule]) -> String {
440    remaps
441        .iter()
442        .map(|r| format!("{}:={}", r.from, r.to))
443        .collect::<Vec<_>>()
444        .join(";")
445}

447// ---------------------------------------------------------------------------
448// Foreground spawn — blocks until first child exits or SIGINT/SIGTERM.
449// ---------------------------------------------------------------------------
450
451fn spawn_foreground(commands: &[PreparedCommand]) -> Result<()> {
452    let children = Arc::new(Mutex::new(Vec::<Child>::new()));
453    for cmd in commands {
454        let child = spawn_one(cmd, false)?;
455        eprintln!("nros launch: spawned {} (pid {})", cmd.pkg, child.id());
456        children.lock().unwrap().push(child);
457    }
458
459    let stop = install_signal_handler()?;
460
461    // Poll loop: stop on first child exit OR signal.
462    loop {
463        if stop.load(Ordering::SeqCst) {
464            break;
465        }
466        let mut guard = children.lock().unwrap();
467        let mut any_exited = false;
468        for child in guard.iter_mut() {
469            match child.try_wait() {
470                Ok(Some(_)) => {
471                    any_exited = true;
472                    break;
473                }
474                Ok(None) => {}
475                Err(e) => {
476                    eprintln!("nros launch: try_wait failed: {e}");
477                    any_exited = true;
478                    break;
479                }
480            }
481        }
482        drop(guard);
483        if any_exited {
484            break;
485        }
486        std::thread::sleep(Duration::from_millis(50));
487    }
488
489    // Phase 212.J.3 — Propagate SIGTERM, wait up to GRACE_PERIOD for clean
490    // shutdown, then escalate to SIGKILL for stragglers.
491    const GRACE_PERIOD: Duration = Duration::from_secs(5);
492    let mut guard = children.lock().unwrap();
493    for child in guard.iter_mut() {
494        send_sigterm(child.id() as i32);
495    }
496    let deadline = std::time::Instant::now() + GRACE_PERIOD;
497    loop {
498        let mut all_done = true;
499        for child in guard.iter_mut() {
500            match child.try_wait() {
501                Ok(Some(_)) | Err(_) => {}
502                Ok(None) => all_done = false,
503            }
504        }
505        if all_done || std::time::Instant::now() >= deadline {
506            break;
507        }
508        std::thread::sleep(Duration::from_millis(50));
509    }
510    // Escalate: SIGKILL anything still alive, then reap.
511    for child in guard.iter_mut() {
512        if matches!(child.try_wait(), Ok(None)) {
513            send_sigkill(child.id() as i32);
514        }
515    }
516    for child in guard.iter_mut() {
517        let _ = child.wait();
518    }
519    Ok(())
520}

522/// Install a SIGINT/SIGTERM handler that flips a shared flag. The flag is a
523/// process-global `OnceLock<Arc<AtomicBool>>` — the libc signal handler is
524/// extern "C" + has no closure state, so the flag must live in static storage.
525/// First call installs the handler; subsequent calls just hand back the same
526/// flag (so `run()` can be re-invoked from tests).
527#[cfg(unix)]
528fn install_signal_handler() -> Result<Arc<AtomicBool>> {
529    use std::sync::OnceLock;
530    static SIGNAL_FLAG: OnceLock<Arc<AtomicBool>> = OnceLock::new();
531
532    extern "C" fn handler(_sig: libc::c_int) {
533        // Async-signal-safe: only an atomic store on an already-init OnceLock.
534        if let Some(f) = STATIC_FLAG_FOR_HANDLER.get() {
535            f.store(true, Ordering::SeqCst);
536        }
537    }
538    // Mirror of SIGNAL_FLAG, accessible to the handler. They point at the
539    // same Arc once the OnceLock is initialised.
540    static STATIC_FLAG_FOR_HANDLER: OnceLock<Arc<AtomicBool>> = OnceLock::new();
541
542    let flag = SIGNAL_FLAG
543        .get_or_init(|| {
544            let f = Arc::new(AtomicBool::new(false));
545            let _ = STATIC_FLAG_FOR_HANDLER.set(f.clone());
546            unsafe {
547                libc::signal(libc::SIGINT, handler as *const () as libc::sighandler_t);
548                libc::signal(libc::SIGTERM, handler as *const () as libc::sighandler_t);
549            }
550            f
551        })
552        .clone();
553    // Each invocation resets the flag so a prior signal doesn't poison the
554    // next run (relevant in tests).
555    flag.store(false, Ordering::SeqCst);
556    Ok(flag)
557}
558
559#[cfg(not(unix))]
560fn install_signal_handler() -> Result<Arc<AtomicBool>> {
561    Ok(Arc::new(AtomicBool::new(false)))
562}

564#[cfg(unix)]
565fn send_sigterm(pid: i32) {
566    unsafe {
567        let _ = libc::kill(pid, libc::SIGTERM);
568    }
569}
570
571#[cfg(not(unix))]
572fn send_sigterm(_pid: i32) {}
573
574#[cfg(unix)]
575fn send_sigkill(pid: i32) {
576    unsafe {
577        let _ = libc::kill(pid, libc::SIGKILL);
578    }
579}

584// ---------------------------------------------------------------------------
585// Detach spawn — write PID file, return immediately.
586// ---------------------------------------------------------------------------
587
588fn spawn_detached(
589    workspace_root: &Path,
590    bringup_dir: &Path,
591    commands: &[PreparedCommand],
592) -> Result<()> {
593    let mut pids: Vec<u32> = Vec::with_capacity(commands.len());
594    let mut spawned: Vec<Child> = Vec::with_capacity(commands.len());
595    for cmd in commands {
596        let child = spawn_one(cmd, true)?;
597        pids.push(child.id());
598        eprintln!("nros launch: spawned {} (pid {})", cmd.pkg, child.id());
599        spawned.push(child);
600    }
601
602    let pidfile = pidfile_path(workspace_root, bringup_dir);
603    if let Some(parent) = pidfile.parent() {
604        fs::create_dir_all(parent).wrap_err_with(|| format!("create {}", parent.display()))?;
605    }
606    let body = render_pidfile(std::process::id(), &pids);
607    fs::write(&pidfile, body).wrap_err_with(|| format!("write {}", pidfile.display()))?;
608    eprintln!("nros launch: detached; pidfile {}", pidfile.display());
609
610    // We intentionally drop `spawned` without waiting — the children are
611    // now daemonized from this process's perspective.
612    std::mem::forget(spawned);
613    Ok(())
614}
615
616fn pidfile_path(workspace_root: &Path, bringup_dir: &Path) -> PathBuf {
617    let stem = bringup_dir
618        .file_name()
619        .and_then(|n| n.to_str())
620        .unwrap_or("bringup");
621    // Phase 212.J.3 — `.nros/launch/<bringup>.pids` (NOT under `target/`,
622    // which `cargo clean` blasts away mid-run).
623    workspace_root
624        .join(".nros")
625        .join("launch")
626        .join(format!("{stem}.pids"))
627}

629/// Pidfile shape: one PID per line. First line is the launcher's own PID
630/// (so `--stop` can be filtered to ignore it if needed); subsequent lines
631/// are children, one per `[[component]]` in declaration order.
632fn render_pidfile(parent: u32, children: &[u32]) -> String {
633    let mut s = String::new();
634    s.push_str(&format!("# nros launch — Phase 212.J\nparent={parent}\n"));
635    for pid in children {
636        s.push_str(&format!("{pid}\n"));
637    }
638    s
639}
640
641fn parse_pidfile(body: &str) -> Vec<i32> {
642    body.lines()
643        .filter_map(|line| {
644            let t = line.trim();
645            if t.is_empty() || t.starts_with('#') || t.starts_with("parent=") {
646                None
647            } else {
648                t.parse::<i32>().ok()
649            }
650        })
651        .collect()
652}
653
654fn stop_pidfile(pidfile: &Path) -> Result<()> {
655    let body =
656        fs::read_to_string(pidfile).wrap_err_with(|| format!("read {}", pidfile.display()))?;
657    let pids = parse_pidfile(&body);
658    if pids.is_empty() {
659        bail!("nros launch --stop: {} has no PIDs", pidfile.display());
660    }
661    for pid in pids {
662        send_sigterm(pid);
663        eprintln!("nros launch: sent SIGTERM to {pid}");
664    }
665    // Phase 212.J.3.b — drop the pidfile so a second `--stop` doesn't
666    // try to re-signal stale PIDs (the OS may have recycled them).
667    if let Err(e) = fs::remove_file(pidfile) {
668        eprintln!(
669            "nros launch: warning: failed to remove {}: {e}",
670            pidfile.display()
671        );
672    }
673    Ok(())
674}

676/// Spawn one prepared command. `detached` suppresses inherited stdio so the
677/// child survives a parent exit cleanly.
678fn spawn_one(cmd: &PreparedCommand, detached: bool) -> Result<Child> {
679    let mut c = Command::new(&cmd.binary);
680    // Clear inherited env then re-add explicitly — the launcher is the SSOT
681    // for runtime env (NROS_LOCATOR / ROS_DOMAIN_ID / params / remaps). The
682    // caller's PATH is preserved because the binary is invoked by absolute
683    // path (no PATH search).
684    c.env_clear();
685    // Preserve a minimal env so children can find shared libs / use sh
686    // (`LD_LIBRARY_PATH`, `PATH`, `HOME`, locale).
687    for key in ["PATH", "HOME", "USER", "LANG", "LC_ALL", "LD_LIBRARY_PATH"] {
688        if let Ok(v) = std::env::var(key) {
689            c.env(key, v);
690        }
691    }
692    for (k, v) in &cmd.env {
693        c.env(k, v);
694    }
695    if detached {
696        c.stdin(Stdio::null())
697            .stdout(Stdio::null())
698            .stderr(Stdio::null());
699    } else {
700        c.stdin(Stdio::null())
701            .stdout(Stdio::inherit())
702            .stderr(Stdio::inherit());
703    }
704    c.spawn().wrap_err_with(|| {
705        format!(
706            "nros launch: spawn {} ({}) failed — is the binary built? \
707             try `cargo build -p {}`",
708            cmd.pkg,
709            cmd.binary.display(),
710            cmd.pkg
711        )
712    })
713}
