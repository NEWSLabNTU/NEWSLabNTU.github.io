15/// Mirrors `nros::MISSING_COMPONENT_EXPORT_ERROR` (in
16/// `packages/core/nros/src/component.rs`) so host-side diagnostics
17/// surface the same human-readable phrase as the in-tree
18/// `ComponentError::MissingExport` runtime variant. Held as a
19/// `const` here to keep the CLI off the `nros` build dependency
20/// closure (the latter is `no_std` + target-feature-gated).
21pub(crate) const MISSING_COMPONENT_EXPORT_ERROR: &str = "package has no exported nros component";
22
23#[derive(Debug, ClapArgs)]
24pub struct Args {
25    /// System package name used for build/<system_pkg>/nros output
26    pub system_pkg: String,
27
28    /// Workspace root containing colcon-like src/* packages
29    #[arg(long)]
30    pub workspace: Option<PathBuf>,
31
32    /// Output root for orchestration artifacts
33    #[arg(long)]
34    pub out_dir: Option<PathBuf>,
35
36    /// Existing source metadata JSON to validate and preserve
37    #[arg(long = "metadata")]
38    pub metadata: Vec<PathBuf>,
39
40    /// Phase 172.E — produce missing source metadata by compiling + running
41    /// each declared component in metadata mode (the metadata-mode build).
42    #[arg(long)]
43    pub build: bool,
44
45    /// nano-ros workspace root for the metadata harness's `nros` dep
46    /// (`--build`); falls back to the `NROS_WORKSPACE` env var.
47    #[arg(long)]
48    pub nano_ros_workspace: Option<PathBuf>,
49}

51/// Derive the metadata-mode build options for a declared component. The
52/// component id is `crate::module` (the registered type is `::Component`);
53/// the metadata header name is its last segment.
54fn metadata_build_options(
55    decl: &ComponentDeclaration,
56    nano_ros: &Path,
57    out_root: &Path,
58) -> MetadataBuildOptions {
59    let id = decl.config.component.clone();
60    let name = id.rsplit("::").next().unwrap_or(&id).to_string();
61    let probe = out_root.join("metadata-probe").join(id.replace("::", "__"));
62    // W.3 (Phase 172): a minimal `[component]` may omit `[linkage]` — derive the
63    // executable + exported symbol from the component name / crate convention.
64    MetadataBuildOptions {
65        component_id: id,
66        package: decl.config.package.clone(),
67        executable: Some(decl.config.linkage.resolved_executable(&name)),
68        exported_symbol: Some(decl.config.linkage.resolved_exported_symbol(&name)),
69        component: name,
70        component_dir: decl.package_root.clone(),
71        nano_ros_workspace: nano_ros.to_path_buf(),
72        output_path: decl.source_metadata_path(),
73        harness_dir: probe,
74    }
75}

77pub fn run(args: Args) -> Result<()> {
78    let root = args.workspace.unwrap_or(std::env::current_dir()?);
79    let out_root = args
80        .out_dir
81        .unwrap_or_else(|| root.join("build").join(&args.system_pkg).join("nros"));
82    let metadata_dir = out_root.join("metadata");
83    fs::create_dir_all(&metadata_dir)?;
84
85    let workspace = Workspace::discover(&root)?;
86
87    // Phase 126.B.7 acceptance — every package that declared itself
88    // a nros component (via `component_nros.toml`) must have actually
89    // produced its source-metadata JSON. A missing file is the host-
90    // side surface for "forgot to write `nros::component!`": the
91    // metadata-mode binary either failed to build or built but exited
92    // before writing the JSON, both shapes leave the declared
93    // `[metadata].source_metadata` path empty. Catch the case here
94    // with the same diagnostic string the in-tree
95    // `ComponentError::MissingExport` runtime variant uses.
96    let declarations = workspace.component_declarations()?;
97    let mut missing: Vec<&ComponentDeclaration> = declarations
98        .iter()
99        .filter(|decl| !decl.source_metadata_path().is_file())
100        .collect();
101
102    // Phase 172.E — `--build` produces the missing source metadata by
103    // compiling + running each declared component in metadata mode.
104    let mut built: Vec<PathBuf> = Vec::new();
105    if !missing.is_empty() && args.build {
106        let nano_ros = args
107            .nano_ros_workspace
108            .clone()
109            .or_else(|| std::env::var_os("NROS_WORKSPACE").map(PathBuf::from))
110            .ok_or_else(|| {
111                eyre!(
112                    "`nros metadata --build` needs the nano-ros workspace — pass \
113                     --nano-ros-workspace <path> or set NROS_WORKSPACE"
114                )
115            })?;
116        for decl in &missing {
117            let opts = metadata_build_options(decl, &nano_ros, &out_root);
118            build_metadata(&opts)
119                .wrap_err_with(|| format!("build source metadata for `{}`", decl.config.package))?;
120            built.push(decl.source_metadata_path());
121        }
122        missing.clear();
123    }
124
125    if !missing.is_empty() {
126        let mut msg = String::from(MISSING_COMPONENT_EXPORT_ERROR);
127        for decl in &missing {
128            msg.push_str(&format!(
129                "\n  - package `{}`: expected source metadata at {}",
130                decl.config.package,
131                decl.source_metadata_path().display()
132            ));
133        }
134        msg.push_str(
135            "\n  hint: add `nros::component!(YourComponent);` to the package's `lib.rs`/`main.rs`, \
136             then run `nros metadata --build` to produce it.",
137        );
138        bail!(msg);
139    }
140
141    // Collect: explicit `--metadata`, then anything `--build` just produced,
142    // then (if neither) the workspace's discovered source-metadata files.
143    let mut inputs = args.metadata;
144    inputs.extend(built);
145    if inputs.is_empty() {
146        inputs.extend(workspace.source_metadata_files());
147    }
148
149    for path in &inputs {
150        let raw = fs::read_to_string(path)
151            .wrap_err_with(|| format!("failed to read source metadata {}", path.display()))?;
152        let _: Value = serde_json::from_str(&raw)
153            .wrap_err_with(|| format!("invalid source metadata JSON {}", path.display()))?;
154        let file_name = path
155            .file_name()
156            .ok_or_else(|| eyre!("metadata path has no file name: {}", path.display()))?;
157        fs::write(metadata_dir.join(file_name), raw)?;
158    }
159
160    eprintln!(
161        "nros metadata: preserved {} metadata artifact(s) in {}",
162        inputs.len(),
163        metadata_dir.display()
164    );
165    Ok(())
166}

168#[cfg(test)]
169mod tests {
170    use super::*;
171    use crate::orchestration::{
172        config::{ComponentConfig, ComponentLinkage, ComponentMetadataConfig, ComponentOverrides},
173        source_metadata::ComponentLanguage,
174        workspace::ComponentDeclaration,
175    };
176
177    #[test]
178    fn metadata_build_options_derives_id_name_and_paths() {
179        let decl = ComponentDeclaration {
180            package_root: PathBuf::from("/ws/src/demo_pkg"),
181            manifest_path: PathBuf::from("/ws/src/demo_pkg/component_nros.toml"),
182            config: ComponentConfig {
183                version: 1,
184                package: "demo_pkg".into(),
185                component: "demo_pkg::talker".into(),
186                language: ComponentLanguage::Rust,
187                linkage: ComponentLinkage {
188                    crate_name: Some("demo_pkg".into()),
189                    executable: Some("talker".into()),
190                    exported_symbol: Some("nros_component_talker".into()),
191                    static_library: None,
192                },
193                metadata: ComponentMetadataConfig {
194                    source_metadata: "talker.metadata.json".into(),
195                    generated_by: None,
196                },
197                overrides: ComponentOverrides {
198                    default_namespace: None,
199                    parameters: Default::default(),
200                    remaps: Vec::new(),
201                },
202            },
203        };
204        let o = metadata_build_options(&decl, Path::new("/nano-ros"), Path::new("/out"));
205        assert_eq!(o.component_id, "demo_pkg::talker");
206        assert_eq!(o.component, "talker"); // last segment of the id
207        assert_eq!(o.package, "demo_pkg");
208        assert_eq!(o.executable.as_deref(), Some("talker"));
209        assert_eq!(o.exported_symbol.as_deref(), Some("nros_component_talker"));
210        assert_eq!(o.component_dir, PathBuf::from("/ws/src/demo_pkg"));
211        assert_eq!(o.nano_ros_workspace, PathBuf::from("/nano-ros"));
212        // source_metadata resolves relative to the package root.
213        assert_eq!(
214            o.output_path,
215            PathBuf::from("/ws/src/demo_pkg/talker.metadata.json")
216        );
217        // probe dir is per-component under the out root.
218        assert_eq!(
219            o.harness_dir,
220            PathBuf::from("/out/metadata-probe/demo_pkg__talker")
221        );
222    }

228        let decl = ComponentDeclaration {

237                metadata: ComponentMetadataConfig {

239                    generated_by: None,

244        let o = metadata_build_options(&decl, Path::new("/nano-ros"), Path::new("/out"));
245        assert_eq!(o.executable.as_deref(), Some("talker"));
246        assert_eq!(o.exported_symbol.as_deref(), Some("nros_component_talker"));
247    }
