1//! Phase 212.I — `nros migrate workspace` migration tool.
2//!
3//! Converts a pre-212 workspace (root `nros.toml` + per-package
4//! `component_nros.toml` / `nros/components/*.toml` + committed `metadata/*.json`
5//! + hand-maintained `package.xml`) to the post-212 shape (workspace-root
6//! `Cargo.toml` `[workspace.metadata.nros]`, per-package
7//! `Cargo.toml.[package.metadata.nros.node(s)]` +
8//! `[package.metadata.ament]`, regenerated `package.xml`, dedicated
9//! `<bringup>/system.toml` + `<bringup>/launch/system.launch.xml`).
10//!
11//! Post-N.12 (Component → Node rename, 2026-06-03): the migrator emits the
12//! **new** `node` / `nodes` spelling for `[package.metadata.nros]` entries —
13//! this verb's whole purpose is to land a current-shape workspace, so it
14//! writes the post-rename terminology even though the legacy input files we
15//! read still call the field `component` / `components`. Pre-212
16//! `component_nros.toml` inputs keep the legacy `component = "..."` field
17//! name (pre-rename); we accept `node = "..."` as a deserialize alias so a
18//! partially-edited pre-212 tree still migrates.
19//!
20//! See `docs/roadmap/phase-212-ux-cargo-native-and-file-consolidation.md` §212.I.
21

41#[derive(Debug, ClapArgs)]
42pub struct Args {
43    /// Workspace root containing the pre-212 `nros.toml` (default: cwd).
44    #[arg(default_value = ".")]
45    pub dir: PathBuf,
46
47    /// Print the migration plan, do not modify any files.
48    #[arg(long)]
49    pub dry_run: bool,
50
51    /// Override the auto-derived bringup package name
52    /// (`<system_name>_bringup`).
53    #[arg(long)]
54    pub bringup_name: Option<String>,
55
56    /// Re-run on an already-migrated tree.
57    #[arg(long)]
58    pub force: bool,
59}
60
61pub fn run(args: Args) -> Result<()> {
62    let ws = args.dir.canonicalize().unwrap_or(args.dir.clone());
63
64    if !args.force && is_already_migrated(&ws) {
65        println!(
66            "nros migrate workspace: {} already migrated (workspace.metadata.nros present); \
67             pass --force to re-run.",
68            ws.display()
69        );
70        return Ok(());
71    }
72
73    let plan = build_plan(&ws, args.bringup_name.as_deref())?;
74    print_plan(&plan);
75
76    if args.dry_run {
77        println!("nros migrate workspace: --dry-run, no files written.");
78        return Ok(());
79    }
80
81    apply_plan(&plan)?;
82    println!(
83        "nros migrate workspace: migrated {} ({} step(s)).",
84        ws.display(),
85        plan.steps.len()
86    );
87    Ok(())
88}

121/// Permissive — pre-212 manifests carry various extra fields we don't read.
122///
123/// Post-N.12 the canonical instance-name field is `node`, but pre-212 files
124/// were authored with `component`. We accept either spelling via a serde
125/// alias — pre-212 files that have already been hand-edited to the new
126/// `node = "..."` field name still parse, while untouched pre-212 files keep

190fn build_plan(ws: &Path, bringup_override: Option<&str>) -> Result<MigrationPlan> {
191    let nros_toml = ws.join("nros.toml");
192    if !nros_toml.is_file() {
193        bail!(
194            "no pre-212 nros.toml at {} — nothing to migrate",
195            nros_toml.display()
196        );
197    }
198    let raw =
199        fs::read_to_string(&nros_toml).with_context(|| format!("read {}", nros_toml.display()))?;
200    let legacy: LegacyWorkspaceNrosToml =
201        toml::from_str(&raw).with_context(|| format!("parse {}", nros_toml.display()))?;
202
203    // System name derivation: the spec says default = `<system_name_from_nros_toml>_bringup`.
204    // The pre-212 schema has no `[system].name`, so we prefer the first
205    // component package name (matches the e2e/composable fixtures' implicit
206    // naming) and fall back to the launch-file stem when no components are
207    // declared yet.
208    let system_name = legacy
209        .system
210        .components
211        .first()
212        .cloned()
213        .or_else(|| {
214            legacy
215                .system
216                .launch
217                .as_deref()
218                .and_then(|p| Path::new(p).file_stem().and_then(|n| n.to_str()))
219                .map(|s| s.trim_end_matches(".launch").to_string())
220        })
221        .unwrap_or_else(|| "demo".to_string());
222
223    let bringup_pkg_name = bringup_override
224        .map(str::to_string)
225        .unwrap_or_else(|| format!("{system_name}_bringup"));
226    let bringup_dir = ws.join(&bringup_pkg_name);
227
228    // Walk component packages.
229    let mut pkg_dirs = Vec::new();
230    let mut components_by_pkg: BTreeMap<PathBuf, Vec<PreComponent>> = BTreeMap::new();
231    for pkg_name in &legacy.system.components {
232        let pkg_dir = ws.join("src").join(pkg_name);
233        if !pkg_dir.is_dir() {
234            bail!(
235                "component pkg dir not found: {} (referenced in [system].components)",
236                pkg_dir.display()
237            );
238        }
239        let comps = discover_components_for_pkg(&pkg_dir)?;
240        if comps.is_empty() {
241            bail!(
242                "no component manifests found under {} (looked for component_nros.toml + \
243                 nros/components/*.toml)",
244                pkg_dir.display()
245            );
246        }
247        components_by_pkg.insert(pkg_dir.clone(), comps);
248        pkg_dirs.push(pkg_dir);
249    }
250
251    // Resolve launch source. The pre-212 fixtures keep it in
252    // `<first pkg>/launch/system.launch.xml`. We move it into the bringup.
253    let launch_src = legacy
254        .system
255        .launch
256        .as_deref()
257        .map(|p| ws.join(p))
258        .filter(|p| p.is_file());
259
260    // Convert [deploy] tables. We keep keys + parse what we can.
261    let deploy = legacy
262        .deploy
263        .iter()
264        .filter_map(|(name, val)| convert_deploy(val).map(|d| (name.clone(), d)))
265        .collect();
266
267    let system_header = SystemHeader {
268        name: system_name,
269        rmw: legacy.system.rmw.unwrap_or_else(|| "zenoh".into()),
270        domain_id: legacy.system.domain_id.unwrap_or(0),
271        locator: legacy.system.locator,
272        default_launch: None,
273        default_target: None,
274    };
275
276    let mut steps = vec![Step::WriteSystemToml, Step::WriteBringupPackageXml];
277    if let Some(src) = &launch_src {
278        steps.push(Step::MoveLaunchXml {
279            from: src.clone(),
280            to: bringup_dir.join("launch").join("system.launch.xml"),
281        });
282    }
283    for pkg_dir in &pkg_dirs {
284        steps.push(Step::PatchPackageCargoToml {
285            pkg_dir: pkg_dir.clone(),
286        });
287        // Delete component manifests for this pkg.
288        for comp in &components_by_pkg[pkg_dir] {
289            steps.push(Step::DeleteComponentManifest {
290                path: comp.src.clone(),
291            });
292        }
293        let metadata_dir = pkg_dir.join("metadata");
294        if metadata_dir.is_dir() {
295            steps.push(Step::DeleteMetadataDir { path: metadata_dir });
296        }
297        steps.push(Step::RegeneratePackageXml {
298            pkg_dir: pkg_dir.clone(),
299        });
300    }
301    steps.push(Step::DeleteNrosToml);
302    steps.push(Step::PatchWorkspaceCargoToml);
303
304    Ok(MigrationPlan {
305        ws: ws.to_path_buf(),
306        bringup_dir,
307        bringup_pkg_name,
308        system_header,
309        deploy,
310        domains: Vec::new(),
311        bridges: Vec::new(),
312        pkg_dirs,
313        components_by_pkg,
314        launch_src,
315        steps,
316    })
317}

319fn convert_deploy(
320    raw: &toml::Value,
321) -> Option<crate::orchestration::cargo_metadata_schema::DeployTarget> {
322    let tbl = raw.as_table()?;
323    let kind = tbl.get("kind")?.as_str()?.to_string();
324    let target = tbl.get("target")?.as_str()?.to_string();
325    Some(crate::orchestration::cargo_metadata_schema::DeployTarget {
326        kind: Some(kind),
327        target: Some(target),
328        launch: tbl
329            .get("launch")
330            .and_then(|v| v.as_str())
331            .map(str::to_string),
332        board: tbl
333            .get("board")
334            .and_then(|v| v.as_str())
335            .map(str::to_string),
336        framework: tbl
337            .get("framework")
338            .and_then(|v| v.as_str())
339            .map(str::to_string),
340        // Phase 255 — migrate a per-deploy rmw override if the legacy block had one.
341        rmw: tbl.get("rmw").and_then(|v| v.as_str()).map(str::to_string),
342    })
343}

345fn discover_components_for_pkg(pkg_dir: &Path) -> Result<Vec<PreComponent>> {
346    let mut out = Vec::new();
347    // Single-component form.
348    let folded = pkg_dir.join("component_nros.toml");
349    if folded.is_file() {
350        let raw =
351            fs::read_to_string(&folded).with_context(|| format!("read {}", folded.display()))?;
352        let cfg: LegacyComponentConfig =
353            toml::from_str(&raw).with_context(|| format!("parse {}", folded.display()))?;
354        out.push(PreComponent { src: folded, cfg });
355    }
356    // Multi-component form.
357    let multi_dir = pkg_dir.join("nros").join("components");
358    if multi_dir.is_dir() {
359        let mut entries: Vec<_> = fs::read_dir(&multi_dir)?
360            .filter_map(|e| e.ok().map(|e| e.path()))
361            .filter(|p| p.extension().and_then(|s| s.to_str()) == Some("toml"))
362            .collect();
363        entries.sort();
364        for path in entries {
365            let raw =
366                fs::read_to_string(&path).with_context(|| format!("read {}", path.display()))?;
367            let cfg: LegacyComponentConfig =
368                toml::from_str(&raw).with_context(|| format!("parse {}", path.display()))?;
369            out.push(PreComponent { src: path, cfg });
370        }
371    }
372    Ok(out)
373}

379fn print_plan(plan: &MigrationPlan) {
380    let ws = &plan.ws;
381    let rel = |p: &Path| -> String { p.strip_prefix(ws).unwrap_or(p).display().to_string() };
382    println!("nros migrate workspace plan:");
383    for step in &plan.steps {
384        match step {
385            Step::WriteSystemToml => println!(
386                "  read    nros.toml → write {}/system.toml",
387                rel(&plan.bringup_dir)
388            ),
389            Step::WriteBringupPackageXml => println!(
390                "  write   {}/package.xml (via nros emit package-xml)",
391                rel(&plan.bringup_dir)
392            ),
393            Step::MoveLaunchXml { from, to } => {
394                println!("  move    {} → {}", rel(from), rel(to))
395            }
396            Step::PatchPackageCargoToml { pkg_dir } => println!(
397                "  patch   {}/Cargo.toml [package.metadata.nros.node(s)] + \
398                 [package.metadata.ament]",
399                rel(pkg_dir)
400            ),
401            Step::DeleteComponentManifest { path } => {
402                println!("  delete  {}", rel(path))
403            }
404            Step::DeleteMetadataDir { path } => {
405                println!("  delete  {}/", rel(path))
406            }
407            Step::RegeneratePackageXml { pkg_dir } => println!(
408                "  write   {}/package.xml (regenerated via nros emit package-xml)",
409                rel(pkg_dir)
410            ),
411            Step::DeleteNrosToml => println!("  delete  nros.toml"),
412            Step::PatchWorkspaceCargoToml => println!(
413                "  patch   Cargo.toml [workspace] exclude = [\"{}\"], [workspace.metadata.nros] \
414                 default_system = \"{}\"",
415                plan.bringup_pkg_name, plan.bringup_pkg_name
416            ),
417        }
418    }
419}

425fn apply_plan(plan: &MigrationPlan) -> Result<()> {
426    // Step 1: build + write system.toml (use SystemToml model).
427    fs::create_dir_all(plan.bringup_dir.join("launch"))
428        .with_context(|| format!("create {}", plan.bringup_dir.display()))?;
429    let system = build_system_toml(plan);
430    let toml_str = toml::to_string_pretty(&system).wrap_err("serialize generated system.toml")?;
431    let header = "# generated by nros migrate workspace — Phase 212.I\n\n";
432    fs::write(
433        plan.bringup_dir.join("system.toml"),
434        format!("{header}{toml_str}"),
435    )?;
436
437    // Step 2: move launch file (if any).
438    if let Some(src) = &plan.launch_src {
439        let dst = plan.bringup_dir.join("launch").join("system.launch.xml");
440        if src != &dst {
441            fs::copy(src, &dst)
442                .with_context(|| format!("copy {} → {}", src.display(), dst.display()))?;
443            fs::remove_file(src).with_context(|| format!("remove old launch {}", src.display()))?;
444        }
445    }
446
447    // Step 3: per-pkg: patch Cargo.toml, delete component manifests, delete metadata/, regen package.xml.
448    for pkg_dir in &plan.pkg_dirs {
449        patch_package_cargo_toml(pkg_dir, &plan.components_by_pkg[pkg_dir])?;
450        for comp in &plan.components_by_pkg[pkg_dir] {
451            if comp.src.is_file() {
452                fs::remove_file(&comp.src)
453                    .with_context(|| format!("remove {}", comp.src.display()))?;
454            }
455        }
456        // Also remove the now-empty nros/components dir.
457        let nros_components = pkg_dir.join("nros").join("components");
458        if nros_components.is_dir() {
459            let _ = fs::remove_dir(&nros_components);
460            let _ = fs::remove_dir(pkg_dir.join("nros"));
461        }
462        let metadata_dir = pkg_dir.join("metadata");
463        if metadata_dir.is_dir() {
464            fs::remove_dir_all(&metadata_dir)
465                .with_context(|| format!("remove {}", metadata_dir.display()))?;
466        }
467        // Regenerate package.xml via emit_package_xml.
468        let regenerated = emit_package_xml::render_for_pkg(pkg_dir)
469            .with_context(|| format!("regenerate {}/package.xml", pkg_dir.display()))?;
470        fs::write(pkg_dir.join("package.xml"), regenerated)
471            .with_context(|| format!("write {}/package.xml", pkg_dir.display()))?;
472    }
473
474    // Step 4: write bringup package.xml AFTER system.toml is on disk.
475    let bringup_pkg_xml = emit_package_xml::render_for_pkg(&plan.bringup_dir)
476        .with_context(|| format!("render {}/package.xml", plan.bringup_dir.display()))?;
477    fs::write(plan.bringup_dir.join("package.xml"), bringup_pkg_xml)?;
478
479    // Step 5: delete root nros.toml.
480    let root_nros = plan.ws.join("nros.toml");
481    if root_nros.is_file() {
482        fs::remove_file(&root_nros).with_context(|| format!("remove {}", root_nros.display()))?;
483    }
484
485    // Step 6: patch (or create) workspace-root Cargo.toml.
486    patch_workspace_cargo_toml(&plan.ws, &plan.bringup_pkg_name, &plan.pkg_dirs)?;
487
488    Ok(())
489}

491fn build_system_toml(plan: &MigrationPlan) -> SystemToml {
492    // Build component entries from each PreComponent.
493    let mut components = Vec::new();
494    for pkg_dir in &plan.pkg_dirs {
495        for c in &plan.components_by_pkg[pkg_dir] {
496            let class = pkg_class_path(&c.cfg);
497            components.push(SystemComponentEntry {
498                pkg: c.cfg.package.clone(),
499                class,
500                name: c.cfg.component.clone(),
501            });
502        }
503    }
504    SystemToml {
505        system: plan.system_header.clone(),
506        components,
507        deploy: plan.deploy.clone(),
508        domains: plan.domains.clone(),
509        bridges: plan.bridges.clone(),
510        tiers: std::collections::BTreeMap::new(),
511        shared_state: Vec::new(),
512        node_overrides: Vec::new(),
513        safety: None,
514        param_services: None,
515        lifecycle: None,
516        param_persistence: None,
517    }
518}

520/// Derive a Rust-module-path style `class` (`<pkg>::<Component>`) from the
521/// legacy linkage block when available, else fall back to `<pkg>::<comp>`.
522fn pkg_class_path(cfg: &LegacyComponentConfig) -> String {
523    if let Some(linkage) = &cfg.linkage
524        && let Some(sym) = linkage
525            .as_table()
526            .and_then(|t| t.get("exported_symbol"))
527            .and_then(|v| v.as_str())
528    {
529        // `nros_component_<name>` → strip the prefix to get the component name.
530        if let Some(stripped) = sym.strip_prefix("nros_component_") {
531            return format!("{}::{}", cfg.package, stripped);
532        }
533    }
534    format!("{}::{}", cfg.package, cfg.component)
535}

541fn patch_package_cargo_toml(pkg_dir: &Path, comps: &[PreComponent]) -> Result<()> {
542    let cargo_toml = pkg_dir.join("Cargo.toml");
543    let mut doc: DocumentMut = if cargo_toml.is_file() {
544        let raw = fs::read_to_string(&cargo_toml)
545            .with_context(|| format!("read {}", cargo_toml.display()))?;
546        raw.parse()
547            .with_context(|| format!("parse {}", cargo_toml.display()))?
548    } else {
549        // Create a stub Cargo.toml so the new pkg shape is self-contained.
550        let pkg_name = comps
551            .first()
552            .map(|c| c.cfg.package.clone())
553            .unwrap_or_else(|| {
554                pkg_dir
555                    .file_name()
556                    .and_then(|n| n.to_str())
557                    .unwrap_or("pkg")
558                    .to_string()
559            });
560        format!("[package]\nname = \"{pkg_name}\"\nversion = \"0.1.0\"\nedition = \"2021\"\n",)
561            .parse()
562            .expect("stub Cargo.toml parses")
563    };
564
565    // Ensure [package] table.
566    let package = doc
567        .entry("package")
568        .or_insert_with(|| Item::Table(Table::new()))
569        .as_table_mut()
570        .ok_or_else(|| eyre::eyre!("[package] is not a table in {}", cargo_toml.display()))?;
571    package
572        .entry("metadata")
573        .or_insert_with(|| Item::Table(Table::new()));
574
575    // Render the nros section.
576    let nros_table = render_nros_metadata_table(comps);
577    doc["package"]["metadata"]["nros"] = Item::Table(nros_table);
578
579    // Render the ament section from package.xml (if present).
580    let pkg_xml_path = pkg_dir.join("package.xml");
581    let ament = if pkg_xml_path.is_file() {
582        extract_ament_from_package_xml(&pkg_xml_path)?
583    } else {
584        PackageMetadataAment::default()
585    };
586    doc["package"]["metadata"]["ament"] = Item::Table(render_ament_metadata_table(&ament));
587
588    fs::write(&cargo_toml, doc.to_string())
589        .with_context(|| format!("write {}", cargo_toml.display()))?;
590    Ok(())
591}

593fn render_nros_metadata_table(comps: &[PreComponent]) -> Table {
594    // Post-N.12 (2026-06-03): emit the new `node` / `nodes` spelling. The
595    // migrator's job is to land a current-shape workspace, so output uses the
596    // post-rename terminology even though the legacy input files we read
597    // still call the field `component` / `components`.
598    let mut nros_tbl = Table::new();
599    if comps.len() == 1 {
600        let c = &comps[0];
601        nros_tbl["node"] = Item::Table(render_component_table(&c.cfg));
602    } else {
603        let mut nodes_tbl = Table::new();
604        nodes_tbl.set_implicit(true);
605        for c in comps {
606            let name = c.cfg.component.clone();
607            nodes_tbl[&name] = Item::Table(render_component_table(&c.cfg));
608        }
609        nros_tbl["nodes"] = Item::Table(nodes_tbl);
610    }
611    nros_tbl
612}

614fn render_component_table(cfg: &LegacyComponentConfig) -> Table {
615    let mut tbl = Table::new();
616    if let Some(ov) = &cfg.overrides {
617        if let Some(ns) = &ov.default_namespace {
618            tbl["default_namespace"] = value(ns);
619        }
620        if !ov.parameters.is_empty() {
621            let mut params_tbl = Table::new();
622            for (k, v) in &ov.parameters {
623                params_tbl[k] = toml_value_to_item(v);
624            }
625            tbl["parameters"] = Item::Table(params_tbl);
626        }
627        if !ov.remaps.is_empty() {
628            let mut arr = toml_edit::ArrayOfTables::new();
629            for r in &ov.remaps {
630                let mut row = Table::new();
631                row["from"] = value(&r.from);
632                row["to"] = value(&r.to);
633                arr.push(row);
634            }
635            tbl["remaps"] = Item::ArrayOfTables(arr);
636        }
637    }
638    tbl
639}

641fn toml_value_to_item(v: &toml::Value) -> Item {
642    // Round-trip through string: keeps the conversion simple + correct.
643    let raw =
644        toml::to_string(&BTreeMap::from([(String::from("v"), v.clone())])).unwrap_or_default();
645    if let Ok(doc) = raw.parse::<DocumentMut>() {
646        if let Some(item) = doc.get("v") {
647            return item.clone();
648        }
649    }
650    value(format!("{v}"))
651}

653fn render_ament_metadata_table(ament: &PackageMetadataAment) -> Table {
654    let mut tbl = Table::new();
655    if !ament.build_depend.is_empty() {
656        let mut a = Array::new();
657        for d in &ament.build_depend {
658            a.push(d);
659        }
660        tbl["build_depend"] = value(a);
661    }
662    if !ament.exec_depend.is_empty() {
663        let mut a = Array::new();
664        for d in &ament.exec_depend {
665            a.push(d);
666        }
667        tbl["exec_depend"] = value(a);
668    }
669    if !ament.test_depend.is_empty() {
670        let mut a = Array::new();
671        for d in &ament.test_depend {
672            a.push(d);
673        }
674        tbl["test_depend"] = value(a);
675    }
676    if let Some(bt) = &ament.build_type {
677        tbl["build_type"] = value(bt);
678    }
679    tbl

682/// Lightweight package.xml dep extractor. Reuses the cargo-nano-ros
683/// `PackageXml` parser but extracts dep entries from raw text — that parser
684/// collapses every dep type into one set, so for the migration we need the
685/// per-tag breakdown.
686fn extract_ament_from_package_xml(path: &Path) -> Result<PackageMetadataAment> {
687    let raw = fs::read_to_string(path).with_context(|| format!("read {}", path.display()))?;
688    let mut out = PackageMetadataAment::default();
689    out.build_depend = extract_tag(&raw, "build_depend");
690    out.exec_depend = extract_tag(&raw, "exec_depend");
691    out.test_depend = extract_tag(&raw, "test_depend");
692    // `<depend>` counts as both build + exec per REP 140.
693    for d in extract_tag(&raw, "depend") {
694        if !out.build_depend.contains(&d) {
695            out.build_depend.push(d.clone());
696        }
697        if !out.exec_depend.contains(&d) {
698            out.exec_depend.push(d);
699        }
700    }
701    // `<export><build_type>…</build_type></export>` → `build_type`.
702    let bts = extract_tag(&raw, "build_type");
703    if let Some(bt) = bts.into_iter().next() {
704        out.build_type = Some(bt);
705    }
706    Ok(out)
707}
708
709fn extract_tag(xml: &str, tag: &str) -> Vec<String> {
710    let open = format!("<{tag}>");
711    let close = format!("</{tag}>");
712    let mut out = Vec::new();
713    let mut cursor = 0;
714    while let Some(start) = xml[cursor..].find(&open) {
715        let body_start = cursor + start + open.len();
716        let Some(end) = xml[body_start..].find(&close) else {
717            break;
718        };
719        let body = xml[body_start..body_start + end].trim().to_string();
720        if !body.is_empty() {
721            out.push(body);
722        }
723        cursor = body_start + end + close.len();
724    }
725    out
726}

728fn patch_workspace_cargo_toml(ws: &Path, bringup_name: &str, pkg_dirs: &[PathBuf]) -> Result<()> {
729    let cargo_toml = ws.join("Cargo.toml");
730    let mut doc: DocumentMut = if cargo_toml.is_file() {
731        let raw = fs::read_to_string(&cargo_toml)
732            .with_context(|| format!("read {}", cargo_toml.display()))?;
733        raw.parse()
734            .with_context(|| format!("parse {}", cargo_toml.display()))?
735    } else {
736        "[workspace]\nresolver = \"2\"\n"
737            .parse()
738            .expect("stub parses")
739    };
740
741    let workspace = doc
742        .entry("workspace")
743        .or_insert_with(|| Item::Table(Table::new()))
744        .as_table_mut()
745        .ok_or_else(|| eyre::eyre!("[workspace] is not a table in {}", cargo_toml.display()))?;
746    if workspace.get("resolver").is_none() {
747        workspace["resolver"] = value("2");
748    }
749
750    // Update members from pkg_dirs (relative to ws).
751    let mut members_arr = workspace
752        .get("members")
753        .and_then(|i| i.as_array())
754        .cloned()
755        .unwrap_or_else(Array::new);
756    let existing: Vec<String> = members_arr
757        .iter()
758        .filter_map(|v| v.as_str().map(str::to_string))
759        .collect();
760    for pkg in pkg_dirs {
761        let rel = pkg.strip_prefix(ws).unwrap_or(pkg).display().to_string();
762        if !existing.contains(&rel) {
763            members_arr.push(rel);
764        }
765    }
766    workspace["members"] = value(members_arr);
767
768    // Update exclude list.
769    let mut exclude_arr = workspace
770        .get("exclude")
771        .and_then(|i| i.as_array())
772        .cloned()
773        .unwrap_or_else(Array::new);
774    if !exclude_arr.iter().any(|v| v.as_str() == Some(bringup_name)) {
775        exclude_arr.push(bringup_name);
776    }
777    workspace["exclude"] = value(exclude_arr);
778
779    // [workspace.metadata.nros].
780    workspace
781        .entry("metadata")
782        .or_insert_with(|| Item::Table(Table::new()));
783    let mut nros_tbl = Table::new();
784    nros_tbl["default_system"] = value(bringup_name);
785    workspace["metadata"]["nros"] = Item::Table(nros_tbl);
786
787    fs::write(&cargo_toml, doc.to_string())
788        .with_context(|| format!("write {}", cargo_toml.display()))?;
789    Ok(())
790}

796fn is_already_migrated(ws: &Path) -> bool {
797    let cargo_toml = ws.join("Cargo.toml");
798    if !cargo_toml.is_file() {
799        return false;
800    }
801    let Ok(raw) = fs::read_to_string(&cargo_toml) else {
802        return false;
803    };
804    let Ok(doc) = raw.parse::<DocumentMut>() else {
805        return false;
806    };
807    doc.get("workspace")
808        .and_then(|w| w.get("metadata"))
809        .and_then(|m| m.get("nros"))
810        .is_some()
811}
