1//! Phase 212.F — `nros new system <name>_bringup --components <list>` scaffolder.
2//!
3//! Materializes a **Path A** bringup package per the multi-node workspace
4//! layout design (`docs/design/multi-node-workspace-layout.md` §4 +
5//! `docs/design/workspace-layout-by-case.md` Case 3). A bringup package is
6//! **pure declarative** — `package.xml` + `system.toml` + `launch/` only. No
7//! `Cargo.toml`, no `CMakeLists.txt`, no `src/`.
8//!
9//! When invoked inside an existing cargo workspace, the bringup pkg name is
10//! appended to the workspace-root `[workspace] exclude` list (Path A keeps
11//! bringup out of `members`).
12//!
13//! Surface (see `cmd/new.rs` dispatcher):
14//!
15//! ```text
16//! nros new system <name>_bringup --components <pkg1,pkg2,...> [--workspace-root <dir>] [--force]
17//! ```
18//!
19//! Each `<pkgN>` becomes one `[[component]]` entry in `system.toml`, one
20//! `<exec_depend>` line in `package.xml`, and one `<node>` block in
21//! `launch/system.launch.xml`. Component crates themselves are **NOT**
22//! scaffolded — that's `nros new --component <name>` (Phase 172 W.3).
23

37#[derive(Debug, ClapArgs)]
38pub struct Args {
39    /// Bringup package directory to create (e.g. `demo_bringup`). Conventionally
40    /// suffixed `_bringup` per the Phase 212 design doc, but any name is
41    /// accepted (`<system>_launch` documented as an alias).
42    pub name: PathBuf,
43
44    /// Comma-separated component package names to wire into the bringup
45    /// (`pkg1,pkg2,…`). One `[[component]]` entry + one `<exec_depend>` +
46    /// one launch `<node>` per name.
47    #[arg(long, value_delimiter = ',', required = true)]
48    pub components: Vec<String>,
49
50    /// Workspace root holding the cargo workspace `Cargo.toml` we should
51    /// update. Defaults to the current dir's nearest workspace root (parent
52    /// of the bringup dir for an absolute path, else cwd). When no workspace
53    /// `Cargo.toml` exists we still scaffold the bringup pkg — only the
54    /// `[workspace] exclude` update is skipped.
55    #[arg(long)]
56    pub workspace_root: Option<PathBuf>,
57
58    /// Overwrite an existing bringup directory.
59    #[arg(long)]
60    pub force: bool,
61}
62
63pub fn run(args: Args) -> Result<()> {
64    let cwd = std::env::current_dir()?;
65    let bringup_dir = if args.name.is_absolute() {
66        args.name.clone()
67    } else {
68        cwd.join(&args.name)
69    };
70
71    let pkg_name = bringup_dir
72        .file_name()
73        .and_then(|n| n.to_str())
74        .ok_or_else(|| eyre::eyre!("invalid bringup package name"))?
75        .to_string();
76
77    // Workspace-root resolution: explicit flag → parent of bringup dir → cwd.
78    let workspace_root = args
79        .workspace_root
80        .clone()
81        .or_else(|| bringup_dir.parent().map(Path::to_path_buf))
82        .unwrap_or_else(|| cwd.clone());
83
84    let _ = scaffold_bringup(&BringupScaffold {
85        bringup_dir,
86        pkg_name,
87        components: args.components,
88        workspace_root,
89        force: args.force,
90    })?;
91    Ok(())
92}

94/// Resolved inputs for [`scaffold_bringup`] — public so tests + the
95/// dispatcher (`cmd/new.rs`) can build it without going through clap.
96pub struct BringupScaffold {
97    pub bringup_dir: PathBuf,
98    pub pkg_name: String,
99    pub components: Vec<String>,
100    pub workspace_root: PathBuf,
101    pub force: bool,
102}
103
104/// Result of a successful scaffold — paths written, for tests + UX.
105#[derive(Debug)]
106pub struct ScaffoldedBringup {
107    pub bringup_dir: PathBuf,
108    pub package_xml: PathBuf,
109    pub system_toml: PathBuf,
110    pub launch_file: PathBuf,
111    pub gitignore: PathBuf,
112    /// `Some(path)` when a workspace `Cargo.toml` existed and got updated.
113    pub workspace_cargo_toml: Option<PathBuf>,
114}
115
116pub fn scaffold_bringup(s: &BringupScaffold) -> Result<ScaffoldedBringup> {
117    if s.components.is_empty() {
118        bail!("at least one --components <pkg> is required");
119    }
120    for c in &s.components {
121        if c.trim().is_empty() {
122            bail!("empty component name in --components list");
123        }
124    }
125
126    if s.bringup_dir.exists() {
127        if !s.force {
128            bail!(
129                "bringup directory already exists at {} — pass --force to overwrite",
130                s.bringup_dir.display()
131            );
132        }
133        fs::remove_dir_all(&s.bringup_dir).wrap_err_with(|| {
134            format!("remove existing {} for --force", s.bringup_dir.display())
135        })?;
136    }
137
138    fs::create_dir_all(s.bringup_dir.join("launch"))
139        .wrap_err_with(|| format!("create {}", s.bringup_dir.display()))?;
140
141    let package_xml = s.bringup_dir.join("package.xml");
142    fs::write(&package_xml, render_package_xml(&s.pkg_name, &s.components))
143        .wrap_err_with(|| format!("write {}", package_xml.display()))?;
144
145    let system_toml = s.bringup_dir.join("system.toml");
146    fs::write(&system_toml, render_system_toml(&s.pkg_name, &s.components)?)
147        .wrap_err_with(|| format!("write {}", system_toml.display()))?;
148
149    let launch_file = s.bringup_dir.join("launch").join("system.launch.xml");
150    fs::write(
151        &launch_file,
152        render_launch_xml(&s.pkg_name, &s.components),
153    )
154    .wrap_err_with(|| format!("write {}", launch_file.display()))?;
155
156    let gitignore = s.bringup_dir.join(".gitignore");
157    fs::write(&gitignore, "/target/\n/build/\n")
158        .wrap_err_with(|| format!("write {}", gitignore.display()))?;
159
160    let workspace_cargo_toml =
161        add_bringup_to_workspace_exclude(&s.workspace_root, &s.pkg_name)?;
162
163    Ok(ScaffoldedBringup {
164        bringup_dir: s.bringup_dir.clone(),
165        package_xml,
166        system_toml,
167        launch_file,
168        gitignore,
169        workspace_cargo_toml,
170    })
171}

173fn render_package_xml(pkg_name: &str, components: &[String]) -> String {
174    let mut s = String::new();
175    s.push_str("<?xml version=\"1.0\"?>\n");
176    s.push_str(
177        "<package format=\"3\">\n\
178         <!-- generated by nros new system — Phase 212.F -->\n",
179    );
180    s.push_str(&format!("  <name>{pkg_name}</name>\n"));
181    s.push_str("  <version>0.1.0</version>\n");
182    s.push_str(&format!(
183        "  <description>Bringup package for {pkg_name} (declarative system spec).</description>\n"
184    ));
185    s.push_str("  <maintainer email=\"nobody@example.invalid\">TODO maintainer</maintainer>\n");
186    s.push_str("  <license>Apache-2.0</license>\n");
187    for c in components {
188        s.push_str(&format!("  <exec_depend>{c}</exec_depend>\n"));
189    }
190    s.push_str("  <export>\n");
191    s.push_str("    <build_type>ament_nros</build_type>\n");
192    s.push_str("  </export>\n");
193    s.push_str("</package>\n");
194    s
195}

197fn render_system_toml(pkg_name: &str, components: &[String]) -> Result<String> {
198    // The pkg_name conventionally ends in `_bringup`; the system name strips
199    // that suffix (`demo_bringup` → `demo`). When the convention is broken we
200    // fall back to the full name.
201    let system_name = pkg_name
202        .strip_suffix("_bringup")
203        .or_else(|| pkg_name.strip_suffix("_launch"))
204        .unwrap_or(pkg_name)
205        .to_string();
206
207    let entries: Vec<SystemComponentEntry> = components
208        .iter()
209        .map(|pkg| SystemComponentEntry {
210            pkg: pkg.clone(),
211            // Placeholder — user fills in real Rust path / C++ class. Documented
212            // as TODO in the file header below.
213            class: format!("{pkg}::TODO"),
214            // Default the node name to the package name; the user typically
215            // edits it to drop the `_pkg` suffix.
216            name: pkg.clone(),
217        })
218        .collect();
219
220    let model = SystemToml {
221        system: SystemHeader {
222            name: system_name,
223            rmw: "zenoh".to_string(),
224            domain_id: 0,
225            locator: None,
226            default_launch: None,
227        },
228        components: entries,
229        deploy: Default::default(),
230        domains: Vec::new(),
231        bridges: Vec::new(),
232    };
233
234    let body = toml::to_string_pretty(&model)
235        .wrap_err("serialize generated system.toml")?;
236
237    let mut out = String::new();
238    out.push_str("# generated by nros new system — Phase 212.F\n");
239    out.push_str("# TODO: fill in real component `class` paths (Rust module-path / C++ class).\n");
240    out.push_str("# See docs/design/workspace-layout-by-case.md §3 for the full schema.\n");
241    out.push_str("\n");
242    out.push_str(&body);
243    Ok(out)
244}

246fn render_launch_xml(pkg_name: &str, components: &[String]) -> String {
247    let mut s = String::new();
248    s.push_str("<?xml version=\"1.0\"?>\n");
249    s.push_str(&format!(
250        "<!-- generated by nros new system — bringup pkg {pkg_name}, Phase 212.F -->\n"
251    ));
252    s.push_str("<launch>\n");
253    for c in components {
254        // node name = pkg name (user edits later); exec = pkg name. Matches
255        // the `cmake` Case 4 example in workspace-layout-by-case.md.
256        s.push_str(&format!(
257            "  <node pkg=\"{c}\" exec=\"{c}\" name=\"{c}\" />\n"
258        ));
259    }
260    s.push_str("</launch>\n");
261    s
262}

264/// Append `pkg_name` to the workspace-root `Cargo.toml`'s `[workspace] exclude`
265/// list (Path A). Returns `None` when no workspace `Cargo.toml` lives at the
266/// given root (still allowed — the bringup pkg works fine outside a cargo
267/// workspace, e.g. for pure-C++ users).
268fn add_bringup_to_workspace_exclude(
269    workspace_root: &Path,
270    pkg_name: &str,
271) -> Result<Option<PathBuf>> {
272    let cargo_toml = workspace_root.join("Cargo.toml");
273    if !cargo_toml.is_file() {
274        return Ok(None);
275    }
276
277    let raw = fs::read_to_string(&cargo_toml)
278        .wrap_err_with(|| format!("read {}", cargo_toml.display()))?;
279    let mut doc: DocumentMut = raw
280        .parse()
281        .wrap_err_with(|| format!("parse {}", cargo_toml.display()))?;
282
283    // Ensure `[workspace]` table.
284    let workspace = doc
285        .entry("workspace")
286        .or_insert_with(|| toml_edit::Item::Table(toml_edit::Table::new()))
287        .as_table_mut()
288        .ok_or_else(|| eyre::eyre!("[workspace] is not a table in {}", cargo_toml.display()))?;
289
290    // Read existing exclude list (default empty).
291    let mut exclude_array = workspace
292        .get("exclude")
293        .and_then(|i| i.as_array())
294        .cloned()
295        .unwrap_or_else(Array::new);
296
297    let already = exclude_array.iter().any(|v| v.as_str() == Some(pkg_name));
298    if !already {
299        exclude_array.push(pkg_name);
300    }
301    workspace["exclude"] = value(exclude_array);
302
303    fs::write(&cargo_toml, doc.to_string())
304        .wrap_err_with(|| format!("write {}", cargo_toml.display()))?;
305
306    Ok(Some(cargo_toml))
307}

318    fn temp_root(tag: &str) -> PathBuf {
319        let stamp = SystemTime::now()
320            .duration_since(UNIX_EPOCH)
321            .unwrap()
322            .as_nanos();
323        let dir = std::env::temp_dir().join(format!(
324            "nros-new-system-{tag}-{}-{stamp}",
325            std::process::id()
326        ));
327        let _ = fs::remove_dir_all(&dir);
328        fs::create_dir_all(&dir).unwrap();
329        dir
330    }

332    fn write_workspace_cargo_toml(root: &Path) {
333        fs::write(
334            root.join("Cargo.toml"),
335            "[workspace]\nresolver = \"2\"\nmembers = [\"talker_pkg\", \"listener_pkg\"]\n",
336        )
337        .unwrap();
338    }

340    #[test]
341    fn nros_new_system_scaffolds_bringup_pkg_with_expected_files() {
342        let root = temp_root("scaffold_files");
343        write_workspace_cargo_toml(&root);
344
345        let out = scaffold_bringup(&BringupScaffold {
346            bringup_dir: root.join("demo_bringup"),
347            pkg_name: "demo_bringup".to_string(),
348            components: vec!["talker_pkg".to_string(), "listener_pkg".to_string()],
349            workspace_root: root.clone(),
350            force: false,
351        })
352        .expect("scaffold succeeds");
353
354        // 1. Expected tree exists.
355        assert!(out.package_xml.is_file(), "package.xml: {:?}", out.package_xml);
356        assert!(out.system_toml.is_file(), "system.toml: {:?}", out.system_toml);
357        assert!(out.launch_file.is_file(), "launch xml: {:?}", out.launch_file);
358        assert!(out.gitignore.is_file(), ".gitignore: {:?}", out.gitignore);
359
360        // 2. NO forbidden files.
361        assert!(!out.bringup_dir.join("Cargo.toml").exists());
362        assert!(!out.bringup_dir.join("CMakeLists.txt").exists());
363        assert!(!out.bringup_dir.join("src").exists());
364
365        // 3. package.xml carries one <exec_depend> per component.
366        let pkg_xml = fs::read_to_string(&out.package_xml).unwrap();
367        assert!(pkg_xml.contains("<name>demo_bringup</name>"));
368        assert!(pkg_xml.contains("<exec_depend>talker_pkg</exec_depend>"));
369        assert!(pkg_xml.contains("<exec_depend>listener_pkg</exec_depend>"));
370
371        // 4. system.toml round-trips through SystemToml.
372        let sys: SystemToml =
373            toml::from_str(&fs::read_to_string(&out.system_toml).unwrap()).unwrap();
374        assert_eq!(sys.system.name, "demo");
375        assert_eq!(sys.system.rmw, "zenoh");
376        assert_eq!(sys.system.domain_id, 0);
377        assert_eq!(sys.components.len(), 2);
378        assert_eq!(sys.components[0].pkg, "talker_pkg");
379        assert_eq!(sys.components[1].pkg, "listener_pkg");
380
381        // 5. .gitignore is the documented two-liner.
382        let gi = fs::read_to_string(&out.gitignore).unwrap();
383        assert!(gi.contains("/target/"));
384        assert!(gi.contains("/build/"));
385    }

389        let root = temp_root("workspace_exclude");
390        write_workspace_cargo_toml(&root);
391
392        let out = scaffold_bringup(&BringupScaffold {

400
401        let cargo_toml_path = out
402            .workspace_cargo_toml

405        let after = fs::read_to_string(cargo_toml_path).unwrap();
406        let doc: toml::Value = toml::from_str(&after).unwrap();
407        let exclude = doc
408            .get("workspace")
409            .and_then(|w| w.get("exclude"))
410            .and_then(|e| e.as_array())
411            .expect("[workspace] exclude is an array");
412        let names: Vec<&str> = exclude.iter().filter_map(|v| v.as_str()).collect();
413        assert!(
414            names.contains(&"demo_bringup"),
415            "exclude must contain demo_bringup, got: {names:?}"
416        );
417
418        // Path A guard: bringup NOT in [workspace] members.
419        let members = doc
420            .get("workspace")
421            .and_then(|w| w.get("members"))
422            .and_then(|m| m.as_array())
423            .expect("members present");
424        let member_names: Vec<&str> = members.iter().filter_map(|v| v.as_str()).collect();
425        assert!(
426            !member_names.contains(&"demo_bringup"),
427            "Path A means bringup must NOT be in [workspace] members"
