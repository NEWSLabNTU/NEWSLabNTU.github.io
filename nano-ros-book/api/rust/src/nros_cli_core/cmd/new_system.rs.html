1//! Phase 212.F — `nros new system <name>_bringup --components <list>` scaffolder.
2//!
3//! Materializes a **Path A** bringup package per the multi-node workspace
4//! layout design (`docs/design/multi-node-workspace-layout.md` §4 + §11.3 +
5//! `docs/design/workspace-layout-by-case.md` Case 3). A bringup package is
6//! **pure declarative** — `package.xml` + `system.toml` + `launch/` (+ optional
7//! `config/` + `README.md`) only. No `Cargo.toml`, no `CMakeLists.txt`, no
8//! `src/`.
9//!
10//! When invoked inside an existing cargo workspace, the bringup pkg name is
11//! appended to the workspace-root `[workspace] exclude` list (Path A keeps
12//! bringup out of `members`).
13//!
14//! Surface (see `cmd/new.rs` dispatcher):
15//!
16//! ```text
17//! nros new system <name>_bringup --components <pkg1,pkg2,...> [--into <dir>] \
18//!     [--workspace-root <dir>] [--no-config] [--no-readme] [--force]
19//! ```
20//!
21//! Each `<pkgN>` becomes one `[[component]]` entry in `system.toml`, one
22//! `<exec_depend>` line in `package.xml`, and one `<node>` block in
23//! `launch/system.launch.xml`. Component crates themselves are **NOT**
24//! scaffolded — that's `nros new --component <name>` (Phase 172 W.3).
25//!
26//! **J.5 policy.** Generated `package.xml` does NOT include
27//! `<buildtool_depend>ament_cmake</buildtool_depend>` — `nros launch` reads
28//! `launch/` directly from the source tree, no install step needed (see
29//! `docs/design/multi-node-workspace-layout.md` §11.1).

44#[derive(Debug, ClapArgs)]
45pub struct Args {
46    /// Bringup package directory to create (e.g. `demo_bringup`). Conventionally
47    /// suffixed `_bringup` per the Phase 212 design doc, but any name is
48    /// accepted (`<system>_launch` documented as an alias).
49    pub name: PathBuf,
50
51    /// Comma-separated component package names to wire into the bringup
52    /// (`pkg1,pkg2,…`). One `[[component]]` entry + one `<exec_depend>` +
53    /// one launch `<node>` per name. May be combined with `--component-name`
54    /// (repeatable). At least one component (across both forms) is required.
55    #[arg(long, value_delimiter = ',')]
56    pub components: Vec<String>,
57
58    /// Repeatable single-component flag. Equivalent to one entry in
59    /// `--components`; provided for cases where commas in the shell are
60    /// awkward.
61    #[arg(long = "component-name")]
62    pub component_name: Vec<String>,
63
64    /// Parent directory under which `<name>_bringup/` is created. Defaults to
65    /// the current working directory. Mutually compatible with
66    /// `--workspace-root` (which controls only the `[workspace] exclude`
67    /// update target).
68    #[arg(long)]
69    pub into: Option<PathBuf>,
70
71    /// Workspace root holding the cargo workspace `Cargo.toml` we should
72    /// update. Defaults to the parent dir of the bringup. When no workspace
73    /// `Cargo.toml` exists we still scaffold the bringup pkg — only the
74    /// `[workspace] exclude` update is skipped.
75    #[arg(long)]
76    pub workspace_root: Option<PathBuf>,
77
78    /// Skip generating the optional `config/` sub-dir (with its `.gitkeep`).
79    #[arg(long)]
80    pub no_config: bool,
81
82    /// Skip generating the optional `README.md`.
83    #[arg(long)]
84    pub no_readme: bool,
85
86    /// Overwrite an existing bringup directory.
87    #[arg(long)]
88    pub force: bool,
89}

91pub fn run(args: Args) -> Result<()> {
92    let cwd = std::env::current_dir()?;
93    // Resolve the parent dir: --into takes precedence over the implicit cwd.
94    let into = args.into.clone().unwrap_or_else(|| cwd.clone());
95    validate_bringup_name(&args.name)?;
96    let bringup_dir = if args.name.is_absolute() {
97        args.name.clone()
98    } else {
99        into.join(&args.name)
100    };
101
102    let pkg_name = bringup_dir
103        .file_name()
104        .and_then(|n| n.to_str())
105        .ok_or_else(|| eyre::eyre!("invalid bringup package name"))?
106        .to_string();
107
108    // Workspace-root resolution: explicit flag → parent of bringup dir → cwd.
109    let workspace_root = args
110        .workspace_root
111        .clone()
112        .or_else(|| bringup_dir.parent().map(Path::to_path_buf))
113        .unwrap_or_else(|| cwd.clone());
114
115    // Merge `--components` + `--component-name` lists (preserving order).
116    let mut components: Vec<String> = args.components.clone();
117    components.extend(args.component_name.clone());
118
119    let _ = scaffold_bringup(&BringupScaffold {
120        bringup_dir,
121        pkg_name,
122        components,
123        workspace_root,
124        emit_config: !args.no_config,
125        emit_readme: !args.no_readme,
126        force: args.force,
127    })?;
128    Ok(())
129}

131/// Validate the user-supplied bringup pkg name. Reject path traversal (`..`)
132/// and absolute / nested paths — the bringup name must be a single directory
133/// segment.
134pub fn validate_bringup_name(name: &Path) -> Result<()> {
135    let display = name.display();
136    if name.as_os_str().is_empty() {
137        bail!("bringup package name is empty");
138    }
139    // Single-component name only: a nested path is ambiguous (`--into` is the
140    // way to place the bringup somewhere else).
141    let components: Vec<_> = name.components().collect();
142    if components.len() != 1 {
143        bail!(
144            "bringup package name {display:?} must be a single path component — \
145             use `--into <dir>` to place the bringup elsewhere"
146        );
147    }
148    let segment = match components[0] {
149        std::path::Component::Normal(s) => s.to_string_lossy().into_owned(),
150        std::path::Component::CurDir
151        | std::path::Component::ParentDir
152        | std::path::Component::RootDir
153        | std::path::Component::Prefix(_) => {
154            bail!(
155                "bringup package name {display:?} is not a regular dir name — \
156                 reserved components (`.`, `..`, `/`) are rejected"
157            );
158        }
159    };
160    // Disallow whitespace and obvious path metacharacters even inside the
161    // single segment (defence in depth — the OS would reject most of these
162    // anyway, but the diagnostic is cleaner here).
163    for bad in ['/', '\\', '\0'] {
164        if segment.contains(bad) {
165            bail!("bringup package name {segment:?} contains forbidden character {bad:?}");
166        }
167    }
168    if segment == "." || segment == ".." || segment.contains("..") {
169        bail!(
170            "bringup package name {segment:?} contains `..` or is a reserved \
171             dir name"
172        );
173    }
174    Ok(())
175}

189/// Result of a successful scaffold — paths written, for tests + UX.
190#[derive(Debug)]
191pub struct ScaffoldedBringup {
192    pub bringup_dir: PathBuf,
193    pub package_xml: PathBuf,
194    pub system_toml: PathBuf,
195    pub launch_file: PathBuf,
196    pub gitignore: PathBuf,
197    /// `Some(path)` when `--no-config` was NOT set; the `config/.gitkeep` path.
198    pub config_gitkeep: Option<PathBuf>,
199    /// `Some(path)` when `--no-readme` was NOT set; the `README.md` path.
200    pub readme: Option<PathBuf>,
201    /// `Some(path)` when a workspace `Cargo.toml` existed and got updated.
202    pub workspace_cargo_toml: Option<PathBuf>,
203}

205pub fn scaffold_bringup(s: &BringupScaffold) -> Result<ScaffoldedBringup> {
206    if s.components.is_empty() {
207        bail!("at least one --components <pkg> (or --component-name <pkg>) is required");
208    }
209    for c in &s.components {
210        if c.trim().is_empty() {
211            bail!("empty component name in --components / --component-name list");
212        }
213    }
214
215    if s.bringup_dir.exists() {
216        if !s.force {
217            bail!(
218                "bringup directory already exists at {} — pass --force to overwrite",
219                s.bringup_dir.display()
220            );
221        }
222        fs::remove_dir_all(&s.bringup_dir)
223            .wrap_err_with(|| format!("remove existing {} for --force", s.bringup_dir.display()))?;
224    }
225
226    fs::create_dir_all(s.bringup_dir.join("launch"))
227        .wrap_err_with(|| format!("create {}", s.bringup_dir.display()))?;
228
229    let package_xml = s.bringup_dir.join("package.xml");
230    fs::write(&package_xml, render_package_xml(&s.pkg_name, &s.components))
231        .wrap_err_with(|| format!("write {}", package_xml.display()))?;
232
233    let system_toml = s.bringup_dir.join("system.toml");
234    fs::write(
235        &system_toml,
236        render_system_toml(&s.pkg_name, &s.components)?,
237    )
238    .wrap_err_with(|| format!("write {}", system_toml.display()))?;
239
240    let launch_file = s.bringup_dir.join("launch").join("system.launch.xml");
241    fs::write(&launch_file, render_launch_xml(&s.pkg_name, &s.components))
242        .wrap_err_with(|| format!("write {}", launch_file.display()))?;
243
244    let gitignore = s.bringup_dir.join(".gitignore");
245    fs::write(&gitignore, "/target/\n/build/\n")
246        .wrap_err_with(|| format!("write {}", gitignore.display()))?;
247
248    let config_gitkeep = if s.emit_config {
249        let dir = s.bringup_dir.join("config");
250        fs::create_dir_all(&dir).wrap_err_with(|| format!("create {}", dir.display()))?;
251        let keep = dir.join(".gitkeep");
252        fs::write(&keep, "").wrap_err_with(|| format!("write {}", keep.display()))?;
253        Some(keep)
254    } else {
255        None
256    };
257
258    let readme = if s.emit_readme {
259        let path = s.bringup_dir.join("README.md");
260        fs::write(&path, render_readme(&s.pkg_name, &s.components))
261            .wrap_err_with(|| format!("write {}", path.display()))?;
262        Some(path)
263    } else {
264        None
265    };
266
267    let workspace_cargo_toml = add_bringup_to_workspace_exclude(&s.workspace_root, &s.pkg_name)?;
268
269    Ok(ScaffoldedBringup {
270        bringup_dir: s.bringup_dir.clone(),
271        package_xml,
272        system_toml,
273        launch_file,
274        gitignore,
275        config_gitkeep,
276        readme,
277        workspace_cargo_toml,
278    })
279}

281fn render_package_xml(pkg_name: &str, components: &[String]) -> String {
282    let mut s = String::new();
283    s.push_str("<?xml version=\"1.0\"?>\n");
284    s.push_str(
285        "<package format=\"3\">\n\
286         <!-- generated by nros new system — Phase 212.F -->\n",
287    );
288    s.push_str(&format!("  <name>{pkg_name}</name>\n"));
289    s.push_str("  <version>0.1.0</version>\n");
290    s.push_str(&format!(
291        "  <description>Bringup package for {pkg_name} (declarative system spec).</description>\n"
292    ));
293    s.push_str("  <maintainer email=\"nobody@example.invalid\">TODO maintainer</maintainer>\n");
294    s.push_str("  <license>Apache-2.0</license>\n");
295    // J.5 policy: NO <buildtool_depend>ament_cmake</buildtool_depend>.
296    // `nros launch` reads launch/ directly from source; users wanting ROS 2
297    // `ros2 launch <bringup>` add the buildtool_depend manually.
298    for c in components {
299        s.push_str(&format!("  <exec_depend>{c}</exec_depend>\n"));
300    }
301    s.push_str("  <export>\n");
302    s.push_str("    <build_type>ament_nros</build_type>\n");
303    s.push_str("  </export>\n");
304    s.push_str("</package>\n");
305    s
306}

308fn render_system_toml(pkg_name: &str, components: &[String]) -> Result<String> {
309    // The pkg_name conventionally ends in `_bringup`; the system name strips
310    // that suffix (`demo_bringup` → `demo`). When the convention is broken we
311    // fall back to the full name.
312    let system_name = pkg_name
313        .strip_suffix("_bringup")
314        .or_else(|| pkg_name.strip_suffix("_launch"))
315        .unwrap_or(pkg_name)
316        .to_string();
317
318    let entries: Vec<SystemComponentEntry> = components
319        .iter()
320        .map(|pkg| SystemComponentEntry {
321            pkg: pkg.clone(),
322            // Placeholder — user fills in real Rust path / C++ class. Documented
323            // as TODO in the file header below.
324            class: format!("{pkg}::TODO"),
325            // Default the node name to the package name; the user typically
326            // edits it to drop the `_pkg` suffix.
327            name: pkg.clone(),
328        })
329        .collect();
330
331    let mut deploy: std::collections::BTreeMap<String, DeployTarget> =
332        std::collections::BTreeMap::new();
333    deploy.insert(
334        "native".to_string(),
335        DeployTarget {
336            kind: Some("self".to_string()),
337            target: None,
338            launch: None,
339            board: None,
340            framework: None,
341        },
342    );
343
344    let model = SystemToml {
345        system: SystemHeader {
346            name: system_name,
347            rmw: "zenoh".to_string(),
348            domain_id: 0,
349            locator: None,
350            default_launch: Some("system.launch.xml".to_string()),
351            default_target: Some("native".to_string()),
352        },
353        components: entries,
354        deploy,
355        domains: Vec::new(),
356        bridges: Vec::new(),
357    };
358
359    let body = toml::to_string_pretty(&model).wrap_err("serialize generated system.toml")?;
360
361    let mut out = String::new();
362    out.push_str("# generated by nros new system — Phase 212.F\n");
363    out.push_str("# TODO: fill in real component `class` paths (Rust module-path / C++ class).\n");
364    out.push_str("# See docs/design/workspace-layout-by-case.md §3 for the full schema.\n");
365    out.push_str("\n");
366    out.push_str(&body);
367    Ok(out)
368}

370fn render_launch_xml(pkg_name: &str, components: &[String]) -> String {
371    let mut s = String::new();
372    s.push_str("<?xml version=\"1.0\"?>\n");
373    s.push_str(&format!(
374        "<!-- generated by nros new system — bringup pkg {pkg_name}, Phase 212.F -->\n"
375    ));
376    s.push_str("<launch>\n");
377    for c in components {
378        // node name = pkg name (user edits later); exec = pkg name. Matches
379        // the `cmake` Case 4 example in workspace-layout-by-case.md.
380        s.push_str(&format!(
381            "  <node pkg=\"{c}\" exec=\"{c}\" name=\"{c}\" />\n"
382        ));
383    }
384    s.push_str("</launch>\n");
385    s
386}

388fn render_readme(pkg_name: &str, components: &[String]) -> String {
389    let component_list = components
390        .iter()
391        .map(|c| format!("- `{c}`"))
392        .collect::<Vec<_>>()
393        .join("\n");
394    format!(
395        "# {pkg_name}\n\
396         \n\
397         This is a **bringup package** — a pure-declarative system spec for the\n\
398         nano-ros multi-node workspace. It carries:\n\
399         \n\
400         - `package.xml` — ROS 2 manifest. Hand-edit `<exec_depend>` to match\n\
401           `system.toml` `[[component]] pkg`.\n\
402         - `system.toml` — components, RMW, deploy targets, bridges. See\n\
403           `docs/design/multi-node-workspace-layout.md` §4 and §11.3 for the\n\
404           full schema.\n\
405         - `launch/*.launch.xml` — ROS 2 launch XML, picked by `nros launch`\n\
406           (or `ros2 launch` after a colcon install). The default is\n\
407           `system.launch.xml`; add more launch files for alternate\n\
408           topologies (nav2 convention; see design doc §11.3).\n\
409         - `config/` — optional `params.yaml`, rviz, etc.\n\
410         \n\
411         No `Cargo.toml`, no `CMakeLists.txt`, no `src/`. Code goes in sibling\n\
412         **Node packages** (one per `[[component]]`).\n\
413         \n\
414         ## Components\n\
415         \n\
416         {component_list}\n\
417         \n\
418         ## Editing\n\
419         \n\
420         1. Open `system.toml` and replace each `[[component]] class = \"<pkg>::TODO\"`\n\
421            with the real Rust module-path / C++ class name.\n\
422         2. Open `launch/system.launch.xml` and adjust `<node exec=\"…\">` to\n\
423            match each Entry pkg binary name.\n\
424         3. Run `nros check` to validate the bringup is well-formed.\n\
425         4. Run `nros launch {pkg_name}` to start the system (or\n\
426            `nros launch {pkg_name} --launch <other>.launch.xml` to pick a\n\
427            different topology).\n",
428    )
429}

431/// Append `pkg_name` to the workspace-root `Cargo.toml`'s `[workspace] exclude`
432/// list (Path A). Returns `None` when no workspace `Cargo.toml` lives at the
433/// given root (still allowed — the bringup pkg works fine outside a cargo
434/// workspace, e.g. for pure-C++ users).
435fn add_bringup_to_workspace_exclude(
436    workspace_root: &Path,
437    pkg_name: &str,
438) -> Result<Option<PathBuf>> {
439    let cargo_toml = workspace_root.join("Cargo.toml");
440    if !cargo_toml.is_file() {
441        return Ok(None);
442    }
443
444    let raw = fs::read_to_string(&cargo_toml)
445        .wrap_err_with(|| format!("read {}", cargo_toml.display()))?;
446    let mut doc: DocumentMut = raw
447        .parse()
448        .wrap_err_with(|| format!("parse {}", cargo_toml.display()))?;
449
450    // Ensure `[workspace]` table.
451    let workspace = doc
452        .entry("workspace")
453        .or_insert_with(|| toml_edit::Item::Table(toml_edit::Table::new()))
454        .as_table_mut()
455        .ok_or_else(|| eyre::eyre!("[workspace] is not a table in {}", cargo_toml.display()))?;
456
457    // Read existing exclude list (default empty).
458    let mut exclude_array = workspace
459        .get("exclude")
460        .and_then(|i| i.as_array())
461        .cloned()
462        .unwrap_or_else(Array::new);
463
464    let already = exclude_array.iter().any(|v| v.as_str() == Some(pkg_name));
465    if !already {
466        exclude_array.push(pkg_name);
467    }
468    workspace["exclude"] = value(exclude_array);
469
470    fs::write(&cargo_toml, doc.to_string())
471        .wrap_err_with(|| format!("write {}", cargo_toml.display()))?;
472
473    Ok(Some(cargo_toml))
474}
