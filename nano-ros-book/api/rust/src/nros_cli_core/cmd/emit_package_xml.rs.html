1//! Internal `package.xml` helpers — Phase 212.G (CLI verb removed).
2//!
3//! Renders a `package.xml` from a component package's `Cargo.toml`
4//! `[package.metadata.ament]` table, or from a bringup package's
5//! `system.toml.[[component]]` block. The `nros emit package-xml`
6//! user-facing verb was retired (users hand-write `package.xml` when
7//! they need ament/colcon interop, or skip it entirely when they use
8//! `nros launch`). The render path stays as an internal helper:
9//!
10//!   * `migrate::run` regenerates `package.xml` for every component
11//!     during the pre-212 → post-212 sweep.
12//!   * `check::run` uses [`check_drift`] to detect a hand-edited
13//!     `package.xml` whose generator marker disagrees with the current
14//!     `Cargo.toml` / `system.toml` (Phase 212.G.2).
15//!
16//! Nothing in this module is reachable from the CLI dispatcher.
17//!
18//! Two pkg shapes, dispatched by the artifacts present in `<pkg-dir>`:
19//!
20//! * **Component pkg** — `Cargo.toml` present, with
21//!   `[package.metadata.nros.component]` (or `…components.<Name>`) and
22//!   `[package.metadata.ament]`. Output carries `<build_depend>`,
23//!   `<exec_depend>`, `<test_depend>`, and `<export><build_type>…</build_type></export>`.
24//! * **Bringup pkg** — `system.toml` present. Output carries one
25//!   `<exec_depend>` per `[system.components[*].pkg`, plus `<name>`/`<description>`
26//!   derived from `[system].name`.
27//!
28//! Output discipline (every contract in §212.G):
29//!
30//! 1. A generated marker comment at the top so `nros check` can detect
31//!    hand-edit drift later (§212.G.2 — implemented as a small `nros check`
32//!    helper below; full lint surface is its own follow-up).
33//! 2. Byte-identical idempotency on re-run with unchanged inputs.
34//! 3. Alphabetised dependency entries so list ordering doesn't drift between
35//!    `Cargo.toml` edits and emitted output.
36

48/// Sentinel header injected at the top of every emitted `package.xml`.
49///
50/// `nros check` uses substring presence of this comment to decide a
51/// `package.xml` is generator-owned (and therefore eligible for drift
52/// diagnostics). Keep it stable — touching the wording invalidates every
53/// previously-emitted file.
54pub const GENERATED_HEADER_COMMENT: &str =
55    "<!-- generated by nros emit package-xml; edits will be overwritten -->";

57const DEFAULT_VERSION: &str = "0.0.0";
58const DEFAULT_LICENSE: &str = "Apache-2.0";
59const DEFAULT_MAINTAINER_EMAIL: &str = "dev@example.com";
60const DEFAULT_MAINTAINER_NAME: &str = "Developer";
61const DEFAULT_BUILD_TYPE_COMPONENT: &str = "ament_cargo";
62const DEFAULT_BUILD_TYPE_BRINGUP: &str = "ament_cmake";
63const DEFAULT_DESCRIPTION_BRINGUP_FMT: &str = "Generated bringup package for ";

69/// Decide which shape `<pkg-dir>` is and render the corresponding XML.
70///
71/// Order matters: a bringup pkg might end up in a cargo workspace's `src/`
72/// tree with no `Cargo.toml` of its own; `system.toml` is the disambiguator.
73pub fn render_for_pkg(pkg_dir: &Path) -> Result<String> {
74    if !pkg_dir.exists() {
75        bail!("pkg dir does not exist: {}", pkg_dir.display());
76    }
77    let cargo_toml = pkg_dir.join("Cargo.toml");
78    let system_toml = pkg_dir.join("system.toml");
79
80    if system_toml.exists() {
81        return render_bringup(&system_toml);
82    }
83    if cargo_toml.exists() {
84        return render_component(&cargo_toml);
85    }
86    bail!(
87        "{}: neither Cargo.toml nor system.toml found — not a component or bringup pkg",
88        pkg_dir.display()
89    );
90}

125fn render_component(cargo_toml: &Path) -> Result<String> {
126    let raw = std::fs::read_to_string(cargo_toml)
127        .with_context(|| format!("read {}", cargo_toml.display()))?;
128    let manifest: CargoManifest =
129        toml::from_str(&raw).with_context(|| format!("parse {}", cargo_toml.display()))?;
130
131    // Validate the nros table early — surfaces the
132    // "both component and components" diagnostic from the schema layer.
133    if let Some(meta) = manifest.package.metadata.as_ref()
134        && let Some(nros) = meta.nros.as_ref()
135    {
136        nros.validate()
137            .map_err(|m| eyre::eyre!("{}: {m}", cargo_toml.display()))?;
138    }
139
140    let ament = manifest
141        .package
142        .metadata
143        .as_ref()
144        .and_then(|m| m.ament.as_ref())
145        .cloned()
146        .unwrap_or_default();
147
148    let pkg_name = manifest.package.name;
149    let version = manifest
150        .package
151        .version
152        .unwrap_or_else(|| DEFAULT_VERSION.into());
153    // Phase 212.B.4 — `[package.metadata.ament]` is the source of
154    // truth for description / license / maintainer. Cargo's
155    // `[package].description` / `[package].license` are kept as
156    // fallbacks (most users author them anyway); the synthesised
157    // default lands only when neither is present.
158    let description = ament
159        .description
160        .clone()
161        .or(manifest.package.description)
162        .unwrap_or_else(|| format!("nano-ros component package {pkg_name}"));
163    let license = ament
164        .license
165        .clone()
166        .or(manifest.package.license)
167        .unwrap_or_else(|| DEFAULT_LICENSE.into());
168    let build_type = ament
169        .build_type
170        .clone()
171        .unwrap_or_else(|| DEFAULT_BUILD_TYPE_COMPONENT.into());
172    let maintainer = ament.maintainer.clone();
173
174    Ok(render_xml(RenderInputs {
175        pkg_name: &pkg_name,
176        version: &version,
177        description: &description,
178        license: &license,
179        maintainer: maintainer.as_ref(),
180        build_depend: sorted_unique(&ament.build_depend),
181        buildtool_depend: sorted_unique(&ament.buildtool_depend),
182        exec_depend: sorted_unique(&ament.exec_depend),
183        test_depend: sorted_unique(&ament.test_depend),
184        build_type: &build_type,
185    }))
186}

196// ---------------------------------------------------------------------------
197// Bringup pkg path
198// ---------------------------------------------------------------------------
199
200// Phase 212.J.5 policy — the generated bringup `package.xml` deliberately
201// does NOT emit `<buildtool_depend>ament_cmake</buildtool_depend>`. The
202// canonical desktop launcher (`nros launch`) reads `launch/*.launch.xml`
203// straight from the bringup pkg's source tree, not from an ament install
204// `share/<pkg>/launch/` path, so a colcon outer build is never required
205// to exercise a workspace. Users who want stock `ros2 launch <bringup>`
206// compatibility add `<buildtool_depend>ament_cmake</buildtool_depend>`
207// themselves (the emit is generator-owned; the drift checker only
208// rejects edits to the marker-bearing region). See
209// `docs/design/multi-node-workspace-layout.md` §11.1 in the nano-ros
210// tree for the policy + rationale.
211fn render_bringup(system_toml: &Path) -> Result<String> {
212    let raw = std::fs::read_to_string(system_toml)
213        .with_context(|| format!("read {}", system_toml.display()))?;
214    let system: SystemToml =
215        toml::from_str(&raw).with_context(|| format!("parse {}", system_toml.display()))?;
216
217    // Bringup pkg name convention is `<system>_bringup`. Per the design doc
218    // we name the emitted `<package><name>` by suffixing if missing — but the
219    // user-authored `[system].name` is the source of truth either way.
220    let system_name = system.system.name.clone();
221    let pkg_name = if system_name.ends_with("_bringup") {
222        system_name.clone()
223    } else {
224        format!("{system_name}_bringup")
225    };
226    let description = format!("{DEFAULT_DESCRIPTION_BRINGUP_FMT}{system_name}");
227    let exec_depend = sorted_unique(&component_pkg_names(&system.components));
228
229    Ok(render_xml(RenderInputs {
230        pkg_name: &pkg_name,
231        version: DEFAULT_VERSION,
232        description: &description,
233        license: DEFAULT_LICENSE,
234        maintainer: None,
235        build_depend: Vec::new(),
236        buildtool_depend: Vec::new(),
237        exec_depend,
238        test_depend: Vec::new(),
239        build_type: DEFAULT_BUILD_TYPE_BRINGUP,
240    }))
241}

243fn component_pkg_names(components: &[SystemComponentEntry]) -> Vec<String> {
244    components.iter().map(|c| c.pkg.clone()).collect()
245}

251struct RenderInputs<'a> {
252    pkg_name: &'a str,
253    version: &'a str,
254    description: &'a str,
255    license: &'a str,
256    /// Phase 212.B.4 — when present, replaces the default
257    /// `Developer <dev@example.com>` placeholder maintainer.
258    maintainer: Option<&'a AmentMaintainer>,
259    build_depend: Vec<String>,
260    /// Phase 212.B.4 — explicit `<buildtool_depend>` rows.
261    buildtool_depend: Vec<String>,
262    exec_depend: Vec<String>,
263    test_depend: Vec<String>,
264    build_type: &'a str,
265}
266
267fn render_xml(inp: RenderInputs<'_>) -> String {
268    let mut out = String::new();
269    out.push_str("<?xml version=\"1.0\"?>\n");
270    out.push_str(GENERATED_HEADER_COMMENT);
271    out.push('\n');
272    out.push_str("<package format=\"3\">\n");
273    push_text_elem(&mut out, "name", inp.pkg_name, 2);
274    push_text_elem(&mut out, "version", inp.version, 2);
275    push_text_elem(&mut out, "description", inp.description, 2);
276    push_maintainer(&mut out, 2, inp.maintainer);
277    push_text_elem(&mut out, "license", inp.license, 2);
278
279    if !inp.buildtool_depend.is_empty() {
280        out.push('\n');
281        for d in &inp.buildtool_depend {
282            push_text_elem(&mut out, "buildtool_depend", d, 2);
283        }
284    }
285    if !inp.build_depend.is_empty() {
286        out.push('\n');
287        for d in &inp.build_depend {
288            push_text_elem(&mut out, "build_depend", d, 2);
289        }
290    }
291    if !inp.exec_depend.is_empty() {
292        out.push('\n');
293        for d in &inp.exec_depend {
294            push_text_elem(&mut out, "exec_depend", d, 2);
295        }
296    }
297    if !inp.test_depend.is_empty() {
298        out.push('\n');
299        for d in &inp.test_depend {
300            push_text_elem(&mut out, "test_depend", d, 2);
301        }
302    }
303
304    out.push('\n');
305    out.push_str("  <export>\n");
306    push_text_elem(&mut out, "build_type", inp.build_type, 4);
307    out.push_str("  </export>\n");
308    out.push_str("</package>\n");
309    out
310}

312fn push_maintainer(out: &mut String, indent: usize, m: Option<&AmentMaintainer>) {
313    for _ in 0..indent {
314        out.push(' ');
315    }
316    let (name, email) = match m {
317        Some(m) => (m.name.as_str(), m.email.as_str()),
318        None => (DEFAULT_MAINTAINER_NAME, DEFAULT_MAINTAINER_EMAIL),
319    };
320    out.push_str("<maintainer email=\"");
321    push_attr_escaped(out, email);
322    out.push_str("\">");
323    push_text_escaped(out, name);
324    out.push_str("</maintainer>\n");
325}
326
327fn push_text_elem(out: &mut String, tag: &str, body: &str, indent: usize) {
328    for _ in 0..indent {
329        out.push(' ');
330    }
331    out.push('<');
332    out.push_str(tag);
333    out.push('>');
334    push_text_escaped(out, body);
335    out.push_str("</");
336    out.push_str(tag);
337    out.push_str(">\n");
338}
339
340fn push_text_escaped(out: &mut String, body: &str) {
341    for ch in body.chars() {
342        match ch {
343            '&' => out.push_str("&amp;"),
344            '<' => out.push_str("&lt;"),
345            '>' => out.push_str("&gt;"),
346            _ => out.push(ch),
347        }
348    }
349}
350
351fn push_attr_escaped(out: &mut String, body: &str) {
352    for ch in body.chars() {
353        match ch {
354            '&' => out.push_str("&amp;"),
355            '<' => out.push_str("&lt;"),
356            '"' => out.push_str("&quot;"),
357            _ => out.push(ch),
358        }
359    }
360}

362fn sorted_unique(src: &[String]) -> Vec<String> {
363    let set: BTreeSet<&String> = src.iter().collect();
364    set.into_iter().cloned().collect()
365}

367// ---------------------------------------------------------------------------
368// Drift detection helper for `nros check`
369// ---------------------------------------------------------------------------
370
371/// Phase 212.G.2 drift check. Re-renders `package.xml` for the given pkg dir
372/// and compares it against the existing file (if present + generator-owned).
373///
374/// Returns:
375/// * `Ok(None)` — file absent, or carries no generated marker (hand-written
376///   package.xml — out of scope for drift; user owns it).
377/// * `Ok(Some(())) ` — file matches a fresh emit. No drift.
378/// * `Err(_)` — file carries the marker but DIFFERS from a fresh emit:
379///   the diagnostic body describes the drift. Caller (`nros check`) prints
380///   it as a warning.
381pub fn check_drift(pkg_dir: &Path) -> Result<DriftStatus> {
382    let on_disk_path = pkg_dir.join("package.xml");
383    if !on_disk_path.exists() {
384        return Ok(DriftStatus::Absent);
385    }
386    let on_disk = std::fs::read_to_string(&on_disk_path)
387        .with_context(|| format!("read {}", on_disk_path.display()))?;
388    if !on_disk.contains(GENERATED_HEADER_COMMENT) {
389        return Ok(DriftStatus::HandWritten);
390    }
391    let fresh = render_for_pkg(pkg_dir)?;
392    if on_disk == fresh {
393        Ok(DriftStatus::Clean)
394    } else {
395        Ok(DriftStatus::Drift { on_disk_path })
396    }
397}
398
399#[derive(Debug)]
400pub enum DriftStatus {
401    /// No `package.xml` on disk.
402    Absent,
403    /// Present but without the generated marker — user-maintained.
404    HandWritten,
405    /// Present, generator-owned, and matches a fresh emit.
406    Clean,
407    /// Present, generator-owned, but differs from a fresh emit.
408    Drift { on_disk_path: PathBuf },
409}

679    // -----------------------------------------------------------------------
680    // Tiny zero-dep tempdir shim, scoped to this module's tests so we don't
681    // add a workspace-level dev-dep.
682    // -----------------------------------------------------------------------
683    mod tempfile_lite {
684        use std::path::{Path, PathBuf};
685        use std::sync::atomic::{AtomicU64, Ordering};
686
687        static SEQ: AtomicU64 = AtomicU64::new(0);
688
689        pub struct TempDir {
690            path: PathBuf,
691        }
692
693        impl TempDir {
694            pub fn new() -> std::io::Result<Self> {
695                let n = SEQ.fetch_add(1, Ordering::Relaxed);
696                let pid = std::process::id();
697                let now = std::time::SystemTime::now()
698                    .duration_since(std::time::UNIX_EPOCH)
699                    .map(|d| d.as_nanos())
700                    .unwrap_or(0);
701                let path = std::env::temp_dir().join(format!("nros-emit-pkgxml-{pid}-{now}-{n}"));
702                std::fs::create_dir_all(&path)?;
703                Ok(Self { path })
704            }
705
706            pub fn path(&self) -> &Path {
707                &self.path
708            }
709        }
710
711        impl Drop for TempDir {
712            fn drop(&mut self) {
713                let _ = std::fs::remove_dir_all(&self.path);
714            }
715        }
716    }
