1//! Internal `package.xml` helpers — Phase 212.G (CLI verb removed).
2//!
3//! Renders a `package.xml` from a component package's `Cargo.toml`
4//! `[package.metadata.ament]` table, or from a bringup package's
5//! `system.toml.[[component]]` block. The `nros emit package-xml`
6//! user-facing verb was retired (users hand-write `package.xml` when
7//! they need ament/colcon interop, or skip it entirely when they use
8//! `nros launch`). The render path stays as an internal helper:
9//!
10//!   * `migrate::run` regenerates `package.xml` for every component
11//!     during the pre-212 → post-212 sweep.
12//!   * `check::run` uses [`check_drift`] to detect a hand-edited
13//!     `package.xml` whose generator marker disagrees with the current
14//!     `Cargo.toml` / `system.toml` (Phase 212.G.2).
15//!
16//! Nothing in this module is reachable from the CLI dispatcher.
17//!
18//! Two pkg shapes, dispatched by the artifacts present in `<pkg-dir>`:
19//!
20//! * **Component pkg** — `Cargo.toml` present, with
21//!   `[package.metadata.nros.component]` (or `…components.<Name>`) and
22//!   `[package.metadata.ament]`. Output carries `<build_depend>`,
23//!   `<exec_depend>`, `<test_depend>`, and `<export><build_type>…</build_type></export>`.
24//! * **Bringup pkg** — `system.toml` present. Output carries one
25//!   `<exec_depend>` per `[system.components[*].pkg`, plus `<name>`/`<description>`
26//!   derived from `[system].name`.
27//!
28//! Output discipline (every contract in §212.G):
29//!
30//! 1. A generated marker comment at the top so `nros check` can detect
31//!    hand-edit drift later (§212.G.2 — implemented as a small `nros check`
32//!    helper below; full lint surface is its own follow-up).
33//! 2. Byte-identical idempotency on re-run with unchanged inputs.
34//! 3. Alphabetised dependency entries so list ordering doesn't drift between
35//!    `Cargo.toml` edits and emitted output.
36

47/// Sentinel header injected at the top of every emitted `package.xml`.
48///
49/// `nros check` uses substring presence of this comment to decide a
50/// `package.xml` is generator-owned (and therefore eligible for drift
51/// diagnostics). Keep it stable — touching the wording invalidates every
52/// previously-emitted file.
53pub const GENERATED_HEADER_COMMENT: &str =
54    "<!-- generated by nros emit package-xml; edits will be overwritten -->";
55
56const DEFAULT_VERSION: &str = "0.0.0";
57const DEFAULT_LICENSE: &str = "Apache-2.0";
58const DEFAULT_MAINTAINER_EMAIL: &str = "dev@example.com";
59const DEFAULT_MAINTAINER_NAME: &str = "Developer";
60const DEFAULT_BUILD_TYPE_COMPONENT: &str = "ament_cargo";
61const DEFAULT_BUILD_TYPE_BRINGUP: &str = "ament_cmake";
62const DEFAULT_DESCRIPTION_BRINGUP_FMT: &str = "Generated bringup package for ";

68/// Decide which shape `<pkg-dir>` is and render the corresponding XML.
69///
70/// Order matters: a bringup pkg might end up in a cargo workspace's `src/`
71/// tree with no `Cargo.toml` of its own; `system.toml` is the disambiguator.
72pub fn render_for_pkg(pkg_dir: &Path) -> Result<String> {
73    if !pkg_dir.exists() {
74        bail!("pkg dir does not exist: {}", pkg_dir.display());
75    }
76    let cargo_toml = pkg_dir.join("Cargo.toml");
77    let system_toml = pkg_dir.join("system.toml");
78
79    if system_toml.exists() {
80        return render_bringup(&system_toml);
81    }
82    if cargo_toml.exists() {
83        return render_component(&cargo_toml);
84    }
85    bail!(
86        "{}: neither Cargo.toml nor system.toml found — not a component or bringup pkg",
87        pkg_dir.display()
88    );
89}

95/// Minimal `Cargo.toml` projection — we only need `[package]` + the two
96/// `[package.metadata.*]` tables. Everything else is opaque so `cargo`'s own
97/// fields don't fight `deny_unknown_fields` discipline elsewhere in the tree.
98#[derive(Debug, Deserialize)]
99struct CargoManifest {
100    package: CargoPackage,
101}

103#[derive(Debug, Deserialize)]
104struct CargoPackage {
105    name: String,
106    #[serde(default)]
107    version: Option<String>,
108    #[serde(default)]
109    description: Option<String>,
110    #[serde(default)]
111    license: Option<String>,
112    #[serde(default)]
113    metadata: Option<CargoPackageMetadata>,
114}

116#[derive(Debug, Deserialize)]
117struct CargoPackageMetadata {
118    #[serde(default)]
119    nros: Option<PackageMetadataNros>,
120    #[serde(default)]
121    ament: Option<PackageMetadataAment>,
122}

124fn render_component(cargo_toml: &Path) -> Result<String> {
125    let raw = std::fs::read_to_string(cargo_toml)
126        .with_context(|| format!("read {}", cargo_toml.display()))?;
127    let manifest: CargoManifest =
128        toml::from_str(&raw).with_context(|| format!("parse {}", cargo_toml.display()))?;
129
130    // Validate the nros table early — surfaces the
131    // "both component and components" diagnostic from the schema layer.
132    if let Some(meta) = manifest.package.metadata.as_ref()
133        && let Some(nros) = meta.nros.as_ref()
134    {
135        nros.validate()
136            .map_err(|m| eyre::eyre!("{}: {m}", cargo_toml.display()))?;
137    }
138
139    let ament = manifest
140        .package
141        .metadata
142        .as_ref()
143        .and_then(|m| m.ament.as_ref())
144        .cloned()
145        .unwrap_or_default();
146
147    let pkg_name = manifest.package.name;
148    let version = manifest
149        .package
150        .version
151        .unwrap_or_else(|| DEFAULT_VERSION.into());
152    let description = manifest
153        .package
154        .description
155        .unwrap_or_else(|| format!("nano-ros component package {pkg_name}"));
156    let license = manifest
157        .package
158        .license
159        .unwrap_or_else(|| DEFAULT_LICENSE.into());
160    let build_type = ament
161        .build_type
162        .clone()
163        .unwrap_or_else(|| DEFAULT_BUILD_TYPE_COMPONENT.into());
164
165    Ok(render_xml(RenderInputs {
166        pkg_name: &pkg_name,
167        version: &version,
168        description: &description,
169        license: &license,
170        build_depend: sorted_unique(&ament.build_depend),
171        exec_depend: sorted_unique(&ament.exec_depend),
172        test_depend: sorted_unique(&ament.test_depend),
173        build_type: &build_type,
174    }))
175}

181fn render_bringup(system_toml: &Path) -> Result<String> {
182    let raw = std::fs::read_to_string(system_toml)
183        .with_context(|| format!("read {}", system_toml.display()))?;
184    let system: SystemToml =
185        toml::from_str(&raw).with_context(|| format!("parse {}", system_toml.display()))?;
186
187    // Bringup pkg name convention is `<system>_bringup`. Per the design doc
188    // we name the emitted `<package><name>` by suffixing if missing — but the
189    // user-authored `[system].name` is the source of truth either way.
190    let system_name = system.system.name.clone();
191    let pkg_name = if system_name.ends_with("_bringup") {
192        system_name.clone()
193    } else {
194        format!("{system_name}_bringup")
195    };
196    let description = format!("{DEFAULT_DESCRIPTION_BRINGUP_FMT}{system_name}");
197    let exec_depend = sorted_unique(&component_pkg_names(&system.components));
198
199    Ok(render_xml(RenderInputs {
200        pkg_name: &pkg_name,
201        version: DEFAULT_VERSION,
202        description: &description,
203        license: DEFAULT_LICENSE,
204        build_depend: Vec::new(),
205        exec_depend,
206        test_depend: Vec::new(),
207        build_type: DEFAULT_BUILD_TYPE_BRINGUP,
208    }))
209}

211fn component_pkg_names(components: &[SystemComponentEntry]) -> Vec<String> {
212    components.iter().map(|c| c.pkg.clone()).collect()
213}

219struct RenderInputs<'a> {
220    pkg_name: &'a str,
221    version: &'a str,
222    description: &'a str,
223    license: &'a str,
224    build_depend: Vec<String>,
225    exec_depend: Vec<String>,
226    test_depend: Vec<String>,
227    build_type: &'a str,
228}
229
230fn render_xml(inp: RenderInputs<'_>) -> String {
231    let mut out = String::new();
232    out.push_str("<?xml version=\"1.0\"?>\n");
233    out.push_str(GENERATED_HEADER_COMMENT);
234    out.push('\n');
235    out.push_str("<package format=\"3\">\n");
236    push_text_elem(&mut out, "name", inp.pkg_name, 2);
237    push_text_elem(&mut out, "version", inp.version, 2);
238    push_text_elem(&mut out, "description", inp.description, 2);
239    push_maintainer(&mut out, 2);
240    push_text_elem(&mut out, "license", inp.license, 2);
241
242    if !inp.build_depend.is_empty() {
243        out.push('\n');
244        for d in &inp.build_depend {
245            push_text_elem(&mut out, "build_depend", d, 2);
246        }
247    }
248    if !inp.exec_depend.is_empty() {
249        out.push('\n');
250        for d in &inp.exec_depend {
251            push_text_elem(&mut out, "exec_depend", d, 2);
252        }
253    }
254    if !inp.test_depend.is_empty() {
255        out.push('\n');
256        for d in &inp.test_depend {
257            push_text_elem(&mut out, "test_depend", d, 2);
258        }
259    }
260
261    out.push('\n');
262    out.push_str("  <export>\n");
263    push_text_elem(&mut out, "build_type", inp.build_type, 4);
264    out.push_str("  </export>\n");
265    out.push_str("</package>\n");
266    out
267}

269fn push_maintainer(out: &mut String, indent: usize) {
270    for _ in 0..indent {
271        out.push(' ');
272    }
273    out.push_str("<maintainer email=\"");
274    push_attr_escaped(out, DEFAULT_MAINTAINER_EMAIL);
275    out.push_str("\">");
276    push_text_escaped(out, DEFAULT_MAINTAINER_NAME);
277    out.push_str("</maintainer>\n");
278}
279
280fn push_text_elem(out: &mut String, tag: &str, body: &str, indent: usize) {
281    for _ in 0..indent {
282        out.push(' ');
283    }
284    out.push('<');
285    out.push_str(tag);
286    out.push('>');
287    push_text_escaped(out, body);
288    out.push_str("</");
289    out.push_str(tag);
290    out.push_str(">\n");
291}
292
293fn push_text_escaped(out: &mut String, body: &str) {
294    for ch in body.chars() {
295        match ch {
296            '&' => out.push_str("&amp;"),
297            '<' => out.push_str("&lt;"),
298            '>' => out.push_str("&gt;"),
299            _ => out.push(ch),
300        }
301    }
302}
303
304fn push_attr_escaped(out: &mut String, body: &str) {
305    for ch in body.chars() {
306        match ch {
307            '&' => out.push_str("&amp;"),
308            '<' => out.push_str("&lt;"),
309            '"' => out.push_str("&quot;"),
310            _ => out.push(ch),
311        }
312    }
313}

315fn sorted_unique(src: &[String]) -> Vec<String> {
316    let set: BTreeSet<&String> = src.iter().collect();
317    set.into_iter().cloned().collect()
318}

320// ---------------------------------------------------------------------------
321// Drift detection helper for `nros check`
322// ---------------------------------------------------------------------------
323
324/// Phase 212.G.2 drift check. Re-renders `package.xml` for the given pkg dir
325/// and compares it against the existing file (if present + generator-owned).
326///
327/// Returns:
328/// * `Ok(None)` — file absent, or carries no generated marker (hand-written
329///   package.xml — out of scope for drift; user owns it).
330/// * `Ok(Some(())) ` — file matches a fresh emit. No drift.
331/// * `Err(_)` — file carries the marker but DIFFERS from a fresh emit:
332///   the diagnostic body describes the drift. Caller (`nros check`) prints
333///   it as a warning.
334pub fn check_drift(pkg_dir: &Path) -> Result<DriftStatus> {
335    let on_disk_path = pkg_dir.join("package.xml");
336    if !on_disk_path.exists() {
337        return Ok(DriftStatus::Absent);
338    }
339    let on_disk = std::fs::read_to_string(&on_disk_path)
340        .with_context(|| format!("read {}", on_disk_path.display()))?;
341    if !on_disk.contains(GENERATED_HEADER_COMMENT) {
342        return Ok(DriftStatus::HandWritten);
343    }
344    let fresh = render_for_pkg(pkg_dir)?;
345    if on_disk == fresh {
346        Ok(DriftStatus::Clean)
347    } else {
348        Ok(DriftStatus::Drift { on_disk_path })
349    }
350}
351
352#[derive(Debug)]
353pub enum DriftStatus {
354    /// No `package.xml` on disk.
355    Absent,
356    /// Present but without the generated marker — user-maintained.
357    HandWritten,
358    /// Present, generator-owned, and matches a fresh emit.
359    Clean,
360    /// Present, generator-owned, but differs from a fresh emit.
361    Drift { on_disk_path: PathBuf },
362}

632    // -----------------------------------------------------------------------
633    // Tiny zero-dep tempdir shim, scoped to this module's tests so we don't
634    // add a workspace-level dev-dep.
635    // -----------------------------------------------------------------------
636    mod tempfile_lite {
637        use std::path::{Path, PathBuf};
638        use std::sync::atomic::{AtomicU64, Ordering};
639
640        static SEQ: AtomicU64 = AtomicU64::new(0);
641
642        pub struct TempDir {
643            path: PathBuf,
644        }
645
646        impl TempDir {
647            pub fn new() -> std::io::Result<Self> {
648                let n = SEQ.fetch_add(1, Ordering::Relaxed);
649                let pid = std::process::id();
650                let now = std::time::SystemTime::now()
651                    .duration_since(std::time::UNIX_EPOCH)
652                    .map(|d| d.as_nanos())
653                    .unwrap_or(0);
654                let path = std::env::temp_dir().join(format!("nros-emit-pkgxml-{pid}-{now}-{n}"));
655                std::fs::create_dir_all(&path)?;
656                Ok(Self { path })
657            }
658
659            pub fn path(&self) -> &Path {
660                &self.path
661            }
662        }
663
664        impl Drop for TempDir {
665            fn drop(&mut self) {
666                let _ = std::fs::remove_dir_all(&self.path);
667            }
668        }
669    }
