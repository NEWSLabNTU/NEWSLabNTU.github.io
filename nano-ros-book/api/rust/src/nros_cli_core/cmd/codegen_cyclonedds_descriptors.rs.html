1//! `nros codegen cyclonedds-descriptors` — Phase 212.K.4
2//!
3//! Per-example Cyclone-DDS topic-descriptor emit.
4//!
5//! Reads one or more `.msg` sources (named by `<pkg>/<Msg>`), synthesises
6//! Cyclone-shaped IDL via [`nros_msg_to_idl`], shells out to the host
7//! `idlc` to produce the matching `<pkg>_<Msg>.{c,h}` pair, then writes
8//! a tiny `register.{c,h}` translation unit that exposes a single
9//! `extern "C"` entry per crate. Also drops a JSON manifest enumerating
10//! every generated artifact so the consumer build script (the
11//! `nros-build::cyclonedds::Descriptors` helper) can feed them into a
12//! `cc::Build`.
13//!
14//! Usage:
15//!
16//!   nros codegen cyclonedds-descriptors \
17//!       --idlc <path> \
18//!       --include <path> \
19//!       --msg std_msgs/Int32=/abs/path/to/Int32.msg \
20//!       [--msg ...] \
21//!       --crate-name <ident> \
22//!       --out <dir>
23//!
24//! Or via JSON `--args-file <path>` mirroring the `nros codegen` surface
25//! (the build-script helper uses argv directly; `--args-file` exists for
26//! parity with the existing `nros codegen` shape).
27//!
28//! The K.2 baked-in defaults in `packages/dds/nros-rmw-cyclonedds-sys/build.rs`
29//! (std_msgs/Int32 + rmw_dds_common_graph) stay as universal fallbacks —
30//! this verb is purely additive (per-example descriptors layered on top).

43#[derive(Debug, ClapArgs)]
44pub struct Args {
45    /// Path to the host `idlc` binary (typically `build/cyclonedds/bin/idlc`).
46    #[arg(long)]
47    pub idlc: Option<PathBuf>,
48
49    /// Path to the Cyclone include dir (`<sysroot>/include`).
50    #[arg(long)]
51    pub include: Option<PathBuf>,
52
53    /// One `<pkg>/<Msg>=<msg-source-path>` per message. Repeatable.
54    #[arg(long = "msg", value_name = "PKG/MSG=PATH")]
55    pub msg: Vec<String>,
56
57    /// Crate / consumer name. Used as the prefix of the generated
58    /// `extern "C" void <name>_register_descriptors(void)` entry point.
59    /// Defaults to `nros_descriptors`.
60    #[arg(long = "crate-name")]
61    pub crate_name: Option<String>,
62
63    /// Output directory. Created if missing. Receives `*.{c,h}`,
64    /// `register.{c,h}`, and `cyclonedds-descriptors.json`.
65    #[arg(long)]
66    pub out: Option<PathBuf>,
67
68    /// Optional JSON file carrying every flag as a structured doc.
69    /// When set, takes precedence over the matching argv flag.
70    #[arg(long)]
71    pub args_file: Option<PathBuf>,
72}

75struct ArgsFile {

78    #[serde(default)]
79    crate_name: Option<String>,
80    out: PathBuf,

114pub fn run(args: Args) -> Result<()> {
115    let resolved = resolve_args(args)?;
116    emit(&resolved).context("emit cyclonedds descriptors")?;
117    Ok(())
118}

120#[derive(Debug)]
121struct ResolvedArgs {
122    idlc: PathBuf,
123    /// Recorded for parity with the spec; `idlc` itself drives include
124    /// resolution via the `-I` flag we don't currently emit (Cyclone's
125    /// in-tree IDL set is fully self-contained for the bundled types).
126    #[allow(dead_code)]
127    include: PathBuf,
128    crate_name: String,
129    out: PathBuf,
130    messages: Vec<(String, String, PathBuf)>, // (pkg, msg, source-path)
131}
132
133fn resolve_args(args: Args) -> Result<ResolvedArgs> {
134    if let Some(path) = args.args_file {
135        let body = fs::read_to_string(&path)
136            .with_context(|| format!("read --args-file {}", path.display()))?;
137        let doc: ArgsFile = serde_json::from_str(&body)
138            .with_context(|| format!("parse --args-file {} as JSON", path.display()))?;
139        let crate_name = doc
140            .crate_name
141            .unwrap_or_else(|| default_crate_name().to_string());
142        let messages = doc
143            .messages
144            .into_iter()
145            .map(|m| {
146                let (pkg, msg) = split_pkg_msg(&m.name)?;
147                Ok::<_, eyre::Report>((pkg, msg, m.source))
148            })
149            .collect::<Result<Vec<_>>>()?;
150        return Ok(ResolvedArgs {
151            idlc: doc.idlc,
152            include: doc.include,
153            crate_name,
154            out: doc.out,
155            messages,
156        });
157    }
158
159    let idlc = args
160        .idlc
161        .ok_or_else(|| eyre::eyre!("--idlc is required (or use --args-file)"))?;
162    let include = args
163        .include
164        .ok_or_else(|| eyre::eyre!("--include is required (or use --args-file)"))?;
165    let out = args
166        .out
167        .ok_or_else(|| eyre::eyre!("--out is required (or use --args-file)"))?;
168    if args.msg.is_empty() {
169        bail!("at least one --msg <pkg/Msg=<path>> is required");
170    }
171
172    let messages = args
173        .msg
174        .iter()
175        .map(|raw| {
176            let (left, source) = raw
177                .split_once('=')
178                .ok_or_else(|| eyre::eyre!("--msg expects `<pkg>/<Msg>=<path>`, got {raw:?}"))?;
179            let (pkg, msg) = split_pkg_msg(left)?;
180            Ok::<_, eyre::Report>((pkg, msg, PathBuf::from(source)))
181        })
182        .collect::<Result<Vec<_>>>()?;
183
184    Ok(ResolvedArgs {
185        idlc,
186        include,
187        crate_name: args
188            .crate_name
189            .unwrap_or_else(|| default_crate_name().to_string()),
190        out,
191        messages,
192    })
193}
194
195fn default_crate_name() -> &'static str {
196    "nros_descriptors"
197}

199fn split_pkg_msg(s: &str) -> Result<(String, String)> {
200    let (p, m) = s
201        .split_once('/')
202        .ok_or_else(|| eyre::eyre!("--msg expects `<pkg>/<Msg>`, got {s:?}"))?;
203    if p.is_empty() || m.is_empty() {
204        bail!("--msg expects non-empty `<pkg>/<Msg>`, got {s:?}");
205    }
206    Ok((p.to_string(), m.to_string()))
207}

209fn emit(args: &ResolvedArgs) -> Result<()> {
210    if !args.idlc.is_file() {
211        bail!(
212            "--idlc points at {} which is not a file",
213            args.idlc.display()
214        );
215    }
216    fs::create_dir_all(&args.out)
217        .with_context(|| format!("create --out dir {}", args.out.display()))?;
218
219    let mut entries: Vec<ManifestEntry> = Vec::with_capacity(args.messages.len());
220    // Track stems to detect collisions.
221    let mut stems: BTreeSet<String> = BTreeSet::new();
222
223    for (pkg, msg, source) in &args.messages {
224        let stem = format!("{pkg}_{msg}");
225        if !stems.insert(stem.clone()) {
226            bail!("duplicate --msg stem `{stem}` (pkg=`{pkg}` msg=`{msg}`)");
227        }
228
229        let msg_source = fs::read_to_string(source).with_context(|| {
230            format!(
231                "read .msg source for {pkg}/{msg} at {}",
232                source.display()
233            )
234        })?;
235        let idl_text = nros_msg_to_idl::msg_to_idl(&msg_source, pkg, msg)
236            .map_err(|e| eyre::eyre!("nros-msg-to-idl({pkg}/{msg}): {e}"))?;
237
238        let idl_path = args.out.join(format!("{stem}.idl"));
239        fs::write(&idl_path, &idl_text)
240            .with_context(|| format!("write {}", idl_path.display()))?;
241
242        // Shell idlc -t -l c -o <out> <idl-file>. Matches the helper
243        // in nros-rmw-cyclonedds-sys/build.rs (Phase 212.K.2).
244        let status = Command::new(&args.idlc)
245            .args(["-t", "-l", "c", "-o"])
246            .arg(&args.out)
247            .arg(&idl_path)
248            .status()
249            .with_context(|| format!("spawn idlc at {}", args.idlc.display()))?;
250        if !status.success() {
251            bail!(
252                "idlc failed on {} (status: {status})",
253                idl_path.display()
254            );
255        }
256
257        // idlc names outputs after the IDL stem.
258        let c_path = args.out.join(format!("{stem}.c"));
259        let h_path = args.out.join(format!("{stem}.h"));
260        if !c_path.is_file() || !h_path.is_file() {
261            bail!(
262                "idlc did not emit the expected `{stem}.{{c,h}}` pair in {}",
263                args.out.display()
264            );
265        }
266
267        let type_name = format!("{pkg}::msg::dds_::{msg}_");
268        let descriptor_symbol = format!("{pkg}_msg_dds__{msg}__desc");
269
270        entries.push(ManifestEntry {
271            pkg: pkg.clone(),
272            msg: msg.clone(),
273            idl_path,
274            c_path,
275            h_path,
276            type_name,
277            descriptor_symbol,
278        });
279    }
280
281    // Emit register.{c,h}.
282    let register_entry = format!("{}_register_descriptors", c_ident(&args.crate_name));
283    let register_h = args.out.join("register.h");
284    let register_c = args.out.join("register.c");
285    fs::write(&register_h, render_register_h(&register_entry))
286        .with_context(|| format!("write {}", register_h.display()))?;
287    fs::write(&register_c, render_register_c(&register_entry, &entries))
288        .with_context(|| format!("write {}", register_c.display()))?;
289
290    let manifest = Manifest {
291        crate_name: args.crate_name.clone(),
292        register_entry,
293        register_c: register_c.clone(),
294        register_h: register_h.clone(),
295        descriptors: entries,
296    };
297    let manifest_path = args.out.join("cyclonedds-descriptors.json");
298    let body = serde_json::to_string_pretty(&manifest)
299        .context("serialize cyclonedds-descriptors.json")?;
300    fs::write(&manifest_path, body)
301        .with_context(|| format!("write {}", manifest_path.display()))?;
302
303    Ok(())
304}

306fn render_register_h(entry: &str) -> String {
307    let mut s = String::new();
308    s.push_str("/* Auto-generated by `nros codegen cyclonedds-descriptors`. */\n");
309    s.push_str("#ifndef NROS_CYCLONEDDS_DESCRIPTORS_REGISTER_H\n");
310    s.push_str("#define NROS_CYCLONEDDS_DESCRIPTORS_REGISTER_H\n\n");
311    s.push_str("#ifdef __cplusplus\nextern \"C\" {\n#endif\n\n");
312    s.push_str(&format!("void {entry}(void);\n\n"));
313    s.push_str("#ifdef __cplusplus\n}\n#endif\n\n");
314    s.push_str("#endif /* NROS_CYCLONEDDS_DESCRIPTORS_REGISTER_H */\n");
315    s
316}
317
318fn render_register_c(entry: &str, descriptors: &[ManifestEntry]) -> String {
319    let mut s = String::new();
320    s.push_str("/* Auto-generated by `nros codegen cyclonedds-descriptors`. */\n");
321    s.push_str("#include \"dds/dds.h\"\n");
322    for d in descriptors {
323        let stem = format!("{}_{}", d.pkg, d.msg);
324        s.push_str(&format!("#include \"{stem}.h\"\n"));
325    }
326    s.push_str("\n");
327    for d in descriptors {
328        s.push_str(&format!(
329            "extern const dds_topic_descriptor_t {};\n",
330            d.descriptor_symbol
331        ));
332    }
333    s.push_str("\n");
334    s.push_str(
335        "extern void nros_rmw_cyclonedds_register_descriptor(\n    const char *type_name, const dds_topic_descriptor_t *desc);\n\n",
336    );
337    s.push_str(&format!("void {entry}(void) {{\n"));
338    for d in descriptors {
339        s.push_str(&format!(
340            "    nros_rmw_cyclonedds_register_descriptor(\n        \"{}\",\n        &{});\n",
341            c_escape(&d.type_name),
342            d.descriptor_symbol,
343        ));
344    }
345    s.push_str("}\n\n");
346    // Also expose a constructor-priority hook so a +whole-archive
347    // link pulls the call-sites in even when the consumer forgets
348    // to invoke the explicit register entry.
349    s.push_str(&format!(
350        "__attribute__((constructor))\nstatic void {entry}_ctor(void) {{\n    {entry}();\n}}\n"
351    ));
352    s
353}

355fn c_ident(s: &str) -> String {
356    let mut out = String::with_capacity(s.len());
357    for c in s.chars() {
358        if c.is_ascii_alphanumeric() || c == '_' {
359            out.push(c);
360        } else {
361            out.push('_');
362        }
363    }
364    if out.chars().next().map_or(false, |c| c.is_ascii_digit()) {
365        out.insert(0, '_');
366    }
367    out
368}
369
370fn c_escape(s: &str) -> String {
371    let mut out = String::with_capacity(s.len());
372    for c in s.chars() {
373        match c {
374            '\\' => out.push_str("\\\\"),
375            '"' => out.push_str("\\\""),
376            '\n' => out.push_str("\\n"),
377            c if (c as u32) < 0x20 => {
378                out.push_str(&format!("\\x{:02x}", c as u32));
379            }
380            c => out.push(c),
381        }
382    }
383    out
384}

407    /// Stub `idlc` that takes `-t -l c -o <out> <idl>` and writes
408    /// `<stem>.c` + `<stem>.h` matching the real idlc's emit shape
409    /// (just enough for the verb's output checks).
410    fn write_stub_idlc(dir: &Path) -> PathBuf {
411        let path = dir.join("idlc");
412        let script = r#"#!/usr/bin/env bash
413set -euo pipefail
414out=""
415idl=""
416while [[ $# -gt 0 ]]; do
417    case "$1" in
418        -t|-l) shift; shift; continue;;
419        -o) out="$2"; shift; shift; continue;;
420        *) idl="$1"; shift;;
421    esac
422done
423stem="$(basename "$idl" .idl)"
424cat > "${out}/${stem}.c" <<EOF
425/* stub idlc output for ${stem} */
426const int ${stem}_marker = 1;
427EOF
428cat > "${out}/${stem}.h" <<EOF
429/* stub idlc header for ${stem} */
430#ifndef ${stem}_H
431#define ${stem}_H
432extern const int ${stem}_marker;
433#endif
434EOF
435"#;
436        fs::write(&path, script).expect("write stub idlc");
437        let mut perm = fs::metadata(&path).unwrap().permissions();
438        perm.set_mode(0o755);
439        fs::set_permissions(&path, perm).unwrap();
440        path
441    }
