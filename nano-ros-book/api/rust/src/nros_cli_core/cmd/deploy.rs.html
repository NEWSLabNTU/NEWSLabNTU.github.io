1//! `nros deploy` — Phase 172 WP-A command-runner.
2//!
3//! Runs a `[deploy.<name>]` target from the root `nros.toml`: assert the
4//! vendor pin → emit the entry-lib form (WP-B; stubbed) → run `build[]` →
5//! `package[]`, substituting `{self}` / `{entry_lib}` / `{entry_src}` /
6//! `{entry_header}` / `{board}` / `{target}` / `{vendor.dir}` into each shell
7//! step. No per-vendor code lives here — the vendor knowledge is the
8//! user-authored `build[]` / `package[]` lines.
9

31/// Tokens the runner substitutes in `build[]` / `package[]` steps. A `{token}`
32/// for one of these that the target can't resolve is an error; any other
33/// `{...}` is left verbatim (shell brace syntax).
34const KNOWN_VARS: &[&str] = &[
35    "self",
36    "entry_lib",
37    "entry_src",
38    "entry_header",
39    "board",
40    "target",
41    "vendor.dir",
42];

63pub fn run(args: Args) -> Result<()> {
64    let config_path = resolve_deploy_config(&args.config)?;
65    // Absolutize so the workspace root (and the component path-deps the
66    // generator renders against it) is independent of the cwd a relative
67    // `--config nros.toml` was run from — otherwise an out-of-tree vendor
68    // build (e.g. Zephyr `west`, which compiles from its own build dir)
69    // can't resolve a `./src/<pkg>` relative dep.
70    let config_path = config_path.canonicalize().unwrap_or(config_path);
71    let cfg = WorkspaceConfig::load(&config_path)?;
72    let root = config_path
73        .parent()
74        .filter(|p| !p.as_os_str().is_empty())
75        .map(Path::to_path_buf)
76        .unwrap_or_else(|| PathBuf::from("."));
77    let (name, deploy) = resolve(&cfg, args.name.as_deref())?;
78    let nano_ros = args
79        .nano_ros_workspace
80        .clone()
81        .or_else(|| std::env::var_os("NROS_WORKSPACE").map(PathBuf::from));
82    deploy_target(
83        &cfg,
84        &root,
85        &name,
86        deploy,
87        nano_ros.as_deref(),
88        args.dry_run,
89    )
90}

92/// Resolve the workspace-root `nros.toml` this deploy operates on (Phase 172
93/// W.1, slice 3). If `--config` already points at a `[workspace]` file, use it.
94/// Otherwise — a `[component]`/`[node]` file, or a missing path — walk up to the
95/// nearest enclosing `[workspace]` root (Cargo-style). A non-workspace manifest
96/// with no enclosing workspace is a clear, kind-specific error: a deploy needs a
97/// system to deploy, which only a `[workspace]` root declares.
98fn resolve_deploy_config(config: &Path) -> Result<PathBuf> {
99    if config.is_file() {
100        match probe_manifest_kind(config)? {
101            ManifestKind::Workspace => return Ok(config.to_path_buf()),
102            kind => {
103                return resolve_workspace_root(config)?
104                    .ok_or_else(|| no_workspace_err(config, kind));
105            }
106        }
107    }
108    // No file at the given path → walk up from its directory.
109    let start = config
110        .parent()
111        .filter(|p| !p.as_os_str().is_empty())
112        .map(Path::to_path_buf)
113        .unwrap_or_else(|| PathBuf::from("."));
114    resolve_workspace_root(&start)?.ok_or_else(|| {
115        eyre!(
116            "nros deploy: no workspace nros.toml found from {} — a deploy needs a [workspace] \
117             root; pass --config <root nros.toml>",
118            config.display()
119        )
120    })
121}

123/// The "this isn't a workspace, and none encloses it" message, tailored to the
124/// manifest's actual kind so the user knows the right next move.
125fn no_workspace_err(config: &Path, kind: ManifestKind) -> eyre::Report {
126    match kind {
127        ManifestKind::DirectNode => eyre!(
128            "{}: this is a direct-mode node config (no [workspace]) — build it directly with \
129             `nros build`, or add it to a workspace's [system].components",
130            config.display()
131        ),
132        ManifestKind::Component => eyre!(
133            "{}: this is a component manifest (no [workspace]) and no enclosing workspace \
134             nros.toml was found — add it to a workspace's [system].components, or run from \
135             inside the workspace",
136            config.display()
137        ),
138        _ => eyre!(
139            "{}: not a workspace nros.toml (no [workspace] table) and no enclosing workspace \
140             was found",
141            config.display()
142        ),
143    }
144}

146/// Resolve the target by name, or fall back to `[workspace].default`.
147fn resolve<'c>(cfg: &'c WorkspaceConfig, name: Option<&str>) -> Result<(String, &'c DeployTarget)> {
148    match name {
149        Some(n) => cfg
150            .deploy
151            .get(n)
152            .map(|d| (n.to_string(), d))
153            .ok_or_else(|| eyre!("no [deploy.{n}] in the root nros.toml")),
154        None => cfg
155            .default_deploy()
156            .map(|(n, d)| (n.clone(), d))
157            .ok_or_else(|| {
158                eyre!("no deploy name given and no [workspace].default set in the root nros.toml")
159            }),
160    }
161}

163fn deploy_target(
164    cfg: &WorkspaceConfig,
165    root: &Path,
166    name: &str,
167    deploy: &DeployTarget,
168    nano_ros: Option<&Path>,
169    dry_run: bool,
170) -> Result<()> {
171    let sys = cfg.system_for(deploy).ok_or_else(|| {
172        eyre!("deploy '{name}': no resolvable [system] (set `system = \"<name>\"`)")
173    })?;
174
175    assert_pin(root, name, deploy)?;
176
177    // Real run: drive metadata → plan → entry-lib emit (WP-B). `--dry-run`
178    // skips the heavy pipeline and just resolves + prints the steps.
179    if !dry_run {
180        let nano_ros = nano_ros.ok_or_else(|| {
181            eyre!(
182                "deploy '{name}': need the nano-ros workspace — pass \
183                 --nano-ros-workspace <path> or set NROS_WORKSPACE"
184            )
185        })?;
186        // Phase 187.6 (Method A): lazy-install the board's host tools, then put
187        // them on PATH for every child we spawn below (opt out: NROS_NO_AUTO_SETUP).
188        let bins = crate::cmd::setup::ensure_tools(
189            deploy.board.as_deref().unwrap_or("native"),
190            Some(nano_ros),
191        )?;
192        crate::cmd::setup::activate_store_path(&bins);
193        emit_entry_lib(root, name, sys, deploy, nano_ros)?;
194    }
195
196    let vars = build_vars(root, name, deploy)?;
197    run_phase("build", &deploy.build, &vars, root, dry_run)?;
198    run_phase("package", &deploy.package, &vars, root, dry_run)?;
199
200    if !dry_run {
201        eprintln!("nros deploy: {name} complete");
202    }
203    Ok(())
204}

206/// Drift guard: when a vendor pin is declared, the vendor dir must resolve +
207/// exist. The exact version compare is vendor-specific (WP-C per-vendor);
208/// here we assert presence and surface the expected pin.
209fn assert_pin(root: &Path, name: &str, deploy: &DeployTarget) -> Result<()> {
210    let Some(vendor) = &deploy.vendor else {
211        return Ok(());
212    };
213    let Some(pin) = &vendor.pin else {
214        return Ok(());
215    };
216    let dir = vendor.dir.resolve().map(|d| abs(root, &d)).ok_or_else(|| {
217        eyre!(
218            "deploy '{name}': vendor pinned at '{pin}' but its dir is unset \
219                 (set the env var or a default)"
220        )
221    })?;
222    if !dir.exists() {
223        bail!(
224            "deploy '{name}': vendor dir {} not found (pin '{pin}' expects it) — \
225             install the SDK or set the env var",
226            dir.display()
227        );
228    }
229    eprintln!(
230        "nros deploy: {name} vendor pinned '{pin}' at {}",
231        dir.display()
232    );
233    Ok(())
234}

236/// Generated-artifact layout for a deploy target. Pure path computation
237/// (deterministic from the target triple + name), so it works in `--dry-run`
238/// before anything is generated and matches what `emit_entry_lib` produces.
239struct EntryPaths {
240    /// Output root: `<root>/build/<name>/nros`.
241    out_root: PathBuf,
242    /// Generated entry crate (`{entry_src}`).
243    src_dir: PathBuf,
244    /// cbindgen C header (`{entry_header}`).
245    header: PathBuf,
246    /// Compiled staticlib (`{entry_lib}`); exists only after a compiled-form build.
247    lib: PathBuf,
248}
249
250fn entry_paths(root: &Path, name: &str, deploy: &DeployTarget) -> Result<EntryPaths> {
251    let triple = deploy.target.as_deref().ok_or_else(|| {
252        eyre!("deploy '{name}': set `target = \"<triple>\"` (needed for the entry lib)")
253    })?;
254    let out_root = root.join("build").join(name).join("nros");
255    let src_dir = out_root.join("generated");
256    let pkg = package_name(name);
257    Ok(EntryPaths {
258        header: src_dir
259            .join("include")
260            .join(format!("{}.h", system_ident(name))),
261        lib: out_root
262            .join("target")
263            .join(triple)
264            .join("debug")
265            .join(format!("lib{}.a", pkg.replace('-', "_"))),
266        src_dir,
267        out_root,
268    })
269}

271/// Generated package name → drives the staticlib name (`lib<pkg>.a`).
272fn package_name(name: &str) -> String {
273    format!("nros-{name}")
274}

276/// System identifier baked into the header name (the `nros plan` system pkg,

282/// Drive metadata → plan → entry-lib emission (WP-B). Compiled-form kinds
283/// (`self` / `vendor-lib`) build the staticlib (+ self-shim binary); the
284/// source form (`vendor-module`) only generates the crate + CMake fragment for
285/// the vendor toolchain to compile.
286fn emit_entry_lib(
287    root: &Path,
288    name: &str,
289    sys: &SystemSection,
290    deploy: &DeployTarget,
291    nano_ros: &Path,
292) -> Result<()> {
293    let launch = sys.launch.as_ref().ok_or_else(|| {
294        eyre!("deploy '{name}': [system].launch is required for a planned deploy")
295    })?;
296    let paths = entry_paths(root, name, deploy)?;
297    let system_pkg = name.to_string();
298
299    std::fs::create_dir_all(&paths.out_root)
300        .wrap_err_with(|| format!("create {}", paths.out_root.display()))?;
301    let overlay = synth_build_overlay(&paths.out_root, sys, deploy)?;
302
303    metadata::run(metadata::Args {
304        system_pkg: system_pkg.clone(),
305        workspace: Some(root.to_path_buf()),
306        out_dir: Some(paths.out_root.clone()),
307        metadata: Vec::new(),
308        // A real deploy produces any missing component source-metadata via the
309        // metadata-mode build (172.E driver), using the same nano-ros workspace.
310        build: true,
311        nano_ros_workspace: Some(nano_ros.to_path_buf()),
312    })
313    .wrap_err("deploy: metadata step")?;
314
315    plan::run(plan::Args {
316        system_pkg: system_pkg.clone(),
317        launch_file: root.join(launch),
318        record: None,
319        file: None,
320        exec: None,
321        workspace: Some(root.to_path_buf()),
322        out_dir: Some(paths.out_root.clone()),
323        metadata: Vec::new(),
324        manifests: Vec::new(),
325        nros_toml: vec![overlay],
326        launch_args: Vec::new(),
327    })
328    .wrap_err("deploy: plan step")?;
329
330    let plan_path = paths.out_root.join("nros-plan.json");
331    // Phase 172.K.5 — stamp each plan node with the ROS domain from its root
332    // `[system].[[domain]]` group, so the generator opens a session per domain
333    // and routes each node to its own (in-binary multi-domain).
334    apply_domain_groups(&plan_path, &sys.domain)?;
335    // Phase 172 — carry the root `[system].[[bridge]]` gateways into the plan;
336    // the generator resolves each forwarded topic's type from the plan's
337    // interfaces and emits the raw sub→pub forwarding.
338    apply_bridges(&plan_path, &sys.bridge)?;
339    let emit = deploy.emit.unwrap_or_else(|| deploy.kind.default_emit());
340    match emit {
341        EmitForm::Source => {
342            generate_package(&GenerateOptions {
343                package_name: package_name(name),
344                output_dir: paths.src_dir.clone(),
345                plan_path,
346                nros_path: nano_ros.join("packages/core/nros"),
347                nros_orchestration_path: nano_ros.join("packages/core/nros-orchestration"),
348                component_workspace: Some(root.to_path_buf()),
349            })
350            .wrap_err("deploy: generate entry-lib source")?;
351            eprintln!(
352                "nros deploy: {name} entry-lib source at {}",
353                paths.src_dir.display()
354            );
355        }
356        EmitForm::Compiled => {
357            orchestration::build::build_generated_package(&orchestration::build::BuildOptions {
358                package_name: package_name(name),
359                output_dir: paths.src_dir.clone(),
360                plan_path,
361                workspace_root: nano_ros.to_path_buf(),
362                component_workspace: Some(root.to_path_buf()),
363                release: false,
364                target: deploy.target.clone(),
365                cargo_args: Vec::new(),
366                force: false,
367            })
368            .wrap_err("deploy: build entry lib")?;
369            eprintln!(
370                "nros deploy: {name} entry-lib staticlib at {}",
371                paths.lib.display()
372            );
373        }
374    }
375    Ok(())
376}

378/// Bridge the root `[system]`/`[deploy]` into the planner's `[build]` overlay
379/// (target/board/rmw) so the generated entry lib lowers the right RMW. Written
380/// into the deploy's build dir (ephemeral).
381fn synth_build_overlay(
382    out_root: &Path,
383    sys: &SystemSection,
384    deploy: &DeployTarget,
385) -> Result<PathBuf> {
386    let target = deploy
387        .target
388        .as_deref()
389        .unwrap_or("x86_64-unknown-linux-gnu");
390    let board = deploy.board.as_deref().unwrap_or("native");
391    let rmw = deploy
392        .rmw
393        .as_deref()
394        .or(sys.rmw.as_deref())
395        .unwrap_or("zenoh");
396    let mut body =
397        format!("[build]\ntarget = \"{target}\"\nboard = \"{board}\"\nrmw = \"{rmw}\"\n");
398    // W.4 — bake the locator (where the agent/peer is) as a synthetic
399    // `[[transport]]`; the generator surfaces it as `TRANSPORT_LOCATOR`. An
400    // ethernet transport with only a locator (no ip) is valid — networking on
401    // native_sim is host-offloaded (NSOS), real boards add ip via their overlay.
402    if let Some(locator) = &deploy.locator {
403        body.push_str(&format!(
404            "\n[[transport]]\nkind = \"ethernet\"\nlocator = \"{locator}\"\n"
405        ));
406    }
407    let path = out_root.join("deploy-build-overlay.toml");
408    std::fs::write(&path, body).wrap_err_with(|| format!("write {}", path.display()))?;
409    Ok(path)
410}

412/// Phase 172.K.5 — stamp each plan node with the ROS domain from its root
413/// `[system].[[domain]]` group (matched by node name). The generator then opens
414/// a session per distinct domain (`SESSION_SPECS` + `open_multi`) and routes
415/// each node to its own via `NodeBuilder::session_idx`. No groups ⇒ no-op.
416fn apply_domain_groups(plan_path: &Path, groups: &[DomainGroup]) -> Result<()> {
417    if groups.is_empty() {
418        return Ok(());
419    }
420    let raw = std::fs::read_to_string(plan_path)
421        .wrap_err_with(|| format!("read plan {}", plan_path.display()))?;
422    let mut plan: orchestration::plan::NrosPlan =
423        serde_json::from_str(&raw).wrap_err("parse plan for domain assignment")?;
424    let mut changed = false;
425    for instance in &mut plan.instances {
426        for node in &mut instance.nodes {
427            if let Some(group) = groups.iter().find(|g| node_in_group(node, g)) {
428                node.domain_id = Some(group.id);
429                changed = true;
430            }
431        }
432    }
433    if changed {
434        let pretty =
435            serde_json::to_string_pretty(&plan).wrap_err("serialize domain-stamped plan")?;
436        std::fs::write(plan_path, pretty)
437            .wrap_err_with(|| format!("write plan {}", plan_path.display()))?;
438    }
439    Ok(())
440}

442/// A `[[domain]]` group matches a plan node by its resolved graph name or that
443/// name's short (last-segment) form.
444fn node_in_group(node: &orchestration::plan::PlanNode, group: &DomainGroup) -> bool {
445    let resolved = node.resolved_name.as_str();
446    let short = resolved.rsplit('/').next().unwrap_or(resolved);
447    group.nodes.iter().any(|n| n == resolved || n == short)
448}

450/// Phase 172 — copy the root `[system].[[bridge]]` gateways into the plan
451/// (`plan.bridges`). The generator resolves each forwarded topic's type from
452/// the plan's `interfaces` and emits the raw sub→pub forwarding across the
453/// connected sessions. No bridges ⇒ no-op.
454fn apply_bridges(plan_path: &Path, bridges: &[BridgeSpec]) -> Result<()> {
455    if bridges.is_empty() {
456        return Ok(());
457    }
458    let raw = std::fs::read_to_string(plan_path)
459        .wrap_err_with(|| format!("read plan {}", plan_path.display()))?;
460    let mut plan: orchestration::plan::NrosPlan =
461        serde_json::from_str(&raw).wrap_err("parse plan for bridge wiring")?;
462    plan.bridges = bridges.iter().map(plan_bridge).collect();
463    let pretty = serde_json::to_string_pretty(&plan).wrap_err("serialize bridge plan")?;
464    std::fs::write(plan_path, pretty)
465        .wrap_err_with(|| format!("write plan {}", plan_path.display()))?;
466    Ok(())
467}

469/// Map a root `[[bridge]]` to its plan form (the generator resolves topic types
470/// from the plan's interfaces + emits the forwarding).
471fn plan_bridge(spec: &BridgeSpec) -> orchestration::plan::PlanBridge {
472    orchestration::plan::PlanBridge {
473        name: spec.name.clone(),
474        connect: spec
475            .connect
476            .iter()
477            .map(|e| orchestration::plan::PlanBridgeEndpoint {
478                rmw: e.rmw.clone(),
479                domain: e.domain,
480                locator: e.locator.clone(),
481            })
482            .collect(),
483        topics: spec.topics.clone(),
484    }
485}

487type Vars = BTreeMap<&'static str, String>;
488
489fn build_vars(root: &Path, name: &str, deploy: &DeployTarget) -> Result<Vars> {
490    let mut v = Vars::new();
491
492    // Real entry-lib artifact paths (match what `emit_entry_lib` produces).
493    let paths = entry_paths(root, name, deploy)?;
494    v.insert("entry_lib", paths.lib.display().to_string());
495    v.insert("entry_src", paths.src_dir.display().to_string());
496    v.insert("entry_header", paths.header.display().to_string());
497
498    if let Some(self_dir) = &deploy.self_dir {
499        v.insert("self", abs(root, Path::new(self_dir)).display().to_string());
500    }
501    if let Some(board) = &deploy.board {
502        v.insert("board", board.clone());
503    }
504    if let Some(target) = &deploy.target {
505        v.insert("target", target.clone());
506    }
507    if let Some(vendor) = &deploy.vendor
508        && let Some(dir) = vendor.dir.resolve()
509    {
510        v.insert("vendor.dir", abs(root, &dir).display().to_string());
511    }
512    Ok(v)
513}

515/// Substitute the known `{token}`s. A referenced-but-undefined known token is
516/// an error (catches a target that forgot `self`/`board`/…); unknown `{...}`
517/// is left for the shell.
518fn substitute(template: &str, vars: &Vars) -> Result<String> {
519    for tok in KNOWN_VARS {
520        if template.contains(&format!("{{{tok}}}")) && !vars.contains_key(*tok) {
521            bail!("deploy step references {{{tok}}} but this target doesn't define it");
522        }
523    }
524    let mut out = template.to_string();
525    for (k, val) in vars {
526        out = out.replace(&format!("{{{k}}}"), val);
527    }
528    Ok(out)
529}

531fn run_phase(phase: &str, steps: &[String], vars: &Vars, root: &Path, dry_run: bool) -> Result<()> {
532    let n = steps.len();
533    for (i, step) in steps.iter().enumerate() {
534        let cmd = substitute(step, vars)?;
535        if dry_run {
536            println!("[{phase} {}/{n}] {cmd}", i + 1);
537            continue;
538        }
539        eprintln!("nros deploy: [{phase} {}/{n}] {cmd}", i + 1);
540        let status = Command::new("sh")
541            .arg("-c")
542            .arg(&cmd)
543            .current_dir(root)
544            .stdout(Stdio::inherit())
545            .stderr(Stdio::inherit())
546            .status()
547            .wrap_err_with(|| format!("spawn {phase} step: {cmd}"))?;
548        if !status.success() {
549            bail!(
550                "{phase} step {}/{n} failed (exit {}): {cmd}",
551                i + 1,
552                status.code().unwrap_or(-1)
553            );
554        }
555    }
556    Ok(())
557}

559/// A relative path is resolved against the workspace root; an absolute path is
560/// kept as-is.
561fn abs(root: &Path, p: &Path) -> PathBuf {
562    if p.is_absolute() {
563        p.to_path_buf()
564    } else {
565        root.join(p)
566    }
