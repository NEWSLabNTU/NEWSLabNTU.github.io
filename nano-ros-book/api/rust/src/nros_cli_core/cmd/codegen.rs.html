1//! `nros codegen` — build-tool-facing C/C++ binding generation.
2//!
3//! Phase 195.A: folds the former standalone `nros-codegen` binary
4//! (`nros-codegen-c`) into the canonical `nros` CLI. Same engine
5//! (`cargo_nano_ros`), same call shape, so the cmake / build.rs consumers
6//! only change the program name (`nros-codegen …` → `nros codegen …`):
7//!
8//!   nros codegen --args-file <path> [--language c|cpp] [--verbose]
9//!   nros codegen resolve-deps --package-xml <path> --output-cmake <path> [--verbose]
10//!
11//! Distinct from `nros generate` (the user-facing, `package.xml`-driven surface):
12//! this is the JSON-`--args-file` contract the build system already speaks.
13
14use clap::{Args as ClapArgs, Subcommand};
15use eyre::{Result, bail, eyre};
16use std::path::PathBuf;
17
18#[derive(Debug, ClapArgs)]
19pub struct Args {
20    #[command(subcommand)]
21    pub command: Option<Sub>,
22
23    /// Path to the JSON arguments file (default generate mode)
24    #[arg(long)]
25    pub args_file: Option<PathBuf>,
26
27    /// Target language: "c" (default) or "cpp"
28    #[arg(long, default_value = "c")]
29    pub language: String,
30
31    /// Verbose output
32    #[arg(long)]
33    pub verbose: bool,
34}
35
36#[derive(Debug, Subcommand)]
37pub enum Sub {
38    /// Resolve interface dependencies from package.xml → a CMake script
39    ResolveDeps {
40        /// Path to package.xml
41        #[arg(long)]
42        package_xml: PathBuf,
43
44        /// Path to output .cmake file
45        #[arg(long)]
46        output_cmake: PathBuf,
47
48        /// Verbose output
49        #[arg(long)]
50        verbose: bool,
51    },
52
53    /// Phase 212.K.4 — emit per-example Cyclone-DDS topic descriptors.
54    ///
55    /// Synthesises Cyclone-shaped IDL from one or more `.msg` sources,
56    /// drives the host `idlc` to produce `<pkg>_<Msg>.{c,h}` pairs, and
57    /// writes a `register.{c,h}` + JSON manifest the consumer build
58    /// script feeds into `cc::Build`.
59    #[command(name = "cyclonedds-descriptors")]
60    CycloneddsDescriptors(super::codegen_cyclonedds_descriptors::Args),
61}
62
63pub fn run(args: Args) -> Result<()> {
64    match args.command {
65        Some(Sub::ResolveDeps {
66            package_xml,
67            output_cmake,
68            verbose,
69        }) => cargo_nano_ros::resolve_deps_from_package_xml(cargo_nano_ros::ResolveDepsConfig {
70            package_xml,
71            output_cmake,
72            verbose,
73        })
74        .map_err(|e| eyre!("{e:#}")),
75        Some(Sub::CycloneddsDescriptors(sub_args)) => {
76            super::codegen_cyclonedds_descriptors::run(sub_args)
77        }
78        None => {
79            let Some(args_file) = args.args_file else {
80                bail!("nros codegen: --args-file is required (or use a subcommand)");
81            };
82            match args.language.as_str() {
83                "c" => cargo_nano_ros::generate_c_from_args_file(cargo_nano_ros::GenerateCConfig {
84                    args_file,
85                    verbose: args.verbose,
86                })
87                .map_err(|e| eyre!("{e:#}")),
88                "cpp" => {
89                    cargo_nano_ros::generate_cpp_from_args_file(cargo_nano_ros::GenerateCppConfig {
90                        args_file,
91                        verbose: args.verbose,
92                    })
93                    .map_err(|e| eyre!("{e:#}"))
94                }
95                other => {
96                    bail!("nros codegen: unsupported language '{other}' (expected 'c' or 'cpp')")
97                }
98            }
99        }
100    }
101}
