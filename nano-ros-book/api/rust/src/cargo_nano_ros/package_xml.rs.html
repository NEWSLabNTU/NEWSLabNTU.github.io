1//! Package.xml parser for extracting ROS 2 dependencies
2//!
3//! This module parses package.xml files to extract interface dependencies
4//! (std_msgs, geometry_msgs, etc.) that need bindings generated.
5
6use eyre::{Result, WrapErr, eyre};
7use quick_xml::{Reader, events::Event};
8use std::{collections::HashSet, path::Path};
9
10/// Parsed package.xml metadata
11#[derive(Debug, Clone)]
12pub struct PackageXml {
13    /// Package name
14    pub name: String,
15    /// Package version
16    pub version: String,
17    /// All dependencies (build, exec, depend)
18    pub dependencies: HashSet<String>,
19}
20
21impl PackageXml {
22    /// Parse a package.xml file
23    pub fn parse(path: &Path) -> Result<Self> {
24        let content = std::fs::read_to_string(path)
25            .wrap_err_with(|| format!("Failed to read {}", path.display()))?;
26
27        Self::parse_str(&content)
28    }
29
30    /// Parse package.xml from string content
31    pub fn parse_str(content: &str) -> Result<Self> {
32        let mut reader = Reader::from_str(content);
33        reader.config_mut().trim_text(true);
34
35        let mut name = None;
36        let mut version = None;
37        let mut dependencies = HashSet::new();
38
39        let mut current_tag = String::new();
40
41        loop {
42            match reader.read_event() {
43                Ok(Event::Start(e)) => {
44                    current_tag = String::from_utf8_lossy(e.name().as_ref()).to_string();
45                }
46                Ok(Event::Text(e)) => {
47                    let text = e.unescape().unwrap_or_default().to_string();
48                    match current_tag.as_str() {
49                        "name" if name.is_none() => {
50                            name = Some(text);
51                        }
52                        "version" if version.is_none() => {
53                            version = Some(text);
54                        }
55                        "depend" | "build_depend" | "exec_depend" | "build_export_depend" => {
56                            dependencies.insert(text);
57                        }
58                        _ => {}
59                    }
60                }
61                Ok(Event::End(_)) => {
62                    current_tag.clear();
63                }
64                Ok(Event::Eof) => break,
65                Err(e) => return Err(eyre!("XML parse error: {}", e)),
66                _ => {}
67            }
68        }
69
70        Ok(PackageXml {
71            name: name.ok_or_else(|| eyre!("Missing <name> in package.xml"))?,
72            version: version.unwrap_or_else(|| "0.0.0".to_string()),
73            dependencies,
74        })
75    }
76
77    /// Get all dependencies
78    pub fn all_dependencies(&self) -> &HashSet<String> {
79        &self.dependencies
80    }
81}

83#[cfg(test)]
84mod tests {
85    use super::*;
86
87    #[test]
88    fn test_parse_simple_package_xml() {
89        let xml = r#"<?xml version="1.0"?>
90<package format="3">
91  <name>my_package</name>
92  <version>1.0.0</version>
93  <description>Test package</description>
94  <maintainer email="test@test.com">Test</maintainer>
95  <license>Apache-2.0</license>
96
97  <depend>std_msgs</depend>
98  <depend>geometry_msgs</depend>
99  <build_depend>rosidl_default_generators</build_depend>
100  <exec_depend>rosidl_default_runtime</exec_depend>
101
102  <export>
103    <build_type>ament_cargo</build_type>
104  </export>
105</package>"#;
106
107        let pkg = PackageXml::parse_str(xml).unwrap();
108        assert_eq!(pkg.name, "my_package");
109        assert_eq!(pkg.version, "1.0.0");
110        assert!(pkg.dependencies.contains("std_msgs"));
111        assert!(pkg.dependencies.contains("geometry_msgs"));
112        assert!(pkg.dependencies.contains("rosidl_default_generators"));
113        assert!(pkg.dependencies.contains("rosidl_default_runtime"));
114    }
115
116    #[test]
117    fn test_parse_minimal_package_xml() {
118        let xml = r#"<?xml version="1.0"?>
119<package format="3">
120  <name>minimal</name>
121</package>"#;
122
123        let pkg = PackageXml::parse_str(xml).unwrap();
124        assert_eq!(pkg.name, "minimal");
125        assert_eq!(pkg.version, "0.0.0");
126        assert!(pkg.dependencies.is_empty());
127    }
