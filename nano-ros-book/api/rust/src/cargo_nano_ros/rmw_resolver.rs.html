1//! RMW selection lowering (Phase 227.2, design-of-record RFC-0031).
2//!
3//! RMW is a declared, language-agnostic value (`system.toml [system].rmw` /
4//! `[deploy.<t>].rmw` / CLI flag). This module is the single place that
5//! validates a declared RMW string and **lowers** it to each language's build
6//! mechanism — a Rust cargo feature and a CMake `-DNANO_ROS_RMW` value — plus the
7//! C `#define` token the bake emits. The cargo feature / CMake var are lowering
8//! targets, not the user-facing knob.
9//!
10//! Phase 248 C5b (RFC-0031 amendment) — the Rust lowering target is the **board
11//! crate's** `rmw-X` feature (the board self-links + registers the concrete
12//! backend), NOT an `nros/rmw-X` feature. `cargo_feature` therefore names the
13//! `rmw-X` feature codegen places on the entry's board dep; the `nros` umbrella
14//! stays RMW-agnostic.
15//!
16//! It lives in `cargo-nano-ros` (the lower crate) so both the scaffolder here
17//! and the orchestration loader in `nros-cli-core` share one mapping.
18
19use std::fmt;

21/// The RMW backends nano-ros supports today. `dust-dds` was retired (Phase 169).
22pub const KNOWN_RMW: &[&str] = &["zenoh", "xrce", "cyclonedds"];
23
24/// A declared RMW value lowered to its per-language build forms.
25#[derive(Debug, Clone, PartialEq, Eq)]
26pub struct ResolvedRmw {
27    /// The canonical declared name, e.g. `"zenoh"`.
28    pub declared: &'static str,
29    /// The board-crate cargo feature codegen lowers the RMW to, e.g.
30    /// `"rmw-zenoh"` (Phase 248 C5b: lands on the entry's board dep, not
31    /// `nros`). Board crate and `nros` share the `rmw-X` naming.
32    pub cargo_feature: &'static str,
33    /// The `-DNANO_ROS_RMW` CMake value, e.g. `"zenoh"`.
34    pub cmake_value: &'static str,
35    /// The C `#define NROS_SYSTEM_RMW_<TOKEN>` token, e.g. `"ZENOH"`.
36    pub c_define_token: &'static str,
37    /// Phase 241 W13/R1 (RFC-0042 §D3 bullet 2) — the **link dispatch** data,
38    /// the one place that records how each backend reaches the final link.
39    /// Consumed by both the W11 synthesized `nros_ws_runtime` crate's nros-cpp
40    /// feature and the cmake link extras (formerly hand-maintained prose).
41    pub dispatch: RmwDispatch,
42}
43
44/// How a backend reaches the final binary's link (Phase 241 W13/R1). One SSoT for
45/// the per-backend link requirement that cmake + the synthesized runtime crate
46/// both consumed via duplicated conditionals before.
47#[derive(Debug, Clone, PartialEq, Eq)]
48pub struct RmwDispatch {
49    /// The C/C++ umbrella (`nros-c`/`nros-cpp`) cffi feature that bundles +
50    /// force-links this backend, e.g. `"rmw-zenoh-cffi"`. The W11 synthesized
51    /// `nros_ws_runtime` crate sets this on its `nros-cpp` dep.
52    pub umbrella_cffi_feature: &'static str,
53    /// The pure-Rust backend crate force-linked **into** the umbrella as an rlib
54    /// dep, e.g. `Some("nros-rmw-zenoh")`. `None` for cyclonedds — it is a C++
55    /// library linked separately (see `extra_link_libs`), not a Rust rlib.
56    pub rlib_dep: Option<&'static str>,
57    /// Extra non-umbrella link libraries the final binary needs for this backend.
58    /// Empty for the pure-Rust backends (zenoh/xrce); cyclonedds pulls its C++
59    /// RMW wrapper + Cyclone + the C++ runtime (incl. for C binaries — the locked
60    /// design choice). Names are cmake link targets / `-l` stems.
61    pub extra_link_libs: &'static [&'static str],
62    /// Whether the final link must use the C++ linker driver (libstdc++ on the
63    /// line). True for cyclonedds (its wrapper is C++), even for C binaries.
64    pub needs_cxx_linker: bool,
65}

67/// A declared RMW value that is not one of [`KNOWN_RMW`].
68#[derive(Debug, Clone, PartialEq, Eq)]
69pub struct UnknownRmw {
70    pub declared: String,
71}
72
73impl fmt::Display for UnknownRmw {
74    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
75        write!(
76            f,
77            "unknown rmw `{}` (known: {})",
78            self.declared,
79            KNOWN_RMW.join(", ")
80        )
81    }
82}
83
84impl std::error::Error for UnknownRmw {}

86/// The canonical backend name (`"zenoh"` / `"xrce"` / `"cyclonedds"`) from any
87/// accepted spelling — the bare name, the `rmw-<x>` feature spelling, or the
88/// legacy `rmw-<x>-cffi`. `None` for empty / unknown.
89///
90/// This is the single alias table; the orchestration layer's `normalize_rmw`
91/// delegates here so there is one source of truth for RMW name recognition.
92pub fn canonical_rmw(input: &str) -> Option<&'static str> {
93    match input {
94        "zenoh" | "rmw-zenoh" | "rmw-zenoh-cffi" => Some("zenoh"),
95        "xrce" | "rmw-xrce" | "rmw-xrce-cffi" => Some("xrce"),
96        "cyclonedds" | "rmw-cyclonedds" | "rmw-cyclonedds-cffi" => Some("cyclonedds"),
97        _ => None,
98    }
99}

101/// Lower a declared RMW string to its per-language build forms.
102///
103/// Accepts the canonical name or any alias ([`canonical_rmw`]). An unknown
104/// value is an error — caught early in the loader so a typo fails with a clear
105/// message rather than producing a broken build.
106pub fn resolve_rmw(declared: &str) -> Result<ResolvedRmw, UnknownRmw> {
107    match canonical_rmw(declared) {
108        Some("zenoh") => Ok(ResolvedRmw {
109            declared: "zenoh",
110            cargo_feature: "rmw-zenoh",
111            cmake_value: "zenoh",
112            c_define_token: "ZENOH",
113            dispatch: RmwDispatch {
114                umbrella_cffi_feature: "rmw-zenoh-cffi",
115                rlib_dep: Some("nros-rmw-zenoh"),
116                extra_link_libs: &[],
117                needs_cxx_linker: false,
118            },
119        }),
120        Some("xrce") => Ok(ResolvedRmw {
121            declared: "xrce",
122            cargo_feature: "rmw-xrce",
123            cmake_value: "xrce",
124            c_define_token: "XRCE",
125            dispatch: RmwDispatch {
126                umbrella_cffi_feature: "rmw-xrce-cffi",
127                rlib_dep: Some("nros-rmw-xrce-cffi"),
128                extra_link_libs: &[],
129                needs_cxx_linker: false,
130            },
131        }),
132        Some("cyclonedds") => Ok(ResolvedRmw {
133            declared: "cyclonedds",
134            cargo_feature: "rmw-cyclonedds",
135            cmake_value: "cyclonedds",
136            c_define_token: "CYCLONEDDS",
137            dispatch: RmwDispatch {
138                umbrella_cffi_feature: "rmw-cyclonedds-cffi",
139                // cyclonedds is a C++ library, NOT a Rust rlib bundled in the
140                // umbrella — linked separately via `extra_link_libs`.
141                rlib_dep: None,
142                extra_link_libs: &["nros_rmw_cyclonedds", "ddsc", "stdc++"],
143                needs_cxx_linker: true,
144            },
145        }),
146        _ => Err(UnknownRmw {
147            declared: declared.to_string(),
148        }),
149    }
150}

152/// Phase 241 W13/R1 — render the per-backend link dispatch ([`RmwDispatch`]) as a
153/// CMake-includable `nros_rmw_dispatch(<rmw>)` function, the **generated** form of the
154/// formerly hand-maintained "RMW backend dispatch" prose (RFC-0042 §D3 bullet 2). The
155/// output is committed at `cmake/NanoRosRmwDispatch.cmake`; `rmw_cmake_dispatch_is_current`
156/// asserts the committed copy matches this renderer, so the SSoT is `resolve_rmw()` and
157/// drift fails the build. Consumed by `NanoRosRuntimeCrate.cmake` (the W11 synthesized
158/// runtime crate's nros-cpp cffi feature) and the cmake Cyclone link block.
159pub fn render_cmake_dispatch() -> String {
160    let mut out = String::new();
161    out.push_str(
162        "# Generated from cargo-nano-ros `resolve_rmw()` — DO NOT EDIT.\n\
163         # Regenerate: `cargo test -p cargo-nano-ros rmw_cmake_dispatch_is_current -- --ignored`\n\
164         # (or run the bin helper). The SSoT is rmw_resolver.rs; this is its CMake lowering.\n\
165         #\n\
166         # nros_rmw_dispatch(<rmw>) sets in the CALLER scope:\n\
167         #   NROS_RMW_UMBRELLA_CFFI_FEATURE  the nros-c/nros-cpp cffi feature (e.g. rmw-zenoh-cffi)\n\
168         #   NROS_RMW_RLIB_DEP               backend rlib crate bundled in the umbrella, or \"\"\n\
169         #   NROS_RMW_EXTRA_LINK_LIBS        ;-list of extra link libs (cyclonedds C++ path), or \"\"\n\
170         #   NROS_RMW_NEEDS_CXX_LINKER       ON/OFF — force the C++ linker driver (libstdc++)\n\
171         function(nros_rmw_dispatch rmw)\n",
172    );
173    let mut first = true;
174    for name in KNOWN_RMW {
175        let r = resolve_rmw(name).expect("KNOWN_RMW resolves");
176        let d = &r.dispatch;
177        let branch = if first { "if" } else { "elseif" };
178        first = false;
179        let rlib = d.rlib_dep.unwrap_or("");
180        let extra = d.extra_link_libs.join(";");
181        let needs_cxx = if d.needs_cxx_linker { "ON" } else { "OFF" };
182        out.push_str(&format!(
183            "    {branch}(rmw STREQUAL \"{decl}\")\n\
184             \x20       set(NROS_RMW_UMBRELLA_CFFI_FEATURE \"{feat}\" PARENT_SCOPE)\n\
185             \x20       set(NROS_RMW_RLIB_DEP \"{rlib}\" PARENT_SCOPE)\n\
186             \x20       set(NROS_RMW_EXTRA_LINK_LIBS \"{extra}\" PARENT_SCOPE)\n\
187             \x20       set(NROS_RMW_NEEDS_CXX_LINKER {needs_cxx} PARENT_SCOPE)\n",
188            decl = r.declared,
189            feat = d.umbrella_cffi_feature,
190        ));
191    }
192    out.push_str(
193        "    else()\n\
194         \x20       message(FATAL_ERROR \"nros_rmw_dispatch: unknown rmw '${rmw}' \"\n\
195         \x20           \"(known: zenoh xrce cyclonedds)\")\n\
196         \x20   endif()\n\
197         endfunction()\n",
198    );
199    out
200}

202/// Repo-relative path of the committed CMake lowering of [`render_cmake_dispatch`].

205#[cfg(test)]
206mod tests {
207    use super::*;

209    #[test]
210    fn zenoh_lowers_to_each_language() {
211        let r = resolve_rmw("zenoh").expect("zenoh is known");
212        assert_eq!(r.cargo_feature, "rmw-zenoh");
213        assert_eq!(r.cmake_value, "zenoh");
214        assert_eq!(r.c_define_token, "ZENOH");
215    }
216
217    #[test]
218    fn xrce_and_cyclonedds_lower_uniformly() {
219        let x = resolve_rmw("xrce").unwrap();
220        assert_eq!(
221            (x.cargo_feature, x.cmake_value, x.c_define_token),
222            ("rmw-xrce", "xrce", "XRCE")
223        );
224        let c = resolve_rmw("cyclonedds").unwrap();
225        assert_eq!(
226            (c.cargo_feature, c.cmake_value, c.c_define_token),
227            ("rmw-cyclonedds", "cyclonedds", "CYCLONEDDS")
228        );
229    }
230
231    #[test]
232    fn every_known_rmw_resolves() {
233        for name in KNOWN_RMW {
234            assert!(resolve_rmw(name).is_ok(), "{name} should resolve");
235        }
236    }
237
238    #[test]
239    fn unknown_rmw_is_rejected_with_known_list() {
240        let err = resolve_rmw("dust-dds").unwrap_err();
241        assert_eq!(err.declared, "dust-dds");
242        let msg = err.to_string();
243        assert!(msg.contains("dust-dds"));
244        assert!(msg.contains("zenoh") && msg.contains("xrce") && msg.contains("cyclonedds"));
245    }
246
247    #[test]
248    fn canonical_rmw_accepts_aliases() {
249        assert_eq!(canonical_rmw("zenoh"), Some("zenoh"));
250        assert_eq!(canonical_rmw("rmw-zenoh"), Some("zenoh"));
251        assert_eq!(canonical_rmw("rmw-zenoh-cffi"), Some("zenoh"));
252        assert_eq!(canonical_rmw("rmw-cyclonedds"), Some("cyclonedds"));
253        assert_eq!(canonical_rmw("nope"), None);
254        // resolve_rmw accepts aliases via canonical_rmw.
255        assert_eq!(resolve_rmw("rmw-xrce").unwrap().cargo_feature, "rmw-xrce");
256    }
257
258    #[test]
259    fn cargo_feature_matches_board_rmw_naming() {
260        // Phase 248 C5b — the cargo lowering target is the board crate's
261        // `rmw-X` feature; guards against drift from the board crates' (and
262        // nros's) `rmw-<name>` feature naming.
263        for name in KNOWN_RMW {
264            let r = resolve_rmw(name).unwrap();
265            assert_eq!(r.cargo_feature, format!("rmw-{name}"));
266        }
267    }

269    fn cmake_dispatch_path() -> std::path::PathBuf {
270        // CARGO_MANIFEST_DIR = packages/cli/cargo-nano-ros → repo root is ../../..
271        std::path::Path::new(env!("CARGO_MANIFEST_DIR"))
272            .join("../../..")
273            .join(CMAKE_DISPATCH_REL_PATH)
274    }
275
276    #[test]
277    fn rmw_cmake_dispatch_is_current() {
278        let want = render_cmake_dispatch();
279        let path = cmake_dispatch_path();
280        let got = std::fs::read_to_string(&path).unwrap_or_else(|e| {
281            panic!(
282                "cannot read {} ({e}); regenerate with `cargo test -p cargo-nano-ros \
283                 regenerate_cmake_dispatch -- --ignored`",
284                path.display()
285            )
286        });
287        assert_eq!(
288            got, want,
289            "{} is stale — resolve_rmw() changed. Regenerate: `cargo test -p cargo-nano-ros \
290             regenerate_cmake_dispatch -- --ignored`",
291            CMAKE_DISPATCH_REL_PATH
292        );
293    }

295    /// Writes the committed CMake lowering from `resolve_rmw()`. Ignored by default
296    /// (it mutates a tracked file); run explicitly to regenerate after editing the
297    /// dispatch data: `cargo test -p cargo-nano-ros regenerate_cmake_dispatch -- --ignored`.
298    #[test]
299    #[ignore = "writes a tracked file; run explicitly to regenerate"]
300    fn regenerate_cmake_dispatch() {
301        let path = cmake_dispatch_path();
302        std::fs::write(&path, render_cmake_dispatch())
303            .unwrap_or_else(|e| panic!("write {}: {e}", path.display()));
304    }
