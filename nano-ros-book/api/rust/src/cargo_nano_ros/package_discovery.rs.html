13/// Discover Cargo packages in the workspace directory
14///
15/// Recursively walks the workspace to find all Cargo.toml files,
16/// extracting package names and paths. Skips build/ and install/ directories.
17///
18/// # Arguments
19/// * `workspace_root` - Root directory of the workspace
20/// * `build_base` - Build directory to skip (e.g., "build/")
21/// * `install_base` - Install directory to skip (e.g., "install/")
22///
23/// # Returns
24/// HashMap of package name -> absolute path to package directory
25pub fn discover_workspace_packages(
26    workspace_root: &Path,
27    build_base: Option<&Path>,
28    install_base: Option<&Path>,
29) -> Result<HashMap<String, PathBuf>> {
30    let mut packages = HashMap::new();
31
32    // Walk the workspace directory
33    fn walk_dir(
34        dir: &Path,
35        build_base: Option<&Path>,
36        install_base: Option<&Path>,
37        packages: &mut HashMap<String, PathBuf>,
38    ) -> Result<()> {
39        if !dir.is_dir() {
40            return Ok(());
41        }
42
43        // Check if we should skip this directory
44        // Skip build/ directories
45        if let Some(build_base) = build_base
46            && (dir == build_base || dir.starts_with(build_base))
47        {
48            return Ok(());
49        }
50
51        // Skip install/ directories (identified by setup.sh)
52        if let Some(install_base) = install_base
53            && (dir == install_base || dir.starts_with(install_base))
54        {
55            return Ok(());
56        }
57
58        // Check for setup.sh (indicates install directory)
59        if dir.join("setup.sh").exists() {
60            return Ok(());
61        }
62
63        // Check for COLCON_IGNORE (build directories have this)
64        if dir.join("COLCON_IGNORE").exists() {
65            return Ok(());
66        }
67
68        // Check if this directory has a Cargo.toml
69        let cargo_toml_path = dir.join("Cargo.toml");
70        if cargo_toml_path.exists()
71            && let Ok(content) = fs::read_to_string(&cargo_toml_path)
72            && let Some(name) = extract_package_name(&content)
73        {
74            packages.insert(name, dir.to_path_buf());
75        }
76
77        // Recursively walk subdirectories
78        if let Ok(entries) = fs::read_dir(dir) {
79            for entry in entries.flatten() {
80                if let Ok(file_type) = entry.file_type()
81                    && file_type.is_dir()
82                {
83                    walk_dir(&entry.path(), build_base, install_base, packages)?;
84                }
85            }
86        }
87
88        Ok(())
89    }
90
91    walk_dir(workspace_root, build_base, install_base, &mut packages)?;
92    Ok(packages)
93}

95/// Discover installed ament packages with Rust bindings
96///
97/// Scans AMENT_PREFIX_PATH to find packages with rust_packages resource index.
98/// Returns a mapping of package names to their Rust binding directories.
99///
100/// # Returns
101/// HashMap of package name -> path to <prefix>/share/<package>/rust/
102pub fn discover_installed_ament_packages() -> Result<HashMap<String, PathBuf>> {
103    let mut packages = HashMap::new();
104
105    // Get AMENT_PREFIX_PATH from environment
106    let ament_prefix_path = env::var("AMENT_PREFIX_PATH").unwrap_or_default();
107
108    if ament_prefix_path.is_empty() {
109        // Not an error - just means no ROS 2 is sourced
110        return Ok(packages);
111    }
112
113    // Split by path separator
114    let prefixes: Vec<&str> = ament_prefix_path
115        .split(if cfg!(windows) { ';' } else { ':' })
116        .collect();
117
118    for prefix in prefixes {
119        let prefix_path = Path::new(prefix);
120
121        // Check for rust_packages resource index
122        let rust_packages_dir = prefix_path
123            .join("share")
124            .join("ament_index")
125            .join("resource_index")
126            .join("rust_packages");
127
128        if rust_packages_dir.exists() {
129            // Read all package names from this directory
130            if let Ok(entries) = fs::read_dir(&rust_packages_dir) {
131                for entry in entries.flatten() {
132                    let package_name = entry.file_name().to_string_lossy().to_string();
133
134                    // The rust bindings are located at <prefix>/share/<package>/rust/
135                    let rust_binding_path =
136                        prefix_path.join("share").join(&package_name).join("rust");
137
138                    if rust_binding_path.exists() {
139                        packages.insert(package_name, rust_binding_path);
140                    }
141                }
142            }
143        }
144    }
145
146    Ok(packages)
147}

149/// Discover interface packages from workspace install directory
150///
151/// Scans the install directory for packages with .msg/.srv/.action files.
152/// Returns a mapping of package name -> share directory path.
153///
154/// # Colcon Dependency Ordering Guarantee
155///
156/// This function discovers from the `install/` directory, which works correctly
157/// because colcon **guarantees** topological dependency ordering:
158///
159/// 1. Dependencies are ALWAYS built and installed BEFORE dependents
160/// 2. When building package B that depends on package A, package A is already in install/A/
161/// 3. The install/ directory is created before any package builds
162///
163/// Example workflow:
164/// ```text
165/// colcon build
166///   └─> Builds robot_interfaces (no dependencies) → install/robot_interfaces/
167///   └─> Builds robot_controller (depends on robot_interfaces)
168///       └─> cargo ros2 ament-build
169///           └─> Discovers robot_interfaces from install/robot_interfaces/ ✓
170///           └─> Generates bindings ✓
171///           └─> Build succeeds ✓
172/// ```
173///
174/// This is colcon's core design: packages discover dependencies from `install/`, not `src/`.
175/// See: <https://colcon.readthedocs.io/en/released/developer/environment.html>
176///
177/// # Edge Cases
178///
179/// The only scenario where this returns empty is when:
180/// - User bypasses colcon and runs `cargo ros2 build` directly in a package subdirectory
181/// - This is expected behavior - users should use colcon for workspace builds
182///
183/// # Arguments
184/// * `install_base` - Install directory path (e.g., "install/")
185///
186/// # Returns
187/// HashMap of package name -> absolute path to package share directory
188pub fn discover_interface_packages_from_workspace(
189    install_base: &Path,
190) -> Result<HashMap<String, PathBuf>> {
191    let mut packages = HashMap::new();
192
193    // If install/ doesn't exist, we're either:
194    // 1. Building the first package in the workspace (no dependencies yet), or
195    // 2. User is building standalone (not via colcon)
196    // In both cases, returning empty is correct.
197    if !install_base.exists() {
198        return Ok(packages);
199    }
200
201    // Iterate through packages in install directory
202    if let Ok(entries) = fs::read_dir(install_base) {
203        for entry in entries.flatten() {
204            if !entry.file_type().map(|ft| ft.is_dir()).unwrap_or(false) {
205                continue;
206            }
207
208            let package_name = entry.file_name().to_string_lossy().to_string();
209            let share_dir = entry.path().join("share").join(&package_name);
210
211            if !share_dir.exists() {
212                continue;
213            }
214
215            // Check if package has interface files
216            let has_interfaces = share_dir.join("msg").exists()
217                || share_dir.join("srv").exists()
218                || share_dir.join("action").exists();
219
220            if has_interfaces {
221                packages.insert(package_name, share_dir);
222            }
223        }
224    }
225
226    Ok(packages)
227}

229/// Extract package name from Cargo.toml content
230///
231/// Simple line-by-line parser to find "name = ..." in [package] section
232fn extract_package_name(cargo_toml: &str) -> Option<String> {
233    for line in cargo_toml.lines() {
234        let trimmed = line.trim();
235        if trimmed.starts_with("name")
236            && let Some(eq_pos) = trimmed.find('=')
237        {
238            let value = &trimmed[eq_pos + 1..].trim();
239            let value = value.trim_matches('"').trim_matches('\'');
240            return Some(value.to_string());
241        }
242    }
243    None
244}

265    #[test]
266    fn test_discover_workspace_packages() {
267        let temp_dir = TempDir::new().unwrap();
268        let workspace = temp_dir.path();
269
270        // Create a test package
271        let pkg_dir = workspace.join("test_pkg");
272        fs::create_dir_all(&pkg_dir).unwrap();
273        fs::write(
274            pkg_dir.join("Cargo.toml"),
275            r#"[package]
276name = "test_pkg"
277version = "0.1.0"
278"#,
279        )
280        .unwrap();
281
282        // Create build directory (should be skipped)
283        let build_dir = workspace.join("build");
284        fs::create_dir_all(&build_dir).unwrap();
285        fs::write(build_dir.join("COLCON_IGNORE"), "").unwrap();
286
287        let pkg_in_build = build_dir.join("test_pkg");
288        fs::create_dir_all(&pkg_in_build).unwrap();
289        fs::write(
290            pkg_in_build.join("Cargo.toml"),
291            r#"[package]
292name = "should_be_skipped"
293version = "0.1.0"
294"#,
295        )
296        .unwrap();
297
298        // Discover packages
299        let packages = discover_workspace_packages(workspace, Some(&build_dir), None).unwrap();
300
301        assert_eq!(packages.len(), 1);
302        assert!(packages.contains_key("test_pkg"));
303        assert!(!packages.contains_key("should_be_skipped"));
304    }
