1//! Capability/feature-axis lowering (Phase 252, design-of-record RFC-0031 §
2//! "Generalization (Phase 250 / issue 0072)").
3//!
4//! A capability is a declared, language-agnostic axis the user toggles in config
5//! (`[safety]`, `[param_services]`, future). It generalizes the RMW/platform
6//! selection model: a declared axis **lowers** to up to three Rust build targets —
7//! the entry `nros/<feat>` umbrella feature, the concrete BACKEND crate's own
8//! feature (the capability's wire behaviour lives there, e.g. the CRC attach +
9//! validate in `nros-rmw-zenoh/safety-e2e`), and the BOARD crate's forwarding
10//! feature (the board is the selection point, RFC-0031 C5b). Cargo features do not
11//! propagate upward, so reaching the backend is NOT automatic from
12//! `nros/<feat>` — this table is the single source of truth for which targets a
13//! given axis lowers to, and which backends carry the backend feature.
14//!
15//! It lives in `cargo-nano-ros` (the lower crate) so both the scaffolder here and
16//! the orchestration `generate` in `nros-cli-core` share one mapping — mirroring
17//! `rmw_resolver`.
18//!
19//! C/C++ (`cmake_token` / `c_define`) is reserved for a future wave — `safety-e2e`
20//! is Rust-only today (the CRC machinery is feature-gated inside the zpico Rust
21//! shim; no `NROS_SAFETY` define exists).
22
23/// A declared capability axis lowered to its Rust build-feature targets.
24#[derive(Debug, Clone, Copy, PartialEq, Eq)]
25pub struct Capability {
26    /// The declared config axis name, e.g. `"safety"` (the `[safety]` block) or
27    /// `"param_services"` (the `[param_services]` block).
28    pub declared: &'static str,
29    /// The `nros` umbrella feature the axis enables on the generated entry
30    /// (target 1) — e.g. `"safety-e2e"` → `nros/safety-e2e`. Always present.
31    pub nros_feature: &'static str,
32    /// The feature name on the concrete BACKEND crate (target 2, the direct
33    /// `nros-rmw-<x>` dep on board-less native) AND the forwarding feature the
34    /// BOARD crate carries (target 3). `None` ⇒ the axis is entry-umbrella-only
35    /// (no backend wire behaviour), e.g. `param_services`.
36    pub backend_feature: Option<&'static str>,
37    /// The RMW backends that declare `backend_feature`. The backend feature is
38    /// emitted only for these; others (no such feature) no-op. Empty when
39    /// `backend_feature` is `None`.
40    pub backends_supporting: &'static [&'static str],
41}
42
43impl Capability {
44    /// `true` if this capability's backend feature applies to `backend` (a
45    /// canonical RMW name, e.g. `"zenoh"`).
46    pub fn backend_supports(&self, backend: &str) -> bool {
47        self.backend_feature.is_some() && self.backends_supporting.contains(&backend)
48    }
49}
50
51/// The capability axes nano-ros lowers today.
52pub const CAPABILITIES: &[Capability] = &[
53    // E2E message integrity. CRC attach (publisher) + validate (subscriber) live
54    // behind the zenoh backend's own `safety-e2e`; xrce / cyclonedds carry no such
55    // feature, so the axis no-ops there (issue 0072).
56    Capability {
57        declared: "safety",
58        nros_feature: "safety-e2e",
59        backend_feature: Some("safety-e2e"),
60        backends_supporting: &["zenoh"],
61    },
62    // The external ROS 2 parameter SERVER. Entry-umbrella-only — the runtime
63    // registers the services; there is no backend wire feature.
64    Capability {
65        declared: "param_services",
66        nros_feature: "param-services",
67        backend_feature: None,
68        backends_supporting: &[],
69    },
70];
71
72/// Look up a declared capability axis. `None` for an unknown axis.
73pub fn capability(declared: &str) -> Option<&'static Capability> {
74    CAPABILITIES.iter().find(|c| c.declared == declared)
75}

77#[cfg(test)]
78mod tests {
79    use super::*;
80
81    #[test]
82    fn safety_lowers_to_all_three_targets_zenoh_only() {
83        let c = capability("safety").expect("safety axis");
84        assert_eq!(c.nros_feature, "safety-e2e");
85        assert_eq!(c.backend_feature, Some("safety-e2e"));
86        assert!(c.backend_supports("zenoh"));
87        // xrce / cyclonedds have no CRC path → no backend feature.
88        assert!(!c.backend_supports("xrce"));
89        assert!(!c.backend_supports("cyclonedds"));
90    }
91
92    #[test]
93    fn param_services_is_entry_umbrella_only() {
94        let c = capability("param_services").expect("param_services axis");
95        assert_eq!(c.nros_feature, "param-services");
96        assert_eq!(c.backend_feature, None);
97        assert!(!c.backend_supports("zenoh"));
98    }
99
100    #[test]
101    fn unknown_axis_is_none() {
102        assert!(capability("nope").is_none());
103    }
104}
