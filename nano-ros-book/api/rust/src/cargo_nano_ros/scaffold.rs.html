1//! Project scaffolder — promoted out of `main.rs` so `nros-cli` and any
2//! other front-end can share it.
3//!
4//! v1 emits a colcon-compatible hello-world per `<lang> × <platform>`.
5//! Use-case (`talker` / `listener` / `service` / `action`) and RMW-choice
6//! diversification arrives once the `templates/` tree lands; until then
7//! both fields are accepted for forward-compat but only surfaced in the
8//! "Next steps" output.
9

15};

18pub struct ScaffoldConfig {
19    pub name: String,
20    pub lang: String,
21    pub platform: String,
22    pub rmw: String,
23    pub use_case: String,
24    pub force: bool,
25}

27pub fn scaffold_package(cfg: &ScaffoldConfig) -> Result<()> {
28    let dir = PathBuf::from(&cfg.name);
29    if dir.exists() {
30        if !cfg.force {
31            bail!(
32                "Directory '{}' already exists (use --force to overwrite)",
33                cfg.name
34            );
35        }
36        fs::remove_dir_all(&dir)?;
37    }
38
39    // Phase 227.4 — validate + lower the declared RMW (RFC-0031). A bad
40    // `--rmw` fails here with the known-list, not as a broken downstream build.
41    let rmw = rmw_resolver::resolve_rmw(&cfg.rmw).map_err(|e| eyre::eyre!("nros new: {e}"))?;
42
43    let build_type = format!("nros.{}.{}", cfg.lang, cfg.platform);
44
45    fs::create_dir_all(dir.join("src"))?;
46
47    let package_xml = format!(
48        r#"<?xml version="1.0"?>
49<package format="3">
50  <name>{name}</name>
51  <version>0.1.0</version>
52  <description>{name} — nano-ros {platform} package</description>
53  <maintainer email="TODO@todo.com">TODO</maintainer>
54  <license>Apache-2.0</license>
55  <depend>std_msgs</depend>
56  <export>
57    <build_type>{build_type}</build_type>
58  </export>
59</package>
60"#,
61        name = cfg.name,
62        platform = cfg.platform,
63    );
64    fs::write(dir.join("package.xml"), package_xml)?;
65
66    match cfg.lang.as_str() {
67        "rust" => scaffold_rust(&cfg.name, &cfg.platform, rmw.cargo_feature, &dir)?,
68        "c" => scaffold_c(&cfg.name, &cfg.platform, rmw.cmake_value, &dir)?,
69        "cpp" => scaffold_cpp(&cfg.name, &cfg.platform, rmw.cmake_value, &dir)?,
70        other => bail!("Unknown language: {other}. Use rust, c, or cpp."),
71    }
72
73    println!("✓ Created nano-ros package '{}'", cfg.name);
74    println!("  Language : {}", cfg.lang);
75    println!("  Platform : {}", cfg.platform);
76    println!("  RMW      : {}", cfg.rmw);
77    println!(
78        "  Use case : {} (template diversification: TODO)",
79        cfg.use_case
80    );
81    println!("  Build    : {build_type}");
82    println!();
83    println!("Next steps:");
84    println!("  cd {}", cfg.name);
85    println!("  export NROS_REPO_DIR=/path/to/nano-ros   # your nano-ros source checkout");
86    println!("  eval \"$(nros ws env)\"   # ROS + interface search path");
87    println!("  nros ws sync          # codegen + write the [patch.crates-io] block (RFC-0040)");
88    println!("  cargo build           # or: cmake --build build / west build / idf.py build");
89
90    Ok(())
91}

94pub struct ComponentScaffoldConfig {
95    pub name: String,
96    /// Node flavor: `talker` / `listener` / `service` / `action`. Template
97    /// diversification is TODO — today every flavor emits the publisher+timer
98    /// shape, named after the flavor.
99    pub use_case: String,
100    /// Source language. `rust` lands the planned-mode `nros::Component` shape;
101    /// `c` / `cpp` land the typed component shape (RFC-0043):
102    /// `nano_ros_workspace_pkg_guard()` + `nano_ros_node_register()` + a
103    /// `configure(::nros::Node&)` (C++) / `NROS_C_COMPONENT` (C) seam.
104    pub lang: String,
105    pub force: bool,
106}
107
108/// Scaffold a **planned-mode component** — a reusable nano-ros node compiled as
109/// a *library* and linked into a system plan. Unlike
110/// the direct-mode hello-world binary `scaffold_package` emits (a `[node]`
111/// manifest), this produces an `nros::Component` impl plus a *folded*
112/// `[component]` table in `nros.toml`. The platform + RMW are chosen later, at
113/// Entry-package build time — not baked here.
114///
115/// The manifest is intentionally minimal: `[linkage]` is omitted (derived —
116/// executable ← component short name, `exported_symbol` ← `nros_component_<n>`,
117/// `crate_name` ← package) and `[overrides]` defaults to empty (Phase 172 W.3).
118pub fn scaffold_component(cfg: &ComponentScaffoldConfig) -> Result<()> {
119    match cfg.lang.as_str() {
120        "rust" => scaffold_component_rust(cfg),
121        "cpp" => scaffold_component_cpp(cfg),
122        "c" => scaffold_component_c(cfg),
123        other => bail!(
124            "`nros new --component --lang {other}` is not supported. Use \
125             `rust`, `c`, or `cpp`."
126        ),
127    }
128}

130fn scaffold_component_rust(cfg: &ComponentScaffoldConfig) -> Result<()> {
131    let dir = PathBuf::from(&cfg.name);
132    if dir.exists() {
133        if !cfg.force {
134            bail!(
135                "Directory '{}' already exists (use --force to overwrite)",
136                cfg.name
137            );
138        }
139        fs::remove_dir_all(&dir)?;
140    }
141    fs::create_dir_all(dir.join("src"))?;
142
143    let crate_name = cfg.name.replace('-', "_");
144    let module = &cfg.use_case; // constrained by the CLI to a valid Rust ident
145
146    let package_xml = format!(
147        r#"<?xml version="1.0"?>
148<package format="3">
149  <name>{name}</name>
150  <version>0.1.0</version>
151  <description>{name} — nano-ros reusable component.</description>
152  <maintainer email="TODO@todo.com">TODO</maintainer>
153  <license>Apache-2.0</license>
154</package>
155"#,
156        name = cfg.name,
157    );
158    fs::write(dir.join("package.xml"), package_xml)?;
159
160    let cargo_toml = format!(
161        r#"[package]
162name = "{name}"
163version = "0.1.0"
164edition = "2024"
165
166# Standalone-buildable: an empty [workspace] makes this its own Cargo root even
167# when dropped under a colcon workspace's src/.
168[workspace]
169
170# A reusable component is a library (rlib); the deployed system links it.
171# nano-ros crates are not published to crates.io (RFC-0040) — `version = "*"` is
172# only the patched left-hand side. Run `nros ws sync` (with NROS_REPO_DIR set) to
173# write the nros-managed [patch.crates-io] block redirecting `nros` to your
174# nano-ros checkout, then `cargo build`.
175[dependencies]
176nros = {{ version = "*", default-features = false }}
177"#,
178        name = cfg.name,
179    );
180    fs::write(dir.join("Cargo.toml"), cargo_toml)?;
181
182    let lib_rs = format!(
183        r#"#![no_std]
184
185//! `{name}` — a reusable nano-ros component (planned mode).
186//!
187//! `nros plan` and Entry codegen link this crate into a system and call
188//! `{module}::Component::register`. `nros metadata --build` records its
189//! declarations into `metadata/{module}.json`. Platform + RMW are chosen at
190//! Entry-package build time, not here.
191
192pub mod {module} {{
193    use nros::{{
194        CallbackId, CdrReader, CdrWriter, ComponentContext, ComponentResult, DeserError,
195        Deserialize, EntityId, NodeId, NodeOptions, RosMessage, SerError, Serialize, TimerDuration,
196    }};
197
198    pub struct Component;
199
200    impl nros::Component for Component {{
201        const NAME: &'static str = "{module}";
202
203        fn register(context: &mut ComponentContext<'_>) -> ComponentResult<()> {{
204            let mut node =
205                context.create_node(NodeId::new("node_{module}"), NodeOptions::new("{module}"))?;
206            let _publisher =
207                node.create_publisher::<StringMsg>(EntityId::new("pub_chatter"), "chatter")?;
208            let _timer = node.create_timer(
209                EntityId::new("timer_publish"),
210                CallbackId::new("cb_timer"),
211                TimerDuration::from_millis(100),
212            )?;
213            Ok(())
214        }}
215    }}
216
217    /// Minimal hand-rolled `std_msgs/String` stand-in. Replace with a generated
218    /// message type (`nros generate-rust`) for real payloads.
219    pub struct StringMsg;
220    impl Serialize for StringMsg {{
221        fn serialize(&self, _writer: &mut CdrWriter) -> Result<(), SerError> {{
222            Ok(())
223        }}
224    }}
225    impl Deserialize for StringMsg {{
226        fn deserialize(_reader: &mut CdrReader) -> Result<Self, DeserError> {{
227            Ok(Self)
228        }}
229    }}
230    impl RosMessage for StringMsg {{
231        const TYPE_NAME: &'static str = "std_msgs::msg::dds_::String_";
232        const TYPE_HASH: &'static str = "std_msgs/String";
233    }}
234}}
235
236// The planner links the component via the Rust type path above. To also expose
237// the C / dynamic registration symbol (`__nros_component_register`), add:
238//     nros::component!({module}::Component);
239"#,
240        name = cfg.name,
241    );
242    fs::write(dir.join("src/lib.rs"), lib_rs)?;
243
244    // Folded `[component]` manifest (Phase 172 W.1). Minimal — no `[linkage]`
245    // (derived) and no `[overrides]` (defaults to empty). The `crate::module`
246    // component id is required by `nros metadata --build`.
247    let nros_toml = format!(
248        r#"# nano-ros component manifest (planned mode). A reusable node linked into a
249# system by `nros plan` and Entry codegen. See
250# docs/design/0004-configuration-and-transports.md.
251
252[component]
253version = 1
254package = "{name}"
255component = "{crate_name}::{module}"
256language = "rust"
257
258[component.metadata]
259source_metadata = "metadata/{module}.json"
260"#,
261        name = cfg.name,
262    );
263    fs::write(dir.join("nros.toml"), nros_toml)?;
264
265    println!("✓ Created nano-ros component '{}'", cfg.name);
266    println!("  Component : {crate_name}::{module}");
267    println!("  Kind      : planned-mode (library, linked by an Entry pkg)");
268    println!();
269    println!("Next steps:");
270    println!("  cd {}", cfg.name);
271    println!("  # add this package to a workspace's [system].components, then:");
272    println!("  nros metadata --build   # record its source metadata");
273
274    Ok(())
275}

277/// Scaffold a **C++ Node pkg** — typed component (RFC-0043). Emits the
278/// §212.L.9 cmake-fn surface (`nano_ros_node_register`) and a
279/// `<UserClass>::configure(::nros::Node&)` real-callback body in the
280/// `<pkg>::` namespace per §212.L.4 (class prefix must equal `PROJECT_NAME`).
281/// `configure` creates a `Publisher` + binds a member timer callback by
282/// identity (no string descriptor, no interpreter) — the typed Entry
283/// constructs the object + calls `configure(node)` on the real executor.
284fn scaffold_component_cpp(cfg: &ComponentScaffoldConfig) -> Result<()> {
285    let dir = PathBuf::from(&cfg.name);
286    if dir.exists() {
287        if !cfg.force {
288            bail!(
289                "Directory '{}' already exists (use --force to overwrite)",
290                cfg.name
291            );
292        }
293        fs::remove_dir_all(&dir)?;
294    }
295    // Header lives at `include/<pkg>/<Class>.hpp` so the typed Entry can
296    // `#include "<pkg>/<Class>.hpp"` (nano_ros_node_register adds `include/`).
297    fs::create_dir_all(dir.join("src"))?;
298    fs::create_dir_all(dir.join("include").join(&cfg.name.replace('-', "_")))?;
299
300    // Pkg name → namespace + class. `my-talker` → ns `my_talker`, class
301    // `Talker` (PascalCase of use_case). §212.L.4 class prefix must equal
302    // PROJECT_NAME (sanitised), so the namespace = the sanitised pkg name.
303    let pkg_sym = cfg.name.replace('-', "_");
304    let class_name = use_case_to_pascal(&cfg.use_case);
305    let node_name = &cfg.use_case;
306
307    let package_xml = format!(
308        r#"<?xml version="1.0"?>
309<package format="3">
310  <name>{name}</name>
311  <version>0.1.0</version>
312  <description>{name} — nano-ros C++ Node pkg.</description>
313  <maintainer email="TODO@todo.com">TODO</maintainer>
314  <license>Apache-2.0</license>
315  <depend>std_msgs</depend>
316  <export>
317    <build_type>cmake</build_type>
318  </export>
319</package>
320"#,
321        name = cfg.name,
322    );
323    fs::write(dir.join("package.xml"), package_xml)?;
324
325    let cmake = format!(
326        r#"cmake_minimum_required(VERSION 3.22)
327# §212.L.4 — class prefix must equal PROJECT_NAME.
328project({pkg_sym} VERSION 0.1.0 LANGUAGES C CXX)
329
330set(CMAKE_CXX_STANDARD 14)
331set(CMAKE_CXX_STANDARD_REQUIRED ON)
332
333{bootstrap}
334nros_find_interfaces(LANGUAGE CPP SKIP_INSTALL)
335
336# Typed component (RFC-0043). No add_executable, no `main()`; the linked
337# Entry pkg's typed runtime constructs this class + calls `configure(node)`
338# on the real executor. `DEPLOY native` also builds a standalone runnable
339# single-node ELF via the typed native carrier.
340nano_ros_node_register(
341    NAME    {node_name}
342    CLASS   {pkg_sym}::{class_name}
343    SOURCES src/{class_name}.cpp
344    DEPLOY  native)
345
346# `nros_find_interfaces` declares an INTERFACE lib per dep
347# (`std_msgs__nano_ros_cpp` etc.) carrying the generated headers' include
348# dirs + FFI-glue link. The typed `Publisher<Int32>` needs them.
349target_link_libraries({pkg_sym}_{node_name}_component
350    PUBLIC std_msgs__nano_ros_cpp)
351"#,
352        bootstrap = NROS_BOOTSTRAP_BLOCK,
353    );
354    fs::write(dir.join("CMakeLists.txt"), cmake)?;
355
356    let class_hpp = format!(
357        r#"#pragma once
358
359#include <nros/component.hpp>
360#include <nros/nros.hpp>
361
362#include "std_msgs.hpp"
363
364namespace {pkg_sym} {{
365
366/// {class_name} — a typed component (RFC-0043). `configure` creates a
367/// publisher on `/chatter` and binds the member `on_tick` (by identity, no
368/// name) as a 1 Hz timer callback that publishes a real counter. The typed
369/// Entry constructs this object + calls `configure(node)`; the executor
370/// dispatches `on_tick` during `spin_once`.
371class {class_name} {{
372    ::nros::Publisher<std_msgs::msg::Int32> pub_;
373    ::nros::Timer timer_;
374    int count_ = 0;
375
376    void on_tick(); // real body; bound via &{class_name}::on_tick (no name)
377
378  public:
379    ::nros::Result configure(::nros::Node& node);
380}};
381
382}} // namespace {pkg_sym}
383"#,
384    );
385    fs::write(
386        dir.join("include")
387            .join(&pkg_sym)
388            .join(format!("{class_name}.hpp")),
389        class_hpp,
390    )?;
391
392    let class_cpp = format!(
393        r#"// Generated by `nros new {name} --component --lang cpp`.
394//
395// Typed component (RFC-0043). `configure` creates a `Publisher<Int32>` on
396// `/chatter` + binds the member `on_tick` by identity as a 1 Hz timer; the
397// callback publishes a real counter. No string descriptor, no interpreter.
398
399#include "{pkg_sym}/{class_name}.hpp"
400
401#include <cstdio>
402
403namespace {pkg_sym} {{
404
405void {class_name}::on_tick() {{
406    std_msgs::msg::Int32 m;
407    m.data = count_++;
408    if (pub_.publish(m).ok()) {{
409        std::printf("Published: %d\n", m.data);
410    }}
411}}
412
413::nros::Result {class_name}::configure(::nros::Node& node) {{
414    std::setvbuf(stdout, nullptr, _IONBF, 0);
415    ::nros::Result r = node.create_publisher(pub_, "/chatter");
416    if (!r.ok()) return r;
417    // Member-fn-pointer-as-template-param → no-alloc trampoline; `this` is ctx.
418    return ::nros::bind_timer<{class_name}, &{class_name}::on_tick>(node, timer_, 1000, this);
419}}
420
421}} // namespace {pkg_sym}
422"#,
423        name = cfg.name,
424    );
425    fs::write(dir.join(format!("src/{class_name}.cpp")), class_cpp)?;
426
427    println!("✓ Created nano-ros C++ Node pkg '{}'", cfg.name);
428    println!("  Class     : {pkg_sym}::{class_name}");
429    println!("  Node      : {node_name}");
430    println!("  Kind      : typed component (RFC-0043)");
431    println!();
432    println!("Next steps:");
433    println!("  cd {}", cfg.name);
434    println!("  # Solo build:");
435    println!("  cmake -S . -B build -DNANO_ROS_ROOT=<path-to-nano-ros>");
436    println!("  cmake --build build");
437    println!();
438    println!("  # Or add it as a SUBDIR in a workspace root that calls");
439    println!("  # nano_ros_workspace(...), then build the workspace.");
440
441    Ok(())
442}

444/// Scaffold a **C Node pkg** — typed component (RFC-0043). Emits a
445/// `NROS_C_COMPONENT(<state>, <configure>)` factory/configure seam: a raw
446/// `/chatter` publisher + 1 Hz timer that publishes a CDR-encoded Int32. The
447/// typed Entry's `__nros_c_component_<pkg>_{create,configure}` calls run it on
448/// the real executor — no declarative descriptor, no interpreter.
449fn scaffold_component_c(cfg: &ComponentScaffoldConfig) -> Result<()> {
450    let dir = PathBuf::from(&cfg.name);
451    if dir.exists() {
452        if !cfg.force {
453            bail!(
454                "Directory '{}' already exists (use --force to overwrite)",
455                cfg.name
456            );
457        }
458        fs::remove_dir_all(&dir)?;
459    }
460    fs::create_dir_all(dir.join("src"))?;
461
462    let pkg_sym = cfg.name.replace('-', "_");
463    let class_name = use_case_to_pascal(&cfg.use_case);
464    let node_name = &cfg.use_case;
465    let state_ty = format!("{pkg_sym}_t");
466    let configure_fn = format!("{node_name}_configure");
467
468    let package_xml = format!(
469        r#"<?xml version="1.0"?>
470<package format="3">
471  <name>{name}</name>
472  <version>0.1.0</version>
473  <description>{name} — nano-ros C Node pkg.</description>
474  <maintainer email="TODO@todo.com">TODO</maintainer>
475  <license>Apache-2.0</license>
476  <export>
477    <build_type>cmake</build_type>
478  </export>
479</package>
480"#,
481        name = cfg.name,
482    );
483    fs::write(dir.join("package.xml"), package_xml)?;
484
485    let cmake = format!(
486        r#"cmake_minimum_required(VERSION 3.22)
487# §212.L.4 — class prefix must equal PROJECT_NAME.
488project({pkg_sym} VERSION 0.1.0 LANGUAGES C CXX)
489
490set(CMAKE_C_STANDARD 11)
491set(CMAKE_C_STANDARD_REQUIRED ON)
492
493{bootstrap}
494# Typed C component (RFC-0043). The raw `/chatter` publisher carries the type
495# name as a string, so no generated C bindings are needed. `DEPLOY native`
496# also builds a standalone runnable single-node ELF via the typed C carrier.
497nano_ros_node_register(
498    NAME     {node_name}
499    CLASS    {pkg_sym}::{class_name}
500    LANGUAGE C
501    TYPED
502    SOURCES  src/{class_name}.c
503    DEPLOY   native)
504"#,
505        bootstrap = NROS_BOOTSTRAP_BLOCK,
506    );
507    fs::write(dir.join("CMakeLists.txt"), cmake)?;
508
509    let source = format!(
510        r#"// Generated by `nros new {name} --component --lang c`.
511//
512// Typed C component (RFC-0043). `{configure_fn}` creates a raw `/chatter`
513// publisher + a 1 Hz timer publishing a CDR-encoded Int32 counter.
514// `NROS_C_COMPONENT` emits the C-ABI factory/configure the typed Entry calls;
515// it runs on the real executor — no declarative descriptor, no interpreter.
516
517#include <stddef.h>
518#include <stdint.h>
519#include <stdio.h>
520
521#include <nros/component.h>
522
523typedef struct {{
524    _Alignas(8) uint8_t pub[NROS_C_PUBLISHER_STORAGE_SIZE];
525    int32_t count;
526}} {state_ty};
527
528static void write_u32_le(uint8_t* p, uint32_t v) {{
529    p[0] = (uint8_t)v;
530    p[1] = (uint8_t)(v >> 8);
531    p[2] = (uint8_t)(v >> 16);
532    p[3] = (uint8_t)(v >> 24);
533}}
534
535static void on_tick(void* ctx) {{
536    {state_ty}* self = ({state_ty}*)ctx;
537    /* std_msgs/Int32 CDR: 4-byte encapsulation header (CDR_LE) + int32 data. */
538    uint8_t buf[8];
539    buf[0] = 0x00;
540    buf[1] = 0x01;
541    buf[2] = 0x00;
542    buf[3] = 0x00;
543    write_u32_le(buf + 4, (uint32_t)self->count);
544    if (nros_cpp_publish_raw(self->pub, buf, sizeof(buf)) == 0) {{
545        printf("Published: %d\n", (int)self->count);
546    }}
547    self->count++;
548}}
549
550static nros_ret_t {configure_fn}(const nros_cpp_node_t* node, void* executor, {state_ty}* self) {{
551    self->count = 0;
552    int32_t rc = nros_cpp_publisher_create(node, "/chatter", "std_msgs::msg::dds_::Int32_", "",
553                                           nros_c_qos_default(), self->pub);
554    if (rc != 0) {{
555        return rc;
556    }}
557    size_t timer_handle;
558    return nros_cpp_timer_create(executor, /*period_ms=*/1000, on_tick, self, &timer_handle);
559}}
560
561NROS_C_COMPONENT({state_ty}, {configure_fn})
562"#,
563        name = cfg.name,
564    );
565    fs::write(dir.join(format!("src/{class_name}.c")), source)?;
566
567    println!("✓ Created nano-ros C Node pkg '{}'", cfg.name);
568    println!("  Class     : {pkg_sym}::{class_name}");
569    println!("  Node      : {node_name}");
570    println!("  Kind      : typed component (RFC-0043)");
571    println!();
572    println!("Next steps:");
573    println!("  cd {}", cfg.name);
574    println!("  # Solo build:");
575    println!("  cmake -S . -B build -DNANO_ROS_ROOT=<path-to-nano-ros>");
576    println!("  cmake --build build");
577    println!();
578    println!("  # Or add it as a SUBDIR in a C++ or Rust Entry workspace.");
579
580    Ok(())
581}

583/// Map `talker` → `Talker`, `service-server` → `ServiceServer`.
584fn use_case_to_pascal(s: &str) -> String {
585    s.split(|c: char| c == '_' || c == '-')
586        .filter(|p| !p.is_empty())
587        .map(|p| {
588            let mut chars = p.chars();
589            match chars.next() {
590                Some(c) => c.to_ascii_uppercase().to_string() + chars.as_str(),
591                None => String::new(),
592            }
593        })
594        .collect()
595}

597fn scaffold_rust(name: &str, platform: &str, rmw_feature: &str, dir: &Path) -> Result<()> {
598    let mut deps = String::new();
599    let is_embedded = platform != "native";
600
601    if is_embedded {
602        // Phase 248 C5b + C5c-platform (RFC-0031) — BOTH axes lower to the BOARD
603        // crate (the board dep below carries `rmw-X` and brings the concrete
604        // `nros-platform/platform-X` impl), so the umbrella `nros` dep stays
605        // agnostic — no `platform-*`/`rmw-*`, vtable-only.
606        deps.push_str(
607            "nros = { version = \"*\", default-features = false, features = [\"ros-humble\"] }\n",
608        );
609        let board_crate = match platform {
610            "freertos" => "nros-board-mps2-an385-freertos",
611            "baremetal" => "nros-board-mps2-an385",
612            "nuttx" => "nros-board-nuttx-qemu-arm",
613            _ => "# TODO: add board crate for this platform",
614        };
615        deps.push_str(&format!(
616            "{board_crate} = {{ version = \"*\", features = [\"{rmw_feature}\"] }}\n"
617        ));
618        deps.push_str("panic-semihosting = \"0.6\"\n");
619    } else {
620        // Phase 248 C5b + C5c-platform (RFC-0031) — native: both axes lower to
621        // `nros-board-native` (its `rmw-X` feature self-links the backend, and it
622        // brings `nros-platform/platform-posix`); `nros` stays agnostic.
623        deps.push_str(&format!(
624            "# nros = {{ version = \"*\", default-features = false, features = [\"std\", \"ros-humble\"] }}\n\
625             # nros-board-native = {{ version = \"*\", features = [\"{rmw_feature}\"] }}\n"
626        ));
627    }
628
629    let cargo_toml = format!(
630        r#"[package]
631name = "{name}"
632version = "0.1.0"
633edition = "2024"
634
635[workspace]
636
637[[bin]]
638name = "{name}"
639path = "src/main.rs"
640
641[dependencies]
642{deps}
643# nano-ros crates are not published to crates.io (RFC-0040) — the `version = "*"`
644# requirements above are patched, not resolved from crates.io. Run `nros ws sync`
645# (with NROS_REPO_DIR set) to write the nros-managed [patch.crates-io] block here
646# (path deps into your nano-ros checkout + any generated msg crates), then build.
647
648# Phase 204.15 inc 3 — named size/speed profiles so the plain-cargo path honours
649# the same intent as `[build].optimize` (`cargo build --profile
650# size|speed`), no hand-editing. (panic is left to the target/profile — embedded
651# triples are already abort; host keeps its default.)
652#
653# Phase 204.3 — `opt-level = "s"`, NOT `"z"`: on smoltcp/IP examples `-Oz`'s
654# weaker DCE keeps a non-inlined socket-buffer accessor that defeats opt-3's
655# per-socket dead-buffer elimination (grew `.bss` +24 KB on a measured talker);
656# `"s"` shrinks `.text` *more* and preserves the DCE.
657[profile.size]
658inherits = "release"
659opt-level = "s"
660lto = "fat"
661codegen-units = 1
662strip = true
663
664[profile.speed]
665inherits = "release"
666opt-level = 3
667lto = "fat"
668codegen-units = 1
669"#
670    );
671    fs::write(dir.join("Cargo.toml"), cargo_toml)?;
672
673    let main_rs = if is_embedded {
674        r#"#![no_std]
675#![no_main]
676
677use nros::prelude::*;
678// TODO: import your board crate
679// use nros_board_mps2_an385_freertos::{Config, run, println};
680use panic_semihosting as _;
681
682#[unsafe(no_mangle)]
683extern "C" fn _start() -> ! {
684    // TODO: replace with your board crate's run()
685    loop {}
686}
687"#
688        .to_string()
689    } else {
690        format!(
691            r#"fn main() {{
692    println!("Hello from {name}!");
693}}
694"#
695        )
696    };
697    fs::write(dir.join("src/main.rs"), main_rs)?;
698
699    if is_embedded {
700        write_default_config_toml(dir)?;
701        write_cargo_config(dir, platform)?;
702    }
703
704    Ok(())
705}

707/// Scaffold `.cargo/config.toml` for the cortex-m cargo-built platforms
708/// (bare-metal / FreeRTOS on the QEMU mps2-an385). Carries the Phase 204
709/// size knobs by default: `--gc-sections` at link (204.8) plus a documented,
710/// commented block for the serial-only IP-stack opt-out (204.7) and the
711/// per-backend static-heap size (204.5). Other embedded platforms (Zephyr,
712/// NuttX, ESP-IDF, ThreadX) build through their own toolchains and don't use
713/// a cargo target triple here, so they get no `.cargo/config.toml`.
714fn write_cargo_config(dir: &Path, platform: &str) -> Result<()> {
715    let triple = match platform {
716        "baremetal" | "freertos" => "thumbv7m-none-eabi",
717        // Non-cargo-target build flows — leave the build config to the
718        // platform's own toolchain integration.
719        _ => return Ok(()),
720    };
721
722    let config = format!(
723        r#"[build]
724target = "{triple}"
725
726[target.{triple}]
727# QEMU mps2-an385 (cortex-m3) + semihosting. Adjust `-machine`/`-cpu` for your board.
728runner = "qemu-system-arm -cpu cortex-m3 -machine mps2-an385 -nographic -semihosting-config enable=on,target=native -kernel"
729rustflags = [
730    # Phase 204.8 — drop unreferenced fns/data at link. `rust-lld` is invoked
731    # directly (no gcc driver), so the bare `--gc-sections`, NOT `-Wl,...`.
732    # `cortex-m-rt`'s `link.x` KEEPs the vector table, so gc is safe.
733    "-C", "link-arg=--gc-sections",
734    "-C", "link-arg=-Tlink.x",
735]
736
737[env]
738# Phase 204.7 — serial-only node: uncomment to drop the IP link layer
739# (zenoh-pico TCP/UDP link C via `Z_FEATURE_LINK_TCP/UDP=0`; `--gc-sections`
740# above then strips the smoltcp residue). Also switch the board to its
741# `serial` feature and use a serial `locator` in `nros.toml`.
742# NROS_LINK_IP = "0"
743# ZPICO_NO_SMOLTCP = "1"
744#
745# Phase 204.5 — size the static heap to the backend's working set
746# (zenoh-pico ~24 KB, XRCE ~8 KB); default is the per-board value (64 KB on
747# mps2-an385). Decimal bytes.
748# NROS_HEAP_SIZE = "24576"
749#
750# Phase 204.2 — a brokered (zenoh/XRCE) client multiplexes over one session;
751# drop the spare smoltcp socket buffers (sized for DDS RTPS by default).
752# NROS_SMOLTCP_MAX_SOCKETS = "1"
753# NROS_SMOLTCP_MAX_UDP_SOCKETS = "1"
754"#
755    );
756
757    let cargo_dir = dir.join(".cargo");
758    fs::create_dir_all(&cargo_dir)?;
759    fs::write(cargo_dir.join("config.toml"), config)?;
760    Ok(())
761}

763/// Standard preamble that bootstraps the nano-ros workspace cmake fns
764/// (`nano_ros_workspace_pkg_guard`, `nano_ros_node_register`,
765/// `nano_ros_entry`, `nros_find_interfaces`, …) regardless of whether the
766/// pkg is built solo or as a workspace member. Lands in every scaffolded
767/// C / C++ CMakeLists at the top, right after `project()`.
768///
769/// Phase 219.I shape — see `cmake/NanoRosWorkspace.cmake`.
770const NROS_BOOTSTRAP_BLOCK: &str = r#"# Phase 219.I — bootstrap nano-ros workspace helpers. Workspace builds
771# inherit the helpers from the parent root; standalone solo builds
772# require `-DNANO_ROS_ROOT=<path-to-nano-ros>` and locate them via the
773# include() below.
774if(NOT COMMAND nano_ros_workspace_pkg_guard)
775    if(NOT NANO_ROS_ROOT)
776        message(FATAL_ERROR
777            "nano-ros: set -DNANO_ROS_ROOT=<path-to-nano-ros> for "
778            "standalone builds, or build via the workspace root.")
779    endif()
780    include("${NANO_ROS_ROOT}/cmake/NanoRosWorkspace.cmake")
781endif()
782nano_ros_workspace_pkg_guard()
783"#;

785fn scaffold_c(name: &str, platform: &str, rmw_value: &str, dir: &Path) -> Result<()> {
786    let _ = platform;
787    let cmake = format!(
788        r#"cmake_minimum_required(VERSION 3.22)
789project({name} VERSION 0.1.0 LANGUAGES C CXX)
790
791set(CMAKE_C_STANDARD 11)
792set(CMAKE_C_STANDARD_REQUIRED ON)
793
794# Phase 227.4 — declared RMW baked in (override at configure with -DNANO_ROS_RMW=…).
795set(NANO_ROS_RMW {rmw_value} CACHE STRING "nano-ros RMW backend")
796
797{bootstrap}
798# Phase 210.E.4 — `nros_find_interfaces` reads `package.xml` and
799# generates msg/srv/action bindings (+ FFI glue) for every transitive
800# interface dep declared via `<depend>` tags.
801nros_find_interfaces(LANGUAGE C SKIP_INSTALL)
802
803# Phase 212.N.6 — Entry pkg shape (single-Node self-bringup; Phase 219
804# adds the `LAUNCH "<bringup>:<file>.launch.xml"` form for multi-Node).
805nano_ros_entry(
806    NAME {name}
807    SOURCES src/main.c
808    DEPLOY native)
809
810target_link_libraries({name} PRIVATE std_msgs__nano_ros_c)
811nros_platform_link_app({name})
812"#,
813        bootstrap = NROS_BOOTSTRAP_BLOCK,
814    );
815    fs::write(dir.join("CMakeLists.txt"), cmake)?;
816
817    let main_c = format!(
818        r#"// Generated by `nros new {name} --lang c`.
819//
820// Minimal nano-ros C talker — publishes one `std_msgs/Int32` message on
821// `/chatter`, then returns. Swap the body for your own logic; see
822// `examples/native/c/talker/src/main.c` for a fuller shape (timer,
823// executor, signal handler).
824
825#include <stdio.h>
826
827#include <nros/init.h>
828#include <nros/node.h>
829#include <nros/publisher.h>
830#include "std_msgs.h"
831
832int main(int argc, char** argv) {{
833    (void)argc;
834    (void)argv;
835
836    nros_support_t support = nros_support_get_zero_initialized();
837    if (nros_support_init(&support, NULL, 0) != NROS_RET_OK) {{
838        fprintf(stderr, "nros_support_init failed\n");
839        return 1;
840    }}
841
842    nros_node_t node = nros_node_get_zero_initialized();
843    if (nros_node_init(&node, &support, "{name}", "/") != NROS_RET_OK) {{
844        fprintf(stderr, "nros_node_init failed\n");
845        return 1;
846    }}
847
848    nros_publisher_t pub = nros_publisher_get_zero_initialized();
849    if (nros_publisher_init(&pub, &node,
850                            std_msgs_msg_int32_get_type_support(),
851                            "/chatter") != NROS_RET_OK) {{
852        fprintf(stderr, "nros_publisher_init failed\n");
853        return 1;
854    }}
855
856    std_msgs_msg_int32 msg;
857    std_msgs_msg_int32_init(&msg);
858    msg.data = 0;
859    (void)std_msgs_msg_int32_publish(&pub, &msg);
860    printf("{name}: published 0 on /chatter\n");
861
862    nros_publisher_fini(&pub);
863    nros_node_fini(&node);
864    nros_support_fini(&support);
865    return 0;
866}}
867"#,
868    );
869    fs::write(dir.join("src/main.c"), main_c)?;
870
871    if platform != "native" {
872        write_default_config_toml(dir)?;
873    }
874    Ok(())
875}

973kind    = "ethernet"
974ip      = "10.0.2.20/24"
