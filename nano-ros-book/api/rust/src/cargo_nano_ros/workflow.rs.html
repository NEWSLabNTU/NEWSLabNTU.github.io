1//! Main workflow orchestration for cargo-ros2
2//!
3//! This module coordinates the entire process:
4//! 1. Discover ROS dependencies from Cargo.toml
5//! 2. Check cache for each package
6//! 3. Generate missing/stale bindings
7//! 4. Update cache
8//! 5. Patch .cargo/config.toml
9//! 6. Invoke cargo build
10

25/// Workflow context
26pub struct WorkflowContext {
27    /// Project root directory
28    pub project_root: PathBuf,
29    /// Output directory for bindings (default: target/ros2_bindings)
30    pub output_dir: PathBuf,
31    /// Cache file path
32    pub cache_file: PathBuf,
33    /// Verbose output

70            project_root,

72            cache_file,

77    /// Discover ROS dependencies via ament index
78    pub fn discover_ament_packages(&self) -> Result<HashMap<String, PathBuf>> {
79        let index =
80            AmentIndex::from_env().wrap_err("Failed to load ament index (is ROS 2 sourced?)")?;
81
82        let mut packages = HashMap::new();
83        for (name, package) in index.packages() {
84            packages.insert(name.clone(), package.share_dir.clone());
85        }
86
87        Ok(packages)
88    }

90    /// Discover ROS dependencies from Cargo.toml
91    pub fn discover_ros_dependencies(&self) -> Result<Vec<RosDependency>> {
92        // Get known ROS packages from ament index
93        let ament_packages = self.discover_ament_packages()?;
94        let known_ros_packages = ament_packages.keys().cloned().collect();
95
96        // Parse Cargo.toml dependencies
97        let parser = DependencyParser::new(known_ros_packages);
98        parser.discover_dependencies(&self.project_root)
99    }

101    /// Check which packages need generation (cache miss or stale)
102    pub fn check_cache(
103        &self,
104        dependencies: &[RosDependency],
105        ament_packages: &HashMap<String, PathBuf>,
106    ) -> Result<Vec<String>> {
107        let cache = Cache::load(&self.cache_file)?;
108        let mut to_generate = Vec::new();
109
110        for dep in dependencies {
111            // Get the package share dir
112            let share_dir = match ament_packages.get(&dep.name) {
113                Some(dir) => dir,
114                None => {
115                    // Package not in ament index, skip
116                    continue;
117                }
118            };
119
120            // Calculate current checksum
121            let current_checksum = cache::calculate_package_checksum(share_dir)
122                .wrap_err_with(|| format!("Failed to calculate checksum for {}", dep.name))?;
123
124            // Check if cache is valid
125            if !cache.is_valid(&dep.name, &current_checksum) {
126                to_generate.push(dep.name.clone());
127            }
128        }
129
130        Ok(to_generate)
131    }

133    /// Generate bindings for a package using cargo-ros2-bindgen
134    pub fn generate_bindings(&self, package_name: &str) -> Result<PathBuf> {
135        if self.verbose {
136            eprintln!("  Generating bindings for {}...", package_name);
137        }
138
139        // Find rosidl-bindgen binary
140        let bindgen_binary = self.find_rosidl_bindgen()?;
141
142        // Build command
143        let output_path = self.output_dir.clone();
144        let mut cmd = Command::new(&bindgen_binary);
145        cmd.arg("--package")
146            .arg(package_name)
147            .arg("--output")
148            .arg(&output_path);
149
150        if self.verbose {
151            cmd.arg("--verbose");
152        }
153
154        // Execute
155        let output = cmd
156            .output()
157            .wrap_err_with(|| format!("Failed to execute {}", bindgen_binary.display()))?;
158
159        if !output.status.success() {
160            return Err(eyre!(
161                "rosidl-bindgen failed for {}: {}",
162                package_name,
163                String::from_utf8_lossy(&output.stderr)
164            ));
165        }
166
167        Ok(output_path.join(package_name))
168    }

170    /// Find rosidl-bindgen binary
171    fn find_rosidl_bindgen(&self) -> Result<PathBuf> {
172        // Try to find in target directory (development)
173        let dev_path = self
174            .project_root
175            .ancestors()
176            .find(|p| p.join("Cargo.toml").exists())
177            .map(|p| p.join("target").join("debug").join("rosidl-bindgen"));
178
179        if let Some(path) = dev_path
180            && path.exists()
181        {
182            return Ok(path);
183        }
184
185        // Try to find in PATH
186        if let Ok(output) = Command::new("which").arg("rosidl-bindgen").output()
187            && output.status.success()
188        {
189            let path_str = String::from_utf8_lossy(&output.stdout).trim().to_string();
190            return Ok(PathBuf::from(path_str));
191        }
192
193        Err(eyre!(
194            "rosidl-bindgen not found. Please build it first with 'cargo build'"
195        ))
196    }

198    /// Update cache after successful generation
199    pub fn update_cache(
200        &self,
201        package_name: &str,
202        package_share_dir: &Path,
203        output_dir: PathBuf,
204    ) -> Result<()> {
205        let mut cache = Cache::load(&self.cache_file)?;
206
207        // Calculate checksum of the source package
208        let checksum = cache::calculate_package_checksum(package_share_dir)
209            .wrap_err_with(|| format!("Failed to calculate checksum for {}", package_name))?;
210
211        let entry = CacheEntry {
212            package_name: package_name.to_string(),
213            checksum,
214            ros_distro: std::env::var("ROS_DISTRO").ok(),
215            package_version: None,
216            timestamp: SystemTime::now()
217                .duration_since(UNIX_EPOCH)
218                .unwrap()
219                .as_secs(),
220            output_dir,
221        };
222
223        cache.insert(entry);
224        cache.save(&self.cache_file)?;
225
226        Ok(())
227    }

229    /// Patch .cargo/config.toml with binding paths
230    pub fn patch_cargo_config(&self, packages: &[(String, PathBuf)]) -> Result<()> {
231        let mut patcher = ConfigPatcher::new(&self.project_root)?;
232
233        for (package_name, package_path) in packages {
234            if self.verbose {
235                eprintln!(
236                    "  Adding patch for {} -> {}",
237                    package_name,
238                    package_path.display()
239                );
240            }
241            patcher.add_patch(package_name, package_path);
242        }
243
244        patcher.save()?;
245        Ok(())
246    }

248    /// Generate bindings for multiple packages in parallel
249    fn generate_bindings_parallel(
250        &self,
251        packages: &[String],
252        ament_packages: &HashMap<String, PathBuf>,
253    ) -> Result<Vec<(String, PathBuf)>> {
254        use indicatif::{ProgressBar, ProgressStyle};
255        use rayon::prelude::*;
256        use std::sync::Mutex;
257
258        // Create progress bar
259        let pb = ProgressBar::new(packages.len() as u64);
260        pb.set_style(
261            ProgressStyle::default_bar()
262                .template(
263                    "{spinner:.green} [{elapsed_precise}] [{bar:40.cyan/blue}] {pos}/{len} {msg}",
264                )
265                .unwrap()
266                .progress_chars("#>-"),
267        );
268
269        // Use Mutex for thread-safe cache updates
270        let cache_file = self.cache_file.clone();
271        let results = Mutex::new(Vec::new());
272        let errors = Mutex::new(Vec::new());
273
274        packages.par_iter().for_each(|package_name| {
275            pb.set_message(format!("Generating {}", package_name));
276
277            match self.generate_bindings(package_name) {
278                Ok(output_dir) => {
279                    // Update cache
280                    if let Some(share_dir) = ament_packages.get(package_name)
281                        && let Err(e) = self.update_cache_threadsafe(
282                            package_name,
283                            share_dir,
284                            output_dir.clone(),
285                            &cache_file,
286                        )
287                    {
288                        errors.lock().unwrap().push(format!(
289                            "Failed to update cache for {}: {}",
290                            package_name, e
291                        ));
292                    }
293
294                    results
295                        .lock()
296                        .unwrap()
297                        .push((package_name.clone(), output_dir));
298                }
299                Err(e) => {
300                    errors
301                        .lock()
302                        .unwrap()
303                        .push(format!("Failed to generate {}: {}", package_name, e));
304                }
305            }
306
307            pb.inc(1);
308        });
309
310        pb.finish_with_message("Generation complete");
311
312        // Check for errors
313        let errors = errors.lock().unwrap();
314        if !errors.is_empty() {
315            return Err(eyre!(
316                "Errors during parallel generation:\n{}",
317                errors.join("\n")
318            ));
319        }
320        drop(errors);
321
322        let results = results.lock().unwrap().clone();
323        Ok(results)
324    }

326    /// Thread-safe cache update for parallel generation
327    fn update_cache_threadsafe(
328        &self,
329        package_name: &str,
330        package_share_dir: &Path,
331        output_dir: PathBuf,
332        cache_file: &Path,
333    ) -> Result<()> {
334        use std::sync::Mutex;
335        static CACHE_LOCK: Mutex<()> = Mutex::new(());
336
337        let _lock = CACHE_LOCK.lock().unwrap();
338
339        let mut cache = Cache::load(cache_file)?;
340
341        // Calculate checksum of the source package
342        let checksum = cache::calculate_package_checksum(package_share_dir)
343            .wrap_err_with(|| format!("Failed to calculate checksum for {}", package_name))?;
344
345        let entry = CacheEntry {
346            package_name: package_name.to_string(),
347            checksum,
348            ros_distro: std::env::var("ROS_DISTRO").ok(),
349            package_version: None,
350            timestamp: SystemTime::now()
351                .duration_since(UNIX_EPOCH)
352                .unwrap()
353                .as_secs(),
354            output_dir,
355        };
356
357        cache.insert(entry);
358        cache.save(cache_file)?;
359
360        Ok(())
361    }

363    /// Run the complete workflow
364    pub fn run(&self, bindings_only: bool) -> Result<()> {
365        if self.verbose {
366            eprintln!("cargo-ros2 workflow starting...");
367        }
368
369        // Step 1: Discover ament packages
370        if self.verbose {
371            eprintln!("Step 1: Discovering ROS packages from ament index...");
372        }
373        let ament_packages = self.discover_ament_packages()?;
374
375        if self.verbose {
376            eprintln!("  Found {} packages in ament index", ament_packages.len());
377        }
378
379        // Step 2: Discover ROS dependencies from Cargo.toml
380        if self.verbose {
381            eprintln!("Step 2: Discovering ROS dependencies from Cargo.toml...");
382        }
383        let mut dependencies = self.discover_ros_dependencies()?;
384
385        if dependencies.is_empty() {
386            eprintln!("No ROS 2 dependencies found in Cargo.toml");
387            if !bindings_only {
388                return self.invoke_cargo_build();
389            }
390            return Ok(());
391        }
392
393        if self.verbose {
394            eprintln!("  Found {} ROS dependencies", dependencies.len());
395        }
396
397        // Step 3: Iteratively discover and generate transitive dependencies
398        let mut all_generated = Vec::new();
399        let mut seen_packages = std::collections::HashSet::new();
400
401        loop {
402            // Check cache for current set of dependencies
403            if self.verbose {
404                eprintln!(
405                    "Step 3: Checking cache for {} packages...",
406                    dependencies.len()
407                );
408            }
409            let to_generate = self.check_cache(&dependencies, &ament_packages)?;
410
411            if to_generate.is_empty() {
412                break; // No more packages to generate
413            }
414
415            if self.verbose {
416                eprintln!("  {} packages need generation", to_generate.len());
417            }
418
419            // Generate bindings (in parallel if multiple packages)
420            let generated_packages = if to_generate.len() > 1 {
421                self.generate_bindings_parallel(&to_generate, &ament_packages)?
422            } else {
423                let mut generated_packages = Vec::new();
424                for package_name in &to_generate {
425                    let output_dir = self.generate_bindings(package_name)?;
426
427                    // Get share dir for checksum calculation
428                    if let Some(share_dir) = ament_packages.get(package_name) {
429                        self.update_cache(package_name, share_dir, output_dir.clone())?;
430                    }
431
432                    generated_packages.push((package_name.clone(), output_dir));
433                }
434                generated_packages
435            };
436
437            all_generated.extend(generated_packages.clone());
438
439            // Mark these packages as seen
440            for (pkg_name, _) in &generated_packages {
441                seen_packages.insert(pkg_name.clone());
442            }
443
444            // Discover transitive dependencies from generated packages
445            let mut new_deps = Vec::new();
446            for (_pkg_name, pkg_path) in &generated_packages {
447                if let Ok(transitive_deps) = self.discover_transitive_dependencies(pkg_path) {
448                    for dep in transitive_deps {
449                        // Only add if we haven't seen it yet and it's a known ROS package
450                        if !seen_packages.contains(&dep) && ament_packages.contains_key(&dep) {
451                            new_deps.push(RosDependency {
452                                name: dep.clone(),
453                                direct: false,
454                            });
455                            seen_packages.insert(dep);
456                        }
457                    }
458                }
459            }
460
461            if new_deps.is_empty() {
462                break; // No new dependencies found
463            }
464
465            if self.verbose {
466                eprintln!("  Discovered {} transitive dependencies", new_deps.len());
467            }
468
469            dependencies = new_deps;
470        }
471
472        // Step 4: Patch .cargo/config.toml
473        if !all_generated.is_empty() {
474            if self.verbose {
475                eprintln!("Step 4: Patching .cargo/config.toml...");
476            }
477            self.patch_cargo_config(&all_generated)?;
478        }
479
480        // Step 5: Invoke cargo build (unless --bindings-only)
481        if !bindings_only {
482            if self.verbose {
483                eprintln!("Step 5: Invoking cargo build...");
484            }
485            self.invoke_cargo_build()?;
486        }
487
488        Ok(())
489    }

491    /// Discover transitive dependencies from a generated package
492    fn discover_transitive_dependencies(&self, package_path: &Path) -> Result<Vec<String>> {
493        use std::fs;
494
495        let cargo_toml_path = package_path.join("Cargo.toml");
496        let contents = fs::read_to_string(&cargo_toml_path)
497            .wrap_err_with(|| format!("Failed to read {}", cargo_toml_path.display()))?;
498
499        let mut dependencies = Vec::new();
500        let mut in_dependencies = false;
501
502        for line in contents.lines() {
503            let trimmed = line.trim();
504
505            if trimmed.starts_with("[dependencies]") {
506                in_dependencies = true;
507                continue;
508            }
509
510            if trimmed.starts_with('[') {
511                in_dependencies = false;
512                continue;
513            }
514
515            if in_dependencies && !trimmed.is_empty() && !trimmed.starts_with('#') {
516                // Extract package name (before '=' or space)
517                if let Some(pkg_name) = trimmed.split('=').next() {
518                    let pkg_name = pkg_name.trim();
519                    // Skip special entries like "serde" or "rosidl-runtime-rs"
520                    if !pkg_name.contains('-') || pkg_name.contains('_') {
521                        dependencies.push(pkg_name.to_string());
522                    }
523                }
524            }
525        }
526
527        Ok(dependencies)
528    }

530    /// Invoke cargo build
531    fn invoke_cargo_build(&self) -> Result<()> {
532        if self.verbose {
533            eprintln!("Step 4: Invoking cargo build...");
534        }
535
536        let status = Command::new("cargo")
537            .arg("build")
538            .current_dir(&self.project_root)
539            .status()
540            .wrap_err("Failed to execute cargo build")?;
541
542        if !status.success() {
543            return Err(eyre!("cargo build failed"));
544        }
545
546        Ok(())
547    }
