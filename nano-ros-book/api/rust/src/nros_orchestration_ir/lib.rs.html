1//! Shared per-tier orchestration IR — schema + resolver (RFC-0015).
2//!
3//! This leaf crate holds the small set of `system.toml` types that
4//! describe scheduling tiers + the [`resolve_tiers`] algorithm that
5//! lowers them (with `[[node_overrides]]` applied) into an ordered,
6//! per-RTOS **resolved tier table**. It is depended on by BOTH:
7//!
8//! - the `nros` CLI (`nros-cli-core`), whose `codegen-system` bakes the
9//!   resolved table into `nros-plan.json`, and
10//! - the `nros::main!()` proc-macro (`nros-macros`), which resolves the
11//!   same table at compile time to emit one task/`Executor` per tier.
12//!
13//! Keeping the schema + resolver here is the single source of truth so
14//! the build-time and codegen paths can never drift. The crate is pure
15//! host code (serde + thiserror); it carries no runtime/`no_std`
16//! dependencies.
17//!
18//! The all-default-tier degenerate case (no `[tiers.*]`, no callback
19//! groups) resolves to a single synthesized `"default"` tier — the
20//! single-task shape that ships today.
21//!
22//! ## Board key mapping
23//!
24//! [`board_path_for`] is also kept here as the **single source of truth**
25//! for the `deploy`-key → board-ZST-path mapping, consumed by both
26//! `nros-macros` and `nros-cli-core`. See [`board_path_for`].
27
28use std::collections::{BTreeMap, BTreeSet};

30use serde::{Deserialize, Serialize};
31use thiserror::Error;

37/// Board deployment key → canonical Rust ZST path (double-colon-prefixed,
38/// ready for `syn::parse_str` or direct string inclusion in generated code).
39///
40/// This is the **single source of truth** for the board-key → board-crate
41/// mapping, consumed by BOTH:
42///
43/// - `nros-macros` (`nros::main!(deploy = "…")`) — wraps the returned string
44///   in `syn::parse_str::<syn::Path>()` to emit a `BoardEntry` impl call, and
45/// - `nros-cli-core` (`nros codegen entry --lang rust`) — includes the string
46///   directly in the emitted `main.rs` body.
47///
48/// Adding a new board requires ONE edit here, not two. Aliases (e.g.
49/// `"qemu-arm-freertos"` for `"freertos"`) are normalised here so callers
50/// need no extra translation.
51///
52/// Keys match the `deploy = "…"` strings in
53/// `[package.metadata.nros.deploy.<board>]` (RFC-0014 §3).
54pub fn board_path_for(key: &str) -> Option<&'static str> {
55    Some(match key {
56        "native" | "posix" => "::nros_board_native::NativeBoard",
57        // FreeRTOS — MPS2-AN385 Cortex-M3 (the only FreeRTOS board today).
58        // The RTOS calls `main()`; the board ZST impls `BoardEntry`.
59        "freertos" | "freertos-qemu-mps2-an385" | "qemu-arm-freertos" => {
60            "::nros_board_mps2_an385_freertos::Mps2An385"
61        }
62        "threadx-linux" => "::nros_board_threadx_linux::ThreadxLinux",
63        "threadx-qemu-riscv64" | "qemu-riscv64-threadx" => {
64            "::nros_board_threadx_qemu_riscv64::ThreadxQemuRiscv64"
65        }
66        "nuttx" | "qemu-arm-nuttx" => "::nros_board_nuttx_qemu_arm::QemuArmVirt",
67        // Phase 225.O — CI-runnable ESP32-C3 QEMU (OpenETH) board. Routed
68        // through `Framework::Esp32` emit shape in the proc-macro.
69        "esp32-qemu" | "qemu-esp32-baremetal" => "::nros_board_esp32_qemu::Esp32QemuEntry",
70        // Phase 225.P — Zephyr owns `main`; the board ZST impls `NetworkWait`
71        // only (NOT `BoardEntry`). The proc-macro routes through
72        // `Framework::Zephyr` and emits a `rust_main` staticlib export.
73        "zephyr" => "::nros_board_zephyr::ZephyrBoard",
74        // Phase 216.B.3 — RTIC + STM32F4. Board ZST impls `RticBoardEntry`.
75        "rtic-stm32f4" => "::nros_board_rtic_stm32f4::RticStm32F4",
76        "rtic-mps2-an385" | "qemu-rtic-mps2-an385" => "::nros_board_rtic_mps2_an385::RticMps2An385",
77        // Phase 244.D1 — pure bare-metal (no-RTOS) MPS2-AN385 direct-exec
78        // board. Board ZST impls `nros_platform::BoardEntry`. Distinct from
79        // `rtic-mps2-an385`, which routes through the RTIC framework emit.
80        "qemu-mps2-an385" | "mps2-an385" => "::nros_board_mps2_an385::Mps2An385",
81        // Phase 244.C5 — pure bare-metal STM32F4 direct-exec board.
82        "stm32f4" => "::nros_board_stm32f4::Stm32F4",
83        // Phase 216.C.3 — Embassy + STM32F4. Board ZST impls `EmbassyBoardEntry`.
84        "embassy-stm32f4" => "::nros_board_embassy_stm32f4::EmbassyStm32F4",
85        _ => return None,
86    })
87}

93/// `[tiers.<name>]` — a symbolic priority tier (RFC-0015 §4.2). Carries the
94/// RTOS-agnostic `spin_period_us` plus a per-RTOS sub-table
95/// (`[tiers.<name>.<rtos>]`) giving the concrete priority/stack for each target.
96#[derive(Clone, Debug, Default, PartialEq, Eq, Serialize, Deserialize)]
97#[serde(deny_unknown_fields)]
98pub struct TierDef {
99    #[serde(default, skip_serializing_if = "Option::is_none")]
100    pub spin_period_us: Option<u64>,
101    // Phase 256 W4 (decision A) — the RTOS-AGNOSTIC real-time policy a callback
102    // group runs under (absorbed from the retired `[[scheduling.contexts]]`
103    // overlay). Per-RTOS placement (priority/stack) stays in the `<rtos>`
104    // sub-tables; these describe HOW it is scheduled, identically on every RTOS.
105    // All optional → a plain priority tier (today's shape) is byte-identical.
106    /// Scheduling class — the plan's `SchedClass` (snake_case): `"best_effort"` |
107    /// `"real_time"` (default for a priority tier) | `"time_triggered"` |
108    /// `"interrupt"`. The W4.2 codegen lowering validates + maps it.
109    #[serde(default, skip_serializing_if = "Option::is_none")]
110    pub class: Option<String>,
111    /// Callback period (µs) for `periodic` / `time_triggered`.
112    #[serde(default, skip_serializing_if = "Option::is_none")]
113    pub period_us: Option<u64>,
114    /// Execution-time budget (µs) — EDF/sporadic.
115    #[serde(default, skip_serializing_if = "Option::is_none")]
116    pub budget_us: Option<u64>,
117    /// Relative deadline (µs) — EDF.
118    #[serde(default, skip_serializing_if = "Option::is_none")]
119    pub deadline_us: Option<u64>,
120    /// On deadline miss — the plan's `DeadlinePolicy` (snake_case): `"ignore"`
121    /// (default) | `"warn"` | `"skip"` | `"fault"`.
122    #[serde(default, skip_serializing_if = "Option::is_none")]
123    pub deadline_policy: Option<String>,
124    /// CPU core to pin the tier task to (SMP); `None` ⇒ unpinned.
125    #[serde(default, skip_serializing_if = "Option::is_none")]
126    pub core: Option<u32>,
127    #[serde(default, skip_serializing_if = "Option::is_none")]
128    pub freertos: Option<TierRtosSpec>,
129    #[serde(default, skip_serializing_if = "Option::is_none")]
130    pub zephyr: Option<TierRtosSpec>,
131    #[serde(default, skip_serializing_if = "Option::is_none")]
132    pub threadx: Option<TierRtosSpec>,
133    #[serde(default, skip_serializing_if = "Option::is_none")]
134    pub nuttx: Option<TierRtosSpec>,
135    #[serde(default, skip_serializing_if = "Option::is_none")]
136    pub posix: Option<TierRtosSpec>,
137}
138
139/// `[tiers.<name>.<rtos>]` — concrete per-RTOS task knobs. One shape for all
140/// RTOSes; `priority` is `i64` to admit Zephyr's negative coop priorities.
141#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]
142#[serde(deny_unknown_fields)]
143pub struct TierRtosSpec {
144    pub priority: i64,
145    #[serde(default, skip_serializing_if = "Option::is_none")]
146    pub stack_bytes: Option<u32>,
147    /// ThreadX preemption threshold (ignored on other RTOSes).
148    #[serde(default, skip_serializing_if = "Option::is_none")]
149    pub preempt_threshold: Option<i64>,
150    /// POSIX scheduler class (e.g. `"SCHED_FIFO"`).
151    #[serde(default, skip_serializing_if = "Option::is_none")]
152    pub sched_class: Option<String>,
153}

155/// `[[node.callback_groups]]` row (Phase 228.A, RFC-0015 §4.1).
156#[derive(Clone, Debug, PartialEq, Eq, Serialize, Deserialize)]

158pub struct CallbackGroupDecl {

168    pub tier: String,
169}

202/// The synthesized tier used when a callback group names no tier (or none are
203/// declared at all). It needs no `[tiers.default]` table.
204pub const DEFAULT_TIER: &str = "default";
205
206/// One resolved tier: a concrete RTOS task to emit.
207#[derive(Clone, Debug, PartialEq, Eq)]
208pub struct ResolvedTier {
209    pub name: String,
210    /// RTOS-specific numeric priority (the spawn-call value).
211    pub priority: i64,
212    pub stack_bytes: Option<u32>,
213    pub spin_period_us: Option<u64>,
214    pub preempt_threshold: Option<i64>,
215    pub sched_class: Option<String>,
216    // Phase 256 W4 — the RTOS-agnostic real-time policy (from `TierDef`), carried
217    // through so the planner can lower a tier to a `PlanSchedContext` (the home the
218    // retired `[[scheduling.contexts]]` overlay used to fill).
219    pub class: Option<String>,
220    pub period_us: Option<u64>,
221    pub budget_us: Option<u64>,
222    pub deadline_us: Option<u64>,
223    pub deadline_policy: Option<String>,
224    pub core: Option<u32>,
225    /// `(node_name, callback_group_id)` pairs assigned to this tier, sorted.
226    pub members: Vec<(String, String)>,
227}

229/// The ordered tier table for one deploy target.
230#[derive(Clone, Debug, PartialEq, Eq)]
231pub struct ResolvedTierTable {
232    /// Tiers ordered highest-priority first (by the RTOS numeric `priority`).
233    pub tiers: Vec<ResolvedTier>,
234}
235
236impl ResolvedTierTable {
237    /// True when this is the single-task degenerate case (one tier, the
238    /// synthesized `default`). Codegen uses this to skip multi-task scaffolding.
239    pub fn is_single_tier(&self) -> bool {
240        self.tiers.len() == 1 && self.tiers[0].name == DEFAULT_TIER
241    }
242}

244#[derive(Debug, Clone, PartialEq, Eq, Error)]
245pub enum TierResolveError {
246    #[error(
247        "callback group `{node}/{group}` names tier `{tier}`, which has no `[tiers.{tier}]` definition"
248    )]
249    UnknownTier {
250        node: String,
251        group: String,
252        tier: String,
253    },
254    #[error("tier `{tier}` has no `[tiers.{tier}.{rtos}]` sub-table for the target RTOS")]
255    MissingRtosSpec { tier: String, rtos: String },
256    #[error("`[[node_overrides]]` targets node `{node}` which is not a component in the system")]
257    UnknownOverrideNode { node: String },
258}
259
260/// Pick a tier's per-RTOS spec by target name.
261fn rtos_spec<'a>(def: &'a TierDef, rtos: &str) -> Option<&'a TierRtosSpec> {
262    match rtos {
263        "freertos" => def.freertos.as_ref(),
264        "zephyr" => def.zephyr.as_ref(),
265        "threadx" => def.threadx.as_ref(),
266        "nuttx" => def.nuttx.as_ref(),
267        "posix" | "native" => def.posix.as_ref(),
268        _ => None,
269    }
270}

272/// Resolve a system's tiers against the per-node callback groups for one
273/// target RTOS.
274///
275/// Inputs are the decomposed `system.toml` pieces so both the CLI (which
276/// holds a full `SystemToml`) and the proc-macro (which parses a leaner
277/// view) can call this without sharing the whole config type:
278/// - `tiers` — the `[tiers.*]` table.
279/// - `node_overrides` — the `[[node_overrides]]` rows.
280/// - `component_names` — the system's component *instance* names, used to
281///   validate override targets.
282/// - `callback_groups` — component-instance-name → its declared groups
283///   (`[package.metadata.nros.node].callback_groups`).
284pub fn resolve_tiers(
285    tiers: &BTreeMap<String, TierDef>,
286    node_overrides: &[NodeOverride],
287    component_names: &BTreeSet<&str>,
288    callback_groups: &BTreeMap<String, Vec<CallbackGroupDecl>>,
289    target_rtos: &str,
290) -> Result<ResolvedTierTable, TierResolveError> {
291    // Per-node group→tier overrides from the system.
292    let mut overrides: BTreeMap<(&str, &str), &str> = BTreeMap::new();
293    for ov in node_overrides {
294        if !component_names.contains(ov.name.as_str()) {
295            return Err(TierResolveError::UnknownOverrideNode {
296                node: ov.name.clone(),
297            });
298        }
299        for cg in &ov.callback_groups {
300            overrides.insert((ov.name.as_str(), cg.id.as_str()), cg.tier.as_str());
301        }
302    }
303
304    // (node, group) → effective tier (override wins over the node's declaration).
305    let mut members_by_tier: BTreeMap<String, Vec<(String, String)>> = BTreeMap::new();
306    for (node, groups) in callback_groups {
307        for g in groups {
308            let tier = overrides
309                .get(&(node.as_str(), g.id.as_str()))
310                .copied()
311                .unwrap_or(g.tier.as_str());
312            members_by_tier
313                .entry(tier.to_string())
314                .or_default()
315                .push((node.clone(), g.id.clone()));
316        }
317    }
318
319    // Phase 273 W4 (RFC-0047): the v1 node-pinned-to-tier rule is lifted.
320    // A single node may now have callback groups in different tiers (sub-node
321    // tiering). Each group is individually bound to its sched context via
322    // bind_group_sched; the caller is responsible for thread-safety when groups
323    // in different tiers run concurrently (RFC-0047 §3).
324
325    // Degenerate: nothing declared → a single synthesized default tier.
326    if members_by_tier.is_empty() {
327        return Ok(ResolvedTierTable {
328            tiers: vec![default_tier(Vec::new())],
329        });
330    }
331
332    let mut out = Vec::with_capacity(members_by_tier.len());
333    for (name, mut members) in members_by_tier {
334        members.sort();
335        if name == DEFAULT_TIER && !tiers.contains_key(DEFAULT_TIER) {
336            // The default tier needs no `[tiers.default]` table.
337            out.push(default_tier(members));
338            continue;
339        }
340        let def = tiers.get(&name).ok_or_else(|| {
341            let (node, group) = members.first().cloned().unwrap_or_default();
342            TierResolveError::UnknownTier {
343                node,
344                group,
345                tier: name.clone(),
346            }
347        })?;
348        let spec =
349            rtos_spec(def, target_rtos).ok_or_else(|| TierResolveError::MissingRtosSpec {
350                tier: name.clone(),
351                rtos: target_rtos.to_string(),
352            })?;
353        out.push(ResolvedTier {
354            name,
355            priority: spec.priority,
356            stack_bytes: spec.stack_bytes,
357            spin_period_us: def.spin_period_us,
358            preempt_threshold: spec.preempt_threshold,
359            sched_class: spec.sched_class.clone(),
360            class: def.class.clone(),
361            period_us: def.period_us,
362            budget_us: def.budget_us,
363            deadline_us: def.deadline_us,
364            deadline_policy: def.deadline_policy.clone(),
365            core: def.core,
366            members,
367        });
368    }
369
370    // Highest RTOS priority first. (The system owner authors numbers correct for
371    // the target RTOS's direction; v1 does not invert.)
372    out.sort_by(|a, b| b.priority.cmp(&a.priority).then(a.name.cmp(&b.name)));
373    Ok(ResolvedTierTable { tiers: out })
374}

376fn default_tier(members: Vec<(String, String)>) -> ResolvedTier {
377    ResolvedTier {
378        name: DEFAULT_TIER.to_string(),
379        priority: 0,
380        stack_bytes: None,
381        spin_period_us: None,
382        preempt_threshold: None,
383        sched_class: None,

386        budget_us: None,
387        deadline_us: None,
388        deadline_policy: None,
389        core: None,
390        members,
391    }
