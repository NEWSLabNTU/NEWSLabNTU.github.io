1//! [`BoardEntry`] — Phase 212.N.1.
2//!
3//! The single boot-driver trait every Entry pkg `main.rs` invokes:
4//!
5//! ```ignore
6//! fn main() {
7//!     let _ = <MyBoard as BoardEntry>::run(|runtime| {
8//!         run_plan(runtime)         // codegen-emitted (212.N.4)
9//!     });
10//! }
11//! ```
12//!
13//! `run` owns the full lifecycle:
14//!
15//! 1. [`super::BoardInit::init_hardware`]
16//! 2. [`super::TransportBringup::init_transport`] (if implemented)
17//! 3. [`super::NetworkWait::wait_link_up`] (if implemented)
18//! 4. Open executor, build `RuntimeCtx`, invoke `setup(runtime)`.
19//! 5. Spin executor to completion (or termination signal).
20//! 6. [`super::BoardExit::exit_success`] / `exit_failure`.
21//!
22//! The exact body lives in the family driver crates (212.N.2); the
23//! trait here pins the signature so codegen + user Entry pkg can
24//! call it without knowing the family.
25
26use super::runtime::RuntimeCtx;
27
28/// Per-board boot driver.
29///
30/// Implementations live in the family driver crates
31/// (`nros-board-posix`, `nros-board-freertos`, …). Per-board crates
32/// (`nros-board-mps2-an385-freertos`, …) plug the family.
33pub trait BoardEntry: super::Board {
34    /// Drive the full boot → user-closure → exit flow.
35    ///
36    /// `setup` receives a `&mut RuntimeCtx` with overlay knobs from
37    /// the launch file / CLI args. Returning `Err` from `setup` makes
38    /// `run` route to [`super::BoardExit::exit_failure`]; `Ok`
39    /// proceeds to executor spin + clean exit.
40    ///
41    /// **Returns `Result`, not `!`.** The legacy
42    /// `nros-board-common::board_init::BoardEntry::run` diverged;
43    /// 212.N keeps the option to return so unit tests can drive it
44    /// in a hosted process without `exit()` killing the test
45    /// harness. Production boards still call `exit_*` from inside
46    /// `run`'s body after spin returns.
47    fn run<F, E>(setup: F) -> Result<(), E>
48    where
49        F: FnOnce(&mut RuntimeCtx<'_>) -> Result<(), E>,
50        E: core::fmt::Debug;
51}
