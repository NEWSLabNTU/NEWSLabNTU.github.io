1//! [`RuntimeCtx`] — Phase 212.N.1.
2//!
3//! Per-invocation runtime context handed to `BoardEntry::run`'s
4//! `setup` callback. Carries the overlay knobs the codegen
5//! `run_plan(runtime)` body reads:
6//!
7//! - **params** — `(key, value)` pairs from launch XML
8//!   `<param name="…" value="…"/>` or `--ros-args -p k:=v`.
9//! - **remaps** — `(from, to)` topic/service renames.
10//! - **env** — environment-style key/value pairs (POSIX `getenv`
11//!   shape) accessible from no_std boards via this struct rather
12//!   than a `libc::getenv` call.
13//!
14//! ## no_std-safe shape
15//!
16//! Slice-of-tuples kept on the boot stack. No allocation, no
17//! `core::collections`. Codegen owns the storage and passes a
18//! `&mut RuntimeCtx<'_>` whose backing slices live in `static`s.
19//!
20//! Hosted boards (POSIX) may instead build a longer-lived owned
21//! variant on the heap; the trait surface is slice-based so
22//! both shapes work.
23
24/// Runtime context handed to `BoardEntry::run(setup)`.
25///
26/// All three slices may be empty. A board's launch overlay typically
27/// populates `params` + `remaps`; `env` is rarely set on embedded.
28#[derive(Debug)]
29pub struct RuntimeCtx<'a> {
30    /// `<param name=… value=…/>` from launch XML, or
31    /// `-p name:=value` CLI overrides.
32    pub params: &'a [(&'a str, &'a str)],
33
34    /// Topic / service / action remaps: `(from, to)`.
35    pub remaps: &'a [(&'a str, &'a str)],
36
37    /// Environment-style key/value pairs (mostly POSIX). Empty on
38    /// embedded boards.
39    pub env: &'a [(&'a str, &'a str)],
40}
41
42impl<'a> RuntimeCtx<'a> {
43    /// An empty `RuntimeCtx` — no params, no remaps, no env. Useful
44    /// as a placeholder when running a launch-less single-node
45    /// example, or in unit tests.
46    pub const EMPTY: Self = Self {
47        params: &[],
48        remaps: &[],
49        env: &[],
50    };
51
52    /// Lookup a param by name; first match wins. Linear scan
53    /// because the slice is typically small (≤ a dozen entries).
54    pub fn param(&self, name: &str) -> Option<&'a str> {
55        self.params
56            .iter()
57            .find(|(k, _)| *k == name)
58            .map(|(_, v)| *v)
59    }
60
61    /// Lookup a remap by the original (`from`) name; returns the
62    /// rewritten name when remapped, else `None`.
63    pub fn remap(&self, from: &str) -> Option<&'a str> {
64        self.remaps
65            .iter()
66            .find(|(k, _)| *k == from)
67            .map(|(_, v)| *v)
68    }
69
70    /// Lookup an env entry by name.
71    pub fn env_var(&self, name: &str) -> Option<&'a str> {
72        self.env.iter().find(|(k, _)| *k == name).map(|(_, v)| *v)
73    }
74}
75
76impl<'a> Default for RuntimeCtx<'a> {
77    fn default() -> Self {
78        Self::EMPTY
79    }
80}

82/// Error returned by the codegen-emitted `run_plan(runtime)` body
83/// (Phase 212.N.4) and by Component pkg `register(runtime)` wrappers
84/// (Phase 212.N.7 step-2).
85///
86/// `no_std`-safe — variants are string-typed so embedded Entry pkgs
87/// don't need to pull `thiserror`/`anyhow` to print. The
88/// out-of-tree `nros-build` codegen library re-exports this type so
89/// emitted code references `::nros_platform::RuntimeError`, NOT
90/// `::nros_build::RuntimeError` — the embedded Entry pkg's runtime
91/// path then doesn't need `nros-build` as a runtime dep (build-dep
92/// only).
93#[derive(Debug)]
94#[non_exhaustive]
95pub enum RuntimeError {
96    /// A component's `register(runtime)` call failed. The string
97    /// carries the component pkg name.
98    ComponentRegister(&'static str),
99}
100
101impl core::fmt::Display for RuntimeError {
102    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
103        match self {
104            Self::ComponentRegister(msg) => write!(f, "component register failed: {msg}"),
105        }
106    }
107}
