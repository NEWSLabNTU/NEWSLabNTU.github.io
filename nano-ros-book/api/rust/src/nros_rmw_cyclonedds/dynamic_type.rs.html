1//! Runtime descriptor builder.
2//!
3//! [`DescriptorBuilder::build`] turns a [`nros_serdes::Message`]'s
4//! static [`nros_serdes::Field`] schema into a Cyclone DDS
5//! `dds_topic_descriptor_t *` by:
6//!
7//! 1. Walking the schema recursively, flattening every
8//!    `&'static FieldType` reference into a stable index in the
9//!    `kinds[]` table.
10//! 2. Building the top-level `fields[]` table.
11//! 3. Calling the C++ bridge in
12//!    [`crate::bridge`] to construct the Cyclone descriptor.
13//! 4. Returning the resulting opaque pointer.
14//!
15//! The walker is `no_std`, alloc-free: it borrows fixed-capacity
16//! stack arrays (`MAX_FIELDS`, `MAX_KINDS`) sized at compile time
17//! by `option_env!` knobs.
18//!
19//! # Bounded sizing
20//!
21//! * [`MAX_FIELDS`] — max top-level fields per message. Default 64.
22//!   Override at compile time via `NROS_CYCLONEDDS_MAX_FIELDS`.
23//! * [`MAX_KINDS`] — max flattened kinds (fields + every nested kind
24//!   reached via recursion). Default 256. Override via
25//!   `NROS_CYCLONEDDS_MAX_KINDS`.
26//! * [`MAX_NESTED_DEPTH`] — recursion ceiling on `Nested` walks.
27//!   Default 8. Override via `NROS_CYCLONEDDS_MAX_NESTED_DEPTH`.
28//!
29//! Overflow returns [`BuildError::RegistryFull`] (for the registry's
30//! type-table) or [`BuildError::NestedDepthExceeded`] / a flat-table
31//! overflow variant — never panics.
32

42/// Maximum number of top-level fields per message. Compile-time
43/// knob: `NROS_CYCLONEDDS_MAX_FIELDS=<N>`. Default 64.
44pub const MAX_FIELDS: usize = parse_env_usize(option_env!("NROS_CYCLONEDDS_MAX_FIELDS"), 64);
45
46/// Maximum number of flattened `FieldType` entries in `kinds[]`
47/// (top-level fields + every nested `FieldType` reached via
48/// recursion). Compile-time knob: `NROS_CYCLONEDDS_MAX_KINDS=<N>`.
49/// Default 256.
50pub const MAX_KINDS: usize = parse_env_usize(option_env!("NROS_CYCLONEDDS_MAX_KINDS"), 256);
51
52/// Recursion ceiling on `FieldType::Nested` walks. Default 8.
53/// Compile-time knob: `NROS_CYCLONEDDS_MAX_NESTED_DEPTH=<N>`.
54pub const MAX_NESTED_DEPTH: usize =
55    parse_env_usize(option_env!("NROS_CYCLONEDDS_MAX_NESTED_DEPTH"), 8);
56
57const _: () = assert!(MAX_FIELDS >= 1, "MAX_FIELDS must be at least 1");
58const _: () = assert!(MAX_KINDS >= 1, "MAX_KINDS must be at least 1");
59const _: () = assert!(MAX_NESTED_DEPTH >= 1, "MAX_NESTED_DEPTH must be at least 1");

61/// Per-name buffer size for the on-stack NUL-terminated copies the
62/// builder hands to the C bridge. 64 bytes covers every ROS field
63/// name (typically ≤ 32) and every Cyclone-mangled type name
64/// (`pkg::msg::dds_::Type_`, ≤ ~60 bytes). Total stack budget:
65/// `(MAX_FIELDS + MAX_KINDS) × NAME_SLOT_LEN` — 20 KiB at default
66/// (64 + 256 = 320 slots × 64 B). Tune via the MAX_* knobs if your
67/// schema's names exceed this.
68pub const NAME_SLOT_LEN: usize = 64;
69
70/// Per-type-name buffer size for the top-level message name. Larger
71/// than [`NAME_SLOT_LEN`] because the message's full ROS path
72/// (`my_long_pkg/msg/SomeNestedThing`) can run to ~80 bytes in the
73/// wild; 256 still rounds up generously.
74pub const TYPE_NAME_BUF_LEN: usize = 256;
75
76/// Compile-time `usize` parser. Mirrors the existing
77/// `option_env!` knob pattern in `nros-platform`. Returns `default`
78/// on `None`; panics in `const` on parse failure (caught at compile
79/// time, surfaces as a clean diagnostic at the call site).
80const fn parse_env_usize(s: Option<&str>, default: usize) -> usize {
81    match s {
82        None => default,
83        Some(s) => {
84            let bytes = s.as_bytes();
85            if bytes.is_empty() {
86                panic!("env knob set but empty");
87            }
88            let mut acc: usize = 0;
89            let mut i = 0;
90            while i < bytes.len() {
91                let b = bytes[i];
92                if b < b'0' || b > b'9' {
93                    panic!("non-digit in env knob");
94                }
95                acc = acc * 10 + (b - b'0') as usize;
96                i += 1;
97            }
98            acc
99        }
100    }
101}

103/// Failure modes for [`DescriptorBuilder::build`].
104#[derive(Debug, Clone, Copy, PartialEq, Eq)]
105pub enum BuildError {
106    /// The C++ bridge does not yet support this `FieldType` variant
107    /// (e.g. `WString` on a Cyclone build without wide-string
108    /// support). The Rust walker recognises every variant; this
109    /// error always surfaces from the bridge.
110    UnsupportedFieldType,
111    /// `MAX_NESTED_DEPTH` exceeded while recursing through
112    /// [`FieldType::Nested`].
113    NestedDepthExceeded,
114    /// `MAX_FIELDS` exceeded — top-level message has too many fields.
115    FieldsOverflow,
116    /// `MAX_KINDS` exceeded — flattening the recursive `FieldType`
117    /// graph would not fit. Bump `NROS_CYCLONEDDS_MAX_KINDS`.
118    KindsOverflow,
119    /// `type_name` was empty (Cyclone rejects empty type names).
120    EmptyTypeName,
121    /// `M::FIELDS` was empty.
122    EmptySchema,
123    /// `M::TYPE_NAME` contained an embedded NUL — can't pass through
124    /// a C string boundary.
125    InvalidTypeName,
126    /// Cyclone's `dds_return_t` reported a failure code (passed
127    /// through from the bridge).
128    DdsError(i32),
129    /// The global type registry's [`heapless::FnvIndexMap`] cap was
130    /// reached. Bump `NROS_CYCLONEDDS_MAX_TYPES` and rebuild. Surfaces
131    /// from [`crate::type_registry::TypeRegistry::get_or_build`] —
132    /// kept on `BuildError` so the registry can share one error type.
133    RegistryFull,
134}
135
136impl BuildError {
137    fn from_bridge(code: c_int) -> Self {
138        match code {
139            x if x == BridgeError::NestedDepthExceeded as i32 => Self::NestedDepthExceeded,
140            x if x == BridgeError::UnsupportedFieldType as i32 => Self::UnsupportedFieldType,
141            x if x == BridgeError::NullPointer as i32 => Self::InvalidTypeName,
142            x if x == BridgeError::EmptySchema as i32 => Self::EmptySchema,
143            other => Self::DdsError(other),
144        }
145    }
146}

157/// Opaque pointer to a Cyclone DDS `dds_topic_descriptor_t`. Returned
158/// by the bridge; consumed by `dds_create_topic`. Treated as
159/// `*const c_void` throughout this crate so the Rust side never needs
160/// the (absent) `cyclonedds-sys` bindgen output.
161pub type DescriptorPtr = *const c_void;

163impl DescriptorBuilder {
164    /// Build a descriptor for `M` and return the resulting pointer.
165    pub fn build<M: Message>() -> Result<DescriptorPtr, BuildError> {
166        Self::build_raw(M::TYPE_NAME, M::FIELDS)
167    }
168
169    /// Build a descriptor from a raw `(type_name, fields)` pair. Used
170    /// internally by [`Self::build`]; also useful for tests that don't
171    /// want to gin up a full `impl Message`.
172    ///
173    /// # Service request/reply prefix injection (Phase 212.K.7 fix)
174    ///
175    /// When `type_name` matches a service request / reply shape — i.e.
176    /// the basename ends with `_Request`, `_Response`, or `_Reply` —
177    /// the descriptor synthesised here gets the 16-byte
178    /// `cdds_request_header_t` prefix (`rmw_writer_guid: u64`,
179    /// `rmw_sequence_number: i64`) injected at offset 0, with every
180    /// caller-supplied field offset shifted by 16. This mirrors stock
181    /// `rmw_cyclonedds_cpp`'s idlc-emitted Request/Response descriptors
182    /// (see `service.cpp` doc block) and is what `service_send_request_raw`
183    /// / `service_try_recv_request` already assume on the C++ wire side
184    /// when they prepend / split the 16-byte header. The user-facing
185    /// `Serialize` / `Deserialize` impls intentionally still serialise
186    /// only user fields — the wire framing (`build_wire_with_header`)
187    /// supplies the 16-byte prefix bytes; the descriptor needs the
188    /// matching field declarations so Cyclone's
189    /// `dds_stream_{read,write}_sample` round-trip preserves the user
190    /// payload that follows the header.
191    ///
192    /// Without this prefix the synthesised descriptor's `m_size = 16`
193    /// (or the size of just the user fields) and Cyclone silently
194    /// truncates the wire CDR's user payload during the take/write
195    /// reserialise pass — manifesting as a server-side `wire_len`
196    /// equal to encap+header with no user bytes (Phase 212.K.7 root
197    /// cause: native-Rust-Cyclone service requests timed out because
198    /// the server received only the 16-byte header, never the `a`/`b`
199    /// payload).
200    pub fn build_raw(
201        type_name: &'static str,
202        fields: &'static [Field],
203    ) -> Result<DescriptorPtr, BuildError> {
204        if type_name.is_empty() {
205            return Err(BuildError::EmptyTypeName);
206        }
207        if fields.is_empty() {
208            return Err(BuildError::EmptySchema);
209        }
210
211        let prepend_header = is_service_request_or_reply(type_name);
212        let header_field_count = if prepend_header {
213            SERVICE_HEADER_FIELDS.len()
214        } else {
215            0
216        };
217        let total_fields = fields.len() + header_field_count;
218        if total_fields > MAX_FIELDS {
219            return Err(BuildError::FieldsOverflow);
220        }
221
222        // The C bridge consumes NUL-terminated `*const c_char`s.
223        // Rust string literals (and codegen-emitted `&'static str`
224        // constants) are NOT guaranteed NUL-terminated, so we
225        // normalise into fixed-size on-stack buffers here. Inputs
226        // that already happen to be NUL-terminated (one trailing
227        // `\0`) are accepted by trimming the terminator before
228        // copying — this also matches future codegen that emits
229        // `c"…"` literals via `CStr::to_bytes`.
230        let mut type_name_buf = [0u8; TYPE_NAME_BUF_LEN];
231        copy_to_buf(type_name, &mut type_name_buf)?;
232
233        // Flatten the schema into bounded stack arrays.
234        let mut walker = SchemaWalker::new();
235        let mut field_descs = [const { NrosFieldDescriptor::zero() }; MAX_FIELDS];
236        // Per-field name buffers + per-nested-kind type-name buffers.
237        // Both bounded by their respective static caps to keep the
238        // stack footprint deterministic. Per-name slot = 64 B (ROS
239        // field names ≤ 32 char; mangled Cyclone type names ≤ ~60).
240        // Total: MAX_FIELDS*64 + MAX_KINDS*64 = (64 + 256)*64 = 20 KiB
241        // worst case at default sizing — large for an embedded stack
242        // but bounded; bump NROS_CYCLONEDDS_MAX_{FIELDS,KINDS} only
243        // if your schemas exceed the defaults.
244        let mut field_names = [const { [0u8; NAME_SLOT_LEN] }; MAX_FIELDS];
245        let mut nested_names = [const { [0u8; NAME_SLOT_LEN] }; MAX_KINDS];
246
247        // 1. Inject the 16-byte cdds_request_header_t prefix if the
248        //    type is a service request / reply (see method docs).
249        if prepend_header {
250            for (i, hf) in SERVICE_HEADER_FIELDS.iter().enumerate() {
251                copy_to_buf(hf.name, &mut field_names[i])?;
252                let kind_idx = walker.push_field_type(&hf.ty, &mut nested_names, 0)? as u32;
253                field_descs[i] = NrosFieldDescriptor {
254                    name: field_names[i].as_ptr() as *const c_char,
255                    offset: hf.offset as u32,
256                    kind: kind_idx,
257                };
258            }
259        }
260
261        // 2. Caller-supplied user fields. For service request/reply types
262        //    we override the codegen-supplied `f.offset` with synthetic
263        //    CDR-walk-order offsets (Phase 212.K.7.4.e).
264        //
265        //    Why: codegen-generated envelope structs (e.g.
266        //    `Fibonacci_SendGoal_Response { accepted: bool, stamp: Time }`)
267        //    are emitted without `#[repr(C)]`. Rust's default `repr(Rust)`
268        //    is free to reorder fields for size — so `offset_of!(.., accepted)`
269        //    can land at 8 (after `stamp`) instead of 0. The runtime wire
270        //    framing for `_SendGoal_*` / `_GetResult_*` in
271        //    `service.cpp::{write_typed, take_typed_wire}` uses raw memcpy
272        //    between the wire CDR (post-encap, post-header) and the typed
273        //    sample, treating the sample as CDR-ordered. That contract
274        //    breaks when Cyclone reads/writes fields at the
275        //    Rust-reordered positions.
276        //
277        //    For service request/reply types the sample memory is
278        //    bridge-internal (the runtime only ever sees the wire CDR), so
279        //    we're free to pick a layout. Using CDR-walk-order offsets
280        //    matches the memcpy paths and round-trips correctly through
281        //    `dds_stream_{read,write}_sample`. The 16-byte header sits at
282        //    offsets 0/8 (already CDR-aligned) so user fields start at 16.
283        //
284        //    For non-service types (plain pub/sub) we keep the codegen
285        //    offsets — those exercise the regular `write_typed` /
286        //    `take_typed_wire` paths that round-trip via Cyclone CDR
287        //    only; consistency between read and write is all that
288        //    matters, and the existing K.7.7 pub/sub e2e proves the
289        //    pattern works there even with `repr(Rust)` codegen.
290        let mut cdr_cursor = if prepend_header {
291            SERVICE_HEADER_BYTES
292        } else {
293            0
294        };
295        for (i, f) in fields.iter().enumerate() {
296            let slot = header_field_count + i;
297            copy_to_buf(f.name, &mut field_names[slot])?;
298            let kind_idx = walker.push_field_type(&f.ty, &mut nested_names, 0)? as u32;
299            let synthetic_offset = if prepend_header {
300                let align = cdr_align_of(&f.ty);
301                cdr_cursor = (cdr_cursor + align - 1) & !(align - 1);
302                let start = cdr_cursor;
303                cdr_cursor = cdr_cursor.saturating_add(cdr_size_of(&f.ty));
304                start
305            } else {
306                f.offset
307            };
308            field_descs[slot] = NrosFieldDescriptor {
309                name: field_names[slot].as_ptr() as *const c_char,
310                offset: synthetic_offset as u32,
311                kind: kind_idx,
312            };
313        }
314
315        let mut err_code: c_int = 0;
316        let descriptor = unsafe {
317            nros_cyclonedds_build_descriptor_from_schema(
318                type_name_buf.as_ptr() as *const c_char,
319                field_descs.as_ptr(),
320                total_fields as u32,
321                walker.kinds.as_ptr(),
322                walker.kind_count as u32,
323                &mut err_code,
324            )
325        };
326
327        if descriptor.is_null() {
328            return Err(BuildError::from_bridge(err_code));
329        }
330        Ok(descriptor)
331    }
332}

334// ── CDR layout helpers (Phase 212.K.7.4.e) ─────────────────────────────
335//
336// Compute the CDR-walk-order (size, alignment) for a `FieldType` so
337// `build_raw` can lay out the bridge-internal typed sample in CDR order
338// for service request/reply types whose Rust struct codegen lacks
339// `#[repr(C)]` and may therefore reorder fields.
340//
341// These mirror Cyclone's `primitive_size_align` in
342// `bridge/dynamic_type_builder.cpp` plus the conservative
343// placeholders used there for nested / sequence / string fields:
344//   * Nested struct: 16 B / align 8 (placeholder — matches the bridge's
345//     `compute_struct_size` branch).
346//   * Sequence / bounded sequence / string / bounded string: 16 B /
347//     align 8 (matches `kSeqSize`/`kStringSize` in the bridge).
348//   * Array `T[N]`: N * size_of(T), aligned to align_of(T).
349
350const fn cdr_size_of(ty: &FieldType) -> usize {
351    match ty {
352        FieldType::Bool | FieldType::Uint8 | FieldType::Int8 => 1,
353        FieldType::Uint16 | FieldType::Int16 => 2,
354        FieldType::Uint32 | FieldType::Int32 | FieldType::Float32 => 4,
355        FieldType::Uint64 | FieldType::Int64 | FieldType::Float64 => 8,
356        FieldType::String
357        | FieldType::BoundedString(_)
358        | FieldType::WString
359        | FieldType::BoundedWString(_)
360        | FieldType::Sequence(_)
361        | FieldType::BoundedSequence(_, _) => 16,
362        FieldType::Nested(n) => cdr_struct_size(n),
363        FieldType::Array(n, inner) => *n * cdr_size_of(inner),
364    }
365}
366
367const fn cdr_align_of(ty: &FieldType) -> usize {
368    match ty {
369        FieldType::Bool | FieldType::Uint8 | FieldType::Int8 => 1,
370        FieldType::Uint16 | FieldType::Int16 => 2,
371        FieldType::Uint32 | FieldType::Int32 | FieldType::Float32 => 4,
372        FieldType::Uint64 | FieldType::Int64 | FieldType::Float64 => 8,
373        FieldType::String
374        | FieldType::BoundedString(_)
375        | FieldType::WString
376        | FieldType::BoundedWString(_)
377        | FieldType::Sequence(_)
378        | FieldType::BoundedSequence(_, _) => 8,
379        FieldType::Nested(n) => cdr_struct_align(n),
380        FieldType::Array(_, inner) => cdr_align_of(inner),
381    }
382}

384const fn cdr_struct_align(n: &NestedType) -> usize {
385    let mut max_align = 1usize;
386    let mut i = 0;
387    while i < n.fields.len() {
388        let a = cdr_align_of(&n.fields[i].ty);
389        if a > max_align {
390            max_align = a;
391        }
392        i += 1;
393    }
394    max_align
395}
396
397const fn cdr_struct_size(n: &NestedType) -> usize {
398    let mut cursor = 0usize;
399    let mut max_align = 1usize;
400    let mut i = 0;
401    while i < n.fields.len() {
402        let a = cdr_align_of(&n.fields[i].ty);
403        let s = cdr_size_of(&n.fields[i].ty);
404        if a > max_align {
405            max_align = a;
406        }
407        // align cursor up to `a`.
408        cursor = (cursor + a - 1) & !(a - 1);
409        cursor = cursor + s;
410        i += 1;
411    }
412    // pad up to struct alignment.
413    (cursor + max_align - 1) & !(max_align - 1)
414}

416// ── Service request / reply CDR header (Phase 212.K.7) ─────────────────
417//
418// Stock `rmw_cyclonedds_cpp` idlc-generated request/reply structs carry
419// a 16-byte `cdds_request_header_t` prefix:
420//
421//     uint64_t rmw_writer_guid;       // lower 8 bytes of RTPS GUID
422//     int64_t  rmw_sequence_number;   // monotonic per-client
423//     /* user fields ... */
424//
425// (See `packages/dds/nros-rmw-cyclonedds/src/service.cpp` top-of-file
426// doc block + `cdds_request_header_t` in upstream
427// `rmw_cyclonedds_cpp/src/serdata.hpp:73-77`.)
428//
429// The Rust runtime descriptor builder injects matching fields into
430// every Request/Reply descriptor so Cyclone's typed sample I/O
431// (`dds_stream_{read,write}_sample`) preserves the user payload that
432// follows the prefix. The user-facing `Serialize`/`Deserialize` impls
433// still only round-trip user fields — the prefix bytes come from
434// `build_wire_with_header` / `split_wire_header` in `service.cpp`.
435
436/// Total byte size of the synthesised `cdds_request_header_t` prefix.
437pub const SERVICE_HEADER_BYTES: usize = 16;
438
439/// The two synthesised header fields, in order.
440///
441/// `offset` is absolute within the typed sample — `rmw_writer_guid` at
442/// 0, `rmw_sequence_number` at 8. Total prefix size 16 B.
443const SERVICE_HEADER_FIELDS: &[Field] = &[
444    Field {
445        name: "rmw_writer_guid",
446        ty: FieldType::Uint64,
447        offset: 0,
448    },
449    Field {
450        name: "rmw_sequence_number",
451        ty: FieldType::Int64,
452        offset: 8,
453    },
454];
455
456/// Returns `true` if `type_name` (in `pkg/srv/<Svc>_Request` /
457/// `_Response` / `_Reply` slash-form *or* the
458/// `pkg::srv::dds_::<Svc>_Request_` mangled form) names a service
459/// request or reply type.
460///
461/// The slash form is the one [`Message::TYPE_NAME`] is documented to
462/// carry; the mangled form is checked defensively in case codegen
463/// emits the wire-form directly (e.g. mirroring `RosMessage::TYPE_NAME`).
464fn is_service_request_or_reply(type_name: &str) -> bool {
465    // Normalise: trim an optional trailing NUL (codegen may include
466    // a NUL terminator in the `&'static str`), then trim a trailing
467    // `_` (mangled Cyclone names end in `_`).
468    let mut s = type_name.as_bytes();
469    while let Some(b'\0') = s.last() {
470        s = &s[..s.len() - 1];
471    }
472    while let Some(b'_') = s.last() {
473        s = &s[..s.len() - 1];
474    }
475    ends_with_ignore_case(s, b"_Request")
476        || ends_with_ignore_case(s, b"_Response")
477        || ends_with_ignore_case(s, b"_Reply")
478}
479
480/// Case-sensitive `ends_with` for byte slices (avoid pulling
481/// `str::ends_with` since `s` is already a `&[u8]`).
482fn ends_with_ignore_case(s: &[u8], suffix: &[u8]) -> bool {
483    if s.len() < suffix.len() {
484        return false;
485    }
486    let tail = &s[s.len() - suffix.len()..];
487    tail == suffix
488}

514/// Internal: flattens `&'static FieldType` recursion into a
515/// bounded `kinds[]` table.
516struct SchemaWalker {
517    kinds: [NrosFieldKindDescriptor; MAX_KINDS],
518    kind_count: usize,
519}
520
521impl SchemaWalker {
522    fn new() -> Self {
523        Self {
524            kinds: [const { NrosFieldKindDescriptor::zero() }; MAX_KINDS],
525            kind_count: 0,
526        }
527    }
528
529    /// Append a kind entry for `ty`, recursing into nested types.
530    /// Returns the index in `kinds[]` of the appended entry.
531    fn push_field_type(
532        &mut self,
533        ty: &FieldType,
534        nested_names: &mut [[u8; NAME_SLOT_LEN]; MAX_KINDS],
535        depth: usize,
536    ) -> Result<usize, BuildError> {
537        if depth >= MAX_NESTED_DEPTH {
538            return Err(BuildError::NestedDepthExceeded);
539        }
540        if self.kind_count >= MAX_KINDS {
541            return Err(BuildError::KindsOverflow);
542        }
543        let my_idx = self.kind_count;
544        self.kind_count += 1;
545        // Pre-fill with the simple primitives; complex variants
546        // overwrite the entry below.
547        self.kinds[my_idx] = match ty {
548            FieldType::Bool => simple(0),
549            FieldType::Uint8 => simple(1),
550            FieldType::Int8 => simple(2),
551            FieldType::Uint16 => simple(3),
552            FieldType::Int16 => simple(4),
553            FieldType::Uint32 => simple(5),
554            FieldType::Int32 => simple(6),
555            FieldType::Uint64 => simple(7),
556            FieldType::Int64 => simple(8),
557            FieldType::Float32 => simple(9),
558            FieldType::Float64 => simple(10),
559            FieldType::String => simple(11),
560            FieldType::WString => simple(12),
561            FieldType::BoundedString(n) => with_bound(13, *n as u32),
562            FieldType::BoundedWString(n) => with_bound(14, *n as u32),
563            FieldType::Nested(_)
564            | FieldType::Array(_, _)
565            | FieldType::Sequence(_)
566            | FieldType::BoundedSequence(_, _) => NrosFieldKindDescriptor::zero(),
567        };
568
569        match ty {
570            FieldType::Nested(nested) => {
571                let NestedType { type_name, fields } = *nested;
572                // Pack `type_name` into a stack-owned NUL-terminated
573                // buffer slot (one per kind index) so the bridge can
574                // read it.
575                copy_to_buf(type_name, &mut nested_names[my_idx])?;
576                // Recurse: each child field's `FieldType` becomes an
577                // entry in `kinds[]`. We record `inner` as the index
578                // of the first child; the bridge walks `bound` (=
579                // fields.len()) children sequentially.
580                let first_child = self.kind_count as u32;
581                for child in fields.iter() {
582                    self.push_field_type(&child.ty, nested_names, depth + 1)?;
583                }
584                self.kinds[my_idx] = NrosFieldKindDescriptor {
585                    kind: 15,
586                    _pad: [0; 3],
587                    bound: fields.len() as u32,
588                    inner: first_child,
589                    nested_name: nested_names[my_idx].as_ptr() as *const c_char,
590                };
591            }
592            FieldType::Array(n, inner) => {
593                let inner_idx = self.push_field_type(inner, nested_names, depth + 1)? as u32;
594                self.kinds[my_idx] = NrosFieldKindDescriptor {
595                    kind: 16,
596                    _pad: [0; 3],
597                    bound: *n as u32,
598                    inner: inner_idx,
599                    nested_name: core::ptr::null(),
600                };
601            }
602            FieldType::Sequence(inner) => {
603                let inner_idx = self.push_field_type(inner, nested_names, depth + 1)? as u32;
604                self.kinds[my_idx] = NrosFieldKindDescriptor {
605                    kind: 17,
606                    _pad: [0; 3],
607                    bound: 0,
608                    inner: inner_idx,
609                    nested_name: core::ptr::null(),
610                };
611            }
612            FieldType::BoundedSequence(n, inner) => {
613                let inner_idx = self.push_field_type(inner, nested_names, depth + 1)? as u32;
614                self.kinds[my_idx] = NrosFieldKindDescriptor {
615                    kind: 18,
616                    _pad: [0; 3],
617                    bound: *n as u32,
618                    inner: inner_idx,
619                    nested_name: core::ptr::null(),
620                };
621            }
622            // Primitives + strings already filled in above.
623            _ => {}
624        }
625
626        Ok(my_idx)
627    }
628}

650/// Copy `s` into `buf` and NUL-terminate.
651///
652/// Accepts both plain `&str` (no trailing NUL) and pre-NUL-terminated
653/// strings (one trailing `\0`, matching `c"…".to_bytes_with_nul()`).
654/// Embedded NULs (other than a single trailing one) are rejected with
655/// [`BuildError::InvalidTypeName`].
656fn copy_to_buf(s: &str, buf: &mut [u8]) -> Result<(), BuildError> {
657    let mut bytes = s.as_bytes();
658    if let Some((&last, rest)) = bytes.split_last()
659        && last == 0
660    {
661        // Already NUL-terminated; strip the trailing NUL before the
662        // embedded-NUL scan.
663        bytes = rest;
664    }
665    if bytes.len() + 1 > buf.len() {
666        return Err(BuildError::InvalidTypeName);
667    }
668    for (i, b) in bytes.iter().enumerate() {
669        if *b == 0 {
670            return Err(BuildError::InvalidTypeName);
671        }
672        buf[i] = *b;
673    }
674    buf[bytes.len()] = 0;
675    Ok(())
676}

918    /// Phase 212.K.7.4.e — codegen-emitted service request/reply types
919    /// like `Fibonacci_SendGoal_Response { accepted: bool, stamp: Time }`
920    /// are NOT `#[repr(C)]`; Rust's default `repr(Rust)` reorders fields
921    /// for size — `offset_of!(.., accepted)` lands at 8, not 0. The
922    /// runtime memcpy paths in `service.cpp::{write_typed,
923    /// take_typed_wire}` for `_SendGoal_*` / `_GetResult_*` treat the
924    /// typed sample as CDR-ordered, so the descriptor MUST place
925    /// `accepted` at the post-header CDR position (16), not whatever
926    /// Rust picked.

928    /// This test mirrors the codegen output exactly (mis-ordered Rust
929    /// offsets) and asserts `build_raw` overrides them with
930    /// CDR-walk-order synthetic offsets (16 for `accepted`, 20 for
931    /// `stamp`). Pre-K.7.4.e it would report (24, 16) — Rust's
932    /// reordered struct shifted by 16.
933    #[test]
934    fn service_response_uses_cdr_walk_order_offsets() {
935        use crate::bridge::test_stub::{LAST_FIELD_COUNT, LAST_FIELDS};
936        use core::sync::atomic::Ordering;
937
938        // Mirror the `Time` nested struct (matches builtin_interfaces/msg/Time).
939        static TIME_FIELDS: &[Field] = &[
940            Field {
941                name: "sec\0",
942                ty: FieldType::Int32,
943                offset: 0,
944            },
945            Field {
946                name: "nanosec\0",
947                ty: FieldType::Uint32,
948                offset: 4,
949            },
950        ];
951        static TIME_NESTED: NestedType = NestedType {
952            type_name: "builtin_interfaces/msg/Time",
953            fields: TIME_FIELDS,
954        };
955
956        // Mirror `Fibonacci_SendGoal_Response { accepted: bool, stamp: Time }`
957        // codegen output: declaration order [accepted, stamp], but Rust's
958        // `repr(Rust)` reorders `stamp` (align 4, size 8) before `accepted`
959        // (align 1) → offsets (8, 0). We feed those reordered offsets to
960        // build_raw; the K.7.4.e synthetic-offset logic must ignore them.
961        static REORDERED_FIELDS: &[Field] = &[
962            Field {
963                name: "accepted\0",
964                ty: FieldType::Bool,
965                offset: 8, // Rust's `repr(Rust)` reordered position.
966            },
967            Field {
968                name: "stamp\0",
969                ty: FieldType::Nested(&TIME_NESTED),
970                offset: 0, // Rust's `repr(Rust)` reordered position.
971            },
972        ];
973
974        // Build a Response-suffixed descriptor.
975        let ptr = DescriptorBuilder::build_raw(
976            "example_interfaces/srv/Fake_SendGoal_Response",
977            REORDERED_FIELDS,
978        )
979        .expect("build send_goal response");
980        assert!(!ptr.is_null());
981
982        // 2 header fields + 2 user fields = 4 captured.
983        let n = LAST_FIELD_COUNT.load(Ordering::SeqCst);
984        assert_eq!(n, 4, "expected 2 header + 2 user fields");
985
986        // SAFETY: the bridge stub has returned, so the slots are no
987        // longer being written.
988        let captured: [(u32, u32); 4] = unsafe {
989            [
990                *LAST_FIELDS[0].0.get(),
991                *LAST_FIELDS[1].0.get(),
992                *LAST_FIELDS[2].0.get(),
993                *LAST_FIELDS[3].0.get(),

996

1002        // the K.7.4.e fix must place `accepted` at CDR offset 16 and
1003        // `stamp` aligned-to-4 right after = 20.
1004        assert_eq!(
1005            captured[2].0, 16,
1006            "accepted MUST land at post-header CDR offset 16 (got {})",
1007            captured[2].0
1008        );
1009        assert_eq!(
1010            captured[3].0, 20,
1011            "stamp MUST land at CDR-walk-aligned offset 20 (got {})",
1012            captured[3].0
