1//! C++ bridge declarations.
2//!
3//! The Cyclone DDS C API (and the C++ headers `dds/ddsi/ddsi_dynamic_type*`)
4//! are not exposed through any in-tree Rust `*-sys` crate — `cyclonedds-sys`
5//! only ships the build-time link recipe, no bindgen-generated types. To
6//! keep the K.7.4 descriptor builder host-language-honest, the actual walk
7//! lives on the C++ side in `bridge/dynamic_type_builder.cpp` and exposes
8//! a single `extern "C"` entry that the Rust shim calls with the schema
9//! laid out for ABI stability.
10//!
11//! The ABI mirrors [`nros_serdes::schema`]: each `FieldType` variant
12//! maps to a tag from [`FieldKind`] plus an `inner` index (for the
13//! recursive variants — `Array` / `Sequence` / `BoundedSequence` /
14//! `Nested`) into a shared `kinds[]` table that flattens the
15//! `&'static FieldType` references the schema uses.
16//!
17//! The C++ TU is responsible for:
18//!
19//! 1. Walking the flattened `FieldDescriptor[]` + `FieldKindDescriptor[]`
20//!    arrays.
21//! 2. Calling Cyclone's `ddsi_dynamic_type_*` constructors to build
22//!    nested types bottom-up.
23//! 3. Finalising into a `dds_topic_descriptor_t` allocated from
24//!    `ddsrt_malloc`.
25//! 4. Returning the pointer (or NULL on error, with `*out_err` set to
26//!    a [`BridgeError`] code).
27//!
28//! See `bridge/dynamic_type_builder.cpp` for the implementation.
29
30use core::ffi::{c_char, c_int};
31
32/// Tags every recognised [`nros_serdes::schema::FieldType`] variant in
33/// a stable C ABI representation. Mirrors `enum NrosFieldKind` in
34/// `bridge/dynamic_type_builder.cpp`.
35#[repr(u8)]
36#[derive(Debug, Clone, Copy, PartialEq, Eq)]
37pub enum FieldKind {
38    Bool = 0,
39    Uint8 = 1,
40    Int8 = 2,
41    Uint16 = 3,
42    Int16 = 4,
43    Uint32 = 5,
44    Int32 = 6,
45    Uint64 = 7,
46    Int64 = 8,
47    Float32 = 9,
48    Float64 = 10,
49    String = 11,
50    WString = 12,
51    BoundedString = 13,
52    BoundedWString = 14,
53    /// `Nested(NestedType { type_name, fields })`. The `bound` slot
54    /// carries `fields.len()`; `inner` indexes the first child kind in
55    /// the flattened `kinds[]` table.
56    Nested = 15,
57    /// `Array(N, inner)` — `bound = N`, `inner` indexes `kinds[]`.
58    Array = 16,
59    /// `Sequence(inner)` — `inner` indexes `kinds[]`, `bound = 0`.
60    Sequence = 17,
61    /// `BoundedSequence(N, inner)` — `bound = N`.
62    BoundedSequence = 18,
63}
64
65/// Mirror of `struct NrosFieldKindDescriptor` in
66/// `bridge/dynamic_type_builder.cpp`.
67///
68/// Together the `kind` + `bound` + `inner` triple captures every
69/// shape of [`nros_serdes::schema::FieldType`]:
70///
71/// | FieldType | kind | bound | inner | nested_name |
72/// |---|---|---|---|---|
73/// | `Bool` … `Float64` | matching tag | 0 | 0 | NULL |
74/// | `String` / `WString` | `String` / `WString` | 0 | 0 | NULL |
75/// | `BoundedString(N)` / `BoundedWString(N)` | matching tag | N | 0 | NULL |
76/// | `Nested(NestedType)` | `Nested` | child fields[].len() | first child idx in kinds[] | type_name |
77/// | `Array(N, inner)` | `Array` | N | inner kind idx | NULL |
78/// | `Sequence(inner)` | `Sequence` | 0 | inner kind idx | NULL |
79/// | `BoundedSequence(N, inner)` | `BoundedSequence` | N | inner kind idx | NULL |
80///
81/// The flattened layout sidesteps Rust's lifetime tracking for the
82/// `&'static FieldType` recursion the schema uses internally.
83#[repr(C)]
84#[derive(Debug, Clone, Copy)]
85pub struct NrosFieldKindDescriptor {
86    pub kind: u8,
87    pub _pad: [u8; 3],
88    pub bound: u32,
89    pub inner: u32,
90    /// NUL-terminated nested type-name pointer (only valid for
91    /// `kind == Nested`; otherwise NULL).
92    pub nested_name: *const c_char,
93}
94
95// SAFETY: only contains POD + a `*const c_char` pointer that lives in
96// `.rodata` (every name in the schema is a `&'static str`). The C++
97// bridge reads it from the calling thread; the registry mutex
98// serialises Rust-side access. Cross-thread send/sync is therefore
99// safe for the descriptor table itself.
100unsafe impl Send for NrosFieldKindDescriptor {}
101unsafe impl Sync for NrosFieldKindDescriptor {}
102
103/// Mirror of `struct NrosFieldDescriptor` — one entry per field of
104/// the top-level message.
105#[repr(C)]
106#[derive(Debug, Clone, Copy)]
107pub struct NrosFieldDescriptor {
108    /// NUL-terminated field name (points into `.rodata`).
109    pub name: *const c_char,
110    /// Byte offset of the field within the host struct.
111    pub offset: u32,
112    /// Index into the flattened `kinds[]` table.
113    pub kind: u32,
114}
115
116unsafe impl Send for NrosFieldDescriptor {}
117unsafe impl Sync for NrosFieldDescriptor {}
118
119/// Bridge return codes. Mirrors `enum NrosBridgeError` in
120/// `bridge/dynamic_type_builder.cpp`.
121///
122/// Negative codes from the underlying `dds_return_t` API pass through
123/// unchanged (Cyclone uses `< 0` for failure).
124#[repr(i32)]
125#[derive(Debug, Clone, Copy, PartialEq, Eq)]
126pub enum BridgeError {
127    /// Recursion limit exceeded while walking [`FieldKind::Nested`].
128    NestedDepthExceeded = -1001,
129    /// One of the schema's `FieldType` variants is recognised but
130    /// not yet wired through to Cyclone's dynamic-type API (e.g.
131    /// `WString` on a build where Cyclone's wide-string variant is
132    /// disabled, or `BoundedSequence` on a Cyclone older than the
133    /// dynamic-bounded-sequence API).
134    UnsupportedFieldType = -1002,
135    /// A `*const c_char` was unexpectedly NULL.
136    NullPointer = -1003,
137    /// `count == 0` — empty schema rejected.
138    EmptySchema = -1004,
139}

141unsafe extern "C" {
142    /// C++ bridge entry — see `bridge/dynamic_type_builder.cpp`.
143    ///
144    /// # Arguments
145    ///
146    /// * `type_name` — NUL-terminated ROS type name in
147    ///   `package/msg/Type` form (the bridge mangles it to the
148    ///   `package::msg::dds_::Type_` Cyclone shape internally).
149    /// * `fields` / `field_count` — top-level field descriptors.
150    /// * `kinds` / `kind_count` — flattened recursive-kind table.
151    /// * `out_err` — set to a [`BridgeError`] (or a Cyclone
152    ///   `dds_return_t`) on NULL return; left untouched on success.
153    ///
154    /// # Returns
155    ///
156    /// Non-NULL `*const dds_topic_descriptor_t` (opaque to Rust) on
157    /// success. NULL on failure — read `*out_err` for the reason.
158    ///
159    /// # Safety
160    ///
161    /// All pointers must be valid for the duration of the call. The
162    /// returned pointer is allocated by Cyclone (`ddsrt_malloc`) and
163    /// is owned by the global registry once stored — callers must not
164    /// free it.
165    pub fn nros_cyclonedds_build_descriptor_from_schema(
166        type_name: *const c_char,
167        fields: *const NrosFieldDescriptor,
168        field_count: u32,
169        kinds: *const NrosFieldKindDescriptor,
170        kind_count: u32,
171        out_err: *mut c_int,
172    ) -> *const core::ffi::c_void;
173
174    /// Register a built descriptor with the existing C++
175    /// `descriptors.cpp` registry so that the C++ pub/sub paths can
176    /// find it via `find_descriptor(type_name)`.
177    ///
178    /// Mirrors the C entry already used by the static-init `_register.c`
179    /// codegen TUs.
180    pub fn nros_rmw_cyclonedds_register_descriptor(
181        type_name: *const c_char,
182        descriptor: *const core::ffi::c_void,
183    );
184}

186// ── Test-only stub ──────────────────────────────────────────────────
187//
188// `cargo test -p nros-rmw-cyclonedds` runs on a hosted toolchain that
189// has no Cyclone DDS C library available. Stub both bridge entries
190// out under `#[cfg(test)]` so the registry + builder logic compiles
191// and exercises end-to-end without dragging in `libddsc`.
192
193#[cfg(any(test, feature = "bridge-stub"))]
194pub mod test_stub {
195    use core::{
196        cell::UnsafeCell,
197        ffi::{c_char, c_int, c_void},
198        sync::atomic::{AtomicI32, AtomicUsize, Ordering},
199    };
200
201    use super::{NrosFieldDescriptor, NrosFieldKindDescriptor};
202
203    /// Set to a non-zero `BridgeError` code to force the stub to
204    /// fail the next call (used by `BuildError` mapping tests).
205    pub static FORCED_ERROR: AtomicI32 = AtomicI32::new(0);
206
207    /// Increments each time the stub fakes a successful build. Let
208    /// tests assert the registry's cache-hit behaviour.
209    pub static BUILD_COUNTER: AtomicUsize = AtomicUsize::new(0);
210
211    /// Captures the last `(offset, kind)` pair per top-level field passed
212    /// to the bridge builder (Phase 212.K.7.4.e — lets unit tests assert
213    /// that service request/reply types get the CDR-walk-order synthetic
214    /// offsets that match the runtime memcpy paths in
215    /// `service.cpp::{write_typed, take_typed_wire}`).
216    pub const LAST_FIELDS_CAP: usize = 16;
217    pub static LAST_FIELD_COUNT: AtomicUsize = AtomicUsize::new(0);
218
219    /// Sync newtype around `UnsafeCell<(u32, u32)>`. Captures per-field
220    /// `(offset, kind)` from the bridge stub. Single-threaded tests only;
221    /// writers are serialised by `BUILD_COUNTER` so readers (in the test
222    /// body after `build_raw` returns) observe stable values.
223    #[repr(transparent)]
224    pub struct FieldCapture(pub UnsafeCell<(u32, u32)>);
225
226    // SAFETY: see `FieldCapture` doc comment — single-threaded test use only.
227    unsafe impl Sync for FieldCapture {}
228
229    /// `(offset, kind)` per captured field. Tests inspect after the
230    /// corresponding `build_raw` call returns.
231    pub static LAST_FIELDS: [FieldCapture; LAST_FIELDS_CAP] =
232        [const { FieldCapture(UnsafeCell::new((0u32, 0u32))) }; LAST_FIELDS_CAP];
233
234    /// Pretend-descriptor pool. We hand out unique non-NULL pointers
235    /// (the index of `STUB_DESCRIPTORS`) so the registry can store +
236    /// retrieve them without aliasing the same value.
237    static STUB_DESCRIPTORS: [u8; 64] = [0; 64];
238
239    #[unsafe(no_mangle)]
240    extern "C" fn nros_cyclonedds_build_descriptor_from_schema(
241        _type_name: *const c_char,
242        fields: *const NrosFieldDescriptor,
243        field_count: u32,
244        _kinds: *const NrosFieldKindDescriptor,
245        _kind_count: u32,
246        out_err: *mut c_int,
247    ) -> *const c_void {
248        let forced = FORCED_ERROR.swap(0, Ordering::SeqCst);
249        if forced != 0 {
250            if !out_err.is_null() {
251                unsafe { *out_err = forced };
252            }
253            return core::ptr::null();
254        }
255        // Capture field (offset, kind) for inspection by tests.
256        if !fields.is_null() {
257            let cap = LAST_FIELDS_CAP.min(field_count as usize);
258            LAST_FIELD_COUNT.store(cap, Ordering::SeqCst);
259            for (i, slot) in LAST_FIELDS.iter().enumerate().take(cap) {
260                // SAFETY: caller guarantees `fields` is valid for
261                // `field_count` reads. Stub only writes into
262                // `LAST_FIELDS` from one thread per call.
263                let f = unsafe { &*fields.add(i) };
264                unsafe {
265                    *slot.0.get() = (f.offset, f.kind);
266                }
267            }
268        }
269        let idx = BUILD_COUNTER.fetch_add(1, Ordering::SeqCst);
270        // Hand back a stable, unique non-NULL pointer per call (bounded
271        // by the size of the stub backing array so we don't pretend to
272        // exceed the test pool).
273        let bounded = idx % STUB_DESCRIPTORS.len();
274        let ptr: *const u8 = &STUB_DESCRIPTORS[bounded];
275        ptr as *const c_void
276    }
277
278    #[unsafe(no_mangle)]
279    extern "C" fn nros_rmw_cyclonedds_register_descriptor(
280        _type_name: *const c_char,
281        _descriptor: *const c_void,
282    ) {
283        // No-op: the C++ registry is irrelevant in unit tests; the
284        // Rust registry already cached the pointer.
285    }
