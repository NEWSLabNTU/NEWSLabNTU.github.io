1//! Platform-selected mutex for the global type registry.
2//!
3//! * Hosted (`target_os = "linux" | "macos" | "windows" | "freebsd" |
4//!   "netbsd" | "openbsd" | "dragonfly"`): [`spin::Mutex`]. Multi-thread,
5//!   no ISR concerns; the registry is touched on every first-use
6//!   `create_publisher` / `create_subscription`, so spin contention is
7//!   negligible (one cache-miss per topic per process).
8//! * Bare-metal (`target_os = "none"`): [`critical_section::Mutex`]
9//!   wrapped in a thin newtype that mirrors the `spin::Mutex` lock
10//!   surface. ISR-safe.
11//! * Other (e.g. RTOS targets compiled with their own `target_os`):
12//!   falls back to `spin::Mutex` — the platform's mutex layer already
13//!   adapts via `nros-platform-critical-section`.
14//!
15//! The newtype-uniform [`RegistryMutex`] alias keeps the call-site in
16//! [`crate::type_registry`] free of `#[cfg]`s.
17
18#[cfg(target_os = "none")]
19pub use cs::RegistryMutex;
20#[cfg(not(target_os = "none"))]
21pub use hosted::RegistryMutex;
22
23#[cfg(not(target_os = "none"))]
24mod hosted {
25    /// Spin lock — fine on hosted multi-thread; the registry is
26    /// touched once per topic per process.
27    pub type RegistryMutex<T> = spin::Mutex<T>;
28}
29
30#[cfg(target_os = "none")]
31mod cs {
32    use core::cell::RefCell;
33
34    /// Thin wrapper that adapts [`critical_section::Mutex`] to the
35    /// `lock() -> Guard` shape the registry uses on hosted targets.
36    ///
37    /// `critical_section` masks interrupts for the duration of the
38    /// closure body, which is fine here — descriptor-build work is
39    /// O(fields) and runs on entity creation (not in the data path).
40    pub struct RegistryMutex<T> {
41        inner: critical_section::Mutex<RefCell<T>>,
42    }
43
44    impl<T> RegistryMutex<T> {
45        pub const fn new(value: T) -> Self {
46            Self {
47                inner: critical_section::Mutex::new(RefCell::new(value)),
48            }
49        }
50
51        /// Internal accessor used by the [`super::RegistryMutexExt`]
52        /// blanket impl on this newtype. Kept `pub(super)` so callers
53        /// always go through the trait — that keeps the
54        /// `.with(|r| …)` call site shape identical between hosted
55        /// and bare-metal arms.
56        pub(super) fn with_inner<R>(&self, f: impl FnOnce(&mut T) -> R) -> R {
57            critical_section::with(|cs| {
58                let mut borrow = self.inner.borrow_ref_mut(cs);
59                f(&mut *borrow)
60            })
61        }
62    }
63}
64
65/// Adapter wrapper that gives [`spin::Mutex`] the same `.with()`
66/// shape as the bare-metal newtype. Lets `type_registry` use one
67/// call style under both `#[cfg]` arms.
68pub trait RegistryMutexExt<T> {
69    fn with<R>(&self, f: impl FnOnce(&mut T) -> R) -> R;
70}
71
72#[cfg(not(target_os = "none"))]
73impl<T> RegistryMutexExt<T> for spin::Mutex<T> {
74    fn with<R>(&self, f: impl FnOnce(&mut T) -> R) -> R {
75        let mut guard = self.lock();
76        f(&mut *guard)
77    }
78}
79
80#[cfg(target_os = "none")]
81impl<T> RegistryMutexExt<T> for cs::RegistryMutex<T> {
82    fn with<R>(&self, f: impl FnOnce(&mut T) -> R) -> R {
83        cs::RegistryMutex::with_inner(self, f)
84    }
85}
