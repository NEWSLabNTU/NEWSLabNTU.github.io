1//! Platform capability sub-traits for nros.
2//!
3//! This crate exists to break the dependency cycle between
4//! `nros-platform` (which depends on every platform crate via its
5//! feature-gated `ConcretePlatform` resolver) and the platform crates
6//! themselves (which need to implement these traits on their ZSTs).
7//! It contains only trait definitions — no implementations, no
8//! dependencies — so platform crates can take a build-time dep on it
9//! without creating a cycle back through `nros-platform`.
10//!
11//! `nros-platform` re-exports everything from this crate, so downstream
12//! code that writes `use nros_platform::PlatformClock;` continues to
13//! work unchanged.
14//!
15//! Each trait covers an independent system capability. Platform
16//! implementations pick which traits to implement based on what the
17//! hardware/RTOS provides. RMW shim crates declare trait bounds for
18//! the capabilities they need.
19
20#![no_std]
21//!
22//! # Naming convention
23//!
24//! Method names drop redundant prefixes when the trait name already
25//! supplies the namespace — e.g., `PlatformTcp::open` rather than
26//! `PlatformTcp::tcp_open`. Dispatch is always through a qualified
27//! path (`<ConcretePlatform as PlatformTcp>::open(...)`), so
28//! trait-to-trait collisions (PlatformTcp::open vs PlatformUdp::open)
29//! are disambiguated at the call site without needing a prefix on the
30//! trait method itself.
31//!
32//! Three categories still keep sub-namespace prefixes internally:
33//!
34//! * `PlatformThreading` — `mutex_*`, `condvar_*`, `task_*` because the
35//!   trait bundles three independent primitive families and unprefixed
36//!   `init` / `drop` would be ambiguous *within* the trait itself.
37//! * `PlatformUdpMulticast` — `mcast_*` because these methods have
38//!   different signatures from `PlatformUdp`'s same-name methods; keeping
39//!   the prefix makes call sites that use both traits self-documenting.
40//! * The `close` method appears on both `PlatformTcp` and
41//!   `PlatformSocketHelpers` — the first is TCP teardown, the second is
42//!   zenoh-pico's generic "shutdown + close" helper. Both live unprefixed
43//!   in their respective traits; call sites disambiguate via the
44//!   qualified path.
45//!
46//! # Status (Phase 84.F4)
47//!
48//! The platform ZSTs (`PosixPlatform`, `ZephyrPlatform`, etc.) do **not**
49//! currently implement these traits — every platform exposes its methods
50//! as *inherent* `impl Platform { fn foo() {} }` blocks, and shim crates
51//! dispatch by name match. 84.F4 migrates each platform to `impl
52//! PlatformX for Platform { fn foo() {} }` one trait at a time, with the
53//! shims switching to `<P as PlatformX>::foo()`. Until that work is
54//! complete the traits here are a target specification.
55

111// ============================================================================
112// Cooperative yield
113// ============================================================================
114
115/// Scheduler yield primitive.
116///
117/// Used inside `socket_wait_event` and similar "let another task make
118/// progress" points where the caller isn't actually waiting for I/O
119/// readability — the background read task already owns that — it just
120/// needs to relinquish the CPU so the real waiter can run.
121///
122/// Prior to Phase 77.22 each backend hand-rolled its own 1-ms busy
123/// sleep (`libc::usleep(1000)`, `vTaskDelay(1)`, `tx_thread_sleep(1)`,
124/// `k_usleep(1000)`, `select(.., 1 ms)`) — all with slightly different
125/// units and no common home.
126///
127/// **ISR-safety**: on the hosted-RTOS backends (FreeRTOS / NuttX /
128/// Zephyr / ThreadX) the underlying primitives panic or error when
129/// invoked from an ISR. Don't call `yield_now()` from an interrupt
130/// handler.
131///
132/// **Bare-metal has no real yield**: there's nothing to yield *to*.
133/// The default bare-metal impl is `core::hint::spin_loop()` (a pure
134/// CPU hint: emits `YIELD` / `PAUSE` / `WFE` depending on the arch,
135/// safe everywhere). Board crates that have armed an IRQ source may
136/// opt in to deep idle (`wfi`) via a separate `BoardIdle` hook — not
137/// part of this trait because calling `wfi` without an IRQ source
138/// deadlocks.
139pub trait PlatformYield {
140    /// Relinquish the CPU long enough for another task / thread to run.
141    ///
142    /// Non-blocking in the sleep sense — returns as soon as the
143    /// scheduler has had an opportunity to pick a different runnable.
144    fn yield_now();
145}

169/// Wall-clock / system time.
170///
171/// Used for logging timestamps and `z_time_now_as_str()`.
172/// On bare-metal without an RTC, return monotonic time or zeros.
173///
174/// The two-function `time_since_epoch_*` split (instead of returning a
175/// struct) was chosen to match the shape that zenoh-pico's C headers
176/// want across the FFI boundary — zpico-platform-shim forwards each
177/// of these directly to a `_z_time_*` symbol, so collapsing them into
178/// a Rust struct would require the shim to decompose the struct on
179/// every call.
180pub trait PlatformTime {
181    /// Returns system time in milliseconds.
182    fn time_now_ms() -> u64;
183
184    /// Seconds component of wall-clock time since the Unix epoch.
185    fn time_since_epoch_secs() -> u32;
186
187    /// Sub-second nanoseconds component of wall-clock time since the
188    /// Unix epoch (i.e. the nanosecond remainder after the seconds are
189    /// stripped; always in `0..1_000_000_000`).
190    fn time_since_epoch_nanos() -> u32;
191}

193// ============================================================================
194// Threading (multi-threaded platforms only)
195// ============================================================================
196//
197// Handle types are opaque `*mut c_void` to match the shape zenoh-pico
198// passes across the FFI boundary. The original draft had typed wrappers
199// (`TaskHandle`, `MutexHandle`, ...) but every shipped platform used
200// `*mut c_void` internally and the shim never materialised the typed
201// forms — keeping them in the trait would have required a pointless
202// cast at every impl site. (F4.1 / F4.5 decision, 2026-04-24.)
203//
204// Mutex / condvar / task method names keep their sub-namespace prefix
205// (`mutex_*`, `condvar_*`, `task_*`) because the trait bundles three
206// independent primitive families and unprefixed `init` / `drop` would
207// be ambiguous *within* the trait itself.
208
209/// Threading primitives: tasks, mutexes, and condition variables.
210///
211/// # Threading
212///
213/// Mutex / condvar operations must be safe under concurrent callers.
214/// **Recursive mutex (`mutex_rec_*`) must support same-thread
215/// re-entrancy** — zenoh-pico relies on this; a non-recursive mutex
216/// backing `mutex_rec_*` deadlocks under load.
217///
218/// # ISR-safety
219///
220/// **None of these methods are ISR-safe** on hosted RTOSes
221/// (FreeRTOS / NuttX / Zephyr / ThreadX) — the underlying primitives
222/// panic or error when invoked from an ISR. Only `core::hint::spin_loop()`
223/// in [`PlatformYield::yield_now`] is.
224///
225/// For single-threaded platforms (bare-metal), all operations should
226/// be no-ops returning `0`, except [`task_init`](Self::task_init)
227/// which should return `-1`.
228pub trait PlatformThreading {
229    // -- Tasks --
230
231    /// Spawn a new task. `task` is opaque caller-provided storage;
232    /// `attr` carries scheduling hints (priority, stack size) or
233    /// `null` for defaults; `entry` is the task entry point; `arg`
234    /// is forwarded to `entry`. Returns `0` on success, non-zero
235    /// on failure.
236    fn task_init(
237        task: *mut c_void,
238        attr: *mut c_void,
239        entry: Option<unsafe extern "C" fn(*mut c_void) -> *mut c_void>,
240        arg: *mut c_void,
241    ) -> i8;
242
243    /// Block until `task` exits. Cleans up the task storage on
244    /// success.
245    fn task_join(task: *mut c_void) -> i8;
246    /// Mark `task` as detached — its storage is reclaimed on exit
247    /// without a join.
248    fn task_detach(task: *mut c_void) -> i8;
249    /// Request `task` to terminate at the next cancellation point.
250    /// Cooperative.
251    fn task_cancel(task: *mut c_void) -> i8;
252    /// Terminate the calling task immediately. Does not return.
253    fn task_exit();
254    /// Free the task storage allocated by `task_init`.
255    fn task_free(task: *mut *mut c_void);
256
257    // -- Mutex --
258
259    /// Initialise a non-recursive mutex in caller-provided storage.
260    fn mutex_init(m: *mut c_void) -> i8;
261    /// Tear down a non-recursive mutex.
262    fn mutex_drop(m: *mut c_void) -> i8;
263    /// Lock; block if held.
264    fn mutex_lock(m: *mut c_void) -> i8;
265    /// Try to lock; non-zero return immediately if held.
266    fn mutex_try_lock(m: *mut c_void) -> i8;
267    /// Unlock; only the owning thread may call this.
268    fn mutex_unlock(m: *mut c_void) -> i8;
269
270    // -- Recursive mutex --
271
272    /// Initialise a *recursive* mutex (same-thread re-entrancy
273    /// permitted). Required by zenoh-pico.
274    fn mutex_rec_init(m: *mut c_void) -> i8;
275    /// Tear down a recursive mutex.
276    fn mutex_rec_drop(m: *mut c_void) -> i8;
277    /// Lock; re-entry from the owning thread must succeed.
278    fn mutex_rec_lock(m: *mut c_void) -> i8;
279    /// Try to lock; same re-entry semantics as `mutex_rec_lock`.
280    fn mutex_rec_try_lock(m: *mut c_void) -> i8;
281    /// Unlock; releases when the lock count returns to zero.
282    fn mutex_rec_unlock(m: *mut c_void) -> i8;
283
284    // -- Condition variables --
285
286    /// Initialise a condition variable in caller-provided storage.
287    fn condvar_init(cv: *mut c_void) -> i8;
288    /// Tear down a condition variable.
289    fn condvar_drop(cv: *mut c_void) -> i8;
290    /// Wake one waiter on the condition variable.
291    fn condvar_signal(cv: *mut c_void) -> i8;
292    /// Wake all waiters on the condition variable.
293    fn condvar_signal_all(cv: *mut c_void) -> i8;
294    /// Atomically release `m` and block on `cv`. The mutex is
295    /// re-acquired before this function returns.
296    fn condvar_wait(cv: *mut c_void, m: *mut c_void) -> i8;
297
298    /// Wait with absolute monotonic deadline (milliseconds since
299    /// the [`PlatformClock::clock_ms`] epoch). Returns non-zero on
300    /// timeout.
301    fn condvar_wait_until(cv: *mut c_void, m: *mut c_void, abstime: u64) -> i8;
302}

319// Networking — TCP

322/// TCP networking.
323///
324/// Socket and endpoint parameters are opaque `*mut c_void` pointers to
325/// platform-specific types (`_z_sys_net_socket_t`, `_z_sys_net_endpoint_t`).
326/// The shim provides correctly-sized `#[repr(C)]` wrappers whose sizes are
327/// auto-detected from C headers at build time (see Phase 80 design).
328///
329/// Read functions return `usize::MAX` on error. Send returns `usize::MAX` on error.
330///
331/// Method names are unprefixed — the trait already namespaces them. Shims
332/// dispatch via `<ConcretePlatform as PlatformTcp>::open(...)` etc.
333pub trait PlatformTcp {
334    /// Resolve address + port strings into an endpoint handle.
335    fn create_endpoint(ep: *mut c_void, address: *const u8, port: *const u8) -> i8;
336    /// Free endpoint resources.
337    fn free_endpoint(ep: *mut c_void);
338    /// Open a TCP client connection. `endpoint` is by-value (opaque bytes on stack).
339    fn open(sock: *mut c_void, endpoint: *const c_void, timeout_ms: u32) -> i8;
340    /// Open a TCP listening socket.
341    fn listen(sock: *mut c_void, endpoint: *const c_void) -> i8;
342    /// Close a TCP socket.
343    fn close(sock: *mut c_void);
344    /// Read up to `len` bytes. Returns bytes read, or `usize::MAX` on error.
345    fn read(sock: *const c_void, buf: *mut u8, len: usize) -> usize;
346    /// Read exactly `len` bytes. Returns `len` on success, `usize::MAX` on error.
347    fn read_exact(sock: *const c_void, buf: *mut u8, len: usize) -> usize;
348    /// Send `len` bytes. Returns bytes sent, or `usize::MAX` on error.
349    fn send(sock: *const c_void, buf: *const u8, len: usize) -> usize;
350}

406// libc stubs (bare-metal only)
407// ============================================================================
408
409/// Standard C library functions needed by zenoh-pico on bare-metal targets.
410///
411/// Platforms with a C runtime (RTOS, POSIX) do NOT need to implement this.
412///
413/// # Dispatch model (Phase 84.F4.6)
414///
415/// This trait is **documentary only** — it is NOT dispatched through by
416/// `zpico-platform-shim` or `xrce-platform-shim`. The C libraries resolve
417/// these symbols (`strlen`, `memcpy`, `errno`, ...) at link time directly
418/// from `#[unsafe(no_mangle)] extern "C" fn` definitions in
419/// `nros-baremetal-common`, which bare-metal platform crates pull in
420/// via the `libc-stubs` feature:
421///
422/// ```text
423///   nros-baremetal-common = { ..., features = ["libc-stubs"] }
424/// ```
425///
426/// The trait is retained in this API surface so that a future shim
427/// refactor could route libc through typed Rust methods without
428/// changing consumers. Today, implementing `PlatformLibc` on a platform
429/// ZST would be pure documentation; the actual contract — "the linker
430/// can resolve `strlen` etc." — is enforced at link time, not at
431/// compile time. No platform crate implements this trait in the
432/// current tree.
433pub trait PlatformLibc {
434    fn strlen(s: *const u8) -> usize;
435    fn strcmp(s1: *const u8, s2: *const u8) -> c_int;
436    fn strncmp(s1: *const u8, s2: *const u8, n: usize) -> c_int;
437    fn strchr(s: *const u8, c: c_int) -> *mut u8;
438    fn strncpy(dest: *mut u8, src: *const u8, n: usize) -> *mut u8;
439    fn memcpy(dest: *mut c_void, src: *const c_void, n: usize) -> *mut c_void;
440    fn memmove(dest: *mut c_void, src: *const c_void, n: usize) -> *mut c_void;
441    fn memset(dest: *mut c_void, c: c_int, n: usize) -> *mut c_void;
442    fn memcmp(s1: *const c_void, s2: *const c_void, n: usize) -> c_int;
443    fn memchr(s: *const c_void, c: c_int, n: usize) -> *mut c_void;
444    fn strtoul(nptr: *const u8, endptr: *mut *mut u8, base: c_int) -> core::ffi::c_ulong;
445    fn errno_ptr() -> *mut c_int;
446}

497// Serial (UART / PTY)
498// ============================================================================
499
500/// Serial (byte-stream) transport.
501///
502/// Used by XRCE-DDS's HDLC-framed serial transport and (once the
503/// bare-metal migration lands) by zenoh-pico's serial link layer.
504///
505/// # Handle model
506///
507/// `open()` returns a platform-defined [`Handle`](PlatformSerial::Handle)
508/// — an FD on POSIX, a port-table index on bare-metal, whatever the
509/// impl wants. Every other method takes the handle back. This lets a
510/// single platform impl service multiple concurrent devices (e.g.,
511/// `zpico-serial`'s two-port table) without the trait constraining
512/// the impl to a single active device.
513///
514/// Single-device platforms return the same handle forever and ignore
515/// it internally; `INVALID` gives a well-defined "no live handle"
516/// sentinel for shims that stash the current handle in a `static`.
517///
518/// # Path conventions
519///
520/// `path` in `open()` is platform-specific: a null-terminated UTF-8
521/// device path on POSIX (e.g., `/dev/ttyUSB0` or a PTY), or a
522/// board-defined port identifier on bare-metal (typically parsed by
523/// the platform's internal handler). Callers pass the locator string
524/// from their config unchanged; interpretation is the platform's job.
525///
526/// # I/O conventions
527///
528/// Read / write return `usize::MAX` on hard error. Read with
529/// `timeout_ms == 0` should block for a platform-chosen default;
530/// positive values are the poll/select deadline in milliseconds.
531/// Returning `0` from `read()` indicates "no data within timeout" and
532/// is **not** an error — both XRCE and zenoh-pico tolerate
533/// timeout-zero reads.
534pub trait PlatformSerial {
535    /// Platform-specific handle type. POSIX returns the FD (`i32`);
536    /// bare-metal returns a port-table index (`u8`). Must be `Copy`
537    /// so shims can stash it in a `static`.
538    type Handle: Copy;
539
540    /// Sentinel handle meaning "not a live device". Shims initialise
541    /// their cached handle to this and compare against it to detect
542    /// "transport not yet opened" states.
543    const INVALID: Self::Handle;
544
545    /// Returns `true` if `h` is a live handle (not [`INVALID`](Self::INVALID)
546    /// and points at a device that was opened and not yet closed).
547    fn is_valid(h: Self::Handle) -> bool;
548
549    /// Open the serial device identified by `path`. Returns a live
550    /// handle on success, [`INVALID`](Self::INVALID) on failure.
551    fn open(path: *const u8) -> Self::Handle;
552
553    /// Close the given handle. No-op if already closed or invalid.
554    fn close(h: Self::Handle);
555
556    /// Configure baud rate (in bits per second). Returns 0 on success,
557    /// -1 on error. Called after `open()`; implementations may choose
558    /// to apply the baud rate during `open()` instead and make this a
559    /// no-op.
560    fn configure(h: Self::Handle, baudrate: u32) -> i8;
561
562    /// Read up to `len` bytes into `buf`. Returns the number of bytes
563    /// read, `0` on timeout, or `usize::MAX` on hard error.
564    fn read(h: Self::Handle, buf: *mut u8, len: usize, timeout_ms: u32) -> usize;
565
566    /// Write `len` bytes from `buf`. Returns bytes written, or
567    /// `usize::MAX` on error.
568    fn write(h: Self::Handle, buf: *const u8, len: usize) -> usize;
569}
