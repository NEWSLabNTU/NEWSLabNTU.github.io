1//! Platform capability sub-traits for nros.
2//!
3//! This crate exists to break the dependency cycle between
4//! `nros-platform` (which depends on every platform crate via its
5//! feature-gated `ConcretePlatform` resolver) and the platform crates
6//! themselves (which need to implement these traits on their ZSTs).
7//! It contains only trait definitions — no implementations, no
8//! dependencies — so platform crates can take a build-time dep on it
9//! without creating a cycle back through `nros-platform`.
10//!
11//! `nros-platform` re-exports everything from this crate, so downstream
12//! code that writes `use nros_platform::PlatformClock;` continues to
13//! work unchanged.
14//!
15//! Each trait covers an independent system capability. Platform
16//! implementations pick which traits to implement based on what the
17//! hardware/RTOS provides. RMW shim crates declare trait bounds for
18//! the capabilities they need.
19
20#![no_std]
21//!
22//! # Naming convention
23//!
24//! Method names drop redundant prefixes when the trait name already
25//! supplies the namespace — e.g., `PlatformTcp::open` rather than
26//! `PlatformTcp::tcp_open`. Dispatch is always through a qualified
27//! path (`<ConcretePlatform as PlatformTcp>::open(...)`), so
28//! trait-to-trait collisions (PlatformTcp::open vs PlatformUdp::open)
29//! are disambiguated at the call site without needing a prefix on the
30//! trait method itself.
31//!
32//! Three categories still keep sub-namespace prefixes internally:
33//!
34//! * `PlatformThreading` — `mutex_*`, `condvar_*`, `task_*` because the
35//!   trait bundles three independent primitive families and unprefixed
36//!   `init` / `drop` would be ambiguous *within* the trait itself.
37//! * `PlatformUdpMulticast` — `mcast_*` because these methods have
38//!   different signatures from `PlatformUdp`'s same-name methods; keeping
39//!   the prefix makes call sites that use both traits self-documenting.
40//! * The `close` method appears on both `PlatformTcp` and
41//!   `PlatformSocketHelpers` — the first is TCP teardown, the second is
42//!   zenoh-pico's generic "shutdown + close" helper. Both live unprefixed
43//!   in their respective traits; call sites disambiguate via the
44//!   qualified path.
45//!
46//! # Status (Phase 84.F4)
47//!
48//! The platform ZSTs (`PosixPlatform`, `ZephyrPlatform`, etc.) do **not**
49//! currently implement these traits — every platform exposes its methods
50//! as *inherent* `impl Platform { fn foo() {} }` blocks, and shim crates
51//! dispatch by name match. 84.F4 migrates each platform to `impl
52//! PlatformX for Platform { fn foo() {} }` one trait at a time, with the
53//! shims switching to `<P as PlatformX>::foo()`. Until that work is
54//! complete the traits here are a target specification.
55

112// Cooperative yield

115/// Scheduler yield primitive.
116///
117/// Used inside `socket_wait_event` and similar "let another task make
118/// progress" points where the caller isn't actually waiting for I/O
119/// readability — the background read task already owns that — it just
120/// needs to relinquish the CPU so the real waiter can run.
121///
122/// Prior to Phase 77.22 each backend hand-rolled its own 1-ms busy
123/// sleep (`libc::usleep(1000)`, `vTaskDelay(1)`, `tx_thread_sleep(1)`,
124/// `k_usleep(1000)`, `select(.., 1 ms)`) — all with slightly different
125/// units and no common home.
126///
127/// **ISR-safety**: on the hosted-RTOS backends (FreeRTOS / NuttX /
128/// Zephyr / ThreadX) the underlying primitives panic or error when
129/// invoked from an ISR. Don't call `yield_now()` from an interrupt
130/// handler.
131///
132/// **Bare-metal has no real yield**: there's nothing to yield *to*.
133/// The default bare-metal impl is `core::hint::spin_loop()` (a pure
134/// CPU hint: emits `YIELD` / `PAUSE` / `WFE` depending on the arch,
135/// safe everywhere). Board crates that have armed an IRQ source may
136/// opt in to deep idle (`wfi`) via a separate `BoardIdle` hook — not
137/// part of this trait because calling `wfi` without an IRQ source
138/// deadlocks.
139pub trait PlatformYield {
140    /// Relinquish the CPU long enough for another task / thread to run.
141    ///
142    /// Non-blocking in the sleep sense — returns as soon as the
143    /// scheduler has had an opportunity to pick a different runnable.
144    fn yield_now();
145}

151/// Errors returned by [`PlatformScheduler`] entry points.
152#[derive(Debug, Clone, Copy, PartialEq, Eq)]
153pub enum SchedError {
154    /// The active platform doesn't expose this control surface
155    /// (e.g. bare-metal with no scheduler, or an RTOS without an
156    /// affinity API).
157    Unsupported,
158    /// The requested policy is valid for this platform but the
159    /// numeric arguments fall outside the platform's accepted range.
160    OutOfRange,
161    /// A platform-specific syscall / kernel call failed; check
162    /// `errno` (POSIX) or the RTOS error code for details.
163    KernelError,
164}
165
166/// Per-thread OS scheduling policy.
167///
168/// `Fifo` / `RoundRobin` / `Deadline` / `Sporadic` map to platform-
169/// native scheduling classes when available. `Platform(...)` is the
170/// escape hatch for RTOS-specific knobs (e.g. ThreadX preempt-
171/// threshold) that don't map cleanly into the portable variants.
172///
173/// User-facing API (`Executor::open_threaded`) takes the abstract
174/// [`Priority`-like](crate) values and `PlatformScheduler` translates
175/// them into platform-native numerics — direction-flipped priority
176/// (Zephyr / ThreadX low-numeric = high-priority vs POSIX / FreeRTOS
177/// high-numeric = high-priority) handled internally. Phase 110.D.
178#[derive(Debug, Clone, Copy, PartialEq, Eq)]
179pub enum SchedPolicy {
180    /// POSIX `SCHED_FIFO`. `os_pri` is the platform-native numeric
181    /// priority; out-of-range values surface `SchedError::OutOfRange`.
182    Fifo { os_pri: u8 },
183    /// POSIX `SCHED_RR` with `quantum_ms` time slice.
184    RoundRobin { os_pri: u8, quantum_ms: u32 },
185    /// Linux `SCHED_DEADLINE` (sched_setattr). All values nanoseconds.
186    Deadline {
187        runtime_ns: u64,
188        period_ns: u64,
189        deadline_ns: u64,
190    },
191    /// NuttX `SCHED_SPORADIC` server. Phase 110.E.
192    Sporadic {
193        budget_us: u32,
194        period_us: u32,
195        hi_pri: u8,
196        lo_pri: u8,
197    },
198}

213/// Errors returned by [`PlatformTimer`] entry points.
214#[derive(Debug, Clone, Copy, PartialEq, Eq)]
215pub enum TimerError {
216    /// The active platform doesn't expose a periodic-timer surface
217    /// (e.g. bare-metal without a board-side `SysTickHook`).
218    Unsupported,
219    /// `period_us` is below the platform's minimum timer resolution
220    /// or above its maximum representable interval.
221    OutOfRange,
222    /// A platform-specific syscall / kernel call failed; check
223    /// `errno` (POSIX) or the RTOS error code for details.
224    KernelError,
225}

227/// Periodic timer for ISR-driven sporadic-server budget refill
228/// (Phase 110.E.b). The trait factors out the per-platform timer
229/// surface so the executor (`nros-node`) can register an atomic
230/// refill callback without becoming generic over the platform —
231/// see `docs/design/phase-110-e-platform-timer.md`.
232///
233/// Default `create_periodic` returns [`TimerError::Unsupported`] so
234/// platforms without a timer surface inherit safe behavior; `destroy`
235/// is a no-op default for the same reason.
236pub trait PlatformTimer {
237    /// Opaque per-platform handle (FreeRTOS `TimerHandle_t`,
238    /// Zephyr `*mut k_timer`, ThreadX `*mut TX_TIMER`, POSIX
239    /// `timer_t`). Must be `Send + Sync + 'static` so the executor
240    /// can stash it across thread boundaries.
241    type TimerHandle: Send + Sync + 'static;
242
243    /// Register a periodic timer that fires `callback(user_data)`
244    /// every `period_us` microseconds. Returns the platform-native
245    /// handle for later destruction.
246    ///
247    /// # Safety
248    ///
249    /// `user_data` must outlive the returned handle. The callback is
250    /// invoked from a platform-defined timer context (direct ISR on
251    /// Zephyr / bare-metal, deferred via `xTimerPendFunctionCall` on
252    /// FreeRTOS, signal handler on POSIX) — bodies should be
253    /// short and use atomic ops only.
254    fn create_periodic(
255        _period_us: u32,
256        _callback: extern "C" fn(*mut c_void),
257        _user_data: *mut c_void,
258    ) -> Result<Self::TimerHandle, TimerError> {
259        Err(TimerError::Unsupported)
260    }
261
262    /// Cancel + free the timer. Idempotent on already-destroyed
263    /// handles. Must drain in-flight callback invocations before
264    /// returning so the user_data pointer is no longer accessed.
265    fn destroy(_handle: Self::TimerHandle) {}
266
267    /// Phase 110.E.b follow-up — register a one-shot timer that
268    /// fires `callback(user_data)` exactly once after `timeout_us`
269    /// microseconds. Used by per-callback runtime measurement: arm
270    /// just before dispatch with `timeout_us = budget_us`; on
271    /// callback completion call `cancel`. If the timer fires first
272    /// the callback overran its budget.
273    ///
274    /// Default returns `Unsupported` so platforms without a oneshot
275    /// surface inherit safe behavior.
276    fn create_oneshot(
277        _timeout_us: u32,
278        _callback: extern "C" fn(*mut c_void),
279        _user_data: *mut c_void,
280    ) -> Result<Self::TimerHandle, TimerError> {
281        Err(TimerError::Unsupported)
282    }
283
284    /// Phase 110.E.b follow-up — cancel a previously-armed oneshot
285    /// timer. Returns `true` when the cancellation prevented the
286    /// callback from firing, `false` when the callback already
287    /// fired (or the timer was already cancelled). Default is a
288    /// no-op returning `false`.
289    fn cancel(_handle: &mut Self::TimerHandle) -> bool {
290        false
291    }
292}

294pub trait PlatformScheduler {
295    /// Apply the requested policy to the calling thread.
296    ///
297    /// Default returns [`SchedError::Unsupported`] so single-core bare-
298    /// metal targets and RTOS impls without per-thread scheduling
299    /// pickup the no-op behavior automatically. Platforms with a
300    /// real scheduler override this.
301    fn set_current_thread_policy(_p: SchedPolicy) -> Result<(), SchedError> {
302        Err(SchedError::Unsupported)
303    }
304
305    /// Cooperative yield. Same semantics as
306    /// [`PlatformYield::yield_now`]; mirrored here so consumers don't
307    /// need to import both traits when only the scheduler control
308    /// surface is in scope. Default is a `core::hint::spin_loop()`.
309    fn yield_now() {
310        core::hint::spin_loop();
311    }
312
313    /// Pin the calling thread to the CPUs whose bit is set in
314    /// `cpu_mask`. Default returns [`SchedError::Unsupported`] —
315    /// platforms with affinity APIs override.
316    fn set_affinity(_cpu_mask: u32) -> Result<(), SchedError> {
317        Err(SchedError::Unsupported)
318    }
319}

367// ============================================================================
368// Threading (multi-threaded platforms only)
369// ============================================================================
370//
371// Handle types are opaque `*mut c_void` to match the shape zenoh-pico
372// passes across the FFI boundary. The original draft had typed wrappers
373// (`TaskHandle`, `MutexHandle`, ...) but every shipped platform used
374// `*mut c_void` internally and the shim never materialised the typed
375// forms — keeping them in the trait would have required a pointless
376// cast at every impl site. (F4.1 / F4.5 decision, 2026-04-24.)
377//
378// Mutex / condvar / task method names keep their sub-namespace prefix
379// (`mutex_*`, `condvar_*`, `task_*`) because the trait bundles three
380// independent primitive families and unprefixed `init` / `drop` would
381// be ambiguous *within* the trait itself.
382
383/// Threading primitives: tasks, mutexes, and condition variables.
384///
385/// # Threading
386///
387/// Mutex / condvar operations must be safe under concurrent callers.
388/// **Recursive mutex (`mutex_rec_*`) must support same-thread
389/// re-entrancy** — zenoh-pico relies on this; a non-recursive mutex
390/// backing `mutex_rec_*` deadlocks under load.
391///
392/// # ISR-safety
393///
394/// **None of these methods are ISR-safe** on hosted RTOSes
395/// (FreeRTOS / NuttX / Zephyr / ThreadX) — the underlying primitives
396/// panic or error when invoked from an ISR. Only `core::hint::spin_loop()`
397/// in [`PlatformYield::yield_now`] is.
398///
399/// For single-threaded platforms (bare-metal), all operations should
400/// be no-ops returning `0`, except [`task_init`](Self::task_init)
401/// which should return `-1`.
402pub trait PlatformThreading {
403    // -- Tasks --
404
405    /// Spawn a new task. `task` is opaque caller-provided storage;
406    /// `attr` carries scheduling hints (priority, stack size) or
407    /// `null` for defaults; `entry` is the task entry point; `arg`
408    /// is forwarded to `entry`. Returns `0` on success, non-zero
409    /// on failure.
410    fn task_init(
411        task: *mut c_void,
412        attr: *mut c_void,
413        entry: Option<unsafe extern "C" fn(*mut c_void) -> *mut c_void>,
414        arg: *mut c_void,
415    ) -> i8;
416
417    /// Block until `task` exits. Cleans up the task storage on
418    /// success.
419    fn task_join(task: *mut c_void) -> i8;
420    /// Mark `task` as detached — its storage is reclaimed on exit
421    /// without a join.
422    fn task_detach(task: *mut c_void) -> i8;
423    /// Request `task` to terminate at the next cancellation point.
424    /// Cooperative.
425    fn task_cancel(task: *mut c_void) -> i8;
426    /// Terminate the calling task immediately. Does not return.
427    fn task_exit();
428    /// Free the task storage allocated by `task_init`.
429    fn task_free(task: *mut *mut c_void);
430
431    // -- Mutex --
432
433    /// Initialise a non-recursive mutex in caller-provided storage.
434    fn mutex_init(m: *mut c_void) -> i8;
435    /// Tear down a non-recursive mutex.
436    fn mutex_drop(m: *mut c_void) -> i8;
437    /// Lock; block if held.
438    fn mutex_lock(m: *mut c_void) -> i8;
439    /// Try to lock; non-zero return immediately if held.
440    fn mutex_try_lock(m: *mut c_void) -> i8;
441    /// Unlock; only the owning thread may call this.
442    fn mutex_unlock(m: *mut c_void) -> i8;
443
444    // -- Recursive mutex --
445
446    /// Initialise a *recursive* mutex (same-thread re-entrancy
447    /// permitted). Required by zenoh-pico.
448    fn mutex_rec_init(m: *mut c_void) -> i8;
449    /// Tear down a recursive mutex.
450    fn mutex_rec_drop(m: *mut c_void) -> i8;
451    /// Lock; re-entry from the owning thread must succeed.
452    fn mutex_rec_lock(m: *mut c_void) -> i8;
453    /// Try to lock; same re-entry semantics as `mutex_rec_lock`.
454    fn mutex_rec_try_lock(m: *mut c_void) -> i8;
455    /// Unlock; releases when the lock count returns to zero.
456    fn mutex_rec_unlock(m: *mut c_void) -> i8;
457
458    // -- Condition variables --
459
460    /// Initialise a condition variable in caller-provided storage.
461    fn condvar_init(cv: *mut c_void) -> i8;
462    /// Tear down a condition variable.
463    fn condvar_drop(cv: *mut c_void) -> i8;
464    /// Wake one waiter on the condition variable.
465    fn condvar_signal(cv: *mut c_void) -> i8;
466    /// Wake all waiters on the condition variable.
467    fn condvar_signal_all(cv: *mut c_void) -> i8;
468    /// Phase 124.B.7.a — ISR-safe variant of [`Self::condvar_signal`].
469    /// Callable from interrupt / signal-handler context. Backends
470    /// implement via async-signal-safe primitives (POSIX:
471    /// `eventfd` write forwarded by a worker thread; RTOS:
472    /// `xSemaphoreGiveFromISR`, `tx_event_flags_set` from ISR,
473    /// `k_sem_give` from ISR). Returns non-zero when the backend
474    /// has no ISR-safe path — caller can fall back to
475    /// `condvar_signal` (with the obvious latency cost).
476    ///
477    /// Default body: forward to `condvar_signal`. POSIX and other
478    /// backends override to use their async-signal-safe primitive.
479    fn condvar_signal_from_isr(cv: *mut c_void) -> i8 {
480        Self::condvar_signal(cv)
481    }
482    /// Atomically release `m` and block on `cv`. The mutex is
483    /// re-acquired before this function returns.
484    fn condvar_wait(cv: *mut c_void, m: *mut c_void) -> i8;
485
486    /// Wait with absolute monotonic deadline (milliseconds since
487    /// the [`PlatformClock::clock_ms`] epoch). Returns non-zero on
488    /// timeout.
489    fn condvar_wait_until(cv: *mut c_void, m: *mut c_void, abstime: u64) -> i8;
490}

510/// Global mutual exclusion against preemption + ISR delivery
511/// (Phase 121.9).
512///
513/// Backs the Rust `critical_section::Impl` registration used by
514/// dust-dds, nros-rmw-{xrce,zenoh}, and any other no_std consumer of
515/// `critical_section::with()`. The token returned by `acquire` is
516/// passed back to `release`; it holds whatever bookkeeping the
517/// platform needs to restore the prior posture (Cortex-M PRIMASK bit,
518/// Cortex-R CPSR I-bit, RISC-V `mstatus.MIE` snapshot, pthread
519/// recursion depth, etc.) and is opaque to callers.
520///
521/// Reentrant by contract: nested `acquire` / `release` pairs must
522/// stack — the platform impl is responsible for nesting (PRIMASK
523/// already stacks; pthread side uses a recursive mutex).
524pub trait PlatformCriticalSection {
525    /// Enter a critical section. Returns an opaque token to pass to
526    /// [`Self::release`].
527    fn acquire() -> u32;
528
529    /// Leave a critical section, restoring the posture captured at
530    /// [`Self::acquire`].
531    fn release(token: u32);
532}

538/// TCP networking.
539///
540/// Socket and endpoint parameters are opaque `*mut c_void` pointers to
541/// platform-specific types (`_z_sys_net_socket_t`, `_z_sys_net_endpoint_t`).
542/// The shim provides correctly-sized `#[repr(C)]` wrappers whose sizes are
543/// auto-detected from C headers at build time (see Phase 80 design).
544///
545/// Read functions return `usize::MAX` on error. Send returns `usize::MAX` on error.
546///
547/// Method names are unprefixed — the trait already namespaces them. Shims
548/// dispatch via `<ConcretePlatform as PlatformTcp>::open(...)` etc.
549pub trait PlatformTcp {
550    /// Resolve address + port strings into an endpoint handle.
551    fn create_endpoint(ep: *mut c_void, address: *const u8, port: *const u8) -> i8;
552    /// Free endpoint resources.
553    fn free_endpoint(ep: *mut c_void);
554    /// Open a TCP client connection. `endpoint` is by-value (opaque bytes on stack).
555    fn open(sock: *mut c_void, endpoint: *const c_void, timeout_ms: u32) -> i8;
556    /// Open a TCP listening socket.
557    fn listen(sock: *mut c_void, endpoint: *const c_void) -> i8;
558    /// Close a TCP socket.
559    fn close(sock: *mut c_void);
560    /// Read up to `len` bytes. Returns bytes read, or `usize::MAX` on error.
561    fn read(sock: *const c_void, buf: *mut u8, len: usize) -> usize;
562    /// Read exactly `len` bytes. Returns `len` on success, `usize::MAX` on error.
563    fn read_exact(sock: *const c_void, buf: *mut u8, len: usize) -> usize;
564    /// Send `len` bytes. Returns bytes sent, or `usize::MAX` on error.
565    fn send(sock: *const c_void, buf: *const u8, len: usize) -> usize;
566}

712// ============================================================================
713// Inter-VM / mailbox transport (NVIDIA IVC and similar)
714// ============================================================================
715
716/// Inter-processor mailbox transport, modelled after NVIDIA Tegra IVC.
717///
718/// IVC (Inter-VM Communication on Tegra; in practice CCPLEX↔SPE on AGX
719/// Orin) is a header-prefixed lock-free SPSC ring in shared DRAM, paired
720/// with a hardware doorbell for wake. One channel is one peer — there
721/// is no discovery, naming, QoS, or fanout — which is why it's a *link*
722/// transport (peer to TCP/UDP/Serial/RawEth inside zenoh-pico) rather
723/// than a new RMW backend.
724///
725/// Channel handles are opaque `*mut c_void` to match the shape zenoh-pico
726/// passes across its FFI boundary. The driver crate (`nvidia-ivc`)
727/// translates between this opaque handle and either NVIDIA's FSP
728/// `tegra_ivc_*` API (`fsp` feature) or a Unix-socket pair
729/// (`unix-mock` feature, host dev + CI).
730///
731/// **Zero-copy contract** (Phase 11.3.A). NVIDIA's FSP IVC API is
732/// fundamentally a "borrow ring slot, fill, commit / release"
733/// pattern. Both backends mirror that here so consumers don't branch
734/// on backend:
735///
736/// - [`Self::rx_get`] returns a pointer into the channel's RX slot
737///   (and writes the frame length to `*len_out`); [`Self::rx_release`]
738///   advances the producer-visible cursor. Returns null +
739///   `*len_out = 0` if the ring is empty.
740/// - [`Self::tx_get`] returns a writable pointer to the next free TX
741///   slot (and writes the slot capacity to `*cap_out`);
742///   [`Self::tx_commit`] makes the slot visible to the peer (and
743///   rings the per-frame doorbell on FSP — see `notify` for batching).
744///   [`Self::tx_abandon`] frees the slot without sending. Returns
745///   null + `*cap_out = 0` if the ring is full.
746///
747/// Single outstanding RX slot and single outstanding TX slot per
748/// channel — borrow, finish, repeat. Multi-frame batching is the
749/// caller's job (loop `tx_get` / fill / `tx_commit`, then one
750/// `notify`).
751///
752/// `frame_size` is the fixed per-channel frame size negotiated at
753/// carveout setup (typical NVIDIA IVC: 64 bytes per frame, 16 frames
754/// per channel). The link layer uses it to pick its reassembly buffer.
755pub trait PlatformIvc {
756    /// Resolve a channel ID into an opaque handle. Returns null on
757    /// failure. The numeric ID matches the NVIDIA channel index
758    /// (`channel 2 = aon_echo`).
759    fn channel_get(id: u32) -> *mut c_void;
760
761    /// Fixed frame size negotiated for this channel, in bytes.
762    fn frame_size(ch: *mut c_void) -> u32;
763
764    /// Borrow the next-available RX frame. Writes the frame length to
765    /// `*len_out` and returns a pointer into the ring. Returns null +
766    /// `*len_out = 0` if no frame is available.
767    fn rx_get(ch: *mut c_void, len_out: *mut usize) -> *const u8;
768
769    /// Release the most recently `rx_get`'d frame back to the
770    /// producer. Pair 1:1 with `rx_get` calls that returned non-null.
771    fn rx_release(ch: *mut c_void);
772
773    /// Borrow the next free TX slot. Writes the slot capacity to
774    /// `*cap_out` and returns a writable pointer. Returns null +
775    /// `*cap_out = 0` if the ring is full.
776    fn tx_get(ch: *mut c_void, cap_out: *mut usize) -> *mut u8;
777
778    /// Commit `len` bytes from the most recently `tx_get`'d slot.
779    /// Slot is then visible to the peer; per-frame doorbell may also
780    /// fire on FSP. Pair 1:1 with `tx_get` calls that returned
781    /// non-null.
782    fn tx_commit(ch: *mut c_void, len: usize);
783
784    /// Abandon the most recently `tx_get`'d slot without sending.
785    /// Pair 1:1 with `tx_get` calls that returned non-null.
786    fn tx_abandon(ch: *mut c_void);
787
788    /// Ring the doorbell. On FSP this is redundant (commit/release
789    /// already invoke `ch->notify_remote`); on unix-mock it's a no-op
790    /// (SOCK_DGRAM wakes the peer naturally). Provided for symmetry
791    /// so callers can batch-commit then notify once.
792    fn notify(ch: *mut c_void);
793}

795// ============================================================================
796// Serial (UART / PTY)
797// ============================================================================
798
799/// Serial (byte-stream) transport.
800///
801/// Used by XRCE-DDS's HDLC-framed serial transport and (once the
802/// bare-metal migration lands) by zenoh-pico's serial link layer.
803///
804/// # Handle model
805///
806/// `open()` returns a platform-defined [`Handle`](PlatformSerial::Handle)
807/// — an FD on POSIX, a port-table index on bare-metal, whatever the
808/// impl wants. Every other method takes the handle back. This lets a
809/// single platform impl service multiple concurrent devices (e.g.,
810/// `zpico-serial`'s two-port table) without the trait constraining
811/// the impl to a single active device.
812///
813/// Single-device platforms return the same handle forever and ignore
814/// it internally; `INVALID` gives a well-defined "no live handle"
815/// sentinel for shims that stash the current handle in a `static`.
816///
817/// # Path conventions
818///
819/// `path` in `open()` is platform-specific: a null-terminated UTF-8
820/// device path on POSIX (e.g., `/dev/ttyUSB0` or a PTY), or a
821/// board-defined port identifier on bare-metal (typically parsed by
822/// the platform's internal handler). Callers pass the locator string
823/// from their config unchanged; interpretation is the platform's job.
824///
825/// # I/O conventions
826///
827/// Read / write return `usize::MAX` on hard error. Read with
828/// `timeout_ms == 0` should block for a platform-chosen default;
829/// positive values are the poll/select deadline in milliseconds.
830/// Returning `0` from `read()` indicates "no data within timeout" and
831/// is **not** an error — both XRCE and zenoh-pico tolerate
832/// timeout-zero reads.
833pub trait PlatformSerial {
834    /// Platform-specific handle type. POSIX returns the FD (`i32`);
835    /// bare-metal returns a port-table index (`u8`). Must be `Copy`
836    /// so shims can stash it in a `static`.
837    type Handle: Copy;
838
839    /// Sentinel handle meaning "not a live device". Shims initialise
840    /// their cached handle to this and compare against it to detect
841    /// "transport not yet opened" states.
842    const INVALID: Self::Handle;
843
844    /// Returns `true` if `h` is a live handle (not [`INVALID`](Self::INVALID)
845    /// and points at a device that was opened and not yet closed).
846    fn is_valid(h: Self::Handle) -> bool;
847
848    /// Open the serial device identified by `path`. Returns a live
849    /// handle on success, [`INVALID`](Self::INVALID) on failure.
850    fn open(path: *const u8) -> Self::Handle;
851
852    /// Close the given handle. No-op if already closed or invalid.
853    fn close(h: Self::Handle);
854
855    /// Configure baud rate (in bits per second). Returns 0 on success,
856    /// -1 on error. Called after `open()`; implementations may choose
857    /// to apply the baud rate during `open()` instead and make this a
858    /// no-op.
859    fn configure(h: Self::Handle, baudrate: u32) -> i8;
860
861    /// Read up to `len` bytes into `buf`. Returns the number of bytes
862    /// read, `0` on timeout, or `usize::MAX` on hard error.
863    fn read(h: Self::Handle, buf: *mut u8, len: usize, timeout_ms: u32) -> usize;
864
865    /// Write `len` bytes from `buf`. Returns bytes written, or
866    /// `usize::MAX` on error.
867    fn write(h: Self::Handle, buf: *const u8, len: usize) -> usize;
868}
