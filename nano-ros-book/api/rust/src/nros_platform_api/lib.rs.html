1//! Platform capability sub-traits for nros.
2//!
3//! This crate exists to break the dependency cycle between
4//! `nros-platform` (which depends on every platform crate via its
5//! feature-gated `ConcretePlatform` resolver) and the platform crates
6//! themselves (which need to implement these traits on their ZSTs).
7//! It contains only trait definitions — no implementations, no
8//! dependencies — so platform crates can take a build-time dep on it
9//! without creating a cycle back through `nros-platform`.
10//!
11//! `nros-platform` re-exports everything from this crate, so downstream
12//! code that writes `use nros_platform::PlatformClock;` continues to
13//! work unchanged.
14//!
15//! Each trait covers an independent system capability. Platform
16//! implementations pick which traits to implement based on what the
17//! hardware/RTOS provides. RMW shim crates declare trait bounds for
18//! the capabilities they need.
19
20#![no_std]
21//!
22//! # Naming convention
23//!
24//! Method names drop redundant prefixes when the trait name already
25//! supplies the namespace — e.g., `PlatformTcp::open` rather than
26//! `PlatformTcp::tcp_open`. Dispatch is always through a qualified
27//! path (`<ConcretePlatform as PlatformTcp>::open(...)`), so
28//! trait-to-trait collisions (PlatformTcp::open vs PlatformUdp::open)
29//! are disambiguated at the call site without needing a prefix on the
30//! trait method itself.
31//!
32//! Three categories still keep sub-namespace prefixes internally:
33//!
34//! * `PlatformThreading` — `mutex_*`, `condvar_*`, `task_*` because the
35//!   trait bundles three independent primitive families and unprefixed
36//!   `init` / `drop` would be ambiguous *within* the trait itself.
37//! * `PlatformUdpMulticast` — `mcast_*` because these methods have
38//!   different signatures from `PlatformUdp`'s same-name methods; keeping
39//!   the prefix makes call sites that use both traits self-documenting.
40//! * The `close` method appears on both `PlatformTcp` and
41//!   `PlatformSocketHelpers` — the first is TCP teardown, the second is
42//!   zenoh-pico's generic "shutdown + close" helper. Both live unprefixed
43//!   in their respective traits; call sites disambiguate via the
44//!   qualified path.
45//!
46//! # Status (Phase 84.F4)
47//!
48//! The platform ZSTs (`PosixPlatform`, `ZephyrPlatform`, etc.) do **not**
49//! currently implement these traits — every platform exposes its methods
50//! as *inherent* `impl Platform { fn foo() {} }` blocks, and shim crates
51//! dispatch by name match. 84.F4 migrates each platform to `impl
52//! PlatformX for Platform { fn foo() {} }` one trait at a time, with the
53//! shims switching to `<P as PlatformX>::foo()`. Until that work is
54//! complete the traits here are a target specification.
55

118/// Scheduler yield primitive.
119///
120/// Used inside `socket_wait_event` and similar "let another task make
121/// progress" points where the caller isn't actually waiting for I/O
122/// readability — the background read task already owns that — it just
123/// needs to relinquish the CPU so the real waiter can run.
124///
125/// Prior to Phase 77.22 each backend hand-rolled its own 1-ms busy
126/// sleep (`libc::usleep(1000)`, `vTaskDelay(1)`, `tx_thread_sleep(1)`,
127/// `k_usleep(1000)`, `select(.., 1 ms)`) — all with slightly different
128/// units and no common home.
129///
130/// **ISR-safety**: on the hosted-RTOS backends (FreeRTOS / NuttX /
131/// Zephyr / ThreadX) the underlying primitives panic or error when
132/// invoked from an ISR. Don't call `yield_now()` from an interrupt
133/// handler.
134///
135/// **Bare-metal has no real yield**: there's nothing to yield *to*.
136/// The default bare-metal impl is `core::hint::spin_loop()` (a pure
137/// CPU hint: emits `YIELD` / `PAUSE` / `WFE` depending on the arch,
138/// safe everywhere). Board crates that have armed an IRQ source may
139/// opt in to deep idle (`wfi`) via a separate `BoardIdle` hook — not
140/// part of this trait because calling `wfi` without an IRQ source
141/// deadlocks.
142pub trait PlatformYield {
143    /// Relinquish the CPU long enough for another task / thread to run.
144    ///
145    /// Non-blocking in the sleep sense — returns as soon as the
146    /// scheduler has had an opportunity to pick a different runnable.
147    fn yield_now();
148}

154/// Errors returned by [`PlatformScheduler`] entry points.
155#[derive(Debug, Clone, Copy, PartialEq, Eq)]
156pub enum SchedError {
157    /// The active platform doesn't expose this control surface
158    /// (e.g. bare-metal with no scheduler, or an RTOS without an
159    /// affinity API).
160    Unsupported,
161    /// The requested policy is valid for this platform but the
162    /// numeric arguments fall outside the platform's accepted range.
163    OutOfRange,
164    /// A platform-specific syscall / kernel call failed; check
165    /// `errno` (POSIX) or the RTOS error code for details.
166    KernelError,
167}
168
169/// Per-thread OS scheduling policy.
170///
171/// `Fifo` / `RoundRobin` / `Deadline` / `Sporadic` map to platform-
172/// native scheduling classes when available. `Platform(...)` is the
173/// escape hatch for RTOS-specific knobs (e.g. ThreadX preempt-
174/// threshold) that don't map cleanly into the portable variants.
175///
176/// User-facing API (`Executor::open_threaded`) takes the abstract
177/// [`Priority`-like](crate) values and `PlatformScheduler` translates
178/// them into platform-native numerics — direction-flipped priority
179/// (Zephyr / ThreadX low-numeric = high-priority vs POSIX / FreeRTOS
180/// high-numeric = high-priority) handled internally. Phase 110.D.
181#[derive(Debug, Clone, Copy, PartialEq, Eq)]
182pub enum SchedPolicy {
183    /// POSIX `SCHED_FIFO`. `os_pri` is the platform-native numeric
184    /// priority; out-of-range values surface `SchedError::OutOfRange`.
185    Fifo { os_pri: u8 },
186    /// POSIX `SCHED_RR` with `quantum_ms` time slice.
187    RoundRobin { os_pri: u8, quantum_ms: u32 },
188    /// Linux `SCHED_DEADLINE` (sched_setattr). All values nanoseconds.
189    Deadline {
190        runtime_ns: u64,
191        period_ns: u64,
192        deadline_ns: u64,
193    },
194    /// NuttX `SCHED_SPORADIC` server. Phase 110.E.
195    Sporadic {
196        budget_us: u32,
197        period_us: u32,
198        hi_pri: u8,
199        lo_pri: u8,
200    },
201}

216/// Errors returned by [`PlatformTimer`] entry points.
217#[derive(Debug, Clone, Copy, PartialEq, Eq)]
218pub enum TimerError {
219    /// The active platform doesn't expose a periodic-timer surface
220    /// (e.g. bare-metal without a board-side `SysTickHook`).
221    Unsupported,
222    /// `period_us` is below the platform's minimum timer resolution
223    /// or above its maximum representable interval.
224    OutOfRange,
225    /// A platform-specific syscall / kernel call failed; check
226    /// `errno` (POSIX) or the RTOS error code for details.
227    KernelError,
228}

230/// Periodic timer for ISR-driven sporadic-server budget refill
231/// (Phase 110.E.b). The trait factors out the per-platform timer
232/// surface so the executor (`nros-node`) can register an atomic
233/// refill callback without becoming generic over the platform —
234/// see `docs/design/phase-110-e-platform-timer.md`.
235///
236/// Default `create_periodic` returns [`TimerError::Unsupported`] so
237/// platforms without a timer surface inherit safe behavior; `destroy`
238/// is a no-op default for the same reason.
239pub trait PlatformTimer {
240    /// Opaque per-platform handle (FreeRTOS `TimerHandle_t`,
241    /// Zephyr `*mut k_timer`, ThreadX `*mut TX_TIMER`, POSIX
242    /// `timer_t`). Must be `Send + Sync + 'static` so the executor
243    /// can stash it across thread boundaries.
244    type TimerHandle: Send + Sync + 'static;
245
246    /// Register a periodic timer that fires `callback(user_data)`
247    /// every `period_us` microseconds. Returns the platform-native
248    /// handle for later destruction.
249    ///
250    /// # Safety
251    ///
252    /// `user_data` must outlive the returned handle. The callback is
253    /// invoked from a platform-defined timer context (direct ISR on
254    /// Zephyr / bare-metal, deferred via `xTimerPendFunctionCall` on
255    /// FreeRTOS, signal handler on POSIX) — bodies should be
256    /// short and use atomic ops only.
257    fn create_periodic(
258        _period_us: u32,
259        _callback: extern "C" fn(*mut c_void),
260        _user_data: *mut c_void,
261    ) -> Result<Self::TimerHandle, TimerError> {
262        Err(TimerError::Unsupported)
263    }
264
265    /// Cancel + free the timer. Idempotent on already-destroyed
266    /// handles. Must drain in-flight callback invocations before
267    /// returning so the user_data pointer is no longer accessed.
268    fn destroy(_handle: Self::TimerHandle) {}
269
270    /// Phase 110.E.b follow-up — register a one-shot timer that
271    /// fires `callback(user_data)` exactly once after `timeout_us`
272    /// microseconds. Used by per-callback runtime measurement: arm
273    /// just before dispatch with `timeout_us = budget_us`; on
274    /// callback completion call `cancel`. If the timer fires first
275    /// the callback overran its budget.
276    ///
277    /// Default returns `Unsupported` so platforms without a oneshot
278    /// surface inherit safe behavior.
279    fn create_oneshot(
280        _timeout_us: u32,
281        _callback: extern "C" fn(*mut c_void),
282        _user_data: *mut c_void,
283    ) -> Result<Self::TimerHandle, TimerError> {
284        Err(TimerError::Unsupported)
285    }
286
287    /// Phase 110.E.b follow-up — cancel a previously-armed oneshot
288    /// timer. Returns `true` when the cancellation prevented the
289    /// callback from firing, `false` when the callback already
290    /// fired (or the timer was already cancelled). Default is a
291    /// no-op returning `false`.
292    fn cancel(_handle: &mut Self::TimerHandle) -> bool {
293        false
294    }
295}

297pub trait PlatformScheduler {
298    /// Apply the requested policy to the calling thread.
299    ///
300    /// Default returns [`SchedError::Unsupported`] so single-core bare-
301    /// metal targets and RTOS impls without per-thread scheduling
302    /// pickup the no-op behavior automatically. Platforms with a
303    /// real scheduler override this.
304    fn set_current_thread_policy(_p: SchedPolicy) -> Result<(), SchedError> {
305        Err(SchedError::Unsupported)
306    }
307
308    /// Cooperative yield. Same semantics as
309    /// [`PlatformYield::yield_now`]; mirrored here so consumers don't
310    /// need to import both traits when only the scheduler control
311    /// surface is in scope. Default is a `core::hint::spin_loop()`.
312    fn yield_now() {
313        core::hint::spin_loop();
314    }
315
316    /// Pin the calling thread to the CPUs whose bit is set in
317    /// `cpu_mask`. Default returns [`SchedError::Unsupported`] —
318    /// platforms with affinity APIs override.
319    fn set_affinity(_cpu_mask: u32) -> Result<(), SchedError> {
320        Err(SchedError::Unsupported)
321    }
322}

370// ============================================================================
371// Threading (multi-threaded platforms only)
372// ============================================================================
373//
374// Handle types are opaque `*mut c_void` to match the shape zenoh-pico
375// passes across the FFI boundary. The original draft had typed wrappers
376// (`TaskHandle`, `MutexHandle`, ...) but every shipped platform used
377// `*mut c_void` internally and the shim never materialised the typed
378// forms — keeping them in the trait would have required a pointless
379// cast at every impl site. (F4.1 / F4.5 decision, 2026-04-24.)
380//
381// Mutex / condvar / task method names keep their sub-namespace prefix
382// (`mutex_*`, `condvar_*`, `task_*`) because the trait bundles three
383// independent primitive families and unprefixed `init` / `drop` would
384// be ambiguous *within* the trait itself.
385
386/// Threading primitives: tasks, mutexes, and condition variables.
387///
388/// # Threading
389///
390/// Mutex / condvar operations must be safe under concurrent callers.
391/// **Recursive mutex (`mutex_rec_*`) must support same-thread
392/// re-entrancy** — zenoh-pico relies on this; a non-recursive mutex
393/// backing `mutex_rec_*` deadlocks under load.
394///
395/// # ISR-safety
396///
397/// **None of these methods are ISR-safe** on hosted RTOSes
398/// (FreeRTOS / NuttX / Zephyr / ThreadX) — the underlying primitives
399/// panic or error when invoked from an ISR. Only `core::hint::spin_loop()`
400/// in [`PlatformYield::yield_now`] is.
401///
402/// For single-threaded platforms (bare-metal), all operations should
403/// be no-ops returning `0`, except [`task_init`](Self::task_init)
404/// which should return `-1`.
405pub trait PlatformThreading {
406    // -- Tasks --
407
408    /// Spawn a new task. `task` is opaque caller-provided storage;
409    /// `attr` carries scheduling hints (priority, stack size) or
410    /// `null` for defaults; `entry` is the task entry point; `arg`
411    /// is forwarded to `entry`. Returns `0` on success, non-zero
412    /// on failure.
413    fn task_init(
414        task: *mut c_void,
415        attr: *mut c_void,
416        entry: Option<unsafe extern "C" fn(*mut c_void) -> *mut c_void>,
417        arg: *mut c_void,
418    ) -> i8;
419
420    /// Block until `task` exits. Cleans up the task storage on
421    /// success.
422    fn task_join(task: *mut c_void) -> i8;
423    /// Mark `task` as detached — its storage is reclaimed on exit
424    /// without a join.
425    fn task_detach(task: *mut c_void) -> i8;
426    /// Request `task` to terminate at the next cancellation point.
427    /// Cooperative.
428    fn task_cancel(task: *mut c_void) -> i8;
429    /// Terminate the calling task immediately. Does not return.
430    fn task_exit();
431    /// Free the task storage allocated by `task_init`.
432    fn task_free(task: *mut *mut c_void);
433
434    // -- Mutex --
435
436    /// Initialise a non-recursive mutex in caller-provided storage.
437    fn mutex_init(m: *mut c_void) -> i8;
438    /// Tear down a non-recursive mutex.
439    fn mutex_drop(m: *mut c_void) -> i8;
440    /// Lock; block if held.
441    fn mutex_lock(m: *mut c_void) -> i8;
442    /// Try to lock; non-zero return immediately if held.
443    fn mutex_try_lock(m: *mut c_void) -> i8;
444    /// Unlock; only the owning thread may call this.
445    fn mutex_unlock(m: *mut c_void) -> i8;
446
447    // -- Recursive mutex --
448
449    /// Initialise a *recursive* mutex (same-thread re-entrancy
450    /// permitted). Required by zenoh-pico.
451    fn mutex_rec_init(m: *mut c_void) -> i8;
452    /// Tear down a recursive mutex.
453    fn mutex_rec_drop(m: *mut c_void) -> i8;
454    /// Lock; re-entry from the owning thread must succeed.
455    fn mutex_rec_lock(m: *mut c_void) -> i8;
456    /// Try to lock; same re-entry semantics as `mutex_rec_lock`.
457    fn mutex_rec_try_lock(m: *mut c_void) -> i8;
458    /// Unlock; releases when the lock count returns to zero.
459    fn mutex_rec_unlock(m: *mut c_void) -> i8;
460
461    // -- Condition variables --
462
463    /// Initialise a condition variable in caller-provided storage.
464    fn condvar_init(cv: *mut c_void) -> i8;
465    /// Tear down a condition variable.
466    fn condvar_drop(cv: *mut c_void) -> i8;
467    /// Wake one waiter on the condition variable.
468    fn condvar_signal(cv: *mut c_void) -> i8;
469    /// Wake all waiters on the condition variable.
470    fn condvar_signal_all(cv: *mut c_void) -> i8;
471    /// Phase 124.B.7.a — ISR-safe variant of [`Self::condvar_signal`].
472    /// Callable from interrupt / signal-handler context. Backends
473    /// implement via async-signal-safe primitives (POSIX:
474    /// `eventfd` write forwarded by a worker thread; RTOS:
475    /// `xSemaphoreGiveFromISR`, `tx_event_flags_set` from ISR,
476    /// `k_sem_give` from ISR). Returns non-zero when the backend
477    /// has no ISR-safe path — caller can fall back to
478    /// `condvar_signal` (with the obvious latency cost).
479    ///
480    /// Default body: forward to `condvar_signal`. POSIX and other
481    /// backends override to use their async-signal-safe primitive.
482    fn condvar_signal_from_isr(cv: *mut c_void) -> i8 {
483        Self::condvar_signal(cv)
484    }
485    /// Atomically release `m` and block on `cv`. The mutex is
486    /// re-acquired before this function returns.
487    fn condvar_wait(cv: *mut c_void, m: *mut c_void) -> i8;
488
489    /// Wait with absolute monotonic deadline (milliseconds since
490    /// the [`PlatformClock::clock_ms`] epoch). Returns non-zero on
491    /// timeout.
492    fn condvar_wait_until(cv: *mut c_void, m: *mut c_void, abstime: u64) -> i8;
493
494    // -- Wake primitive (Phase 129) --
495    //
496    // Binary-semaphore-shaped primitive for the executor's wake_flag
497    // / spin_once cv-wait pair. Default bodies return "unsupported"
498    // (`-1`, size 0) so existing single-thread bare-metal platforms
499    // don't need to override; platforms that want event-driven wake
500    // (POSIX, Zephyr, FreeRTOS, NuttX, ThreadX) override with their
501    // native binary semaphore.
502
503    /// Initialise a binary-semaphore-shaped wake primitive in
504    /// caller-provided storage. See `<nros/platform.h>` for the
505    /// per-platform backing primitive (POSIX `sem_t`, Zephyr
506    /// `k_sem`, FreeRTOS `xSemaphoreBinary`, …). Default returns
507    /// `-1` (unsupported).
508    fn wake_init(_w: *mut c_void) -> i8 {
509        -1
510    }
511    /// Tear down a wake primitive. Default returns `-1`
512    /// (unsupported).
513    fn wake_drop(_w: *mut c_void) -> i8 {
514        -1
515    }
516    /// Block until signaled or `timeout_ms` elapses. Returns `0` on
517    /// signal, `1` on timeout, `-1` on error. Default returns `-1`
518    /// (unsupported).
519    fn wake_wait_ms(_w: *mut c_void, _timeout_ms: u32) -> i8 {
520        -1
521    }
522    /// Wake one waiter. Idempotent — a signal pending when another
523    /// arrives is coalesced (the primitive stays at value 1).
524    /// Default returns `-1` (unsupported).
525    fn wake_signal(_w: *mut c_void) -> i8 {
526        -1
527    }
528    /// ISR-safe signal. Returns `-1` when the backend has no ISR
529    /// path; callers may fall back to `wake_signal` (with the
530    /// obvious latency cost). Default forwards to `wake_signal`.
531    fn wake_signal_from_isr(w: *mut c_void) -> i8 {
532        Self::wake_signal(w)
533    }
534    /// Caller-storage size requirement (bytes). Default `0` —
535    /// signals "no wake primitive available". May be called before
536    /// `wake_init`.
537    fn wake_storage_size() -> usize {
538        0
539    }
540    /// Caller-storage alignment requirement (bytes). Default `1`.
541    fn wake_storage_align() -> usize {
542        1
543    }
544}

562/// Global mutual exclusion against preemption + ISR delivery
563/// (Phase 121.9).
564///
565/// Backs the Rust `critical_section::Impl` registration used by
566/// dust-dds, nros-rmw-{xrce,zenoh}, and any other no_std consumer of
567/// `critical_section::with()`. The token returned by `acquire` is
568/// passed back to `release`; it holds whatever bookkeeping the
569/// platform needs to restore the prior posture (Cortex-M PRIMASK bit,
570/// Cortex-R CPSR I-bit, RISC-V `mstatus.MIE` snapshot, pthread
571/// recursion depth, etc.) and is opaque to callers.
572///
573/// Reentrant by contract: nested `acquire` / `release` pairs must
574/// stack — the platform impl is responsible for nesting (PRIMASK
575/// already stacks; pthread side uses a recursive mutex).
576pub trait PlatformCriticalSection {
577    /// Enter a critical section. Returns an opaque token to pass to
578    /// [`Self::release`].
579    fn acquire() -> u32;
580
581    /// Leave a critical section, restoring the posture captured at
582    /// [`Self::acquire`].
583    fn release(token: u32);
584}

590/// TCP networking.
591///
592/// Socket and endpoint parameters are opaque `*mut c_void` pointers to
593/// platform-specific types (`_z_sys_net_socket_t`, `_z_sys_net_endpoint_t`).
594/// The shim provides correctly-sized `#[repr(C)]` wrappers whose sizes are
595/// auto-detected from C headers at build time (see Phase 80 design).
596///
597/// Read functions return `usize::MAX` on error. Send returns `usize::MAX` on error.
598///
599/// Method names are unprefixed — the trait already namespaces them. Shims
600/// dispatch via `<ConcretePlatform as PlatformTcp>::open(...)` etc.
601pub trait PlatformTcp {
602    /// Resolve address + port strings into an endpoint handle.
603    fn create_endpoint(ep: *mut c_void, address: *const u8, port: *const u8) -> i8;
604    /// Free endpoint resources.
605    fn free_endpoint(ep: *mut c_void);
606    /// Open a TCP client connection. `endpoint` is by-value (opaque bytes on stack).
607    fn open(sock: *mut c_void, endpoint: *const c_void, timeout_ms: u32) -> i8;
608    /// Open a TCP listening socket.
609    fn listen(sock: *mut c_void, endpoint: *const c_void) -> i8;
610    /// Close a TCP socket.
611    fn close(sock: *mut c_void);
612    /// Read up to `len` bytes. Returns bytes read, or `usize::MAX` on error.
613    fn read(sock: *const c_void, buf: *mut u8, len: usize) -> usize;
614    /// Read exactly `len` bytes. Returns `len` on success, `usize::MAX` on error.
615    fn read_exact(sock: *const c_void, buf: *mut u8, len: usize) -> usize;
616    /// Send `len` bytes. Returns bytes sent, or `usize::MAX` on error.
617    fn send(sock: *const c_void, buf: *const u8, len: usize) -> usize;
618}

764// ============================================================================
765// Inter-VM / mailbox transport (NVIDIA IVC and similar)
766// ============================================================================
767
768/// Inter-processor mailbox transport, modelled after NVIDIA Tegra IVC.
769///
770/// IVC (Inter-VM Communication on Tegra; in practice CCPLEX↔SPE on AGX
771/// Orin) is a header-prefixed lock-free SPSC ring in shared DRAM, paired
772/// with a hardware doorbell for wake. One channel is one peer — there
773/// is no discovery, naming, QoS, or fanout — which is why it's a *link*
774/// transport (peer to TCP/UDP/Serial/RawEth inside zenoh-pico) rather
775/// than a new RMW backend.
776///
777/// Channel handles are opaque `*mut c_void` to match the shape zenoh-pico
778/// passes across its FFI boundary. The driver crate (`nvidia-ivc`)
779/// translates between this opaque handle and either NVIDIA's FSP
780/// `tegra_ivc_*` API (`fsp` feature) or a Unix-socket pair
781/// (`unix-mock` feature, host dev + CI).
782///
783/// **Zero-copy contract** (Phase 11.3.A). NVIDIA's FSP IVC API is
784/// fundamentally a "borrow ring slot, fill, commit / release"
785/// pattern. Both backends mirror that here so consumers don't branch
786/// on backend:
787///
788/// - [`Self::rx_get`] returns a pointer into the channel's RX slot
789///   (and writes the frame length to `*len_out`); [`Self::rx_release`]
790///   advances the producer-visible cursor. Returns null +
791///   `*len_out = 0` if the ring is empty.
792/// - [`Self::tx_get`] returns a writable pointer to the next free TX
793///   slot (and writes the slot capacity to `*cap_out`);
794///   [`Self::tx_commit`] makes the slot visible to the peer (and
795///   rings the per-frame doorbell on FSP — see `notify` for batching).
796///   [`Self::tx_abandon`] frees the slot without sending. Returns
797///   null + `*cap_out = 0` if the ring is full.
798///
799/// Single outstanding RX slot and single outstanding TX slot per
800/// channel — borrow, finish, repeat. Multi-frame batching is the
801/// caller's job (loop `tx_get` / fill / `tx_commit`, then one
802/// `notify`).
803///
804/// `frame_size` is the fixed per-channel frame size negotiated at
805/// carveout setup (typical NVIDIA IVC: 64 bytes per frame, 16 frames
806/// per channel). The link layer uses it to pick its reassembly buffer.
807pub trait PlatformIvc {
808    /// Resolve a channel ID into an opaque handle. Returns null on
809    /// failure. The numeric ID matches the NVIDIA channel index
810    /// (`channel 2 = aon_echo`).
811    fn channel_get(id: u32) -> *mut c_void;
812
813    /// Fixed frame size negotiated for this channel, in bytes.
814    fn frame_size(ch: *mut c_void) -> u32;
815
816    /// Borrow the next-available RX frame. Writes the frame length to
817    /// `*len_out` and returns a pointer into the ring. Returns null +
818    /// `*len_out = 0` if no frame is available.
819    fn rx_get(ch: *mut c_void, len_out: *mut usize) -> *const u8;
820
821    /// Release the most recently `rx_get`'d frame back to the
822    /// producer. Pair 1:1 with `rx_get` calls that returned non-null.
823    fn rx_release(ch: *mut c_void);
824
825    /// Borrow the next free TX slot. Writes the slot capacity to
826    /// `*cap_out` and returns a writable pointer. Returns null +
827    /// `*cap_out = 0` if the ring is full.
828    fn tx_get(ch: *mut c_void, cap_out: *mut usize) -> *mut u8;
829
830    /// Commit `len` bytes from the most recently `tx_get`'d slot.
831    /// Slot is then visible to the peer; per-frame doorbell may also
832    /// fire on FSP. Pair 1:1 with `tx_get` calls that returned
833    /// non-null.
834    fn tx_commit(ch: *mut c_void, len: usize);
835
836    /// Abandon the most recently `tx_get`'d slot without sending.
837    /// Pair 1:1 with `tx_get` calls that returned non-null.
838    fn tx_abandon(ch: *mut c_void);
839
840    /// Ring the doorbell. On FSP this is redundant (commit/release
841    /// already invoke `ch->notify_remote`); on unix-mock it's a no-op
842    /// (SOCK_DGRAM wakes the peer naturally). Provided for symmetry
843    /// so callers can batch-commit then notify once.
844    fn notify(ch: *mut c_void);
845}

847// ============================================================================
848// Serial (UART / PTY)
849// ============================================================================
850
851/// Serial (byte-stream) transport.
852///
853/// Used by XRCE-DDS's HDLC-framed serial transport and (once the
854/// bare-metal migration lands) by zenoh-pico's serial link layer.
855///
856/// # Handle model
857///
858/// `open()` returns a platform-defined [`Handle`](PlatformSerial::Handle)
859/// — an FD on POSIX, a port-table index on bare-metal, whatever the
860/// impl wants. Every other method takes the handle back. This lets a
861/// single platform impl service multiple concurrent devices (e.g.,
862/// `zpico-serial`'s two-port table) without the trait constraining
863/// the impl to a single active device.
864///
865/// Single-device platforms return the same handle forever and ignore
866/// it internally; `INVALID` gives a well-defined "no live handle"
867/// sentinel for shims that stash the current handle in a `static`.
868///
869/// # Path conventions
870///
871/// `path` in `open()` is platform-specific: a null-terminated UTF-8
872/// device path on POSIX (e.g., `/dev/ttyUSB0` or a PTY), or a
873/// board-defined port identifier on bare-metal (typically parsed by
874/// the platform's internal handler). Callers pass the locator string
875/// from their config unchanged; interpretation is the platform's job.
876///
877/// # I/O conventions
878///
879/// Read / write return `usize::MAX` on hard error. Read with
880/// `timeout_ms == 0` should block for a platform-chosen default;
881/// positive values are the poll/select deadline in milliseconds.
882/// Returning `0` from `read()` indicates "no data within timeout" and
883/// is **not** an error — both XRCE and zenoh-pico tolerate
884/// timeout-zero reads.
885pub trait PlatformSerial {
886    /// Platform-specific handle type. POSIX returns the FD (`i32`);
887    /// bare-metal returns a port-table index (`u8`). Must be `Copy`
888    /// so shims can stash it in a `static`.
889    type Handle: Copy;
890
891    /// Sentinel handle meaning "not a live device". Shims initialise
892    /// their cached handle to this and compare against it to detect
893    /// "transport not yet opened" states.
894    const INVALID: Self::Handle;
895
896    /// Returns `true` if `h` is a live handle (not [`INVALID`](Self::INVALID)
897    /// and points at a device that was opened and not yet closed).
898    fn is_valid(h: Self::Handle) -> bool;
899
900    /// Open the serial device identified by `path`. Returns a live
901    /// handle on success, [`INVALID`](Self::INVALID) on failure.
902    fn open(path: *const u8) -> Self::Handle;
903
904    /// Close the given handle. No-op if already closed or invalid.
905    fn close(h: Self::Handle);
906
907    /// Configure baud rate (in bits per second). Returns 0 on success,
908    /// -1 on error. Called after `open()`; implementations may choose
909    /// to apply the baud rate during `open()` instead and make this a
910    /// no-op.
911    fn configure(h: Self::Handle, baudrate: u32) -> i8;
912
913    /// Read up to `len` bytes into `buf`. Returns the number of bytes
914    /// read, `0` on timeout, or `usize::MAX` on hard error.
915    fn read(h: Self::Handle, buf: *mut u8, len: usize, timeout_ms: u32) -> usize;
916
917    /// Write `len` bytes from `buf`. Returns bytes written, or
918    /// `usize::MAX` on error.
919    fn write(h: Self::Handle, buf: *const u8, len: usize) -> usize;
920}
