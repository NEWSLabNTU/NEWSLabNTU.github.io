1//! Phase 129 — ergonomic Rust wrapper around
2//! [`PlatformThreading`]'s wake primitive.
3//!
4//! Wraps a fixed-size, aligned scratch buffer so the executor can
5//! drop a `PlatformWake<P>` into a struct field without having to
6//! call `nros_platform_wake_storage_size()` at compile time.
7//! `WAKE_STORAGE_BYTES` is sized to cover every supported
8//! platform's binary semaphore (POSIX `sem_t` ~32 B, Zephyr
9//! `k_sem` ~16 B, FreeRTOS `xSemaphoreHandle` ~ptr, NuttX `sem_t`,
10//! ThreadX `tx_semaphore` ~56 B, macOS pthread cond+mutex+flag
11//! ~72 B). The probe-vs-buffer invariant is asserted at runtime
12//! during construction.
13//!
14//! Allocation-free: `Wake<P>` lives inline. Suitable for `no_std`
15//! consumers that can't depend on `alloc`.
16//!
17//! `Wake<P>` is `Sync` (the underlying primitive is the wake
18//! contract) but not `Send` after construction — the storage must
19//! stay put because the platform impl stores backing-primitive
20//! pointers that reference it.
21
22use core::cell::UnsafeCell;
23use core::ffi::c_void;
24use core::marker::PhantomData;
25use core::mem::MaybeUninit;
26
27use crate::PlatformThreading;
28
29/// Maximum bytes the wake primitive may occupy on any supported
30/// platform. Sized generously so we never have to bump for a new
31/// RTOS port. Asserted against the runtime probe in [`Wake::new`].
32pub const WAKE_STORAGE_BYTES: usize = 128;
33
34/// Alignment of the inline storage buffer. Wide enough for every
35/// pointer-aligned primitive on 64-bit hosts.
36pub const WAKE_STORAGE_ALIGN: usize = 16;
37
38#[repr(C, align(16))]
39struct WakeStorage {
40    bytes: UnsafeCell<[MaybeUninit<u8>; WAKE_STORAGE_BYTES]>,
41}
42
43/// Reason `wait_ms` returned.
44#[derive(Debug, Copy, Clone, PartialEq, Eq)]
45pub enum WakeReason {
46    /// `wake_signal` (or `wake_signal_from_isr`) fired and the
47    /// pending signal was consumed.
48    Signaled,
49    /// The timeout deadline expired before any signal.
50    Timeout,
51    /// Backend error (e.g. ISR-unsafe call on a platform that
52    /// rejects it). Treated as fatal in the executor's wake loop.
53    Error,
54}
55
56/// Errors at construction time.
57#[derive(Debug, Copy, Clone, PartialEq, Eq)]
58pub enum WakeInitError {
59    /// `P::wake_storage_size()` returned more bytes than the
60    /// inline buffer can hold. The platform impl needs more room
61    /// than the API contract reserved — bump
62    /// [`WAKE_STORAGE_BYTES`] or switch the consumer to a heap
63    /// path.
64    StorageTooSmall { needed: usize, available: usize },
65    /// `P::wake_storage_align()` returned a stricter alignment
66    /// than [`WAKE_STORAGE_ALIGN`].
67    AlignmentTooStrict { needed: usize, available: usize },
68    /// `P::wake_init` returned `-1`. Platform doesn't implement a
69    /// wake primitive (single-thread bare-metal).
70    Unsupported,
71}
72
73/// RAII wrapper around the platform's wake primitive.
74pub struct Wake<P: PlatformThreading> {
75    storage: WakeStorage,
76    _marker: PhantomData<P>,
77}
78
79// SAFETY: the wake contract is explicitly cross-thread. Backends
80// store either inline kernel structures (Zephyr k_sem, POSIX sem_t)
81// or pointers to handles owned by the kernel — both are safe to
82// access concurrently per their platform spec.
83unsafe impl<P: PlatformThreading> Sync for Wake<P> {}
84
85impl<P: PlatformThreading> Wake<P> {
86    /// Construct a new wake primitive in the inline buffer.
87    pub fn new() -> Result<Self, WakeInitError> {
88        let needed = P::wake_storage_size();
89        if needed == 0 {
90            return Err(WakeInitError::Unsupported);
91        }
92        if needed > WAKE_STORAGE_BYTES {
93            return Err(WakeInitError::StorageTooSmall {
94                needed,
95                available: WAKE_STORAGE_BYTES,
96            });
97        }
98        let align = P::wake_storage_align();
99        if align > WAKE_STORAGE_ALIGN {
100            return Err(WakeInitError::AlignmentTooStrict {
101                needed: align,
102                available: WAKE_STORAGE_ALIGN,
103            });
104        }
105
106        let storage = WakeStorage {
107            bytes: UnsafeCell::new([MaybeUninit::uninit(); WAKE_STORAGE_BYTES]),
108        };
109        let ptr = storage.bytes.get() as *mut c_void;
110        // `ptr` points at `WAKE_STORAGE_BYTES` aligned to 16 —
111        // both invariants verified above. `wake_init` either
112        // initialises the primitive in place or returns non-zero
113        // (in which case we drop `storage` without calling
114        // `wake_drop`, matching the "init failed, no teardown
115        // needed" contract).
116        let rc = P::wake_init(ptr);
117        if rc != 0 {
118            return Err(WakeInitError::Unsupported);
119        }
120        Ok(Self {
121            storage,
122            _marker: PhantomData,
123        })
124    }
125
126    /// Block until signaled or `timeout_ms` elapses.
127    pub fn wait_ms(&self, timeout_ms: u32) -> WakeReason {
128        let ptr = self.storage.bytes.get() as *mut c_void;
129        match P::wake_wait_ms(ptr, timeout_ms) {
130            0 => WakeReason::Signaled,
131            1 => WakeReason::Timeout,
132            _ => WakeReason::Error,
133        }
134    }
135
136    /// Wake one waiter. Safe to call from any thread.
137    pub fn signal(&self) {
138        let ptr = self.storage.bytes.get() as *mut c_void;
139        let _ = P::wake_signal(ptr);
140    }
141
142    /// ISR-safe variant. Returns `false` when the backend has no
143    /// ISR path and the caller should fall back to
144    /// [`Self::signal`].
145    pub fn signal_from_isr(&self) -> bool {
146        let ptr = self.storage.bytes.get() as *mut c_void;
147        P::wake_signal_from_isr(ptr) == 0
148    }
149}
150
151impl<P: PlatformThreading> Drop for Wake<P> {
152    fn drop(&mut self) {
153        let ptr = self.storage.bytes.get() as *mut c_void;
154        // ptr was init'd in `new`; this is the matching teardown
155        // call. Any in-flight `wait_ms` is the caller's bug —
156        // `Wake` is consumed by-value so callers cannot legally
157        // hold a `&self` past Drop.
158        let _ = P::wake_drop(ptr);
159    }
160}
