1//! C function table adapter for nros RMW backends.
2//!
3//! This crate provides a vtable-based bridge so that backends written in C,
4//! C++, Zig, Ada, or any language with a C-compatible ABI can implement the
5//! nros `Session` / `Publisher` / `Subscriber` / service traits without
6//! writing Rust code.
7//!
8//! # Usage (C backend implementor)
9//!
10//! 1. Include `<nros/rmw_vtable.h>`
11//! 2. Implement all function pointers in `nros_rmw_vtable_t`
12//! 3. Call `nros_rmw_cffi_register(&my_vtable)` before creating sessions
13//!
14//! # Usage (Rust consumer)
15//!
16//! Enable the `rmw-cffi` feature on `nros` and use `Executor<CffiSession>`.
17

34// Mirrors the macro constants in `<nros/rmw_ret.h>`. The C side uses

39/// Signed 32-bit status code mirroring the C `nros_rmw_ret_t` typedef.
40/// Zero on success; negative on error.
41pub type NrosRmwRet = i32;
42
43/// Operation completed successfully.
44pub const NROS_RMW_RET_OK: NrosRmwRet = 0;
45/// Generic failure not covered by a more specific code.
46pub const NROS_RMW_RET_ERROR: NrosRmwRet = -1;
47/// Operation deadline elapsed before completion.
48pub const NROS_RMW_RET_TIMEOUT: NrosRmwRet = -2;
49/// Memory allocation failed.
50pub const NROS_RMW_RET_BAD_ALLOC: NrosRmwRet = -3;
51/// Caller supplied a NULL pointer or an out-of-range value.
52pub const NROS_RMW_RET_INVALID_ARGUMENT: NrosRmwRet = -4;
53/// The backend does not implement this operation.
54pub const NROS_RMW_RET_UNSUPPORTED: NrosRmwRet = -5;
55/// QoS profiles incompatible in a way the backend cannot reconcile.
56pub const NROS_RMW_RET_INCOMPATIBLE_QOS: NrosRmwRet = -6;
57/// Topic, service, or action name failed validation.
58pub const NROS_RMW_RET_TOPIC_NAME_INVALID: NrosRmwRet = -7;
59/// Request referenced a node that does not exist in this session.
60pub const NROS_RMW_RET_NODE_NAME_NON_EXISTENT: NrosRmwRet = -8;
61/// Backend does not support loaned messages on this entity, or slot in use.
62pub const NROS_RMW_RET_LOAN_NOT_SUPPORTED: NrosRmwRet = -9;
63/// No data on a non-blocking receive (distinct from `TIMEOUT`).
64pub const NROS_RMW_RET_NO_DATA: NrosRmwRet = -10;
65/// Resource momentarily unavailable; caller should retry.
66pub const NROS_RMW_RET_WOULD_BLOCK: NrosRmwRet = -11;
67/// Caller buffer smaller than the data the backend wants to deliver.
68pub const NROS_RMW_RET_BUFFER_TOO_SMALL: NrosRmwRet = -12;
69/// Incoming message exceeded the backend's static capacity.
70pub const NROS_RMW_RET_MESSAGE_TOO_LARGE: NrosRmwRet = -13;

72/// Map a `TransportError` to the corresponding `nros_rmw_ret_t` code.
73///
74/// By-reference because `TransportError` carries a `String` on its
75/// dynamic-diagnostic variant and is not `Copy`. The string itself is
76/// dropped at the boundary — embedded RMW callers cannot afford a
77/// thread-local error buffer.
78pub fn ret_from_error(err: &TransportError) -> NrosRmwRet {
79    match err {
80        TransportError::Timeout => NROS_RMW_RET_TIMEOUT,
81        TransportError::WouldBlock => NROS_RMW_RET_WOULD_BLOCK,
82        TransportError::TooLarge => NROS_RMW_RET_MESSAGE_TOO_LARGE,
83        TransportError::BufferTooSmall => NROS_RMW_RET_BUFFER_TOO_SMALL,
84        TransportError::MessageTooLarge => NROS_RMW_RET_MESSAGE_TOO_LARGE,
85        TransportError::InvalidArgument => NROS_RMW_RET_INVALID_ARGUMENT,
86        TransportError::InvalidConfig => NROS_RMW_RET_INVALID_ARGUMENT,
87        TransportError::Unsupported => NROS_RMW_RET_UNSUPPORTED,
88        TransportError::BadAlloc => NROS_RMW_RET_BAD_ALLOC,
89        TransportError::IncompatibleQos => NROS_RMW_RET_INCOMPATIBLE_QOS,
90        TransportError::TopicNameInvalid => NROS_RMW_RET_TOPIC_NAME_INVALID,
91        TransportError::NodeNameNonExistent => NROS_RMW_RET_NODE_NAME_NON_EXISTENT,
92        TransportError::LoanNotSupported => NROS_RMW_RET_LOAN_NOT_SUPPORTED,
93        TransportError::NoData => NROS_RMW_RET_NO_DATA,
94        // Everything else collapses to NROS_RMW_RET_ERROR. Backends
95        // that want fine-grained reporting should adopt the named
96        // variants above (Phase 102.2 sweep).
97        _ => NROS_RMW_RET_ERROR,
98    }
99}

101/// Map a `nros_rmw_ret_t` returned by a C-side vtable function back to
102/// a `TransportError` for the Rust caller. Inverse of `ret_from_error`
103/// — used when `nros-rmw-cffi`'s `CffiSession` etc. receive a code
104/// from the registered C backend.
105///
106/// `NROS_RMW_RET_OK` is mapped to `TransportError::Backend("ok")` as a
107/// programming-error sentinel; callers should branch on the success
108/// path before calling this. Unknown negative codes collapse to the
109/// generic `TransportError::Backend("unknown rmw_ret_t")` so a future
110/// constant added to the C header degrades gracefully on the Rust side.
111pub fn error_from_ret(ret: NrosRmwRet) -> TransportError {
112    match ret {
113        NROS_RMW_RET_OK => {
114            TransportError::Backend("ok (logic error: positive ret_t at error site)")
115        }
116        NROS_RMW_RET_ERROR => TransportError::Backend("rmw_ret error"),
117        NROS_RMW_RET_TIMEOUT => TransportError::Timeout,
118        NROS_RMW_RET_BAD_ALLOC => TransportError::BadAlloc,
119        NROS_RMW_RET_INVALID_ARGUMENT => TransportError::InvalidArgument,
120        NROS_RMW_RET_UNSUPPORTED => TransportError::Unsupported,
121        NROS_RMW_RET_INCOMPATIBLE_QOS => TransportError::IncompatibleQos,
122        NROS_RMW_RET_TOPIC_NAME_INVALID => TransportError::TopicNameInvalid,
123        NROS_RMW_RET_NODE_NAME_NON_EXISTENT => TransportError::NodeNameNonExistent,
124        NROS_RMW_RET_LOAN_NOT_SUPPORTED => TransportError::LoanNotSupported,
125        NROS_RMW_RET_NO_DATA => TransportError::NoData,
126        NROS_RMW_RET_WOULD_BLOCK => TransportError::WouldBlock,
127        NROS_RMW_RET_BUFFER_TOO_SMALL => TransportError::BufferTooSmall,
128        NROS_RMW_RET_MESSAGE_TOO_LARGE => TransportError::MessageTooLarge,
129        _ => TransportError::Backend("unknown rmw_ret_t"),
130    }
131}

519/// Register a custom RMW backend vtable.
520///
521/// # Safety
522///
523/// The vtable pointer must remain valid for the lifetime of the program.
524/// All function pointers in the vtable must be valid.
525#[unsafe(no_mangle)]
526pub unsafe extern "C" fn nros_rmw_cffi_register(vtable: *const NrosRmwVtable) -> NrosRmwRet {
527    VTABLE.store(vtable as *mut NrosRmwVtable, Ordering::Release);
528    NROS_RMW_RET_OK
529}

531fn get_vtable() -> Result<&'static NrosRmwVtable, TransportError> {
532    let ptr = VTABLE.load(Ordering::Acquire);
533    if ptr.is_null() {
534        // No vtable registered — caller forgot nros_rmw_cffi_register.
535        return Err(TransportError::InvalidArgument);
536    }
537    // SAFETY: Registration ensures the pointer is valid and 'static.
538    Ok(unsafe { &*ptr })
539}

545/// Write a Rust `&str` as a null-terminated byte sequence into a fixed buffer.
546/// Returns a pointer to the buffer start.
547fn to_c_str<const N: usize>(s: &str, buf: &mut [u8; N]) -> *const u8 {
548    let len = s.len().min(N - 1);
549    buf[..len].copy_from_slice(&s.as_bytes()[..len]);
550    buf[len] = 0;
551    buf.as_ptr()
552}

555/// as a `&str`, stopping at the first NUL byte. Used by the
556/// `topic_name()` / `type_name()` / `node_name()` accessors on the
557/// `Cffi*` types so callers can introspect without round-tripping
558/// through the vtable. Phase 102.5.
559fn cstr_buf_to_str<const N: usize>(buf: &[u8; N]) -> &str {
560    let len = buf.iter().position(|&b| b == 0).unwrap_or(N);
561    // The buffers are written via `to_c_str` from a `&str`, so the
562    // bytes between [..len] are guaranteed valid UTF-8. `from_utf8`
563    // handles the (impossible) corruption case by returning empty.
564    core::str::from_utf8(&buf[..len]).unwrap_or("")
565}

567// ============================================================================
568// CffiSession
569// ============================================================================
570//
571// Storage discipline:
572// * Each Cffi* struct owns null-terminated name buffers as inline
573//   arrays. The C-side typed entity struct is rebuilt fresh on every
574//   FFI call via `make_*_view`, so move-invalidation of pointers
575//   into the buffer is impossible — the pointer always points to the
576//   *current* address of the buffer, computed at call time.
577// * The backend writes `backend_data` (and `can_loan_messages` for
578//   pub/sub entities)
579//   into the FFI view; we copy the writes back into the Cffi*
580//   struct's fields after the call.
581// * Strings ARE immutable for the entity's lifetime, so backends that
582//   stash the topic_name pointer for diagnostics see stable storage
583//   *as long as the Cffi* struct is not moved.* The Phase 102.4
584//   contract is "do not move a Cffi* struct after construction" —
585//   nano-ros embeds them inside the executor arena, which doesn't
586//   relocate.
587
588const NAME_BUF_LEN: usize = 256;
589const HASH_BUF_LEN: usize = 128;

591/// Session backed by a C vtable.
592pub struct CffiSession {
593    vtable: &'static NrosRmwVtable,
594    /// Borrowed-pointer storage for `node_name`. Outlives the session.
595    node_name_buf: [u8; NAME_BUF_LEN],
596    /// Borrowed-pointer storage for `namespace_`. Empty for now —
597    /// `RmwConfig` does not yet carry a namespace through the cffi
598    /// path; reserved for future use.
599    namespace_buf: [u8; NAME_BUF_LEN],
600    /// Backend-private state, written by `vtable.open`.
601    backend_data: *mut c_void,
602}
603
604impl CffiSession {
605    fn make_view(&mut self) -> NrosRmwSession {
606        NrosRmwSession {
607            node_name: self.node_name_buf.as_ptr(),
608            namespace_: self.namespace_buf.as_ptr(),
609            _reserved: [0u8; 8],
610            backend_data: self.backend_data,
611        }
612    }
613
614    /// Node name passed at session-open time.
615    pub fn node_name(&self) -> &str {
616        cstr_buf_to_str(&self.node_name_buf)
617    }
618
619    /// Open a new session via the registered vtable.
620    pub fn open(
621        locator: &str,
622        mode: u8,
623        domain_id: u32,
624        node_name: &str,
625    ) -> Result<Self, TransportError> {
626        let vtable = get_vtable()?;
627        let mut loc_buf = [0u8; NAME_BUF_LEN];
628        let loc_ptr = to_c_str(locator, &mut loc_buf);
629
630        let mut session = Self {
631            vtable,
632            node_name_buf: [0u8; NAME_BUF_LEN],
633            namespace_buf: [0u8; NAME_BUF_LEN],
634            backend_data: core::ptr::null_mut(),
635        };
636        let _ = to_c_str(node_name, &mut session.node_name_buf);
637
638        let mut view = NrosRmwSession {
639            node_name: session.node_name_buf.as_ptr(),
640            namespace_: session.namespace_buf.as_ptr(),
641            _reserved: [0u8; 8],
642            backend_data: core::ptr::null_mut(),
643        };
644        let ret = unsafe {
645            (vtable.open)(
646                loc_ptr,
647                mode,
648                domain_id,
649                session.node_name_buf.as_ptr(),
650                &mut view,
651            )
652        };
653        if ret != NROS_RMW_RET_OK {
654            return Err(error_from_ret(ret));
655        }
656        if view.backend_data.is_null() {
657            return Err(TransportError::ConnectionFailed);
658        }
659        session.backend_data = view.backend_data;
660        Ok(session)
661    }
662}

1201                request.as_ptr(),
1202                request.len(),
1203                reply_buf.as_mut_ptr(),
1204                reply_buf.len(),
1205            )
1206        };
1207        if rc < 0 {
1208            return Err(error_from_ret(rc));
1209        }
1210        Ok(rc as usize)
1211    }
1212
1213    fn send_request_raw(&mut self, request: &[u8]) -> Result<(), TransportError> {
1214        if request.len() > self.pending_request.len() {
1215            return Err(TransportError::BufferTooSmall);
1216        }
1217        self.pending_request[..request.len()].copy_from_slice(request);
1218        self.pending_len = request.len();
1219        Ok(())
1220    }
1221
1222    fn try_recv_reply_raw(
1223        &mut self,
1224        reply_buf: &mut [u8],
1225    ) -> Result<Option<usize>, TransportError> {
1226        if self.pending_len == 0 {
1227            return Ok(None);
1228        }
1229        // Blocking fallback: copy request to stack, then call_raw
1230        let mut req_copy = [0u8; 4096];
1231        let req_len = self.pending_len;
1232        req_copy[..req_len].copy_from_slice(&self.pending_request[..req_len]);
1233        self.pending_len = 0;
1234        #[allow(deprecated)]
1235        let len = self.call_raw(&req_copy[..req_len], reply_buf)?;
1236        Ok(Some(len))
1237    }
1238}

1240impl Drop for CffiServiceClient {
1241    fn drop(&mut self) {
1242        if !self.backend_data.is_null() {
1243            let mut view = self.make_view();
1244            unsafe { (self.vtable.destroy_service_client)(&mut view) };
1245        }
1246    }
