1//! Linker-section registry discovery (Phase 128.A).
2//!
3//! At static link time every `nros-rmw-<name>` crate or static lib
4//! contributes exactly one [`RmwInitEntry`] function pointer to the
5//! [`RMW_INIT_ENTRIES`] distributed slice. The runtime calls
6//! [`walk_init_section`] / [`nros_rmw_cffi_walk_init_section`] on
7//! first `Executor::open` / `nros::init` to invoke every entry; each
8//! entry in turn calls [`crate::nros_rmw_cffi_register_named`] with
9//! its canonical name and vtable pointer.
10//!
11//! This is *static* discovery — no `dlopen`, no plugin system. The
12//! linker section is populated at link time and frozen for the
13//! program's lifetime.
14//!
15//! The section anchoring (cross-platform `__start_/__stop_`
16//! synthesis, `KEEP` semantics under `--gc-sections`, Mach-O
17//! `__DATA,__nros_rmw_init` placement, COFF chunked sections) is
18//! delegated to the [`linkme`] crate so the discipline works
19//! identically on every target the crate supports: Linux / macOS /
20//! Windows / FreeBSD / illumos / `target_os = "none"` (bare-metal
21//! ELF).
22//!
23//! # Unsupported targets (RTOS / ESP-IDF)
24//!
25//! Targets `linkme` does NOT recognise (`target_os = "nuttx"`,
26//! `"zephyr"`, `"espidf"`, `"vxworks"`, …) fall back to a no-op
27//! [`RMW_INIT_ENTRIES`] stub and a walker that always returns 0.
28//! Backends on those targets still register via the explicit
29//! `nros_rmw_<x>::register()` call from main (the rlib-pull anchor
30//! pattern documented in phase 128.B.1's commit) — the section
31//! walker just doesn't add anything on top.
32//!
33//! # C / C++ backends
34//!
35//! Static-lib backends emit their entry via the
36//! [`NROS_RMW_REGISTER_BACKEND`] macro in `<nros/rmw_vtable.h>`. The
37//! macro lands the function pointer in `linkm2_RMW_INIT_ENTRIES`
38//! (the section name `linkme` uses for [`RMW_INIT_ENTRIES`]),
39//! interoperating with the Rust-side entries through pure linker
40//! discipline.
41//!
42//! [`NROS_RMW_REGISTER_BACKEND`]: ../../include/nros/rmw_vtable.h
43

48/// Public type of every entry in [`RMW_INIT_ENTRIES`]. The function
49/// takes no arguments, returns nothing, and is expected to call
50/// [`crate::nros_rmw_cffi_register_named`] with its own canonical
51/// name + vtable. A non-zero return from the registration call is
52/// silently ignored here — the walker only guarantees that every
53/// entry is invoked; per-backend failures surface later at
54/// `Executor::open` when the registry lookup misses.
55pub type RmwInitEntry = unsafe extern "C" fn();

147/// Macro emitted by backend crates to contribute one entry to
148/// [`RMW_INIT_ENTRIES`]. Expands to a `linkme::distributed_slice`
149/// item on targets `linkme` supports; expands to nothing on
150/// unsupported targets (NuttX, Zephyr, ESP-IDF, …) where the
151/// section walker is a no-op and backends must rely on the
152/// explicit `register()` call pattern.
153///
154/// Usage (inside the backend crate):
155///
156/// ```ignore
157/// nros_rmw_cffi::nros_rmw_register_backend! {
158///     fn() { let _ = nros_rmw_zenoh_register(); }
159/// }
160/// ```
161#[macro_export]
162macro_rules! nros_rmw_register_backend {
163    (fn() $body:block) => {
164        // Phase 134.fix — also gate on the `linkme-register` Cargo
165        // feature of `nros-rmw-cffi`. When the consuming staticlib
166        // builds with `default-features = false`, the macro expands
167        // to nothing (no linkme entry). Auto-registration falls to
168        // an explicit `.init_array` ctor in the staticlib's lib.rs.
169        #[cfg(all(
170            feature = "linkme-register",
171            any(
172                target_os = "linux",
173                target_os = "android",
174                target_os = "fuchsia",
175                target_os = "psp",
176                target_os = "freebsd",
177                target_os = "openbsd",
178                target_os = "macos",
179                target_os = "ios",
180                target_os = "tvos",
181                target_os = "watchos",
182                target_os = "windows",
183                target_os = "illumos",
184                // Phase 142 — `target_os = "none"` dropped here
185                // too so the macro expansion in bare-metal
186                // backend crates also collapses to a no-op,
187                // matching the mod-gating above.
188            )
189        ))]
190        const _: () = {
191            unsafe extern "C" fn __nros_rmw_backend_section_entry() $body
192            #[$crate::linkme::distributed_slice($crate::RMW_INIT_ENTRIES)]
193            static __NROS_RMW_BACKEND_SECTION_ENTRY: $crate::RmwInitEntry =
194                __nros_rmw_backend_section_entry;
195        };
196    };
197}

199/// Idempotency guard. Set after the first successful walk. Subsequent
200/// `Executor::open` calls skip re-walking.
201static WALKED: AtomicBool = AtomicBool::new(false);
202
203/// Walk every entry in [`RMW_INIT_ENTRIES`]. Idempotent — subsequent
204/// calls are no-ops. Safe to invoke from `Executor::open` /
205/// `nros::init` on every entry into the runtime.
206///
207/// Returns the number of entries actually invoked on this call. A
208/// return of 0 from the first call indicates no `nros-rmw-*` backend
209/// was linked into this binary; the caller (typically
210/// `Executor::open`'s resolution policy) should surface
211/// [`crate::NROS_RMW_RET_NO_BACKEND`].
212///
213/// # Safety
214///
215/// Every entry in `RMW_INIT_ENTRIES` is contributed by an in-tree
216/// `nros-rmw-<name>` crate (or static lib) via the
217/// `nros_rmw_register_backend!` macro / `NROS_RMW_REGISTER_BACKEND`
218/// C macro; both expand to a no-arg `extern "C" fn` whose only side
219/// effect is calling [`crate::nros_rmw_cffi_register_named`]. Calling
220/// the walker is therefore safe provided no third-party code has
221/// injected a non-conforming entry into the same linker section.
222#[unsafe(no_mangle)]
223pub unsafe extern "C" fn nros_rmw_cffi_walk_init_section() -> usize {
224    if WALKED.swap(true, Ordering::AcqRel) {
225        return 0;
226    }
227    let mut invoked = 0usize;
228    for entry in RMW_INIT_ENTRIES.iter() {
229        // SAFETY: every backend's ctor calls
230        // `nros_rmw_cffi_register_named`, which is documented as
231        // safe to call before any session creation. The walker fires
232        // before `Executor::open` finishes; nothing else races.
233        unsafe { entry() };
234        invoked += 1;
235    }
236    invoked
237}

242/// fire-once.
243#[cfg(test)]
244#[allow(dead_code)]
245pub(crate) fn reset_walked_for_test() {
