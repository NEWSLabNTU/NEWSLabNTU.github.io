1//! Phase 115.L.0 — Generic Rust-trait → C-vtable adapter.
2//!
3//! `RustBackendAdapter<R>` converts any `R: nros_rmw::Rmw` whose
4//! associated types implement the matching `Session` / `Publisher` /
5//! `Subscriber` / `ServiceServerTrait` / `ServiceClientTrait` traits
6//! into a `static NrosRmwVtable`. Each per-backend cffi crate then
7//! collapses to ~10 LOC:
8//!
9//! ```ignore
10//! #[unsafe(no_mangle)]
11//! pub extern "C" fn nros_rmw_dds_register() -> nros_rmw_cffi::NrosRmwRet {
12//!     nros_rmw_cffi::RustBackendAdapter::<nros_rmw_dds::DdsRmw>::register()
13//! }
14//! ```
15//!
16//! # Storage discipline
17//!
18//! - Session: `Box::into_raw(Box::new(session))` is stashed in
19//!   `NrosRmwSession::backend_data`. `close` reclaims via
20//!   `Box::from_raw` (drops the box, runs `Drop`, frees the alloc).
21//! - Publisher / Subscriber / ServiceServer / ServiceClient: same
22//!   pattern with their respective handle types.
23//!
24//! # 'static
25//!
26//! Every handle must be `'static` (the `Box` outlives the call). We
27//! deliberately do **not** require `Send`: the C runtime hands the
28//! `backend_data` pointer back to the same caller that minted it,
29//! and the executor's single-thread-per-session invariant matches
30//! the Rust trait surface. Zenoh-pico's `ZenohSession`, for
31//! instance, holds an unmovable `*const Context` which the
32//! upstream library does not mark `Send`.
33//!
34//! # Bounds
35//!
36//! All error types must be `TransportError` (or `Into<TransportError>`).
37//! This matches every in-tree backend today.
38

56// ============================================================================
57// Trait alias bundle
58// ============================================================================
59//
60// `RustBackend` ties together every constraint the adapter needs. The
61// alternative — long `where` clauses on every fn — is unreadable. This
62// trait is sealed: backends don't impl it directly; the blanket impl
63// below picks it up automatically for any `R: Rmw` whose associated
64// types line up.
65
66/// Bundle of trait bounds an `Rmw` backend must satisfy to be exposed
67/// through [`RustBackendAdapter`]. Implemented automatically for any
68/// `R: Rmw` whose handle types use `TransportError` and are
69/// `'static`.
70pub trait RustBackend: Sized {
71    type Session: Session<
72            Error = TransportError,
73            PublisherHandle = Self::Publisher,
74            SubscriberHandle = Self::Subscriber,
75            ServiceServerHandle = Self::ServiceServer,
76            ServiceClientHandle = Self::ServiceClient,
77        > + 'static;
78    type Publisher: Publisher<Error = TransportError> + 'static;
79    type Subscriber: Subscriber<Error = TransportError> + 'static;
80    type ServiceServer: ServiceServerTrait<Error = TransportError> + 'static;
81    type ServiceClient: ServiceClientTrait<Error = TransportError> + 'static;
82
83    /// Construct a fresh factory instance. Called inside the `open`
84    /// trampoline. Equivalent to `R::default()` for backends that
85    /// `derive(Default)`; the indirection keeps the door open for
86    /// future per-backend registration knobs.
87    fn factory() -> Self;
88
89    /// Move the factory into a session per the `Rmw::open` contract.
90    fn open(self, config: &RmwConfig) -> Result<Self::Session, TransportError>;
91}
92
93impl<R> RustBackend for R
94where
95    R: Rmw<Error = TransportError> + Default + Sized,
96    R::Session: Session<Error = TransportError> + 'static,
97    <R::Session as Session>::PublisherHandle: Publisher<Error = TransportError> + 'static,
98    <R::Session as Session>::SubscriberHandle: Subscriber<Error = TransportError> + 'static,
99    <R::Session as Session>::ServiceServerHandle:
100        ServiceServerTrait<Error = TransportError> + 'static,
101    <R::Session as Session>::ServiceClientHandle:
102        ServiceClientTrait<Error = TransportError> + 'static,
103{
104    type Session = R::Session;
105    type Publisher = <R::Session as Session>::PublisherHandle;
106    type Subscriber = <R::Session as Session>::SubscriberHandle;
107    type ServiceServer = <R::Session as Session>::ServiceServerHandle;
108    type ServiceClient = <R::Session as Session>::ServiceClientHandle;
109
110    fn factory() -> Self {
111        R::default()
112    }
113
114    fn open(self, config: &RmwConfig) -> Result<Self::Session, TransportError> {
115        Rmw::open(self, config)
116    }
117}

123/// Read a null-terminated C string into a Rust `&str`. Returns the
124/// empty string if `ptr` is null or contains invalid UTF-8.
125///
126/// # Safety
127///
128/// `ptr`, if non-null, must point to a valid null-terminated byte
129/// sequence that outlives the returned borrow.
130unsafe fn cstr_to_str<'a>(ptr: *const u8) -> &'a str {
131    if ptr.is_null() {
132        return "";
133    }
134    let mut len = 0usize;
135    // Bound the scan so a missing terminator can't read off the end of
136    // a small caller buffer. 1 KiB matches the typical NAME_BUF_LEN
137    // (256) plus headroom for type-name / hash strings.
138    while len < 4096 && unsafe { *ptr.add(len) } != 0 {
139        len += 1;
140    }
141    let slice = unsafe { core::slice::from_raw_parts(ptr, len) };
142    core::str::from_utf8(slice).unwrap_or("")
143}

145/// Convert the cffi QoS view back into a `nros_rmw::QosSettings`. The
146/// adapter trampolines call this when forwarding `create_publisher` /
147/// `create_subscriber` into the Rust trait.
148fn qos_from_cffi(q: &NrosRmwQos) -> QosSettings {
149    use nros_rmw::{
150        QosDurabilityPolicy, QosHistoryPolicy, QosLivelinessPolicy, QosReliabilityPolicy,
151    };
152    QosSettings {
153        reliability: if q.reliability == 0 {
154            QosReliabilityPolicy::BestEffort
155        } else {
156            QosReliabilityPolicy::Reliable
157        },
158        durability: if q.durability == 0 {
159            QosDurabilityPolicy::Volatile
160        } else {
161            QosDurabilityPolicy::TransientLocal
162        },
163        history: if q.history == 0 {
164            QosHistoryPolicy::KeepLast
165        } else {
166            QosHistoryPolicy::KeepAll
167        },
168        depth: q.depth as u32,
169        deadline_ms: q.deadline_ms,
170        lifespan_ms: q.lifespan_ms,
171        liveliness_kind: match q.liveliness_kind {
172            1 => QosLivelinessPolicy::Automatic,
173            2 => QosLivelinessPolicy::ManualByTopic,
174            3 => QosLivelinessPolicy::ManualByNode,
175            _ => QosLivelinessPolicy::None,
176        },
177        liveliness_lease_ms: q.liveliness_lease_ms,
178        avoid_ros_namespace_conventions: q.avoid_ros_namespace_conventions != 0,
179    }
180}

216        next_deadline_ms: Some(next_deadline_ms_trampoline::<R>),
217        set_wake_callback: Some(set_wake_callback_trampoline::<R>),
218        // Phase 124.A — zero-copy slots default to NULL on the
219        // generic adapter; per-backend opt-in via dedicated trampolines
220        // (see `nros-rmw-zenoh` for the first implementation in 124.A.4).
221        // Runtime falls back to the arena path when these are NULL.
222        pub_loan: None,

231    };

233    /// Install the per-`R` vtable into the cffi registry under the
234    /// implicit name `"default"`. Equivalent to calling
235    /// `nros_rmw_cffi_register(&Self::VTABLE)`. Idempotent — re-
236    /// registering the same vtable is a no-op from the runtime's
237    /// perspective.
238    ///
239    /// Most backends should use [`register_named`](Self::register_named)
240    /// instead so they show up in the registry under their canonical
241    /// name (`"zenoh"`, `"dds"`, `"xrce"`, …).
242    pub fn register() -> NrosRmwRet {
243        // SAFETY: `&Self::VTABLE` is a reference to a const-promoted
244        // static; the address is stable for the life of the program.
245        unsafe { nros_rmw_cffi_register(&Self::VTABLE) }
246    }
247
248    /// Phase 104.B.2 — install the per-`R` vtable under a stable
249    /// name. Multiple backends can coexist via this entry point.
250    ///
251    /// # Safety
252    /// `name` must be a valid NUL-terminated UTF-8 string.
253    pub unsafe fn register_named(name: *const core::ffi::c_char) -> NrosRmwRet {
254        // SAFETY: `&Self::VTABLE` is a reference to a const-promoted
255        // static; address stable for the program's lifetime.
256        unsafe { crate::nros_rmw_cffi_register_named(name, &Self::VTABLE) }
257    }
258}

264unsafe extern "C" fn open_trampoline<R: RustBackend>(
265    locator: *const u8,
266    mode: u8,
267    domain_id: u32,
268    node_name: *const u8,
269    out: *mut NrosRmwSession,
270) -> NrosRmwRet {
271    if out.is_null() {
272        return NROS_RMW_RET_INVALID_ARGUMENT;
273    }
274    let cfg = RmwConfig {
275        locator: unsafe { cstr_to_str(locator) },
276        mode: if mode == 0 {
277            SessionMode::Client
278        } else {
279            SessionMode::Peer
280        },
281        domain_id,
282        node_name: unsafe { cstr_to_str(node_name) },
283        namespace: "",
284        properties: &[],
285    };
286    let factory = R::factory();
287    match factory.open(&cfg) {
288        Ok(session) => {
289            let boxed = Box::into_raw(Box::new(session));
290            unsafe {
291                (*out).backend_data = boxed as *mut c_void;
292            }
293            NROS_RMW_RET_OK
294        }
295        Err(e) => ret_from_error(&e),
296    }
297}
298
299unsafe extern "C" fn close_trampoline<R: RustBackend>(session: *mut NrosRmwSession) -> NrosRmwRet {
300    let Some(boxed) = (unsafe { take_box::<R::Session>(session_data_mut(session)) }) else {
301        return NROS_RMW_RET_INVALID_ARGUMENT;
302    };
303    let mut s = boxed;
304    let ret = match Session::close(&mut *s) {
305        Ok(()) => NROS_RMW_RET_OK,
306        Err(e) => ret_from_error(&e),
307    };
308    drop(s);
309    ret
310}

435unsafe extern "C" fn try_recv_raw_trampoline<R: RustBackend>(
436    subscriber: *mut NrosRmwSubscriber,
437    buf: *mut u8,
438    buf_len: usize,
439) -> i32 {
440    let Some(s) = (unsafe { subscriber_mut::<R::Subscriber>(subscriber) }) else {

451    }

507unsafe extern "C" fn try_recv_request_trampoline<R: RustBackend>(
508    server: *mut NrosRmwServiceServer,
509    buf: *mut u8,
510    buf_len: usize,
511    seq_out: *mut i64,
512) -> i32 {
513    let Some(s) = (unsafe { service_server_mut::<R::ServiceServer>(server) }) else {
514        return NROS_RMW_RET_INVALID_ARGUMENT;
515    };
516    if (buf.is_null() && buf_len != 0) || seq_out.is_null() {
517        return NROS_RMW_RET_INVALID_ARGUMENT;
518    }
519    let slice = unsafe { core::slice::from_raw_parts_mut(buf, buf_len) };
520    let buf_start = slice.as_ptr() as usize;
521    match ServiceServerTrait::try_recv_request(s, slice) {
522        Ok(Some(req)) => {
523            // The handle's `data` slice borrows from `buf`. Use its
524            // offset within the caller's buffer to compute the payload
525            // length; some backends prepend an envelope header so the
526            // payload doesn't start at offset 0.
527            let offset = (req.data.as_ptr() as usize).saturating_sub(buf_start);
528            let len = req.data.len();
529            unsafe {
530                *seq_out = req.sequence_number;
531            }
532            // Move the payload to the start of the buffer so the C
533            // caller sees a `(buf, len)` pair starting at offset 0,
534            // matching the cyclonedds backend's contract.
535            if offset != 0 {
536                let total = offset + len;
537                // SAFETY: `slice[..total]` is initialised by the
538                // backend; copy_within respects overlapping ranges.
539                slice.copy_within(offset..total, 0);
540            }
541            len as i32
542        }
543        Ok(None) => NROS_RMW_RET_NO_DATA,
544        Err(e) => ret_from_error(&e),
545    }
546}

594    }
595    let Some(s) = (unsafe { session_mut::<R::Session>(session) }) else {
596        return NROS_RMW_RET_INVALID_ARGUMENT;
597    };
598    let info = nros_rmw::ServiceInfo {
599        name: unsafe { cstr_to_str(service_name) },
600        type_name: unsafe { cstr_to_str(type_name) },
601        type_hash: unsafe { cstr_to_str(type_hash) },
602        domain_id,
603        node_name: None,
604        namespace: "/",
605    };
606    match Session::create_service_client(s, &info) {
607        Ok(client) => {
608            let boxed = Box::into_raw(Box::new(client));
609            unsafe {
610                (*out).backend_data = boxed as *mut c_void;
611            }
612            NROS_RMW_RET_OK
613        }
614        Err(e) => ret_from_error(&e),
615    }
616}

624unsafe extern "C" fn call_raw_trampoline<R: RustBackend>(
625    client: *mut NrosRmwServiceClient,
626    request: *const u8,
627    req_len: usize,
628    reply_buf: *mut u8,
629    reply_buf_len: usize,
630) -> i32 {
631    let Some(c) = (unsafe { service_client_mut::<R::ServiceClient>(client) }) else {
632        return NROS_RMW_RET_INVALID_ARGUMENT;
633    };
634    if (request.is_null() && req_len != 0) || (reply_buf.is_null() && reply_buf_len != 0) {
635        return NROS_RMW_RET_INVALID_ARGUMENT;
636    }
637    let req = unsafe { core::slice::from_raw_parts(request, req_len) };
638    let reply = unsafe { core::slice::from_raw_parts_mut(reply_buf, reply_buf_len) };
639    // `call_raw` is marked deprecated on the trait (the executor-driven
640    // `Client::call` → `Promise::wait` path is the supported flow), but
641    // the cffi vtable still exposes a blocking call entry point for C
642    // consumers that don't have an executor handle. The deprecation is
643    // about API ergonomics, not correctness, so the adapter forwards
644    // through unchanged.
645    #[allow(deprecated)]
646    match ServiceClientTrait::call_raw(c, req, reply) {
647        Ok(n) => n as i32,
648        Err(e) => ret_from_error(&e),
649    }
650}

656// ============================================================================
657// Event-callback bridge (Phase 115.L.0.events)
658// ============================================================================
659//
660// `NrosRmwEventCallback` (cffi shape) and `nros_rmw::EventCallback`
661// (trait shape) have *layout-identical* arguments:
662//
663//   * NrosRmwEventKind  ↔ EventKind     — both `#[repr(u8)]`, same variant ids.
664//   * *const NrosRmwEventPayload ↔ *const c_void  — fn-ptr-width pointer; the
665//     trait callback dereferences as `*const LivelinessChangedStatus` /
666//     `*const CountStatus`, the cffi callback dereferences as the matching
667//     union member. The field-by-field layout matches today (see the
668//     `_event_payload_layout_match` const-asserts below).
669//   * *mut c_void       ↔ *mut c_void   — identical.
670//
671// Therefore the cffi callback can be transmuted into a trait callback
672// pointer; the receiving Rust trait code calls it via the trait
673// signature, but the bytes on the wire are the cffi struct.
674
675const _: () = {
676    use core::mem::{align_of, size_of};
677    assert!(
678        size_of::<crate::NrosRmwLivelinessChangedStatus>()
679            == size_of::<nros_rmw::LivelinessChangedStatus>()
680    );
681    assert!(
682        align_of::<crate::NrosRmwLivelinessChangedStatus>()
683            == align_of::<nros_rmw::LivelinessChangedStatus>()
684    );
685    assert!(size_of::<crate::NrosRmwCountStatus>() == size_of::<nros_rmw::CountStatus>());
686    assert!(align_of::<crate::NrosRmwCountStatus>() == align_of::<nros_rmw::CountStatus>());
687    // EventKind enum tags must round-trip 0..=4 between the two
688    // `#[repr(u8)]` definitions.
689    assert!(
690        NrosRmwEventKind::LivelinessChanged as u8 == nros_rmw::EventKind::LivelinessChanged as u8
691    );
692    assert!(
693        NrosRmwEventKind::RequestedDeadlineMissed as u8
694            == nros_rmw::EventKind::RequestedDeadlineMissed as u8
695    );
696    assert!(NrosRmwEventKind::MessageLost as u8 == nros_rmw::EventKind::MessageLost as u8);
697    assert!(NrosRmwEventKind::LivelinessLost as u8 == nros_rmw::EventKind::LivelinessLost as u8);
698    assert!(
699        NrosRmwEventKind::OfferedDeadlineMissed as u8
700            == nros_rmw::EventKind::OfferedDeadlineMissed as u8
701    );
702};

704unsafe extern "C" fn register_subscriber_event_trampoline<R: RustBackend>(
705    subscriber: *mut NrosRmwSubscriber,
706    kind: NrosRmwEventKind,
707    deadline_ms: u32,
708    cb: NrosRmwEventCallback,
709    user_context: *mut c_void,
710) -> NrosRmwRet {
711    let Some(s) = (unsafe { subscriber_mut::<R::Subscriber>(subscriber) }) else {
712        return NROS_RMW_RET_INVALID_ARGUMENT;
713    };
714    // SAFETY: cffi `NrosRmwEventKind` and trait `EventKind` are both
715    // `#[repr(u8)]` with identical variant numbering (asserted at
716    // compile time above).
717    let trait_kind: nros_rmw::EventKind = unsafe { core::mem::transmute(kind) };
718    // SAFETY: see module-level note. `cb`'s parameter types
719    // `(NrosRmwEventKind, *const NrosRmwEventPayload, *mut c_void)` are
720    // ABI-compatible with the trait's `(EventKind, *const c_void, *mut c_void)`.
721    let trait_cb: nros_rmw::EventCallback = unsafe { core::mem::transmute(cb) };
722    let res = unsafe {
723        Subscriber::register_event_callback(s, trait_kind, deadline_ms, trait_cb, user_context)
724    };
725    match res {
726        Ok(()) => NROS_RMW_RET_OK,
727        Err(e) => {
728            // Backend's `Unsupported` mapping for events is its
729            // `serialization_error()` by trait-doc default; the cffi
730            // contract for "this event kind unsupported" is its own
731            // ret code, so map any error here to UNSUPPORTED rather
732            // than e.g. SERIALIZATION_ERROR — the caller's mental
733            // model is "vtable said no events," not "marshalling
734            // failed."
735            let _ = e;
736            NROS_RMW_RET_UNSUPPORTED
737        }
738    }
739}

741unsafe extern "C" fn register_publisher_event_trampoline<R: RustBackend>(
742    publisher: *mut NrosRmwPublisher,
743    kind: NrosRmwEventKind,
744    deadline_ms: u32,
745    cb: NrosRmwEventCallback,
746    user_context: *mut c_void,
747) -> NrosRmwRet {
748    // Publisher::register_event_callback takes `&mut self`. Need a
749    // mut-ptr to the boxed handle.
750    if publisher.is_null() {
751        return NROS_RMW_RET_INVALID_ARGUMENT;
752    }
753    let p_ptr = unsafe { (*publisher).backend_data } as *mut R::Publisher;
754    if p_ptr.is_null() {
755        return NROS_RMW_RET_INVALID_ARGUMENT;
756    }
757    let p = unsafe { &mut *p_ptr };
758    let trait_kind: nros_rmw::EventKind = unsafe { core::mem::transmute(kind) };
759    let trait_cb: nros_rmw::EventCallback = unsafe { core::mem::transmute(cb) };
760    let res = unsafe {
761        Publisher::register_event_callback(p, trait_kind, deadline_ms, trait_cb, user_context)
762    };
763    match res {
764        Ok(()) => NROS_RMW_RET_OK,
765        Err(_) => NROS_RMW_RET_UNSUPPORTED,
766    }
767}

863unsafe extern "C" fn try_recv_sequence_trampoline<R: RustBackend>(
864    subscriber: *mut NrosRmwSubscriber,
865    buf: *mut u8,
866    per_msg_cap: usize,
867    max_msgs: usize,
868    out_lens: *mut usize,
869) -> i32 {
870    // Phase 124.D.1 — delegate to the Rust backend's
871    // `Subscriber::try_recv_sequence` impl. Default trait body
872    // loop-drives `try_recv_raw`; concrete backends opt in by
873    // overriding for a native batch take.
874    let Some(s) = (unsafe { subscriber_mut::<R::Subscriber>(subscriber) }) else {
875        return NROS_RMW_RET_INVALID_ARGUMENT;
876    };
877    if buf.is_null() || out_lens.is_null() || per_msg_cap == 0 {
878        return NROS_RMW_RET_INVALID_ARGUMENT;
879    }
880    // SAFETY: caller pinky-promised a contiguous block of
881    // `max_msgs * per_msg_cap` bytes at `buf` and at least
882    // `max_msgs` `usize` slots at `out_lens`. The buffer is
883    // exclusively borrowed for the duration of this call.
884    let buf_slice =
885        unsafe { core::slice::from_raw_parts_mut(buf, max_msgs.saturating_mul(per_msg_cap)) };
886    let lens_slice = unsafe { core::slice::from_raw_parts_mut(out_lens, max_msgs) };
887    match Subscriber::try_recv_sequence(s, buf_slice, per_msg_cap, max_msgs, lens_slice) {
888        Ok(count) => count as i32,
889        Err(e) => ret_from_error(&e),
890    }
891}

895// `backend_data` is dereferenced.
896// ============================================================================
897//
898// Validity contract:
899//
900//   - The C-side wrappers (`CffiSession`, `CffiPublisher`, …) hand
901//     each trampoline an entity-struct pointer whose `backend_data`
902//     was last written by *this* adapter's create-trampoline. That
903//     write is `Box::into_raw(Box::new(handle))` of `R::Session` /
904//     `R::Publisher` / … so the pointer aliases a valid Box<T>.
905//   - `_mut` variants give `&mut T`; the runtime serialises access
906//     per the executor's single-thread invariant.
907//   - `take_box` reclaims ownership; subsequent calls see a null
908//     `backend_data` and return `INVALID_ARGUMENT`.

1039#[inline]
1040unsafe fn take_box<T>(slot: &mut *mut c_void) -> Option<Box<T>> {
1041    if slot.is_null() {
1042        return None;
1043    }
1044    let ptr = *slot as *mut T;
1045    *slot = core::ptr::null_mut();
1046    if ptr.is_null() {
1047        None
1048    } else {
1049        // SAFETY: this pointer was minted by `Box::into_raw` in the
1050        // corresponding create-trampoline; we are taking ownership
1051        // back and the slot has been cleared so no second take is
1052        // possible.
1053        Some(unsafe { Box::from_raw(ptr) })
1054    }
