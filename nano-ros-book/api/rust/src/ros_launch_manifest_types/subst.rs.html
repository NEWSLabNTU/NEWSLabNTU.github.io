1//! `$(var ...)` substitution engine for manifest args.
2//!
3//! Resolves `$(var name)` references in all string fields of a manifest
4//! using a provided args map. Runs before static checks and namespace resolution.
5
6use crate::types::ArgDecl;
7use std::collections::{BTreeMap, HashMap};
8
9/// Error during substitution.
10#[derive(Debug, thiserror::Error)]
11pub enum SubstError {
12    #[error("unresolved variable: $(var {name}) — not in args")]
13    Unresolved { name: String },
14    #[error("required arg '{name}' has no default and was not provided")]
15    RequiredArgMissing { name: String },
16    #[error("arg '{name}' has invalid value '{value}' — expected one of {expected}")]
17    InvalidArgValue {
18        name: String,
19        value: String,
20        expected: String,
21    },
22    #[error("malformed substitution: {expr}")]
23    Malformed { expr: String },
24}
25
26/// Resolve manifest args from caller-provided scope args.
27///
28/// All manifest args are mandatory. Errors if any declared arg is missing
29/// from `caller_args`. Validates values against type constraints (bool, choices).
30/// Caller args not declared in the manifest are passed through.
31pub fn resolve_args(
32    manifest_args: &BTreeMap<String, ArgDecl>,
33    caller_args: &HashMap<String, String>,
34) -> Result<HashMap<String, String>, SubstError> {
35    let mut resolved = HashMap::new();
36
37    for (name, decl) in manifest_args {
38        if let Some(value) = caller_args.get(name) {
39            // Validate against type constraint
40            if let Some(valid) = decl.valid_values()
41                && !valid.contains(&value.as_str())
42            {
43                return Err(SubstError::InvalidArgValue {
44                    name: name.clone(),
45                    value: value.clone(),
46                    expected: format!("{:?}", valid),
47                });
48            }
49            resolved.insert(name.clone(), value.clone());
50        } else {
51            return Err(SubstError::RequiredArgMissing { name: name.clone() });
52        }
53    }
54
55    // Also include caller args not declared in manifest (pass-through)
56    for (name, value) in caller_args {
57        if !resolved.contains_key(name) {
58            resolved.insert(name.clone(), value.clone());
59        }
60    }
61
62    Ok(resolved)
63}

65/// Replace all `$(var name)` in a string with values from the args map.
66pub fn substitute_str(s: &str, args: &HashMap<String, String>) -> Result<String, SubstError> {
67    let mut result = String::with_capacity(s.len());
68    let mut remaining = s;
69
70    while let Some(start) = remaining.find("$(var ") {
71        result.push_str(&remaining[..start]);
72        let after_prefix = &remaining[start + 6..]; // skip "$(var "
73        let end = after_prefix
74            .find(')')
75            .ok_or_else(|| SubstError::Malformed {
76                expr: remaining[start..].to_string(),
77            })?;
78        let var_name = after_prefix[..end].trim();
79        let value = args.get(var_name).ok_or_else(|| SubstError::Unresolved {
80            name: var_name.to_string(),
81        })?;
82        result.push_str(value);
83        remaining = &after_prefix[end + 1..];
84    }
85
86    result.push_str(remaining);
87    Ok(result)
88}

90/// Substitute all `$(var ...)` in a Vec<String>.
91pub fn substitute_vec(
92    v: &[String],
93    args: &HashMap<String, String>,
94) -> Result<Vec<String>, SubstError> {
95    v.iter().map(|s| substitute_str(s, args)).collect()
96}

98/// Substitute all `$(var ...)` in an Option<String>.
99pub fn substitute_opt(
100    o: &Option<String>,
101    args: &HashMap<String, String>,
102) -> Result<Option<String>, SubstError> {
103    match o {
104        Some(s) => Ok(Some(substitute_str(s, args)?)),
105        None => Ok(None),
106    }
107}

109/// Substitute all `$(var ...)` references in a manifest's string fields.
110///
111/// This walks all topics, services, imports, exports, and endpoint references.
112/// Does NOT modify node/topic keys (they are manifest-local identifiers).
113pub fn substitute_manifest(
114    manifest: &crate::Manifest,
115    args: &HashMap<String, String>,
116) -> Result<crate::Manifest, SubstError> {
117    let mut m = manifest.clone();
118
119    // Nodes: conditions
120    for node in m.nodes.values_mut() {
121        node.if_condition = substitute_opt(&node.if_condition, args)?;
122        node.unless_condition = substitute_opt(&node.unless_condition, args)?;
123        // Node-level paths: conditions + fields
124        for path in node.paths.values_mut() {
125            path.if_condition = substitute_opt(&path.if_condition, args)?;
126            path.unless_condition = substitute_opt(&path.unless_condition, args)?;
127            path.input = substitute_vec(&path.input, args)?;
128            path.output = substitute_vec(&path.output, args)?;
129            path.correlation = substitute_opt(&path.correlation, args)?;
130        }
131    }
132
133    // Topics: conditions + fields
134    for topic in m.topics.values_mut() {
135        topic.if_condition = substitute_opt(&topic.if_condition, args)?;
136        topic.unless_condition = substitute_opt(&topic.unless_condition, args)?;
137        topic.msg_type = substitute_str(&topic.msg_type, args)?;
138        topic.publishers = substitute_vec(&topic.publishers, args)?;
139        topic.subscribers = substitute_vec(&topic.subscribers, args)?;
140    }
141
142    // Services: conditions + fields
143    for svc in m.services.values_mut() {
144        svc.if_condition = substitute_opt(&svc.if_condition, args)?;
145        svc.unless_condition = substitute_opt(&svc.unless_condition, args)?;
146        svc.srv_type = substitute_str(&svc.srv_type, args)?;
147        svc.server = substitute_vec(&svc.server, args)?;
148        svc.client = substitute_vec(&svc.client, args)?;
149    }
150
151    // Actions: conditions + fields
152    for act in m.actions.values_mut() {
153        act.if_condition = substitute_opt(&act.if_condition, args)?;
154        act.unless_condition = substitute_opt(&act.unless_condition, args)?;
155        act.action_type = substitute_str(&act.action_type, args)?;
156        act.server = substitute_vec(&act.server, args)?;
157        act.client = substitute_vec(&act.client, args)?;
158    }
159
160    // Scope paths: conditions + fields
161    for path in m.paths.values_mut() {
162        path.if_condition = substitute_opt(&path.if_condition, args)?;
163        path.unless_condition = substitute_opt(&path.unless_condition, args)?;
164        path.input = substitute_vec(&path.input, args)?;
165        path.output = substitute_vec(&path.output, args)?;
166        path.correlation = substitute_opt(&path.correlation, args)?;
167    }
168
169    // Includes: external manifest path
170    for include in m.includes.values_mut() {
171        if let crate::IncludeDecl::External { manifest } = include {
172            *manifest = substitute_str(manifest, args)?;
173        }
174    }
175
176    // External topics: type field
177    for ext in m.external_topics.values_mut() {
178        if let Some(t) = &ext.msg_type {
179            ext.msg_type = Some(substitute_str(t, args)?);
180        }
181    }
182
183    Ok(m)
184}
