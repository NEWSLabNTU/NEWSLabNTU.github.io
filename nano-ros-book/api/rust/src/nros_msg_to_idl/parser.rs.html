6/// Primitive type tokens recognised in `.msg` files (mirror of
7/// `rosidl_adapter.parser.PRIMITIVE_TYPES`).
8const PRIMITIVE_TYPES: &[&str] = &[
9    "bool", "byte", "char", "float32", "float64", "int8", "uint8", "int16", "uint16", "int32",
10    "uint32", "int64", "uint64", "string", "wstring", "duration", "time",
11];
12
13const STRING_UPPER_BOUND_TOKEN: &str = "<=";
14const ARRAY_UPPER_BOUND_TOKEN: &str = "<=";
15
16/// A parsed type expression. Mirrors `rosidl_adapter.parser.Type`.
17#[derive(Debug, Clone, PartialEq, Eq)]
18pub struct RosType {
19    /// Package name for non-primitive types (`Some("std_msgs")`)
20    /// or `None` for primitives.
21    pub pkg: Option<String>,
22    /// Bare type name without package prefix or array brackets.
23    pub base: String,
24    /// `string<=N` upper bound, if set.
25    pub string_upper_bound: Option<u32>,
26    /// `true` if the type carries `[]`/`[N]`/`[<=N]`.
27    pub is_array: bool,
28    /// Numeric bound parsed out of the array brackets, if any.
29    pub array_size: Option<u32>,
30    /// `true` for `[<=N]` (vs `[N]`).
31    pub is_upper_bound: bool,
32}
33
34impl RosType {
35    pub fn is_primitive(&self) -> bool {
36        self.pkg.is_none()
37    }
38    pub fn is_fixed_size_array(&self) -> bool {
39        self.is_array && self.array_size.is_some() && !self.is_upper_bound
40    }
41}

43/// A field in a parsed `.msg`.
44#[derive(Debug, Clone, PartialEq, Eq)]
45pub struct Field {
46    pub ty: RosType,
47    pub name: String,
48    /// Comment lines attached to this field (in order). Mirrors
49    /// `rosidl_adapter`'s `annotations['comment']` *after*
50    /// `process_comments` dedent / blank-strip.
51    pub comments: Vec<String>,
52}

54/// A parsed `.msg` file. `constants` is reserved — none of the
55/// fixture set uses constants and the python action / srv flows
56/// don't either, so the port leaves it as a stub.
57#[derive(Debug, Clone, PartialEq, Eq)]
58pub struct Message {
59    pub package: String,
60    pub name: String,
61    pub comments: Vec<String>,
62    pub fields: Vec<Field>,
63}

65/// Parse a `.msg` file. Mirrors
66/// `rosidl_adapter.parser.parse_message_string`.
67pub fn parse_msg(package: &str, name: &str, source: &str) -> Result<Message, ConvertError> {
68    // Mirror the python `replace('\t', ' ')` and `splitlines()`.
69    let normalized = source.replace('\t', " ");
70    let raw_lines: Vec<&str> = normalized.split('\n').collect();
71    // python `splitlines()` drops a trailing empty line if the
72    // string ends in `\n`. Mimic that.
73    let mut all_lines: Vec<&str> = raw_lines.clone();
74    if let Some(last) = all_lines.last() {
75        if last.is_empty() {
76            all_lines.pop();
77        }
78    }
79
80    // 1. Extract file-level (message) comments: every leading line
81    //    that starts with `#`, stripping one or more leading `#`.
82    let split_idx = all_lines
83        .iter()
84        .position(|l| !l.starts_with('#'))
85        .unwrap_or(all_lines.len());
86    let mut message_comments: Vec<String> = all_lines[..split_idx]
87        .iter()
88        .map(|l| lstrip_chars(l, '#'))
89        .collect();
90    let body = &all_lines[split_idx..];
91
92    let mut fields: Vec<Field> = Vec::new();
93    let mut current_comments: Vec<String> = Vec::new();
94
95    for raw in body {
96        // Mirror python `line.rstrip()` — strips ASCII whitespace
97        // from the end. `str::trim_end` matches.
98        let line = raw.trim_end();
99        if line.is_empty() {
100            continue;
101        }
102
103        // Find the comment delimiter `#`.
104        let (code, comment): (&str, Option<String>) = match line.find('#') {
105            Some(idx) => {
106                let raw_comment = &line[idx..];
107                (&line[..idx], Some(lstrip_chars(raw_comment, '#')))
108            }
109            None => (line, None),
110        };
111
112        if let Some(comment_text) = comment {
113            // Python: `if line and not line.strip()` — indented
114            // comment, attaches to previous field.
115            if !code.is_empty() && code.trim().is_empty() {
116                if let Some(last) = fields.last_mut() {
117                    last.comments.push(comment_text);
118                }
119                continue;
120            }
121            // Otherwise it's a free-floating comment ahead of a field.
122            current_comments.push(comment_text);
123            // And fall through with `code` for the field parse.
124        }
125        let code = code.trim_end();
126        if code.is_empty() {
127            continue;
128        }
129
130        // Split off `<type> <rest>`.
131        let (ty_tok, rest) = match code.split_once(' ') {
132            Some((t, r)) => (t, r.trim_start()),
133            None => return Err(ConvertError::InvalidFieldDefinition(code.to_string())),
134        };
135        if rest.is_empty() {
136            return Err(ConvertError::InvalidFieldDefinition(code.to_string()));
137        }
138
139        // Detect constants (`=` separator). The fixture set has
140        // none; reject explicitly so future callers see an error
141        // instead of silent skip.
142        if rest.contains('=') {
143            return Err(ConvertError::Unsupported(format!(
144                "constant declarations not yet supported: {code}"
145            )));
146        }
147
148        // Default values (third whitespace-separated token) — also
149        // unsupported by the fixture set; if present, fold into
150        // the type token's `rest` and parse only the field name.
151        let (field_name, default) = match rest.split_once(' ') {
152            Some((n, d)) => (n, Some(d.trim_start().to_string())),
153            None => (rest, None),
154        };
155        if let Some(d) = &default {
156            if !d.is_empty() {
157                return Err(ConvertError::Unsupported(format!(
158                    "default values not yet supported: {code}"
159                )));
160            }
161        }
162
163        let ty = parse_type(ty_tok, package)?;
164
165        let mut field = Field {
166            ty,
167            name: field_name.to_string(),
168            comments: Vec::new(),
169        };
170        field.comments.extend(current_comments.drain(..));
171        fields.push(field);
172    }
173
174    // Run the per-comment-list condense pass on the message + every
175    // field, mirroring `process_comments`.
176    process_comments(&mut message_comments);
177    for f in fields.iter_mut() {
178        process_comments(&mut f.comments);
179    }
180
181    Ok(Message {
182        package: package.to_string(),
183        name: name.to_string(),
184        comments: message_comments,
185        fields,
186    })
187}

189/// Mirror of `rosidl_adapter.parser.process_comments` (only the
190/// trimming branch — unit-extraction omitted because none of the
191/// fixture set carries a `[unit]` annotation).
192fn process_comments(lines: &mut Vec<String>) {
193    // Strip leading empties.
194    while lines.first().map(|s| s.is_empty()).unwrap_or(false) {
195        lines.remove(0);
196    }
197    // Strip trailing empties.
198    while lines.last().map(|s| s.is_empty()).unwrap_or(false) {
199        lines.pop();
200    }
201    // Collapse consecutive empties.
202    let mut i = 1;
203    while i < lines.len() {
204        if lines[i].is_empty() && lines[i - 1].is_empty() {
205            lines.remove(i);
206            continue;
207        }
208        i += 1;
209    }
210    if lines.is_empty() {
211        return;
212    }
213    // Apply `textwrap.dedent`. Implemented inline for the ASCII-only
214    // comment input set.
215    let dedented = textwrap_dedent(lines);
216    *lines = dedented;
217}

219/// Tiny textwrap.dedent: find the longest common leading-whitespace
220/// prefix across all non-empty lines and strip it from every line.
221fn textwrap_dedent(lines: &[String]) -> Vec<String> {
222    let prefixes: Vec<&str> = lines
223        .iter()
224        .filter(|l| !l.trim().is_empty())
225        .map(|l| {
226            let end = l.find(|c: char| !c.is_whitespace()).unwrap_or(l.len());
227            &l[..end]
228        })
229        .collect();
230    let common = if prefixes.is_empty() {
231        ""
232    } else {
233        let mut common = prefixes[0];
234        for p in &prefixes[1..] {
235            common = common_prefix(common, p);
236            if common.is_empty() {
237                break;
238            }
239        }
240        common
241    };
242    lines
243        .iter()
244        .map(|l| {
245            if l.starts_with(common) {
246                l[common.len()..].to_string()
247            } else {
248                l.clone()
249            }
250        })
251        .collect()
252}

254fn common_prefix<'a>(a: &'a str, b: &str) -> &'a str {
255    let n = a
256        .as_bytes()
257        .iter()
258        .zip(b.as_bytes().iter())
259        .take_while(|(x, y)| x == y)
260        .count();
261    &a[..n]
262}

264/// Strip one or more leading occurrences of `ch` from `s`. Mirrors
265/// python's `str.lstrip(COMMENT_DELIMITER)` (which strips a SET of
266/// chars, but only `#` is in the set).
267fn lstrip_chars(s: &str, ch: char) -> String {
268    let trimmed = s.trim_start_matches(ch);
269    trimmed.to_string()
270}

272/// Parse a single type token (after array / string-bound handling).
273/// Mirrors `rosidl_adapter.parser.Type::__init__` + `BaseType::__init__`.
274fn parse_type(tok: &str, context_package: &str) -> Result<RosType, ConvertError> {
275    let mut is_array = false;
276    let mut array_size: Option<u32> = None;
277    let mut is_upper_bound = false;
278    let mut core_tok = tok;
279
280    if tok.ends_with(']') {
281        is_array = true;
282        let idx = tok
283            .rfind('[')
284            .ok_or_else(|| ConvertError::MalformedArray(tok.to_string()))?;
285        let bracket_inner = &tok[idx + 1..tok.len() - 1];
286        if !bracket_inner.is_empty() {
287            let (rest, upper) =
288                if let Some(stripped) = bracket_inner.strip_prefix(ARRAY_UPPER_BOUND_TOKEN) {
289                    (stripped, true)
290                } else {
291                    (bracket_inner, false)
292                };
293            let n: u32 = rest
294                .parse()
295                .map_err(|_| ConvertError::InvalidBound(tok.to_string()))?;
296            if n == 0 {
297                return Err(ConvertError::InvalidBound(tok.to_string()));
298            }
299            array_size = Some(n);
300            is_upper_bound = upper;
301        }
302        core_tok = &tok[..idx];
303    }
304
305    // Now `core_tok` is the un-arrayed type, e.g. `int32`, `string<=20`,
306    // `std_msgs/Header`.
307    let (pkg, base, string_upper_bound) = parse_base_type(core_tok, context_package)?;
308
309    Ok(RosType {
310        pkg,
311        base,
312        string_upper_bound,
313        is_array,
314        array_size,
315        is_upper_bound,
316    })
317}

319fn parse_base_type(
320    tok: &str,
321    context_package: &str,
322) -> Result<(Option<String>, String, Option<u32>), ConvertError> {
323    if PRIMITIVE_TYPES.contains(&tok) {
324        return Ok((None, tok.to_string(), None));
325    }
326
327    // string<=N / wstring<=N
328    for prefix in ["string", "wstring"] {
329        let bound_prefix = format!("{prefix}{STRING_UPPER_BOUND_TOKEN}");
330        if let Some(rest) = tok.strip_prefix(&bound_prefix) {
331            let n: u32 = rest
332                .parse()
333                .map_err(|_| ConvertError::InvalidBound(tok.to_string()))?;
334            if n == 0 {
335                return Err(ConvertError::InvalidBound(tok.to_string()));
336            }
337            return Ok((None, prefix.to_string(), Some(n)));
338        }
339    }
340
341    // `<pkg>/<Type>` or bare `<Type>` (using context_package).
342    let (pkg, base) = match tok.split_once('/') {
343        Some((p, b)) => (p.to_string(), b.to_string()),
344        None => (context_package.to_string(), tok.to_string()),
345    };
346    if pkg.is_empty() || base.is_empty() {
347        return Err(ConvertError::InvalidResourceName(tok.to_string()));
348    }
349
350    // No primitive-vs-namespaced disambiguation needed — the
351    // primitive check above already covered it.
352    Ok((Some(pkg), base, None))
353}
