8/// Errors raised by the `.msg` parser / IDL emitter.
9#[derive(Debug, Clone, PartialEq, Eq)]
10pub enum ConvertError {
11    /// A field line could not be split into `<type> <name>`.
12    InvalidFieldDefinition(String),
13    /// A type token contains array brackets but no opening `[`.
14    MalformedArray(String),
15    /// A string-bound or array-bound is not a positive integer.
16    InvalidBound(String),
17    /// `<pkg>/<Type>` reference is missing the package half and
18    /// no context package was passed in.
19    InvalidResourceName(String),
20    /// Unsupported feature in the input `.msg` (constants with
21    /// default values, etc.) — currently the port targets the
22    /// `rcl_interfaces`/`std_msgs`/`geometry_msgs`/`sensor_msgs`
23    /// common case; richer surfaces become explicit
24    /// `Unsupported` errors so callers can fall back to python.
25    Unsupported(String),
26}
27
28impl fmt::Display for ConvertError {
29    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
30        match self {
31            Self::InvalidFieldDefinition(s) => write!(f, "invalid field definition: {s}"),
32            Self::MalformedArray(s) => write!(f, "malformed array type: {s}"),
33            Self::InvalidBound(s) => write!(f, "invalid bound: {s}"),
34            Self::InvalidResourceName(s) => write!(f, "invalid resource name: {s}"),
35            Self::Unsupported(s) => write!(f, "unsupported .msg feature: {s}"),
36        }
37    }
38}
39
40impl std::error::Error for ConvertError {}

42/// Mirror of `rosidl_adapter.msg.MSG_TYPE_TO_IDL`.
43pub(crate) fn primitive_idl(name: &str) -> Option<&'static str> {
44    Some(match name {
45        "bool" => "boolean",
46        "byte" => "octet",
47        "char" => "uint8",
48        "int8" => "int8",
49        "uint8" => "uint8",
50        "int16" => "int16",
51        "uint16" => "uint16",
52        "int32" => "int32",
53        "uint32" => "uint32",
54        "int64" => "int64",
55        "uint64" => "uint64",
56        "float32" => "float",
57        "float64" => "double",
58        "string" => "string",
59        "wstring" => "wstring",
60        _ => return None,
61    })
62}

64/// Equivalent of `rosidl_adapter.msg.get_idl_type(type_)`.
65///
66/// Returns the IDL spelling of a parsed ROS type, e.g.
67///
68/// - `int32` → `"int32"`
69/// - `string<=20` → `"string<20>"`
70/// - `geometry_msgs/Vector3` → `"geometry_msgs::msg::Vector3"`
71/// - `uint8[10]` → `"uint8[10]"`
72/// - `uint8[]` → `"sequence<uint8>"`
73/// - `uint8[<=5]` → `"sequence<uint8, 5>"`
74pub fn idl_type_for(ty: &RosType) -> String {
75    let base = if let Some(prim) = primitive_idl(&ty.base) {
76        if (prim == "string" || prim == "wstring") && ty.string_upper_bound.is_some() {
77            format!("{prim}<{}>", ty.string_upper_bound.unwrap())
78        } else {
79            prim.to_string()
80        }
81    } else {
82        // Non-primitive — `<pkg>::msg::<Type>`.
83        format!("{}::msg::{}", ty.pkg.as_deref().unwrap_or(""), ty.base)
84    };
85
86    if !ty.is_array {
87        return base;
88    }
89
90    if ty.is_fixed_size_array() {
91        return format!("{base}[{}]", ty.array_size.unwrap());
92    }
93
94    match (ty.array_size, ty.is_upper_bound) {
95        (Some(n), true) => format!("sequence<{base}, {n}>"),
96        _ => format!("sequence<{base}>"),
97    }
98}
