20/// Convert a `.msg` source string to Cyclone-DDS-shaped IDL.
21///
22/// Output matches the python `msg_to_cyclone_idl.py` byte-for-byte
23/// for the common-case `.msg` subset (primitives, strings, nested
24/// types, fixed-size arrays, unbounded / bounded sequences).
25pub fn msg_to_idl(msg_source: &str, package: &str, message: &str) -> Result<String, ConvertError> {
26    Converter::new(package, message).convert(msg_source)
27}
28
29/// Builder-style entry point. Use this when the caller has more
30/// context than the bare `(package, message)` pair — e.g. a `.srv`
31/// half that needs the 16-byte request-header injected.
32#[derive(Clone, Debug)]
33pub struct Converter<'a> {
34    package: &'a str,
35    message: &'a str,
36    /// Mirror of the python script's `inject_service_header` flag.
37    /// Forces the two `cdds_request_header_t` fields
38    /// (`unsigned long long rmw_writer_guid` + `long long
39    /// rmw_sequence_number`) to be inlined as the first two members
40    /// of every rewritten struct.
41    inject_service_header: bool,
42}
43
44impl<'a> Converter<'a> {
45    pub fn new(package: &'a str, message: &'a str) -> Self {
46        Self {
47            package,
48            message,
49            inject_service_header: false,
50        }
51    }
52
53    pub fn with_service_header(mut self, on: bool) -> Self {
54        self.inject_service_header = on;
55        self
56    }
57
58    pub fn convert(&self, msg_source: &str) -> Result<String, ConvertError> {
59        let msg = parse_msg(self.package, self.message, msg_source)?;
60        let raw = emit_idl(self.package, self.message, &msg);
61        Ok(mangle_idl(&raw, self.inject_service_header))
62    }
63}
