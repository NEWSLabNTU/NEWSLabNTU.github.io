1//! Phase 212.N.9 — `nros::main!()` proc-macro implementation.
2//!
3//! Replaces the legacy Entry pkg `build.rs + include!()` shape with a
4//! one-line `main.rs`. Four forms (per design-doc §11.6):
5//!
6//! ```ignore
7//! nros::main!();                                          // single-node self-bringup
8//! nros::main!(board = NativeBoard);                       // single-node, explicit board
9//! nros::main!(launch = "demo_bringup");                   // multi-node, default launch
10//! nros::main!(launch = "demo_bringup:sim.launch.xml");    // multi-node, explicit file
11//! nros::main!(board = X, launch = "Y:Z.xml", args = [("k", "v")]);
12//! ```
13//!
14//! Form 1 reads `[package.metadata.nros.entry] deploy = "<board>"` from
15//! the Entry pkg's own `Cargo.toml` and maps the board key to a board
16//! crate via a small lookup table. Forms 3+ resolve the bringup pkg
17//! through the N.10 workspace pkg-index, walk the N.11 launch.xml
18//! parser, and emit one `<pkg_ident>::register(runtime)?;` call per
19//! `<node pkg=… exec=…>` entry.
20//!
21//! ## Rebuild-correctness workaround
22//!
23//! Stable Rust proc-macros can't use the unstable
24//! `proc_macro::tracked_path::path()` API. Instead we emit
25//! `const _: &[u8] = include_bytes!("/abs/path");` for every file the
26//! macro read (launch.xml, every `package.xml` the pkg-index walked,
27//! the bringup's `system.toml`). Cargo's `include_bytes!` is tracked,
28//! so touching any of these files invalidates the Entry pkg's
29//! compilation cache.
30

43/// Parsed `nros::main!(...)` argument set.
44///
45/// Each form maps to a different population of these fields:
46///   - Form 1 (no args): all `None`.
47///   - Form 2 (`board = X`): `board = Some(X)`, launch / args `None`.
48///   - Form 3 (`launch = "Y"`): `launch = Some("Y")`, board derived
49///     from `[package.metadata.nros.entry] deploy`.
50///   - Form 4 (board + launch + args): all populated.
51#[derive(Default)]
52struct MainArgs {
53    /// Explicit board ident — used verbatim when supplied. None
54    /// triggers the `Cargo.toml [package.metadata.nros.entry]
55    /// deploy = "<board>"` lookup.
56    board: Option<SynPath>,
57    /// `"<bringup_pkg>"` or `"<bringup_pkg>:<file.launch.xml>"`. None
58    /// triggers the single-node self-bringup path (emit
59    /// `<this_pkg>::register(runtime)?;`).
60    launch: Option<LitStr>,
61    /// `args = [("k", "v"), ...]`. Forwarded to the launch parser as
62    /// argument overrides.
63    args: Vec<(String, String)>,
64}
65
66impl Parse for MainArgs {
67    fn parse(input: ParseStream) -> syn::Result<Self> {
68        let mut out = MainArgs::default();
69        if input.is_empty() {
70            return Ok(out);
71        }
72        // Parse `key = value` pairs, comma-separated. Allow a trailing
73        // comma — matches every other Rust attribute-arg syntax.
74        let pairs: Punctuated<KeyValue, Token![,]> = Punctuated::parse_terminated(input)?;
75        for KeyValue { key, value } in pairs {
76            match key.to_string().as_str() {
77                "board" => {
78                    let path = match value {
79                        KvValue::Path(p) => p,
80                        KvValue::Str(s) => {
81                            return Err(syn::Error::new(
82                                s.span(),
83                                "expected board ident (e.g. `NativeBoard`), got string literal",
84                            ));
85                        }
86                        KvValue::Args(_) => {
87                            return Err(syn::Error::new(
88                                key.span(),
89                                "`board = ` takes a type path, not a list",
90                            ));
91                        }
92                    };
93                    out.board = Some(path);
94                }
95                "launch" => {
96                    let s = match value {
97                        KvValue::Str(s) => s,
98                        _ => {
99                            return Err(syn::Error::new(
100                                key.span(),
101                                "`launch = ` takes a string literal",
102                            ));
103                        }
104                    };
105                    out.launch = Some(s);
106                }
107                "args" => {
108                    let list = match value {
109                        KvValue::Args(pairs) => pairs,
110                        _ => {
111                            return Err(syn::Error::new(
112                                key.span(),
113                                "`args = ` takes a list of `(\"key\", \"value\")` tuples",
114                            ));
115                        }
116                    };
117                    out.args = list;
118                }
119                other => {
120                    return Err(syn::Error::new(
121                        key.span(),
122                        format!(
123                            "unknown `nros::main!` argument `{other}` \
124                             (expected one of: board, launch, args)"
125                        ),
126                    ));
127                }
128            }
129        }
130        Ok(out)
131    }
132}

134struct KeyValue {
135    key: Ident,
136    value: KvValue,
137}
138
139enum KvValue {
140    /// `board = NativeBoard` / `board = ::nros_board_native::NativeBoard`
141    Path(SynPath),
142    /// `launch = "demo_bringup:sim.launch.xml"`
143    Str(LitStr),
144    /// `args = [("use_sim", "true"), ...]`
145    Args(Vec<(String, String)>),
146}
147
148impl Parse for KeyValue {
149    fn parse(input: ParseStream) -> syn::Result<Self> {
150        let key: Ident = input.parse()?;
151        input.parse::<Token![=]>()?;
152        // Try the array form first (args = [...]) — `Expr::parse`
153        // would also handle it, but extracting the tuple values is
154        // cleaner via a dedicated branch.
155        if input.peek(syn::token::Bracket) {
156            let content;
157            syn::bracketed!(content in input);
158            let pairs: Punctuated<TupleStrPair, Token![,]> =
159                Punctuated::parse_terminated(&content)?;
160            let collected = pairs
161                .into_iter()
162                .map(|p| (p.k.value(), p.v.value()))
163                .collect();
164            return Ok(KeyValue {
165                key,
166                value: KvValue::Args(collected),
167            });
168        }
169        // A bare string literal -> KvValue::Str. Anything else -> Path.
170        if input.peek(LitStr) {
171            let s: LitStr = input.parse()?;
172            return Ok(KeyValue {
173                key,
174                value: KvValue::Str(s),
175            });
176        }
177        let path: SynPath = input.parse()?;
178        Ok(KeyValue {
179            key,
180            value: KvValue::Path(path),
181        })
182    }
183}
184
185struct TupleStrPair {
186    k: LitStr,
187    v: LitStr,
188}
189
190impl Parse for TupleStrPair {
191    fn parse(input: ParseStream) -> syn::Result<Self> {
192        let content;
193        syn::parenthesized!(content in input);
194        let k: LitStr = content.parse()?;
195        content.parse::<Token![,]>()?;
196        let v: LitStr = content.parse()?;
197        Ok(TupleStrPair { k, v })
198    }
199}

201/// Entry point — emits the `fn main()` body. Errors surface as
202/// `compile_error!()` spans pointing at the macro invocation.
203pub fn expand(input: TokenStream) -> TokenStream {
204    let args = parse_macro_input!(input as MainArgs);
205    match build_main(args) {
206        Ok(ts) => ts.into(),
207        Err(err) => err.to_compile_error().into(),
208    }
209}
210
211/// Errors carry a span — when we have a relevant token, attach it; when
212/// the failure is environmental (Cargo.toml missing, launch parse
213/// failed), fall back to `Span::call_site()`.
214type MacroResult<T> = std::result::Result<T, syn::Error>;
215
216fn build_main(args: MainArgs) -> MacroResult<proc_macro2::TokenStream> {
217    // CARGO_MANIFEST_DIR is set by cargo when compiling proc-macro
218    // consumers; if missing we fail loud — proc-macros without a
219    // manifest dir would have no way to find Cargo.toml or workspace
220    // root.
221    let manifest_dir = std::env::var_os("CARGO_MANIFEST_DIR").ok_or_else(|| {
222        syn::Error::new(
223            Span::call_site(),
224            "nros::main!: CARGO_MANIFEST_DIR not set (cargo must drive the build)",
225        )
226    })?;
227    let manifest_dir = PathBuf::from(manifest_dir);
228
229    // List of files that participated in the macro's decision so we
230    // can emit `include_bytes!` rebuild stamps below. Always
231    // canonicalised so the paths survive cargo's relocation tricks.
232    let mut tracked: Vec<PathBuf> = Vec::new();
233
234    // --- Board resolution ---
235    let board_path = match &args.board {
236        Some(p) => p.clone(),
237        None => {
238            let cargo_toml = manifest_dir.join("Cargo.toml");
239            tracked.push(cargo_toml.clone());
240            let deploy = read_entry_deploy(&cargo_toml).map_err(|e| {
241                syn::Error::new(
242                    Span::call_site(),
243                    format!(
244                        "nros::main!: failed to read `[package.metadata.nros.entry] deploy` \
245                         from `{}`: {e}\n  Hint: add `[package.metadata.nros.entry] deploy = \
246                         \"<board>\"` (e.g. `\"native\"`, `\"freertos\"`, `\"zephyr\"`) \
247                         to your Cargo.toml, or pass `board = MyBoard` to the macro.",
248                        cargo_toml.display()
249                    ),
250                )
251            })?;
252            board_path_for(&deploy).ok_or_else(|| {
253                syn::Error::new(
254                    Span::call_site(),
255                    format!(
256                        "nros::main!: unknown board `{deploy}` in \
257                         `[package.metadata.nros.entry] deploy`. \
258                         Known boards: {}.\n  Pass `board = <YourBoardZst>` explicitly \
259                         if your board crate is not in the default table.",
260                        known_boards_csv()
261                    ),
262                )
263            })?
264        }
265    };
266
267    // --- Launch resolution → list of <pkg_ident> register calls ---
268    let pkg_idents: Vec<Ident> = match &args.launch {
269        None => {
270            // Form 1 / Form 2 — single-node self-bringup. Use this
271            // pkg's own name as the register target. The codegen-
272            // emitted ident is snake_case; we read CARGO_PKG_NAME and
273            // sanitise via the same `-` → `_` rule the existing
274            // `nros::node!()` macro uses (see `lib.rs::sanitize_pkg_name_for_symbol`).
275            //
276            // The bin target depends on the lib target of the same
277            // pkg via Cargo's automatic `extern crate <my_pkg>;`,
278            // so `<my_pkg>::register(runtime)?;` resolves at build
279            // time.
280            let pkg_name = std::env::var("CARGO_PKG_NAME").map_err(|_| {
281                syn::Error::new(Span::call_site(), "nros::main!: CARGO_PKG_NAME not set")
282            })?;
283            let crate_ident = pkg_to_crate_ident(&pkg_name);
284            // Form 1 self-bringup is opt-in: the user's lib crate
285            // must expose a `pub fn register(runtime)`. If this
286            // Entry pkg is binary-only (no companion lib), the
287            // emitted `<this_pkg>::register(...)` will fail to
288            // compile with a clear error; we don't try to detect
289            // that here.
290            vec![Ident::new(&crate_ident, Span::call_site())]
291        }
292        Some(launch_lit) => {
293            let launch_value = launch_lit.value();
294            // Walk the workspace from the Entry pkg's manifest dir.
295            let workspace_root = nros_build::pkg_index::detect_workspace_root(&manifest_dir)
296                .map_err(|e| {
297                    syn::Error::new(
298                        launch_lit.span(),
299                        format!("nros::main!: detect_workspace_root: {e}"),
300                    )
301                })?;
302            let pkg_index =
303                nros_build::pkg_index::build_pkg_index(&workspace_root).map_err(|e| {
304                    syn::Error::new(
305                        launch_lit.span(),
306                        format!("nros::main!: build_pkg_index: {e}"),
307                    )
308                })?;
309            // Track every package.xml the index walked.
310            for (_, pkg_dir) in pkg_index.pkgs() {
311                tracked.push(pkg_dir.join("package.xml"));
312            }
313
314            // Split `"bringup_pkg[:file.launch.xml]"`.
315            let (bringup_name, file_override) = match launch_value.split_once(':') {
316                Some((b, f)) => (b.trim().to_string(), Some(f.trim().to_string())),
317                None => (launch_value.trim().to_string(), None),
318            };
319            if bringup_name.is_empty() {
320                return Err(syn::Error::new(
321                    launch_lit.span(),
322                    "nros::main!: empty bringup pkg name in `launch = \"...\"`",
323                ));
324            }
325            let bringup_dir = pkg_index
326                .resolve_pkg(&bringup_name)
327                .map_err(|e| syn::Error::new(launch_lit.span(), format!("nros::main!: {e}")))?;
328
329            // Resolve the launch file. If no override, consult
330            // `system.toml::[system] default_launch` (default
331            // `system.launch.xml`).
332            let launch_filename = match file_override {
333                Some(s) => s,
334                None => {
335                    let system_toml = bringup_dir.join("system.toml");
336                    if system_toml.exists() {
337                        tracked.push(system_toml.clone());
338                        read_default_launch(&system_toml)
339                            .map_err(|e| {
340                                syn::Error::new(
341                                    launch_lit.span(),
342                                    format!("nros::main!: parse `{}`: {e}", system_toml.display()),
343                                )
344                            })?
345                            .unwrap_or_else(|| "system.launch.xml".to_string())
346                    } else {
347                        "system.launch.xml".to_string()
348                    }
349                }
350            };
351            let launch_path = bringup_dir.join("launch").join(&launch_filename);
352            tracked.push(launch_path.clone());
353            if !launch_path.exists() {
354                return Err(syn::Error::new(
355                    launch_lit.span(),
356                    format!(
357                        "nros::main!: launch file not found: `{}`",
358                        launch_path.display()
359                    ),
360                ));
361            }
362
363            // Parse the launch file via N.11.
364            let arg_overrides: Vec<(String, String)> = args.args.clone();
365            let desc = nros_build::launch_parser::parse_launch_file(
366                &launch_path,
367                &pkg_index,
368                &arg_overrides,
369            )
370            .map_err(|e| {
371                syn::Error::new(
372                    launch_lit.span(),
373                    format!(
374                        "nros::main!: parse launch file `{}`: {e}",
375                        launch_path.display()
376                    ),
377                )
378            })?;
379
380            // Walk every `<node>` (top-level + inside groups) and
381            // resolve to its Rust crate ident.
382            let mut node_specs = Vec::new();
383            for n in &desc.nodes {
384                node_specs.push(n.clone());
385            }
386            for g in &desc.groups {
387                for n in &g.nodes {
388                    node_specs.push(n.clone());
389                }
390            }
391
392            let mut idents = Vec::new();
393            for node in node_specs {
394                // Look up the node pkg in the index, then derive the
395                // Rust crate ident from its `package.xml::<name>`.
396                let node_pkg_dir = pkg_index.resolve_pkg(&node.pkg).map_err(|e| {
397                    syn::Error::new(
398                        launch_lit.span(),
399                        format!("nros::main!: node pkg `{}`: {e}", node.pkg),
400                    )
401                })?;
402                tracked.push(node_pkg_dir.join("package.xml"));
403                // Is this a Rust pkg? Check for `Cargo.toml`. If
404                // missing, surface a clear error — C++ Node pkgs are
405                // out of scope for v1.
406                let cargo_toml = node_pkg_dir.join("Cargo.toml");
407                if !cargo_toml.exists() {
408                    return Err(syn::Error::new(
409                        launch_lit.span(),
410                        format!(
411                            "nros::main!: node pkg `{}` has no Cargo.toml at `{}`. \
412                             C++ Node pkgs are not yet supported in Rust Entry pkgs.",
413                            node.pkg,
414                            node_pkg_dir.display()
415                        ),
416                    ));
417                }
418                let crate_ident = pkg_to_crate_ident(&node.pkg);
419                idents.push(Ident::new(&crate_ident, Span::call_site()));
420            }
421            idents
422        }
423    };
424
425    // De-duplicate the tracked list — pkg-index walks can revisit
426    // a pkg dir's `package.xml` from multiple paths.
427    tracked.sort();
428    tracked.dedup();
429
430    // --- Emit ---
431    let tracked_consts = tracked.iter().filter_map(|p| {
432        // Skip non-existent paths. The macro may have stuffed a
433        // path that the filesystem doesn't surface (e.g. a missing
434        // `package.xml` when build_pkg_index synthesised a dir).
435        // `include_bytes!` on a missing path is a hard compile
436        // error, so be defensive.
437        if !p.exists() {
438            return None;
439        }
440        let s = p.to_string_lossy().into_owned();
441        let lit = LitStr::new(&s, Span::call_site());
442        Some(quote! {
443            const _: &[u8] = ::core::include_bytes!(#lit);
444        })
445    });
446
447    let register_calls = pkg_idents.iter().map(|ident| {
448        quote! {
449            ::#ident::register(runtime)?;
450        }
451    });
452
453    let expanded = quote! {
454        // Phase 212.N.9 — rebuild-tracking workaround. Stable Rust
455        // proc-macros can't use `proc_macro::tracked_path::path()`;
456        // anonymous `const _: &[u8] = include_bytes!(...)` items are
457        // tracked by cargo's `include_bytes!` and force a recompile
458        // when any tracked file changes.
459        #( #tracked_consts )*
460
461        fn main() {
462            let outcome = <#board_path as ::nros::__macro_support::nros_platform::BoardEntry>::run(
463                |runtime: &mut ::nros::__macro_support::nros_platform::RuntimeCtx<'_>|
464                    -> ::core::result::Result<
465                        (),
466                        ::nros::__macro_support::nros_platform::RuntimeError,
467                    >
468                {
469                    #( #register_calls )*
470                    ::core::result::Result::Ok(())
471                },
472            );
473            if let ::core::result::Result::Err(e) = outcome {
474                ::std::eprintln!("{}: {}", ::core::env!("CARGO_PKG_NAME"), e);
475                ::std::process::exit(1);
476            }
477        }
478    };
479
480    Ok(expanded)
481}

483/// Read `[package.metadata.nros.entry] deploy = "<board>"` from
484/// `Cargo.toml`. The key is mandatory for form-1 (no-arg) invocations.
485fn read_entry_deploy(cargo_toml: &Path) -> Result<String, String> {
486    let raw = std::fs::read_to_string(cargo_toml).map_err(|e| format!("read: {e}"))?;
487    let v: toml::Value = toml::from_str(&raw).map_err(|e| format!("parse toml: {e}"))?;
488    let deploy = v
489        .get("package")
490        .and_then(|p| p.get("metadata"))
491        .and_then(|m| m.get("nros"))
492        .and_then(|n| n.get("entry"))
493        .and_then(|e| e.get("deploy"))
494        .and_then(|d| d.as_str())
495        .ok_or_else(|| {
496            "missing `[package.metadata.nros.entry] deploy = \"<board>\"`".to_string()
497        })?;
498    Ok(deploy.to_string())
499}

501/// Read `[system] default_launch = "<file>"` from a bringup pkg's
502/// `system.toml`. Returns `Ok(None)` when the key is absent (caller
503/// falls back to `"system.launch.xml"`).
504fn read_default_launch(system_toml: &Path) -> Result<Option<String>, String> {
505    let raw = std::fs::read_to_string(system_toml).map_err(|e| format!("read: {e}"))?;
506    let v: toml::Value = toml::from_str(&raw).map_err(|e| format!("parse toml: {e}"))?;
507    Ok(v.get("system")
508        .and_then(|s| s.get("default_launch"))
509        .and_then(|d| d.as_str())
510        .map(str::to_string))
511}

513/// Map a board key from `[package.metadata.nros.entry] deploy = "X"`
514/// to the tier-1 board crate's ZST type path. Adding a new board
515/// requires a workspace-wide edit anyway; the table stays local.
516///
517/// Per the Phase 212.N.3 tier-1 board crates inventory (run
518/// `ls packages/boards/`):
519fn board_path_for(deploy: &str) -> Option<SynPath> {
520    let path_str = match deploy {
521        "native" | "posix" => "::nros_board_native::NativeBoard",
522        "freertos" | "freertos-qemu-mps2-an385" | "qemu-arm-freertos" => {
523            "::nros_board_mps2_an385_freertos::Mps2An385"
524        }
525        "threadx-linux" => "::nros_board_threadx_linux::ThreadxLinux",
526        "threadx-qemu-riscv64" | "qemu-riscv64-threadx" => {
527            "::nros_board_threadx_qemu_riscv64::ThreadxQemuRiscv64"
528        }
529        "nuttx" | "qemu-arm-nuttx" => "::nros_board_qemu_arm_nuttx::QemuArmNuttx",
530        "esp32" => "::nros_board_esp32::Esp32",
531        "zephyr" => "::nros_board_zephyr::Zephyr",
532        _ => return None,
533    };
534    syn::parse_str::<SynPath>(path_str).ok()
535}

541/// Sanitise a pkg name into a valid Rust crate ident. Cargo allows
542/// `-` in pkg names; Rust idents don't. Matches the
543/// `sanitize_pkg_name_for_symbol` rule the existing `nros::node!()`
544/// macro uses, so the codegen + Entry-pkg sides round-trip.
545fn pkg_to_crate_ident(pkg: &str) -> String {
546    let mut out = String::with_capacity(pkg.len());
547    for c in pkg.chars() {
548        if c.is_ascii_alphanumeric() || c == '_' {
549            out.push(c);
550        } else {
551            out.push('_');
552        }
553    }
554    out
555}

557/// Silence the unused-import warning in proc_macro2 — Expr / ExprLit /
558/// Lit are imported so future extensions (e.g. parsing
559/// `args = vec![("k","v")]`) can reach them without re-importing.
560#[allow(dead_code)]
561fn _unused() {
562    let _ = std::marker::PhantomData::<(Expr, ExprLit, Lit)>;
563}
