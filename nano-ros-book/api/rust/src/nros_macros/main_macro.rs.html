1//! Phase 212.N.9 — `nros::main!()` proc-macro implementation.
2//!
3//! Replaces the legacy Entry pkg `build.rs + include!()` shape with a
4//! one-line `main.rs`. Four forms (per design-doc §11.6):
5//!
6//! ```ignore
7//! nros::main!();                                          // single-node self-bringup
8//! nros::main!(board = NativeBoard);                       // single-node, explicit board
9//! nros::main!(launch = "demo_bringup");                   // multi-node, default launch
10//! nros::main!(launch = "demo_bringup:sim.launch.xml");    // multi-node, explicit file
11//! nros::main!(board = X, launch = "Y:Z.xml", args = [("k", "v")]);
12//! ```
13//!
14//! Form 1 reads `[package.metadata.nros.entry] deploy = "<board>"` from
15//! the Entry pkg's own `Cargo.toml` and maps the board key to a board
16//! crate via a small lookup table. Forms 3+ resolve the bringup pkg
17//! through the N.10 workspace pkg-index, walk the N.11 launch.xml
18//! parser, and emit one `<pkg_ident>::register(runtime)?;` call per
19//! `<node pkg=… exec=…>` entry.
20//!
21//! ## Rebuild-correctness workaround
22//!
23//! Stable Rust proc-macros can't use the unstable
24//! `proc_macro::tracked_path::path()` API. Instead we emit
25//! `const _: &[u8] = include_bytes!("/abs/path");` for every file the
26//! macro read (launch.xml, every `package.xml` the pkg-index walked,
27//! the bringup's `system.toml`). Cargo's `include_bytes!` is tracked,
28//! so touching any of these files invalidates the Entry pkg's
29//! compilation cache.
30

43/// Parsed `nros::main!(...)` argument set.
44///
45/// Each form maps to a different population of these fields:
46///   - Form 1 (no args): all `None`.
47///   - Form 2 (`board = X`): `board = Some(X)`, launch / args `None`.
48///   - Form 3 (`launch = "Y"`): `launch = Some("Y")`, board derived
49///     from `[package.metadata.nros.entry] deploy`.
50///   - Form 4 (board + launch + args): all populated.
51#[derive(Default)]
52struct MainArgs {
53    /// Explicit board ident — used verbatim when supplied. None
54    /// triggers the `Cargo.toml [package.metadata.nros.entry]
55    /// deploy = "<board>"` lookup.
56    board: Option<SynPath>,
57    /// `"<bringup_pkg>"` or `"<bringup_pkg>:<file.launch.xml>"`. None
58    /// triggers the single-node self-bringup path (emit
59    /// `<this_pkg>::register(runtime)?;`).
60    launch: Option<LitStr>,
61    /// `args = [("k", "v"), ...]`. Forwarded to the launch parser as
62    /// argument overrides.
63    args: Vec<(String, String)>,
64    /// Phase 216.B.4 — `custom_tasks = [adc_sample, ui_redraw]`. Each
65    /// ident becomes an extra `#[task]` trampoline inside the
66    /// generated `mod __nros_app` body when the dispatched framework
67    /// is RTIC. `None` = not supplied (the default — no extra tasks).
68    /// `Some(vec![])` = supplied as an empty list (still no extra
69    /// tasks, but the key was present — distinguished so the
70    /// OwnedSpin/Embassy misuse error fires even on `[]`).
71    custom_tasks: Option<Vec<Ident>>,
72    /// Span of the `custom_tasks` key, retained for diagnostics when
73    /// rejecting the key under a non-RTIC framework.
74    custom_tasks_span: Option<Span>,
75}
76
77impl Parse for MainArgs {
78    fn parse(input: ParseStream) -> syn::Result<Self> {
79        let mut out = MainArgs::default();
80        if input.is_empty() {
81            return Ok(out);
82        }
83        // Parse `key = value` pairs, comma-separated. Allow a trailing
84        // comma — matches every other Rust attribute-arg syntax.
85        let pairs: Punctuated<KeyValue, Token![,]> = Punctuated::parse_terminated(input)?;
86        for KeyValue { key, value } in pairs {
87            match key.to_string().as_str() {
88                "board" => {
89                    let path = match value {
90                        KvValue::Path(p) => p,
91                        KvValue::Str(s) => {
92                            return Err(syn::Error::new(
93                                s.span(),
94                                "expected board ident (e.g. `NativeBoard`), got string literal",
95                            ));
96                        }
97                        KvValue::Args(_) | KvValue::IdentList(_) => {
98                            return Err(syn::Error::new(
99                                key.span(),
100                                "`board = ` takes a type path, not a list",
101                            ));
102                        }
103                    };
104                    out.board = Some(path);
105                }
106                "launch" => {
107                    let s = match value {
108                        KvValue::Str(s) => s,
109                        _ => {
110                            return Err(syn::Error::new(
111                                key.span(),
112                                "`launch = ` takes a string literal",
113                            ));
114                        }
115                    };
116                    out.launch = Some(s);
117                }
118                "args" => {
119                    let list = match value {
120                        KvValue::Args(pairs) => pairs,
121                        _ => {
122                            return Err(syn::Error::new(
123                                key.span(),
124                                "`args = ` takes a list of `(\"key\", \"value\")` tuples",
125                            ));
126                        }
127                    };
128                    out.args = list;
129                }
130                "custom_tasks" => {
131                    // Phase 216.B.4 — `custom_tasks = [ident, ident,
132                    // ...]`. Stored even when empty so the framework-
133                    // dispatch error fires regardless of list length.
134                    let idents = match value {
135                        KvValue::IdentList(v) => v,
136                        _ => {
137                            return Err(syn::Error::new(
138                                key.span(),
139                                "`custom_tasks = ` takes a list of fn idents, \
140                                 e.g. `custom_tasks = [adc_sample, ui_redraw]`",
141                            ));
142                        }
143                    };
144                    out.custom_tasks = Some(idents);
145                    out.custom_tasks_span = Some(key.span());
146                }
147                other => {
148                    return Err(syn::Error::new(
149                        key.span(),
150                        format!(
151                            "unknown `nros::main!` argument `{other}` \
152                             (expected one of: board, launch, args, custom_tasks)"
153                        ),
154                    ));
155                }
156            }
157        }
158        Ok(out)
159    }
160}

162struct KeyValue {
163    key: Ident,
164    value: KvValue,
165}
166
167enum KvValue {
168    /// `board = NativeBoard` / `board = ::nros_board_native::NativeBoard`
169    Path(SynPath),
170    /// `launch = "demo_bringup:sim.launch.xml"`
171    Str(LitStr),
172    /// `args = [("use_sim", "true"), ...]`
173    Args(Vec<(String, String)>),
174    /// Phase 216.B.4 — `custom_tasks = [adc_sample, ui_redraw, ...]`.
175    /// Empty list is valid (parses to `vec![]`).
176    IdentList(Vec<Ident>),
177}
178
179impl Parse for KeyValue {
180    fn parse(input: ParseStream) -> syn::Result<Self> {
181        let key: Ident = input.parse()?;
182        input.parse::<Token![=]>()?;
183        // Try the array form first — two shapes share the bracket:
184        // `args = [("k","v"), ...]` (tuple-string pairs) and
185        // `custom_tasks = [foo, bar]` (bare idents, Phase 216.B.4).
186        // Dispatch on the key name so each form's parser sees only
187        // its expected token shape (cleaner diagnostics).
188        if input.peek(syn::token::Bracket) {
189            let content;
190            syn::bracketed!(content in input);
191            if key == "custom_tasks" {
192                // Bare-ident list; empty `[]` parses to `vec![]`.
193                let idents: Punctuated<Ident, Token![,]> = Punctuated::parse_terminated(&content)?;
194                let collected: Vec<Ident> = idents.into_iter().collect();
195                return Ok(KeyValue {
196                    key,
197                    value: KvValue::IdentList(collected),
198                });
199            }
200            let pairs: Punctuated<TupleStrPair, Token![,]> =
201                Punctuated::parse_terminated(&content)?;
202            let collected = pairs
203                .into_iter()
204                .map(|p| (p.k.value(), p.v.value()))
205                .collect();
206            return Ok(KeyValue {
207                key,
208                value: KvValue::Args(collected),
209            });
210        }
211        // A bare string literal -> KvValue::Str. Anything else -> Path.
212        if input.peek(LitStr) {
213            let s: LitStr = input.parse()?;
214            return Ok(KeyValue {
215                key,
216                value: KvValue::Str(s),
217            });
218        }
219        let path: SynPath = input.parse()?;
220        Ok(KeyValue {
221            key,
222            value: KvValue::Path(path),
223        })
224    }
225}

227struct TupleStrPair {
228    k: LitStr,
229    v: LitStr,
230}
231
232impl Parse for TupleStrPair {
233    fn parse(input: ParseStream) -> syn::Result<Self> {
234        let content;
235        syn::parenthesized!(content in input);
236        let k: LitStr = content.parse()?;
237        content.parse::<Token![,]>()?;
238        let v: LitStr = content.parse()?;
239        Ok(TupleStrPair { k, v })
240    }
241}

243/// Entry point — emits the `fn main()` body. Errors surface as
244/// `compile_error!()` spans pointing at the macro invocation.
245pub fn expand(input: TokenStream) -> TokenStream {
246    let args = parse_macro_input!(input as MainArgs);
247    match build_main(args) {
248        Ok(ts) => ts.into(),
249        Err(err) => err.to_compile_error().into(),
250    }
251}
252
253/// Errors carry a span — when we have a relevant token, attach it; when
254/// the failure is environmental (Cargo.toml missing, launch parse
255/// failed), fall back to `Span::call_site()`.
256type MacroResult<T> = std::result::Result<T, syn::Error>;
257
258fn build_main(args: MainArgs) -> MacroResult<proc_macro2::TokenStream> {
259    // CARGO_MANIFEST_DIR is set by cargo when compiling proc-macro
260    // consumers; if missing we fail loud — proc-macros without a
261    // manifest dir would have no way to find Cargo.toml or workspace
262    // root.
263    let manifest_dir = std::env::var_os("CARGO_MANIFEST_DIR").ok_or_else(|| {
264        syn::Error::new(
265            Span::call_site(),
266            "nros::main!: CARGO_MANIFEST_DIR not set (cargo must drive the build)",
267        )
268    })?;
269    let manifest_dir = PathBuf::from(manifest_dir);
270
271    // List of files that participated in the macro's decision so we
272    // can emit `include_bytes!` rebuild stamps below. Always
273    // canonicalised so the paths survive cargo's relocation tricks.
274    let mut tracked: Vec<PathBuf> = Vec::new();
275
276    // --- Board resolution ---
277    // `deploy` is `Some(...)` only when the macro had to read the
278    // Entry pkg's `[package.metadata.nros.entry] deploy = "..."` key
279    // (form 1). When the user passes `board = X` directly we have no
280    // deploy string and default to the `OwnedSpin` framework — RTIC
281    // / Embassy require the `deploy = "rtic-stm32f4"` / `deploy =
282    // "embassy-stm32f4"` opt-in for now.
283    let (board_path, deploy_for_framework): (SynPath, Option<String>) = match &args.board {
284        Some(p) => (p.clone(), None),
285        None => {
286            let cargo_toml = manifest_dir.join("Cargo.toml");
287            tracked.push(cargo_toml.clone());
288            let deploy = read_entry_deploy(&cargo_toml).map_err(|e| {
289                syn::Error::new(
290                    Span::call_site(),
291                    format!(
292                        "nros::main!: failed to read `[package.metadata.nros.entry] deploy` \
293                         from `{}`: {e}\n  Hint: add `[package.metadata.nros.entry] deploy = \
294                         \"<board>\"` (e.g. `\"native\"`, `\"freertos\"`, `\"zephyr\"`) \
295                         to your Cargo.toml, or pass `board = MyBoard` to the macro.",
296                        cargo_toml.display()
297                    ),
298                )
299            })?;
300            let resolved = board_path_for(&deploy).ok_or_else(|| {
301                syn::Error::new(
302                    Span::call_site(),
303                    format!(
304                        "nros::main!: unknown board `{deploy}` in \
305                         `[package.metadata.nros.entry] deploy`. \
306                         Known boards: {}.\n  Pass `board = <YourBoardZst>` explicitly \
307                         if your board crate is not in the default table.",
308                        known_boards_csv()
309                    ),
310                )
311            })?;
312            (resolved, Some(deploy))
313        }
314    };
315    let framework = match deploy_for_framework.as_deref() {
316        Some(d) => framework_for(d),
317        None => Framework::OwnedSpin,
318    };
319
320    // Phase 216.B.4 — `custom_tasks = [...]` only applies to the
321    // RTIC emit. OwnedSpin / Embassy have no `mod __nros_app { ... }`
322    // body for the macro to splice into, so flag misuse early with a
323    // span pointing at the key.
324    if args.custom_tasks.is_some() && framework != Framework::Rtic {
325        let span = args.custom_tasks_span.unwrap_or_else(Span::call_site);
326        let framework_label = match framework {
327            Framework::OwnedSpin => "OwnedSpin",
328            Framework::Embassy => "Embassy",
329            Framework::Rtic => unreachable!(),
330        };
331        return Err(syn::Error::new(
332            span,
333            format!(
334                "nros::main!: `custom_tasks = [...]` is only valid for the RTIC framework \
335                 (current framework: {framework_label}). \
336                 RTIC splices each ident as a `#[task]` inside the generated `mod __nros_app` \
337                 body; other frameworks have no equivalent splice point."
338            ),
339        ));
340    }
341
342    // --- Launch resolution → list of <pkg_ident> register calls ---
343    let pkg_idents: Vec<Ident> = match &args.launch {
344        None => {
345            // Form 1 / Form 2 — single-node self-bringup. Use this
346            // pkg's own name as the register target. The codegen-
347            // emitted ident is snake_case; we read CARGO_PKG_NAME and
348            // sanitise via the same `-` → `_` rule the existing
349            // `nros::node!()` macro uses (see `lib.rs::sanitize_pkg_name_for_symbol`).
350            //
351            // The bin target depends on the lib target of the same
352            // pkg via Cargo's automatic `extern crate <my_pkg>;`,
353            // so `<my_pkg>::register(runtime)?;` resolves at build
354            // time.
355            let pkg_name = std::env::var("CARGO_PKG_NAME").map_err(|_| {
356                syn::Error::new(Span::call_site(), "nros::main!: CARGO_PKG_NAME not set")
357            })?;
358            let crate_ident = pkg_to_crate_ident(&pkg_name);
359            // Form 1 self-bringup is opt-in: the user's lib crate
360            // must expose a `pub fn register(runtime)`. If this
361            // Entry pkg is binary-only (no companion lib), the
362            // emitted `<this_pkg>::register(...)` will fail to
363            // compile with a clear error; we don't try to detect
364            // that here.
365            vec![Ident::new(&crate_ident, Span::call_site())]
366        }
367        Some(launch_lit) => {
368            let launch_value = launch_lit.value();
369            // Walk the workspace from the Entry pkg's manifest dir.
370            let workspace_root = nros_build::pkg_index::detect_workspace_root(&manifest_dir)
371                .map_err(|e| {
372                    syn::Error::new(
373                        launch_lit.span(),
374                        format!("nros::main!: detect_workspace_root: {e}"),
375                    )
376                })?;
377            let pkg_index =
378                nros_build::pkg_index::build_pkg_index(&workspace_root).map_err(|e| {
379                    syn::Error::new(
380                        launch_lit.span(),
381                        format!("nros::main!: build_pkg_index: {e}"),
382                    )
383                })?;
384            // Track every package.xml the index walked.
385            for (_, pkg_dir) in pkg_index.pkgs() {
386                tracked.push(pkg_dir.join("package.xml"));
387            }
388
389            // Split `"bringup_pkg[:file.launch.xml]"`.
390            let (bringup_name, file_override) = match launch_value.split_once(':') {
391                Some((b, f)) => (b.trim().to_string(), Some(f.trim().to_string())),
392                None => (launch_value.trim().to_string(), None),
393            };
394            if bringup_name.is_empty() {
395                return Err(syn::Error::new(
396                    launch_lit.span(),
397                    "nros::main!: empty bringup pkg name in `launch = \"...\"`",
398                ));
399            }
400            let bringup_dir = pkg_index
401                .resolve_pkg(&bringup_name)
402                .map_err(|e| syn::Error::new(launch_lit.span(), format!("nros::main!: {e}")))?;
403
404            // Resolve the launch file. If no override, consult
405            // `system.toml::[system] default_launch` (default
406            // `system.launch.xml`).
407            let launch_filename = match file_override {
408                Some(s) => s,
409                None => {
410                    let system_toml = bringup_dir.join("system.toml");
411                    if system_toml.exists() {
412                        tracked.push(system_toml.clone());
413                        read_default_launch(&system_toml)
414                            .map_err(|e| {
415                                syn::Error::new(
416                                    launch_lit.span(),
417                                    format!("nros::main!: parse `{}`: {e}", system_toml.display()),
418                                )
419                            })?
420                            .unwrap_or_else(|| "system.launch.xml".to_string())
421                    } else {
422                        "system.launch.xml".to_string()
423                    }
424                }
425            };
426            let launch_path = bringup_dir.join("launch").join(&launch_filename);
427            tracked.push(launch_path.clone());
428            if !launch_path.exists() {
429                return Err(syn::Error::new(
430                    launch_lit.span(),
431                    format!(
432                        "nros::main!: launch file not found: `{}`",
433                        launch_path.display()
434                    ),
435                ));
436            }
437
438            // Parse the launch file via N.11.
439            let arg_overrides: Vec<(String, String)> = args.args.clone();
440            let desc = nros_build::launch_parser::parse_launch_file(
441                &launch_path,
442                &pkg_index,
443                &arg_overrides,
444            )
445            .map_err(|e| {
446                syn::Error::new(
447                    launch_lit.span(),
448                    format!(
449                        "nros::main!: parse launch file `{}`: {e}",
450                        launch_path.display()
451                    ),
452                )
453            })?;
454
455            // Walk every `<node>` (top-level + inside groups) and
456            // resolve to its Rust crate ident.
457            let mut node_specs = Vec::new();
458            for n in &desc.nodes {
459                node_specs.push(n.clone());
460            }
461            for g in &desc.groups {
462                for n in &g.nodes {
463                    node_specs.push(n.clone());
464                }
465            }
466
467            let mut idents = Vec::new();
468            for node in node_specs {
469                // Look up the node pkg in the index, then derive the
470                // Rust crate ident from its `package.xml::<name>`.
471                let node_pkg_dir = pkg_index.resolve_pkg(&node.pkg).map_err(|e| {
472                    syn::Error::new(
473                        launch_lit.span(),
474                        format!("nros::main!: node pkg `{}`: {e}", node.pkg),
475                    )
476                })?;
477                tracked.push(node_pkg_dir.join("package.xml"));
478                // Is this a Rust pkg? Check for `Cargo.toml`. If
479                // missing, surface a clear error — C++ Node pkgs are
480                // out of scope for v1.
481                let cargo_toml = node_pkg_dir.join("Cargo.toml");
482                if !cargo_toml.exists() {
483                    return Err(syn::Error::new(
484                        launch_lit.span(),
485                        format!(
486                            "nros::main!: node pkg `{}` has no Cargo.toml at `{}`. \
487                             C++ Node pkgs are not yet supported in Rust Entry pkgs.",
488                            node.pkg,
489                            node_pkg_dir.display()
490                        ),
491                    ));
492                }
493                let crate_ident = pkg_to_crate_ident(&node.pkg);
494                idents.push(Ident::new(&crate_ident, Span::call_site()));
495            }
496            idents
497        }
498    };
499
500    // De-duplicate the tracked list — pkg-index walks can revisit
501    // a pkg dir's `package.xml` from multiple paths.
502    tracked.sort();
503    tracked.dedup();
504
505    // --- Emit ---
506    let tracked_consts = tracked.iter().filter_map(|p| {
507        // Skip non-existent paths. The macro may have stuffed a
508        // path that the filesystem doesn't surface (e.g. a missing
509        // `package.xml` when build_pkg_index synthesised a dir).
510        // `include_bytes!` on a missing path is a hard compile
511        // error, so be defensive.
512        if !p.exists() {
513            return None;
514        }
515        let s = p.to_string_lossy().into_owned();
516        let lit = LitStr::new(&s, Span::call_site());
517        Some(quote! {
518            const _: &[u8] = ::core::include_bytes!(#lit);
519        })
520    });
521
522    let register_calls: Vec<proc_macro2::TokenStream> = pkg_idents
523        .iter()
524        .map(|ident| {
525            quote! {
526                ::#ident::register(runtime)?;
527            }
528        })
529        .collect();
530
531    // Phase 216.B.3 — framework-dispatched emit body. `OwnedSpin`
532    // keeps the long-standing `fn __nros_entry_run + fn main` shape
533    // (BoardEntry::run owns the spin loop). `Rtic` emits a
534    // `#[rtic::app(...)]` skeleton that delegates to
535    // `RticBoardEntry::init_hardware` from the framework-generated
536    // `#[init]` body. `Embassy` is a hard error pointing at the
537    // 216.C.3 sibling that lands the emit body.
538    let body_ts: proc_macro2::TokenStream = match framework {
539        Framework::OwnedSpin => quote! {
540            // Phase 213.C follow-up — emit two cfg-gated entry shapes so
541            // both hosted (POSIX / NuttX / threadx-linux) and embedded
542            // (FreeRTOS / bare-metal `target_os = "none"`) targets resolve
543            // a working `main`. The shared body is factored into a private
544            // `__nros_entry_run` returning `Result` so neither arm
545            // duplicates the closure logic.
546            fn __nros_entry_run() -> ::core::result::Result<
547                (),
548                ::nros::__macro_support::nros_platform::RuntimeError,
549            > {
550                <#board_path as ::nros::__macro_support::nros_platform::BoardEntry>::run(
551                    |runtime: &mut ::nros::__macro_support::nros_platform::RuntimeCtx<'_>|
552                        -> ::core::result::Result<
553                            (),
554                            ::nros::__macro_support::nros_platform::RuntimeError,
555                        >
556                    {
557                        #( #register_calls )*
558                        ::core::result::Result::Ok(())
559                    },
560                )
561            }
562
563            #[cfg(not(target_os = "none"))]
564            fn main() {
565                if let ::core::result::Result::Err(e) = __nros_entry_run() {
566                    ::std::eprintln!("{}: {}", ::core::env!("CARGO_PKG_NAME"), e);
567                    ::std::process::exit(1);
568                }
569            }
570
571            #[cfg(target_os = "none")]
572            #[unsafe(no_mangle)]
573            pub extern "C" fn main() -> i32 {
574                match __nros_entry_run() {
575                    ::core::result::Result::Ok(()) => 0,
576                    ::core::result::Result::Err(_) => 1,
577                }
578            }
579        },
580        Framework::Rtic => {
581            // Phase 216.B.3 SKELETON — `#[rtic::app(...)]` module that
582            // delegates boot to `RticBoardEntry::init_hardware`. The
583            // full body (the `__nros_spin` + `__nros_dispatch` software
584            // tasks + per-Node register/spawn wiring) lands in a
585            // 216.B.3 follow-up. Today's emit only needs to compile so
586            // the route through `framework_for(deploy)` is observable
587            // — runtime use surfaces the board crate's `todo!()` in
588            // `init_hardware` (intentional).
589            //
590            // Phase 216.B.4 adds the `custom_tasks = [...]` splice:
591            // each user-listed ident `f` becomes a thin `#[task]`
592            // trampoline that awaits `super::<f>_impl(cx).await`. The
593            // user supplies the impl fn (and its `Context` type-alias
594            // arg — RTIC generates `<f>::Context` from the task ident)
595            // outside the macro; the macro just declares the task.
596            //
597            // Hardcoded `dispatchers = [USART1, USART2]` matches
598            // `RticStm32F4::DISPATCHERS`. A follow-up reads the const
599            // from the board crate at macro-expansion time (requires a
600            // build-graph fs round-trip we want to defer).
601            //
602            // The `#![no_std]`/`#![no_main]` inner attrs are NOT emitted
603            // here — the Entry pkg's `main.rs` already declares those
604            // at file scope (see the talker-rtic example). Emitting
605            // them inside the macro would double-declare.
606            let custom_task_items: Vec<proc_macro2::TokenStream> = args
607                .custom_tasks
608                .as_ref()
609                .map(|v| v.as_slice())
610                .unwrap_or(&[])
611                .iter()
612                .map(|f| {
613                    // Sibling impl fn name: `<f>_impl`. Defined at
614                    // module scope outside the `mod __nros_app` body
615                    // by the user — the trampoline reaches it via
616                    // `super::<f>_impl`.
617                    let impl_ident = Ident::new(&format!("{}_impl", f), f.span());
618                    quote! {
619                        #[task(priority = 1)]
620                        async fn #f(cx: #f::Context) {
621                            super::#impl_ident(cx).await;
622                        }
623                    }
624                })
625                .collect();
626
627            quote! {
628                use #board_path as __NrosBoard;
629
630                #[::rtic::app(
631                    device = stm32f4xx_hal::pac,
632                    dispatchers = [USART1, USART2]
633                )]
634                mod __nros_app {
635                    use super::*;
636
637                    #[shared]
638                    struct Shared {}
639
640                    /// Phase 216.B.3 follow-up — stashes the
641                    /// `(Executor, Runtime)` pair returned by
642                    /// `RticBoardEntry::init_hardware` so the
643                    /// `__nros_spin` / `__nros_dispatch` software
644                    /// tasks can take ownership through the RTIC
645                    /// `local = [<field>]` attribute. The assoc-type
646                    /// projection keeps the macro emit board-agnostic
647                    /// — every `RticBoardEntry` impl picks its own
648                    /// concrete `Executor` / `Runtime` types.
649                    #[local]
650                    struct Local {
651                        executor: <__NrosBoard as ::nros::__macro_support::nros_platform::RticBoardEntry>::Executor,
652                        runtime: <__NrosBoard as ::nros::__macro_support::nros_platform::RticBoardEntry>::Runtime,
653                    }
654
655                    #[init]
656                    fn init(cx: init::Context) -> (Shared, Local) {
657                        // Phase 216.B.3 follow-up — board bring-up
658                        // hands back the `(Executor, Runtime)` pair;
659                        // we stash both into `Local` and spawn the two
660                        // software tasks that will drive them. The
661                        // `run_plan(&mut runtime)` per-Node
662                        // register-dispatch-slot call still belongs to
663                        // a separate follow-up (the trampoline
664                        // registration story spans the macro +
665                        // `nros::node!()` emit + the runtime trait
666                        // surface; same deferred story as the Embassy
667                        // sibling in the C.3 follow-up).
668                        let (executor, runtime) =
669                            <__NrosBoard as ::nros::__macro_support::nros_platform::RticBoardEntry>::init_hardware(
670                                cx.device,
671                                cx.core,
672                            );
673                        // todo: run_plan(&mut runtime) — register each
674                        // Node pkg's dispatch slot via
675                        // `register_dispatch_slot_dyn` before the
676                        // collapsed `__nros_run` task starts consuming
677                        // envelopes. Tracked alongside the per-Node
678                        // trampoline registry story (linkme / Phase 216
679                        // follow-up).
680                        __nros_run::spawn().unwrap();
681                        (Shared {}, Local { executor, runtime })
682                    }
683
684                    /// RTIC run task — collapsed `__nros_spin` +
685                    /// `__nros_dispatch` (Phase 216.B.3 follow-up).
686                    ///
687                    /// The earlier split-task shape had `__nros_spin`
688                    /// own the `Executor` half and `__nros_dispatch`
689                    /// own the `Runtime` half. RTIC `#[local]` fields
690                    /// are claimed by a single task (the
691                    /// `local = [<field>]` attribute is exclusive),
692                    /// and the dispatch task needs the executor to
693                    /// drive the per-Node trampolines that run inside
694                    /// the executor's spin loop. Collapsing into one
695                    /// task that owns both fields side-steps the
696                    /// exclusivity rule and gives the spin / dequeue
697                    /// loop a single coherent borrow.
698                    ///
699                    /// Body:
700                    ///   1. Claim the board's SPSC consumer half via
701                    ///      `take_dispatch_consumer()` (stashed by
702                    ///      `RticBoardEntry::init_hardware`).
703                    ///   2. Drive `executor.spin_once(small_dur)`
704                    ///      — small budget so the dequeue loop runs
705                    ///      between executor iterations.
706                    ///   3. Drain whatever the SPSC has for this
707                    ///      cycle. Today each dequeued envelope is
708                    ///      dropped with a TODO: per-Node trampoline
709                    ///      routing needs an `ExecutorNodeRuntime`-
710                    ///      wrapped sink that the macro emit hasn't
711                    ///      plumbed yet (the `dispatch_callback`
712                    ///      entry on `ExecutorNodeRuntime` is wired
713                    ///      separately; the trampoline registry that
714                    ///      pairs `cb_id` → Node pkg is the next 216
715                    ///      follow-up — likely via `linkme`).
716                    ///
717                    /// Splitting the tasks back apart once the
718                    /// `ExecutorNodeRuntime` sink is plumbed (so the
719                    /// spin task can run at lower priority than the
720                    /// dispatch task) is a separate follow-up.
721                    #[task(local = [executor, runtime], priority = 1)]
722                    async fn __nros_run(cx: __nros_run::Context) {
723                        let executor = cx.local.executor;
724                        // The board-side runtime owns the SPSC
725                        // producer half. Today's collapse keeps it in
726                        // `Local` for symmetry with the planned split
727                        // — once `ExecutorNodeRuntime`-wrapped routing
728                        // lands the runtime's `signal_callback` will
729                        // be the producer-side bridge between executor
730                        // callbacks and the SPSC consumer drained
731                        // below.
732                        let _runtime = cx.local.runtime;
733                        let mut consumer =
734                            ::nros_board_rtic_stm32f4::take_dispatch_consumer()
735                                .expect("RTIC dispatch consumer take");
736                        loop {
737                            let _ = executor.spin_once(
738                                ::core::time::Duration::from_millis(1),
739                            );
740                            // Phase 216 final dispatch wiring — drain
741                            // every envelope the board's SPSC has
742                            // queued this cycle and forward each one
743                            // through `Executor::dispatch_callback`.
744                            // The layer-clean `(cb_id: &str,
745                            // ctx_ptr: *mut c_void)` shape matches both
746                            // the dequeued `SignaledCallback<'static>`
747                            // payload and the executor's stable entry
748                            // point — no type-translation gymnastics.
749                            // The executor-side body is a no-op stub
750                            // today; the per-Node trampoline registry
751                            // (linkme / Phase 216 follow-up) fills it
752                            // in by resolving `cb_id` →
753                            // `__nros_node_<pkg>_on_callback` and
754                            // invoking with the per-pkg state blob.
755                            // What this commit closes is the gap
756                            // where the dequeued envelope was being
757                            // silently dropped with a TODO — values
758                            // now flow into the executor's stable
759                            // surface and the registry lands as a
760                            // body fill, not a macro rewrite.
761                            while let Some(envelope) = consumer.dequeue() {
762                                let cb = envelope.into_inner();
763                                executor.dispatch_callback(cb.cb_id, cb.ctx_ptr);
764                            }
765                        }
766                    }
767
768                    // Phase 216.B.4 — user-supplied `#[task]` trampolines.
769                    // Each calls a sibling `<name>_impl` fn the user
770                    // defines at module scope; signatures are kept
771                    // simple to dodge cross-pkg type-alias plumbing.
772                    #( #custom_task_items )*
773                }
774            }
775        }
776        Framework::Embassy => {
777            // Phase 216.C.3 follow-up — sibling of the Rtic emit above.
778            // Emits `#[embassy_executor::main] async fn main(spawner)`
779            // that delegates to `EmbassyBoardEntry::init_hardware`
780            // (which is **sync** — see `embassy_entry.rs`'s "Sync
781            // `init_hardware`" note; matches `RticBoardEntry`) and
782            // then spawns two `#[embassy_executor::task]` fns:
783            //
784            // - `__nros_spin_task(executor)` — long-lived task that
785            //   drives the executor. The real body will dequeue from
786            //   the board's `CALLBACK_CHANNEL` and invoke per-Node
787            //   trampolines; today it parks on `Timer::after_secs` so
788            //   the macro emit compiles standalone. The dequeue +
789            //   trampoline-lookup integration lands alongside the
790            //   B.3-equivalent RTIC `__nros_spin` body fill — the
791            //   trampoline registration story spans the macro, the
792            //   `nros::node!()` emit, and
793            //   `register_dispatch_slot_dyn`, which is substantial
794            //   integration work for a separate follow-up.
795            //
796            // - `__nros_dispatch_task(runtime)` — long-lived task that
797            //   calls `runtime.spin_once(timeout_ms)` in a loop with an
798            //   `embassy_time` yield between iterations. Same
799            //   placeholder shape as the spin task — the
800            //   `register_dispatch_slot_dyn(...)` registration call
801            //   (the `run_plan(&mut runtime)` story) is the
802            //   integration work deferred to the same follow-up.
803            //
804            // Task argument types resolve via the board's
805            // `EmbassyBoardEntry::{Executor, Runtime}` associated
806            // types; `#[embassy_executor::task]` doesn't accept
807            // generic params, so we name them concretely through the
808            // assoc-type projection.
809            //
810            // The `#![no_std]`/`#![no_main]` inner attrs are NOT
811            // emitted here — the Entry pkg's `main.rs` already
812            // declares those at file scope (mirrors the Rtic branch).
813            quote! {
814                use #board_path as __NrosBoard;
815
816                /// Embassy run task — collapsed `__nros_spin_task` +
817                /// `__nros_dispatch_task` (Phase 216.C.3 follow-up).
818                ///
819                /// `#[embassy_executor::task]` doesn't accept multiple
820                /// generic params and the spin + dispatch loops need
821                /// to share the `(Executor, Runtime)` pair so the
822                /// per-callback routing (once plumbed via the
823                /// `ExecutorNodeRuntime::dispatch_callback` sink in
824                /// `packages/core/nros/src/node_runtime.rs`) can drain
825                /// the board's static `CALLBACK_CHANNEL` between
826                /// executor iterations. Collapsing the two tasks into
827                /// one gives the spin + drain loops a single coherent
828                /// borrow over both halves.
829                ///
830                /// Body:
831                ///   1. Drive `executor.spin_once(small_dur)` — a
832                ///      small budget so the loop can yield between
833                ///      iterations.
834                ///   2. Yield to the Embassy scheduler via
835                ///      `Timer::after_millis(1)` so other tasks
836                ///      (Ethernet driver, user-spawned tasks) run.
837                ///
838                /// What's still placeholder: a dequeue + dispatch
839                /// step. `EmbassyRuntime` owns a `&'static` borrow
840                /// of the board's private `CALLBACK_CHANNEL`, but
841                /// no public accessor exposes the receiver half today
842                /// — adding one is the next 216.C follow-up, paired
843                /// with the per-Node trampoline registry (linkme) and
844                /// the `ExecutorNodeRuntime`-wrapped sink the macro
845                /// emit needs to plumb in order to call
846                /// `dispatch_callback`. Splitting the tasks back
847                /// apart once that lands is a separate follow-up.
848                #[::embassy_executor::task]
849                async fn __nros_run_task(
850                    mut executor: <__NrosBoard as ::nros::__macro_support::nros_platform::EmbassyBoardEntry>::Executor,
851                    runtime: <__NrosBoard as ::nros::__macro_support::nros_platform::EmbassyBoardEntry>::Runtime,
852                ) {
853                    loop {
854                        let _ = executor.spin_once(
855                            ::core::time::Duration::from_millis(1),
856                        );
857                        // Phase 216 final dispatch wiring — drain
858                        // every envelope the board's
859                        // `CALLBACK_CHANNEL` has queued this cycle
860                        // and forward each one through
861                        // `Executor::dispatch_callback`.
862                        // `EmbassyRuntime::try_recv()` (Phase 216
863                        // final, sibling of the RTIC SPSC `dequeue`
864                        // path) is non-blocking so the spin loop
865                        // keeps yielding even when no callback is
866                        // signaled — the `Timer::after_millis(1)`
867                        // below paces the executor poll without
868                        // needing `embassy_futures::select` (not a
869                        // current dep). The same layer-clean
870                        // `(cb_id: &str, ctx_ptr: *mut c_void)` shape
871                        // applies — see the RTIC sibling for the
872                        // matching commentary. The executor-side
873                        // dispatch body is a no-op stub today; the
874                        // per-Node trampoline registry (linkme /
875                        // Phase 216 follow-up) fills it in.
876                        while let Some(envelope) = runtime.try_recv() {
877                            let cb = envelope.into_inner();
878                            executor.dispatch_callback(cb.cb_id, cb.ctx_ptr);
879                        }
880                        ::embassy_time::Timer::after_millis(1).await;
881                    }
882                }
883
884                #[::embassy_executor::main]
885                async fn main(spawner: ::embassy_executor::Spawner) {
886                    // Sync `init_hardware` — see the
887                    // `EmbassyBoardEntry` trait "Sync
888                    // `init_hardware`" note; matches `RticBoardEntry`.
889                    let (executor, runtime) =
890                        <__NrosBoard as ::nros::__macro_support::nros_platform::EmbassyBoardEntry>::init_hardware(
891                            spawner,
892                        );
893                    // todo: register_dispatch_slot_dyn(...) per Node
894                    // pkg via `run_plan(&mut runtime)` — the
895                    // trampoline-registration story spans the macro
896                    // + `nros::node!()` emit + the runtime trait
897                    // and lands in a separate follow-up.
898                    spawner.spawn(__nros_run_task(executor, runtime)).unwrap();
899                }
900            }
901        }
902    };
903
904    let expanded = quote! {
905        // Phase 212.N.9 — rebuild-tracking workaround. Stable Rust
906        // proc-macros can't use `proc_macro::tracked_path::path()`;
907        // anonymous `const _: &[u8] = include_bytes!(...)` items are
908        // tracked by cargo's `include_bytes!` and force a recompile
909        // when any tracked file changes.
910        #( #tracked_consts )*
911
912        #body_ts
913    };
914
915    Ok(expanded)
916}

918/// Read `[package.metadata.nros.entry] deploy = "<board>"` from
919/// `Cargo.toml`. The key is mandatory for form-1 (no-arg) invocations.
920fn read_entry_deploy(cargo_toml: &Path) -> Result<String, String> {
921    let raw = std::fs::read_to_string(cargo_toml).map_err(|e| format!("read: {e}"))?;
922    let v: toml::Value = toml::from_str(&raw).map_err(|e| format!("parse toml: {e}"))?;
923    let deploy = v
924        .get("package")
925        .and_then(|p| p.get("metadata"))
926        .and_then(|m| m.get("nros"))
927        .and_then(|n| n.get("entry"))
928        .and_then(|e| e.get("deploy"))
929        .and_then(|d| d.as_str())
930        .ok_or_else(|| {
931            "missing `[package.metadata.nros.entry] deploy = \"<board>\"`".to_string()
932        })?;
933    Ok(deploy.to_string())
934}

989/// Phase 216.B.3 — boot-framework dispatch for `nros::main!()`.
990///
991/// Distinct from [`board_path_for`] (which only resolves the board
992/// crate ZST). Frameworks are orthogonal to RMW + platform: each
993/// board crate carries its own
994/// `[package.metadata.nros.board] framework = "<f>"` knob (consumed
995/// by 216.D.1 `nros check`). The macro keys off the Entry pkg's
996/// `deploy = "..."` value directly to avoid a fs round-trip into the
997/// board crate's manifest at proc-macro expansion time; the long-term
998/// spec reads the board's manifest, but the skeleton hardcodes the
999/// table to match the `board_path_for` row above.
1000#[derive(Debug, Clone, Copy, PartialEq, Eq)]
1001enum Framework {
1002    /// Board owns the spin loop (`BoardEntry::run`). Default for
1003    /// every board key not explicitly routed below.
1004    OwnedSpin,
1005    /// RTIC framework owns the spin loop. The macro emits a
1006    /// `#[rtic::app(...)]` module + `#[init]` body that delegates to
1007    /// `RticBoardEntry::init_hardware`.
1008    Rtic,
1009    /// Embassy framework owns the spin loop. The macro emits a
1010    /// `#[embassy_executor::main] async fn main(spawner: Spawner)` body
1011    /// that delegates to `EmbassyBoardEntry::init_hardware`.
1012    Embassy,
1013}
1014
1015fn framework_for(deploy: &str) -> Framework {
1016    match deploy {
1017        "rtic-stm32f4" => Framework::Rtic,
1018        "embassy-stm32f4" => Framework::Embassy,
1019        _ => Framework::OwnedSpin,
1020    }
1021}

1023/// Sanitise a pkg name into a valid Rust crate ident. Cargo allows
1024/// `-` in pkg names; Rust idents don't. Matches the
1025/// `sanitize_pkg_name_for_symbol` rule the existing `nros::node!()`
1026/// macro uses, so the codegen + Entry-pkg sides round-trip.
1027fn pkg_to_crate_ident(pkg: &str) -> String {
1028    let mut out = String::with_capacity(pkg.len());
1029    for c in pkg.chars() {
1030        if c.is_ascii_alphanumeric() || c == '_' {
1031            out.push(c);
1032        } else {
1033            out.push('_');
1034        }
1035    }
1036    out
1037}
