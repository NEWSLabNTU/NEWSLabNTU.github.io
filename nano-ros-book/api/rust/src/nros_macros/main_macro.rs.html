1//! Phase 212.N.9 — `nros::main!()` proc-macro implementation.
2//!
3//! Replaces the legacy Entry pkg `build.rs + include!()` shape with a
4//! one-line `main.rs`. Four forms (per design-doc §11.6):
5//!
6//! ```ignore
7//! nros::main!();                                          // single-node self-bringup
8//! nros::main!(board = NativeBoard);                       // single-node, explicit board
9//! nros::main!(launch = "demo_bringup");                   // multi-node, default launch
10//! nros::main!(launch = "demo_bringup:sim.launch.xml");    // multi-node, explicit file
11//! nros::main!(board = X, launch = "Y:Z.xml", args = [("k", "v")]);
12//! ```
13//!
14//! Form 1 reads `[package.metadata.nros.entry] deploy = "<board>"` from
15//! the Entry pkg's own `Cargo.toml` and maps the board key to a board
16//! crate via a small lookup table. Forms 3+ resolve the bringup pkg
17//! through the N.10 workspace pkg-index, walk the N.11 launch.xml
18//! parser, and emit one `<pkg_ident>::register(runtime)?;` call per
19//! `<node pkg=… exec=…>` entry.
20//!
21//! ## Rebuild-correctness workaround
22//!
23//! Stable Rust proc-macros can't use the unstable
24//! `proc_macro::tracked_path::path()` API. Instead we emit
25//! `const _: &[u8] = include_bytes!("/abs/path");` for every file the
26//! macro read (launch.xml, every `package.xml` the pkg-index walked,
27//! the bringup's `system.toml`). Cargo's `include_bytes!` is tracked,
28//! so touching any of these files invalidates the Entry pkg's
29//! compilation cache.

49/// Parsed `nros::main!(...)` argument set.
50///
51/// Each form maps to a different population of these fields:
52///   - Form 1 (no args): all `None`.
53///   - Form 2 (`board = X`): `board = Some(X)`, launch / args `None`.
54///   - Form 3 (`launch = "Y"`): `launch = Some("Y")`, board derived
55///     from `[package.metadata.nros.entry] deploy`.
56///   - Form 4 (board + launch + args): all populated.
57#[derive(Default)]
58struct MainArgs {
59    /// Explicit board ident — used verbatim when supplied. None
60    /// triggers the `Cargo.toml [package.metadata.nros.entry]
61    /// deploy = "<board>"` lookup.
62    board: Option<SynPath>,
63    /// `"<bringup_pkg>"` or `"<bringup_pkg>:<file.launch.xml>"`. None
64    /// triggers the single-node self-bringup path (emit
65    /// `<this_pkg>::register(runtime)?;`).
66    launch: Option<LitStr>,
67    /// `args = [("k", "v"), ...]`. Forwarded to the launch parser as
68    /// argument overrides.
69    args: Vec<(String, String)>,
70    /// Phase 211.F — `host = "<id>"`: multi-host partition. When set, keep only
71    /// launch nodes whose `<node machine="…">` equals `<id>` plus all unhosted
72    /// (shared) nodes — mirrors `nros codegen entry --host` / `Plan::for_host`.
73    /// `None` = single-host / unfiltered (every node). A per-host Entry pkg
74    /// passes its target host so a multi-host launch bakes one runnable entry
75    /// per host (CLI/macro parity, now that nros-macros builds against the
76    /// in-tree nros-cli-core carrying `NodeSpec.machine`).
77    host: Option<String>,
78    /// Phase 216.B.4 — `custom_tasks = [adc_sample, ui_redraw]`. Each
79    /// ident becomes an extra `#[task]` trampoline inside the
80    /// generated `mod __nros_app` body when the dispatched framework
81    /// is RTIC. `None` = not supplied (the default — no extra tasks).
82    /// `Some(vec![])` = supplied as an empty list (still no extra
83    /// tasks, but the key was present — distinguished so the
84    /// OwnedSpin/Embassy misuse error fires even on `[]`).
85    custom_tasks: Option<Vec<Ident>>,
86    /// Span of the `custom_tasks` key, retained for diagnostics when
87    /// rejecting the key under a non-RTIC framework.
88    custom_tasks_span: Option<Span>,
89}
90
91impl Parse for MainArgs {
92    fn parse(input: ParseStream) -> syn::Result<Self> {
93        let mut out = MainArgs::default();
94        if input.is_empty() {
95            return Ok(out);
96        }
97        // Parse `key = value` pairs, comma-separated. Allow a trailing
98        // comma — matches every other Rust attribute-arg syntax.
99        let pairs: Punctuated<KeyValue, Token![,]> = Punctuated::parse_terminated(input)?;
100        for KeyValue { key, value } in pairs {
101            match key.to_string().as_str() {
102                "board" => {
103                    let path = match value {
104                        KvValue::Path(p) => p,
105                        KvValue::Str(s) => {
106                            return Err(syn::Error::new(
107                                s.span(),
108                                "expected board ident (e.g. `NativeBoard`), got string literal",
109                            ));
110                        }
111                        KvValue::Args(_) | KvValue::IdentList(_) => {
112                            return Err(syn::Error::new(
113                                key.span(),
114                                "`board = ` takes a type path, not a list",
115                            ));
116                        }
117                    };
118                    out.board = Some(path);
119                }
120                "launch" => {
121                    let s = match value {
122                        KvValue::Str(s) => s,
123                        _ => {
124                            return Err(syn::Error::new(
125                                key.span(),
126                                "`launch = ` takes a string literal",
127                            ));
128                        }
129                    };
130                    out.launch = Some(s);
131                }
132                "host" => {
133                    let s = match value {
134                        KvValue::Str(s) => s,
135                        _ => {
136                            return Err(syn::Error::new(
137                                key.span(),
138                                "`host = ` takes a string literal (the target machine id)",
139                            ));
140                        }
141                    };
142                    out.host = Some(s.value());
143                }
144                "args" => {
145                    let list = match value {
146                        KvValue::Args(pairs) => pairs,
147                        _ => {
148                            return Err(syn::Error::new(
149                                key.span(),
150                                "`args = ` takes a list of `(\"key\", \"value\")` tuples",
151                            ));
152                        }
153                    };
154                    out.args = list;
155                }
156                "custom_tasks" => {
157                    // Phase 216.B.4 — `custom_tasks = [ident, ident,
158                    // ...]`. Stored even when empty so the framework-
159                    // dispatch error fires regardless of list length.
160                    let idents = match value {
161                        KvValue::IdentList(v) => v,
162                        _ => {
163                            return Err(syn::Error::new(
164                                key.span(),
165                                "`custom_tasks = ` takes a list of fn idents, \
166                                 e.g. `custom_tasks = [adc_sample, ui_redraw]`",
167                            ));
168                        }
169                    };
170                    out.custom_tasks = Some(idents);
171                    out.custom_tasks_span = Some(key.span());
172                }
173                other => {
174                    return Err(syn::Error::new(
175                        key.span(),
176                        format!(
177                            "unknown `nros::main!` argument `{other}` \
178                             (expected one of: board, launch, host, args, custom_tasks)"
179                        ),
180                    ));
181                }
182            }
183        }
184        Ok(out)
185    }
186}

188struct KeyValue {
189    key: Ident,
190    value: KvValue,
191}
192
193enum KvValue {
194    /// `board = NativeBoard` / `board = ::nros_board_native::NativeBoard`
195    Path(SynPath),
196    /// `launch = "demo_bringup:sim.launch.xml"`
197    Str(LitStr),
198    /// `args = [("use_sim", "true"), ...]`
199    Args(Vec<(String, String)>),
200    /// Phase 216.B.4 — `custom_tasks = [adc_sample, ui_redraw, ...]`.
201    /// Empty list is valid (parses to `vec![]`).
202    IdentList(Vec<Ident>),
203}
204
205impl Parse for KeyValue {
206    fn parse(input: ParseStream) -> syn::Result<Self> {
207        let key: Ident = input.parse()?;
208        input.parse::<Token![=]>()?;
209        // Try the array form first — two shapes share the bracket:
210        // `args = [("k","v"), ...]` (tuple-string pairs) and
211        // `custom_tasks = [foo, bar]` (bare idents, Phase 216.B.4).
212        // Dispatch on the key name so each form's parser sees only
213        // its expected token shape (cleaner diagnostics).
214        if input.peek(syn::token::Bracket) {
215            let content;
216            syn::bracketed!(content in input);
217            if key == "custom_tasks" {
218                // Bare-ident list; empty `[]` parses to `vec![]`.
219                let idents: Punctuated<Ident, Token![,]> = Punctuated::parse_terminated(&content)?;
220                let collected: Vec<Ident> = idents.into_iter().collect();
221                return Ok(KeyValue {
222                    key,
223                    value: KvValue::IdentList(collected),
224                });
225            }
226            let pairs: Punctuated<TupleStrPair, Token![,]> =
227                Punctuated::parse_terminated(&content)?;
228            let collected = pairs
229                .into_iter()
230                .map(|p| (p.k.value(), p.v.value()))
231                .collect();
232            return Ok(KeyValue {
233                key,
234                value: KvValue::Args(collected),
235            });
236        }
237        // A bare string literal -> KvValue::Str. Anything else -> Path.
238        if input.peek(LitStr) {
239            let s: LitStr = input.parse()?;
240            return Ok(KeyValue {
241                key,
242                value: KvValue::Str(s),
243            });
244        }
245        let path: SynPath = input.parse()?;
246        Ok(KeyValue {
247            key,
248            value: KvValue::Path(path),
249        })
250    }
251}

253struct TupleStrPair {
254    k: LitStr,
255    v: LitStr,
256}
257
258impl Parse for TupleStrPair {
259    fn parse(input: ParseStream) -> syn::Result<Self> {
260        let content;
261        syn::parenthesized!(content in input);
262        let k: LitStr = content.parse()?;
263        content.parse::<Token![,]>()?;
264        let v: LitStr = content.parse()?;
265        Ok(TupleStrPair { k, v })
266    }

269/// Entry point — emits the `fn main()` body. Errors surface as
270/// `compile_error!()` spans pointing at the macro invocation.
271pub fn expand(input: TokenStream) -> TokenStream {
272    let args = parse_macro_input!(input as MainArgs);
273    match build_main(args) {
274        Ok(ts) => ts.into(),
275        Err(err) => err.to_compile_error().into(),
276    }
277}

279/// Errors carry a span — when we have a relevant token, attach it; when
280/// the failure is environmental (Cargo.toml missing, launch parse
281/// failed), fall back to `Span::call_site()`.
282type MacroResult<T> = std::result::Result<T, syn::Error>;
283
284fn build_main(args: MainArgs) -> MacroResult<proc_macro2::TokenStream> {
285    // CARGO_MANIFEST_DIR is set by cargo when compiling proc-macro
286    // consumers; if missing we fail loud — proc-macros without a
287    // manifest dir would have no way to find Cargo.toml or workspace
288    // root.
289    let manifest_dir = std::env::var_os("CARGO_MANIFEST_DIR").ok_or_else(|| {
290        syn::Error::new(
291            Span::call_site(),
292            "nros::main!: CARGO_MANIFEST_DIR not set (cargo must drive the build)",
293        )
294    })?;
295    let manifest_dir = PathBuf::from(manifest_dir);
296
297    // List of files that participated in the macro's decision so we
298    // can emit `include_bytes!` rebuild stamps below. Always
299    // canonicalised so the paths survive cargo's relocation tricks.
300    let mut tracked: Vec<PathBuf> = Vec::new();
301
302    // --- Board resolution ---
303    // `deploy` is `Some(...)` only when the macro had to read the
304    // Entry pkg's `[package.metadata.nros.entry] deploy = "..."` key
305    // (form 1). When the user passes `board = X` directly we have no
306    // deploy string and default to the `OwnedSpin` framework — RTIC
307    // / Embassy require the `deploy = "rtic-stm32f4"` / `deploy =
308    // "embassy-stm32f4"` opt-in for now.
309    let (board_path, deploy_for_framework): (SynPath, Option<String>) = match &args.board {
310        Some(p) => (p.clone(), None),
311        None => {
312            let cargo_toml = manifest_dir.join("Cargo.toml");
313            tracked.push(cargo_toml.clone());
314            let deploy = read_entry_deploy(&cargo_toml).map_err(|e| {
315                syn::Error::new(
316                    Span::call_site(),
317                    format!(
318                        "nros::main!: failed to read `[package.metadata.nros.entry] deploy` \
319                         from `{}`: {e}\n  Hint: add `[package.metadata.nros.entry] deploy = \
320                         \"<board>\"` (e.g. `\"native\"`, `\"freertos\"`, `\"zephyr\"`) \
321                         to your Cargo.toml, or pass `board = MyBoard` to the macro.",
322                        cargo_toml.display()
323                    ),
324                )
325            })?;
326            let resolved = board_path_for(&deploy).ok_or_else(|| {
327                syn::Error::new(
328                    Span::call_site(),
329                    format!(
330                        "nros::main!: unknown board `{deploy}` in \
331                         `[package.metadata.nros.entry] deploy`. \
332                         Known boards: {}.\n  Pass `board = <YourBoardZst>` explicitly \
333                         if your board crate is not in the default table.",
334                        known_boards_csv()
335                    ),
336                )
337            })?;
338            (resolved, Some(deploy))
339        }
340    };
341    let framework = match deploy_for_framework.as_deref() {
342        Some(d) => framework_for(d),
343        None => Framework::OwnedSpin,
344    };
345
346    // Phase 216.B.4 — `custom_tasks = [...]` only applies to the
347    // RTIC emit. OwnedSpin / Embassy have no `mod __nros_app { ... }`
348    // body for the macro to splice into, so flag misuse early with a
349    // span pointing at the key.
350    if args.custom_tasks.is_some() && framework != Framework::Rtic {
351        let span = args.custom_tasks_span.unwrap_or_else(Span::call_site);
352        let framework_label = match framework {
353            Framework::OwnedSpin => "OwnedSpin",
354            Framework::Embassy => "Embassy",
355            Framework::Zephyr => "Zephyr",
356            Framework::Esp32 => "Esp32",
357            Framework::Rtic => unreachable!(),
358        };
359        return Err(syn::Error::new(
360            span,
361            format!(
362                "nros::main!: `custom_tasks = [...]` is only valid for the RTIC framework \
363                 (current framework: {framework_label}). \
364                 RTIC splices each ident as a `#[task]` inside the generated `mod __nros_app` \
365                 body; other frameworks have no equivalent splice point."
366            ),
367        ));
368    }
369
370    // --- Phase 228.G — per-tier resolution state (RFC-0032 §6) ---
371    // Populated only in the launch arm (where bringup_dir + node pkgs are in
372    // scope). `node_groups` maps a node *instance* name → its declared callback
373    // groups; `node_instances` is every launch node name (for the
374    // instance-identity check vs `[[component]]`). `resolved_tiers` stays `None`
375    // unless `system.toml` declares `[tiers.*]`.
376    let mut node_groups: BTreeMap<String, Vec<CallbackGroupDecl>> = BTreeMap::new();
377    let mut node_instances: Vec<String> = Vec::new();
378    let mut resolved_tiers: Option<ResolvedTierTable> = None;
379    // Phase 264 W2 — `[lifecycle]` boot autostart from `system.toml` (launch arm only).
380    let mut lifecycle_code: Option<u8> = None;
381    // Phase 264 W4b — `[param_services]` declared in `system.toml` (launch arm only) →
382    // register the ROS 2 param services + seed the volatile store from the baked params.
383    let mut param_services_enabled = false;
384    // phase-267 W1c/C4 — when `system.toml` declares a `[[bridge]]` AND `nros sync`
385    // has generated `<bringup>/nros-bridge.toml`, the entry is a cross-RMW bridge:
386    // the macro emits a `run_from_config_str(include_str!(<that file>))` main
387    // instead of the ordinary register/spin entry. Holds the absolute path.
388    let mut bridge_config_path: Option<PathBuf> = None;
389    // Issue 0106 — the RMW backends a `[[bridge]]` Entry uses, read from
390    // `system.toml` (resolved via `[[bridge]]` endpoints + `[[domain]]` rmws).
391    // The macro emits `nros_rmw_<x>::register()` for each so the linker doesn't
392    // dead-strip the backend's self-register `.init_array` ctor (the data-driven
393    // `run_from_config` path references no backend symbol on its own).
394    let mut bridge_rmws: Vec<String> = Vec::new();
395    // Phase 264 W4a — per-node launch `<param name=… value=…/>` initials, parallel to
396    // `pkg_idents` (launch arm only). Each entry is the baked `(name, value)` slice that
397    // seeds the node's `NodeContext::param` at register time (RFC-0004 §10). Empty in the
398    // self-bringup arm (no launch file → no `<param>`).
399    let mut node_param_bakes: Vec<Vec<(String, String)>> = Vec::new();
400    // Phase 268 W1 — per-node launch `<node name= namespace=>` identity, parallel to
401    // `node_param_bakes` (launch arm only). Each entry is the baked `(name, namespace)`
402    // pair that `ExecutorSink::create_node` uses instead of the `NodeOptions` default
403    // (RFC-0046). Empty (`None`) in the self-bringup arm.
404    let mut node_identity_bakes: Vec<(String, String)> = Vec::new();
405
406    // --- Launch resolution → list of <pkg_ident> register calls ---
407    let pkg_idents: Vec<Ident> = match &args.launch {
408        None => {
409            // Form 1 / Form 2 — single-node self-bringup. Use this
410            // pkg's own name as the register target. The codegen-
411            // emitted ident is snake_case; we read CARGO_PKG_NAME and
412            // sanitise via the same `-` → `_` rule the existing
413            // `nros::node!()` macro uses (see `lib.rs::sanitize_pkg_name_for_symbol`).
414            //
415            // The bin target depends on the lib target of the same
416            // pkg via Cargo's automatic `extern crate <my_pkg>;`,
417            // so `<my_pkg>::register(runtime)?;` resolves at build
418            // time.
419            let pkg_name = std::env::var("CARGO_PKG_NAME").map_err(|_| {
420                syn::Error::new(Span::call_site(), "nros::main!: CARGO_PKG_NAME not set")
421            })?;
422            let crate_ident = pkg_to_crate_ident(&pkg_name);
423            // Form 1 self-bringup is opt-in: the user's lib crate
424            // must expose a `pub fn register(runtime)`. If this
425            // Entry pkg is binary-only (no companion lib), the
426            // emitted `<this_pkg>::register(...)` will fail to
427            // compile with a clear error; we don't try to detect
428            // that here.
429            vec![Ident::new(&crate_ident, Span::call_site())]
430        }
431        Some(launch_lit) => {
432            let launch_value = launch_lit.value();
433            // Walk the workspace from the Entry pkg's manifest dir.
434            let workspace_root =
435                nros_pkg_index::detect_workspace_root(&manifest_dir).map_err(|e| {
436                    syn::Error::new(
437                        launch_lit.span(),
438                        format!("nros::main!: detect_workspace_root: {e}"),
439                    )
440                })?;
441            let pkg_index = nros_pkg_index::build_pkg_index(&workspace_root).map_err(|e| {
442                syn::Error::new(
443                    launch_lit.span(),
444                    format!("nros::main!: build_pkg_index: {e}"),
445                )
446            })?;
447            // Track every package.xml the index walked.
448            for (_, pkg_dir) in pkg_index.pkgs() {
449                tracked.push(pkg_dir.join("package.xml"));
450            }
451
452            // Split `"bringup_pkg[:file.launch.xml]"`.
453            let (bringup_name, file_override) = match launch_value.split_once(':') {
454                Some((b, f)) => (b.trim().to_string(), Some(f.trim().to_string())),
455                None => (launch_value.trim().to_string(), None),
456            };
457            if bringup_name.is_empty() {
458                return Err(syn::Error::new(
459                    launch_lit.span(),
460                    "nros::main!: empty bringup pkg name in `launch = \"...\"`",
461                ));
462            }
463            let bringup_dir = pkg_index
464                .resolve_pkg(&bringup_name)
465                .map_err(|e| syn::Error::new(launch_lit.span(), format!("nros::main!: {e}")))?;
466
467            // Phase 264 W2 — read `[lifecycle]` so the macro can emit the REP-2002
468            // service registration + autostart (mirrors the bake). Unconditional
469            // (independent of the launch-file override below). Phase 264 W4b — also
470            // read `[param_services]` so the macro can emit the parameter-service
471            // registration + store seeding.
472            {
473                let st = bringup_dir.join("system.toml");
474                if st.exists() {
475                    tracked.push(st.clone());
476                    lifecycle_code = read_lifecycle_autostart(&st);
477                    param_services_enabled = read_param_services_enabled(&st);
478                    if read_has_bridge(&st) {
479                        let cfg = bringup_dir.join("nros-bridge.toml");
480                        if cfg.exists() {
481                            tracked.push(cfg.clone());
482                            bridge_config_path = Some(cfg);
483                            // Issue 0106 — collect the bridge's RMW backends so
484                            // the entry force-registers them (anti dead-strip).
485                            bridge_rmws = read_bridge_rmws(&st);
486                        }
487                    }
488                }
489            }
490
491            // Resolve the launch file. If no override, consult
492            // `system.toml::[system] default_launch` (default
493            // `system.launch.xml`).
494            let launch_filename = match file_override {
495                Some(s) => s,
496                None => {
497                    let system_toml = bringup_dir.join("system.toml");
498                    if system_toml.exists() {
499                        tracked.push(system_toml.clone());
500                        read_default_launch(&system_toml)
501                            .map_err(|e| {
502                                syn::Error::new(
503                                    launch_lit.span(),
504                                    format!("nros::main!: parse `{}`: {e}", system_toml.display()),
505                                )
506                            })?
507                            .unwrap_or_else(|| "system.launch.xml".to_string())
508                    } else {
509                        "system.launch.xml".to_string()
510                    }
511                }
512            };
513            let launch_path = bringup_dir.join("launch").join(&launch_filename);
514            tracked.push(launch_path.clone());
515            if !launch_path.exists() {
516                return Err(syn::Error::new(
517                    launch_lit.span(),
518                    format!(
519                        "nros::main!: launch file not found: `{}`",
520                        launch_path.display()
521                    ),
522                ));
523            }
524
525            // Parse the launch file via N.11.
526            let arg_overrides: Vec<(String, String)> = args.args.clone();
527            let desc =
528                nros_launch_parser::parse_launch_file(&launch_path, &pkg_index, &arg_overrides)
529                    .map_err(|e| {
530                        syn::Error::new(
531                            launch_lit.span(),
532                            format!(
533                                "nros::main!: parse launch file `{}`: {e}",
534                                launch_path.display()
535                            ),
536                        )
537                    })?;
538
539            // Walk every `<node>` (top-level + inside groups) and
540            // resolve to its Rust crate ident.
541            let mut node_specs = Vec::new();
542            for n in &desc.nodes {
543                node_specs.push(n.clone());
544            }
545            for g in &desc.groups {
546                for n in &g.nodes {
547                    node_specs.push(n.clone());
548                }
549            }
550
551            // Phase 211.F — multi-host partition: when `host = "<id>"` is set,
552            // keep only this host's nodes (`<node machine="<id>">`) + all
553            // unhosted (shared) nodes. Mirrors `nros codegen entry --host` /
554            // `Plan::for_host`, giving the macro path parity so a per-host Entry
555            // pkg (`nros::main!(launch = …, host = "<id>")`) bakes a runnable
556            // single-host slice of a multi-host launch.
557            if let Some(host) = args.host.as_deref() {
558                node_specs.retain(|n| match &n.machine {
559                    Some(m) => m == host,
560                    None => true,
561                });
562                if node_specs.is_empty() {
563                    return Err(syn::Error::new(
564                        launch_lit.span(),
565                        format!(
566                            "nros::main!(host = {host:?}): no nodes for this host in `{}` \
567                             (no `<node machine={host:?}>` and no unhosted/shared node)",
568                            launch_path.display()
569                        ),
570                    ));
571                }
572            }
573
574            let mut idents = Vec::new();
575            for node in node_specs {
576                // Look up the node pkg in the index, then derive the
577                // Rust crate ident from its `package.xml::<name>`.
578                let node_pkg_dir = pkg_index.resolve_pkg(&node.pkg).map_err(|e| {
579                    syn::Error::new(
580                        launch_lit.span(),
581                        format!("nros::main!: node pkg `{}`: {e}", node.pkg),
582                    )
583                })?;
584                tracked.push(node_pkg_dir.join("package.xml"));
585                // Is this a Rust pkg? Check for `Cargo.toml`. If
586                // missing, surface a clear error — C++ Node pkgs are
587                // out of scope for v1.
588                let cargo_toml = node_pkg_dir.join("Cargo.toml");
589                if !cargo_toml.exists() {
590                    return Err(syn::Error::new(
591                        launch_lit.span(),
592                        format!(
593                            "nros::main!: node pkg `{}` has no Cargo.toml at `{}`. \
594                             C++ Node pkgs are not yet supported in Rust Entry pkgs.",
595                            node.pkg,
596                            node_pkg_dir.display()
597                        ),
598                    ));
599                }
600                let crate_ident = pkg_to_crate_ident(&node.pkg);
601                idents.push(Ident::new(&crate_ident, Span::call_site()));
602
603                // Phase 264 W4a — bake this node's launch `<param>` initials (RFC-0004 §10).
604                // Parallel to `idents`: index i's params seed pkg `idents[i]`'s NodeContext.
605                node_param_bakes.push(
606                    node.params
607                        .iter()
608                        .map(|p| (p.name.clone(), p.value.clone()))
609                        .collect(),
610                );
611
612                // Phase 228.G — collect the node instance + its declared
613                // callback groups for tier resolution. The instance name keys
614                // the group map (RFC-0032 §7); when the launch `<node>` has no
615                // explicit name, fall back to the executable name.
616                let instance = node
617                    .name
618                    .clone()
619                    .filter(|s| !s.is_empty())
620                    .unwrap_or_else(|| node.exec.clone());
621
622                // Phase 268 W1 — bake the launch `<node name= namespace=>` identity
623                // (RFC-0046). The instance name is already resolved above (same logic as
624                // the W4a param rail). Namespace: `node.namespace` when present and
625                // non-empty; default `"/"` (ROS convention). `NodeSpec::namespace` is
626                // `Option<String>` in the launch parser.
627                {
628                    let baked_name = instance.clone();
629                    let baked_ns = node
630                        .namespace
631                        .as_deref()
632                        .filter(|s| !s.is_empty())
633                        .unwrap_or("/")
634                        .to_string();
635                    node_identity_bakes.push((baked_name, baked_ns));
636                }
637                let groups = read_node_callback_groups(&cargo_toml);
638                if !groups.is_empty() {
639                    node_groups.insert(instance.clone(), groups);
640                }
641                node_instances.push(instance);
642            }
643
644            // Phase 228.G — resolve the tier table from `system.toml` when it
645            // declares `[tiers.*]`. No tiers → leave `resolved_tiers = None`
646            // (the single-tier `BoardEntry::run` emit, byte-identical).
647            let system_toml = bringup_dir.join("system.toml");
648            if system_toml.exists() {
649                let cfg = read_system_tier_config(&system_toml).map_err(|e| {
650                    syn::Error::new(
651                        launch_lit.span(),
652                        format!(
653                            "nros::main!: parse tiers in `{}`: {e}",
654                            system_toml.display()
655                        ),
656                    )
657                })?;
658                if cfg.has_tiers {
659                    tracked.push(system_toml.clone());
660                    // Instance-identity (RFC-0032 §7): every launch node that
661                    // carries callback groups must name a `[[component]]`.
662                    let component_names: BTreeSet<&str> =
663                        cfg.component_names.iter().map(String::as_str).collect();
664                    for inst in node_groups.keys() {
665                        if !component_names.contains(inst.as_str()) {
666                            return Err(syn::Error::new(
667                                launch_lit.span(),
668                                format!(
669                                    "nros::main!: launch node `{inst}` declares callback groups \
670                                     but is not a `[[component]]` in `{}` (the launch \
671                                     `<node name>` must match a `system.toml` component name).",
672                                    system_toml.display()
673                                ),
674                            ));
675                        }
676                    }
677                    let rtos = derive_target_rtos(deploy_for_framework.as_deref());
678                    let table = resolve_tiers(
679                        &cfg.tiers,
680                        &cfg.node_overrides,
681                        &component_names,
682                        &node_groups,
683                        &rtos,
684                    )
685                    .map_err(|e| {
686                        syn::Error::new(
687                            launch_lit.span(),
688                            format!("nros::main!: tier resolution: {e}"),
689                        )
690                    })?;
691                    resolved_tiers = Some(table);
692                }
693            }
694
695            idents
696        }
697    };

699    // De-duplicate the tracked list — pkg-index walks can revisit
700    // a pkg dir's `package.xml` from multiple paths.
701    tracked.sort();
702    tracked.dedup();
703
704    // --- Emit ---
705    let tracked_consts = tracked.iter().filter_map(|p| {
706        // Skip non-existent paths. The macro may have stuffed a
707        // path that the filesystem doesn't surface (e.g. a missing
708        // `package.xml` when build_pkg_index synthesised a dir).
709        // `include_bytes!` on a missing path is a hard compile
710        // error, so be defensive.
711        if !p.exists() {
712            return None;
713        }
714        let s = p.to_string_lossy().into_owned();
715        let lit = LitStr::new(&s, Span::call_site());
716        Some(quote! {
717            const _: &[u8] = ::core::include_bytes!(#lit);
718        })
719    });

721    let register_calls: Vec<proc_macro2::TokenStream> = pkg_idents
722        .iter()
723        .enumerate()
724        .map(|(i, ident)| {
725            // Phase 264 W4a — set `runtime.params` to this node's baked launch `<param>`
726            // initials (a promoted `&'static` slice) before the register call, so the
727            // node's `register`/`init` observes its launch values via `ctx.param(name)`.
728            // Empty (self-bringup arm, or a node with no `<param>`) → reset to `&[]` so a
729            // prior node's params never leak into the next register.
730            let params = node_param_bakes.get(i).cloned().unwrap_or_default();
731            let param_lits = params.iter().map(|(name, value)| {
732                let n = LitStr::new(name, Span::call_site());
733                let v = LitStr::new(value, Span::call_site());
734                quote! { (#n, #v) }
735            });
736            // Phase 268 W1 — set `runtime.node_identity` to this node's launch
737            // `<node name= namespace=>` identity before the register call (RFC-0046).
738            // `None` in the self-bringup arm (no launch, `node_identity_bakes` empty) so
739            // no identity leaks between components (same discipline as the params reset).
740            let identity_emit = match node_identity_bakes.get(i) {
741                Some((name, ns)) => {
742                    let n = LitStr::new(name, Span::call_site());
743                    let s = LitStr::new(ns, Span::call_site());
744                    quote! {
745                        runtime.node_identity = ::core::option::Option::Some((#n, #s));
746                    }
747                }
748                None => quote! {
749                    runtime.node_identity = ::core::option::Option::None;
750                },
751            };
752            quote! {
753                runtime.params = &[ #( #param_lits ),* ];
754                #identity_emit
755                ::#ident::register(runtime)?;
756            }
757        })
758        .collect();
759    // Node count for the Zephyr framework boot banner (literal baked at
760    // expansion time so the runtime body needs no extra import).
761    let num_register_calls = register_calls.len();
762
763    // Phase 264 W2 — `[lifecycle]` wiring: when `system.toml` declares it, register
764    // the REP-2002 services + drive boot autostart right after the per-node
765    // `register` calls (the executor is built, the nodes are installed). No-op token
766    // stream when absent. `apply_lifecycle` is a no-op unless the Entry enabled
767    // `nros/lifecycle-services`, so this is inert without the feature.
768    let lifecycle_call: proc_macro2::TokenStream = match lifecycle_code {
769        Some(code) => quote! {
770            runtime.apply_lifecycle(#code).map_err(
771                |_| ::nros::__macro_support::nros_platform::RuntimeError::NodeRegister("lifecycle"),
772            )?;
773        },
774        None => quote! {},
775    };
776
777    // Phase 264 W4b — `[param_services]` wiring: when `system.toml` declares it, register
778    // the 6 ROS 2 parameter services + seed the volatile param store with the aggregate
779    // of every node's launch-baked `<param>` initials, right after the per-node `register`
780    // calls. No-op token stream when absent. `apply_param_services` is a no-op unless the
781    // Entry enabled `nros/param-services`, so this is inert without the feature. The seed
782    // values are the raw launch strings; the runtime infers each `ParameterValue` type.
783    let param_services_call: proc_macro2::TokenStream = if param_services_enabled {
784        let seed_lits = node_param_bakes.iter().flatten().map(|(name, value)| {
785            let n = LitStr::new(name, Span::call_site());
786            let v = LitStr::new(value, Span::call_site());
787            quote! { (#n, #v) }
788        });
789        quote! {
790            runtime.apply_param_services(&[ #( #seed_lits ),* ]).map_err(
791                |_| ::nros::__macro_support::nros_platform::RuntimeError::NodeRegister("param_services"),
792            )?;
793        }
794    } else {
795        quote! {}
796    };
797
798    // Phase 228.G (RFC-0032 §5) — the OwnedSpin entry call. Multi-tier
799    // (`[tiers.*]` present, more than the synthesized `default` tier) emits
800    // `<Board>::run_tiers(TIERS, register-only-closure)`; the board owns the
801    // per-tier spin. Single-tier / no tiers keeps the unchanged
802    // `BoardEntry::run` path (`setup` owns the bounded hosted spin) — so the
803    // emitted TU is byte-identical to pre-228 for every current example.
804    // Issue #48 cause 1 — bake the deploy overlay from
805    // `[package.metadata.nros.deploy.<board>]`. Only Form 1 (deploy key present)
806    // has a board key to read; Form 2 (explicit `board = X`) gets an all-`None`
807    // overlay, so `run_with_deploy` then behaves exactly like `run`.
808    let mut deploy_overlay_lit = match deploy_for_framework.as_deref() {
809        Some(board_key) => read_deploy_overlay(&manifest_dir.join("Cargo.toml"), board_key),
810        None => DeployOverlayLit::default(),
811    };
812    // Issue #98 — a single-node launch names the primary session (the ROS graph
813    // node name) after that node, instead of the board default `"node"`. With
814    // multiple nodes they share one primary session, so naming it after one would
815    // be wrong — per-node naming is the deferred multi-node piece, so leave the
816    // overlay name unset and the board keeps `"node"`.
817    if let [only] = node_instances.as_slice() {
818        deploy_overlay_lit.node_name = Some(only.clone());
819    }
820    let deploy_overlay_ts = deploy_overlay_tokens(&deploy_overlay_lit);
821
822    // W4b — bake the single-node boot identity into a `NROS_BOOT_CONFIG` static
823    // placed in the `.nros_boot_config` linker section (bare-metal only; hosted
824    // gets a plain `#[unsafe(no_mangle)] #[used]` static that is still referenceable).
825    // The static is emitted once, before `body_ts`, so `&NROS_BOOT_CONFIG`
826    // resolves in every framework arm's `deploy_overlay_ts` use site.
827    let boot_config_static_ts = {
828        let node_name_opt = match &deploy_overlay_lit.node_name {
829            Some(s) => quote! { ::core::option::Option::Some(#s) },
830            None => quote! { ::core::option::Option::None },
831        };
832        let locator_opt = match &deploy_overlay_lit.locator {
833            Some(s) => quote! { ::core::option::Option::Some(#s) },
834            None => quote! { ::core::option::Option::None },
835        };
836        let domain_opt = match deploy_overlay_lit.domain_id {
837            Some(d) => quote! { ::core::option::Option::Some(#d) },
838            None => quote! { ::core::option::Option::None },
839        };
840        quote! {
841            #[used]
842            #[cfg_attr(target_os = "none", unsafe(link_section = ".nros_boot_config"))]
843            #[unsafe(no_mangle)]
844            static NROS_BOOT_CONFIG: ::nros::BakedBootConfig =
845                ::nros::BakedBootConfig::new(
846                    #node_name_opt,
847                    #locator_opt,
848                    #domain_opt,
849                    ::core::option::Option::None,
850                );
851        }
852    };
853
854    // Phase 244.D1 — `target_os = "none"` entry shape for the OwnedSpin
855    // framework. FreeRTOS / threadx-linux have a C runtime that calls `main`,
856    // so they keep the `extern "C" fn main`. A pure bare-metal Cortex-M image
857    // has no C runtime; its reset vector needs a `#[cortex_m_rt::entry]`. Both
858    // funnel through the shared `__nros_entry_run`.
859    let none_entry_ts: proc_macro2::TokenStream =
860        if is_baremetal_cortexm_deploy(deploy_for_framework.as_deref()) {
861            quote! {
862                #[cfg(target_os = "none")]
863                #[::cortex_m_rt::entry]
864                fn __nros_cortex_m_reset() -> ! {
865                    // `run_with_deploy` loops forever on success (firmware
866                    // lifetime) and exits via the board on spin error; reaching
867                    // here means `setup()` returned `Err` before the spin loop.
868                    let _ = __nros_entry_run();
869                    loop {
870                        ::cortex_m::asm::wfi();
871                    }
872                }
873            }
874        } else {
875            quote! {
876                #[cfg(target_os = "none")]
877                #[unsafe(no_mangle)]
878                pub extern "C" fn main() -> i32 {
879                    match __nros_entry_run() {
880                        ::core::result::Result::Ok(()) => 0,
881                        ::core::result::Result::Err(_) => 1,
882                    }
883                }
884            }
885        };
886
887    let multi_tier = resolved_tiers.as_ref().filter(|t| !t.is_single_tier());
888    let entry_call: proc_macro2::TokenStream = match multi_tier {
889        Some(table) => {
890            let tiers_ts = tier_specs_tokens(table);
891            quote! {
892                <#board_path>::run_tiers(
893                    // Issue #48 cause 1 — thread the deploy overlay into the
894                    // multi-tier path too (firmware boards apply it to their boot
895                    // `Config`; hosted boards ignore it).
896                    &#deploy_overlay_ts,
897                    #tiers_ts,
898                    |runtime: &mut ::nros::__macro_support::nros_platform::RuntimeCtx<'_>|
899                        -> ::core::result::Result<
900                            (),
901                            ::nros::__macro_support::nros_platform::RuntimeError,
902                        >
903                    {
904                        // Register-only: the board sets each tier's
905                        // `active_groups` filter and owns the spin loop.
906                        // W4c — param services BEFORE the node registers, so the store
907                        // exists when each cell captures it (cell → `ctx.parameter`).
908                        #param_services_call
909                        #( #register_calls )*
910                        #lifecycle_call
911                        ::core::result::Result::Ok(())
912                    },
913                )
914            }
915        }
916        None => quote! {
917            // Issue #48 cause 1 — `run_with_deploy` applies the deploy-metadata
918            // overlay (locator / ip / gateway / domain) to the board's boot
919            // config. The default trait body ignores the overlay and forwards
920            // to `run`, so hosted / framework boards are byte-identical; the
921            // FreeRTOS / bare-metal boards override it to stop the
922            // `[deploy.<board>]` block being inert.
923            <#board_path as ::nros::__macro_support::nros_platform::BoardEntry>::run_with_deploy(
924                &#deploy_overlay_ts,
925                |runtime: &mut ::nros::__macro_support::nros_platform::RuntimeCtx<'_>|
926                    -> ::core::result::Result<
927                        (),
928                        ::nros::__macro_support::nros_platform::RuntimeError,
929                    >
930                {
931                    // W4c — param services BEFORE the node registers, so the store
932                    // exists when each cell captures it (cell → `ctx.parameter`).
933                    #param_services_call
934                    #( #register_calls )*
935                    #lifecycle_call
936                    #[cfg(not(target_os = "none"))]
937                    __nros_hosted_spin_if_requested(runtime)?;
938                    ::core::result::Result::Ok(())
939                },
940            )
941        },
942    };

944    // Phase 216 final wave — Node-pkg registration for framework
945    // targets (RTIC + Embassy). The OwnedSpin branch keeps its
946    // existing `<pkg>::register(runtime)?;` flow inside the
947    // `BoardEntry::run` closure; the framework branches instead emit
948    // `<pkg>::register_dispatch(&mut executor)?;` calls into the
949    // generated `#[init]` body, which push the per-Node
950    // `(state, on_callback)` pair into the `Executor`'s dispatch-slot
951    // registry that the framework dispatch task walks.
952    //
953    // Source of truth: `[package.metadata.nros.entry] node_pkgs =
954    // ["pkg_a", "pkg_b"]` in the Entry pkg's `Cargo.toml`. When the
955    // key is absent we fall back to the Entry pkg's own name with a
956    // conventional `_entry` suffix stripped — covers the common
957    // self-bringup shape where a single Node pkg `foo_pkg` has a
958    // sibling Entry pkg `foo_pkg_entry` (though the framework
959    // examples today don't follow this — they always declare
960    // `node_pkgs = [...]` explicitly).
961    let framework_node_pkg_idents: Vec<Ident> = match framework {
962        // OwnedSpin (incl. NuttX) + Zephyr + Esp32 all register via the
963        // launch-resolved `register_calls` (the `RuntimeCtx`-based
964        // `<pkg>::register` flow), NOT the RTIC/Embassy
965        // `register_dispatch` splice.
966        Framework::OwnedSpin | Framework::Zephyr | Framework::Esp32 => Vec::new(),
967        Framework::Rtic | Framework::Embassy => {
968            let cargo_toml = manifest_dir.join("Cargo.toml");
969            let from_metadata = read_entry_node_pkgs(&cargo_toml).map_err(|e| {
970                syn::Error::new(
971                    Span::call_site(),
972                    format!(
973                        "nros::main!: failed to read `[package.metadata.nros.entry] node_pkgs` \
974                         from `{}`: {e}",
975                        cargo_toml.display()
976                    ),
977                )
978            })?;
979            let names: Vec<String> = match from_metadata {
980                Some(v) => v,
981                None => {
982                    // Self-bringup fallback: strip `_entry` suffix
983                    // from the Entry pkg's own name.
984                    let pkg_name = std::env::var("CARGO_PKG_NAME").map_err(|_| {
985                        syn::Error::new(Span::call_site(), "nros::main!: CARGO_PKG_NAME not set")
986                    })?;
987                    let stripped = pkg_name
988                        .strip_suffix("_entry")
989                        .or_else(|| pkg_name.strip_suffix("-entry"))
990                        .unwrap_or(&pkg_name);
991                    vec![stripped.to_string()]
992                }
993            };
994            names
995                .into_iter()
996                .map(|n| Ident::new(&pkg_to_crate_ident(&n), Span::call_site()))
997                .collect()
998        }
999    };
1000    let framework_register_dispatch_calls: Vec<proc_macro2::TokenStream> =
1001        framework_node_pkg_idents
1002            .iter()
1003            .map(|ident| {
1004                quote! {
1005                    ::#ident::register_dispatch(&mut executor)
1006                        .expect("nros::main!: register_dispatch — executor dispatch-slot table full");
1007                }
1008            })
1009            .collect();
1010
1011    // Phase 216.B.3 — framework-dispatched emit body. `OwnedSpin`
1012    // keeps the long-standing `fn __nros_entry_run + fn main` shape
1013    // (BoardEntry::run owns the spin loop). `Rtic` emits a
1014    // `#[rtic::app(...)]` skeleton that delegates to
1015    // `RticBoardEntry::init_hardware` from the framework-generated
1016    // `#[init]` body. `Embassy` is a hard error pointing at the
1017    // 216.C.3 sibling that lands the emit body.
1018    let body_ts: proc_macro2::TokenStream = match framework {
1019        Framework::OwnedSpin => quote! {
1020            // Phase 213.C follow-up — emit two cfg-gated entry shapes so
1021            // both hosted (POSIX / NuttX / threadx-linux) and embedded
1022            // (FreeRTOS / bare-metal `target_os = "none"`) targets resolve
1023            // a working `main`. The shared body is factored into a private
1024            // `__nros_entry_run` returning `Result` so neither arm
1025            // duplicates the closure logic.
1026            fn __nros_entry_run() -> ::core::result::Result<
1027                (),
1028                ::nros::__macro_support::nros_platform::RuntimeError,
1029            > {
1030                // Phase 244.D1 — custom-transport install seam. Install a
1031                // board custom transport (e.g. the XRCE-over-UART vtable)
1032                // selected by `deploy.transport`, BEFORE the RMW registers.
1033                // XRCE's `set_custom_transport_ops` must precede `register`.
1034                // The default `setup_transport` is a no-op so boards with a
1035                // built-in transport are byte-identical. This call is always
1036                // emitted (it is not dead code): see `BoardEntry::setup_transport`
1037                // for the design rationale and the current override in
1038                // `nros-board-mps2-an385` (`xrce-transport` feature).
1039                #[cfg(target_os = "none")]
1040                <#board_path as ::nros::__macro_support::nros_platform::BoardEntry>::setup_transport(
1041                    &#deploy_overlay_ts,
1042                );
1043                // Phase 249 P1 — the RMW backend register is OWNED BY THE BOARD
1044                // (Phase 248 C5a: each board's boot path calls its linked
1045                // `nros_rmw_<x>::register()`, gated on the board's `rmw-<x>` feature),
1046                // ordered after `setup_transport` inside the board's `run`. The former
1047                // `#[cfg(target_os="none")] ::nros::__register_linked_rmw()` emit here
1048                // was a no-op vestige of the retired linkme-walk path — removed.
1049                #entry_call
1050            }
1051
1052            // `#[allow(dead_code)]` — the multi-tier `run_tiers` entry path
1053            // (Phase 228.G) owns its own spin and never calls these, so they
1054            // are unused in that emit; harmless in the single-tier path.
1055            #[cfg(not(target_os = "none"))]
1056            #[allow(dead_code)]
1057            fn __nros_env_usize(name: &str, default: usize) -> usize {
1058                ::std::env::var(name)
1059                    .ok()
1060                    .and_then(|v| v.parse::<usize>().ok())
1061                    .unwrap_or(default)
1062            }
1063
1064            #[cfg(not(target_os = "none"))]
1065            #[allow(dead_code)]
1066            fn __nros_hosted_spin_if_requested(
1067                runtime: &mut ::nros::__macro_support::nros_platform::RuntimeCtx<'_>,
1068            ) -> ::core::result::Result<
1069                (),
1070                ::nros::__macro_support::nros_platform::RuntimeError,
1071            > {
1072                let total_ms = __nros_env_usize("NROS_ENTRY_SPIN_MS", 0);
1073                if total_ms == 0 {
1074                    return ::core::result::Result::Ok(());
1075                }
1076
1077                let step_ms = __nros_env_usize("NROS_ENTRY_SPIN_STEP_MS", 10)
1078                    .clamp(1, total_ms.max(1));
1079                let expect_messages = __nros_env_usize("NROS_ENTRY_EXPECT_MESSAGE_CALLBACKS", 0);
1080                let deadline = ::std::time::Instant::now()
1081                    + ::std::time::Duration::from_millis(total_ms as u64);
1082
1083                loop {
1084                    runtime
1085                        .runtime
1086                        .spin_once(step_ms as u32)
1087                        .map_err(|_| ::nros::__macro_support::nros_platform::RuntimeError::Spin)?;
1088
1089                    let (callbacks, messages) = runtime.runtime.observed_callback_counts();

1712/// Pure half of [`read_bridge_rmws`] — parse the rmw set from `system.toml` text.
1713fn parse_bridge_rmws(raw: &str) -> Vec<String> {
1714    let Ok(v) = toml::from_str::<toml::Value>(raw) else {
1715        return Vec::new();
1716    };

1718        v.get("domain")
1719            .and_then(|d| d.as_array())?
1720            .iter()
1721            .find(|item| item.get("name").and_then(|n| n.as_str()) == Some(name))
1722            .and_then(|d| d.get("rmw"))

1724            .map(String::from)
1725    };

1727    if let Some(bridges) = v.get("bridge").and_then(|b| b.as_array()) {
1728        for bridge in bridges {
1729            for key in ["from", "to"] {

1734                    Some((r, _)) => Some(r.to_string()),

1736                };

1738                    && !r.is_empty()
1739                    && !rmws.contains(&r)
1740                {
1741                    rmws.push(r);
1742                }
1743            }
1744        }
1745    }

1821/// Parse a dotted IPv4 string (`"10.0.2.15"`) into 4 octets. Returns `None`

1823/// default rather than baking garbage.
1824fn parse_ipv4_lit(s: &str) -> Option<[u8; 4]> {
1825    let mut out = [0u8; 4];
1826    let mut n = 0usize;
1827    for part in s.split('.') {
1828        if n >= 4 {
1829            return None;
1830        }
1831        out[n] = part.parse::<u8>().ok()?;
1832        n += 1;
1833    }
1834    if n == 4 { Some(out) } else { None }

2100/// Emit a `&[TierSpec]` literal from the resolved tier table (Phase 228.G,
2101/// RFC-0032 §5). `priority` is the raw per-RTOS value; `groups` is the tier's
2102/// distinct callback-group ids (the executor's `active_groups` filter).
2103fn tier_specs_tokens(table: &ResolvedTierTable) -> proc_macro2::TokenStream {
2104    let entries = table.tiers.iter().map(|t| {
2105        let name = &t.name;
2106        let mut groups: Vec<&str> = t.members.iter().map(|(_, g)| g.as_str()).collect();
2107        groups.sort();
2108        groups.dedup();
2109        let priority = t.priority;
2110        let stack_bytes = t.stack_bytes.unwrap_or(0) as usize;
2111        let spin_period_us = t.spin_period_us.unwrap_or(1000);
2112        quote! {
2113            ::nros::__macro_support::nros_platform::TierSpec {
2114                name: #name,
2115                groups: &[ #(#groups),* ],
2116                priority: #priority,
2117                stack_bytes: #stack_bytes,
2118                spin_period_us: #spin_period_us,
2119            }
2120        }
2121    });
2122    quote! { &[ #(#entries),* ] }
2123}

2182/// Phase 216.B.3 — boot-framework dispatch for `nros::main!()`.
2183///
2184/// Distinct from [`board_path_for`] (which only resolves the board
2185/// crate ZST). Frameworks are orthogonal to RMW + platform: each
2186/// board crate carries its own
2187/// `[package.metadata.nros.board] framework = "<f>"` knob (consumed
2188/// by 216.D.1 `nros check`). The macro keys off the Entry pkg's
2189/// `deploy = "..."` value directly to avoid a fs round-trip into the
2190/// board crate's manifest at proc-macro expansion time; the long-term
2191/// spec reads the board's manifest, but the skeleton hardcodes the
2192/// table to match the `board_path_for` row above.
2193#[derive(Debug, Clone, Copy, PartialEq, Eq)]
2194enum Framework {
2195    /// Board owns the spin loop (`BoardEntry::run`). Default for
2196    /// every board key not explicitly routed below.
2197    OwnedSpin,
2198    /// RTIC framework owns the spin loop. The macro emits a
2199    /// `#[rtic::app(...)]` module + `#[init]` body that delegates to
2200    /// `RticBoardEntry::init_hardware`.
2201    Rtic,
2202    /// Embassy framework owns the spin loop. The macro emits a
2203    /// `#[embassy_executor::main] async fn main(spawner: Spawner)` body
2204    /// that delegates to `EmbassyBoardEntry::init_hardware`.
2205    Embassy,
2206    /// Phase 225.P — Zephyr RTOS owns boot + `main`. The macro emits a
2207    /// `#[unsafe(no_mangle)] pub extern "C" fn rust_main()` staticlib
2208    /// export (consumed by `zephyr-lang-rust`'s `rust_cargo_application`)
2209    /// that gates on `ZephyrBoard::wait_link_up`, opens an `Executor`,
2210    /// wraps it in `ExecutorNodeRuntime`, registers each launch-named
2211    /// Node pkg, then spins — bounded on hosted `native_sim`, forever
2212    /// otherwise. There is NO `BoardEntry::run` (Zephyr forbids a Rust
2213    /// `fn main`).
2214    Zephyr,
2215    /// Phase 225.O — ESP32-C3 (esp-hal). esp-riscv-rt's `_start` calls
2216    /// the esp-hal entry registration, so a bare `extern "C" fn main`
2217    /// (the `OwnedSpin` `target_os = "none"` shape) does not boot. The
2218    /// macro emits `#[::esp_hal::main] fn main() -> !` that delegates to
2219    /// the real-runtime `BoardEntry::run` (which never returns), then
2220    /// spins defensively. The Entry crate provides the panic handler
2221    /// (`esp-backtrace`) + app descriptor (`esp_app_desc!`).
2222    Esp32,
2223}

2225// Phase 225.O follow-up (known-issue #18) — NOTE on NuttX. NuttX does
2226// NOT get its own `Framework` variant: it rides `Framework::OwnedSpin`.
2227// The NuttX flat-build init task calls `CONFIG_INIT_ENTRYPOINT="nsh_main"`,
2228// but the board crate (`nros-board-nuttx-qemu-arm`'s `entry.rs`) already
2229// exports a `#[no_mangle] nsh_main` that runs `nsh_initialize()` (virtio
2230// FDT discovery + network bringup) and then calls the Rust `main`
2231// lang-start symbol. OwnedSpin emits exactly that `fn main()` (NuttX is
2232// `target_os = "nuttx"`, the `not(target_os = "none")` hosted arm), which
2233// delegates to `<QemuArmVirt as BoardEntry>::run`. So no NuttX-specific
2234// emit is needed; emitting our own `nsh_main` would both collide with
2235// the board's and skip the critical `nsh_initialize()` network bringup.
2236
2237fn framework_for(deploy: &str) -> Framework {
2238    match deploy {
2239        "rtic-stm32f4" | "rtic-mps2-an385" | "qemu-rtic-mps2-an385" => Framework::Rtic,
2240        "embassy-stm32f4" => Framework::Embassy,
2241        "zephyr" => Framework::Zephyr,
2242        "esp32-qemu" | "qemu-esp32-baremetal" => Framework::Esp32,
2243        // NuttX ("nuttx" / "qemu-arm-nuttx") rides OwnedSpin — the board
2244        // crate's `nsh_main` bridges the kernel init task to the emitted
2245        // `fn main()`. See the `Framework` enum NOTE.
2246        _ => Framework::OwnedSpin,
2247    }
2248}

2250/// Sanitise a pkg name into a valid Rust crate ident. Cargo allows
2251/// `-` in pkg names; Rust idents don't. Matches the
2252/// `sanitize_pkg_name_for_symbol` rule the existing `nros::node!()`

2254fn pkg_to_crate_ident(pkg: &str) -> String {
2255    let mut out = String::with_capacity(pkg.len());
2256    for c in pkg.chars() {
2257        if c.is_ascii_alphanumeric() || c == '_' {
2258            out.push(c);
2259        } else {
2260            out.push('_');
2261        }
2262    }
2263    out
2264}
