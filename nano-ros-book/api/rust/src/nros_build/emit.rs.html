1//! Emit `$OUT_DIR/run_plan.rs` from an [`NrosPlan`].
2//!
3//! The emitted file is `include!`d from the Entry pkg's `main.rs`.
4//! It declares one fn:
5//!
6//! ```ignore
7//! pub fn run_plan(
8//!     runtime: &mut ::nros_platform::RuntimeCtx<'_>,
9//! ) -> ::core::result::Result<(), ::nros_build::RuntimeError> {
10//!     // For each plan.instances entry:
11//!     //   ::<pkg_ident>::register(runtime).map_err(|_|
12//!     //       ::nros_build::RuntimeError::ComponentRegister("<pkg>"))?;
13//!     Ok(())
14//! }
15//! ```
16//!
17//! The fn body deliberately keeps no per-instance state — it is a
18//! flat sequence of register calls in plan-order (which the planner
19//! already orders so producers come before consumers). Lifecycle
20//! hooks / shared-state regions / param overlays are NOT yet emitted;
21//! they live in the plan and will be folded in over follow-up
22//! patches.
23
24use core::fmt::Write as _;
25
26use nros_cli_core::orchestration::plan::NrosPlan;

28/// Render the body of `$OUT_DIR/run_plan.rs` for the given plan.
29///
30/// Returns a UTF-8 string the caller writes verbatim. Doc-comments
31/// in the emitted code spell out the lineage (planner version,
32/// system name, instance count) so a curious user can introspect
33/// without re-running codegen.
34pub fn emit_run_plan(plan: &NrosPlan) -> String {
35    let mut out = String::with_capacity(4096);
36    let _ = writeln!(
37        out,
38        "// AUTO-GENERATED by nros-build::emit_run_plan (Phase 212.N.4)."
39    );
40    let _ = writeln!(out, "// DO NOT EDIT — regenerated by Entry pkg build.rs.");
41    let _ = writeln!(out, "//");
42    let _ = writeln!(out, "// plan.version : {}", plan.version);
43    let _ = writeln!(out, "// plan.system  : {}", plan.system);
44    let _ = writeln!(out, "// instance count: {}", plan.instances.len());
45    out.push('\n');
46
47    out.push_str(
48        "pub fn run_plan(\n    runtime: &mut ::nros_platform::RuntimeCtx<'_>,\n) -> ::core::result::Result<(), ::nros_platform::RuntimeError> {\n    let _ = runtime; // silence unused-warn on empty plans\n",
49    );
50
51    for inst in &plan.instances {
52        // Match the planner's record-side pkg field (a launch <node
53        // pkg="X" exec="Y"/> → record.node.package). The plan's
54        // PlanInstance carries the unsanitised pkg name; here we
55        // produce the dash-stripped ident that matches Rust's
56        // crate-name → ident projection (`-` → `_`).
57        let pkg_ident = pkg_ident_from(inst);
58        let pkg_label = pkg_label_from(inst);
59        let _ = writeln!(
60            out,
61            "    ::{pkg_ident}::register(runtime).map_err(|_| ::nros_platform::RuntimeError::ComponentRegister({pkg_label:?}))?;"
62        );
63    }
64
65    out.push_str("    Ok(())\n}\n");
66    out
67}

69fn pkg_ident_from(inst: &nros_cli_core::orchestration::plan::PlanInstance) -> String {
70    // The plan's `PlanInstance` carries `component` (= component-pkg
71    // name) — use whatever serde_json field actually exists. We
72    // accept either `component` or `pkg` to stay compatible across
73    // planner versions. Convert dashes → underscores for the Rust
74    // ident.
75    let raw = instance_pkg_name(inst);
76    raw.replace('-', "_")
77}
78
79fn pkg_label_from(inst: &nros_cli_core::orchestration::plan::PlanInstance) -> String {
80    instance_pkg_name(inst)
81}
82
83fn instance_pkg_name(inst: &nros_cli_core::orchestration::plan::PlanInstance) -> String {
84    // Round-trip through serde_json::Value so the emitter doesn't
85    // require a knowledge of `PlanInstance`'s precise field layout
86    // (which has churned across planner versions). The planner emits
87    // one of `pkg` / `component` / `name`; we walk those in priority
88    // order.
89    let v = serde_json::to_value(inst).unwrap_or(serde_json::Value::Null);
90    for key in ["component", "pkg", "package", "name"] {
91        if let Some(s) = v.get(key).and_then(|x| x.as_str()) {
92            return s.to_string();
93        }
94    }
95    "unknown".to_string()
96}
